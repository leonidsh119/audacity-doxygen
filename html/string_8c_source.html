78 serd_strtod(const char* str, char** endptr)
79 {
80  double result = 0.0;
81 
82  // Point s at the first non-whitespace character
83  const char* s = str;
84  while (is_space(*s)) { ++s; }
85 
86  // Read leading sign if necessary
87  const double sign = read_sign(&s);
88 
89  // Parse integer part
90  for (; is_digit(*s); ++s) {
91  result = (result * 10.0) + (*s - '0');
92  }
93 
94  // Parse fractional part
95  if (*s == '.') {
96  double denom = 10.0;
97  for (++s; is_digit(*s); ++s) {
98  result += (*s - '0') / denom;
99  denom *= 10.0;
100  }
101  }
102 
103  // Parse exponent
104  if (*s == 'e' || *s == 'E') {
105  ++s;
106  double expt = 0.0;
107  double expt_sign = read_sign(&s);
108  for (; is_digit(*s); ++s) {
109  expt = (expt * 10.0) + (*s - '0');
110  }
111  result *= pow(10, expt * expt_sign);
112  }
113 
114  if (endptr) {
115  *endptr = (char*)s;
116  }
117 
118  return result * sign;
119 }
