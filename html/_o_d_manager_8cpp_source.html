141 void ODManager::AddNewTask(ODTask* task, bool lockMutex)
142 {
143  ODWaveTrackTaskQueue* queue = NULL;
144 
145  if(lockMutex)
146  mQueuesMutex.Lock();
147  for(unsigned int i=0;i<mQueues.size();i++)
148  {
149  //search for a task containing the lead track. wavetrack removal is threadsafe and bound to the mQueuesMutex
150  //note that GetWaveTrack is not threadsafe, but we are assuming task is not running on a different thread yet.
151  if(mQueues[i]->ContainsWaveTrack(task->GetWaveTrack(0)))

155  if(queue)
156  {
157  //Add it to the existing queue but keep the lock since this reference can be deleted.

431 bool ODManager::MakeWaveTrackDependent(WaveTrack* dependentTrack,WaveTrack* masterTrack)
432 {
433  //First, check to see if the task lists are mergeable. If so, we can simply add this track to the other task and queue,
434  //then DELETE this one.
435  ODWaveTrackTaskQueue* masterQueue=NULL;
436  ODWaveTrackTaskQueue* dependentQueue=NULL;
437  unsigned int dependentIndex = 0;
438  bool canMerge = false;
439 
440  mQueuesMutex.Lock();
441  for(unsigned int i=0;i<mQueues.size();i++)
442  {
443  if(mQueues[i]->ContainsWaveTrack(masterTrack))
444  {
445  masterQueue = mQueues[i];
446  }
447  else if(mQueues[i]->ContainsWaveTrack(dependentTrack))
448  {
449  dependentQueue = mQueues[i];
450  dependentIndex = i;
451  }
452 
453  }
454  if(masterQueue&&dependentQueue)
455  canMerge=masterQueue->CanMergeWith(dependentQueue);
456 
457  //otherwise we need to let dependentTrack's queue live on. We'll have to wait till the conflicting tasks are done.
458  if(!canMerge)
459  {
460  mQueuesMutex.Unlock();
461  return false;
462  }
463  //then we add dependentTrack to the masterTrack's queue - this will allow future ODScheduling to affect them together.
464  //this sets the NeedODUpdateFlag since we don't want the head task to finish without haven't dealt with the depednent
465  masterQueue->MergeWaveTrack(dependentTrack);
466 
467  //finally remove the dependent track
468  mQueues.erase(mQueues.begin()+dependentIndex);
469  mQueuesMutex.Unlock();
470  delete dependentQueue; //note this locks the ODManager's current task queue.
471  return true;
472 }

490 void ODManager::UpdateQueues()
491 {
492  mQueuesMutex.Lock();
493  for(unsigned int i=0;i<mQueues.size();i++)
494  {
495  if(mQueues[i]->IsFrontTaskComplete())
496  {
497  //this should DELETE and remove the front task instance.
498  mQueues[i]->RemoveFrontTask();
499  //schedule next.
500  if(!mQueues[i]->IsEmpty())
501  {
502  //we need to release the lock on the queue vector before using the task vector's lock or we deadlock
503  //so get a temp.
504  ODWaveTrackTaskQueue* queue;
505  queue = mQueues[i];
506 
507  AddTask(queue->GetFrontTask());
508  }
509  }
510 
511  //if the queue is empty DELETE it.
512  if(mQueues[i]->IsEmpty())
513  {
514  delete mQueues[i];
515  mQueues.erase(mQueues.begin()+i);
516  i--;
517  }
518  }
519  mQueuesMutex.Unlock();
520 }
