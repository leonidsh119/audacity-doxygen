137 LOCAL void freesegs()
138 {
139  SEGMENT *seg;
140  SEGMENT *next;
141 
142  // Free up as many nodes as possible
143  gc();
144 
145  // Reset free node tracking
146  fnodes = NIL;
147  nfree = 0L;
148 
149  // Reset the last segment pointer
150  lastseg = NULL;
151 
152  // Scan all segments
153  for (seg = segs; seg != NULL; seg = next) {
154  int n = seg->sg_size;
155  int empty = TRUE;
156  int i;
157  LVAL p;
158 
159  // Check this segment for in-use nodes
160  p = &seg->sg_nodes[0];
161  for (i = n; --i >= 0; ++p) {
162  if (ntype(p) != FREE_NODE) {
163  empty = FALSE;
164  break;
165  }
166  }
167 
168  // Retain pointer to next segment
169  next = seg->sg_next;
170 
171  // Was the current segment empty?
172  if (empty) {
173  // Free the segment;
174  free((void *) seg);
175 
176  // Unlink it from the list. No need to worry about a NULL lastseg
177  // pointer here since the fixnum and char segments will always exist
178  // at the head of the list and they will always have nodes. So, lastseg
179  // will have been set before we find any empty nodes.
180  lastseg->sg_next = next;
181 
182  // Reduce the stats
183  total -= (long) segsize(n);
184  nsegs--;
185  nnodes -= n;
186  }
187  else {
188  // Not empty, so remember this node as the last segment
189  lastseg = seg;
190 
191  // Add all of the free nodes in this segment to the free list
192  p = &seg->sg_nodes[0];
193  for (i = n; --i >= 0; ++p) {
194  if (ntype(p) == FREE_NODE) {
195  rplaca(p, NIL);
196  rplacd(p, fnodes);
197  fnodes = p;
198  nfree++;
199  }
200  }
201  }
202  }

207 // Copy a node (recursively if appropriate)
208 LOCAL LVAL nyx_dup_value(LVAL val)
209 {
210  LVAL nval = val;
211 
212  // Protect old and new values
213  xlprot1(val);
214  xlprot1(nval);
215 
216  // Copy the node
217  if (val != NIL) {
218  switch (ntype(val))
219  {
220  case FIXNUM:
221  nval = cvfixnum(getfixnum(val));
222  break;
223 
224  case FLONUM:
225  nval = cvflonum(getflonum(val));
226  break;
227 
228  case CHAR:
229  nval = cvchar(getchcode(val));
230  break;
231 
232  case STRING:
233  nval = cvstring((char *) getstring(val));
234  break;
235 
236  case VECTOR:
237  {
238  int len = getsize(val);
239  int i;
240 
241  nval = newvector(len);
242  nval->n_type = ntype(val);
243 
244  for (i = 0; i < len; i++) {
245  if (getelement(val, i) == val) {
246  setelement(nval, i, val);
247  }
248  else {
249  setelement(nval, i, nyx_dup_value(getelement(val, i)));
250  }
251  }
252  }
253  break;
254 
255  case CONS:
256  nval = cons(nyx_dup_value(car(val)), nyx_dup_value(cdr(val)));
257  break;
258 
259  case SUBR:
260  case FSUBR:
261  nval = cvsubr(getsubr(val), ntype(val), getoffset(val));
262  break;
263 
264  // Symbols should never be copied since their addresses are cached
265  // all over the place.
266  case SYMBOL:
267  nval = val;
268  break;
269 
270  // Streams are not copied (although USTREAM could be) and reference
271  // the original value.
272  case USTREAM:
273  case STREAM:
274  nval = val;
275  break;
276 
277  // Externals aren't copied because I'm not entirely certain they can be.
278  case EXTERN:
279  nval = val;
280  break;
281 
282  // For all other types, just allow them to reference the original
283  // value. Probably not the right thing to do, but easier.
284  case OBJECT:
285  case CLOSURE:
286  default:
287  nval = val;
288  break;
289  }
290  }
291 
292  xlpop();
293  xlpop();
294 
295  return nval;
296 }

632 LOCAL int nyx_is_labels(LVAL expr)
633 {
634  /* make sure that we have a list whose first element is a
635  list of the form (time "label") */
636 
637  LVAL label;
638  LVAL first;
639  LVAL second;
640  LVAL third;
641 
642  if (expr == NULL) {
643  return 0;
644  }
645 
646  while (expr != NULL) {
647  if (!consp(expr))
648  return 0;
649 
650  label = car(expr);
651 
652  if (!consp(label))
653  return 0;
654 
655  first = car(label);
656  if (!(floatp(first) || fixp(first)))
657  return 0;
658 
659  if (!consp(cdr(label)))
660  return 0;
661 
662  second = car(cdr(label));
663 
664  if (floatp(second) || fixp(second)) {
665  if (!consp(cdr(cdr(label))))
666  return 0;
667  third = car(cdr(cdr(label)));
668  if (!(stringp(third)))
669  return 0;
670  }
671  else
672  if (!(stringp(second)))
673  return 0;
674 
675  expr = cdr(expr);
676  }
677 
678  return 1;
679 }

895  }

898  }
