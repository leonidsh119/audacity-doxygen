303  streamContext** streams,
304  unsigned int numStreams)
305 {
306  streamContext *sc = NULL;
307  AVPacket pkt;
308 
309  if (av_read_frame(formatContext,&pkt) < 0)
310  {
311  return NULL;
312  }
313 
314  // Find a stream to which this frame belongs to

317  if (streams[i]->m_stream->index == pkt.stream_index)

339 int import_ffmpeg_decode_frame(streamContext *sc, bool flushing)
340 {
341  int nBytesDecoded;
342  wxUint8 *pDecode = sc->m_pktDataPtr;
343  int nDecodeSiz = sc->m_pktRemainingSiz;
344 
345  sc->m_frameValid = 0;
346 
347  if (flushing)
348  {
349  // If we're flushing the decoders we don't actually have any NEW data to decode.
350  pDecode = NULL;
351  nDecodeSiz = 0;
352  }
353  else
354  {
355  if (!sc->m_pktValid || (sc->m_pktRemainingSiz <= 0))
356  {
357  //No more data
358  return -1;
359  }
360  }
361 
362  AVPacket avpkt;
363  av_init_packet(&avpkt);
364  avpkt.data = pDecode;
365  avpkt.size = nDecodeSiz;
366 
367  AVFrame *frame = av_frame_alloc();
368  int got_output = 0;
369 
370  nBytesDecoded =
371  avcodec_decode_audio4(sc->m_codecCtx,
372  frame, // out
373  &got_output, // out
374  &avpkt); // in
375 
376  if (nBytesDecoded < 0)
377  {
378  // Decoding failed. Don't stop.
379  return -1;
380  }
381 
382  sc->m_samplefmt = sc->m_codecCtx->sample_fmt;
383  sc->m_samplesize = av_get_bytes_per_sample(sc->m_samplefmt);
384 
385  int channels = sc->m_codecCtx->channels;
386  unsigned int newsize = sc->m_samplesize * frame->nb_samples * channels;
387  sc->m_decodedAudioSamplesValidSiz = newsize;
388  // Reallocate the audio sample buffer if it's smaller than the frame size.
389  if (newsize > sc->m_decodedAudioSamplesSiz )
390  {
391  if (sc->m_decodedAudioSamples)
392  {
393  av_free(sc->m_decodedAudioSamples);
394  }
395 
396  sc->m_decodedAudioSamples = (uint8_t *) av_malloc(newsize);
397  sc->m_decodedAudioSamplesSiz = newsize;
398 
399  if (sc->m_decodedAudioSamples == NULL)
400  {
401  //Can't allocate bytes
402  return -1;
403  }
404  }
405  if (frame->data[1]) {
406  for (int i = 0; i<frame->nb_samples; i++) {
407  for (int ch = 0; ch<channels; ch++) {
408  memcpy(sc->m_decodedAudioSamples + sc->m_samplesize * (ch + channels*i),
409  frame->extended_data[ch] + sc->m_samplesize*i,
410  sc->m_samplesize);
411  }
412  }
413  } else {
414  memcpy(sc->m_decodedAudioSamples, frame->data[0], newsize);
415  }
416 
417  av_frame_free(&frame);
418 
419  // We may not have read all of the data from this packet. If so, the user can call again.
420  // Whether or not they do depends on if m_pktRemainingSiz == 0 (they can check).
421  sc->m_pktDataPtr += nBytesDecoded;
422  sc->m_pktRemainingSiz -= nBytesDecoded;
423 
424  // At this point it's normally safe to assume that we've read some samples. However, the MPEG
425  // audio decoder is broken. If this is the case then we just return with m_frameValid == 0
426  // but m_pktRemainingSiz perhaps != 0, so the user can call again.
427  if (sc->m_decodedAudioSamplesValidSiz > 0)
428  {
429  sc->m_frameValid = 1;
430  }
431  return 0;
432 }
