42 void process(Alg_seq_ptr seq, bool tempo_flag, double tempo,
43  bool flatten_flag)
44 {
45  // the tempo changing operation replaces the tempo track with
46  // a fixed tempo. This changes the timing of the result.
47  if (tempo_flag) {
48  seq->convert_to_beats(); // preserve beats
49  } else if (flatten_flag) {
50  seq->convert_to_seconds(); // preserve timing
51  } else return;
52  // the following finishes both tempo and flatten processing...
53  seq->get_time_map()->beats.len = 1; // remove contents of tempo map
54  seq->get_time_map()->last_tempo = tempo / 60.0; // set the new fixed tempo
55  // (allegro uses beats/second so divide bpm by 60)
56  seq->get_time_map()->last_tempo_flag = true;
57 }
