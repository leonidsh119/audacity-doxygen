183 void HelpSystem::ShowHelpDialog(wxWindow *parent,
184  const wxString &localFileName,
185  const wxString &remoteURL,
186  bool bModal)
187 {
188  AudacityProject * pProj = GetActiveProject();
189  wxString HelpMode = wxT("Local");
190 
191  if( pProj )
192  {
193  HelpMode = pProj->mHelpPref;
194  // these next lines are for legacy cfg files (pre 2.0) where we had different modes
195  if( (HelpMode == wxT("Standard")) || (HelpMode == wxT("InBrowser")) )
196  {
197  HelpMode = wxT("Local");
198  pProj->mHelpPref = HelpMode;
199  gPrefs->Write(wxT("/GUI/Help"), HelpMode);
200  gPrefs->Flush();
201  }
202  }
203 
204  // Anchors (URLs with a '#' in them) are not supported by many OSs for local file names
205  // See, for example, https://groups.google.com/forum/#!topic/wx-users/pC0uOZJalRQ
206  // Problems have been reported on Win, Mac and some versions of Linux.
207  // So we set HelpMode to use the internet if an anchor is found.
208  if (localFileName.Find('#', true) != wxNOT_FOUND)
209  HelpMode = wxT("FromInternet");
210  // Until a solution is found for this, the next few lines are irrelevant.
211 
212  // Obtain the local file system file name, without the anchor if present.
213  wxString localfile;
214  if (localFileName.Find('#', true) != wxNOT_FOUND)
215  localfile = localFileName.BeforeLast('#');
216  else
217  localfile = localFileName;
218 
219  if( (HelpMode == wxT("FromInternet")) && !remoteURL.IsEmpty() )
220  {
221  // Always go to remote URL. Use External browser.
222  OpenInDefaultBrowser( remoteURL );
223  }
224  else if( !wxFileExists( localfile ))
225  {
226  // If you give an empty remote URL, you should have already ensured
227  // that the file exists!
228  wxASSERT( !remoteURL.IsEmpty() );
229  // I can't find it'.
230  // Use Built-in browser to suggest you use the remote url.
231  wxString Text = HelpText( wxT("remotehelp") );
232  Text.Replace( wxT("*URL*"), remoteURL );
233  ShowHtmlText( parent, _("Help on the Internet"), Text, false, bModal );
234  }
235  else if( HelpMode == wxT("Local") )
236  {
237  // Local file, External browser
238  OpenInDefaultBrowser( wxString(wxT("file:"))+localFileName );
239  }
240  else
241  {
242  // Local file, Built-in browser
243  ShowHtmlText( parent, wxT(""), localFileName, true, bModal );
244  }

247 void HelpSystem::ShowHelpDialog(wxWindow *parent,
248  const wxString &PageName,
249  bool bModal)
250 {
251  wxString localHelpPage;
252  wxString webHelpPath;
253  wxString webHelpPage;
254  wxString releasePageName;
255  wxString anchor; // optional part of URL after (and including) the '#'
256  if (PageName.Find('#', true) != wxNOT_FOUND)
257  { // need to split anchor off into separate variable
258  releasePageName= PageName.BeforeLast('#');
259  anchor = wxT("#") + PageName.AfterLast('#');
260  }
261  else
262  {
263  releasePageName = PageName;
264  anchor = wxT("");
265  }
266  // The wiki pages are transformed to static HTML by
267  // scripts/mw2html_audacity/mw2html.py
268  // The name is first transformed to lower case, then all
269  // 'special characters' are replaced by underscores. Spaces are
270  // transformed to "+".
271  //
272  // The transformations are handled in mw2html by first applying
273  // 'urllib.parse.quote_plus' (escape chars that are not in "always safe" list)
274  // then replacing escape characters (%xx) with underscores,
275  // and finally removing duplicate / redundant underscores.
276  //
277  // The front page and 'quick_help' are treated as special cases and placed in
278  // the root of the help directory rather than the "/man/" sub-directory.
279  if (releasePageName == wxT("Main_Page"))
280  {
281  releasePageName = wxT("index") + HelpSystem::ReleaseSuffix + anchor;
282  localHelpPage = wxFileName(FileNames::HtmlHelpDir(), releasePageName).GetFullPath();
283  webHelpPath = wxT("http://")+HelpSystem::HelpHostname+HelpSystem::HelpServerHomeDir;
284  }
285  else if (releasePageName == wxT("Quick_Help"))
286  {
287  releasePageName = wxT("quick_help") + HelpSystem::ReleaseSuffix + anchor;
288  localHelpPage = wxFileName(FileNames::HtmlHelpDir(), releasePageName).GetFullPath();
289  webHelpPath = wxT("http://")+HelpSystem::HelpHostname+HelpSystem::HelpServerHomeDir;
290  }
291  else
292  {
293  // Handle all other pages.
294  // Change to lower case.
295  releasePageName = releasePageName.Lower();
296  wxRegEx re;
297  // replace 'special characters' with underscores.
298  // RFC 2396 defines the characters a-z, A-Z, 0-9 and ".-_" as "always safe"
299  // mw2html also replaces "-" with "_" so replace that too.
300 
301  // If PageName contains a %xx code, mw2html will transform it:
302  // '%xx' => '%25xx' => '_'
303  re.Compile(wxT("%.."));
304  re.ReplaceAll(&releasePageName, (wxT("_")));
305  // Now replace all other 'not-safe' characters.
306  re.Compile(wxT("[^[:alnum:] . [:space:]]"));
307  re.ReplaceAll(&releasePageName, (wxT("_")));
308  // Replace spaces with "+"
309  releasePageName.Replace(wxT(" "), wxT("+"), true);
310  // Reduce multiple underscores to single underscores
311  re.Compile(wxT("__+"));
312  re.ReplaceAll(&releasePageName, (wxT("_")));
313  // Replace "_." with "."
314  releasePageName.Replace(wxT("_."), wxT("."), true);
315  // Concatenate file name with file extension and anchor.
316  releasePageName = releasePageName + HelpSystem::ReleaseSuffix + anchor;
317  // Other than index and quick_help, all local pages are in subdirectory 'LocalHelpManDir'.
318  localHelpPage = wxFileName(FileNames::HtmlHelpDir() + LocalHelpManDir, releasePageName).GetFullPath();
319  // Other than index and quick_help, all on-line pages are in subdirectory 'HelpServerManDir'.
320  webHelpPath = wxT("http://")+HelpSystem::HelpHostname+HelpSystem::HelpServerManDir;
321  }
322 
323 #if IS_ALPHA
324  webHelpPage = webHelpPath + PageName;
325 #else
326  webHelpPage = webHelpPath + releasePageName;
327 #endif
328 
329  wxLogMessage(wxT("Help button pressed: PageName %s, releasePageName %s"),
330  PageName.c_str(), releasePageName.c_str());
331  wxLogMessage(wxT("webHelpPage %s, localHelpPage %s"),
332  webHelpPage.c_str(), localHelpPage.c_str());
333 
334  HelpSystem::ShowHelpDialog(
335  parent,
336  localHelpPage,
337  webHelpPage,
338  bModal);
339 }
