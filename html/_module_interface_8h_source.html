64 class ModuleInterface /* not final */ : public IdentInterface
65 {
66 public:
67  virtual ~ModuleInterface() {};

69  // Called immediately after creation to give the instance a chance to
70  // initialize. Return "true" if initialziation was successful.
71  virtual bool Initialize() = 0;
72 
73  // Called just prior to deletion to allow releasing any resources.
74  virtual void Terminate() = 0;
75 
76  // Modules providing a single or static set of plugins may use
77  // AutoRegisterPlugins() to register those plugins.
78  virtual bool AutoRegisterPlugins(PluginManagerInterface & pluginManager) = 0;
79 
80  // For modules providing an interface to other dynamically loaded plugins,
81  // the module returns a list of path names that will be presented to the
82  // user for enablement.
83  virtual wxArrayString FindPlugins(PluginManagerInterface & pluginManager) = 0;
84 
85  // Once the user selects desired paths from FindPlugins(), a call to RegisterPlugin()
86  // will be made to request registration of that plugin. If the module must create
87  // an instance of the plugin to register it, then then instance should be deleted
88  // after registration.
89  virtual bool RegisterPlugin(PluginManagerInterface & pluginManager,
90  const wxString & path) = 0;
91 
92  // For modules providing an interface to other dynamically loaded plugins,
93  // the module returns true if the plugin is still valid, otherwise false.
94  virtual bool IsPluginValid(const wxString & path) = 0;
95 
96  // When appropriate, CreateInstance() will be called to instantiate the plugin.
97  virtual IdentInterface *CreateInstance(const wxString & path) = 0;
98 
99  // When appropriate, DeleteInstance() will be called to delete the plugin.
100  virtual void DeleteInstance(IdentInterface *instance) = 0;
