285 static inline LV2_Atom_Forge_Ref
286 lv2_atom_forge_raw(LV2_Atom_Forge* forge, const void* data, uint32_t size)
287 {
288  LV2_Atom_Forge_Ref out = 0;
289  if (forge->sink) {
290  out = forge->sink(forge->handle, data, size);
291  } else {
292  out = (LV2_Atom_Forge_Ref)forge->buf + forge->offset;
293  uint8_t* mem = forge->buf + forge->offset;
294  if (forge->offset + size > forge->size) {
295  return 0;
296  }
297  forge->offset += size;
298  memcpy(mem, data, size);
299  }
300  for (LV2_Atom_Forge_Frame* f = forge->stack; f; f = f->parent) {
301  lv2_atom_forge_deref(forge, f->ref)->size += size;
302  }
303  return out;
304 }
