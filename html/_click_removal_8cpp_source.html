165  SelectedTrackListOfKindIterator iter(Track::Wave, mOutputTracks);

199 bool EffectClickRemoval::ProcessOne(int count, WaveTrack * track, sampleCount start, sampleCount len)
200 {
201  if (len <= windowSize/2)
202  {
203  wxMessageBox(
204  wxString::Format(_("Selection must be larger than %d samples."), windowSize/2),
205  GetName(),
206  wxOK | wxICON_ERROR);
207  return false;
208  }
209 
210  sampleCount idealBlockLen = track->GetMaxBlockSize() * 4;
211  if (idealBlockLen % windowSize != 0)
212  idealBlockLen += (windowSize - (idealBlockLen % windowSize));
213 
214  bool bResult = true;
215  sampleCount s = 0;
216  float *buffer = new float[idealBlockLen];
217  float *datawindow = new float[windowSize];
218  while ((s < len) && ((len - s) > windowSize/2))
219  {
220  sampleCount block = idealBlockLen;
221  if (s + block > len)
222  block = len - s;
223 
224  track->Get((samplePtr) buffer, floatSample, start + s, block);
225 
226  for (int i=0; i < (block-windowSize/2); i += windowSize/2)
227  {
228  int wcopy = windowSize;
229  if (i + wcopy > block)
230  wcopy = block - i;
231 
232  int j;
233  for(j=0; j<wcopy; j++)
234  datawindow[j] = buffer[i+j];
235  for(j=wcopy; j<windowSize; j++)
236  datawindow[j] = 0;
237 
238  mbDidSomething |= RemoveClicks(windowSize, datawindow);
239 
240  for(j=0; j<wcopy; j++)
241  buffer[i+j] = datawindow[j];
242  }
243 
244  if (mbDidSomething) // RemoveClicks() actually did something.
245  track->Set((samplePtr) buffer, floatSample, start + s, block);
246 
247  s += block;
248 
249  if (TrackProgress(count, s / (double) len)) {
250  bResult = false;
251  break;
252  }
253  }
254 
255  delete[] buffer;
256  delete[] datawindow;
257 
258  return bResult;
259 }

261 bool EffectClickRemoval::RemoveClicks(sampleCount len, float *buffer)
262 {
263  bool bResult = false; // This effect usually does nothing.
264  int i;
265  int j;
266  int left = 0;
267 
268  float msw;
269  int ww;
270  int s2 = sep/2;
271  float *ms_seq = new float[len];
272  float *b2 = new float[len];
273 
274  for( i=0; i<len; i++)
275  b2[i] = buffer[i]*buffer[i];
276 
277  /* Shortcut for rms - multiple passes through b2, accumulating
278  * as we go.
279  */
280  for(i=0;i<len;i++)
281  ms_seq[i]=b2[i];
282 
283  for(i=1; i < sep; i *= 2) {
284  for(j=0;j<len-i; j++)
285  ms_seq[j] += ms_seq[j+i];
286  }
287 
288  /* Cheat by truncating sep to next-lower power of two... */
289  sep = i;
290 
291  for( i=0; i<len-sep; i++ ) {
292  ms_seq[i] /= sep;
293  }
294  /* ww runs from about 4 to mClickWidth. wrc is the reciprocal;
295  * chosen so that integer roundoff doesn't clobber us.
296  */
297  int wrc;
298  for(wrc=mClickWidth/4; wrc>=1; wrc /= 2) {
299  ww = mClickWidth/wrc;
300 
301  for( i=0; i<len-sep; i++ ){
302  msw = 0;
303  for( j=0; j<ww; j++) {
304  msw += b2[i+s2+j];
305  }
306  msw /= ww;
307 
308  if(msw >= mThresholdLevel * ms_seq[i]/10) {
309  if( left == 0 ) {
310  left = i+s2;
311  }
312  } else {
313  if(left != 0 && i-left+s2 <= ww*2) {
314  float lv = buffer[left];
315  float rv = buffer[i+ww+s2];
316  for(j=left; j<i+ww+s2; j++) {
317  bResult = true;
318  buffer[j]= (rv*(j-left) + lv*(i+ww+s2-j))/(float)(i+ww+s2-left);
319  b2[j] = buffer[j]*buffer[j];
320  }
321  left=0;
322  } else if(left != 0) {
323  left = 0;
324  }
325  }
326  }
327  }
328  delete[] ms_seq;
329  delete[] b2;
330  return bResult;
331 }
