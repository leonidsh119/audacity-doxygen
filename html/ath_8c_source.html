35 FLOAT ath_db(FLOAT f, FLOAT value)
36 {
37  /* from Painter & Spanias modified by Gabriel Bouvigne to better fit the reality ath = 3.640 *
38  pow(f,-0.8) - 6.800 * exp(-0.6*pow(f-3.4,2.0)) + 6.000 * exp(-0.15*pow(f-8.7,2.0)) + 0.6*
39  0.001 * pow(f,4.0);
40 
41 
42  In the past LAME was using the Painter &Spanias formula. But we had some recurrent problems
43  with HF content. We measured real ATH values, and found the older formula to be inacurate in
44  the higher part. So we made this new formula and this solved most of HF problematic
45  testcases. The tradeoff is that in VBR mode it increases a lot the bitrate. */
46 
47 
48  /* this curve can be udjusted according to the VBR scale: it adjusts from something close to
49  Painter & Spanias on V9 up to Bouvigne's formula for V0. This way the VBR bitrate is more
50  balanced according to the -V value. */
51 
52  FLOAT ath;
53  FLOAT valueold = 0.0;
54 
55  if (f < -.3)
56  f = 3410;
57 
58  f /= 1000; // convert to khz
59  f = MAX(0.01, f);
60  f = MIN(18.0, f);
61 
62  ath = 3.640 * pow(f, -0.8)
63  - 6.800 * exp(-0.6 * pow(f - 3.4, 2.0))
64  + 6.000 * exp(-0.15 * pow(f - 8.7, 2.0))
65  + (0.6 + 0.04 * valueold) * 0.001 * pow(f, 4.0);
66 
67  /* MFC Feb 2003 I've changed the fudge technique on the code. The "-l [float]" value
68  raises/lowers the ATH by this many dB */
69  return (ath + value);
70 }

73 /* Convert ATH values from dB into energy values as required by the psycho model */
74 FLOAT ath_energy(FLOAT freq, FLOAT value)
75 {
76  FLOAT db;
77  db = ath_db(freq, 0) + value; // Originally: ath_db(freq,value)
78  /* The values in the standard, and from the ATH formula are in dB. In the psycho model we are
79  working in the energy domain. Hence the values that are in the absthr_X tables are not in
80  dB. This function converts from dB into the energy domain. As noted on the LAME mailing list
81  from years ago (MFC FIX find the reference), the absolute threhsold of hearing values in the
82  tables in the standard are dodgy - the ATH in the tables do not correspond to any previously
83  known values of the ATH. From ISO 11172 Tables D.4.x "A value of 0dB represents a level in
84  the absolute threshold calculation of 96dB below the energy of a sine wave of amplitude
85  32760." But I still don't know why the factor of 41.837375 is the value that it is. MFC Feb
86  2003 */
87  return (pow(10.0, (db + 41.837375) * 0.1));
88 }

94 FLOAT ath_freq2bark(FLOAT freq)
95 {
96  if (freq < 0)
97  freq = 0;
98  freq = freq * 0.001;
99  return 13.0 * atan(.76 * freq) + 3.5 * atan(freq * freq / (7.5 * 7.5));
