95 void MidiIOPrefs::GetNamesAndLabels() {

97  Pm_Terminate(); // close and open to refresh device lists
98  Pm_Initialize();
99  int nDevices = Pm_CountDevices();

101  const PmDeviceInfo *info = Pm_GetDeviceInfo(i);
102  if (info->output || info->input) { //should always happen

104  if (mHostNames.Index(name) == wxNOT_FOUND) {
105  mHostNames.Add(name);

108  }

176 void MidiIOPrefs::OnHost(wxCommandEvent & e)
177 {
178  int index = mHost->GetCurrentSelection();
179  wxString itemAtIndex = mHostNames.Item(index);
180  int nDevices = Pm_CountDevices();
181 
182  if (nDevices == 0) {
183  mHost->Clear();
184  mHost->Append(_("No MIDI interfaces"), (void *) NULL);
185  mHost->SetSelection(0);
186  }
187 
188  mPlay->Clear();
189 #ifdef EXPERIMENTAL_MIDI_IN
190  mRecord->Clear();
191 #endif
192 
193  wxArrayString playnames;
194  wxArrayString recordnames;
195 
196  for (int i = 0; i < nDevices; i++) {
197  const PmDeviceInfo *info = Pm_GetDeviceInfo(i);
198  wxString interf = wxSafeConvertMB2WX(info->interf);
199  if (itemAtIndex.IsSameAs(interf)) {
200  wxString name = wxSafeConvertMB2WX(info->name);
201  wxString device = wxString::Format(wxT("%s: %s"),
202  interf.c_str(),
203  name.c_str());
204  int index;
205 
206  if (info->output) {
207  playnames.Add(name);
208  index = mPlay->Append(name, (void *) info);
209  if (device == mPlayDevice) {
210  mPlay->SetSelection(index);
211  }
212  }
213 #ifdef EXPERIMENTAL_MIDI_IN
214  if (info->input) {
215  recordnames.Add(name);
216  index = mRecord->Append(name, (void *) info);
217  if (device == mRecordDevice) {
218  mRecord->SetSelection(index);
219  }
220  }
221 #endif
222  }
223  }
224 
225  if (mPlay->GetCount() == 0) {
226  playnames.Add(_("No devices found"));
227  mPlay->Append(playnames[0], (void *) NULL);
228  }
229 #ifdef EXPERIMENTAL_MIDI_IN
230  if (mRecord->GetCount() == 0) {
231  recordnames.Add(_("No devices found"));
232  mRecord->Append(recordnames[0], (void *) NULL);
233  }
234 #endif
235  if (mPlay->GetCount() && mPlay->GetSelection() == wxNOT_FOUND) {
236  mPlay->SetSelection(0);
237  }
238 #ifdef EXPERIMENTAL_MIDI_IN
239  if (mRecord->GetCount() && mRecord->GetSelection() == wxNOT_FOUND) {
240  mRecord->SetSelection(0);
241  }
242 #endif
243  ShuttleGui S(this, eIsCreating);
244  S.SetSizeHints(mPlay, playnames);
245 #ifdef EXPERIMENTAL_MIDI_IN
246  S.SetSizeHints(mRecord, recordnames);
247 #endif
248 // OnDevice(e);
249 }
