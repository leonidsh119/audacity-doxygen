205 int MP3ImportFileHandle::Import(TrackFactory *trackFactory, Track ***outTracks,
206  int *outNumTracks, Tags *tags)
207 {
208  int chn;
209 
210  CreateProgress();
211 
212  /* Prepare decoder data, initialize decoder */
213 
214  mPrivateData.file = mFile;
215  mPrivateData.inputBuffer = new unsigned char [INPUT_BUFFER_SIZE];
216  mPrivateData.progress = mProgress.get();
217  mPrivateData.channels = NULL;
218  mPrivateData.updateResult= eProgressSuccess;
219  mPrivateData.id3checked = false;
220  mPrivateData.numChannels = 0;
221  mPrivateData.trackFactory= trackFactory;
222 
223  mad_decoder_init(&mDecoder, &mPrivateData, input_cb, 0, 0, output_cb, error_cb, 0);
224 
225  /* and send the decoder on its way! */
226 
227  bool res = (mad_decoder_run(&mDecoder, MAD_DECODER_MODE_SYNC) == 0) &&
228  (mPrivateData.numChannels > 0) &&
229  !(mPrivateData.updateResult == eProgressCancelled) &&
230  !(mPrivateData.updateResult == eProgressFailed);
231 
232  mad_decoder_finish(&mDecoder);
233 
234  delete[] mPrivateData.inputBuffer;
235 
236  if (!res) {
237  /* failure */
238  /* printf("failure\n"); */
239 
240  /* DELETE everything */
241  for (chn = 0; chn < mPrivateData.numChannels; chn++) {
242  delete mPrivateData.channels[chn];
243  }
244  delete[] mPrivateData.channels;
245 
246  return (mPrivateData.updateResult);
247  }
248 
249  /* success */
250  /* printf("success\n"); */
251 
252  /* copy the WaveTrack pointers into the Track pointer list that
253  * we are expected to fill */
254  *outNumTracks = mPrivateData.numChannels;
255  *outTracks = new Track* [mPrivateData.numChannels];
256  for(chn = 0; chn < mPrivateData.numChannels; chn++) {
257  mPrivateData.channels[chn]->Flush();
258  (*outTracks)[chn] = mPrivateData.channels[chn];
259  }
260  delete[] mPrivateData.channels;
261 
262  /* Read in any metadata */
263  ImportID3(tags);
264 
265  return mPrivateData.updateResult;
266  }

411 enum mad_flow input_cb(void *_data, struct mad_stream *stream)
412 {
413  struct private_data *data = (struct private_data *)_data;
414 
415  data->updateResult = data->progress->Update((wxULongLong_t)data->file->Tell(),
416  (wxULongLong_t)data->file->Length() != 0 ?
417  (wxULongLong_t)data->file->Length() : 1);
418  if(data->updateResult != eProgressSuccess)
419  return MAD_FLOW_STOP;
420 
421  if(data->file->Eof()) {
422  return MAD_FLOW_STOP;
423  }
424 
425 #ifdef USE_LIBID3TAG
426  if (!data->id3checked) {
427  data->file->Read(data->inputBuffer, ID3_TAG_QUERYSIZE);
428  int len = id3_tag_query(data->inputBuffer, ID3_TAG_QUERYSIZE);
429  if (len > 0) {
430  data->file->Seek(len, wxFromStart);
431  }
432  else {
433  data->file->Seek(0);
434  }
435 
436  data->id3checked = true;
437  }
438 #endif
439 
440  /* "Each time you refill your buffer, you need to preserve the data in
441  * your existing buffer from stream.next_frame to the end.
442  *
443  * This usually amounts to calling memmove() on this unconsumed portion
444  * of the buffer and appending NEW data after it, before calling
445  * mad_stream_buffer()"
446  * -- Rob Leslie, on the mad-dev mailing list */
447 
448  unsigned int unconsumedBytes;
449  if(stream->next_frame) {
450  unconsumedBytes = data->inputBuffer + INPUT_BUFFER_SIZE - stream->next_frame;
451  memmove(data->inputBuffer, stream->next_frame, unconsumedBytes);
452  }
453  else
454  unconsumedBytes = 0;
455 
456 
457  off_t read = data->file->Read(data->inputBuffer + unconsumedBytes,
458  INPUT_BUFFER_SIZE - unconsumedBytes);
459 
460  mad_stream_buffer(stream, data->inputBuffer, read + unconsumedBytes);
461 
462  return MAD_FLOW_CONTINUE;
463 }

493  data->channels[chn]->SetChannel(Track::MonoChannel);

499  data->channels[1]->SetChannel(Track::RightChannel);
500  data->channels[0]->SetLinked(true);

503  }
504  else {
505  // This is not the first run, protect us from libmad glitching
506  // on the number of channels
507  channels = data->numChannels;
508  }
509 
510  /* TODO: get rid of this by adding fixed-point support to SampleFormat.
511  * For now, we allocate temporary float buffers to convert the fixed
512  * point samples into something we can feed to the WaveTrack. Allocating
513  * big blocks of data like this isn't a great idea, but it's temporary.
514  */

516  for(chn = 0; chn < channels; chn++)

518 
519  for(smpl = 0; smpl < samples; smpl++)
520  for(chn = 0; chn < channels; chn++)

522 
523  for(chn = 0; chn < channels; chn++)
524  data->channels[chn]->Append((samplePtr)channelBuffers[chn],
525  floatSample,
526  samples);
527 
528  for(chn = 0; chn < channels; chn++)
529  delete[] channelBuffers[chn];
530  delete[] channelBuffers;
531 
532  return MAD_FLOW_CONTINUE;
533 }
