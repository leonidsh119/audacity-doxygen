71 void UndoManager::CalculateSpaceUsage()
72 {
73  TIMER_START( "CalculateSpaceUsage", space_calc );
74  TrackListOfKindIterator iter(Track::Wave);
75 
76  space.clear();
77  space.resize(stack.size(), 0);
78 
79  Set s1, s2;
80  Set *prev = &s1;
81  Set *cur = &s2;
82 
83  for (size_t i = 0, cnt = stack.size(); i < cnt; i++)
84  {
85  // Swap map pointers
86  std::swap(cur, prev);
87 
88  // And clean out the NEW current map
89  cur->clear();
90 
91  // Scan all tracks at current level
92  WaveTrack *wt = (WaveTrack *) iter.First(stack[i]->state.tracks.get());
93  while (wt)
94  {
95  // Scan all clips within current track
96  WaveClipList::compatibility_iterator it = wt->GetClipIterator();
97  while (it)
98  {
99  // Scan all blockfiles within current clip
100  BlockArray *blocks = it->GetData()->GetSequenceBlockArray();
101  for (size_t b = 0, cnt = blocks->size(); b < cnt; b++)
102  {
103  BlockFile *file = (*blocks)[b].f;
104 
105  // Accumulate space used by the file if the file didn't exist
106  // in the previous level
107  if (prev->count(file) == 0 && cur->count(file) == 0)
108  {
109  space[i] += file->GetSpaceUsage().GetValue();
110  }
111 
112  // Add file to current set
113  cur->insert(file);
114  }
115 
116  it = it->GetNext();
117  }
118 
119  wt = (WaveTrack *) iter.Next();
120  }
121  }
122 
123  TIMER_STOP( space_calc );
124 }

200 void UndoManager::ModifyState(const TrackList * l,
201  const SelectedRegion &selectedRegion,
202  const std::shared_ptr<Tags> &tags)
203 {
204  if (current == wxNOT_FOUND) {
205  return;
206  }
207 
208  SonifyBeginModifyState();
209  // Delete current -- not necessary, but let's reclaim space early
210  stack[current]->state.tracks.reset();
211 
212  // Duplicate
213  auto tracksCopy = std::make_unique<TrackList>(true);
214  TrackListConstIterator iter(l);
215  const Track *t = iter.First();
216  while (t) {
217  tracksCopy->Add(t->Duplicate());
218  t = iter.Next();
219  }
220 
221  // Replace
222  stack[current]->state.tracks = std::move(tracksCopy);
223  stack[current]->state.tags = tags;
224 
225  stack[current]->state.selectedRegion = selectedRegion;
226  SonifyEndModifyState();
227 }

229 void UndoManager::PushState(const TrackList * l,
230  const SelectedRegion &selectedRegion,
231  const std::shared_ptr<Tags> &tags,
232  const wxString &longDescription,
233  const wxString &shortDescription,
234  UndoPush flags)
235 {
236  unsigned int i;
237 
238  // If consolidate is set to true, group up to 3 identical operations.
239  if (((flags & UndoPush::CONSOLIDATE) != UndoPush::MINIMAL) && lastAction == longDescription &&
240  consolidationCount < 2) {
241  consolidationCount++;
242  ModifyState(l, selectedRegion, tags);
243  // MB: If the "saved" state was modified by ModifyState, reset
244  // it so that UnsavedChanges returns true.
245  if (current == saved) {
246  saved = -1;
247  }
248  return;
249  }
250 
251  consolidationCount = 0;
252 
253  i = current + 1;
254  while (i < stack.size()) {
255  RemoveStateAt(i);
256  }
257 
258  auto tracksCopy = std::make_unique<TrackList>(true);
259  TrackListConstIterator iter(l);
260  const Track *t = iter.First();
261  while (t) {
262  tracksCopy->Add(t->Duplicate());
263  t = iter.Next();
264  }
265 
266  // Assume tags was duplicted before any changes.
267  // Just save a new shared_ptr to it.
268  stack.push_back(
269  make_movable<UndoStackElem>
270  (std::move(tracksCopy),
271  longDescription, shortDescription, selectedRegion, tags)
272  );
273 
274  current++;
275 
276  if (saved >= current) {
277  saved = -1;
278  }
279 
280  lastAction = longDescription;
