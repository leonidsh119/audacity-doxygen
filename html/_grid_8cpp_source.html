432 void Grid::OnKeyDown(wxKeyEvent &event)
433 {
434  switch (event.GetKeyCode())
435  {
436  case WXK_LEFT:
437  case WXK_RIGHT:
438  {
439  int rows = GetNumberRows();
440  int cols = GetNumberCols();
441  int crow = GetGridCursorRow();
442  int ccol = GetGridCursorCol();
443 
444  if (event.GetKeyCode() == WXK_LEFT) {
445  if (crow == 0 && ccol == 0) {
446  // do nothing
447  }
448  else if (ccol == 0) {
449  SetGridCursor(crow - 1, cols - 1);
450  }
451  else {
452  SetGridCursor(crow, ccol - 1);
453  }
454  }
455  else {
456  if (crow == rows - 1 && ccol == cols - 1) {
457  // do nothing
458  }
459  else if (ccol == cols - 1) {
460  SetGridCursor(crow + 1, 0);
461  }
462  else {
463  SetGridCursor(crow, ccol + 1);
464  }
465  }
466 
467 #if wxUSE_ACCESSIBILITY
468  // Make sure the NEW cell is made available to the screen reader
469  mAx->SetCurrentCell(GetGridCursorRow(), GetGridCursorCol());
470 #endif
471  }
472  break;
473 
474  case WXK_TAB:
475  {
476  int rows = GetNumberRows();
477  int cols = GetNumberCols();
478  int crow = GetGridCursorRow();
479  int ccol = GetGridCursorCol();
480 
481  if (event.ControlDown()) {
482  int flags = wxNavigationKeyEvent::FromTab |
483  ( event.ShiftDown() ?
484  wxNavigationKeyEvent::IsBackward :
485  wxNavigationKeyEvent::IsForward );
486  Navigate(flags);
487  return;
488  }
489  else if (event.ShiftDown()) {
490  if (crow == 0 && ccol == 0) {
491  Navigate(wxNavigationKeyEvent::FromTab | wxNavigationKeyEvent::IsBackward);
492  return;
493  }
494  else if (ccol == 0) {
495  SetGridCursor(crow - 1, cols - 1);
496  }
497  else {
498  SetGridCursor(crow, ccol - 1);
499  }
500  }
501  else {
502  if (crow == rows - 1 && ccol == cols - 1) {
503  Navigate(wxNavigationKeyEvent::FromTab | wxNavigationKeyEvent::IsForward);
504  return;
505  }
506  else if (ccol == cols - 1) {
507  SetGridCursor(crow + 1, 0);
508  }
509  else {
510  SetGridCursor(crow, ccol + 1);
511  }
512  }
513  MakeCellVisible(GetGridCursorRow(), GetGridCursorCol());
514 
515 #if wxUSE_ACCESSIBILITY
516  // Make sure the NEW cell is made available to the screen reader
517  mAx->SetCurrentCell(GetGridCursorRow(), GetGridCursorCol());
518 #endif
519  }
520  break;
521 
522  case WXK_RETURN:
523  case WXK_NUMPAD_ENTER:
524  {
525  if (!IsCellEditControlShown()) {
526  wxTopLevelWindow *tlw = wxDynamicCast(wxGetTopLevelParent(this), wxTopLevelWindow);
527  wxWindow *def = tlw->GetDefaultItem();
528  if (def && def->IsEnabled()) {
529  wxCommandEvent cevent(wxEVT_COMMAND_BUTTON_CLICKED,
530  def->GetId());
531  GetParent()->GetEventHandler()->ProcessEvent(cevent);
532  }
533  }
534  else {
535  wxGrid::OnKeyDown(event);
536 
537  // This looks strange, but what it does is selects the cell when
538  // enter is pressed after editing. Without it, Jaws and Window-Eyes
539  // do not speak the NEW cell contents (the one below the edited one).
540  SetGridCursor(GetGridCursorRow(), GetGridCursorCol());
541  }
542  break;
543  }
544 
545  default:
546  wxGrid::OnKeyDown(event);
547  break;
548  }
549 }
