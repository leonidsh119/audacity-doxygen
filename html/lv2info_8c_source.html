347 int
348 main(int argc, char** argv)
349 {
350  if (argc == 1) {
351  print_usage();
352  return 1;
353  }
354 
355  const char* plugin_file = NULL;
356  const char* manifest_file = NULL;
357  const char* plugin_uri = NULL;
358  for (int i = 1; i < argc; ++i) {
359  if (!strcmp(argv[i], "--version")) {
360  print_version();
361  return 0;
362  } else if (!strcmp(argv[i], "--help")) {
363  print_usage();
364  return 0;
365  } else if (!strcmp(argv[i], "-p")) {
366  plugin_file = argv[++i];
367  } else if (!strcmp(argv[i], "-m")) {
368  manifest_file = argv[++i];
369  } else if (argv[i][0] == '-') {
370  print_usage();
371  return 1;
372  } else if (i == argc - 1) {
373  plugin_uri = argv[i];
374  }
375  }
376 
377  int ret = 0;
378 
379  LilvWorld* world = lilv_world_new();
380  lilv_world_load_all(world);
381 
382  LilvNode* uri = lilv_new_uri(world, plugin_uri);
383  if (!uri) {
384  fprintf(stderr, "Invalid plugin URI\n");
385  lilv_world_free(world);
386  return 1;
387  }
388 
389  applies_to_pred = lilv_new_uri(world, LV2_CORE__appliesTo);
390  control_class = lilv_new_uri(world, LILV_URI_CONTROL_PORT);
391  event_class = lilv_new_uri(world, LILV_URI_EVENT_PORT);
392  group_pred = lilv_new_uri(world, LV2_PORT_GROUPS__group);
393  label_pred = lilv_new_uri(world, LILV_NS_RDFS "label");
394  preset_class = lilv_new_uri(world, LV2_PRESETS__Preset);
395  designation_pred = lilv_new_uri(world, LV2_CORE__designation);
396  supports_event_pred = lilv_new_uri(world, LV2_EVENT__supportsEvent);
397 
398  const LilvPlugins* plugins = lilv_world_get_all_plugins(world);
399  const LilvPlugin* p = lilv_plugins_get_by_uri(plugins, uri);
400 
401  if (p && plugin_file) {
402  LilvNode* base = lilv_new_file_uri(world, NULL, plugin_file);
403 
404  FILE* plugin_fd = fopen(plugin_file, "a");
405  if (plugin_fd) {
406  lilv_plugin_write_description(world, p, base, plugin_fd);
407  fclose(plugin_fd);
408  } else {
409  fprintf(stderr, "error: Failed to open %s\n", plugin_file);
410  }
411 
412  if (manifest_file) {
413  FILE* manifest_fd = fopen(manifest_file, "a");
414  if (manifest_fd) {
415  lilv_plugin_write_manifest_entry(
416  world, p, base, manifest_fd, plugin_file);
417  fclose(manifest_fd);
418  } else {
419  fprintf(stderr, "error: Failed to open %s\n", manifest_file);
420  }
421  }
422  lilv_node_free(base);
423  } else if (p) {
424  print_plugin(world, p);
425  } else {
426  fprintf(stderr, "Plugin not found.\n");
427  }
428 
429  ret = (p != NULL ? 0 : -1);
430 
431  lilv_node_free(uri);
432 
433  lilv_node_free(supports_event_pred);
434  lilv_node_free(designation_pred);
435  lilv_node_free(preset_class);
436  lilv_node_free(label_pred);
437  lilv_node_free(group_pred);
438  lilv_node_free(event_class);
439  lilv_node_free(control_class);
440  lilv_node_free(applies_to_pred);
441 
442  lilv_world_free(world);
443  return ret;
444 }
