62 static bool
63 x_window_is_valid(SuilX11Wrapper* socket)
64 {
65  GdkWindow* window = gtk_widget_get_window(GTK_WIDGET(socket->plug));
66  Window root = 0;
67  Window parent = 0;
68  Window* children = NULL;
69  unsigned childcount = 0;
70 
71  XQueryTree(GDK_WINDOW_XDISPLAY(window),
72  GDK_WINDOW_XID(window),
73  &root, &parent, &children, &childcount);
74  for (unsigned i = 0; i < childcount; ++i) {
75  if (children[i] == (Window)socket->instance->ui_widget) {
76  return true;
77  }
78  }
79  return false;
80 }

142 static void
143 forward_key_event(SuilX11Wrapper* socket,
144  GdkEvent* gdk_event)
145 {
146  GdkWindow* window = gtk_widget_get_window(GTK_WIDGET(socket->plug));
147  GdkScreen* screen = gdk_visual_get_screen(gdk_window_get_visual(window));
148 
149  XKeyEvent xev;
150  memset(&xev, 0, sizeof(xev));
151  xev.type = (gdk_event->type == GDK_KEY_PRESS) ? KeyPress : KeyRelease;
152  xev.root = GDK_WINDOW_XID(gdk_screen_get_root_window(screen));
153  xev.window = GDK_WINDOW_XID(window);
154  xev.subwindow = None;
155  xev.time = gdk_event->key.time;
156  xev.state = gdk_event->key.state;
157  xev.keycode = gdk_event->key.hardware_keycode;
158 
159  XSendEvent(GDK_WINDOW_XDISPLAY(window),
160  (Window)socket->instance->ui_widget,
161  False,
162  NoEventMask,
163  (XEvent*)&xev);
164 }

338  LV2_URID ui_updateRate = map->map(map->handle, LV2_UI__updateRate);

341  wrap->idle_ms = 1000.0f / *(const float*)o->value;
