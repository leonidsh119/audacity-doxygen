115 static void
116 write_statement(SordModel* sord,
117  SerdWriter* writer,
118  SordQuad tup,
119  SerdStatementFlags flags)
120 {
121  const SordNode* s = tup[SORD_SUBJECT];
122  const SordNode* p = tup[SORD_PREDICATE];
123  const SordNode* o = tup[SORD_OBJECT];
124  const SordNode* d = sord_node_get_datatype(o);
125  const SerdNode* ss = sord_node_to_serd_node(s);
126  const SerdNode* sp = sord_node_to_serd_node(p);
127  const SerdNode* so = sord_node_to_serd_node(o);
128  const SerdNode* sd = sord_node_to_serd_node(d);
129 
130  const char* lang_str = sord_node_get_language(o);
131  size_t lang_len = lang_str ? strlen(lang_str) : 0;
132  SerdNode language = SERD_NODE_NULL;
133  if (lang_str) {
134  language.type = SERD_LITERAL;
135  language.n_bytes = lang_len;
136  language.n_chars = lang_len;
137  language.buf = (const uint8_t*)lang_str;
138  };
139 
140  // TODO: Subject abbreviation
141 
142  if (sord_node_is_inline_object(s) && !(flags & SERD_ANON_CONT)) {
143  return;
144  }
145 
146  SerdStatus st = SERD_SUCCESS;
147  if (sord_node_is_inline_object(o)) {
148  SordQuad sub_pat = { o, 0, 0, 0 };
149  SordIter* sub_iter = sord_find(sord, sub_pat);
150 
151  SerdStatementFlags start_flags = flags
152  | ((sub_iter) ? SERD_ANON_O_BEGIN : SERD_EMPTY_O);
153 
154  st = serd_writer_write_statement(
155  writer, start_flags, NULL, ss, sp, so, sd, &language);
156 
157  if (!st && sub_iter) {
158  flags |= SERD_ANON_CONT;
159  for (; !sord_iter_end(sub_iter); sord_iter_next(sub_iter)) {
160  SordQuad sub_tup;
161  sord_iter_get(sub_iter, sub_tup);
162  write_statement(sord, writer, sub_tup, flags);
163  }
164  sord_iter_free(sub_iter);
165  serd_writer_end_anon(writer, so);
166  }
167  } else {
168  st = serd_writer_write_statement(
169  writer, flags, NULL, ss, sp, so, sd, &language);
170  }
171 
172  if (st) {
173  fprintf(stderr, "Failed to write statement (%s)\n",
174  serd_strerror(st));
175  return;
176  }
177 }
