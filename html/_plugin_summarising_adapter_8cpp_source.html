283 Plugin::FeatureList
284 PluginSummarisingAdapter::Impl::getSummaryForOutput(int output,
285  SummaryType type,
286  AveragingMethod avg)
287 {
288  if (!m_reduced) {
289  accumulateFinalDurations();
290  segment();
291  reduce();
292  m_reduced = true;
293  }
294 
295  bool continuous = (avg == ContinuousTimeAverage);
296 
297  FeatureList fl;
298  for (SummarySegmentMap::const_iterator i = m_summaries[output].begin();
299  i != m_summaries[output].end(); ++i) {
300 
301  Feature f;
302 
303  f.hasTimestamp = true;
304  f.timestamp = i->first;
305 
306  f.hasDuration = true;
307  SummarySegmentMap::const_iterator ii = i;
308  if (++ii == m_summaries[output].end()) {
309  f.duration = m_endTime - f.timestamp;
310  } else {
311  f.duration = ii->first - f.timestamp;
312  }
313 
314  f.label = getSummaryLabel(type, avg);
315 
316  for (OutputSummary::const_iterator j = i->second.begin();
317  j != i->second.end(); ++j) {
318 
319  // these will be ordered by bin number, and no bin numbers
320  // will be missing except at the end (because of the way
321  // the accumulators were initially filled in accumulate())
322 
323  const OutputBinSummary &summary = j->second;
324  double result = 0.f;
325 
326  switch (type) {
327 
328  case Minimum:
329  result = summary.minimum;
330  break;
331 
332  case Maximum:
333  result = summary.maximum;
334  break;
335 
336  case Mean:
337  if (continuous) {
338  result = summary.mean_c;
339  } else if (summary.count) {
340  result = summary.sum / summary.count;
341  }
342  break;
343 
344  case Median:
345  if (continuous) result = summary.median_c;
346  else result = summary.median;
347  break;
348 
349  case Mode:
350  if (continuous) result = summary.mode_c;
351  else result = summary.mode;
352  break;
353 
354  case Sum:
355  result = summary.sum;
356  break;
357 
358  case Variance:
359  if (continuous) result = summary.variance_c;
360  else result = summary.variance;
361  break;
362 
363  case StandardDeviation:
364  if (continuous) result = sqrtf(summary.variance_c);
365  else result = sqrtf(summary.variance);
366  break;
367 
368  case Count:
369  result = summary.count;
370  break;
371 
372  case UnknownSummaryType:
373  break;
374 
375  default:
376  break;
377  }
378 
379  f.values.push_back(result);
380  }
381 
382  fl.push_back(f);
383  }
384  return fl;
385 }

450 void
451 PluginSummarisingAdapter::Impl::accumulate(int output,
452  const Feature &f,
453  RealTime timestamp,
454  bool final)
455 {
456  // What should happen if a feature's duration spans a segment
457  // boundary? I think we probably want to chop it, and pretend
458  // that it appears in both. A very long feature (e.g. key, if the
459  // whole audio is in a single key) might span many or all
460  // segments, and we want that to be reflected in the results
461  // (e.g. it is the modal key in all of those segments, not just
462  // the first). This is actually quite complicated to do.
463 
464  // If features spanning a boundary should be chopped, then we need
465  // to have per-segment accumulators (and the feature value goes
466  // into both -- with a separate phase to split the accumulator up
467  // into segments).
468 
469 #ifdef DEBUG_PLUGIN_SUMMARISING_ADAPTER
470  std::cerr << "output " << output << ": timestamp " << timestamp << ", prev timestamp " << m_prevTimestamps[output] << ", final " << final << std::endl;
471 #endif
472 
473  // At each process step, accumulate() is called once for each
474  // feature on each output within that process's returned feature
475  // list, and with the timestamp passed in being that of the start
476  // of the process block.
477 
478  // At the end (in getRemainingFeatures), accumulate() is called
479  // once for each feature on each output within the feature list
480  // returned by getRemainingFeatures, and with the timestamp being
481  // the same as the last process block and final set to true.
482 
483  // (What if getRemainingFeatures doesn't return any features? We
484  // still need to ensure that the final duration is written. Need
485  // a separate function to close the durations.)
486 
487  // At each call, we pull out the value for the feature and stuff
488  // it into the accumulator's appropriate values array; and we
489  // calculate the duration for the _previous_ feature, or pull it
490  // from the prevDurations array if the previous feature had a
491  // duration in its structure, and stuff that into the
492  // accumulator's appropriate durations array.
493 
494  if (m_prevDurations.find(output) != m_prevDurations.end()) {
495 
496  // Not the first time accumulate has been called for this
497  // output -- there has been a previous feature
498 
499  RealTime prevDuration;
500 
501  // Note that m_prevDurations[output] only contains the
502  // duration field that was contained in the previous feature.
503  // If it didn't have an explicit duration,
504  // m_prevDurations[output] should be INVALID_DURATION and we
505  // will have to calculate the duration from the previous and
506  // current timestamps.
507 
508  if (m_prevDurations[output] != INVALID_DURATION) {
509  prevDuration = m_prevDurations[output];
510 #ifdef DEBUG_PLUGIN_SUMMARISING_ADAPTER
511  std::cerr << "Previous duration from previous feature: " << prevDuration << std::endl;
512 #endif
513  } else {
514  prevDuration = timestamp - m_prevTimestamps[output];
515 #ifdef DEBUG_PLUGIN_SUMMARISING_ADAPTER
516  std::cerr << "Previous duration from diff: " << timestamp << " - "
517  << m_prevTimestamps[output] << std::endl;
518 #endif
519  }
520 
521 #ifdef DEBUG_PLUGIN_SUMMARISING_ADAPTER
522  std::cerr << "output " << output << ": ";
523  std::cerr << "Pushing previous duration as " << prevDuration << std::endl;
524 #endif
525 
526  m_accumulators[output].results
527  [m_accumulators[output].results.size() - 1]
528  .duration = prevDuration;
529  }
530 
531  if (f.hasDuration) m_prevDurations[output] = f.duration;
532  else m_prevDurations[output] = INVALID_DURATION;
533 
534  m_prevTimestamps[output] = timestamp;
535 
536  if (f.hasDuration) {
537  RealTime et = timestamp;
538  et = et + f.duration;
539  if (et > m_endTime) m_endTime = et;
540  }
541 
542  Result result;
543  result.time = timestamp;
544  result.duration = INVALID_DURATION;
545 
546  if (int(f.values.size()) > m_accumulators[output].bins) {
547  m_accumulators[output].bins = f.values.size();
548  }
549 
550  for (int i = 0; i < int(f.values.size()); ++i) {
551  result.values.push_back(f.values[i]);
552  }
553 
554  m_accumulators[output].results.push_back(result);
555 }

557 void
558 PluginSummarisingAdapter::Impl::accumulateFinalDurations()
559 {
560  for (OutputTimestampMap::iterator i = m_prevTimestamps.begin();
561  i != m_prevTimestamps.end(); ++i) {
562 
563  int output = i->first;
564 
565  int acount = m_accumulators[output].results.size();
566 
567  if (acount == 0) continue;
568 
569 #ifdef DEBUG_PLUGIN_SUMMARISING_ADAPTER
570  std::cerr << "output " << output << ": ";
571 #endif
572 
573  if (m_prevDurations.find(output) != m_prevDurations.end() &&
574  m_prevDurations[output] != INVALID_DURATION) {
575 
576 #ifdef DEBUG_PLUGIN_SUMMARISING_ADAPTER
577  std::cerr << "Pushing final duration from feature as " << m_prevDurations[output] << std::endl;
578 #endif
579 
580  m_accumulators[output].results[acount - 1].duration =
581  m_prevDurations[output];
582 
583  } else {
584 
585 #ifdef DEBUG_PLUGIN_SUMMARISING_ADAPTER
586  std::cerr << "Pushing final duration from diff as " << m_endTime << " - " << m_prevTimestamps[output] << std::endl;
587 #endif
588 
589  m_accumulators[output].results[acount - 1].duration =
590  m_endTime - m_prevTimestamps[output];
591  }
592 
593 #ifdef DEBUG_PLUGIN_SUMMARISING_ADAPTER
594  std::cerr << "so duration for result no " << acount-1 << " is "
595  << m_accumulators[output].results[acount-1].duration
596  << std::endl;
597 #endif
598  }

629 void
630 PluginSummarisingAdapter::Impl::segment()
631 {
632  SegmentBoundaries::iterator boundaryitr = m_boundaries.begin();
633 
634 #ifdef DEBUG_PLUGIN_SUMMARISING_ADAPTER_SEGMENT
635  std::cerr << "segment: starting" << std::endl;
636 #endif
637 
638  for (OutputAccumulatorMap::iterator i = m_accumulators.begin();
639  i != m_accumulators.end(); ++i) {
640 
641  int output = i->first;
642  OutputAccumulator &source = i->second;
643 
644 #ifdef DEBUG_PLUGIN_SUMMARISING_ADAPTER_SEGMENT
645  std::cerr << "segment: total results for output " << output << " = "
646  << source.results.size() << std::endl;
647 #endif
648 
649  // This is basically nonsense if the results have no values
650  // (i.e. their times and counts are the only things of
651  // interest)... but perhaps it's the user's problem if they
652  // ask for segmentation (or any summary at all) in that case
653 
654  for (int n = 0; n < int(source.results.size()); ++n) {
655 
656  // This result spans source.results[n].time to
657  // source.results[n].time + source.results[n].duration.
658  // We need to dispose it into segments appropriately
659 
660  RealTime resultStart = source.results[n].time;
661  RealTime resultEnd = resultStart + source.results[n].duration;
662 
663 #ifdef DEBUG_PLUGIN_SUMMARISING_ADAPTER_SEGMENT
664  std::cerr << "output: " << output << ", result start = " << resultStart << ", end = " << resultEnd << std::endl;
665 #endif
666 
667  RealTime segmentStart = RealTime::zeroTime;
668  RealTime segmentEnd = resultEnd - RealTime(1, 0);
669 
670  RealTime prevSegmentStart = segmentStart - RealTime(1, 0);
671 
672  while (segmentEnd < resultEnd) {
673 
674 #ifdef DEBUG_PLUGIN_SUMMARISING_ADAPTER_SEGMENT
675  std::cerr << "segment end " << segmentEnd << " < result end "
676  << resultEnd << " (with result start " << resultStart << ")" << std::endl;
677 #endif
678 
679  findSegmentBounds(resultStart, segmentStart, segmentEnd);
680 
681  if (segmentStart == prevSegmentStart) {
682  // This can happen when we reach the end of the
683  // input, if a feature's end time overruns the
684  // input audio end time
685  break;
686  }
687  prevSegmentStart = segmentStart;
688 
689  RealTime chunkStart = resultStart;
690  if (chunkStart < segmentStart) chunkStart = segmentStart;
691 
692  RealTime chunkEnd = resultEnd;
693  if (chunkEnd > segmentEnd) chunkEnd = segmentEnd;
694 
695  m_segmentedAccumulators[output][segmentStart].bins = source.bins;
696 
697  Result chunk;
698  chunk.time = chunkStart;
699  chunk.duration = chunkEnd - chunkStart;
700  chunk.values = source.results[n].values;
701 
702 #ifdef DEBUG_PLUGIN_SUMMARISING_ADAPTER_SEGMENT
703  std::cerr << "chunk for segment " << segmentStart << ": from " << chunk.time << ", duration " << chunk.duration << std::endl;
704 #endif
705 
706  m_segmentedAccumulators[output][segmentStart].results
707  .push_back(chunk);
708 
709  resultStart = chunkEnd;
710  }
711  }
712  }

767  << std::endl;
768 #endif

770  accumulator.results[sz-1].duration) -

772  }

775 
776  // work on all values over time for a single bin
777 
778  OutputBinSummary summary;
779

781

798 
799  for (int k = 0; k < sz; ++k) {
800  while (int(accumulator.results[k].values.size()) <
801  accumulator.bins) {
802  accumulator.results[k].values.push_back(0.f);
803  }
804  }
805 
806  for (int k = 0; k < sz; ++k) {
807  float value = accumulator.results[k].values[bin];
808  valvec.push_back(ValueDurationFloatPair
809  (value,
810  toSec(accumulator.results[k].duration)));
811  }
812 
813  std::sort(valvec.begin(), valvec.end());
814 
815  summary.minimum = valvec[0].value;
816  summary.maximum = valvec[sz-1].value;
817

820 #endif

829 */
830 #endif
831

833  summary.median = valvec[sz/2].value;
834  } else {
835  summary.median = (valvec[sz/2].value + valvec[sz/2 + 1].value) / 2;
836  }
837

839  summary.median_c = valvec[sz-1].value;
840

842  duracc += valvec[k].duration;
843  if (duracc > totalDuration/2) {
844  summary.median_c = valvec[k].value;
845  break;
846  }
847  }
848 
849 #ifdef DEBUG_PLUGIN_SUMMARISING_ADAPTER
850  std::cerr << "median_c = " << summary.median_c << std::endl;
851  std::cerr << "median = " << summary.median << std::endl;
852 #endif
853

855 
856  for (int k = 0; k < sz; ++k) {
857  summary.sum += accumulator.results[k].values[bin];
858  distribution[accumulator.results[k].values[bin]] += 1;
859  }
860 
861  int md = 0;
862 
863  for (std::map<float, int>::iterator di = distribution.begin();
864  di != distribution.end(); ++di) {
865  if (di->second > md) {
866  md = di->second;
867  summary.mode = di->first;
868  }
869  }
870 
871  distribution.clear();
872 
873  std::map<float, double> distribution_c;
874 
875  for (int k = 0; k < sz; ++k) {
876  distribution_c[accumulator.results[k].values[bin]]
877  += toSec(accumulator.results[k].duration);
878  }
879 
880  double mrd = 0.0;
881 
882  for (std::map<float, double>::iterator di = distribution_c.begin();
883  di != distribution_c.end(); ++di) {
884  if (di->second > mrd) {
885  mrd = di->second;
886  summary.mode_c = di->first;
887  }
888  }
889 
890  distribution_c.clear();
891 
892  if (totalDuration > 0.0) {
893 
894  double sum_c = 0.0;
895 
896  for (int k = 0; k < sz; ++k) {
897  double value = accumulator.results[k].values[bin]
898  * toSec(accumulator.results[k].duration);
899  sum_c += value;
900  }
901 
902 #ifdef DEBUG_PLUGIN_SUMMARISING_ADAPTER
903  std::cerr << "mean_c = " << sum_c << " / " << totalDuration << " = "
904  << sum_c / totalDuration << " (sz = " << sz << ")" << std::endl;
905 #endif
906 
907  summary.mean_c = sum_c / totalDuration;
908 
909  for (int k = 0; k < sz; ++k) {
910  double value = accumulator.results[k].values[bin];
911 // * toSec(accumulator.results[k].duration);
912  summary.variance_c +=
913  (value - summary.mean_c) * (value - summary.mean_c)
914  * toSec(accumulator.results[k].duration);
915  }
916 
917 // summary.variance_c /= summary.count;
918  summary.variance_c /= totalDuration;
919  }
920 
921  double mean = summary.sum / summary.count;
922 
923 #ifdef DEBUG_PLUGIN_SUMMARISING_ADAPTER
924  std::cerr << "mean = " << summary.sum << " / " << summary.count << " = "
925  << summary.sum / summary.count << std::endl;
926 #endif
927 
928  for (int k = 0; k < sz; ++k) {
929  float value = accumulator.results[k].values[bin];
930  summary.variance += (value - mean) * (value - mean);
931  }
932  summary.variance /= summary.count;
933 
934  m_summaries[output][segmentStart][bin] = summary;
935  }
936  }
