781 def should_follow(url):
782  """
783  Returns a boolean for whether url should be spidered
784 
785  Given that 'url' was linked to from site, return whether
786  'url' should be spidered as well.
787  """
788  global config
789 
790  # we don't have search on the local version
791  if (url.endswith('#searchInput')):
792  return False
793 
794  # False if different domains.
795  nurl = normalize_url(url)
796  droot = get_domain(config.rooturl)
797  dn = get_domain(nurl)
798  if droot != dn and not (dn.endswith(droot) or droot.endswith(dn)):
799  if config.debug:
800  print url, 'not in the same domain'
801  return False
802 
803  # False if multiple query fields or parameters found
804  if (url.count('&') >= 1 or url.count(';') > 0) and not any(x in url for x in ('.css', 'gen=css')):
805  if config.debug:
806  print url, 'with multiple query fields'
807  return False
808 
809  if any(x in url for x in ('Special:', 'Image:', 'Talk:', 'User:', 'Help:', 'User_talk:', 'MediaWiki_talk:', 'File:', 'action=edit', 'title=-')):
810  if config.debug:
811  print url, 'is a forbidden wiki page'
812  return False
813 
814  if config.no_images and any(url.strip().lower().endswith(suffix) for suffix in ('.jpg', '.gif', '.png', '.ico')):
815  if config.debug:
816  print url, 'is a image and you are in no-images mode'
817  return False
818 
819  if any(url.strip().lower().endswith(suffix) for suffix in ('.zip', '.7z')):
820  if config.debug:
821  print url, 'is a compressed file'
822  return False
823 
824 
825  # limit_parent support
826  ncurl = normalize_url(config.rooturl)
827 
828  if config.limit_parent and not nurl.startswith(ncurl):
829  L = nurl.split('/')
830  if ('.' not in L[-1]):
831  if config.debug:
832  print url, 'is a file outside of scope with unknown extension'
833  return False
834 
835  # JKC: we do allow css from 'strange' places.
836  if '.css' in L[-1]:
837  return True
838 
839  forbidden_parents = ['.php', '.html', '.htm']
840  for fp in forbidden_parents:
841  if fp in L[-1]:
842  if config.debug:
843  print url, 'is a page outside of scope'
844  return False
845

848 def parse_html(doc, url):
849  """
850  Returns (modified_doc, new_urls), where new_urls are absolute URLs for
851  all links we want to spider in the HTML.
852  """
853  global config
854 
855  BEGIN_COMMENT_REPLACE = '<BEGINCOMMENT-' + str(random.random()) + '>'
856  END_COMMENT_REPLACE = '<ENDCOMMENT-' + str(random.random()) + '>'
857 
858  new_urls = []
859 
860  doc = pre_html_transform(doc, url)
861  # Temporarily "get rid" of comments so htmldata will find the URLs
862  # in the funky "<!--[if" HTML hackery for IE.
863  doc = doc.replace('<!--', BEGIN_COMMENT_REPLACE)
864  doc = doc.replace('-->', END_COMMENT_REPLACE)
865 
866  L = htmldata.urlextract(doc, url, 'text/html')
867 
868  # in this code we change each absolute url in L
869  # into a relative one.
870  # we also kick-off zillions of subthreads to collect
871  # more pages.
872  for item in L:
873  u = item.url
874  follow = should_follow(u)
875  if follow:
876  if config.debug:
877  print 'ACCEPTED - ', u
878  # Store url locally.
879  new_urls += [u]
880  item.url = url_to_relative(u, url)
881  else:
882  # James, let's keep everything by default (but not follow it).
883  # if not any( license in u for license in ('creativecommons.org', 'wxwidgets.org', 'gnu.org', 'mediawiki.org') ):
884  # item.url = ''
885  if config.debug:
886  print 'NOT INCLUDED - ', u
887 
888  newdoc = htmldata.urljoin(doc, L)
889  newdoc = newdoc.replace(BEGIN_COMMENT_REPLACE, '<!--')
890  newdoc = newdoc.replace(END_COMMENT_REPLACE, '-->')
891 
892  newdoc = pos_html_transform(newdoc, url)
893 
894  return (newdoc, new_urls)

897 def run(out=sys.stdout):
898  """
899  Code interface.
900  """
901  global conn, domain, counter, redir_cache, config
902 
903  if urlparse.urlparse(config.rooturl)[1].lower().endswith('wikipedia.org'):
904  out.write('Please do not use robots with the Wikipedia site.\n')

909  # Number of files saved
910  n = 0
911 
912  if not config.overwrite and os.path.exists(config.outdir):
913  out.write('Error: Directory exists: ' + str(config.outdir))
914  sys.exit(1)
915 
916  domain = get_domain(config.rooturl)
917  conn = httplib.HTTPConnection(domain)
918  print 'connection established to:', domain

920  pending = set([config.rooturl])
921

933  continue
934 
935  complete.add(nurl)

937 
938  #this is needed for the first path as it doesn't know if it is a redirect or not in the begining

941  start = False

943  for redir in redir_cache.iterkeys():
944  aux_url = normalize_url(redir)

947  complete.add(aux_url)

949  nurl = normalize_url(redir_cache[nurl])
950

952  continue

955  print "ERROR: ", url, '\n'
956  continue

959  doc = f.read()
960  f.close()
961  new_urls = []
962 
963  if filename.endswith('.html'):
964  (doc, new_urls) = parse_html(doc, url)
965  elif filename.endswith('.css'):
966  (doc, new_urls) = parse_css(doc, url)
967 
968  # Save document changes to disk
969  # The unmodified file already exists on disk.
970  update = False
971  text_ext = ('txt', 'html', 'rtf', 'css', 'sgml', 'xml')
972  for ext in text_ext:
973  if filename.endswith(ext):
974  update = True
975  break
976 
977  if update:
978  f = open(filename, 'w')
979  f.write(doc)
980  f.close()
981 
982  if config.debug:
983  out.write(url + '\n => ' + filename + '\n\n')
984  n += 1
985 
986  # Enqueue URLs that we haven't yet spidered.
987  for u in new_urls:
988  if normalize_url(u) not in complete:
989  # Strip off any #section link.
990  if '#' in u:
991  u = u[:u.index('#')]
992  pending.add(u)
993 
994  conn.close()
995  print "connection to", domain, "closed."
996  out.write(str(n) + ' files saved\n')

998  print errors, "errors not recovered"
