1 /* trigger.c -- return zero until input transitions from <=0 to >0, then
2  evaluate a closure to get a signal and convert to an add
3  of the new signal and a copy of this trigger object.
4  The sample rate of the output is the sample rate of the input, and
5  sounds returned by the closure must also have a matching sample rate.
6  The trigger will take place on the first input sample (zero delay) if the
7  first sample of the input is >0.
8  The input scale factor is assumed to be 1, so caller should force scaling
9  especially if the scale factor is negative (!)
10  The trigger terminates when the input signal terminates (but any adds
11  continue to run until all their inputs terminate).
12 
13 Some implementation notes:
14 
15 The closure gets evaluated at the time of the positive sample.
16 When the positive sample is encountered, first close off the
17 current output block.
18 
19 Next, evaluate the closure, clone the trigger, and convert
20 the current trigger to an add. The next fetch will therefore
21 go to the add susp and it will add the closure result to the
22 zeros that continue to be generated by (a clone of) the trigger.
23 It would be simple if we could back the clone up one sample:
24 on the first call to the add, it will ask for samples from the
25 trigger clone and the closure, but the trigger clone has already
26 processed the positive sample, so it is one sample ahead of
27 everyone else. Since we've just read a sample, we CAN back up
28 just by carefully updating the input pointer to one less than
29 we actually read, forcing a reread later. (We'll still store
30 the previous value so re-reading will not re-trigger.)
31 */

99 void trigger_fetch(snd_susp_type a_susp, snd_list_type snd_list)
100 {
101  trigger_susp_type susp = (trigger_susp_type) a_susp;
102  int cnt = 0; /* how many samples computed */
103  int togo;
104  int n;
105  sample_block_type out;
106  register sample_block_values_type out_ptr;
107  register sample_block_values_type out_ptr_reg;
108  register sample_block_values_type input_ptr_reg;
109  falloc_sample_block(out, "trigger_fetch");
110  out_ptr = out->samples;
111  snd_list->block = out;
112 
113  while (cnt < max_sample_block_len) { /* outer loop */
114  /* first compute how many samples to generate in inner loop: */
115  /* don't overflow the output sample block */
116  togo = max_sample_block_len - cnt;
117 
118  /* don't run past the input sample block: */
119  susp_check_term_samples(s1, s1_ptr, s1_cnt);
120  togo = min(togo, susp->s1_cnt);
121 
122  /* don't run past terminate time */
123  if (susp->terminate_cnt != UNKNOWN &&
124  susp->terminate_cnt <= susp->susp.current + cnt + togo) {
125  togo = susp->terminate_cnt - (susp->susp.current + cnt);
126  if (togo == 0) break;
127  }
128 
129  n = togo;
130  input_ptr_reg = susp->s1_ptr;
131  out_ptr_reg = out_ptr;
132  if (n) do { /* the inner sample computation loop */
133  sample_type s = *input_ptr_reg++;
134  if (susp->previous <= 0 && s > 0) {
135  trigger_susp_type new_trigger;
136  sound_type new_trigger_snd;
137  LVAL result;
138  long delay; /* sample delay to s2 */
139  time_type now;
140 
141  susp->previous = s; /* don't retrigger */
142 
143  /**** close off block ****/
144  togo = togo - n;
145  susp->s1_ptr += togo;
146  susp_took(s1_cnt, togo);
147  cnt += togo;
148  snd_list->block_len = cnt;
149  susp->susp.current += cnt;
150  now = susp->susp.t0 + susp->susp.current / susp->susp.sr;
151 
152  /**** eval closure and add result ****/
153 D nyquist_printf("trigger_fetch: about to eval closure at %g, "
154  "susp->susp.t0 %g, susp.current %d:\n",
155  now, susp->susp.t0, (int)susp->susp.current);
156  xlsave1(result);
157  result = xleval(cons(susp->closure, consa(cvflonum(now))));
158  if (exttypep(result, a_sound)) {
159  susp->s2 = sound_copy(getsound(result));
160 D nyquist_printf("trigger: copied result from closure is %p\n",
161  susp->s2);
162  } else xlerror("closure did not return a (monophonic) sound",
163  result);
164 D nyquist_printf("in trigger: after evaluation; "
165  "%p returned from evform\n",
166  susp->s2);
167  result = NIL;
168 
169  /**** cloan this trigger to become s1 ****/
170  falloc_generic(new_trigger, trigger_susp_node,
171  "new_trigger");
172  memcpy(new_trigger, susp, sizeof(trigger_susp_node));
173  /* don't copy s2 -- it should only be referenced by add */
174  new_trigger->s2 = NULL;
175  new_trigger_snd = sound_create((snd_susp_type) new_trigger,
176  now, susp->susp.sr, 1.0F);
177  susp->s1 = new_trigger_snd;
178  /* add will have to ask new_trigger for samples, new_trigger
179  * will continue reading samples from s1 (the original input)
180  */
181  susp->s1_cnt = 0;
182  susp->s1_ptr = NULL;
183 
184  /**** convert to add ****/
185  susp->susp.mark = add_mark;
186  /* logical stop will be recomputed by add: */
187  susp->susp.log_stop_cnt = UNKNOWN;
188  susp->susp.print_tree = add_print_tree;
189 
190  /* assume sample rates are the same */
191  if (susp->s1->sr != susp->s2->sr)
192  xlfail("in trigger: sample rates must match");
193 
194  /* take care of scale factor, if any */
195  if (susp->s2->scale != 1.0) {
196  // stdputstr("normalizing next sound in a seq\n");
197  susp->s2 = snd_make_normalize(susp->s2);
198  }
199 
200  /* figure out which add fetch routine to use */
201  delay = ROUND((susp->s2->t0 - now) * susp->s1->sr);
202  if (delay > 0) { /* fill hole between s1 and s2 */
203  D stdputstr("using add_s1_nn_fetch\n");
204  susp->susp.fetch = add_s1_nn_fetch;
205  susp->susp.name = "trigger:add_s1_nn_fetch";
206  } else {
207  susp->susp.fetch = add_s1_s2_nn_fetch;
208  susp->susp.name = "trigger:add_s1_s2_nn_fetch";
209  }
210 
211 D stdputstr("in trigger: calling add's fetch\n");
212  /* fetch will get called later ..
213  (*(susp->susp.fetch))(a_susp, snd_list); */
214 D stdputstr("in trigger: returned from add's fetch\n");
215  xlpop();
216 
217  susp->closure = NULL; /* allow garbage collection now */
218  /**** calculation tree modified, time to exit ****/
219  /* but if cnt == 0, then we haven't computed any samples */
220  /* call on new fetch routine to get some samples */
221  if (cnt == 0) {
222  // because adder will reallocate
223  ffree_sample_block(out, "trigger-pre-adder");
224  (*susp->susp.fetch)(a_susp, snd_list);
225  }
226  return;
227  } else {
228  susp->previous = s;
229  /* output zero until ready to add in closure */
230  *out_ptr_reg++ = 0;
231  }
232  } while (--n); /* inner loop */
233 
234  /* using input_ptr_reg is a bad idea on RS/6000: */
235  susp->s1_ptr += togo;
236  out_ptr += togo;
237  susp_took(s1_cnt, togo);
238  cnt += togo;
239  } /* outer loop */
240 
241  if (togo == 0 && cnt == 0) {
242  snd_list_terminate(snd_list);
243  } else {
244  snd_list->block_len = cnt;
245  susp->susp.current += cnt;
246  }
247 } /* trigger_fetch */
