413 void Tags::SetTag(const wxString & name, const wxString & value)
414 {
415  // We don't like empty names
416  if (name.IsEmpty()) {
417  return;
418  }
419 
420  // All keys are uppercase
421  wxString key = name;
422  key.UpperCase();
423 
424  // Look it up
425  TagMap::iterator iter = mXref.find(key);
426 
427  // Didn't find the tag
428  if (iter == mXref.end()) {
429 
430  // Add a NEW tag
431  mXref[key] = name;
432  mMap[name] = value;
433  return;
434  }
435 
436  // Update the value
437  mMap[iter->second] = value;
438 }

445 bool Tags::HandleXMLTag(const wxChar *tag, const wxChar **attrs)
446 {
447  if (wxStrcmp(tag, wxT("tags")) == 0) {
448  return true;
449  }
450 
451  if (wxStrcmp(tag, wxT("tag")) == 0) {
452  wxString n, v;
453 
454  while (*attrs) {
455  wxString attr = *attrs++;
456  if (attr.IsEmpty())
457  break;
458  wxString value = *attrs++;
459 
460  if (!XMLValueChecker::IsGoodString(attr) ||
461  !XMLValueChecker::IsGoodString(value)) {
462  break;
463  }
464 
465  if (attr == wxT("name")) {
466  n = value;
467  }
468  else if (attr == wxT("value")) {
469  v = value;
470  }
471  }
472 
473  if (n == wxT("id3v2")) {
474  // LLL: This is obsolute, but it must be handled and ignored.
475  }
476  else {
477  SetTag(n, v);
478  }
479 
480  return true;
481  }
482 
483  return false;
484 }
