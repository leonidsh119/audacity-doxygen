145 //if the wavetrack is in this queue, and is not the only wavetrack, clones the tasks and schedules it.
146 void ODWaveTrackTaskQueue::MakeWaveTrackIndependent(WaveTrack* track)
147 {
148 
149  mTracksMutex.Lock();
150  if(mTracks.size()<2)
151  {
152  //if there is only one track, it is already independent.
153  mTracksMutex.Unlock();
154  return;
155  }
156 
157  for(unsigned int i=0;i<mTracks.size();i++)
158  {
159  if(mTracks[i]==track)
160  {
161  mTracksMutex.Unlock();//release the lock, since RemoveWaveTrack is a public threadsafe method.
162  RemoveWaveTrack(mTracks[i]);
163 
164  //clone the items in order and add them to the ODManager.
165  mTasksMutex.Lock();
166  ODTask* task;
167  for(unsigned int j=0;j<mTasks.size();j++)
168  {
169  task=mTasks[j]->Clone();
170  task->AddWaveTrack(track);
171  //AddNewTask requires us to relinquish this lock. However, it is safe because ODManager::MakeWaveTrackIndependent
172  //has already locked the m_queuesMutex.
173  mTasksMutex.Unlock();
174  //AddNewTask locks the m_queuesMutex which is already locked by ODManager::MakeWaveTrackIndependent,
175  //so we pass a boolean flag telling it not to lock again.
176  ODManager::Instance()->AddNewTask(task,false);
177  mTasksMutex.Lock();
178  }
179  mTasksMutex.Unlock();
180  mTracksMutex.Lock();
181  break;
182  }
183  }
184  mTracksMutex.Unlock();
185 }
