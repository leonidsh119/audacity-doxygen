217 {
218  mFormat = int16Sample;//start with the smallest and move up in the metadata_callback.

222  if(mFile)
223  delete mFile;

225 
226 
227  if (!mHandle.Open(mFName, wxT("rb"))) {
228  return false;
229  }
230 
231  // Even though there is an init() method that takes a filename, use the one that
232  // takes a file handle because wxWidgets can open a file with a Unicode name and
233  // libflac can't (under Windows).
234  //
235  // Responsibility for closing the file is passed to libflac.
236  // (it happens when mFile->finish() is called)
237  bool result = mFile->init(mHandle.fp())?true:false;
238  mHandle.Detach();
239 
240  if (result != FLAC__STREAM_DECODER_INIT_STATUS_OK) {
241  return false;
242  }
243 
244  //this will call the metadata_callback when it is done
245  mFile->process_until_end_of_metadata();
246  // not necessary to check state, error callback will catch errors, but here's how:
247  if (mFile->get_state() > FLAC__STREAM_DECODER_READ_FRAME) {
248  return false;
249  }
250 
251  if (!mFile->is_valid() || mFile->get_was_error()) {
252  // This probably is not a FLAC file at all
253  return false;
254  }
255
