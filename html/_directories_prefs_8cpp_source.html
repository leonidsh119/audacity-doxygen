126 void DirectoriesPrefs::OnChooseTempDir(wxCommandEvent & e)
127 {
128  wxString oldTempDir = gPrefs->Read(wxT("/Directories/TempDir"), wxGetApp().defaultTempDir);
129 
130  // Because we went through InitTempDir() during initialisation,
131  // the old temp directory name in prefs should already be OK. Just in case there is
132  // some way we hadn't thought of for it to be not OK,
133  // we avoid prompting with it in that case and use the suggested default instead.
134  if( !AudacityApp::IsTempDirectoryNameOK( oldTempDir ) )
135  oldTempDir = wxGetApp().defaultTempDir;
136 
137  wxDirDialog dlog(this,
138  _("Choose a location to place the temporary directory"),
139  oldTempDir );
140  int retval = dlog.ShowModal();
141  if (retval != wxID_CANCEL && dlog.GetPath() != wxT("")) {
142  wxFileName tmpDirPath;
143  tmpDirPath.AssignDir(dlog.GetPath());
144 
145  // Append an "audacity_temp" directory to this path if necessary (the
146  // default, the existing pref (as stored in the control), and any path
147  // ending in a directory with the same name as what we'd add should be OK
148  // already)
149  wxString newDirName;
150 #if defined(__WXMAC__)
151  newDirName = wxT("audacity_temp");
152 #elif defined(__WXMSW__)
153  // Clearing Bug 1271 residual issue. Let's NOT have temp in the name.
154  // [Arguably we should do this for all platforms, not just Windows
155  // which has cleaners that clean up temp data. For another day].
156  newDirName = wxT("SessionData");
157 #else
158  newDirName = wxT(".audacity_temp");
159 #endif
160  wxArrayString dirsInPath = tmpDirPath.GetDirs();
161 
162  // If the default temp dir or user's pref dir don't end in '/' they cause
163  // wxFileName's == operator to construct a wxFileName representing a file
164  // (that doesn't exist) -- hence the constructor calls
165  if (tmpDirPath != wxFileName(wxGetApp().defaultTempDir, wxT("")) &&
166  tmpDirPath != wxFileName(mTempDir->GetValue(), wxT("")) &&
167  (dirsInPath.GetCount() == 0 ||
168  dirsInPath[dirsInPath.GetCount()-1] != newDirName))
169  {
170  tmpDirPath.AppendDir(newDirName);
171  }
172 
173  mTempDir->SetValue(tmpDirPath.GetPath(wxPATH_GET_VOLUME|wxPATH_GET_SEPARATOR));
174  UpdateFreeSpace(e);
175  }
176 }

202 bool DirectoriesPrefs::Validate()
203 {
204  wxFileName tempDir;
205  tempDir.SetPath(mTempDir->GetValue());
206 
207  if( !AudacityApp::IsTempDirectoryNameOK( tempDir.GetPath() ) ) {
208  wxMessageBox(
209  wxString::Format(_("Directory %s is not suitable (at risk of being cleaned out)"),
210  tempDir.GetPath().c_str()),
211  _("Error"),
212  wxOK | wxICON_ERROR);
213  return false;
214  }
215  if (!tempDir.DirExists()) {
216  int ans = wxMessageBox(
217  wxString::Format(_("Directory %s does not exist. Create it?"),
218  tempDir.GetPath().c_str()),
219  _("New Temporary Directory"),
220  wxYES_NO | wxCENTRE | wxICON_EXCLAMATION);
221 
222  if (ans != wxYES) {
223  return false;
224  }
225 
226  if (!tempDir.Mkdir(0755, wxPATH_MKDIR_FULL)) {
227  /* wxWidgets throws up a decent looking dialog */
228  return false;
229  }
230  }
231  else {
232  /* If the directory already exists, make sure it is writable */
233  wxLogNull logNo;
234  tempDir.AppendDir(wxT("canicreate"));
235  if (!tempDir.Mkdir(0755)) {
236  wxMessageBox(
237  wxString::Format(_("Directory %s is not writable"),
238  tempDir.GetPath().c_str()),
239  _("Error"),
240  wxOK | wxICON_ERROR);
241  return false;
242  }
243  tempDir.Rmdir();
244  tempDir.RemoveLastDir();
245  }
246 
247  wxFileName oldDir;
248  oldDir.SetPath(gPrefs->Read(wxT("/Directories/TempDir")));
249  if (tempDir != oldDir) {
250  wxMessageBox(
251  _("Changes to temporary directory will not take effect until Audacity is restarted"),
252  _("Temp Directory Update"),
253  wxOK | wxCENTRE | wxICON_INFORMATION);
254  }
255 
256  return true;
257 }
