147 bool EffectChangeTempo::Process()
148 {
149  mSoundTouch = new SoundTouch();
150  mSoundTouch->setTempoChange(m_PercentChange);
151  double mT1Dashed = mT0 + (mT1 - mT0)/(m_PercentChange/100.0 + 1.0);
152  SetTimeWarper(new RegionTimeWarper(mT0, mT1,
153  new LinearTimeWarper(mT0, mT0, mT1, mT1Dashed )));
154  bool success = EffectSoundTouch::Process();
155  if( success )
156  mT1 = mT0 + (mT1 - mT0)/(m_PercentChange/100 + 1.);
157  return success;
158 }
