95 void LinkingHtmlWindow::OnLinkClicked(const wxHtmlLinkInfo& link)
96 {
97  wxString href = link.GetHref();
98  if( href.StartsWith(wxT("innerlink:")) )
99  {
100  wxString FileName =
101  wxFileName( FileNames::HtmlHelpDir(), href.Mid( 10 ) + wxT(".htm") ).GetFullPath();
102  if( wxFileExists( FileName ) )
103  {
104  HelpSystem::ShowHelpDialog(NULL, FileName, wxT(""));
105  return;
106  }
107  else
108  {
109  SetPage( HelpText( href.Mid( 10 )));
110  wxGetTopLevelParent(this)->SetLabel( TitleText( href.Mid( 10 )));
111  }
112  }
113  else if( href.StartsWith(wxT("mailto:")) || href.StartsWith(wxT("file:")) )
114  {
115  OpenInDefaultBrowser( link );
116  return;
117  }
118  else if( !href.StartsWith( wxT("http:")) && !href.StartsWith( wxT("https:")) )
119  {
120  HtmlWindow::OnLinkClicked( link );
121  }
122  else
123  {
124  OpenInDefaultBrowser(link);
125  return;
126  }

128  if( pDlg )
