58 // Finds real 'top' of a peak hump from neighnourhood of the given 'peakpos'.
59 int PeakFinder::findTop(const float *data, int peakpos) const
60 {
61  int i;
62  int start, end;
63  float refvalue;
64 
65  refvalue = data[peakpos];
66 
67  // seek within ±10 points
68  start = peakpos - 10;
69  if (start < minPos) start = minPos;
70  end = peakpos + 10;
71  if (end > maxPos) end = maxPos;
72 
73  for (i = start; i <= end; i ++)
74  {
75  if (data[i] > refvalue)
76  {
77  peakpos = i;
78  refvalue = data[i];
79  }
80  }
81 
82  // failure if max value is at edges of seek range => it's not peak, it's at slope.
83  if ((peakpos == start) || (peakpos == end)) return 0;
84 
85  return peakpos;
86 }

92 int PeakFinder::findGround(const float *data, int peakpos, int direction) const
93 {
94  int lowpos;
95  int pos;
96  int climb_count;
97  float refvalue;
98  float delta;
99 
100  climb_count = 0;
101  refvalue = data[peakpos];
102  lowpos = peakpos;
103 
104  pos = peakpos;
105 
106  while ((pos > minPos+1) && (pos < maxPos-1))
107  {
108  int prevpos;
109 
110  prevpos = pos;
111  pos += direction;
112 
113  // calculate derivate
114  delta = data[pos] - data[prevpos];
115  if (delta <= 0)
116  {
117  // going downhill, ok
118  if (climb_count)
119  {
120  climb_count --; // decrease climb count
121  }
122 
123  // check if new minimum found
124  if (data[pos] < refvalue)
125  {
126  // new minimum found
127  lowpos = pos;
128  refvalue = data[pos];
129  }
130  }
131  else
132  {
133  // going uphill, increase climbing counter
134  climb_count ++;
135  if (climb_count > 5) break; // we've been climbing too long => it's next uphill => quit
136  }
137  }
138  return lowpos;
139 }

183 double PeakFinder::getPeakCenter(const float *data, int peakpos) const
184 {
185  float peakLevel; // peak level
186  int crosspos1, crosspos2; // position where the peak 'hump' crosses cutting level
187  float cutLevel; // cutting value
188  float groundLevel; // ground level of the peak
189  int gp1, gp2; // bottom positions of the peak 'hump'
190 
191  // find ground positions.
192  gp1 = findGround(data, peakpos, -1);
193  gp2 = findGround(data, peakpos, 1);
194 
195  groundLevel = 0.5f * (data[gp1] + data[gp2]);
196  peakLevel = data[peakpos];
197 
198  // calculate 70%-level of the peak
199  cutLevel = 0.70f * peakLevel + 0.30f * groundLevel;
200  // find mid-level crossings
201  crosspos1 = findCrossingLevel(data, cutLevel, peakpos, -1);
202  crosspos2 = findCrossingLevel(data, cutLevel, peakpos, 1);
203 
204  if ((crosspos1 < 0) || (crosspos2 < 0)) return 0; // no crossing, no peak..
205 
206  // calculate mass center of the peak surroundings
207  return calcMassCenter(data, crosspos1, crosspos2);
208 }

212 double PeakFinder::detectPeak(const float *data, int aminPos, int amaxPos)
213 {
214 
215  int i;
216  int peakpos; // position of peak level
217  double highPeak, peak;
218 
219  this->minPos = aminPos;
220  this->maxPos = amaxPos;
221 
222  // find absolute peak
223  peakpos = minPos;
224  peak = data[minPos];
225  for (i = minPos + 1; i < maxPos; i ++)
226  {
227  if (data[i] > peak)
228  {
229  peak = data[i];
230  peakpos = i;
231  }
232  }
233 
234  // Calculate exact location of the highest peak mass center
235  highPeak = getPeakCenter(data, peakpos);
236  peak = highPeak;
237 
238  // Now check if the highest peak were in fact harmonic of the true base beat peak
239  // - sometimes the highest peak can be Nth harmonic of the true base peak yet
240  // just a slightly higher than the true base
241 
242  for (i = 3; i < 10; i ++)
243  {
244  double peaktmp, harmonic;
245  int i1,i2;
246 
247  harmonic = (double)i * 0.5;
248  peakpos = (int)(highPeak / harmonic + 0.5f);
249  if (peakpos < minPos) break;
250  peakpos = findTop(data, peakpos); // seek true local maximum index
251  if (peakpos == 0) continue; // no local max here
252 
253  // calculate mass-center of possible harmonic peak
254  peaktmp = getPeakCenter(data, peakpos);
255 
256  // accept harmonic peak if
257  // (a) it is found
258  // (b) is within ±4% of the expected harmonic interval
259  // (c) has at least half x-corr value of the max. peak
260 
261  double diff = harmonic * peaktmp / highPeak;
262  if ((diff < 0.96) || (diff > 1.04)) continue; // peak too afar from expected
263 
264  // now compare to highest detected peak
265  i1 = (int)(highPeak + 0.5);
266  i2 = (int)(peaktmp + 0.5);
267  if (data[i2] >= 0.4*data[i1])
268  {
269  // The harmonic is at least half as high primary peak,
270  // thus use the harmonic peak instead
271  peak = peaktmp;
272  }
273  }
274 
275  return peak;
276 }
