549  if(mEffectEqualization48x)

557  bool bGoodResult = true;
558 
559  SelectedTrackListOfKindIterator iter(Track::Wave, mOutputTracks);
560  WaveTrack *track = (WaveTrack *) iter.First();
561  int count = 0;
562  while (track) {
563  double trackStart = track->GetStartTime();
564  double trackEnd = track->GetEndTime();
565  double t0 = mT0 < trackStart? trackStart: mT0;
566  double t1 = mT1 > trackEnd? trackEnd: mT1;
567 
568  if (t1 > t0) {
569  sampleCount start = track->TimeToLongSamples(t0);
570  sampleCount end = track->TimeToLongSamples(t1);
571  sampleCount len = (sampleCount)(end - start);
572 
573  if (!ProcessOne(count, track, start, len))
574  {
575  bGoodResult = false;
576  break;
577  }
578  }
579 
580  track = (WaveTrack *) iter.Next();
581  count++;
582  }
583 
584  this->ReplaceProcessedTracks(bGoodResult);
585  return bGoodResult;

1074 bool EffectEqualization::ProcessOne(int count, WaveTrack * t,
1075  sampleCount start, sampleCount len)
1076 {
1077  // create a NEW WaveTrack to hold all of the output, including 'tails' each end
1078  AudacityProject *p = GetActiveProject();
1079  WaveTrack *output = p->GetTrackFactory()->NewWaveTrack(floatSample, t->GetRate());
1080 
1081  int L = windowSize - (mM - 1); //Process L samples at a go
1082  sampleCount s = start;
1083  sampleCount idealBlockLen = t->GetMaxBlockSize() * 4;
1084  if (idealBlockLen % L != 0)
1085  idealBlockLen += (L - (idealBlockLen % L));
1086 
1087  float *buffer = new float[idealBlockLen];
1088 
1089  float *window1 = new float[windowSize];
1090  float *window2 = new float[windowSize];
1091  float *thisWindow = window1;
1092  float *lastWindow = window2;
1093 
1094  sampleCount originalLen = len;
1095 
1096  int i,j;
1097  for(i=0; i<windowSize; i++)
1098  lastWindow[i] = 0;
1099 
1100  TrackProgress(count, 0.);
1101  bool bLoopSuccess = true;
1102  int wcopy = 0;
1103  int offset = (mM - 1)/2;
1104 
1105  while(len)
1106  {
1107  sampleCount block = idealBlockLen;
1108  if (block > len)
1109  block = len;
1110 
1111  t->Get((samplePtr)buffer, floatSample, s, block);
1112 
1113  for(i=0; i<block; i+=L) //go through block in lumps of length L
1114  {
1115  wcopy = L;
1116  if (i + wcopy > block) //if last lump would exceed block
1117  wcopy = block - i; //shorten it
1118  for(j=0; j<wcopy; j++)
1119  thisWindow[j] = buffer[i+j]; //copy the L (or remaining) samples
1120  for(j=wcopy; j<windowSize; j++)
1121  thisWindow[j] = 0; //this includes the padding
1122 
1123  Filter(windowSize, thisWindow);
1124 
1125  // Overlap - Add
1126  for(j=0; (j<mM-1) && (j<wcopy); j++)
1127  buffer[i+j] = thisWindow[j] + lastWindow[L + j];
1128  for(j=mM-1; j<wcopy; j++)
1129  buffer[i+j] = thisWindow[j];
1130 
1131  float *tempP = thisWindow;
1132  thisWindow = lastWindow;
1133  lastWindow = tempP;
1134  } //next i, lump of this block
1135 
1136  output->Append((samplePtr)buffer, floatSample, block);
1137  len -= block;
1138  s += block;
1139 
1140  if (TrackProgress(count, (s-start)/(double)originalLen))
1141  {
1142  bLoopSuccess = false;
1143  break;
1144  }
1145  }
1146 
1147  if(bLoopSuccess)
1148  {
1149  // mM-1 samples of 'tail' left in lastWindow, get them now
1150  if(wcopy < (mM-1)) {
1151  // Still have some overlap left to process
1152  // (note that lastWindow and thisWindow have been exchanged at this point
1153  // so that 'thisWindow' is really the window prior to 'lastWindow')
1154  for(j=0; j<mM-1-wcopy; j++)
1155  buffer[j] = lastWindow[wcopy + j] + thisWindow[L + wcopy + j];
1156  // And fill in the remainder after the overlap
1157  for( ; j<mM-1; j++)
1158  buffer[j] = lastWindow[wcopy + j];
1159  } else {
1160  for(j=0; j<mM-1; j++)
1161  buffer[j] = lastWindow[wcopy + j];
1162  }
1163  output->Append((samplePtr)buffer, floatSample, mM-1);
1164  output->Flush();
1165 
1166  // now move the appropriate bit of the output back to the track
1167  // (this could be enhanced in the future to use the tails)
1168  double offsetT0 = t->LongSamplesToTime((sampleCount)offset);
1169  double lenT = t->LongSamplesToTime(originalLen);
1170  // 'start' is the sample offset in 't', the passed in track
1171  // 'startT' is the equivalent time value
1172  // 'output' starts at zero
1173  double startT = t->LongSamplesToTime(start);
1174 
1175  //output has one waveclip for the total length, even though
1176  //t might have whitespace seperating multiple clips
1177  //we want to maintain the original clip structure, so
1178  //only paste the intersections of the NEW clip.
1179 
1180  //Find the bits of clips that need replacing
1181  std::vector<std::pair<double, double> > clipStartEndTimes;
1182  std::vector<std::pair<double, double> > clipRealStartEndTimes; //the above may be truncated due to a clip being partially selected
1183  for (WaveClipList::compatibility_iterator it=t->GetClipIterator(); it; it=it->GetNext())
1184  {
1185  WaveClip *clip;
1186  double clipStartT;
1187  double clipEndT;
1188 
1189  clip = it->GetData();
1190  clipStartT = clip->GetStartTime();
1191  clipEndT = clip->GetEndTime();
1192  if( clipEndT <= startT )
1193  continue; // clip is not within selection
1194  if( clipStartT >= startT + lenT )
1195  continue; // clip is not within selection
1196 
1197  //save the actual clip start/end so that we can rejoin them after we paste.
1198  clipRealStartEndTimes.push_back(std::pair<double,double>(clipStartT,clipEndT));
1199 
1200  if( clipStartT < startT ) // does selection cover the whole clip?
1201  clipStartT = startT; // don't copy all the NEW clip
1202  if( clipEndT > startT + lenT ) // does selection cover the whole clip?
1203  clipEndT = startT + lenT; // don't copy all the NEW clip
1204 
1205  //save them
1206  clipStartEndTimes.push_back(std::pair<double,double>(clipStartT,clipEndT));
1207  }
1208  //now go thru and replace the old clips with NEW
1209  for(unsigned int i=0;i<clipStartEndTimes.size();i++)
1210  {
1211  Track *toClipOutput;
1212  //remove the old audio and get the NEW
1213  t->Clear(clipStartEndTimes[i].first,clipStartEndTimes[i].second);
1214  output->Copy(clipStartEndTimes[i].first-startT+offsetT0,clipStartEndTimes[i].second-startT+offsetT0, &toClipOutput);
1215  if(toClipOutput)
1216  {
1217  //put the processed audio in
1218  bool bResult = t->Paste(clipStartEndTimes[i].first, toClipOutput);
1219  wxASSERT(bResult); // TO DO: Actually handle this.
1220  wxUnusedVar(bResult);
1221  //if the clip was only partially selected, the Paste will have created a split line. Join is needed to take care of this
1222  //This is not true when the selection is fully contained within one clip (second half of conditional)
1223  if( (clipRealStartEndTimes[i].first != clipStartEndTimes[i].first ||
1224  clipRealStartEndTimes[i].second != clipStartEndTimes[i].second) &&
1225  !(clipRealStartEndTimes[i].first <= startT &&
1226  clipRealStartEndTimes[i].second >= startT+lenT) )
1227  t->Join(clipRealStartEndTimes[i].first,clipRealStartEndTimes[i].second);
1228  delete toClipOutput;
1229  }
1230  }
1231  }
1232 
1233  delete[] buffer;
1234  delete[] window1;
1235  delete[] window2;
1236  delete output;
1237 
1238  return bLoopSuccess;
1239 }

1241 bool EffectEqualization::CalcFilter()
1242 {
1243  double loLog = log10(mLoFreq);
1244  double hiLog = log10(mHiFreq);
1245  double denom = hiLog - loLog;
1246 
1247  double delta = mHiFreq / ((double)(mWindowSize/2.));
1248  double val0;
1249  double val1;
1250 
1251  if( IsLinear() )
1252  {
1253  val0 = mLinEnvelope->GetValue(0.0); //no scaling required - saved as dB
1254  val1 = mLinEnvelope->GetValue(1.0);
1255  }
1256  else
1257  {
1258  val0 = mLogEnvelope->GetValue(0.0); //no scaling required - saved as dB
1259  val1 = mLogEnvelope->GetValue(1.0);
1260  }
1261  mFilterFuncR[0] = val0;
1262  double freq = delta;
1263 
1264  int i;
1265  for(i=1; i<=mWindowSize/2; i++)
1266  {
1267  double when;
1268  if( IsLinear() )
1269  when = freq/mHiFreq;
1270  else
1271  when = (log10(freq) - loLog)/denom;
1272  if(when < 0.)
1273  {
1274  mFilterFuncR[i] = val0;
1275  }
1276  else if(when > 1.0)
1277  {
1278  mFilterFuncR[i] = val1;
1279  }
1280  else
1281  {
1282  if( IsLinear() )
1283  mFilterFuncR[i] = mLinEnvelope->GetValue(when);
1284  else
1285  mFilterFuncR[i] = mLogEnvelope->GetValue(when);
1286  }
1287  freq += delta;
1288  }
1289  mFilterFuncR[mWindowSize/2] = val1;
1290 
1291  mFilterFuncR[0] = DB_TO_LINEAR(mFilterFuncR[0]);
1292  for(i=1;i<mWindowSize/2;i++)
1293  {
1294  mFilterFuncR[i] = DB_TO_LINEAR(mFilterFuncR[i]);
1295  mFilterFuncR[mWindowSize-i]=mFilterFuncR[i]; //Fill entire array
1296  }
1297  mFilterFuncR[i] = DB_TO_LINEAR(mFilterFuncR[i]); //do last one
1298 
1299  //transfer to time domain to do the padding and windowing
1300  float *outr = new float[mWindowSize];
1301  float *outi = new float[mWindowSize];
1302 #ifdef EXPERIMENTAL_USE_REALFFTF
1303  InverseRealFFT(mWindowSize, mFilterFuncR, NULL, outr); // To time domain
1304 #else
1305  FFT(mWindowSize,true,mFilterFuncR,NULL,outr,outi); //To time domain
1306 #endif
1307 
1308  for(i=0;i<=(mM-1)/2;i++)
1309  { //Windowing - could give a choice, fixed for now - MJS
1310  // double mult=0.54-0.46*cos(2*M_PI*(i+(mM-1)/2.0)/(mM-1)); //Hamming
1311  //Blackman
1312  double mult=0.42-0.5*cos(2*M_PI*(i+(mM-1)/2.0)/(mM-1))+.08*cos(4*M_PI*(i+(mM-1)/2.0)/(mM-1));
1313  outr[i]*=mult;
1314  if(i!=0){
1315  outr[mWindowSize-i]*=mult;
1316  }
1317  }
1318  for(;i<=mWindowSize/2;i++)
1319  { //Padding
1320  outr[i]=0;
1321  outr[mWindowSize-i]=0;
1322  }
1323  float *tempr = new float[mM];
1324  for(i=0;i<(mM-1)/2;i++)
1325  { //shift so that padding on right
1326  tempr[(mM-1)/2+i]=outr[i];

1668 void EffectEqualization::setCurve(int currentCurve)

1671  Select(currentCurve);
1672  wxASSERT( currentCurve < (int) mCurves.GetCount() );
1673

1675  int numPoints = (int) mCurves[currentCurve].points.GetCount();
1676 
1677  if (mLin) { // linear freq mode
1678  env = mLinEnvelope;
1679  }
1680  else { // log freq mode
1681  env = mLogEnvelope;
1682  }
1683  env->Flatten(0.);
1684  env->SetTrackLen(1.0);
1685 
1686  // Handle special case of no points.
1687  if (numPoints == 0) {
1688  ForceRecalc();
1689  return;
1690  }
1691 
1692  double when, value;
1693 
1694  // Handle special case 1 point.
1695  if (numPoints == 1) {
1696  // only one point, so ensure it is in range then return.
1697  when = mCurves[currentCurve].points[0].Freq;
1698  if (mLin) {
1699  when = when / mHiFreq;
1700  }
1701  else { // log scale
1702  // We don't go below loFreqI (20 Hz) in log view.
1703  double loLog = log10((double)loFreqI);
1704  double hiLog = log10(mHiFreq);
1705  double denom = hiLog - loLog;
1706  when = (log10(std::max((double) loFreqI, when)) - loLog)/denom;
1707  }
1708  value = mCurves[currentCurve].points[0].dB;
1709  env->Insert(std::min(1.0, std::max(0.0, when)), value);
1710  ForceRecalc();
1711  return;
1712  }
1713 
1714  // We have at least two points, so ensure they are in frequency order.
1715  mCurves[currentCurve].points.Sort(SortCurvePoints);
1716 
1717  if (mCurves[currentCurve].points[0].Freq < 0) {
1718  // Corrupt or invalid curve, so bail.
1719  ForceRecalc();
1720  return;
1721  }
1722 
1723  if(mLin) { // linear Hz scale
1724  for(int pointCount = 0; pointCount < numPoints; pointCount++) {
1725  when = mCurves[currentCurve].points[pointCount].Freq / mHiFreq;
1726  value = mCurves[currentCurve].points[pointCount].dB;
1727  if(when <= 1) {
1728  env->Insert(when, value);
1729  }
1730  else {
1731  // There are more points at higher freqs, so interpolate next one then stop.
1732  when = 1.0;
1733  double lastF = mCurves[currentCurve].points[pointCount-1].Freq;
1734  double nextF = mCurves[currentCurve].points[pointCount].Freq;
1735  double lastDB = mCurves[currentCurve].points[pointCount-1].dB;
1736  double nextDB = mCurves[currentCurve].points[pointCount].dB;
1737  value = lastDB + ((nextDB - lastDB) * ((mHiFreq - lastF) / (nextF - lastF)));
1738  env->Insert(when, value);
1739  break;
1740  }
1741  }
1742  }
1743  else { // log Hz scale
1744  double loLog = log10((double) loFreqI);
1745  double hiLog = log10(mHiFreq);
1746  double denom = hiLog - loLog;
1747  int firstAbove20Hz;
1748 
1749  // log scale EQ starts at 20 Hz (threshold of hearing).
1750  // so find the first point (if any) above 20 Hz.
1751  for (firstAbove20Hz = 0; firstAbove20Hz < numPoints; firstAbove20Hz++) {
1752  if (mCurves[currentCurve].points[firstAbove20Hz].Freq > loFreqI)
1753  break;
1754  }
1755 
1756  if (firstAbove20Hz == numPoints) {
1757  // All points below 20 Hz, so just use final point.
1758  when = 0.0;
1759  value = mCurves[currentCurve].points[numPoints-1].dB;
1760  env->Insert(when, value);
1761  ForceRecalc();
1762  return;
1763  }
1764 
1765  if (firstAbove20Hz > 0) {
1766  // At least one point is before 20 Hz and there are more
1767  // beyond 20 Hz, so interpolate the first
1768  double prevF = mCurves[currentCurve].points[firstAbove20Hz-1].Freq;
1769  prevF = log10(std::max(1.0, prevF)); // log zero is bad.
1770  double prevDB = mCurves[currentCurve].points[firstAbove20Hz-1].dB;
1771  double nextF = log10(mCurves[currentCurve].points[firstAbove20Hz].Freq);
1772  double nextDB = mCurves[currentCurve].points[firstAbove20Hz].dB;
1773  when = 0.0;
1774  value = nextDB - ((nextDB - prevDB) * ((nextF - loLog) / (nextF - prevF)));
1775  env->Insert(when, value);
1776  }
1777 
1778  // Now get the rest.
1779  for(int pointCount = firstAbove20Hz; pointCount < numPoints; pointCount++)
1780  {
1781  double flog = log10(mCurves[currentCurve].points[pointCount].Freq);
1782  wxASSERT(mCurves[currentCurve].points[pointCount].Freq >= loFreqI);
1783 
1784  when = (flog - loLog)/denom;
1785  value = mCurves[currentCurve].points[pointCount].dB;
1786  if(when <= 1.0) {
1787  env->Insert(when, value);
1788  }
1789  else {
1790  // This looks weird when adjusting curve in Draw mode if
1791  // there is a point off-screen.
1792 
1793  /*
1794  // we have a point beyond fs/2. Insert it so that env code can use it.
1795  // but just this one, we have no use for the rest
1796  env->SetTrackLen(when); // can't Insert if the envelope isn't long enough
1797  env->Insert(when, value);
1798  break;
1799  */
1800 
1801  // interpolate the final point instead
1802  when = 1.0;
1803  double logLastF = log10(mCurves[currentCurve].points[pointCount-1].Freq);
1804  double lastDB = mCurves[currentCurve].points[pointCount-1].dB;
1805  value = lastDB + ((value - lastDB) * ((log10(mHiFreq) - logLastF) / (flog - logLastF)));
1806  env->Insert(when, value);
1807  break;
1808  }
1809  }
1810  }
1811  ForceRecalc();
1812 }
