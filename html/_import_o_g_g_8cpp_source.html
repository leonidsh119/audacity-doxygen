236 int OggImportFileHandle::Import(TrackFactory *trackFactory, Track ***outTracks,
237  int *outNumTracks, Tags *tags)
238 {
239  wxASSERT(mFile->IsOpened());
240 
241  CreateProgress();
242 
243  //Number of streams used may be less than mVorbisFile->links,
244  //but this way bitstream matches array index.
245  mChannels = new WaveTrack **[mVorbisFile->links];
246 
247  int i,c;
248  for (i = 0; i < mVorbisFile->links; i++)
249  {
250  //Stream is not used
251  if (mStreamUsage[i] == 0)
252  {
253  //This is just a padding to keep bitstream number and
254  //array indices matched.
255  mChannels[i] = NULL;
256  continue;
257  }
258 
259  vorbis_info *vi = ov_info(mVorbisFile, i);
260 
261  mChannels[i] = new WaveTrack *[vi->channels];
262 
263  for (c = 0; c < vi->channels; c++) {
264  mChannels[i][c] = trackFactory->NewWaveTrack(mFormat, vi->rate);
265 
266  if (vi->channels == 2) {
267  switch (c) {
268  case 0:
269  mChannels[i][c]->SetChannel(Track::LeftChannel);
270  mChannels[i][c]->SetLinked(true);
271  break;
272  case 1:
273  mChannels[i][c]->SetChannel(Track::RightChannel);
274  break;
275  }
276  }
277  else {
278  mChannels[i][c]->SetChannel(Track::MonoChannel);
279  }
280  }
281  }

283 /* The number of bytes to get from the codec in each run */
284 #define CODEC_TRANSFER_SIZE 4096
285 
286 /* The number of samples to read between calls to the callback.
287  * Balance between responsiveness of the GUI and throughput of import. */
288 #define SAMPLES_PER_CALLBACK 100000
289 
290  short *mainBuffer = new short[CODEC_TRANSFER_SIZE];
291 
292  /* determine endianness (clever trick courtesy of Nicholas Devillard,
293  * (http://www.eso.org/~ndevilla/endian/) */
294  int testvar = 1, endian;
295  if(*(char *)&testvar)
296  endian = 0; // little endian
297  else
298  endian = 1; // big endian
299 
300  /* number of samples currently in each channel's buffer */
301  int updateResult = eProgressSuccess;
302  long bytesRead = 0;
303  long samplesRead = 0;
304  int bitstream = 0;
305  int samplesSinceLastCallback = 0;
306 
307  // You would think that the stream would already be seeked to 0, and
308  // indeed it is if the file is legit. But I had several ogg files on
309  // my hard drive that have malformed headers, and this added call
310  // causes them to be read correctly. Otherwise they have lots of
311  // zeros inserted at the beginning
312  ov_pcm_seek(mVorbisFile, 0);
313 
314  do {
315  /* get data from the decoder */
316  bytesRead = ov_read(mVorbisFile, (char *) mainBuffer,
317  CODEC_TRANSFER_SIZE,
318  endian,
319  2, // word length (2 for 16 bit samples)
320  1, // signed
321  &bitstream);
322 
323  if (bytesRead == OV_HOLE) {
324  wxFileName f(mFilename);
325  wxLogError(wxT("Ogg Vorbis importer: file %s is malformed, ov_read() reported a hole"),
326  f.GetFullName().c_str());
327  /* http://lists.xiph.org/pipermail/vorbis-dev/2001-February/003223.html
328  * is the justification for doing this - best effort for malformed file,
329  * hence the message.
330  */
331  continue;
332  } else if (bytesRead < 0) {
333  /* Malformed Ogg Vorbis file. */
334  /* TODO: Return some sort of meaningful error. */
335  wxLogError(wxT("Ogg Vorbis importer: ov_read() returned error %i"),
336  bytesRead);
337  break;
338  }
339 
340  samplesRead = bytesRead / mVorbisFile->vi[bitstream].channels / sizeof(short);
341 
342  /* give the data to the wavetracks */
343  if (mStreamUsage[bitstream] != 0)
344  {
345  for (c = 0; c < mVorbisFile->vi[bitstream].channels; c++)
346  mChannels[bitstream][c]->Append((char *)(mainBuffer + c),
347  int16Sample,
348  samplesRead,
349  mVorbisFile->vi[bitstream].channels);
350  }
351 
352  samplesSinceLastCallback += samplesRead;
353  if (samplesSinceLastCallback > SAMPLES_PER_CALLBACK) {
354  updateResult = mProgress->Update(ov_time_tell(mVorbisFile),
355  ov_time_total(mVorbisFile, bitstream));
356  samplesSinceLastCallback -= SAMPLES_PER_CALLBACK;
357 
358  }
359  } while (updateResult == eProgressSuccess && bytesRead != 0);
360 
361  delete[]mainBuffer;
362 
363  int res = updateResult;
364  if (bytesRead < 0)
365  res = eProgressFailed;
366 
367  if (res == eProgressFailed || res == eProgressCancelled) {
368  for (i = 0; i < mVorbisFile->links; i++)
369  {
370  if (mChannels[i])
371  {
372  for(c = 0; c < mVorbisFile->vi[bitstream].channels; c++) {
373  if (mChannels[i][c])
374  delete mChannels[i][c];
375  }
376  delete[] mChannels[i];
377  }
378  }
379  delete[] mChannels;
380  return res;
381  }
382 
383  *outNumTracks = 0;
384  for (int s = 0; s < mVorbisFile->links; s++)
385  {
386  if (mStreamUsage[s] != 0)
387  *outNumTracks += mVorbisFile->vi[s].channels;
388  }
389 
390  *outTracks = new Track *[*outNumTracks];
391 
392  int trackindex = 0;
393  for (i = 0; i < mVorbisFile->links; i++)
394  {
395  if (mChannels[i])
396  {
397  for (c = 0; c < mVorbisFile->vi[i].channels; c++) {
398  mChannels[i][c]->Flush();
399  (*outTracks)[trackindex++] = mChannels[i][c];
400  }
401  delete[] mChannels[i];
402  }
403  }
404  delete[] mChannels;
405 
406  //\todo { Extract comments from each stream? }
407  if (mVorbisFile->vc[0].comments > 0) {
408  tags->Clear();
409  for (c = 0; c < mVorbisFile->vc[0].comments; c++) {
410  wxString comment = UTF8CTOWX(mVorbisFile->vc[0].user_comments[c]);
411  wxString name = comment.BeforeFirst(wxT('='));
412  wxString value = comment.AfterFirst(wxT('='));
413  if (name.Upper() == wxT("DATE") && !tags->HasTag(TAG_YEAR)) {
414  long val;
415  if (value.Length() == 4 && value.ToLong(&val)) {
416  name = TAG_YEAR;
417  }
418  }
419  tags->SetTag(name, value);
420  }
421  }
422 
423  return res;
424 }
