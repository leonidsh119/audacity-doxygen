77 float CubicMaximize(float y0, float y1, float y2, float y3)
78 {
79  // Find coefficients of cubic
80 
81  float a, b, c, d;
82  float da, db, dc;
83  float discriminant;
84  float x1, x2;
85  float dda, ddb;
86 
87  a = (float) (y0/-6.0 + y1/2.0 - y2/2.0 + y3/6.0);
88  b = (float) (y0 - 5.0*y1/2.0 + 2.0*y2 - y3/2.0);
89  c = (float) (-11.0*y0/6.0 + 3.0*y1 - 3.0*y2/2.0 + y3/3.0);
90  d = y0;
91 
92  // Take derivative
93 
94  da = 3*a;
95  db = 2*b;
96  dc = c;
97 
98  // Find zeroes of derivative using quadratic equation
99 
100  discriminant = db*db - 4*da*dc;
101  if (discriminant < 0.0)
102  return -1.0; // error
103 
104  x1 = (float) ((-db + sqrt(discriminant)) / (2 * da));
105  x2 = (float) ((-db - sqrt(discriminant)) / (2 * da));
106 
107  // The one which corresponds to a local _maximum_ in the
108  // cubic is the one we want - the one with a negative
109  // second derivative
110 
111  dda = 2*da;
112  ddb = db;
113 
114  if (dda*x1 + ddb < 0)
115  return x1;
116  else
117  return x2;
118 }

121 float parabolic_interp(float x1, float x2, float x3, float y1, float y2,
122  float y3, float *min)
123 {
124  float a, b, c;
125  float pos;
126 
127  // y1=a*x1^2+b*x1+c
128  // y2=a*x2^2+b*x2+c
129  // y3=a*x3^2+b*x3+c
130 
131  // y1-y2=a*(x1^2-x2^2)+b*(x1-x2)
132  // y2-y3=a*(x2^2-x3^2)+b*(x2-x3)
133 
134  // (y1-y2)/(x1-x2)=a*(x1+x2)+b
135  // (y2-y3)/(x2-x3)=a*(x2+x3)+b
136 
137  a = ((y1 - y2) / (x1 - x2) - (y2 - y3) / (x2 - x3)) / (x1 - x3);
138  b = (y1 - y2) / (x1 - x2) - a * (x1 + x2);
139  c = y1 - a * x1 * x1 - b * x1;
140 
141  // dy/dx = 2a*x + b = 0
142 
143  pos = (float) (-b / (a + a));
144  *min = /* ax^2 + bx + c */ (a * pos + b) * pos + c;
145  return pos;

196  results[i - m] = non_periodic;

198  }

238 /* yin_fetch - compute F0 and harmonicity using YIN approach. */
239 /*
240  * The pitch (F0) is determined by finding two periods whose
241  * inner product accounts for almost all of the energy. Let X and Y
242  * be adjacent vectors of length N in the sample stream. Then,
243  * if 2X*Y > threshold * (X*X + Y*Y)
244  * then the period is given by N
245  * In the algorithm, we compute different sizes until we find a
246  * peak above threshold. Then, we use cubic interpolation to get
247  * a precise value. If no peak above threshold is found, we return
248  * the first peak. The second channel returns the value 2X*Y/(X*X+Y*Y)
249  * which is refered to as the "harmonicity" -- the amount of energy
250  * accounted for by periodicity.
251  *
252  * Low sample rates are advised because of the high cost of computing
253  * inner products (fast autocorrelation is not used).
254  *
255  * The result is a 2-channel signal running at the requested rate.
256  * The first channel is the estimated pitch, and the second channel
257  * is the harmonicity.
258  *
259  * This code is adopted from multiread, currently the only other
260  * multichannel suspension in Nyquist. Comments from multiread include:
261  * The susp is shared by all channels. The susp has backpointers
262  * to the tail-most snd_list node of each channel, and it is by
263  * extending the list at these nodes that sounds are read in.
264  * To avoid a circularity, the reference counts on snd_list nodes
265  * do not include the backpointers from this susp. When a snd_list
266  * node refcount goes to zero, the yin susp's free routine
267  * is called. This must scan the backpointers to find the node that
268  * has a zero refcount (the free routine is called before the node
269  * is deallocated, so this is safe). The backpointer is then set
270  * to NULL. When all backpointers are NULL, the susp itself is
271  * deallocated, because it can only be referenced through the
272  * snd_list nodes to which there are backpointers.
273  */
