274  int k, i;
275  int l1ido = l1*ido;
276  if (ido <= 2) {
277  for (k=0; k < l1ido; k += ido, ch += ido, cc+= 2*ido) {
278  ch[0] = VADD(cc[0], cc[ido+0]);
279  ch[l1ido] = VSUB(cc[0], cc[ido+0]);
280  ch[1] = VADD(cc[1], cc[ido+1]);
281  ch[l1ido + 1] = VSUB(cc[1], cc[ido+1]);
282  }
283  } else {
284  for (k=0; k < l1ido; k += ido, ch += ido, cc += 2*ido) {
285  for (i=0; i<ido-1; i+=2) {
286  v4sf tr2 = VSUB(cc[i+0], cc[i+ido+0]);
287  v4sf ti2 = VSUB(cc[i+1], cc[i+ido+1]);
288  v4sf wr = LD_PS1(wa1[i]), wi = VMUL(LD_PS1(fsign), LD_PS1(wa1[i+1]));
289  ch[i] = VADD(cc[i+0], cc[i+ido+0]);
290  ch[i+1] = VADD(cc[i+1], cc[i+ido+1]);
291  VCPLXMUL(tr2, ti2, wr, wi);
292  ch[i+l1ido] = tr2;
293  ch[i+l1ido+1] = ti2;
294  }
295  }
296  }
