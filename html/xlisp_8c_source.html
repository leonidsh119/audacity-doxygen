62 long random() {
63 // note that this takes a seed and returns a big number,
64 // whereas I think XLisp's RANDOM is defined differently
65  long k1;
66 
67  /* algorithm taken from Dr. Dobbs Journal, November 1985, page 91 */
68  k1 = randomseed / 127773L;
69  if ((randomseed = 16807L * (randomseed - k1 * 127773L) - k1 * 2836L) < 0L)
70  randomseed += 2147483647L;
71 
72  /* return a random number between 0 and MAXFIX */
73  return randomseed;
74 }
75 #endif

183 LVAL xlisp_eval(LVAL expr)
184 {
185  int was_in_a_context = in_a_context;
186  XLCONTEXT cntxt;
187 
188  if (in_a_context == FALSE) {
189  /* create an execution context */
190  xlbegin(&cntxt,CF_TOPLEVEL|CF_CLEANUP|CF_BRKLEVEL,s_true);
191  if (_setjmp(cntxt.c_jmpbuf)) {
192  setvalue(s_evalhook,NIL);

194  xltrcindent = 0;
195  xldebug = 0;
196  xlflush();

198  in_a_context = FALSE;

200  }

214 /* xlisp_main -- run normal lisp read-eval-print loop */
215 void xlisp_main()
216 {
217  LVAL expr;
218  XLCONTEXT cntxt;
219 
220  /* build an outer-most context */

222  in_a_context = TRUE;
223

227

229  xlsave1(expr);

231  /* main command processing loop */
232  for (xl_main_loop = TRUE; xl_main_loop;) {
233 
234  /* setup the error return */
235  if (_setjmp(cntxt.c_jmpbuf)) {
236  setvalue(s_evalhook,NIL);
237  setvalue(s_applyhook,NIL);
238  xltrcindent = 0;

244  /* print a prompt */
245  stdputstr("> ");
246  #endif
247 
248  /* read an expression */
249  if (!xlread(getvalue(s_stdin),&expr,FALSE))
250  break;
251 
252  /* save the input expression */
253  xlrdsave(expr);
254
