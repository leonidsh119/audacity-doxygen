170 static gboolean
171 on_expose_event(GtkWidget* widget, GdkEventExpose* ev, gpointer data)
172 {
173  EgScopeUI* ui = (EgScopeUI*)data;
174  const float gain = gtk_spin_button_get_value(GTK_SPIN_BUTTON(ui->spb_amp));
175 
176  // Get cairo type for the gtk window
177  cairo_t* cr;
178  cr = gdk_cairo_create(ui->darea->window);
179 
180  // Limit cairo-drawing to exposed area
181  cairo_rectangle(cr, ev->area.x, ev->area.y, ev->area.width, ev->area.height);
182  cairo_clip(cr);
183 
184  // Clear background
185  cairo_set_source_rgba(cr, 0.0, 0.0, 0.0, 1.0);
186  cairo_rectangle(cr, 0, 0, DAWIDTH, DAHEIGHT * ui->n_channels);
187  cairo_fill(cr);
188 
189  cairo_set_line_width(cr, 1.0);
190 
191  const uint32_t start = ev->area.x;
192  const uint32_t end = ev->area.x + ev->area.width;
193 
194  assert(start < DAWIDTH);
195  assert(end <= DAWIDTH);
196  assert(start < end);
197 
198  for (uint32_t c = 0; c < ui->n_channels; ++c) {
199  ScoChan* chn = &ui->chn[c];
200 
201  /* Drawing area Y-position of given sample-value.
202  * Note: cairo-pixel at 0 spans -0.5 .. +0.5, hence (DAHEIGHT / 2.0 -0.5)
203  * also the cairo Y-axis points upwards (hence 'minus value')
204  *
205  * == ( DAHEIGHT * (CHN) // channel offset
206  * + (DAHEIGHT / 2) - 0.5 // vertical center -- '0'
207  * - (DAHEIGHT / 2) * (VAL) * (GAIN)
208  * )
209  */
210  const float chn_y_offset = DAHEIGHT * c + DAHEIGHT * 0.5f - 0.5f;
211  const float chn_y_scale = DAHEIGHT * 0.5f * gain;
212 
213 #define CYPOS(VAL) (chn_y_offset - (VAL) * chn_y_scale)
214 
215  cairo_save(cr);
216 
217  /* Restrict drawing to current channel area, don't bleed drawing into
218  neighboring channels. */
219  cairo_rectangle(cr, 0, DAHEIGHT * c, DAWIDTH, DAHEIGHT);
220  cairo_clip(cr);
221 
222  // Set color of wave-form
223  cairo_set_source_rgba(cr, 0.0, 1.0, 0.0, 1.0);
224 
225  /* This is a somewhat 'smart' mechanism to plot audio data using
226  alternating up/down line-directions. It works well for both cases:
227  1 pixel <= 1 sample and 1 pixel represents more than 1 sample, but
228  is not ideal for either. */
229  if (start == chn->idx) {
230  cairo_move_to(cr, start - 0.5, CYPOS(0));
231  } else {
232  cairo_move_to(cr, start - 0.5, CYPOS(chn->data_max[start]));
233  }
234 
235  uint32_t pathlength = 0;
236  for (uint32_t i = start; i < end; ++i) {
237  if (i == chn->idx) {
238  continue;
239  } else if (i % 2) {
240  cairo_line_to(cr, i - .5, CYPOS(chn->data_min[i]));
241  cairo_line_to(cr, i - .5, CYPOS(chn->data_max[i]));
242  ++pathlength;
243  } else {
244  cairo_line_to(cr, i - .5, CYPOS(chn->data_max[i]));
245  cairo_line_to(cr, i - .5, CYPOS(chn->data_min[i]));
246  ++pathlength;
247  }
248 
252  if (pathlength > MAX_CAIRO_PATH) {
253  pathlength = 0;
254  cairo_stroke(cr);
255  if (i % 2) {
256  cairo_move_to(cr, i - .5, CYPOS(chn->data_max[i]));
257  } else {
258  cairo_move_to(cr, i - .5, CYPOS(chn->data_min[i]));
259  }
260  }
261  }
262 
263  if (pathlength > 0) {
264  cairo_stroke(cr);
265  }
266 
267  // Draw current position vertical line if display is slow
268  if (ui->stride >= ui->rate / 4800.0f || ui->paused) {
269  cairo_set_source_rgba(cr, .9, .2, .2, .6);
270  cairo_move_to(cr, chn->idx - .5, DAHEIGHT * c);
271  cairo_line_to(cr, chn->idx - .5, DAHEIGHT * (c + 1));
272  cairo_stroke(cr);
273  }
274 
275  // Undo the 'clipping' restriction
276  cairo_restore(cr);
277 
278  // Channel separator
279  if (c > 0) {
280  cairo_set_source_rgba(cr, .5, .5, .5, 1.0);
281  cairo_move_to(cr, 0, DAHEIGHT * c - .5);
282  cairo_line_to(cr, DAWIDTH, DAHEIGHT * c - .5);
283  cairo_stroke(cr);
284  }
285 
286  // Zero scale line
287  cairo_set_source_rgba(cr, .3, .3, .7, .5);
288  cairo_move_to(cr, 0, DAHEIGHT * (c + .5) - .5);
289  cairo_line_to(cr, DAWIDTH, DAHEIGHT * (c + .5) - .5);
290  cairo_stroke(cr);
291  }
292 
293  cairo_destroy(cr);
294  return TRUE;
295 }

315 static int
316 process_channel(EgScopeUI* ui,
317  ScoChan* chn,
318  const size_t n_elem,
319  float const* data,
320  uint32_t* idx_start,
321  uint32_t* idx_end)
322 {
323  int overflow = 0;
324  *idx_start = chn->idx;
325  for (size_t i = 0; i < n_elem; ++i) {
326  if (data[i] < chn->data_min[chn->idx]) {
327  chn->data_min[chn->idx] = data[i];
328  }
329  if (data[i] > chn->data_max[chn->idx]) {
330  chn->data_max[chn->idx] = data[i];
331  }
332  if (++chn->sub >= ui->stride) {
333  chn->sub = 0;
334  chn->idx = (chn->idx + 1) % DAWIDTH;
335  if (chn->idx == 0) {
336  ++overflow;
337  }
338  chn->data_min[chn->idx] = 1.0;
339  chn->data_max[chn->idx] = -1.0;
340  }
341  }
342  *idx_end = chn->idx;
343  return overflow;
344 }

350 static void
351 update_scope(EgScopeUI* ui,
352  const int32_t channel,
353  const size_t n_elem,
354  float const* data)
355 {
356  // Never trust input data which could lead to application failure.
357  if (channel < 0 || (uint32_t)channel > ui->n_channels) {
358  return;
359  }
360 
361  // Update state in sync with 1st channel
362  if (channel == 0) {
363  ui->stride = gtk_spin_button_get_value(GTK_SPIN_BUTTON(ui->spb_speed));
364  const bool paused = gtk_toggle_button_get_active(
365  GTK_TOGGLE_BUTTON(ui->btn_pause));
366 
367  if (paused != ui->paused) {
368  ui->paused = paused;
369  gtk_widget_queue_draw(ui->darea);
370  }
371  }
372  if (ui->paused) {
373  return;
374  }
375 
376  uint32_t idx_start; // Display pixel start
377  uint32_t idx_end; // Display pixel end
378  int overflow; // Received more audio-data than display-pixel
379 
380  // Process this channel's audio-data for display
381  ScoChan* chn = &ui->chn[channel];
382  overflow = process_channel(ui, chn, n_elem, data, &idx_start, &idx_end);
383 
384  // Signal gtk's main thread to redraw the widget after the last channel
385  if ((uint32_t)channel + 1 == ui->n_channels) {
386  if (overflow > 1) {
387  // Redraw complete widget
388  gtk_widget_queue_draw(ui->darea);
389  } else if (idx_end > idx_start) {
390  // Redraw area between start -> end pixel
391  gtk_widget_queue_draw_area(ui->darea, idx_start - 2, 0, 3
392  + idx_end - idx_start,
393  DAHEIGHT * ui->n_channels);
394  } else if (idx_end < idx_start) {
395  // Wrap-around: redraw area between 0->start AND end->right-end
396  gtk_widget_queue_draw_area(
397  ui->darea,
398  idx_start - 2, 0,
399  3 + DAWIDTH - idx_start, DAHEIGHT * ui->n_channels);
400  gtk_widget_queue_draw_area(
401  ui->darea,
402  0, 0,
403  idx_end + 1, DAHEIGHT * ui->n_channels);
404  }
405  }
406 }

531 static int
532 recv_raw_audio(EgScopeUI* ui, const LV2_Atom_Object* obj)
533 {
534  const LV2_Atom* chan_val = NULL;
535  const LV2_Atom* data_val = NULL;
536  const int n_props = lv2_atom_object_get(
537  obj,
538  ui->uris.channelID, &chan_val,
539  ui->uris.audioData, &data_val,
540  NULL);
541 
542  if (n_props != 2 ||
543  chan_val->type != ui->uris.atom_Int ||
544  data_val->type != ui->uris.atom_Vector) {
545  // Object does not have the required properties with correct types
546  fprintf(stderr, "eg-scope.lv2 UI error: Corrupt audio message\n");
547  return 1;
548  }
549 
550  // Get the values we need from the body of the property value atoms
551  const int32_t chn = ((const LV2_Atom_Int*)chan_val)->body;
552  const LV2_Atom_Vector* vec = (const LV2_Atom_Vector*)data_val;
553  if (vec->body.child_type != ui->uris.atom_Float) {
554  return 1; // Vector has incorrect element type
555  }
556 
557  // Number of elements = (total size - header size) / element size
558  const size_t n_elem = ((data_val->size - sizeof(LV2_Atom_Vector_Body))
559  / sizeof(float));
560 
561  // Float elements immediately follow the vector body header
562  const float* data = (const float*)(&vec->body + 1);
563 
564  // Update display
565  update_scope(ui, chn, n_elem, data);
566  return 0;
567 }
