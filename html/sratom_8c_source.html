139 static void
140 list_append(Sratom* sratom,
141  LV2_URID_Unmap* unmap,
142  unsigned* flags,
143  SerdNode* s,
144  SerdNode* p,
145  SerdNode* node,
146  uint32_t size,
147  uint32_t type,
148  const void* body)
149 {
150  // Generate a list node
151  gensym(node, 'l', sratom->next_id);
152  sratom->write_statement(sratom->handle, *flags, NULL,
153  s, p, node, NULL, NULL);
154 
155  // _:node rdf:first value
156  *flags = SERD_LIST_CONT;
157  *p = serd_node_from_string(SERD_URI, NS_RDF "first");
158  sratom_write(sratom, unmap, *flags, node, p, type, size, body);
159 
160  // Set subject to node and predicate to rdf:rest for next time
161  gensym(node, 'l', ++sratom->next_id);
162  *s = *node;
163  *p = serd_node_from_string(SERD_URI, NS_RDF "rest");
164 }

178 static void
179 start_object(Sratom* sratom,
180  uint32_t* flags,
181  const SerdNode* subject,
182  const SerdNode* predicate,
183  const SerdNode* node,
184  const char* type)
185 {
186  if (subject && predicate) {
187  sratom->write_statement(sratom->handle, *flags|SERD_ANON_O_BEGIN, NULL,
188  subject, predicate, node, NULL, NULL);
189  // Start abbreviating object properties
190  *flags |= SERD_ANON_CONT;
191 
192  // Object is in a list, stop list abbreviating if necessary
193  *flags &= ~SERD_LIST_CONT;
194  }
195  if (type) {
196  SerdNode p = serd_node_from_string(SERD_URI, NS_RDF "type");
197  SerdNode o = serd_node_from_string(SERD_URI, USTR(type));
198  sratom->write_statement(sratom->handle, *flags, NULL,
199  node, &p, &o, NULL, NULL);
200  }
201 }

211 SRATOM_API
212 int
213 sratom_write(Sratom* sratom,
214  LV2_URID_Unmap* unmap,
215  uint32_t flags,
216  const SerdNode* subject,
217  const SerdNode* predicate,
218  uint32_t type_urid,
219  uint32_t size,
220  const void* body)
221 {
222  const char* const type = unmap->unmap(unmap->handle, type_urid);
223  uint8_t idbuf[12] = "b0000000000";
224  SerdNode id = serd_node_from_string(SERD_BLANK, idbuf);
225  uint8_t nodebuf[12] = "b0000000000";
226  SerdNode node = serd_node_from_string(SERD_BLANK, nodebuf);
227  SerdNode object = SERD_NODE_NULL;
228  SerdNode datatype = SERD_NODE_NULL;
229  SerdNode language = SERD_NODE_NULL;
230  bool new_node = false;
231  if (type_urid == 0 && size == 0) {
232  object = serd_node_from_string(SERD_URI, USTR(NS_RDF "nil"));
233  } else if (type_urid == sratom->forge.String) {
234  object = serd_node_from_string(SERD_LITERAL, (const uint8_t*)body);
235  } else if (type_urid == sratom->forge.Chunk) {
236  datatype = serd_node_from_string(SERD_URI, NS_XSD "base64Binary");
237  object = serd_node_new_blob(body, size, true);
238  new_node = true;
239  } else if (type_urid == sratom->forge.Literal) {
240  const LV2_Atom_Literal_Body* lit = (const LV2_Atom_Literal_Body*)body;
241  const uint8_t* str = USTR(lit + 1);
242  object = serd_node_from_string(SERD_LITERAL, str);
243  if (lit->datatype) {
244  datatype = serd_node_from_string(
245  SERD_URI, USTR(unmap->unmap(unmap->handle, lit->datatype)));
246  } else if (lit->lang) {
247  const char* lang = unmap->unmap(unmap->handle, lit->lang);
248  const char* prefix = "http://lexvo.org/id/iso639-3/";
249  const size_t prefix_len = strlen(prefix);
250  if (lang && !strncmp(lang, prefix, prefix_len)) {
251  language = serd_node_from_string(
252  SERD_LITERAL, USTR(lang + prefix_len));
253  } else {
254  fprintf(stderr, "Unknown language URID %d\n", lit->lang);
255  }
256  }
257  } else if (type_urid == sratom->forge.URID) {
258  const uint32_t urid = *(const uint32_t*)body;
259  const uint8_t* str = USTR(unmap->unmap(unmap->handle, urid));
260  object = serd_node_from_string(SERD_URI, str);
261  } else if (type_urid == sratom->forge.Path) {
262  const uint8_t* str = USTR(body);
263  if (path_is_absolute((const char*)str)) {
264  new_node = true;
265  object = serd_node_new_file_uri(str, NULL, NULL, true);
266  } else {
267  SerdURI base_uri = SERD_URI_NULL;
268  if (!sratom->base_uri.buf ||
269  strncmp((const char*)sratom->base_uri.buf, "file://", 7)) {
270  fprintf(stderr, "warning: Relative path but base is not a file URI.\n");
271  fprintf(stderr, "warning: Writing ambiguous atom:Path literal.\n");
272  object = serd_node_from_string(SERD_LITERAL, str);
273  datatype = serd_node_from_string(SERD_URI, USTR(LV2_ATOM__Path));
274  } else {
275  if (sratom->base_uri.buf) {
276  serd_uri_parse(sratom->base_uri.buf, &base_uri);
277  }
278  new_node = true;
279  SerdNode rel = serd_node_new_file_uri(str, NULL, NULL, true);
280  object = serd_node_new_uri_from_node(&rel, &base_uri, NULL);
281  serd_node_free(&rel);
282  }
283  }
284  } else if (type_urid == sratom->forge.URI) {
285  const uint8_t* str = USTR(body);
286  object = serd_node_from_string(SERD_URI, str);
287  } else if (type_urid == sratom->forge.Int) {
288  new_node = true;
289  object = serd_node_new_integer(*(const int32_t*)body);
290  datatype = serd_node_from_string(SERD_URI, (sratom->pretty_numbers)
291  ? NS_XSD "integer" : NS_XSD "int");
292  } else if (type_urid == sratom->forge.Long) {
293  new_node = true;
294  object = serd_node_new_integer(*(const int64_t*)body);
295  datatype = serd_node_from_string(SERD_URI, (sratom->pretty_numbers)
296  ? NS_XSD "integer" : NS_XSD "long");
297  } else if (type_urid == sratom->forge.Float) {
298  new_node = true;
299  object = serd_node_new_decimal(*(const float*)body, 8);
300  datatype = serd_node_from_string(SERD_URI, (sratom->pretty_numbers)
301  ? NS_XSD "decimal" : NS_XSD "float");
302  } else if (type_urid == sratom->forge.Double) {
303  new_node = true;
304  object = serd_node_new_decimal(*(const double*)body, 16);
305  datatype = serd_node_from_string(SERD_URI, (sratom->pretty_numbers)
306  ? NS_XSD "decimal" : NS_XSD "double");
307  } else if (type_urid == sratom->forge.Bool) {
308  const int32_t val = *(const int32_t*)body;
309  datatype = serd_node_from_string(SERD_URI, NS_XSD "boolean");
310  object = serd_node_from_string(SERD_LITERAL,
311  USTR(val ? "true" : "false"));
312  } else if (type_urid == sratom->midi_MidiEvent) {
313  new_node = true;
314  datatype = serd_node_from_string(SERD_URI, USTR(LV2_MIDI__MidiEvent));
315  uint8_t* str = (uint8_t*)calloc(size * 2 + 1, 1);
316  for (uint32_t i = 0; i < size; ++i) {
317  snprintf((char*)str + (2 * i), size * 2 + 1, "%02X",
318  (unsigned)(uint8_t)*((const uint8_t*)body + i));
319  }
320  object = serd_node_from_string(SERD_LITERAL, USTR(str));
321  } else if (type_urid == sratom->atom_Event) {
322  const LV2_Atom_Event* ev = (const LV2_Atom_Event*)body;
323  gensym(&id, 'e', sratom->next_id++);
324  start_object(sratom, &flags, subject, predicate, &id, NULL);
325  // TODO: beat time
326  SerdNode time = serd_node_new_integer(ev->time.frames);
327  SerdNode p = serd_node_from_string(SERD_URI,
328  USTR(LV2_ATOM__frameTime));
329  datatype = serd_node_from_string(SERD_URI, NS_XSD "decimal");
330  sratom->write_statement(sratom->handle, SERD_ANON_CONT, NULL,
331  &id, &p, &time, &datatype, &language);
332  serd_node_free(&time);
333 
334  p = serd_node_from_string(SERD_URI, NS_RDF "value");
335  sratom_write(sratom, unmap, SERD_ANON_CONT, &id, &p,
336  ev->body.type, ev->body.size, LV2_ATOM_BODY(&ev->body));
337  if (sratom->end_anon) {
338  sratom->end_anon(sratom->handle, &id);
339  }
340  } else if (type_urid == sratom->forge.Tuple) {
341  gensym(&id, 't', sratom->next_id++);
342  start_object(sratom, &flags, subject, predicate, &id, type);
343  SerdNode p = serd_node_from_string(SERD_URI, NS_RDF "value");
344  flags |= SERD_LIST_O_BEGIN;
345  LV2_ATOM_TUPLE_BODY_FOREACH(body, size, i) {
346  list_append(sratom, unmap, &flags, &id, &p, &node,
347  i->size, i->type, LV2_ATOM_BODY(i));
348  }
349  list_end(sratom->write_statement, sratom->handle, &flags, &id, &p);
350  if (sratom->end_anon) {
351  sratom->end_anon(sratom->handle, &id);
352  }
353  } else if (type_urid == sratom->forge.Vector) {
354  const LV2_Atom_Vector_Body* vec = (const LV2_Atom_Vector_Body*)body;
355  gensym(&id, 'v', sratom->next_id++);
356  start_object(sratom, &flags, subject, predicate, &id, type);
357  SerdNode p = serd_node_from_string(SERD_URI, (const uint8_t*)LV2_ATOM__childType);
358  SerdNode child_type = serd_node_from_string(
359  SERD_URI, (const uint8_t*)unmap->unmap(unmap->handle, vec->child_type));
360  sratom->write_statement(sratom->handle, flags, NULL, &id, &p, &child_type, NULL, NULL);
361  p = serd_node_from_string(SERD_URI, NS_RDF "value");
362  flags |= SERD_LIST_O_BEGIN;
363  for (const char* i = (const char*)(vec + 1);
364  i < (const char*)vec + size;
365  i += vec->child_size) {
366  list_append(sratom, unmap, &flags, &id, &p, &node,
367  vec->child_size, vec->child_type, i);
368  }
369  list_end(sratom->write_statement, sratom->handle, &flags, &id, &p);
370  if (sratom->end_anon) {
371  sratom->end_anon(sratom->handle, &id);
372  }
373  } else if (lv2_atom_forge_is_object_type(&sratom->forge, type_urid)) {
374  const LV2_Atom_Object_Body* obj = (const LV2_Atom_Object_Body*)body;
375  const char* otype = unmap->unmap(unmap->handle,
376  obj->otype);
377 
378  if (lv2_atom_forge_is_blank(&sratom->forge, type_urid, obj)) {
379  gensym(&id, 'b', sratom->next_id++);
380  start_object(sratom, &flags, subject, predicate, &id, otype);
381  } else {
382  id = serd_node_from_string(
383  SERD_URI, (const uint8_t*)unmap->unmap(unmap->handle, obj->id));
384  flags = 0;
385  start_object(sratom, &flags, NULL, NULL, &id, otype);
386  }
387  LV2_ATOM_OBJECT_BODY_FOREACH(obj, size, prop) {
388  const char* const key = unmap->unmap(unmap->handle, prop->key);
389  SerdNode pred = serd_node_from_string(SERD_URI, USTR(key));
390  sratom_write(sratom, unmap, flags, &id, &pred,
391  prop->value.type, prop->value.size,
392  LV2_ATOM_BODY(&prop->value));
393  }
394  if (sratom->end_anon && (flags & SERD_ANON_CONT)) {
395  sratom->end_anon(sratom->handle, &id);
396  }
397  } else if (type_urid == sratom->forge.Sequence) {
398  const LV2_Atom_Sequence_Body* seq = (const LV2_Atom_Sequence_Body*)body;
399  gensym(&id, 'v', sratom->next_id++);
400  start_object(sratom, &flags, subject, predicate, &id, type);
401  SerdNode p = serd_node_from_string(SERD_URI, NS_RDF "value");
402  flags |= SERD_LIST_O_BEGIN;
403  LV2_ATOM_SEQUENCE_BODY_FOREACH(seq, size, ev) {
404  list_append(sratom, unmap, &flags, &id, &p, &node,
405  sizeof(LV2_Atom_Event) + ev->body.size,
406  sratom->atom_Event,
407  ev);
408  }
409  list_end(sratom->write_statement, sratom->handle, &flags, &id, &p);
410  if (sratom->end_anon && subject && predicate) {
411  sratom->end_anon(sratom->handle, &id);
412  }
413  } else {
414  gensym(&id, 'b', sratom->next_id++);
415  start_object(sratom, &flags, subject, predicate, &id, type);
416  SerdNode p = serd_node_from_string(SERD_URI, NS_RDF "value");
417  SerdNode o = serd_node_new_blob(body, size, true);
418  datatype = serd_node_from_string(SERD_URI, NS_XSD "base64Binary");
419  sratom->write_statement(sratom->handle, flags, NULL, &id, &p, &o, &datatype, NULL);
420  if (sratom->end_anon && subject && predicate) {
421  sratom->end_anon(sratom->handle, &id);
422  }
423  serd_node_free(&o);
424  }
425 
426  if (object.buf) {
427  SerdNode def_s = serd_node_from_string(SERD_BLANK, USTR("atom"));
428  SerdNode def_p = serd_node_from_string(SERD_URI, USTR(NS_RDF "value"));
429  if (!subject) {
430  subject = &def_s;
431  }
432  if (!predicate) {
433  predicate = &def_p;
434  }
435  sratom->write_statement(sratom->handle, flags, NULL,
436  subject, predicate, &object, &datatype, &language);
437  }
438 
439  if (new_node) {
440  serd_node_free(&object);
441  }
442 
443  return 0;
444 }

446 SRATOM_API
447 char*
448 sratom_to_turtle(Sratom* sratom,
449  LV2_URID_Unmap* unmap,
450  const char* base_uri,
451  const SerdNode* subject,
452  const SerdNode* predicate,
453  uint32_t type,
454  uint32_t size,
455  const void* body)
456 {
457  SerdURI buri = SERD_URI_NULL;
458  SerdNode base = serd_node_new_uri_from_string(USTR(base_uri), NULL, &buri);
459  SerdEnv* env = serd_env_new(&base);
460  SerdChunk str = { NULL, 0 };
461 
462  serd_env_set_prefix_from_strings(env, USTR("midi"),
463  USTR(LV2_MIDI_PREFIX));
464  serd_env_set_prefix_from_strings(env, USTR("atom"),
465  USTR(LV2_ATOM_URI "#"));
466  serd_env_set_prefix_from_strings(env, USTR("rdf"), NS_RDF);
467  serd_env_set_prefix_from_strings(env, USTR("xsd"), NS_XSD);
468 
469  SerdWriter* writer = serd_writer_new(
470  SERD_TURTLE,
471  (SerdStyle)(SERD_STYLE_ABBREVIATED |
472  SERD_STYLE_RESOLVED |
473  SERD_STYLE_CURIED),
474  env, &buri, serd_chunk_sink, &str);
475 
476  // Write @prefix directives
477  serd_env_foreach(env,
478  (SerdPrefixSink)serd_writer_set_prefix,
479  writer);
480 
481  sratom_set_sink(sratom, base_uri,
482  (SerdStatementSink)serd_writer_write_statement,
483  (SerdEndSink)serd_writer_end_anon,
484  writer);
485  sratom_write(sratom, unmap, SERD_EMPTY_S,
486  subject, predicate, type, size, body);
487  serd_writer_finish(writer);
488 
489  serd_writer_free(writer);
490  serd_env_free(env);
491  serd_node_free(&base);
492  return (char*)serd_chunk_sink_finish(&str);
493 }

561 static void
562 read_node(Sratom* sratom,
563  LV2_Atom_Forge* forge,
564  SordWorld* world,
565  SordModel* model,
566  const SordNode* node,
567  ReadMode mode)
568 {
569  LV2_URID_Map* map = sratom->map;
570  size_t len = 0;
571  const char* str = (const char*)sord_node_get_string_counted(node, &len);
572  if (sord_node_get_type(node) == SORD_LITERAL) {
573  char* endptr;
574  SordNode* datatype = sord_node_get_datatype(node);
575  const char* language = sord_node_get_language(node);
576  if (datatype) {
577  const char* type_uri = (const char*)sord_node_get_string(datatype);
578  if (!strcmp(type_uri, (const char*)NS_XSD "int") ||
579  !strcmp(type_uri, (const char*)NS_XSD "integer")) {
580  lv2_atom_forge_int(forge, strtol(str, &endptr, 10));
581  } else if (!strcmp(type_uri, (const char*)NS_XSD "long")) {
582  lv2_atom_forge_long(forge, strtol(str, &endptr, 10));
583  } else if (!strcmp(type_uri, (const char*)NS_XSD "float") ||
584  !strcmp(type_uri, (const char*)NS_XSD "decimal")) {
585  lv2_atom_forge_float(forge, serd_strtod(str, &endptr));
586  } else if (!strcmp(type_uri, (const char*)NS_XSD "double")) {
587  lv2_atom_forge_double(forge, serd_strtod(str, &endptr));
588  } else if (!strcmp(type_uri, (const char*)NS_XSD "boolean")) {
589  lv2_atom_forge_bool(forge, !strcmp(str, "true"));
590  } else if (!strcmp(type_uri, (const char*)NS_XSD "base64Binary")) {
591  size_t size = 0;
592  void* body = serd_base64_decode(USTR(str), len, &size);
593  lv2_atom_forge_atom(forge, size, forge->Chunk);
594  lv2_atom_forge_write(forge, body, size);
595  free(body);
596  } else if (!strcmp(type_uri, LV2_ATOM__Path)) {
597  lv2_atom_forge_path(forge, str, len);
598  } else if (!strcmp(type_uri, LV2_MIDI__MidiEvent)) {
599  lv2_atom_forge_atom(forge, len / 2, sratom->midi_MidiEvent);
600  for (const char* s = str; s < str + len; s += 2) {
601  unsigned num;
602  sscanf(s, "%2X", &num);
603  assert(num < UINT8_MAX);
604  const uint8_t c = num;
605  lv2_atom_forge_raw(forge, &c, 1);
606  }
607  lv2_atom_forge_pad(forge, len / 2);
608  } else {
609  lv2_atom_forge_literal(
610  forge, str, len,
611  sratom->map->map(sratom->map->handle, type_uri),
612  0);
613  }
614  } else if (language) {
615  const char* prefix = "http://lexvo.org/id/iso639-3/";
616  const size_t lang_len = strlen(prefix) + strlen(language);
617  char* lang_uri = (char*)calloc(lang_len + 1, 1);
618  snprintf(lang_uri, lang_len + 1, "%s%s", prefix, language);
619  lv2_atom_forge_literal(
620  forge, str, len, 0,
621  sratom->map->map(sratom->map->handle, lang_uri));
622  free(lang_uri);
623  } else {
624  lv2_atom_forge_string(forge, str, len);
625  }
626  } else if (sord_node_get_type(node) == SORD_URI &&
627  !(sratom->object_mode == SRATOM_OBJECT_MODE_BLANK_SUBJECT
628  && mode == MODE_SUBJECT)) {
629  if (!strcmp(str, (const char*)NS_RDF "nil")) {
630  lv2_atom_forge_atom(forge, 0, 0);
631  } else if (!strncmp(str, "file://", 7)) {
632  uint8_t* path = serd_file_uri_parse((const uint8_t*)str, NULL);
633  lv2_atom_forge_path(forge, (const char*)path, strlen((const char*)path));
634  free(path);
635  } else {
636  lv2_atom_forge_urid(forge, map->map(map->handle, str));
637  }
638  } else {
639  SordNode* type = sord_get(
640  model, node, sratom->nodes.rdf_type, NULL, NULL);
641  SordNode* value = sord_get(
642  model, node, sratom->nodes.rdf_value, NULL, NULL);
643 
644  const uint8_t* type_uri = NULL;
645  uint32_t type_urid = 0;
646  if (type) {
647  type_uri = sord_node_get_string(type);
648  type_urid = map->map(map->handle, (const char*)type_uri);
649  }
650 
651  LV2_Atom_Forge_Frame frame = { 0, 0 };
652  if (mode == MODE_SEQUENCE) {
653  SordNode* frame_time = sord_get(
654  model, node, sratom->nodes.atom_frameTime, NULL, NULL);
655  const char* frame_time_str = frame_time
656  ? (const char*)sord_node_get_string(frame_time)
657  : "";
658  lv2_atom_forge_frame_time(forge, serd_strtod(frame_time_str, NULL));
659  read_node(sratom, forge, world, model, value, MODE_BODY);
660  sord_node_free(world, frame_time);
661  } else if (type_urid == sratom->forge.Tuple) {
662  lv2_atom_forge_tuple(forge, &frame);
663  read_list_value(sratom, forge, world, model, value, MODE_BODY);
664  } else if (type_urid == sratom->forge.Sequence) {
665  lv2_atom_forge_sequence_head(forge, &frame, 0);
666  read_list_value(sratom, forge, world, model, value, MODE_SEQUENCE);
667  } else if (type_urid == sratom->forge.Vector) {
668  SordNode* child_type_node = sord_get(
669  model, node, sratom->nodes.atom_childType, NULL, NULL);
670  uint32_t child_type = map->map(
671  map->handle, (const char*)sord_node_get_string(child_type_node));
672  uint32_t child_size = atom_size(sratom, child_type);
673  if (child_size > 0) {
674  lv2_atom_forge_vector_head(forge, &frame, child_size, child_type);
675  read_list_value(sratom, forge, world, model, value, MODE_BODY);
676  }
677  sord_node_free(world, child_type_node);
678  } else if (value && sord_node_equals(sord_node_get_datatype(value),
679  sratom->nodes.xsd_base64Binary)) {
680  size_t vlen = 0;
681  const uint8_t* vstr = sord_node_get_string_counted(value, &vlen);
682  size_t size = 0;
683  void* body = serd_base64_decode(vstr, vlen, &size);
684  lv2_atom_forge_atom(forge, size, type_urid);
685  lv2_atom_forge_write(forge, body, size);
686  free(body);
687  } else if (sord_node_get_type(node) == SORD_URI) {
688  lv2_atom_forge_object(
689  forge, &frame, map->map(map->handle, str), type_urid);
690  read_resource(sratom, forge, world, model, node, type_urid);
691  } else {
692  lv2_atom_forge_object(forge, &frame, 0, type_urid);
693  read_resource(sratom, forge, world, model, node, type_urid);
694  }
695  if (frame.ref) {
696  lv2_atom_forge_pop(forge, &frame);
697  }
698  sord_node_free(world, value);
699  sord_node_free(world, type);
700  }
701 }
