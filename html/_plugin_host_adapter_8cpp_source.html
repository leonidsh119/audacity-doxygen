67 std::vector<std::string>
68 PluginHostAdapter::getPluginPath()
69 {
70  std::vector<std::string> path;
71  std::string envPath;
72 
73  char *cpath = getenv("VAMP_PATH");
74  if (cpath) envPath = cpath;
75 
76 #ifdef _WIN32
77 #define PATH_SEPARATOR ';'
78 #define DEFAULT_VAMP_PATH "%ProgramFiles%\\Vamp Plugins"
79 #else
80 #define PATH_SEPARATOR ':'
81 #ifdef __APPLE__
82 #define DEFAULT_VAMP_PATH "$HOME/Library/Audio/Plug-Ins/Vamp:/Library/Audio/Plug-Ins/Vamp"
83 #else
84 #define DEFAULT_VAMP_PATH "$HOME/vamp:$HOME/.vamp:/usr/local/lib/vamp:/usr/lib/vamp"
85 #endif
86 #endif
87 
88  if (envPath == "") {
89  envPath = DEFAULT_VAMP_PATH;
90  char *chome = getenv("HOME");
91  if (chome) {
92  std::string home(chome);
93  std::string::size_type f;
94  while ((f = envPath.find("$HOME")) != std::string::npos &&
95  f < envPath.length()) {
96  envPath.replace(f, 5, home);
97  }
98  }
99 #ifdef _WIN32
100  char *cpfiles = getenv("ProgramFiles");
101  if (!cpfiles) cpfiles = (char *)"C:\\Program Files";
102  std::string pfiles(cpfiles);
103  std::string::size_type f;
104  while ((f = envPath.find("%ProgramFiles%")) != std::string::npos &&
105  f < envPath.length()) {
106  envPath.replace(f, 14, pfiles);
107  }
108 #endif
109  }
110 
111  std::string::size_type index = 0, newindex = 0;
112 
113  while ((newindex = envPath.find(PATH_SEPARATOR, index)) < envPath.size()) {
114  path.push_back(envPath.substr(index, newindex - index));
115  index = newindex + 1;
116  }
117 
118  path.push_back(envPath.substr(index));
119 
120  return path;
121 }

316 PluginHostAdapter::OutputList
317 PluginHostAdapter::getOutputDescriptors() const
318 {
319  OutputList list;
320  if (!m_handle) {
321 // std::cerr << "PluginHostAdapter::getOutputDescriptors: no handle " << std::endl;
322  return list;
323  }
324 
325  unsigned int count = m_descriptor->getOutputCount(m_handle);
326 
327  for (unsigned int i = 0; i < count; ++i) {
328  VampOutputDescriptor *sd = m_descriptor->getOutputDescriptor(m_handle, i);
329  OutputDescriptor d;
330  d.identifier = sd->identifier;
331  d.name = sd->name;
332  d.description = sd->description;
333  d.unit = sd->unit;
334  d.hasFixedBinCount = sd->hasFixedBinCount;
335  d.binCount = sd->binCount;
336  if (d.hasFixedBinCount && sd->binNames) {
337  for (unsigned int j = 0; j < sd->binCount; ++j) {
338  d.binNames.push_back(sd->binNames[j] ? sd->binNames[j] : "");
339  }
340  }
341  d.hasKnownExtents = sd->hasKnownExtents;
342  d.minValue = sd->minValue;
343  d.maxValue = sd->maxValue;
344  d.isQuantized = sd->isQuantized;
345  d.quantizeStep = sd->quantizeStep;
346 
347  switch (sd->sampleType) {
348  case vampOneSamplePerStep:
349  d.sampleType = OutputDescriptor::OneSamplePerStep; break;
350  case vampFixedSampleRate:
351  d.sampleType = OutputDescriptor::FixedSampleRate; break;
352  case vampVariableSampleRate:
353  d.sampleType = OutputDescriptor::VariableSampleRate; break;
354  }
355 
356  d.sampleRate = sd->sampleRate;
357 
358  if (m_descriptor->vampApiVersion >= 2) {
359  d.hasDuration = sd->hasDuration;
360  } else {
361  d.hasDuration = false;
362  }
363 
364  list.push_back(d);
365 
366  m_descriptor->releaseOutputDescriptor(sd);
367  }
368 
369  return list;

404 void
405 PluginHostAdapter::convertFeatures(VampFeatureList *features,
406  FeatureSet &fs)
407 {
408  if (!features) return;
409 
410  unsigned int outputs = m_descriptor->getOutputCount(m_handle);
411 
412  for (unsigned int i = 0; i < outputs; ++i) {
413 
414  VampFeatureList &list = features[i];
415 
416  if (list.featureCount > 0) {
417 
418  Feature feature;
419  feature.values.reserve(list.features[0].v1.valueCount);
420 
421  for (unsigned int j = 0; j < list.featureCount; ++j) {
422 
423  feature.hasTimestamp = list.features[j].v1.hasTimestamp;
424  feature.timestamp = RealTime(list.features[j].v1.sec,
425  list.features[j].v1.nsec);
426  feature.hasDuration = false;
427 
428  if (m_descriptor->vampApiVersion >= 2) {
429  unsigned int j2 = j + list.featureCount;
430  feature.hasDuration = list.features[j2].v2.hasDuration;
431  feature.duration = RealTime(list.features[j2].v2.durationSec,
432  list.features[j2].v2.durationNsec);
433  }
434 
435  for (unsigned int k = 0; k < list.features[j].v1.valueCount; ++k) {
436  feature.values.push_back(list.features[j].v1.values[k]);
437  }
438 
439  if (list.features[j].v1.label) {
440  feature.label = list.features[j].v1.label;
441  }
442 
443  fs[i].push_back(feature);
444 
445  if (list.features[j].v1.valueCount > 0) {
446  feature.values.clear();
447  }
448 
449  if (list.features[j].v1.label) {
450  feature.label = "";
451  }
452  }
453  }
454  }
455 }
