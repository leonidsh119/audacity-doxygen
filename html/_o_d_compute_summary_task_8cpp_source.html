59 void ODComputeSummaryTask::DoSomeInternal()
60 {
61  if(mBlockFiles.size()<=0)
62  {
63  mPercentCompleteMutex.Lock();
64  mPercentComplete = 1.0;
65  mPercentCompleteMutex.Unlock();
66  return;
67  }
68 
69  ODPCMAliasBlockFile* bf;
70  sampleCount blockStartSample = 0;
71  sampleCount blockEndSample = 0;
72  bool success =false;
73 
74  mBlockFilesMutex.Lock();
75  for(size_t i=0; i < mWaveTracks.size() && mBlockFiles.size();i++)
76  {
77  bf = mBlockFiles[0];
78 
79  //first check to see if the ref count is at least 2. It should have one
80  //from when we added it to this instance's mBlockFiles array, and one from
81  //the Wavetrack/sequence. If it doesn't it has been deleted and we should forget it.
82  if(bf->RefCount()>=2)
83  {
84  bf->DoWriteSummary();
85  success = true;
86  blockStartSample = bf->GetStart();
87  blockEndSample = blockStartSample + bf->GetLength();
88  mComputedBlockFiles++;
89  }
90  else
91  {
92  //the waveform in the wavetrack now is shorter, so we need to update mMaxBlockFiles
93  //because now there is less work to do.
94  mMaxBlockFiles--;
95  }
96 
97  //Release the refcount we placed on it.
98  bf->Deref();
99  //take it out of the array - we are done with it.
100  mBlockFiles.erase(mBlockFiles.begin());
101 
102  //This is a bit of a convenience in case someone tries to terminate the task by closing the trackpanel or window.
103  //ODComputeSummaryTask::Terminate() uses this lock to remove everything, and we don't want it to wait since the UI is being blocked.
104  mBlockFilesMutex.Unlock();
105  wxThread::This()->Yield();
106  mBlockFilesMutex.Lock();
107 
108  //upddate the gui for all associated blocks. It doesn't matter that we're hitting more wavetracks then we should
109  //because this loop runs a number of times equal to the number of tracks, they probably are getting processed in
110  //the next iteration at the same sample window.
111  mWaveTrackMutex.Lock();
112  for(size_t i=0;i<mWaveTracks.size();i++)
113  {
114  if(success && mWaveTracks[i])
115  mWaveTracks[i]->AddInvalidRegion(blockStartSample,blockEndSample);
116  }
117  mWaveTrackMutex.Unlock();
118  }
119 
120  mBlockFilesMutex.Unlock();
121 
122  //update percentage complete.
123  CalculatePercentComplete();
124 }

171 void ODComputeSummaryTask::Update()
172 {
173  std::vector<ODPCMAliasBlockFile*> tempBlocks;
174 
175  mWaveTrackMutex.Lock();
176 
177  for(size_t j=0;j<mWaveTracks.size();j++)
178  {
179  if(mWaveTracks[j])
180  {
181  WaveClip *clip;
182  BlockArray *blocks;
183  Sequence *seq;
184 
185  //gather all the blockfiles that we should process in the wavetrack.
186  WaveClipList::compatibility_iterator node = mWaveTracks[j]->GetClipIterator();
187 
188  while(node) {
189  clip = node->GetData();
190  seq = clip->GetSequence();
191  //This lock may be way too big since the whole file is one sequence.
192  //TODO: test for large files and find a way to break it down.
193  Sequence::DeleteUpdateMutexLocker locker(*seq);
194 
195  //See Sequence::Delete() for why need this for now..
196  //We don't need the mBlockFilesMutex here because it is only for the vector list.
197  //These are existing blocks, and its wavetrack or blockfiles won't be deleted because
198  //of the respective mWaveTrackMutex lock and LockDeleteUpdateMutex() call.
199  blocks = clip->GetSequenceBlockArray();
200  int i;
201  int insertCursor;
202 
203  insertCursor =0;//OD TODO:see if this works, removed from inner loop (bfore was n*n)
204 
205  for(i=0; i<(int)blocks->size(); i++)
206  {
207  //if there is data but no summary, this blockfile needs summarizing.
208  SeqBlock &block = (*blocks)[i];
209  BlockFile *const file = block.f;
210  if(file->IsDataAvailable() && !file->IsSummaryAvailable())
211  {
212  file->Ref();
213  ODPCMAliasBlockFile *const odpcmaFile = static_cast<ODPCMAliasBlockFile*>(file);
214  odpcmaFile->SetStart(block.start);
215  odpcmaFile->SetClipOffset((sampleCount)(clip->GetStartTime()*clip->GetRate()));
216 
217  //these will always be linear within a sequence-lets take advantage of this by keeping a cursor.
218  while(insertCursor<(int)tempBlocks.size()&&
219  (sampleCount)(tempBlocks[insertCursor]->GetStart()+tempBlocks[insertCursor]->GetClipOffset()) <
220  (sampleCount)(odpcmaFile->GetStart()+odpcmaFile->GetClipOffset()))
221  insertCursor++;
222 
223  tempBlocks.insert(tempBlocks.begin() + insertCursor++, odpcmaFile);
224  }
225  }
226  node = node->GetNext();
227  }
228  }
229  }
230  mWaveTrackMutex.Unlock();
231 
232  //get the NEW order.
233  mBlockFilesMutex.Lock();
234  OrderBlockFiles(tempBlocks);
235  mBlockFilesMutex.Unlock();
236

243 void ODComputeSummaryTask::OrderBlockFiles(std::vector<ODPCMAliasBlockFile*> &unorderedBlocks)
244 {
245  //we are going to take things out of the array. But first deref them since we ref them when we put them in.
246  for(unsigned int i=0;i<mBlockFiles.size();i++)
247  mBlockFiles[i]->Deref();
248  mBlockFiles.clear();
249  //Order the blockfiles into our queue in a fancy convenient way. (this could be user-prefs)
250  //for now just put them in linear. We start the order from the first block that includes the ondemand sample
251  //(which the user sets by clicking.)
252  //Note that this code assumes that the array is sorted in time.
253 
254  //find the startpoint
255  sampleCount processStartSample = GetDemandSample();
256  for(int i= ((int)unorderedBlocks.size())-1;i>= 0;i--)
257  {
258  //check to see if the refcount is at least two before we add it to the list.
259  //There should be one Ref() from the one added by this ODTask, and one from the track.
260  //If there isn't, then the block was deleted for some reason and we should ignore it.
261  if(unorderedBlocks[i]->RefCount()>=2)
262  {
263  //test if the blockfiles are near the task cursor. we use the last mBlockFiles[0] as our point of reference
264  //and add ones that are closer.
265  if(mBlockFiles.size() &&
266  unorderedBlocks[i]->GetGlobalEnd() >= processStartSample &&
267  ( mBlockFiles[0]->GetGlobalEnd() < processStartSample ||
268  unorderedBlocks[i]->GetGlobalStart() <= mBlockFiles[0]->GetGlobalStart())
269  )
270  {
271  //insert at the front of the list if we get blockfiles that are after the demand sample
272  mBlockFiles.insert(mBlockFiles.begin()+0,unorderedBlocks[i]);
273  }
274  else
275  {
276  //otherwise no priority
277  mBlockFiles.push_back(unorderedBlocks[i]);
278  }
279  if(mMaxBlockFiles< (int) mBlockFiles.size())
280  mMaxBlockFiles = mBlockFiles.size();
281  }
282  else
283  {
284  //Otherwise, let it be deleted and forget about it.
285  unorderedBlocks[i]->Deref();
286  }
287  }
