26 /* pv_fetch -- this is an example, but it doesn't really do
27  * phase vocoding. Instead, it will just multiply f, g, and x
28  *
29  * To make things a bit more interesting, we will assume g has
30  * an arbitrary sample rate with respect to f, and will interpolate.
31  *
32  */
33 long pv_fetch(pvshell_type susp,
34  sample_block_values_type out, long *n)
35 {
36  int i;
37  for (i = 0; i < *n; i++) {
38  long new_flags;
39  sample_type f;
40  double g;
41  /* NOTE: in DSP terms, this is poor code because of the
42  * division operations -- it could be made faster
43  */
44  /* To get a value from g, first compute the time */
45  double f_time = susp->F_COUNT / susp->f->sr;
46  /* Now compute g count that is past the time */
47  double g_count = f_time * susp->g->sr;
48  while (susp->G_COUNT < g_count) {
49  PVSHELL_TEST_G(susp); /* prepare to get a sample */
50  /* ignore flags from g -- we could, if we wanted,
51  * terminate when either f or g terminated, etc.
52  */
53  susp->G_PREV = susp->G_NEXT;
54  susp->G_NEXT = PVSHELL_FETCH_G(susp);
55  susp->G_COUNT++;
56  }
57  /* now interpolate to get the value of g at f_time */
58  g = susp->G_PREV + (susp->G_NEXT - susp->G_PREV) *
59  (g_count - (susp->G_COUNT - 1));
60  new_flags = PVSHELL_TEST_F(susp);
61  susp->flags |= new_flags;
62  if (new_flags) break;
63  f = PVSHELL_FETCH_F(susp);
64  susp->F_COUNT++; /* count how many samples we have taken */
65 
66  /* now we have f, g, x */
67  *out++ = f * g * susp->X_VALUE;
68  }
69  /* i is the number of samples we acutally computed */
70  *n = i;
71  /* if we computed samples, we want to return them before
72  * returning flags that say we're done or stopped
73  */
74  return (i ? 0 : susp->flags);
75 }

78 sound_type snd_phasevocoder(sound_type f, sound_type g, double x)
79 {
80  /* we're using 5 doubles of state. The first is a parameter,
81  * and the rest are initialized to zero except for state[2],
82  * aka G_COUNT. This is the number of samples we have read
83  * from G. Since we're interpolating we need a one-sample
84  * lookahead, and initializing the count to -1 causes an
85  * extra fetch and hence 1-sample lookahead. This state is copied
86  * into the pvshell structure, so we don't need to allocate
87  * a vector on the heap.
88  */
89  double state[5] = {0, 0, -1, 0, 0};
90  state[0] = x;
91  /* If f and g do not start at the same time, we should really
92  * should do something about it, but we'll just throw an error.
93  * Be careful to allow small differences (within one sample).
94  */
95  if (fabs(f->t0 - g->t0) * f->sr > 0.5) {
96  xlfail("phasevocoder inputs must start at the same time");
97  }
98  /* output the same sample rate and start time as f */
99  return snd_make_pvshell("snd_phasevocoder", f->sr, f->t0,
100  &pv_fetch, f, g,
101  state, sizeof(state) / sizeof(state[0]));
