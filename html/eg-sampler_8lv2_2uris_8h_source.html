98 static inline const LV2_Atom*
99 read_set_file(const SamplerURIs* uris,
100  const LV2_Atom_Object* obj)
101 {
102  if (obj->body.otype != uris->patch_Set) {
103  fprintf(stderr, "Ignoring unknown message type %d\n", obj->body.otype);
104  return NULL;
105  }
106 
107  /* Get property URI. */
108  const LV2_Atom* property = NULL;
109  lv2_atom_object_get(obj, uris->patch_property, &property, 0);
110  if (!property) {
111  fprintf(stderr, "Malformed set message has no body.\n");
112  return NULL;
113  } else if (property->type != uris->atom_URID) {
114  fprintf(stderr, "Malformed set message has non-URID property.\n");
115  return NULL;
116  } else if (((const LV2_Atom_URID*)property)->body != uris->eg_sample) {
117  fprintf(stderr, "Set message for unknown property.\n");
118  return NULL;
119  }
120 
121  /* Get value. */
122  const LV2_Atom* file_path = NULL;
123  lv2_atom_object_get(obj, uris->patch_value, &file_path, 0);
124  if (!file_path) {
125  fprintf(stderr, "Malformed set message has no value.\n");
126  return NULL;
127  } else if (file_path->type != uris->atom_Path) {
128  fprintf(stderr, "Set message value is not a Path.\n");
129  return NULL;
130  }
131 
132  return file_path;
133 }
