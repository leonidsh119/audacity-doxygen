111 static Sample*
112 load_sample(Sampler* self, const char* path)
113 {
114  const size_t path_len = strlen(path);
115 
116  lv2_log_trace(&self->logger, "Loading sample %s\n", path);
117 
118  Sample* const sample = (Sample*)malloc(sizeof(Sample));
119  SF_INFO* const info = &sample->info;
120  SNDFILE* const sndfile = sf_open(path, SFM_READ, info);
121 
122  if (!sndfile || !info->frames || (info->channels != 1)) {
123  lv2_log_error(&self->logger, "Failed to open sample '%s'\n", path);
124  free(sample);
125  return NULL;
126  }
127 
128  // Read data
129  float* const data = malloc(sizeof(float) * info->frames);
130  if (!data) {
131  lv2_log_error(&self->logger, "Failed to allocate memory for sample\n");
132  return NULL;
133  }
134  sf_seek(sndfile, 0ul, SEEK_SET);
135  sf_read_float(sndfile, data, info->frames);
136  sf_close(sndfile);
137 
138  // Fill sample struct and return it
139  sample->data = data;
140  sample->path = (char*)malloc(path_len + 1);
141  sample->path_len = (uint32_t)path_len;
142  memcpy(sample->path, path, path_len + 1);
143 
144  return sample;
145 }

165 static LV2_Worker_Status
166 work(LV2_Handle instance,
167  LV2_Worker_Respond_Function respond,
168  LV2_Worker_Respond_Handle handle,
169  uint32_t size,
170  const void* data)
171 {
172  Sampler* self = (Sampler*)instance;
173  const LV2_Atom* atom = (const LV2_Atom*)data;
174  if (atom->type == self->uris.eg_freeSample) {
175  // Free old sample
176  const SampleMessage* msg = (const SampleMessage*)data;
177  free_sample(self, msg->sample);
178  } else {
179  // Handle set message (load sample).
180  const LV2_Atom_Object* obj = (const LV2_Atom_Object*)data;
181 
182  // Get file path from message
183  const LV2_Atom* file_path = read_set_file(&self->uris, obj);
184  if (!file_path) {
185  return LV2_WORKER_ERR_UNKNOWN;
186  }
187 
188  // Load sample.
189  Sample* sample = load_sample(self, LV2_ATOM_BODY_CONST(file_path));
190  if (sample) {
191  // Loaded sample, send it to run() to be applied.
192  respond(handle, sizeof(sample), &sample);
193  }
194  }
195 
196  return LV2_WORKER_SUCCESS;
197 }

206 static LV2_Worker_Status
207 work_response(LV2_Handle instance,
208  uint32_t size,
209  const void* data)
210 {
211  Sampler* self = (Sampler*)instance;
212 
213  SampleMessage msg = { { sizeof(Sample*), self->uris.eg_freeSample },
214  self->sample };
215 
216  // Send a message to the worker to free the current sample
217  self->schedule->schedule_work(self->schedule->handle, sizeof(msg), &msg);
218 
219  // Install the new sample
220  self->sample = *(Sample*const*)data;
221 
222  // Send a notification that we're using a new sample.
223  lv2_atom_forge_frame_time(&self->forge, self->frame_offset);
224  write_set_file(&self->forge, &self->uris,
225  self->sample->path,
226  self->sample->path_len);
227 
228  return LV2_WORKER_SUCCESS;

252 static LV2_Handle
253 instantiate(const LV2_Descriptor* descriptor,
254  double rate,
255  const char* path,
256  const LV2_Feature* const* features)
257 {
258  // Allocate and initialise instance structure.
259  Sampler* self = (Sampler*)malloc(sizeof(Sampler));
260  if (!self) {
261  return NULL;
262  }
263  memset(self, 0, sizeof(Sampler));
264 
265  // Get host features
266  for (int i = 0; features[i]; ++i) {
267  if (!strcmp(features[i]->URI, LV2_URID__map)) {
268  self->map = (LV2_URID_Map*)features[i]->data;
269  } else if (!strcmp(features[i]->URI, LV2_WORKER__schedule)) {
270  self->schedule = (LV2_Worker_Schedule*)features[i]->data;
271  } else if (!strcmp(features[i]->URI, LV2_LOG__log)) {
272  self->log = (LV2_Log_Log*)features[i]->data;
273  }
274  }
275  if (!self->map) {
276  lv2_log_error(&self->logger, "Missing feature urid:map\n");
277  goto fail;
278  } else if (!self->schedule) {
279  lv2_log_error(&self->logger, "Missing feature work:schedule\n");
280  goto fail;
281  }
282 
283  // Map URIs and initialise forge/logger
284  map_sampler_uris(self->map, &self->uris);
285  lv2_atom_forge_init(&self->forge, self->map);
286  lv2_log_logger_init(&self->logger, self->map, self->log);
287 
288  // Load the default sample file
289  const size_t path_len = strlen(path);
290  const size_t file_len = strlen(default_sample_file);
291  const size_t len = path_len + file_len;
292  char* sample_path = (char*)malloc(len + 1);
293  snprintf(sample_path, len + 1, "%s%s", path, default_sample_file);
294  self->sample = load_sample(self, sample_path);
295  free(sample_path);
296 
297  return (LV2_Handle)self;
298 
299 fail:
300  free(self);
301  return 0;
302 }

312 static void
313 run(LV2_Handle instance,
314  uint32_t sample_count)
315 {
316  Sampler* self = (Sampler*)instance;
317  SamplerURIs* uris = &self->uris;
318  sf_count_t start_frame = 0;
319  sf_count_t pos = 0;
320  float* output = self->output_port;
321 
322  // Set up forge to write directly to notify output port.
323  const uint32_t notify_capacity = self->notify_port->atom.size;
324  lv2_atom_forge_set_buffer(&self->forge,
325  (uint8_t*)self->notify_port,
326  notify_capacity);
327 
328  // Start a sequence in the notify output port.
329  lv2_atom_forge_sequence_head(&self->forge, &self->notify_frame, 0);
330 
331  // Read incoming events
332  LV2_ATOM_SEQUENCE_FOREACH(self->control_port, ev) {
333  self->frame_offset = ev->time.frames;
334  if (ev->body.type == uris->midi_Event) {
335  const uint8_t* const msg = (const uint8_t*)(ev + 1);
336  switch (lv2_midi_message_type(msg)) {
337  case LV2_MIDI_MSG_NOTE_ON:
338  start_frame = ev->time.frames;
339  self->frame = 0;
340  self->play = true;
341  break;
342  default:
343  break;
344  }
345  } else if (lv2_atom_forge_is_object_type(&self->forge, ev->body.type)) {
346  const LV2_Atom_Object* obj = (const LV2_Atom_Object*)&ev->body;
347  if (obj->body.otype == uris->patch_Set) {
348  // Received a set message, send it to the worker.
349  lv2_log_trace(&self->logger, "Queueing set message\n");
350  self->schedule->schedule_work(self->schedule->handle,
351  lv2_atom_total_size(&ev->body),
352  &ev->body);
353  } else {
354  lv2_log_trace(&self->logger,
355  "Unknown object type %d\n", obj->body.otype);
356  }
357  } else {
358  lv2_log_trace(&self->logger,
359  "Unknown event type %d\n", ev->body.type);
360  }
361  }
362 
363  // Render the sample (possibly already in progress)
364  if (self->play) {
365  uint32_t f = self->frame;
366  const uint32_t lf = self->sample->info.frames;
367 
368  for (pos = 0; pos < start_frame; ++pos) {
369  output[pos] = 0;
370  }
371 
372  for (; pos < sample_count && f < lf; ++pos, ++f) {
373  output[pos] = self->sample->data[f];
374  }
375 
376  self->frame = f;
377 
378  if (f == lf) {
379  self->play = false;
380  }
381  }
382 
383  // Add zeros to end if sample not long enough (or not playing)
384  for (; pos < sample_count; ++pos) {
385  output[pos] = 0.0f;
386  }
387 }
