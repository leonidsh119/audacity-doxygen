222 {
223  // Retrieve data about all LV2 plugins
224  const LilvPlugins *plugs = lilv_world_get_all_plugins(gWorld);
225 
226  // Iterate over all plugins retrieve their URI
227  wxArrayString plugins;
228  LILV_FOREACH(plugins, i, plugs)

239  plugins.Add(LilvString(lilv_plugin_get_uri(plug)));
