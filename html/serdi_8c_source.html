40 static int
41 print_usage(const char* name, bool error)
42 {
43  FILE* const os = error ? stderr : stdout;
44  fprintf(os, "Usage: %s [OPTION]... INPUT [BASE_URI]\n", name);
45  fprintf(os, "Read and write RDF syntax.\n");
46  fprintf(os, "Use - for INPUT to read from standard input.\n\n");
47  fprintf(os, " -b Fast bulk output for large serialisations.\n");
48  fprintf(os, " -c PREFIX Chop PREFIX from matching blank node IDs.\n");
49  fprintf(os, " -e Eat input one character at a time.\n");
50  fprintf(os, " -f Keep full URIs in input (don't qualify).\n");
51  fprintf(os, " -h Display this help and exit.\n");
52  fprintf(os, " -i SYNTAX Input syntax (`turtle' or `ntriples').\n");
53  fprintf(os, " -o SYNTAX Output syntax (`turtle' or `ntriples').\n");
54  fprintf(os, " -p PREFIX Add PREFIX to blank node IDs.\n");
55  fprintf(os, " -q Suppress all output except data.\n");
56  fprintf(os, " -r ROOT_URI Keep relative URIs within ROOT_URI.\n");
57  fprintf(os, " -s INPUT Parse INPUT as string (terminates options).\n");
58  fprintf(os, " -v Display version information and exit.\n");
59  return error ? 1 : 0;
60 }

89 int
90 main(int argc, char** argv)
91 {
92  if (argc < 2) {
93  return print_usage(argv[0], true);
94  }
95 
96  FILE* in_fd = NULL;
97  SerdSyntax input_syntax = SERD_TURTLE;
98  SerdSyntax output_syntax = SERD_NTRIPLES;
99  bool from_file = true;
100  bool bulk_read = true;
101  bool bulk_write = false;
102  bool full_uris = false;
103  bool quiet = false;
104  const uint8_t* in_name = NULL;
105  const uint8_t* add_prefix = NULL;
106  const uint8_t* chop_prefix = NULL;
107  const uint8_t* root_uri = NULL;
108  int a = 1;
109  for (; a < argc && argv[a][0] == '-'; ++a) {
110  if (argv[a][1] == '\0') {
111  in_name = (const uint8_t*)"(stdin)";
112  in_fd = stdin;
113  break;
114  } else if (argv[a][1] == 'b') {
115  bulk_write = true;
116  } else if (argv[a][1] == 'e') {
117  bulk_read = false;
118  } else if (argv[a][1] == 'f') {
119  full_uris = true;
120  } else if (argv[a][1] == 'h') {
121  return print_usage(argv[0], false);
122  } else if (argv[a][1] == 'q') {
123  quiet = true;
124  } else if (argv[a][1] == 'v') {
125  return print_version();
126  } else if (argv[a][1] == 's') {
127  in_name = (const uint8_t*)"(string)";
128  from_file = false;
129  ++a;
130  break;
131  } else if (argv[a][1] == 'i') {
132  if (++a == argc || !set_syntax(&input_syntax, argv[a])) {
133  return bad_arg(argv[0], 'i');
134  }
135  } else if (argv[a][1] == 'o') {
136  if (++a == argc || !set_syntax(&output_syntax, argv[a])) {
137  return bad_arg(argv[0], 'o');
138  }
139  } else if (argv[a][1] == 'p') {
140  if (++a == argc) {
141  return bad_arg(argv[0], 'p');
142  }
143  add_prefix = (const uint8_t*)argv[a];
144  } else if (argv[a][1] == 'c') {
145  if (++a == argc) {
146  return bad_arg(argv[0], 'c');
147  }
148  chop_prefix = (const uint8_t*)argv[a];
149  } else if (argv[a][1] == 'r') {
150  if (++a == argc) {
151  return bad_arg(argv[0], 'r');
152  }
153  root_uri = (const uint8_t*)argv[a];
154  } else {
155  fprintf(stderr, "%s: Unknown option `%s'\n", argv[0], argv[a]);
156  return print_usage(argv[0], true);
157  }
158  }
159 
160  if (a == argc) {
161  fprintf(stderr, "%s: Missing input\n", argv[0]);
162  return 1;
163  }
164 
165  const uint8_t* input = (const uint8_t*)argv[a++];
166  if (from_file) {
167  in_name = in_name ? in_name : input;
168  if (!in_fd) {
169  input = serd_uri_to_path(in_name);
170  if (!input || !(in_fd = serd_fopen((const char*)input, "r"))) {
171  return 1;
172  }
173  }
174  }
175 
176  SerdURI base_uri = SERD_URI_NULL;
177  SerdNode base = SERD_NODE_NULL;
178  if (a < argc) { // Base URI given on command line
179  base = serd_node_new_uri_from_string(
180  (const uint8_t*)argv[a], NULL, &base_uri);
181  } else if (from_file && in_fd != stdin) { // Use input file URI
182  base = serd_node_new_file_uri(input, NULL, &base_uri, false);
183  }
184 
185  FILE* out_fd = stdout;
186  SerdEnv* env = serd_env_new(&base);
187 
188  int output_style = 0;
189  if (output_syntax == SERD_NTRIPLES) {
190  output_style |= SERD_STYLE_ASCII;
191  } else {
192  output_style |= SERD_STYLE_ABBREVIATED;
193  if (!full_uris) {
194  output_style |= SERD_STYLE_CURIED;
195  }
196  }
197 
198  if (input_syntax != SERD_NTRIPLES || (output_style & SERD_STYLE_CURIED)) {
199  // Base URI may change and/or we're abbreviating URIs, so must resolve
200  output_style |= SERD_STYLE_RESOLVED; // Base may chan
201  }
202 
203  if (bulk_write) {
204  output_style |= SERD_STYLE_BULK;
205  }
206 
207  SerdWriter* writer = serd_writer_new(
208  output_syntax, (SerdStyle)output_style,
209  env, &base_uri, serd_file_sink, out_fd);
210 
211  SerdReader* reader = serd_reader_new(
212  input_syntax, writer, NULL,
213  (SerdBaseSink)serd_writer_set_base_uri,
214  (SerdPrefixSink)serd_writer_set_prefix,
215  (SerdStatementSink)serd_writer_write_statement,
216  (SerdEndSink)serd_writer_end_anon);
217 
218  if (quiet) {
219  serd_reader_set_error_sink(reader, quiet_error_sink, NULL);
220  serd_writer_set_error_sink(writer, quiet_error_sink, NULL);
221  }
222 
223  SerdNode root = serd_node_from_string(SERD_URI, root_uri);
224  serd_writer_set_root_uri(writer, &root);
225  serd_writer_chop_blank_prefix(writer, chop_prefix);
226  serd_reader_add_blank_prefix(reader, add_prefix);
227 
228  SerdStatus status = SERD_SUCCESS;
229  if (!from_file) {
230  status = serd_reader_read_string(reader, input);
231  } else if (bulk_read) {
232  status = serd_reader_read_file_handle(reader, in_fd, in_name);
233  } else {
234  status = serd_reader_start_stream(reader, in_fd, in_name, false);
235  while (!status) {
236  status = serd_reader_read_chunk(reader);
237  }
238  serd_reader_end_stream(reader);
239  }
240 
241  serd_reader_free(reader);
242 
243  if (from_file) {
244  fclose(in_fd);
245  }
246 
247  serd_writer_finish(writer);
248  serd_writer_free(writer);
249  serd_env_free(env);
250  serd_node_free(&base);
251 
252  return (status > SERD_FAILURE) ? 1 : 0;
253 }
