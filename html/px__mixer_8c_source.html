112 PxMixer *Px_OpenMixer(PaStream *pa_stream, int i)
113 {
114  px_mixer *Px;
115  int good = TRUE;
116 
117  Px = (px_mixer *) malloc(sizeof(px_mixer));
118  if (Px == NULL) {
119  return NULL;
120  }
121 
122  Px->magic = PX_MIXER_MAGIC;
123  Px->pa_stream = pa_stream;
124  Px->info = NULL;
125 
126  if (!initialize(Px)) {
127  free(Px);
128  return NULL;
129  }
130 
131  switch (Pa_GetStreamHostApiType(pa_stream))
132  {
133 #if defined(PX_USE_WIN_MME)
134  case paMME:
135  good = OpenMixer_Win_MME(Px, i);
136  break;
137 #endif
138 
139 #if defined(PX_USE_WIN_DSOUND)
140  case paDirectSound:
141  good = OpenMixer_Win_DirectSound(Px, i);
142  break;
143 #endif
144 
145 #if defined(PX_USE_WIN_WASAPI)
146  case paWASAPI:
147  good = OpenMixer_Win_WASAPI(Px, i);
148  break;
149 #endif
150 
151 #if defined(PX_USE_MAC_COREAUDIO)
152  case paCoreAudio:
153  good = OpenMixer_Mac_CoreAudio(Px, i);
154  break;
155 #endif
156 
157 #if defined(PX_USE_UNIX_OSS)
158  case paOSS:
159  good = OpenMixer_Unix_OSS(Px, i);
160  break;
161 #endif
162 
163 #if defined(PX_USE_LINUX_ALSA)
164  case paALSA:
165  good = OpenMixer_Linux_ALSA(Px, i);
166  break;
167 #endif
168 
169  default:
170  good = FALSE;
171  break;
172  }
173 
174  if (!good) {
175  free(Px);
176  return NULL;
177  }
178 
179  return Px;
180 }
