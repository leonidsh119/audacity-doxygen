365 bool EffectTruncSilence::FindSilences
366  (RegionList &silences, TrackList *list, Track *firstTrack, Track *lastTrack)
367 {
368  // Start with the whole selection silent
369  silences.push_back(Region(mT0, mT1));
370 
371  // Remove non-silent regions in each track
372  SelectedTrackListOfKindIterator iter(Track::Wave, list);
373  int whichTrack = 0;
374  bool lastSeen = false;
375  for (Track *t = iter.StartWith(firstTrack); !lastSeen && t; t = iter.Next())
376  {
377  lastSeen = (t == lastTrack);
378  WaveTrack *const wt = static_cast<WaveTrack *>(t);
379 
380  // Smallest silent region to detect in frames
381  sampleCount minSilenceFrames =
382  sampleCount(std::max(mInitialAllowedSilence, DEF_MinTruncMs) * wt->GetRate());
383 
384  //
385  // Scan the track for silences
386  //
387  RegionList trackSilences;
388 
389  sampleCount index = wt->TimeToLongSamples(mT0);
390  sampleCount silentFrame = 0;
391 
392  // Detect silences
393  bool cancelled = !(Analyze(silences, trackSilences, wt, &silentFrame, &index, whichTrack));
394 
395  // Buffer has been freed, so we're OK to return if cancelled
396  if (cancelled)
397  {
398  ReplaceProcessedTracks(false);
399  return false;
400  }
401 
402  if (silentFrame >= minSilenceFrames)
403  {
404  // Track ended in silence -- record region
405  trackSilences.push_back(Region(
406  wt->LongSamplesToTime(index - silentFrame),
407  wt->LongSamplesToTime(index)
408  ));
409  }
410 
411  // Intersect with the overall silent region list
412  Intersect(silences, trackSilences);
413  whichTrack++;
414  }
415 
416  return true;

419 bool EffectTruncSilence::DoRemoval
420 (const RegionList &silences, unsigned iGroup, unsigned nGroups, Track *firstTrack, Track *lastTrack,
421  double &totalCutLen)
422 {
423  //
424  // Now remove the silent regions from all selected / sync-lock selected tracks.
425  //
426 
427  // Loop over detected regions in reverse (so cuts don't change time values
428  // down the line)
429  int whichReg = 0;
430  RegionList::const_reverse_iterator rit;
431  for (rit = silences.rbegin(); rit != silences.rend(); ++rit)
432  {
433  const Region &region = *rit;
434  const Region *const r = &region;
435 
436  // Progress dialog and cancellation. Do additional cleanup before return.
437  const double frac = detectFrac +
438  (1 - detectFrac) * (iGroup + whichReg / double(silences.size())) / nGroups;
439  if (TotalProgress(frac))
440  {
441  ReplaceProcessedTracks(false);
442  return false;
443  }
444 
445  // Intersection may create regions smaller than allowed; ignore them.
446  // Allow one nanosecond extra for consistent results with exact milliseconds of allowed silence.
447  if ((r->end - r->start) < (mInitialAllowedSilence - 0.000000001))
448  continue;
449 
450  // Find NEW silence length as requested
451  double inLength = r->end - r->start;
452  double outLength;
453 
454  switch (mActionIndex)
455  {
456  case kTruncate:
457  outLength = std::min(mTruncLongestAllowedSilence, inLength);
458  break;
459  case kCompress:
460  outLength = mInitialAllowedSilence +
461  (inLength - mInitialAllowedSilence) * mSilenceCompressPercent / 100.0;
462  break;
463  default: // Not currently used.
464  outLength = std::min(mInitialAllowedSilence +
465  (inLength - mInitialAllowedSilence) * mSilenceCompressPercent / 100.0,
466  mTruncLongestAllowedSilence);
467  }
468 
469  double cutLen = inLength - outLength;
470  totalCutLen += cutLen;
471 
472  TrackListIterator iterOut(mOutputTracks);
473  bool lastSeen = false;
474  for (Track *t = iterOut.StartWith(firstTrack); t && !lastSeen; t = iterOut.Next())
475  {
476  lastSeen = (t == lastTrack);
477  if (!(t->GetSelected() || t->IsSyncLockSelected()))
478  continue;
479 
480  // Don't waste time past the end of a track
481  if (t->GetEndTime() < r->start)
482  continue;
483 
484  double cutStart = (r->start + r->end - cutLen) / 2;
485  double cutEnd = cutStart + cutLen;
486  if (t->GetKind() == Track::Wave)
487  {
488  // In WaveTracks, clear with a cross-fade
489  WaveTrack *const wt = static_cast<WaveTrack*>(t);
490  sampleCount blendFrames = mBlendFrameCount;
491  // Round start/end times to frame boundaries
492  cutStart = wt->LongSamplesToTime(wt->TimeToLongSamples(cutStart));
493  cutEnd = wt->LongSamplesToTime(wt->TimeToLongSamples(cutEnd));
494 
495  // Make sure the cross-fade does not affect non-silent frames
496  if (wt->LongSamplesToTime(blendFrames) > inLength)
497  {
498  blendFrames = wt->TimeToLongSamples(inLength);
499  }
500 
501  // Perform cross-fade in memory
502  float *buf1 = new float[blendFrames];
503  float *buf2 = new float[blendFrames];
504  sampleCount t1 = wt->TimeToLongSamples(cutStart) - blendFrames / 2;
505  sampleCount t2 = wt->TimeToLongSamples(cutEnd) - blendFrames / 2;
506 
507  wt->Get((samplePtr)buf1, floatSample, t1, blendFrames);
508  wt->Get((samplePtr)buf2, floatSample, t2, blendFrames);
509 
510  for (sampleCount i = 0; i < blendFrames; ++i)
511  {
512  buf1[i] = ((blendFrames-i) * buf1[i] + i * buf2[i]) /
513  (double)blendFrames;
514  }
515 
516  // Perform the cut
517  wt->Clear(cutStart, cutEnd);
518 
519  // Write cross-faded data
520  wt->Set((samplePtr)buf1, floatSample, t1, blendFrames);
521 
522  delete [] buf1;
523  delete [] buf2;
524  }
525  else
526  // Non-wave tracks: just do a sync-lock adjust
527  t->SyncLockAdjust(cutEnd, cutStart);
528  }
529  ++whichReg;
530  }
531 
532  return true;
533 }

535 bool EffectTruncSilence::Analyze(RegionList& silenceList,
536  RegionList& trackSilences,
537  WaveTrack* wt,
538  sampleCount* silentFrame,
539  sampleCount* index,
540  int whichTrack,
541  double* inputLength /*= NULL*/,
542  double* minInputLength /*= NULL*/)
543 {
544  // Smallest silent region to detect in frames
545  sampleCount minSilenceFrames = sampleCount(std::max( mInitialAllowedSilence, DEF_MinTruncMs) * wt->GetRate());
546 
547  double truncDbSilenceThreshold = Enums::Db2Signal[mTruncDbChoiceIndex];
548  sampleCount blockLen = wt->GetMaxBlockSize();
549  sampleCount start = wt->TimeToLongSamples(mT0);
550  sampleCount end = wt->TimeToLongSamples(mT1);
551  sampleCount outLength = 0;
552 
553  double previewLength;
554  gPrefs->Read(wxT("/AudioIO/EffectsPreviewLen"), &previewLength, 6.0);
555  // Minimum required length in samples.
556  const sampleCount previewLen = previewLength * wt->GetRate();
557 
558  // Keep position in overall silences list for optimization
559  RegionList::iterator rit(silenceList.begin());
560 
561  // Allocate buffer
562  float *buffer = new float[blockLen];
563 
564  // Loop through current track
565  while (*index < end) {
566  if (inputLength && ((outLength >= previewLen) || (*index - start > wt->TimeToLongSamples(*minInputLength)))) {
567  *inputLength = std::min<double>(*inputLength, *minInputLength);
568  if (outLength >= previewLen) {
569  *minInputLength = *inputLength;
570  }
571  return true;
572  }
573 
574  if (!inputLength) {
575  // Show progress dialog, test for cancellation
576  bool cancelled = TotalProgress(
577  detectFrac * (whichTrack + (*index - start) / (double)(end - start)) /
578  (double)GetNumWaveTracks());
579  if (cancelled) {
580  delete [] buffer;
581  return false;
582  }
583  }
584 
585  // Optimization: if not in a silent region skip ahead to the next one
586 
587  double curTime = wt->LongSamplesToTime(*index);
588  for ( ; rit != silenceList.end(); ++rit) {
589  // Find the first silent region ending after current time
590  if (rit->end >= curTime) {
591  break;
592  }
593  }
594 
595  if (rit == silenceList.end()) {
596  // No more regions -- no need to process the rest of the track
597  if (inputLength) {
598  // Add available samples up to previewLength.
599  sampleCount remainingTrackSamples = wt->TimeToLongSamples(wt->GetEndTime()) - *index;
600  sampleCount requiredTrackSamples = previewLen - outLength;
601  outLength += (remainingTrackSamples > requiredTrackSamples)? requiredTrackSamples : remainingTrackSamples;
602  }
603 
604  break;
605  }
606  else if (rit->start > curTime) {
607  // End current silent region, skip ahead
608  if (*silentFrame >= minSilenceFrames) {
609  trackSilences.push_back(Region(
610  wt->LongSamplesToTime(*index - *silentFrame),
611  wt->LongSamplesToTime(*index)
612  ));
613  }
614  *silentFrame = 0;
615  sampleCount newIndex = wt->TimeToLongSamples(rit->start);
616  if (inputLength) {
617  sampleCount requiredTrackSamples = previewLen - outLength;
618  // Add non-silent sample to outLength
619  outLength += ((newIndex - *index) > requiredTrackSamples)? requiredTrackSamples : newIndex - *index;
620  }
621 
622  *index = newIndex;
623  }
624  // End of optimization
625 
626  // Limit size of current block if we've reached the end
627  sampleCount count = blockLen;
628  if ((*index + count) > end) {
629  count = end - *index;
630  }
631 
632  // Fill buffer
633  wt->Get((samplePtr)(buffer), floatSample, *index, count);
634 
635  // Look for silenceList in current block
636  for (sampleCount i = 0; i < count; ++i) {
637  if (inputLength && ((outLength >= previewLen) || (outLength > wt->TimeToLongSamples(*minInputLength)))) {
638  *inputLength = wt->LongSamplesToTime(*index + i) - wt->LongSamplesToTime(start);
639  break;
640  }
641 
642  if (fabs(buffer[i]) < truncDbSilenceThreshold) {
643  (*silentFrame)++;
644  }
645  else {
646  sampleCount allowed = 0;
647  if (*silentFrame >= minSilenceFrames) {
648  if (inputLength) {
649  switch (mActionIndex) {
650  case kTruncate:
651  outLength += wt->TimeToLongSamples(mTruncLongestAllowedSilence);
652  break;
653  case kCompress:
654  allowed = wt->TimeToLongSamples(mInitialAllowedSilence);
655  outLength += allowed +
656  (*silentFrame - allowed) * mSilenceCompressPercent / 100.0;
657  break;
658  // default: // Not currently used.
659  }
660  }
661 
662  // Record the silent region
663  Region *r = new Region;
664  r->start = wt->LongSamplesToTime(*index + i - *silentFrame);
665  r->end = wt->LongSamplesToTime(*index + i);
666  trackSilences.push_back(Region(
667  wt->LongSamplesToTime(*index + i - *silentFrame),
668  wt->LongSamplesToTime(*index + i)
669  ));
670  }
671  else if (inputLength) { // included as part of non-silence
672  outLength += *silentFrame;
673  }
674  *silentFrame = 0;
675  if (inputLength) {
676  ++outLength; // Add non-silent sample to outLength
677  }
678  }
679  }
680  // Next block
681  *index += count;
682  }
683  delete [] buffer;
684 
685  if (inputLength) {
686  *inputLength = std::min<double>(*inputLength, *minInputLength);
687  if (outLength >= previewLen) {
688  *minInputLength = *inputLength;
689  }
690  }
691 
692  return true;
693 }

793 // Finds the intersection of the ordered region lists, stores in dest
794 void EffectTruncSilence::Intersect(RegionList &dest, const RegionList &src)
795 {
796  RegionList::iterator destIter;
797  destIter = dest.begin();
798  // Any time we reach the end of the dest list we're finished
799  if (destIter == dest.end())
800  return;
801  RegionList::iterator curDest = destIter;
802 
803  // Operation: find non-silent regions in src, remove them from dest.
804  double nsStart = curDest->start;
805  double nsEnd;
806  bool lastRun = false; // must run the loop one extra time
807 
808  RegionList::const_iterator srcIter = src.begin();
809 
810  // This logic, causing the loop to run once after end of src, must occur
811  // each time srcIter is updated
812  if (srcIter == src.end())
813  {
814  lastRun = true;
815  }
816 
817  while (srcIter != src.end() || lastRun)
818  {
819  // Don't use curSrc unless lastRun is false!
820  RegionList::const_iterator curSrc;
821 
822  if (lastRun)
823  {
824  // The last non-silent region extends as far as possible
825  nsEnd = std::numeric_limits<double>::max();
826  }
827  else
828  {
829  curSrc = srcIter;
830  nsEnd = curSrc->start;
831  }
832 
833  if (nsEnd > nsStart)
834  {
835  // Increment through dest until we have a region that could be affected
836  while (curDest->end <= nsStart)
837  {
838  ++destIter;
839  if (destIter == dest.end())
840  {
841  return;
842  }
843  curDest = destIter;
844  }
845 
846  // Check for splitting dest region in two
847  if (nsStart > curDest->start && nsEnd < curDest->end)
848  {
849  // The second region
850  Region r(nsEnd, curDest->end);
851 
852  // The first region
853  curDest->end = nsStart;
854 
855  // Insert second region after first
856  RegionList::iterator nextIt(destIter);
857  ++nextIt;
858 
859  // This should just read: destIter = dest.insert(nextIt, r); but we
860  // work around two two wxList::insert() bugs. First, in some
861  // versions it returns the wrong value. Second, in some versions,
862  // it crashes when you insert at list end.
863  if (nextIt == dest.end())
864  dest.push_back(r);
865  else
866  dest.insert(nextIt, r);
867  ++destIter; // (now points at the newly-inserted region)
868 
869  curDest = destIter;
870  }
871 
872  // Check for truncating the end of dest region
873  if (nsStart > curDest->start && nsStart < curDest->end &&
874  nsEnd >= curDest->end)
875  {
876  curDest->end = nsStart;
877 
878  ++destIter;
879  if (destIter == dest.end())
880  {
881  return;
882  }
883  curDest = destIter;
884  }
885 
886  // Check for all dest regions that need to be removed completely
887  while (nsStart <= curDest->start && nsEnd >= curDest->end)
888  {
889  destIter = dest.erase(destIter);
890  if (destIter == dest.end())
891  {
892  return;
893  }
894  curDest = destIter;
895  }
896 
897  // Check for truncating the beginning of dest region
898  if (nsStart <= curDest->start &&
899  nsEnd > curDest->start && nsEnd < curDest->end)
900  {
901  curDest->start = nsEnd;
902  }
903  }
904 
905  if (lastRun)
906  {
907  // done
908  lastRun = false;
909  }
910  else
911  {
912  // Next non-silent region starts at the end of this silent region
913  nsStart = curSrc->end;
914  ++srcIter;
915  if (srcIter == src.end())
916  {
917  lastRun = true;
918  }
919  }
920  }
921 }
