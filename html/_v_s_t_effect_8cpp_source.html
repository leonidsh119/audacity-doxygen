452 bool VSTEffectsModule::RegisterPlugin(PluginManagerInterface & pm, const wxString & path)
453 {
454  // TODO: Fix this for external usage
455  wxString cmdpath = PlatformCompatibility::GetExecutablePath();
456 
457  wxString effectIDs = wxT("0;");
458  wxStringTokenizer effectTzr(effectIDs, wxT(";"));
459 
460  Maybe<wxProgressDialog> progress{};
461  size_t idCnt = 0;
462  size_t idNdx = 0;
463 
464  bool valid = false;
465  bool cont = true;
466 
467  while (effectTzr.HasMoreTokens() && cont)
468  {
469  wxString effectID = effectTzr.GetNextToken();
470 
471  wxString cmd;
472  cmd.Printf(wxT("\"%s\" %s \"%s;%s\""), cmdpath.c_str(), VSTCMDKEY, path.c_str(), effectID.c_str());
473 
474  VSTSubProcess proc;
475  try
476  {
477  int flags = wxEXEC_SYNC | wxEXEC_NODISABLE;
478 #if defined(__WXMSW__)
479  flags += wxEXEC_NOHIDE;
480 #endif
481  wxExecute(cmd, flags, &proc);
482  }
483  catch (...)
484  {
485  wxLogMessage(_("VST plugin registration failed for %s\n"), path.c_str());
486  return false;
487  }
488 
489  wxString output;
490  wxStringOutputStream ss(&output);
491  proc.GetInputStream()->Read(ss);
492 
493  int keycount = 0;
494  bool haveBegin = false;
495  wxStringTokenizer tzr(output, wxT("\n"));
496  while (tzr.HasMoreTokens())
497  {
498  wxString line = tzr.GetNextToken();
499 
500  // Our output may follow any output the plugin may have written.
501  if (!line.StartsWith(OUTPUTKEY))
502  {
503  continue;
504  }
505 
506  long key;
507  if (!line.Mid(wxStrlen(OUTPUTKEY)).BeforeFirst(wxT('=')).ToLong(&key))
508  {
509  continue;
510  }
511  wxString val = line.AfterFirst(wxT('=')).BeforeFirst(wxT('\r'));
512 
513  switch (key)
514  {
515  case kKeySubIDs:
516  effectIDs = val;
517  effectTzr.Reinit(effectIDs);
518  idCnt = effectTzr.CountTokens();
519  if (idCnt > 3)
520  {
521  progress.create( _("Scanning Shell VST"),
522  wxString::Format(_("Registering %d of %d: %-64.64s"), 0, idCnt, proc.GetName().c_str()),
523  static_cast<int>(idCnt),
524  nullptr,
525  wxPD_APP_MODAL |
526  wxPD_AUTO_HIDE |
527  wxPD_CAN_ABORT |
528  wxPD_ELAPSED_TIME |
529  wxPD_ESTIMATED_TIME |
530  wxPD_REMAINING_TIME );
531  progress->Show();
532  }
533  break;
534 
535  case kKeyBegin:
536  haveBegin = true;
537  keycount++;
538  break;
539 
540  case kKeyName:
541  proc.mName = val;
542  keycount++;
543  break;
544 
545  case kKeyPath:
546  proc.mPath = val;
547  keycount++;
548  break;
549 
550  case kKeyVendor:
551  proc.mVendor = val;
552  keycount++;
553  break;
554 
555  case kKeyVersion:
556  proc.mVersion = val;
557  keycount++;
558  break;
559 
560  case kKeyDescription:
561  proc.mDescription = val;
562  keycount++;
563  break;
564 
565  case kKeyEffectType:
566  long type;
567  val.ToLong(&type);
568  proc.mType = (EffectType) type;
569  keycount++;
570  break;
571 
572  case kKeyInteractive:
573  proc.mInteractive = val.IsSameAs(wxT("1"));
574  keycount++;
575  break;
576 
577  case kKeyAutomatable:
578  proc.mAutomatable = val.IsSameAs(wxT("1"));
579  keycount++;
580  break;
581 
582  case kKeyEnd:
583  {
584  if (!haveBegin || ++keycount != kKeyEnd)
585  {
586  keycount = 0;
587  haveBegin = false;
588  continue;
589  }
590 
591  bool skip = false;
592  if (progress)
593  {
594  idNdx++;
595  cont = progress->Update(idNdx,
596  wxString::Format(_("Registering %d of %d: %-64.64s"), idNdx, idCnt, proc.GetName().c_str()));
597  }
598 
599  if (!skip && cont)
600  {
601  valid = true;
602  pm.RegisterPlugin(this, &proc);
603  }
604  }
605  break;
606 
607  default:
608  keycount = 0;
609  haveBegin = false;
610  break;
611  }
612  }
613  }
614 
615  return valid;
616 }

1545 //
1546 // Some history...
1547 //
1548 // Before we ran into the Antress plugin problem with buffer size limitations,
1549 // (see below) we just had a plain old effect loop...get the input samples, pass
1550 // them to the effect, save the output samples.
1551 //
1552 // But, the hack I put in to limit the buffer size to only 8k (normally 512k or so)
1553 // severely impacted performance. So, Michael C. added some intermediate buffering
1554 // that sped things up quite a bit and this is how things have worked for quite a
1555 // while. It still didn't get the performance back to the pre-hack stage, but it
1556 // was a definite benefit.
1557 //
1558 // History over...
1559 //
1560 // I've recently (May 2014) tried newer versions of the Antress effects and they
1561 // no longer seem to have a problem with buffer size. So, I've made a bit of a
1562 // compromise...I've made the buffer size user configurable. Should have done this
1563 // from the beginning. I've left the default 8k, just in case, but now the user
1564 // can set the buffering based on their specific setup and needs.
1565 //
1566 // And at the same time I added buffer delay compensation, which allows Audacity
1567 // to account for latency introduced by some effects. This is based on information
1568 // provided by the effect, so it will not work with all effects since they don't
1569 // all provide the information (kn0ck0ut is one).
