215 bool RecordingPrefs::Apply()
216 {
217  ShuttleGui S(this, eIsSavingToPrefs);
218  PopulateOrExchange(S);
219 
220  double latencyDuration = DEFAULT_LATENCY_DURATION;
221  gPrefs->Read(wxT("/AudioIO/LatencyDuration"), &latencyDuration);
222  if (latencyDuration < 0) {
223  gPrefs->Write(wxT("/AudioIO/LatencyDuration"), DEFAULT_LATENCY_DURATION);
224  }
225 
226  #ifdef EXPERIMENTAL_AUTOMATED_INPUT_LEVEL_ADJUSTMENT
227  double targetpeak, deltapeak;
228  gPrefs->Read(wxT("/AudioIO/TargetPeak"), &targetpeak);
229  gPrefs->Read(wxT("/AudioIO/DeltaPeakVolume"), &deltapeak);
230  if (targetpeak + deltapeak > 100.0 || targetpeak - deltapeak < 0.0)
231  {
232  gPrefs->Write(wxT("/AudioIO/DeltaPeakVolume"), min(100.0 - targetpeak, targetpeak));
233  }
234 
235  int value;
236  gPrefs->Read(wxT("/AudioIO/AnalysisTime"), &value);
237  if (value <= 0)
238  gPrefs->Write(wxT("/AudioIO/AnalysisTime"), AILA_DEF_ANALYSIS_TIME);
239 
240  gPrefs->Read(wxT("/AudioIO/NumberAnalysis"), &value);
241  if (value < 0)
242  gPrefs->Write(wxT("/AudioIO/NumberAnalysis"), AILA_DEF_NUMBER_ANALYSIS);
243  #endif
244  return true;
245 }
