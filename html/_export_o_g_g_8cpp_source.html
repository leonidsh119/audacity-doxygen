167 int ExportOGG::Export(AudacityProject *project,
168  int numChannels,
169  const wxString &fName,
170  bool selectionOnly,
171  double t0,
172  double t1,
173  MixerSpec *mixerSpec,
174  const Tags *metadata,
175  int WXUNUSED(subformat))
176 {
177  double rate = project->GetRate();
178  const TrackList *tracks = project->GetTracks();
179  double quality = (gPrefs->Read(wxT("/FileFormats/OggExportQuality"), 50)/(float)100.0);
180 
181  wxLogNull logNo; // temporarily disable wxWidgets error messages
182  int updateResult = eProgressSuccess;
183  int eos = 0;
184 
185  FileIO outFile(fName, FileIO::Output);
186 
187  if (!outFile.IsOpened()) {
188  wxMessageBox(_("Unable to open target file for writing"));
189  return false;
190  }
191 
192  // All the Ogg and Vorbis encoding data
193  ogg_stream_state stream;
194  ogg_page page;
195  ogg_packet packet;
196 
197  vorbis_info info;
198  vorbis_comment comment;
199  vorbis_dsp_state dsp;
200  vorbis_block block;
201 
202  // Encoding setup
203  vorbis_info_init(&info);
204  vorbis_encode_init_vbr(&info, numChannels, int(rate + 0.5), quality);
205 
206  // Retrieve tags
207  if (!FillComment(project, &comment, metadata)) {
208  return false;
209  }
210 
211  // Set up analysis state and auxiliary encoding storage
212  vorbis_analysis_init(&dsp, &info);
213  vorbis_block_init(&dsp, &block);
214 
215  // Set up packet->stream encoder. According to encoder example,
216  // a random serial number makes it more likely that you can make
217  // chained streams with concatenation.
218  srand(time(NULL));
219  ogg_stream_init(&stream, rand());
220 
221  // First we need to write the required headers:
222  // 1. The Ogg bitstream header, which contains codec setup params
223  // 2. The Vorbis comment header
224  // 3. The bitstream codebook.
225  //
226  // After we create those our responsibility is complete, libvorbis will
227  // take care of any other ogg bistream constraints (again, according
228  // to the example encoder source)
229  ogg_packet bitstream_header;
230  ogg_packet comment_header;
231  ogg_packet codebook_header;
232 
233  vorbis_analysis_headerout(&dsp, &comment, &bitstream_header, &comment_header,
234  &codebook_header);
235 
236  // Place these headers into the stream
237  ogg_stream_packetin(&stream, &bitstream_header);
238  ogg_stream_packetin(&stream, &comment_header);
239  ogg_stream_packetin(&stream, &codebook_header);
240 
241  // Flushing these headers now guarentees that audio data will
242  // start on a NEW page, which apparently makes streaming easier
243  while (ogg_stream_flush(&stream, &page)) {
244  outFile.Write(page.header, page.header_len);
245  outFile.Write(page.body, page.body_len);
246  }
247 
248  const WaveTrackConstArray waveTracks =
249  tracks->GetWaveTrackConstArray(selectionOnly, false);
250  Mixer *mixer = CreateMixer(waveTracks,
251  tracks->GetTimeTrack(),
252  t0, t1,
253  numChannels, SAMPLES_PER_RUN, false,
254  rate, floatSample, true, mixerSpec);
255 
256  {
257  ProgressDialog progress(wxFileName(fName).GetName(),
258  selectionOnly ?
259  _("Exporting the selected audio as Ogg Vorbis") :
260  _("Exporting the entire project as Ogg Vorbis"));
261 
262  while (updateResult == eProgressSuccess && !eos) {
263  float **vorbis_buffer = vorbis_analysis_buffer(&dsp, SAMPLES_PER_RUN);
264  sampleCount samplesThisRun = mixer->Process(SAMPLES_PER_RUN);
265 
266  if (samplesThisRun == 0) {
267  // Tell the library that we wrote 0 bytes - signalling the end.
268  vorbis_analysis_wrote(&dsp, 0);
269  }
270  else {
271 
272  for (int i = 0; i < numChannels; i++) {
273  float *temp = (float *)mixer->GetBuffer(i);
274  memcpy(vorbis_buffer[i], temp, sizeof(float)*SAMPLES_PER_RUN);
275  }
276 
277  // tell the encoder how many samples we have
278  vorbis_analysis_wrote(&dsp, samplesThisRun);
279  }
280 
281  // I don't understand what this call does, so here is the comment
282  // from the example, verbatim:
283  //
284  // vorbis does some data preanalysis, then divvies up blocks
285  // for more involved (potentially parallel) processing. Get
286  // a single block for encoding now
287  while (vorbis_analysis_blockout(&dsp, &block) == 1) {
288 
289  // analysis, assume we want to use bitrate management
290  vorbis_analysis(&block, NULL);
291  vorbis_bitrate_addblock(&block);
292 
293  while (vorbis_bitrate_flushpacket(&dsp, &packet)) {
294 
295  // add the packet to the bitstream
296  ogg_stream_packetin(&stream, &packet);
297 
298  // From vorbis-tools-1.0/oggenc/encode.c:
299  // If we've gone over a page boundary, we can do actual output,
300  // so do so (for however many pages are available).
301 
302  while (!eos) {
303  int result = ogg_stream_pageout(&stream, &page);
304  if (!result) {
305  break;
306  }
307 
308  outFile.Write(page.header, page.header_len);
309  outFile.Write(page.body, page.body_len);
310 
311  if (ogg_page_eos(&page)) {
312  eos = 1;
313  }
314  }
315  }
316  }
317 
318  updateResult = progress.Update(mixer->MixGetCurrentTime() - t0, t1 - t0);
319  }
320  }
321 
322  delete mixer;
323 
324  ogg_stream_clear(&stream);
325 
326  vorbis_block_clear(&block);
327  vorbis_dsp_clear(&dsp);
328  vorbis_info_clear(&info);
329  vorbis_comment_clear(&comment);
330 
331  outFile.Close();
332 
333  return updateResult;
334 }

342 bool ExportOGG::FillComment(AudacityProject *project, vorbis_comment *comment, const Tags *metadata)
343 {
344  // Retrieve tags from project if not over-ridden
345  if (metadata == NULL)
346  metadata = project->GetTags();
347 
348  vorbis_comment_init(comment);
349 
350  wxString n;
351  for (const auto &pair : metadata->GetRange()) {
352  n = pair.first;
353  const auto &v = pair.second;
354  if (n == TAG_YEAR) {
355  n = wxT("DATE");
356  }
357  vorbis_comment_add_tag(comment,
358  (char *) (const char *) n.mb_str(wxConvUTF8),
359  (char *) (const char *) v.mb_str(wxConvUTF8));
360  }
361 
362  return true;
363 }
