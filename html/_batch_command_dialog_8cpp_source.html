166 void BatchCommandDialog::OnItemSelected(wxListEvent &event)
167 {
168  wxString command = mChoices->GetItemText(event.GetIndex());
169 
170  EffectManager & em = EffectManager::Get();
171  PluginID ID = em.GetEffectByIdentifier(command);
172 
173  // If ID is empty, then the effect wasn't found, in which case, the user must have
174  // selected one of the "special" commands.
175  mEditParams->Enable(!ID.IsEmpty());
176  mUsePreset->Enable(em.HasPresets(ID));
177 
178  if (command == mCommand->GetValue())
179  {
180  return;
181  }
182 
183  mCommand->SetValue(command);
184 
185  wxString params = BatchCommands::GetCurrentParamsFor(command);
186  if (params.IsEmpty())
187  {
188  params = em.GetDefaultPreset(ID);
189  }
190 
191  mParameters->SetValue(params);
192 }
