98 {
99  int i, j, length = num_coefs4 * num_phases;
100  sample_t * result = malloc((size_t)(length * (interp_order + 1)) * sizeof(*result));
101  double fm1 = coefs[0], f1 = 0, f2 = 0;
102 
103  for (i = num_coefs4 - 1; i >= 0; --i)
104  for (j = num_phases - 1; j >= 0; --j) {
105  double f0 = fm1, b = 0, c = 0, d = 0; /* = 0 to kill compiler warning */
106  int pos = i * num_phases + j - 1;
107  fm1 = coefs4_check(i) && pos > 0 ? coefs[pos - 1] * multiplier : 0;
108  switch (interp_order) {
109  case 1: b = f1 - f0; break;
110  case 2: b = f1 - (.5 * (f2+f0) - f1) - f0; c = .5 * (f2+f0) - f1; break;
111  case 3: c=.5*(f1+fm1)-f0;d=(1/6.)*(f2-f1+fm1-f0-4*c);b=f1-f0-d-c; break;
112  default: if (interp_order) assert(0);
113  }

115  coef(result, interp_order, num_coefs4, j, x, num_coefs4 - 1 - i)
116  coef_coef(0) = (sample_t)f0;
117  if (interp_order > 0) coef_coef(1) = (sample_t)b;
118  if (interp_order > 1) coef_coef(2) = (sample_t)c;
119  if (interp_order > 2) coef_coef(3) = (sample_t)d;

218  sample_t const * input = fifo_read_ptr(&p->fifo);

220  num_in -= divd.quot;
221

223 
224  if (lsx_is_power_of_2(p->L)) { /* F-domain */
225  int portion = f->dft_length / p->L;
226  memcpy(dft_out, input, (unsigned)portion * sizeof(*dft_out));
227  rdft_oforward(portion, f->dft_forward_setup, dft_out, p->dft_scratch);
228  for (i = portion + 2; i < (portion << 1); i += 2) /* Mirror image. */
229  dft_out[i] = dft_out[(portion << 1) - i],
230  dft_out[i+1] = -dft_out[(portion << 1) - i + 1];
231  dft_out[portion] = dft_out[1];
232  dft_out[portion + 1] = 0;
233  dft_out[1] = dft_out[0];
234 
235  for (portion <<= 1; i < f->dft_length; i += portion, portion <<= 1) {
236  memcpy(dft_out + i, dft_out, (size_t)portion * sizeof(*dft_out));
237  dft_out[i + 1] = 0;
238  }
239  if (p->step.integer > 0)
240  rdft_reorder_back(f->dft_length, f->dft_backward_setup, dft_out, p->dft_scratch);
241  } else {
242  if (p->L == 1)
243  memcpy(dft_out, input, (size_t)f->dft_length * sizeof(*dft_out));
244  else {
245  memset(dft_out, 0, (size_t)f->dft_length * sizeof(*dft_out));
246  for (j = 0, i = p->at.integer; i < f->dft_length; ++j, i += p->L)
247  dft_out[i] = input[j];
248  p->at.integer = p->L - 1 - divd.rem;
249  }
250  if (p->step.integer > 0)
251  rdft_forward(f->dft_length, f->dft_forward_setup, dft_out, p->dft_scratch);
252  else
253  rdft_oforward(f->dft_length, f->dft_forward_setup, dft_out, p->dft_scratch);
254  }
255 
256  if (p->step.integer > 0) {
257  rdft_convolve(f->dft_length, f->dft_backward_setup, dft_out, f->coefs);
258  rdft_backward(f->dft_length, f->dft_backward_setup, dft_out, p->dft_scratch);
259 #if RATE_SIMD
260  if (p->step.integer == 1)
261  memcpy(output, dft_out, (size_t)f->dft_length * sizeof(sample_t));
262 #endif
263  if (p->step.integer != 1) {
264  for (j = 0, i = p->remM; i < f->dft_length - overlap; ++j,
265  i += p->step.integer)
266  output[j] = dft_out[i];
267  p->remM = i - (f->dft_length - overlap);
268  fifo_trim_by(output_fifo, f->dft_length - j);
269  }
270  else fifo_trim_by(output_fifo, overlap);
271  }
272  else { /* F-domain */
273  int m = -p->step.integer;
274  rdft_convolve_portion(f->dft_length >> m, dft_out, f->coefs);
275  rdft_obackward(f->dft_length >> m, f->dft_backward_setup, dft_out, p->dft_scratch);
276 #if RATE_SIMD
277  memcpy(output, dft_out, (size_t)(f->dft_length >> m) * sizeof(sample_t));
278 #endif
279  fifo_trim_by(output_fifo, (((1 << m) - 1) * f->dft_length + overlap) >>m);
280  }
281  }
