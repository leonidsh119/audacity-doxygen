167 void NumberFormatter::AddThousandsSeparators(wxString& s)
168 {
169  wxChar thousandsSep;
170  if ( !GetThousandsSeparatorIfUsed(&thousandsSep) )
171  return;
172 
173  size_t pos = s.find(GetDecimalSeparator());
174  if ( pos == wxString::npos )
175  {
176  // Start grouping at the end of an integer number.
177  pos = s.length();
178  }
179 
180  // End grouping at the beginning of the digits -- there could be at a sign
181  // before their start.
182  const size_t start = s.find_first_of(wxT("0123456789"));
183 
184  // We currently group digits by 3 independently of the locale. This is not
185  // the right thing to do and we should use lconv::grouping (under POSIX)
186  // and GetLocaleInfo(LOCALE_SGROUPING) (under MSW) to get information about
187  // the correct grouping to use. This is something that needs to be done at
188  // wxLocale level first and then used here in the future (TODO).
189  const size_t GROUP_LEN = 3;
190 
191  while ( pos > start + GROUP_LEN )
192  {
193  pos -= GROUP_LEN;
194  s.insert(pos, thousandsSep);
195  }
196 }

198 void NumberFormatter::RemoveTrailingZeroes(wxString& s, size_t retain /* = 0 */)
199 {
200  const size_t posDecSep = s.find(GetDecimalSeparator());
201  wxCHECK_RET( posDecSep != wxString::npos,
202  wxString::Format(wxT("No decimal separator in \"%s\""), s.c_str()) );
203  wxCHECK_RET( posDecSep, wxT("Can't start with decimal separator" ));
204 
205  // Find the last character to keep.
206  size_t posLastCharacterToKeep = s.find_last_not_of(wxT("0"));
207 
208  // If it's the decimal separator itself, remove it.
209  if ((posLastCharacterToKeep == posDecSep) && (retain == 0)) {
210  posLastCharacterToKeep--;
211  } else if ((posLastCharacterToKeep - posDecSep) < retain) {
212  posLastCharacterToKeep = retain + posDecSep;
213  }
214 
215  s.erase(posLastCharacterToKeep + 1);
216 }
