339  data.freqofs = mFreqOfs / 100.0;

344 sampleCount EffectWahwah::InstanceProcess(EffectWahwahState & data, float **inBlock, float **outBlock, sampleCount blockLen)
345 {
346  float *ibuf = inBlock[0];
347  float *obuf = outBlock[0];
348  double frequency, omega, sn, cs, alpha;
349  double in, out;
350 
351  data.lfoskip = mFreq * 2 * M_PI / data.samplerate;
352  data.depth = mDepth / 100.0;
353  data.freqofs = mFreqOfs / 100.0;
354 
355  data.phase = mPhase * M_PI / 180.0;
356  data.outgain = DB_TO_LINEAR(mOutGain);
357 
358  for (int i = 0; i < blockLen; i++)
359  {
360  in = (double) ibuf[i];
361 
362  if ((data.skipcount++) % lfoskipsamples == 0)
363  {
364  frequency = (1 + cos(data.skipcount * data.lfoskip + data.phase)) / 2;
365  frequency = frequency * data.depth * (1 - data.freqofs) + data.freqofs;
366  frequency = exp((frequency - 1) * 6);
367  omega = M_PI * frequency;
368  sn = sin(omega);
369  cs = cos(omega);
370  alpha = sn / (2 * mRes);
371  data.b0 = (1 - cs) / 2;
372  data.b1 = 1 - cs;
373  data.b2 = (1 - cs) / 2;
374  data.a0 = 1 + alpha;
375  data.a1 = -2 * cs;
376  data.a2 = 1 - alpha;
377  };
378  out = (data.b0 * in + data.b1 * data.xn1 + data.b2 * data.xn2 - data.a1 * data.yn1 - data.a2 * data.yn2) / data.a0;
379  data.xn2 = data.xn1;
380  data.xn1 = in;
381  data.yn2 = data.yn1;
382  data.yn1 = out;
383  out *= data.outgain;
384 
385  obuf[i] = (float) out;
386  }
387 
388  return blockLen;
389 }
