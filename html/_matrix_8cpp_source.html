337 {
338  // Very straightforward implementation of
339  // Gauss-Jordan elimination to invert a matrix.
340  // Returns true if successful
341 
342  wxASSERT(input.Rows() == input.Cols());
343  int N = input.Rows();
344  int i, j, k;
345 
346  Matrix M = input;
347  Minv = IdentityMatrix(N);
348 
349  // Do the elimination one column at a time
350  for(i=0; i<N; i++) {
351  // Pivot the row with the largest absolute value in
352  // column i, into row i
353  double absmax = 0.0;
354  int argmax=0;
355 
356  for(j=i; j<N; j++)
357  if (fabs(M[j][i]) > absmax) {
358  absmax = fabs(M[j][i]);
359  argmax = j;
360  }
361 
362  // If no row has a nonzero value in that column,
363  // the matrix is singular and we have to give up.
364  if (absmax == 0)
365  return false;
366 
367  if (i != argmax) {
368  M.SwapRows(i, argmax);
369  Minv.SwapRows(i, argmax);
370  }
371 
372  // Divide this row by the value of M[i][i]
373  double factor = 1.0 / M[i][i];
374  M[i] = M[i] * factor;
375  Minv[i] = Minv[i] * factor;
376 
377  // Eliminate the rest of the column
378  for(j=0; j<N; j++) {
379  if (j==i)
380  continue;
381  if (fabs(M[j][i]) > 0) {
382  // Subtract a multiple of row i from row j
383  double factor = M[j][i];

385  M[j][k] -= (M[i][k] * factor);
