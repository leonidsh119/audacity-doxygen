259 bool ExportFFmpeg::Init(const char *shortname, AudacityProject *project, const Tags *metadata, int subformat)
260 {
261  int err;
262  //FFmpegLibsInst->LoadLibs(NULL,true); //Loaded at startup or from Prefs now
263 
264  if (!FFmpegLibsInst->ValidLibsLoaded()) return false;
265 
266  av_log_set_callback(av_log_wx_callback);
267 
268  // See if libavformat has modules that can write our output format. If so, mEncFormatDesc
269  // will describe the functions used to write the format (used internally by libavformat)
270  // and the default video/audio codecs that the format uses.
271  if ((mEncFormatDesc = av_guess_format(shortname, OSINPUT(mName), NULL)) == NULL)
272  {
273  wxMessageBox(wxString::Format(_("FFmpeg : ERROR - Can't determine format description for file \"%s\"."), mName.c_str()),
274  _("FFmpeg Error"), wxOK|wxCENTER|wxICON_EXCLAMATION);
275  return false;
276  }
277 
278  // mEncFormatCtx is used by libavformat to carry around context data re our output file.
279  if ((mEncFormatCtx = avformat_alloc_context()) == NULL)
280  {
281  wxMessageBox(wxString::Format(_("FFmpeg : ERROR - Can't allocate output format context.")),
282  _("FFmpeg Error"), wxOK|wxCENTER|wxICON_EXCLAMATION);
283  return false;
284  }
285 
286  // Initialise the output format context.
287  mEncFormatCtx->oformat = mEncFormatDesc;
288  memcpy(mEncFormatCtx->filename, OSINPUT(mName), strlen(OSINPUT(mName))+1);
289 
290  // At the moment Audacity can export only one audio stream
291  if ((mEncAudioStream = avformat_new_stream(mEncFormatCtx, NULL)) == NULL)
292  {
293  wxMessageBox(wxString::Format(_("FFmpeg : ERROR - Can't add audio stream to output file \"%s\"."), mName.c_str()),
294  _("FFmpeg Error"), wxOK|wxCENTER|wxICON_EXCLAMATION);
295  return false;
296  }
297 
298  mEncAudioStream->id = 0;
299 
300  // Open the output file.
301  if (!(mEncFormatDesc->flags & AVFMT_NOFILE))
302  {
303  if ((err = ufile_fopen(&mEncFormatCtx->pb, mName, AVIO_FLAG_WRITE)) < 0)
304  {
305  wxMessageBox(wxString::Format(wxT("FFmpeg : ERROR - Can't open output file \"%s\" to write. Error code is %d."), mName.c_str(), err),
306  _("FFmpeg Error"), wxOK|wxCENTER|wxICON_EXCLAMATION);
307  return false;
308  }
309  }
310 
311  // Open the audio stream's codec and initialise any stream related data.
312  if (!InitCodecs(project))
313  return false;
314 
315  if (metadata == NULL)
316  metadata = project->GetTags();
317 
318  // Add metadata BEFORE writing the header.
319  // At the moment that works with ffmpeg-git and ffmpeg-0.5 for MP4.
320  if (GetCanMetaData(subformat))
321  {
322  mSupportsUTF8 = ExportFFmpegOptions::fmts[mSubFormat].canutf8;
323  AddTags(metadata);
324  }
325 
326  // Write headers to the output file.
327  if ((err = avformat_write_header(mEncFormatCtx, NULL)) < 0)
328  {
329  wxMessageBox(wxString::Format(_("FFmpeg : ERROR - Can't write headers to output file \"%s\". Error code is %d."), mName.c_str(),err),
330  _("FFmpeg Error"), wxOK|wxCENTER|wxICON_EXCLAMATION);
331  return false;
332  }
333 
334  return true;
335 }

352 bool ExportFFmpeg::InitCodecs(AudacityProject *project)
353 {
354  AVCodec *codec = NULL;
355  AVDictionary *options = NULL;
356 
357  // Configure the audio stream's codec context.
358  mEncAudioCodecCtx = mEncAudioStream->codec;
359 
360  mEncAudioCodecCtx->codec_id = ExportFFmpegOptions::fmts[mSubFormat].codecid;
361  mEncAudioCodecCtx->codec_type = AVMEDIA_TYPE_AUDIO;
362  mEncAudioCodecCtx->codec_tag = av_codec_get_tag((const AVCodecTag **)mEncFormatCtx->oformat->codec_tag,mEncAudioCodecCtx->codec_id);
363  mSampleRate = (int)project->GetRate();
364  mEncAudioCodecCtx->global_quality = -99999; //quality mode is off by default;
365 
366  // Each export type has its own settings
367  switch (mSubFormat)
368  {
369  case FMT_M4A:
370  mEncAudioCodecCtx->bit_rate = 98000;
371  mEncAudioCodecCtx->bit_rate *= mChannels;
372  mEncAudioCodecCtx->profile = FF_PROFILE_AAC_LOW;
373  mEncAudioCodecCtx->cutoff = 0;
374  mEncAudioCodecCtx->global_quality = gPrefs->Read(wxT("/FileFormats/AACQuality"),-99999);
375  if (!CheckSampleRate(mSampleRate,
376  ExportFFmpegOptions::iAACSampleRates[0],
377  ExportFFmpegOptions::iAACSampleRates[11],
378  &ExportFFmpegOptions::iAACSampleRates[0]))
379  {
380  mSampleRate = AskResample(mEncAudioCodecCtx->bit_rate,mSampleRate,
381  ExportFFmpegOptions::iAACSampleRates[0],
382  ExportFFmpegOptions::iAACSampleRates[11],
383  &ExportFFmpegOptions::iAACSampleRates[0]);
384  }
385  break;
386  case FMT_AC3:
387  mEncAudioCodecCtx->bit_rate = gPrefs->Read(wxT("/FileFormats/AC3BitRate"), 192000);
388  if (!CheckSampleRate(mSampleRate,ExportFFmpegAC3Options::iAC3SampleRates[0], ExportFFmpegAC3Options::iAC3SampleRates[2], &ExportFFmpegAC3Options::iAC3SampleRates[0]))
389  mSampleRate = AskResample(mEncAudioCodecCtx->bit_rate,mSampleRate, ExportFFmpegAC3Options::iAC3SampleRates[0], ExportFFmpegAC3Options::iAC3SampleRates[2], &ExportFFmpegAC3Options::iAC3SampleRates[0]);
390  break;
391  case FMT_AMRNB:
392  mSampleRate = 8000;
393  mEncAudioCodecCtx->bit_rate = gPrefs->Read(wxT("/FileFormats/AMRNBBitRate"), 12200);
394  break;
395  case FMT_WMA2:
396  mEncAudioCodecCtx->bit_rate = gPrefs->Read(wxT("/FileFormats/WMABitRate"), 198000);
397  if (!CheckSampleRate(mSampleRate,ExportFFmpegWMAOptions::iWMASampleRates[0], ExportFFmpegWMAOptions::iWMASampleRates[4], &ExportFFmpegWMAOptions::iWMASampleRates[0]))
398  mSampleRate = AskResample(mEncAudioCodecCtx->bit_rate,mSampleRate, ExportFFmpegWMAOptions::iWMASampleRates[0], ExportFFmpegWMAOptions::iWMASampleRates[4], &ExportFFmpegWMAOptions::iWMASampleRates[0]);
399  break;
400  case FMT_OTHER:
401  av_dict_set(&mEncAudioStream->metadata, "language", gPrefs->Read(wxT("/FileFormats/FFmpegLanguage"),wxT("")).ToUTF8(), 0);
402  mEncAudioCodecCtx->sample_rate = gPrefs->Read(wxT("/FileFormats/FFmpegSampleRate"),(long)0);
403  if (mEncAudioCodecCtx->sample_rate != 0) mSampleRate = mEncAudioCodecCtx->sample_rate;
404  mEncAudioCodecCtx->bit_rate = gPrefs->Read(wxT("/FileFormats/FFmpegBitRate"), (long)0);
405  strncpy((char *)&mEncAudioCodecCtx->codec_tag,gPrefs->Read(wxT("/FileFormats/FFmpegTag"),wxT("")).mb_str(wxConvUTF8),4);
406  mEncAudioCodecCtx->global_quality = gPrefs->Read(wxT("/FileFormats/FFmpegQuality"),(long)-99999);
407  mEncAudioCodecCtx->cutoff = gPrefs->Read(wxT("/FileFormats/FFmpegCutOff"),(long)0);
408  mEncAudioCodecCtx->flags2 = 0;
409  if (gPrefs->Read(wxT("/FileFormats/FFmpegBitReservoir"),true))
410  av_dict_set(&options, "reservoir", "1", 0);
411  if (gPrefs->Read(wxT("/FileFormats/FFmpegVariableBlockLen"),true)) mEncAudioCodecCtx->flags2 |= 0x0004; //WMA only?
412  mEncAudioCodecCtx->compression_level = gPrefs->Read(wxT("/FileFormats/FFmpegCompLevel"),-1);
413  mEncAudioCodecCtx->frame_size = gPrefs->Read(wxT("/FileFormats/FFmpegFrameSize"),(long)0);
414 
415 //FIXME The list of supported options for the seleced encoder should be extracted instead of a few hardcoded
416  set_dict_int(&options, "lpc_coeff_precision", gPrefs->Read(wxT("/FileFormats/FFmpegLPCCoefPrec"),(long)0));
417  set_dict_int(&options, "min_prediction_order", gPrefs->Read(wxT("/FileFormats/FFmpegMinPredOrder"),(long)-1));
418  set_dict_int(&options, "max_prediction_order", gPrefs->Read(wxT("/FileFormats/FFmpegMaxPredOrder"),(long)-1));
419  set_dict_int(&options, "min_partition_order", gPrefs->Read(wxT("/FileFormats/FFmpegMinPartOrder"),(long)-1));
420  set_dict_int(&options, "max_partition_order", gPrefs->Read(wxT("/FileFormats/FFmpegMaxPartOrder"),(long)-1));
421  set_dict_int(&options, "prediction_order_method", gPrefs->Read(wxT("/FileFormats/FFmpegPredOrderMethod"),(long)0));
422  set_dict_int(&options, "muxrate", gPrefs->Read(wxT("/FileFormats/FFmpegMuxRate"),(long)0));
423  mEncFormatCtx->packet_size = gPrefs->Read(wxT("/FileFormats/FFmpegPacketSize"),(long)0);
424  codec = avcodec_find_encoder_by_name(gPrefs->Read(wxT("/FileFormats/FFmpegCodec")).ToUTF8());
425  if (!codec)
426  mEncAudioCodecCtx->codec_id = mEncFormatDesc->audio_codec;
427  break;
428  default:
429  return false;
430  }
431 
432  // This happens if user refused to resample the project
433  if (mSampleRate == 0) return false;
434 
435  if (mEncAudioCodecCtx->global_quality >= 0)
436  {
437  mEncAudioCodecCtx->flags |= CODEC_FLAG_QSCALE;
438  }
439  else mEncAudioCodecCtx->global_quality = 0;
440  mEncAudioCodecCtx->global_quality = mEncAudioCodecCtx->global_quality * FF_QP2LAMBDA;
441  mEncAudioCodecCtx->sample_rate = mSampleRate;
442  mEncAudioCodecCtx->channels = mChannels;
443  mEncAudioCodecCtx->time_base.num = 1;
444  mEncAudioCodecCtx->time_base.den = mEncAudioCodecCtx->sample_rate;
445  mEncAudioCodecCtx->sample_fmt = AV_SAMPLE_FMT_S16;
446  mEncAudioCodecCtx->strict_std_compliance = FF_COMPLIANCE_EXPERIMENTAL;
447 
448  if (mEncAudioCodecCtx->codec_id == AV_CODEC_ID_AC3)
449  {
450  // As of Jan 4, 2011, the default AC3 encoder only accept SAMPLE_FMT_FLT samples.
451  // But, currently, Audacity only supports SAMPLE_FMT_S16. So, for now, look for the
452  // "older" AC3 codec. this is not a proper solution, but will suffice until other
453  // encoders no longer support SAMPLE_FMT_S16.
454  codec = avcodec_find_encoder_by_name("ac3_fixed");
455  }
456 
457  if (!codec)
458  {
459  codec = avcodec_find_encoder(mEncAudioCodecCtx->codec_id);
460  }
461 
462  // Is the required audio codec compiled into libavcodec?
463  if (codec == NULL)
464  {
465  wxMessageBox(wxString::Format(_("FFmpeg cannot find audio codec 0x%x.\nSupport for this codec is probably not compiled in."), (unsigned int) mEncAudioCodecCtx->codec_id),
466  _("FFmpeg Error"), wxOK|wxCENTER|wxICON_EXCLAMATION);
467  return false;
468  }
469 
470  if (codec->sample_fmts) {
471  for (int i=0; codec->sample_fmts[i] != AV_SAMPLE_FMT_NONE; i++) {
472  enum AVSampleFormat fmt = codec->sample_fmts[i];
473  if ( fmt == AV_SAMPLE_FMT_U8
474  || fmt == AV_SAMPLE_FMT_U8P
475  || fmt == AV_SAMPLE_FMT_S16
476  || fmt == AV_SAMPLE_FMT_S16P
477  || fmt == AV_SAMPLE_FMT_S32
478  || fmt == AV_SAMPLE_FMT_S32P
479  || fmt == AV_SAMPLE_FMT_FLT
480  || fmt == AV_SAMPLE_FMT_FLTP) {
481  mEncAudioCodecCtx->sample_fmt = fmt;
482  }
483  if ( fmt == AV_SAMPLE_FMT_S16
484  || fmt == AV_SAMPLE_FMT_S16P)
485  break;
486  }
487  }
488 
489  if (mEncFormatCtx->oformat->flags & AVFMT_GLOBALHEADER)
490  {
491  mEncAudioCodecCtx->flags |= CODEC_FLAG_GLOBAL_HEADER;
492  mEncFormatCtx->flags |= CODEC_FLAG_GLOBAL_HEADER;
493  }
494 
495  // Open the codec.
496  if (avcodec_open2(mEncAudioCodecCtx, codec, &options) < 0)
497  {
498  wxMessageBox(wxString::Format(_("FFmpeg : ERROR - Can't open audio codec 0x%x."),mEncAudioCodecCtx->codec_id),
499  _("FFmpeg Error"), wxOK|wxCENTER|wxICON_EXCLAMATION);
500  return false;
501  }
502 
503  default_frame_size = mEncAudioCodecCtx->frame_size;
504  if (default_frame_size == 0)
505  default_frame_size = 1024; // arbitrary non zero value;
506 
507  wxLogDebug(wxT("FFmpeg : Audio Output Codec Frame Size: %d samples."), mEncAudioCodecCtx->frame_size);
508 
509  // The encoder may require a minimum number of raw audio samples for each encoding but we can't
510  // guarantee we'll get this minimum each time an audio frame is decoded from the input file so
511  // we use a FIFO to store up incoming raw samples until we have enough for one call to the codec.
512  mEncAudioFifo = av_fifo_alloc(1024);
513 
514  mEncAudioFifoOutBufSiz = 2*MAX_AUDIO_PACKET_SIZE;
515  // Allocate a buffer to read OUT of the FIFO into. The FIFO maintains its own buffer internally.
516  if ((mEncAudioFifoOutBuf = (uint8_t*)av_malloc(mEncAudioFifoOutBufSiz)) == NULL)
517  {
518  wxMessageBox(wxString::Format(_("FFmpeg : ERROR - Can't allocate buffer to read into from audio FIFO.")),
519  _("FFmpeg Error"), wxOK|wxCENTER|wxICON_EXCLAMATION);
520  return false;
521  }
522 
523  return true;
524 }

526 static int encode_audio(AVCodecContext *avctx, AVPacket *pkt, int16_t *audio_samples, int nb_samples)
527 {
528  int i, ch, buffer_size, ret, got_output = 0;
529  void *samples = NULL;
530  AVFrame *frame = NULL;
531 
532  if (audio_samples) {
533  frame = av_frame_alloc();
534  if (!frame)
535  return AVERROR(ENOMEM);
536 
537  frame->nb_samples = nb_samples;
538  frame->format = avctx->sample_fmt;
539 #if !defined(DISABLE_DYNAMIC_LOADING_FFMPEG) || (LIBAVCODEC_VERSION_INT >= AV_VERSION_INT(54, 13, 0))
540  frame->channel_layout = avctx->channel_layout;
541 #endif
542 
543  buffer_size = av_samples_get_buffer_size(NULL, avctx->channels, frame->nb_samples,
544  avctx->sample_fmt, 0);
545  if (buffer_size < 0) {
546  wxMessageBox(wxString::Format(_("FFmpeg : ERROR - Could not get sample buffer size")),
547  _("FFmpeg Error"), wxOK|wxCENTER|wxICON_EXCLAMATION);
548  return buffer_size;
549  }
550  samples = av_malloc(buffer_size);
551  if (!samples) {
552  wxMessageBox(wxString::Format(_("FFmpeg : ERROR - Could not allocate bytes for samples buffer")),
553  _("FFmpeg Error"), wxOK|wxCENTER|wxICON_EXCLAMATION);
554  return AVERROR(ENOMEM);
555  }
556  /* setup the data pointers in the AVFrame */
557  ret = avcodec_fill_audio_frame(frame, avctx->channels, avctx->sample_fmt,
558  (const uint8_t*)samples, buffer_size, 0);
559  if (ret < 0) {
560  wxMessageBox(wxString::Format(_("FFmpeg : ERROR - Could not setup audio frame")),
561  _("FFmpeg Error"), wxOK|wxCENTER|wxICON_EXCLAMATION);
562  return ret;
563  }

565  for (ch = 0; ch < avctx->channels; ch++) {
566  for (i = 0; i < frame->nb_samples; i++) {
567  switch(avctx->sample_fmt) {
568  case AV_SAMPLE_FMT_U8:
569  ((uint8_t*)(frame->data[0]))[ch + i*avctx->channels] = audio_samples[ch + i*avctx->channels]/258 + 128;
570  break;
571  case AV_SAMPLE_FMT_U8P:
572  ((uint8_t*)(frame->data[ch]))[i] = audio_samples[ch + i*avctx->channels]/258 + 128;
573  break;
574  case AV_SAMPLE_FMT_S16:
575  ((int16_t*)(frame->data[0]))[ch + i*avctx->channels] = audio_samples[ch + i*avctx->channels];
576  break;
577  case AV_SAMPLE_FMT_S16P:
578  ((int16_t*)(frame->data[ch]))[i] = audio_samples[ch + i*avctx->channels];
579  break;
580  case AV_SAMPLE_FMT_S32:
581  ((int32_t*)(frame->data[0]))[ch + i*avctx->channels] = audio_samples[ch + i*avctx->channels]<<16;
582  break;
583  case AV_SAMPLE_FMT_S32P:
584  ((int32_t*)(frame->data[ch]))[i] = audio_samples[ch + i*avctx->channels]<<16;
585  break;
586  case AV_SAMPLE_FMT_FLT:
587  ((float*)(frame->data[0]))[ch + i*avctx->channels] = audio_samples[ch + i*avctx->channels] / 32767.0;
588  break;
589  case AV_SAMPLE_FMT_FLTP:
590  ((float*)(frame->data[ch]))[i] = audio_samples[ch + i*avctx->channels] / 32767.;
591  break;
592  case AV_SAMPLE_FMT_NONE:
593  case AV_SAMPLE_FMT_DBL:
594  case AV_SAMPLE_FMT_DBLP:
595  case AV_SAMPLE_FMT_NB:
596  wxASSERT(false);
597  break;
598  }
599  }
600  }
601  }
602  av_init_packet(pkt);
603  pkt->data = NULL; // packet data will be allocated by the encoder
604  pkt->size = 0;
605 
606  ret = avcodec_encode_audio2(avctx, pkt, frame, &got_output);
607  if (ret < 0) {
608  wxMessageBox(wxString::Format(_("FFmpeg : ERROR - encoding frame failed")),
609  _("FFmpeg Error"), wxOK|wxCENTER|wxICON_EXCLAMATION);
610  return ret;
611  }
612 
613  pkt->dts = pkt->pts = AV_NOPTS_VALUE; // we dont set frame.pts thus dont trust the AVPacket ts
614 
615  av_frame_free(&frame);
616  av_freep(&samples);
617 
618  return got_output;
619 }

622 bool ExportFFmpeg::Finalize()
623 {
624  int i, nEncodedBytes;
625 
626  // Flush the audio FIFO and encoder.
627  for (;;)
628  {
629  AVPacket pkt;
630  int nFifoBytes = av_fifo_size(mEncAudioFifo); // any bytes left in audio FIFO?
631 
632  av_init_packet(&pkt);
633 
634  nEncodedBytes = 0;
635  int nAudioFrameSizeOut = default_frame_size * mEncAudioCodecCtx->channels * sizeof(int16_t);
636 
637  if (nAudioFrameSizeOut > mEncAudioFifoOutBufSiz || nFifoBytes > mEncAudioFifoOutBufSiz) {
638  wxMessageBox(wxString::Format(_("FFmpeg : ERROR - Too much remaining data.")),
639  _("FFmpeg Error"), wxOK|wxCENTER|wxICON_EXCLAMATION);
640  return false;
641  }
642 
643  // Flush the audio FIFO first if necessary. It won't contain a _full_ audio frame because
644  // if it did we'd have pulled it from the FIFO during the last encodeAudioFrame() call -
645  // the encoder must support short/incomplete frames for this to work.
646  if (nFifoBytes > 0)
647  {
648  // Fill audio buffer with zeroes. If codec tries to read the whole buffer,
649  // it will just read silence. If not - who cares?
650  memset(mEncAudioFifoOutBuf,0,mEncAudioFifoOutBufSiz);
651  const AVCodec *codec = mEncAudioCodecCtx->codec;
652 
653  // We have an incomplete buffer of samples left. Is it OK to encode it?
654  // If codec supports CODEC_CAP_SMALL_LAST_FRAME, we can feed it with smaller frame
655  // Or if codec is FLAC, feed it anyway (it doesn't have CODEC_CAP_SMALL_LAST_FRAME, but it works)
656  // Or if frame_size is 1, then it's some kind of PCM codec, they don't have frames and will be fine with the samples
657  // Or if user configured the exporter to pad with silence, then we'll send audio + silence as a frame.
658  if ((codec->capabilities & (CODEC_CAP_SMALL_LAST_FRAME|CODEC_CAP_VARIABLE_FRAME_SIZE))
659  || mEncAudioCodecCtx->frame_size <= 1
660  || gPrefs->Read(wxT("/FileFormats/OverrideSmallLastFrame"), true)
661  )
662  {
663  int frame_size = default_frame_size;
664 
665  // The last frame is going to contain a smaller than usual number of samples.
666  // For codecs without CODEC_CAP_SMALL_LAST_FRAME use normal frame size
667  if (codec->capabilities & (CODEC_CAP_SMALL_LAST_FRAME|CODEC_CAP_VARIABLE_FRAME_SIZE))
668  frame_size = nFifoBytes / (mEncAudioCodecCtx->channels * sizeof(int16_t));
669 
670  wxLogDebug(wxT("FFmpeg : Audio FIFO still contains %d bytes, writing %d sample frame ..."),
671  nFifoBytes, frame_size);
672 
673  // Pull the bytes out from the FIFO and feed them to the encoder.
674  if (av_fifo_generic_read(mEncAudioFifo, mEncAudioFifoOutBuf, nFifoBytes, NULL) == 0)
675  {
676  nEncodedBytes = encode_audio(mEncAudioCodecCtx, &pkt, (int16_t*)mEncAudioFifoOutBuf, frame_size);
677  }
678  }
679  }
680 
681  // Now flush the encoder.
682  if (nEncodedBytes <= 0)
683  nEncodedBytes = encode_audio(mEncAudioCodecCtx, &pkt, NULL, 0);
684 
685  if (nEncodedBytes <= 0)
686  break;
687 
688  pkt.stream_index = mEncAudioStream->index;
689 
690  // Set presentation time of frame (currently in the codec's timebase) in the stream timebase.
691  if(pkt.pts != int64_t(AV_NOPTS_VALUE))
692  pkt.pts = av_rescale_q(pkt.pts, mEncAudioCodecCtx->time_base, mEncAudioStream->time_base);
693  if(pkt.dts != int64_t(AV_NOPTS_VALUE))
694  pkt.dts = av_rescale_q(pkt.dts, mEncAudioCodecCtx->time_base, mEncAudioStream->time_base);
695 
696  if (av_interleaved_write_frame(mEncFormatCtx, &pkt) != 0)
697  {
698  wxMessageBox(wxString::Format(_("FFmpeg : ERROR - Couldn't write last audio frame to output file.")),
699  _("FFmpeg Error"), wxOK|wxCENTER|wxICON_EXCLAMATION);
700  break;
701  }
702  av_free_packet(&pkt);
703  }
704 
705  // Write any file trailers.
706  av_write_trailer(mEncFormatCtx);
707 
708  // Close the codecs.
709  if (mEncAudioStream != NULL)
710  avcodec_close(mEncAudioStream->codec);
711 
712  for (i = 0; i < (int)mEncFormatCtx->nb_streams; i++)
713  {
714  av_freep(&mEncFormatCtx->streams[i]->codec);
715  av_freep(&mEncFormatCtx->streams[i]);
716  }
717 
718  // Close the output file if we created it.
719  if (!(mEncFormatDesc->flags & AVFMT_NOFILE))
720  ufile_close(mEncFormatCtx->pb);
721 
722  // Free any buffers or structures we allocated.
723  av_free(mEncFormatCtx);
724 
725  av_freep(&mEncAudioFifoOutBuf);
726  mEncAudioFifoOutBufSiz = 0;
727 
728  av_fifo_free(mEncAudioFifo);
729 
730  mEncAudioFifo = NULL;
731 
732  return true;
733 }

735 bool ExportFFmpeg::EncodeAudioFrame(int16_t *pFrame, int frameSize)
736 {
737  AVPacket pkt;
738  int nBytesToWrite = 0;
739  uint8_t *pRawSamples = NULL;
740  int nAudioFrameSizeOut = default_frame_size * mEncAudioCodecCtx->channels * sizeof(int16_t);
741  int ret;
742 
743  nBytesToWrite = frameSize;
744  pRawSamples = (uint8_t*)pFrame;
745  av_fifo_realloc2(mEncAudioFifo, av_fifo_size(mEncAudioFifo) + frameSize);
746 
747  // Put the raw audio samples into the FIFO.
748  ret = av_fifo_generic_write(mEncAudioFifo, pRawSamples, nBytesToWrite,NULL);
749 
750  wxASSERT(ret == nBytesToWrite);
751 
752  if (nAudioFrameSizeOut > mEncAudioFifoOutBufSiz) {
753  wxMessageBox(wxString::Format(_("FFmpeg : ERROR - nAudioFrameSizeOut too large.")),
754  _("FFmpeg Error"), wxOK|wxCENTER|wxICON_EXCLAMATION);
755  return false;
756  }
757 
758  // Read raw audio samples out of the FIFO in nAudioFrameSizeOut byte-sized groups to encode.
759  while ((ret = av_fifo_size(mEncAudioFifo)) >= nAudioFrameSizeOut)
760  {
761  ret = av_fifo_generic_read(mEncAudioFifo, mEncAudioFifoOutBuf, nAudioFrameSizeOut, NULL);
762 
763  av_init_packet(&pkt);
764 
765  int ret= encode_audio(mEncAudioCodecCtx,
766  &pkt, // out
767  (int16_t*)mEncAudioFifoOutBuf, // in
768  default_frame_size);
769  if (ret < 0)
770  {
771  wxMessageBox(wxString::Format(_("FFmpeg : ERROR - Can't encode audio frame.")),
772  _("FFmpeg Error"), wxOK|wxCENTER|wxICON_EXCLAMATION);
773  return false;
774  }
775  if (ret == 0)
776  continue;
777 
778  // Rescale from the codec time_base to the AVStream time_base.
779  if (pkt.pts != int64_t(AV_NOPTS_VALUE))
780  pkt.pts = av_rescale_q(pkt.pts, mEncAudioCodecCtx->time_base, mEncAudioStream->time_base);
781  if (pkt.dts != int64_t(AV_NOPTS_VALUE))
782  pkt.dts = av_rescale_q(pkt.dts, mEncAudioCodecCtx->time_base, mEncAudioStream->time_base);
783  //wxLogDebug(wxT("FFmpeg : (%d) Writing audio frame with PTS: %lld."), mEncAudioCodecCtx->frame_number, (long long) pkt.pts);
784 
785  pkt.stream_index = mEncAudioStream->index;
786 
787  // Write the encoded audio frame to the output file.
788  if ((ret = av_interleaved_write_frame(mEncFormatCtx, &pkt)) < 0)
789  {
790  wxMessageBox(wxString::Format(_("FFmpeg : ERROR - Failed to write audio frame to file.")),
791  _("FFmpeg Error"), wxOK|wxCENTER|wxICON_EXCLAMATION);
792  return false;
793  }
794  av_free_packet(&pkt);
795  }
796  return true;
797 }
