375 // returns true if the combo is constrained and false otherwise
376 // @param toolbarWidth the width of the toolbar in pixels
377 // @param ratio an in/out for the desired and resultant width ratio.
378 // @param flex the amount of extra space allowed to have past the available.
379 // the amount used is subtracted.
380 static bool RepositionCombo(wxWindow *combo, int toolbarWidth, wxSize desiredSize,
381  float &ratio, float &flex, int marginPixels, bool changesRatio)
382 {
383  float ratioChange;
384  bool constrained = false;
385 
386  // push margin pixels
387  desiredSize.x += marginPixels;
388 
389  // truncate the window size if necessary
390  if (desiredSize.x > toolbarWidth * (flex + ratio)) {
391  constrained = true;
392  desiredSize.SetWidth(toolbarWidth * (flex + ratio));
393  if (desiredSize.GetWidth() - marginPixels < 0)
394  desiredSize.SetWidth(marginPixels);
395  }
396 
397  // keep track of how much space gained or lost so it can be used by other combos.
398  if (changesRatio) {
399  ratioChange = (desiredSize.x / ((float) toolbarWidth)) - ratio;
400  ratio += ratioChange;
401  flex -= ratioChange;
402  }
403 
404  // pop the margin pixels
405  desiredSize.x -= marginPixels;
406 
407  combo->SetMinSize(desiredSize);
408 
409  return constrained;
410 }

415 #define kInputWidthRatio 0.32f
416 #define kOutputWidthRatio 0.32f
417 #define kChannelsWidthRatio 0.18f

419 void DeviceToolBar::RepositionCombos()
420 {
421  int w, h, dockw, dockh;
422  float ratioUnused;
423  bool constrained = true;
424  wxWindow *window;
425  wxSize desiredInput, desiredOutput, desiredHost, desiredChannels;
426  float hostRatio, outputRatio, inputRatio, channelsRatio;
427  // if the toolbar is docked then the width we should use is the project width.
428  // as the toolbar's with can extend past this.
429  GetClientSize(&w, &h);
430 
431  // FIXME: Note that there's some bug in here, in that even if the prefs show the toolbar
432  // docked, on initialization, this call to IsDocked() returns false.
433  if (IsDocked()) {
434  // If the toolbar is docked its width can be larger than what is actually viewable
435  // So take the min. We don't need to worry about having another toolbar to the left off us
436  // because if we are larger than the dock size we always get our own row.
437  // and if smaller then we don't use the dock size (because we take the min).
438  window = GetDock();
439  window->GetClientSize(&dockw, &dockh);
440  if (dockw < w)
441  w = dockw;
442  }
443  // subtract the main grabber on the left and the resizer as well
444  w -= grabberWidth + GetResizeGrabberWidth();
445  if (w <= 0)
446  return;
447 
448  // set up initial sizes and ratios
449  hostRatio = kHostWidthRatio;
450  inputRatio = kInputWidthRatio;
451  outputRatio = kOutputWidthRatio;
452  channelsRatio = kChannelsWidthRatio;
453 
454  desiredHost = mHost->GetBestSize();
455  desiredInput = mInput->GetBestSize();
456  desiredOutput = mOutput->GetBestSize();
457  desiredChannels = mInputChannels->GetBestSize();
458 
459  // wxGtk has larger comboboxes than the other platforms. For DeviceToolBar this will cause
460  // the height to be double because of the discrete grid layout. So we shrink it to prevent this.
461 #ifdef __WXGTK__
462  desiredHost.SetHeight(desiredHost.GetHeight() -4);
463  desiredInput.SetHeight(desiredHost.GetHeight());
464  desiredOutput.SetHeight(desiredHost.GetHeight());
465  desiredChannels.SetHeight(desiredHost.GetHeight());
466 #endif
467 
468  ratioUnused = 0.995f - (kHostWidthRatio + kInputWidthRatio + kOutputWidthRatio + kChannelsWidthRatio);
469  int i = 0;
470  // limit the amount of times we solve contraints to 5
471  while (constrained && ratioUnused > 0.01f && i < 5) {
472  i++;
473  constrained = RepositionCombo(mHost, w, desiredHost, hostRatio, ratioUnused, 0, true);
474  constrained |= RepositionCombo(mInput, w, desiredInput, inputRatio, ratioUnused, mRecordBitmap->GetWidth(), true);
475  constrained |= RepositionCombo(mOutput, w, desiredOutput, outputRatio, ratioUnused, mPlayBitmap->GetWidth(), true);
476  constrained |= RepositionCombo(mInputChannels, w, desiredChannels, channelsRatio, ratioUnused, 0, true);
477  }
478 
479  Update();
480 }

507 void DeviceToolBar::FillHostDevices()
508 {
509  const std::vector<DeviceSourceMap> &inMaps = DeviceManager::Instance()->GetInputDeviceMaps();
510  const std::vector<DeviceSourceMap> &outMaps = DeviceManager::Instance()->GetOutputDeviceMaps();
511 
512  //read what is in the prefs
513  wxString host = gPrefs->Read(wxT("/AudioIO/Host"), wxT(""));
514  size_t i;
515  int foundHostIndex = -1;
516 
517  // if the host is not in the hosts combo then we rescanned.
518  // set it to blank so we search for another host.
519  if (mHost->FindString(host) == wxNOT_FOUND)
520  host = wxT("");
521 
522  for (i = 0; i < outMaps.size(); i++) {
523  if (outMaps[i].hostString == host) {
524  foundHostIndex = outMaps[i].hostIndex;
525  break;
526  }
527  }
528 
529  if (foundHostIndex == -1) {
530  for (i = 0; i < inMaps.size(); i++) {
531  if (inMaps[i].hostString == host) {
532  foundHostIndex = inMaps[i].hostIndex;
533  break;
534  }
535  }
536  }
537 
538  // If no host was found based on the prefs device host, load the first available one
539  if (foundHostIndex == -1) {
540  if (outMaps.size())
541  foundHostIndex = outMaps[0].hostIndex;
542  else if (inMaps.size())
543  foundHostIndex = inMaps[0].hostIndex;
544  }
545 
546  // Make sure in/out are clear in case no host was found
547  mInput->Clear();
548  mOutput->Clear();
549 
550  // If we still have no host it means no devices, in which case do nothing.
551  if (foundHostIndex == -1)
552  return;
553 
554  // Repopulate the Input/Output device list available to the user
555  for (i = 0; i < inMaps.size(); i++) {
556  if (foundHostIndex == inMaps[i].hostIndex) {
557  mInput->Append(MakeDeviceSourceString(&inMaps[i]));
558  if (host == wxT("")) {
559  host = inMaps[i].hostString;
560  gPrefs->Write(wxT("/AudioIO/Host"), host);
561  mHost->SetStringSelection(host);
562  }
563  }
564  }
565  mInput->Enable(mInput->GetCount() ? true : false);
566 
567  mInput->InvalidateBestSize();
568  mInput->SetMaxSize(mInput->GetBestSize());
569 
570  for (i = 0; i < outMaps.size(); i++) {
571  if (foundHostIndex == outMaps[i].hostIndex) {
572  mOutput->Append(MakeDeviceSourceString(&outMaps[i]));
573  if (host == wxT("")) {
574  host = outMaps[i].hostString;
575  gPrefs->Write(wxT("/AudioIO/Host"), host);
576  gPrefs->Flush();
577  mHost->SetStringSelection(host);
578  }
579  }
580  }
581  mOutput->Enable(mOutput->GetCount() ? true : false);
582 
583  mOutput->InvalidateBestSize();
584  mOutput->SetMaxSize(mOutput->GetBestSize());
585 
586  // The setting of the Device is left up to OnChoice
587 }

720 void DeviceToolBar::OnChoice(wxCommandEvent &event)
721 {
722  wxObject *eventObject = event.GetEventObject();
723  //if we've changed hosts, we've handled the device switching already.
724  if (eventObject == mHost) {
725  ChangeHost();
726  } else if (eventObject == mInputChannels) {
727  int channelsSelectionIndex = mInputChannels->GetSelection();
728  if (channelsSelectionIndex >= 0)
729  gPrefs->Write(wxT("/AudioIO/RecordChannels"),channelsSelectionIndex + 1);
730  } else if (eventObject == mInput) {
731  ChangeDevice(true);
732  }
733  else if (eventObject == mOutput) {
734  ChangeDevice(false);
735  }
736 
737  if (gAudioIO) {
738  // We cannot have gotten here if gAudioIO->IsAudioTokenActive(),
739  // per the setting of AudioIONotBusyFlag and AudioIOBusyFlag in
740  // AudacityProject::GetUpdateFlags().
741  // However, we can have an invalid audio token (so IsAudioTokenActive()
742  // is false), but be monitoring.
743  // If monitoring, have to stop the stream, so HandleDeviceChange() can work.
744  // We could disable the Preferences command while monitoring, i.e.,
745  // set AudioIONotBusyFlag/AudioIOBusyFlag according to monitoring, as well.
746  // Instead allow it because unlike recording, for example, monitoring
747  // is not clearly something that should prohibit changing device.
748  // TODO: We *could* be smarter in this method and call HandleDeviceChange()
749  // only when the device choices actually changed. True of lots of prefs!
750  // As is, we always stop monitoring before handling the device change.
751  if (gAudioIO->IsMonitoring())
752  {
753  gAudioIO->StopStream();
754  while (gAudioIO->IsBusy())
755  wxMilliSleep(100);
756  }
757  gAudioIO->HandleDeviceChange();
758  }
759 
760  // Update all projects' DeviceToolBar.
761  for (size_t i = 0; i < gAudacityProjects.GetCount(); i++) {
762  gAudacityProjects[i]->GetDeviceToolBar()->UpdatePrefs();
763  }
764 }
