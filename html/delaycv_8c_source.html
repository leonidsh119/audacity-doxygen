838 sound_type snd_make_delaycv(sound_type s, time_type delay, sound_type feedback)
839 {
840  register delaycv_susp_type susp;
841  rate_type sr = max(s->sr, feedback->sr);
842  time_type t0 = max(s->t0, feedback->t0);
843  int interp_desc = 0;
844  sample_type scale_factor = 1.0F;
845  time_type t0_min = t0;
846  /* combine scale factors of linear inputs (S) */
847  scale_factor *= s->scale;
848  s->scale = 1.0F;
849 
850  /* try to push scale_factor back to a low sr input */
851  if (s->sr < sr) { s->scale = scale_factor; scale_factor = 1.0F; }
852 
853  falloc_generic(susp, delaycv_susp_node, "snd_make_delaycv");
854  susp->delaylen = round(s->sr * delay);
855  susp->delaybuf = (sample_type *) calloc (sizeof(double), susp->delaylen);
856  susp->delayptr = susp->delaybuf;
857  susp->endptr = susp->delaybuf + susp->delaylen;
858 
859  /* select a susp fn based on sample rates */
860  interp_desc = (interp_desc << 2) + interp_style(s, sr);
861  interp_desc = (interp_desc << 2) + interp_style(feedback, sr);
862  switch (interp_desc) {
863  case INTERP_nn: susp->susp.fetch = delaycv_nn_fetch; break;
864  case INTERP_ns: susp->susp.fetch = delaycv_ns_fetch; break;
865  case INTERP_ni: susp->susp.fetch = delaycv_ni_fetch; break;
866  case INTERP_nr: susp->susp.fetch = delaycv_nr_fetch; break;
867  case INTERP_in: susp->susp.fetch = delaycv_in_fetch; break;
868  case INTERP_is: susp->susp.fetch = delaycv_is_fetch; break;
869  case INTERP_rn: susp->susp.fetch = delaycv_rn_fetch; break;
870  case INTERP_rs: susp->susp.fetch = delaycv_rs_fetch; break;
871  default: snd_badsr(); break;
872  }
873 
874  susp->terminate_cnt = UNKNOWN;
875  /* handle unequal start times, if any */
876  if (t0 < s->t0) sound_prepend_zeros(s, t0);
877  if (t0 < feedback->t0) sound_prepend_zeros(feedback, t0);
878  /* minimum start time over all inputs: */
879  t0_min = min(s->t0, min(feedback->t0, t0));
880  /* how many samples to toss before t0: */
881  susp->susp.toss_cnt = (long) ((t0 - t0_min) * sr + 0.5);
882  if (susp->susp.toss_cnt > 0) {
883  susp->susp.keep_fetch = susp->susp.fetch;
884  susp->susp.fetch = delaycv_toss_fetch;
885  }
886 
887  /* initialize susp state */
888  susp->susp.free = delaycv_free;
889  susp->susp.sr = sr;
890  susp->susp.t0 = t0;
891  susp->susp.mark = delaycv_mark;
892  susp->susp.print_tree = delaycv_print_tree;
893  susp->susp.name = "delaycv";
894  susp->susp.log_stop_cnt = UNKNOWN;
895  susp->started = false;
896  susp->susp.current = 0;
897  susp->s = s;
898  susp->s_cnt = 0;
899  susp->s_pHaSe = 0.0;
900  susp->s_pHaSe_iNcR = s->sr / sr;
901  susp->s_n = 0;
902  susp->output_per_s = sr / s->sr;
903  susp->feedback = feedback;
904  susp->feedback_cnt = 0;
905  susp->feedback_pHaSe = 0.0;
906  susp->feedback_pHaSe_iNcR = feedback->sr / sr;
907  susp->feedback_n = 0;
908  susp->output_per_feedback = sr / feedback->sr;
909  return sound_create((snd_susp_type)susp, t0, sr, scale_factor);
910 }
