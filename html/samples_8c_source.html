181 /* snd_fetch_array -- fetch a lisp array of samples */
182 /*
183  * storage layout: the extra field points to extra state that we'll use
184  * extra[0] -> length of extra storage
185  * extra[1] -> CNT (number of samples in current block)
186  * extra[2] -> INDEX (current sample index in current block)
187  * extra[3] -> FILLCNT (how many samples in buffer)
188  * extra[4] -> TERMCNT (how many samples until termination)
189  * extra[4 .. 4+len-1] -> samples (stored as floats)
190  *
191  * Termination details:
192  * Return NIL when the sound terminates.
193  * Termination is defined as the point where all original
194  * signal samples have been shifted out of the samples buffer
195  * so that all that's left are zeros from beyond the termination
196  * point.
197  * Implementation: when termination is discovered, set TERMCNT
198  * to the number of samples to be shifted out. TERMCNT is initially
199  * -1 as a flag that we haven't seen the termination yet.
200  * Each time samples are shifted, decrement TERMCNT by the shift amount.
201  * When TERMCNT goes to zero, return NULL.
202  */
203 /* these macros define entries in extra, more macros are defined above */
204 #define FILLCNT extra[3]
205 #define TERMCNT extra[4]
206 #define OFFSET 5
207 
208 LVAL snd_fetch_array(sound_type s, long len, long step)
209 {
210  long i, maxlen, skip, fillptr;
211  float *samples;
212  LVAL result;
213  LVAL rslt_symbol = xlenter("*RSLT*");
214 
215  setvalue(rslt_symbol, NULL);
216 
217  if (len < 1) xlfail("len < 1");
218 
219  if (!s->extra) { /* this is the first call, so fix up s */
220  s->extra = (long *) malloc(sizeof(long) * (len + OFFSET));
221  s->extra[0] = sizeof(long) * (len + OFFSET);
222  s->CNT = s->INDEX = s->FILLCNT = 0;
223  s->TERMCNT = -1;
224  maxlen = len;
225  } else {

227  if (maxlen < 1) xlfail("sound in use by another iterator");
228  if (maxlen < len) xlfail("len grew");
229  }
230  samples = (float *) &(s->extra[OFFSET]);
231 
232  /* step 1: refill buffer with samples */

234  while (fillptr < maxlen) {
235  if (s->INDEX == s->CNT) {
236  sound_get_next(s, &(s->CNT));
237  if (s->SAMPLES == zero_block->samples) {
238  setvalue(rslt_symbol, cvfixnum(fillptr));
239  if (s->TERMCNT < 0) s->TERMCNT = fillptr;
240  }
241  s->INDEX = 0;
242  }
243  samples[fillptr++] = s->SAMPLES[s->INDEX++] * s->scale;
244  }
245  s->FILLCNT = fillptr;
246

248  * if fillptr WAS 0 when we hit the zero_block, then filling the
249  * buffer will set TERMCNT to 0.
250  */
251  if (s->TERMCNT == 0) return NULL;
252

254  * you would need a way to return the logical stop time to
255  * the caller.
256  */
257

259  xlsave1(result);
260  result = newvector(len);
261

263  setelement(result, i, cvflonum(samples[i]));
264  }

268  s->FILLCNT -= step;
269  if (s->FILLCNT < 0) s->FILLCNT = 0;
270  for (i = 0; i < s->FILLCNT; i++) {
271  samples[i] = samples[i + step];
272  }
273 
274 
275  if (s->TERMCNT >= 0) {
276  s->TERMCNT -= step;
277  if (s->TERMCNT < 0) s->TERMCNT = 0;
278  }
279 
280 
281  /* step 4: advance in sound to next sample we need
282  * (only does work if step > size of buffer)
283  */
284  skip = step - maxlen;
285  while (skip > 0) {
286  long remaining = s->CNT - s->INDEX;
287  if (remaining >= skip) {
288  s->INDEX += skip;
289  skip = 0;
290  } else {
291  skip -= remaining;
292  sound_get_next(s, &(s->CNT));
293  s->INDEX = 0;
294  }
295  }
296

298  xlpop();
299  return result;
