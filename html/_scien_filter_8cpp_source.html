619 bool EffectScienFilter::CalcFilter()
620 {
621  // Set up the coefficients in all the biquads
622  float fNorm = mCutoff / mNyquist;
623  if (fNorm >= 0.9999)
624  fNorm = 0.9999F;
625  float fC = tan (PI * fNorm / 2);
626  float fDCPoleDistSqr = 1.0F;
627  float fZPoleX, fZPoleY;
628  float fZZeroX, fZZeroY;
629  float beta = cos (fNorm*PI);
630  switch (mFilterType)
631  {
632  case kButterworth: // Butterworth
633  if ((mOrder & 1) == 0)
634  {
635  // Even order
636  for (int iPair = 0; iPair < mOrder/2; iPair++)
637  {
638  float fSPoleX = fC * cos (PI - (iPair + 0.5) * PI / mOrder);
639  float fSPoleY = fC * sin (PI - (iPair + 0.5) * PI / mOrder);
640  BilinTransform (fSPoleX, fSPoleY, &fZPoleX, &fZPoleY);
641  mpBiquad[iPair].fNumerCoeffs [0] = 1;
642  if (mFilterSubtype == kLowPass) // LOWPASS
643  mpBiquad[iPair].fNumerCoeffs [1] = 2;
644  else
645  mpBiquad[iPair].fNumerCoeffs [1] = -2;
646  mpBiquad[iPair].fNumerCoeffs [2] = 1;
647  mpBiquad[iPair].fDenomCoeffs [0] = -2 * fZPoleX;
648  mpBiquad[iPair].fDenomCoeffs [1] = square(fZPoleX) + square(fZPoleY);
649  if (mFilterSubtype == kLowPass) // LOWPASS
650  fDCPoleDistSqr *= Calc2D_DistSqr (1, 0, fZPoleX, fZPoleY);
651  else
652  fDCPoleDistSqr *= Calc2D_DistSqr (-1, 0, fZPoleX, fZPoleY); // distance from Nyquist
653  }
654  }
655  else
656  {
657  // Odd order - first do the 1st-order section
658  float fSPoleX = -fC;
659  float fSPoleY = 0;
660  BilinTransform (fSPoleX, fSPoleY, &fZPoleX, &fZPoleY);
661  mpBiquad[0].fNumerCoeffs [0] = 1;
662  if (mFilterSubtype == kLowPass) // LOWPASS
663  mpBiquad[0].fNumerCoeffs [1] = 1;
664  else
665  mpBiquad[0].fNumerCoeffs [1] = -1;
666  mpBiquad[0].fNumerCoeffs [2] = 0;
667  mpBiquad[0].fDenomCoeffs [0] = -fZPoleX;
668  mpBiquad[0].fDenomCoeffs [1] = 0;
669  if (mFilterSubtype == kLowPass) // LOWPASS
670  fDCPoleDistSqr = 1 - fZPoleX;
671  else
672  fDCPoleDistSqr = fZPoleX + 1; // dist from Nyquist
673  for (int iPair = 1; iPair <= mOrder/2; iPair++)
674  {
675  float fSPoleX = fC * cos (PI - iPair * PI / mOrder);
676  float fSPoleY = fC * sin (PI - iPair * PI / mOrder);
677  BilinTransform (fSPoleX, fSPoleY, &fZPoleX, &fZPoleY);
678  mpBiquad[iPair].fNumerCoeffs [0] = 1;
679  if (mFilterSubtype == kLowPass) // LOWPASS
680  mpBiquad[iPair].fNumerCoeffs [1] = 2;
681  else
682  mpBiquad[iPair].fNumerCoeffs [1] = -2;
683  mpBiquad[iPair].fNumerCoeffs [2] = 1;
684  mpBiquad[iPair].fDenomCoeffs [0] = -2 * fZPoleX;
685  mpBiquad[iPair].fDenomCoeffs [1] = square(fZPoleX) + square(fZPoleY);
686  if (mFilterSubtype == kLowPass) // LOWPASS
687  fDCPoleDistSqr *= Calc2D_DistSqr (1, 0, fZPoleX, fZPoleY);
688  else
689  fDCPoleDistSqr *= Calc2D_DistSqr (-1, 0, fZPoleX, fZPoleY); // distance from Nyquist
690  }
691  }
692  mpBiquad[0].fNumerCoeffs [0] *= fDCPoleDistSqr / (1 << mOrder); // mult by DC dist from poles, divide by dist from zeroes
693  mpBiquad[0].fNumerCoeffs [1] *= fDCPoleDistSqr / (1 << mOrder);
694  mpBiquad[0].fNumerCoeffs [2] *= fDCPoleDistSqr / (1 << mOrder);
695  break;
696 
697  case kChebyshevTypeI: // Chebyshev Type 1
698  double eps; eps = sqrt (pow (10.0, wxMax(0.001, mRipple) / 10.0) - 1);
699  double a; a = log (1 / eps + sqrt(1 / square(eps) + 1)) / mOrder;
700  // Assume even order to start
701  for (int iPair = 0; iPair < mOrder/2; iPair++)
702  {
703  float fSPoleX = -fC * sinh (a) * sin ((2*iPair + 1) * PI / (2 * mOrder));
704  float fSPoleY = fC * cosh (a) * cos ((2*iPair + 1) * PI / (2 * mOrder));
705  BilinTransform (fSPoleX, fSPoleY, &fZPoleX, &fZPoleY);
706  if (mFilterSubtype == kLowPass) // LOWPASS
707  {
708  fZZeroX = -1;
709  fDCPoleDistSqr = Calc2D_DistSqr (1, 0, fZPoleX, fZPoleY);
710  fDCPoleDistSqr /= 2*2; // dist from zero at Nyquist
711  }
712  else
713  {
714  // Highpass - do the digital LP->HP transform on the poles and zeroes
715  ComplexDiv (beta - fZPoleX, -fZPoleY, 1 - beta * fZPoleX, -beta * fZPoleY, &fZPoleX, &fZPoleY);
716  fZZeroX = 1;
717  fDCPoleDistSqr = Calc2D_DistSqr (-1, 0, fZPoleX, fZPoleY); // distance from Nyquist
718  fDCPoleDistSqr /= 2*2; // dist from zero at Nyquist
719  }
720  mpBiquad[iPair].fNumerCoeffs [0] = fDCPoleDistSqr;
721  mpBiquad[iPair].fNumerCoeffs [1] = -2 * fZZeroX * fDCPoleDistSqr;
722  mpBiquad[iPair].fNumerCoeffs [2] = fDCPoleDistSqr;
723  mpBiquad[iPair].fDenomCoeffs [0] = -2 * fZPoleX;
724  mpBiquad[iPair].fDenomCoeffs [1] = square(fZPoleX) + square(fZPoleY);
725  }
726  if ((mOrder & 1) == 0)
727  {
728  float fTemp = DB_TO_LINEAR(-wxMax(0.001, mRipple)); // at DC the response is down R dB (for even-order)
729  mpBiquad[0].fNumerCoeffs [0] *= fTemp;
730  mpBiquad[0].fNumerCoeffs [1] *= fTemp;
731  mpBiquad[0].fNumerCoeffs [2] *= fTemp;
732  }
733  else
734  {
735  // Odd order - now do the 1st-order section
736  float fSPoleX = -fC * sinh (a);
737  float fSPoleY = 0;
738  BilinTransform (fSPoleX, fSPoleY, &fZPoleX, &fZPoleY);
739  if (mFilterSubtype == kLowPass) // LOWPASS
740  {
741  fZZeroX = -1;
742  fDCPoleDistSqr = sqrt(Calc2D_DistSqr (1, 0, fZPoleX, fZPoleY));
743  fDCPoleDistSqr /= 2; // dist from zero at Nyquist
744  }
745  else
746  {
747  // Highpass - do the digital LP->HP transform on the poles and zeroes
748  ComplexDiv (beta - fZPoleX, -fZPoleY, 1 - beta * fZPoleX, -beta * fZPoleY, &fZPoleX, &fZPoleY);
749  fZZeroX = 1;
750  fDCPoleDistSqr = sqrt(Calc2D_DistSqr (-1, 0, fZPoleX, fZPoleY)); // distance from Nyquist
751  fDCPoleDistSqr /= 2; // dist from zero at Nyquist
752  }
753  mpBiquad[(mOrder-1)/2].fNumerCoeffs [0] = fDCPoleDistSqr;
754  mpBiquad[(mOrder-1)/2].fNumerCoeffs [1] = -fZZeroX * fDCPoleDistSqr;
755  mpBiquad[(mOrder-1)/2].fNumerCoeffs [2] = 0;
756  mpBiquad[(mOrder-1)/2].fDenomCoeffs [0] = -fZPoleX;
757  mpBiquad[(mOrder-1)/2].fDenomCoeffs [1] = 0;
758  }
759  break;
760 
761  case kChebyshevTypeII: // Chebyshev Type 2
762  float fSZeroX, fSZeroY;
763  float fSPoleX, fSPoleY;
764  eps = DB_TO_LINEAR(-wxMax(0.001, mStopbandRipple));
765  a = log (1 / eps + sqrt(1 / square(eps) + 1)) / mOrder;
766 
767  // Assume even order
768  for (int iPair = 0; iPair < mOrder/2; iPair++)
769  {
770  ComplexDiv (fC, 0, -sinh (a) * sin ((2*iPair + 1) * PI / (2 * mOrder)),
771  cosh (a) * cos ((2*iPair + 1) * PI / (2 * mOrder)),
772  &fSPoleX, &fSPoleY);
773  BilinTransform (fSPoleX, fSPoleY, &fZPoleX, &fZPoleY);
774  fSZeroX = 0;
775  fSZeroY = fC / cos (((2 * iPair) + 1) * PI / (2 * mOrder));
776  BilinTransform (fSZeroX, fSZeroY, &fZZeroX, &fZZeroY);
777 
778  if (mFilterSubtype == kLowPass) // LOWPASS
779  {
780  fDCPoleDistSqr = Calc2D_DistSqr (1, 0, fZPoleX, fZPoleY);
781  fDCPoleDistSqr /= Calc2D_DistSqr (1, 0, fZZeroX, fZZeroY);
782  }
783  else
784  {
785  // Highpass - do the digital LP->HP transform on the poles and zeroes
786  ComplexDiv (beta - fZPoleX, -fZPoleY, 1 - beta * fZPoleX, -beta * fZPoleY, &fZPoleX, &fZPoleY);
787  ComplexDiv (beta - fZZeroX, -fZZeroY, 1 - beta * fZZeroX, -beta * fZZeroY, &fZZeroX, &fZZeroY);
788  fDCPoleDistSqr = Calc2D_DistSqr (-1, 0, fZPoleX, fZPoleY); // distance from Nyquist
789  fDCPoleDistSqr /= Calc2D_DistSqr (-1, 0, fZZeroX, fZZeroY);
790  }
791  mpBiquad[iPair].fNumerCoeffs [0] = fDCPoleDistSqr;
792  mpBiquad[iPair].fNumerCoeffs [1] = -2 * fZZeroX * fDCPoleDistSqr;
793  mpBiquad[iPair].fNumerCoeffs [2] = (square(fZZeroX) + square(fZZeroY)) * fDCPoleDistSqr;
794  mpBiquad[iPair].fDenomCoeffs [0] = -2 * fZPoleX;
795  mpBiquad[iPair].fDenomCoeffs [1] = square(fZPoleX) + square(fZPoleY);
796  }
797  // Now, if it's odd order, we have one more to do
798  if (mOrder & 1)
799  {
800  int iPair = (mOrder-1)/2; // we'll do it as a biquad, but it's just first-order
801  ComplexDiv (fC, 0, -sinh (a) * sin ((2*iPair + 1) * PI / (2 * mOrder)),
802  cosh (a) * cos ((2*iPair + 1) * PI / (2 * mOrder)),
803  &fSPoleX, &fSPoleY);
804  BilinTransform (fSPoleX, fSPoleY, &fZPoleX, &fZPoleY);
805  fZZeroX = -1; // in the s-plane, the zero is at infinity
806  fZZeroY = 0;
807  if (mFilterSubtype == kLowPass) // LOWPASS
808  {
809  fDCPoleDistSqr = sqrt(Calc2D_DistSqr (1, 0, fZPoleX, fZPoleY));
810  fDCPoleDistSqr /= 2;
811  }
812  else
813  {
814  // Highpass - do the digital LP->HP transform on the poles and zeroes
815  ComplexDiv (beta - fZPoleX, -fZPoleY, 1 - beta * fZPoleX, -fZPoleY, &fZPoleX, &fZPoleY);
816  fZZeroX = 1;
817  fDCPoleDistSqr = sqrt(Calc2D_DistSqr (-1, 0, fZPoleX, fZPoleY)); // distance from Nyquist
818  fDCPoleDistSqr /= 2;
819  }
820  mpBiquad[iPair].fNumerCoeffs [0] = fDCPoleDistSqr;
821  mpBiquad[iPair].fNumerCoeffs [1] = -fZZeroX * fDCPoleDistSqr;
822  mpBiquad[iPair].fNumerCoeffs [2] = 0;
823  mpBiquad[iPair].fDenomCoeffs [0] = -fZPoleX;
824  mpBiquad[iPair].fDenomCoeffs [1] = 0;
825  }
826  break;
827  }
828  return true;

845 float EffectScienFilter::FilterMagnAtFreq(float Freq)
846 {
847  float Magn;
848  if (Freq >= mNyquist)
849  Freq = mNyquist - 1; // prevent tan(PI/2)
850  float FreqWarped = tan (PI * Freq/(2*mNyquist));
851  if (mCutoff >= mNyquist)
852  mCutoff = mNyquist - 1;
853  float CutoffWarped = tan (PI * mCutoff/(2*mNyquist));
854  float fOverflowThresh = pow (10.0, 12.0 / (2*mOrder)); // once we exceed 10^12 there's not much to be gained and overflow could happen
855 
856  switch (mFilterType)
857  {
858  case kButterworth: // Butterworth
859  default:
860  switch (mFilterSubtype)
861  {
862  case kLowPass: // lowpass
863  default:
864  if (FreqWarped/CutoffWarped > fOverflowThresh) // prevent pow() overflow
865  Magn = 0;
866  else
867  Magn = sqrt (1 / (1 + pow (FreqWarped/CutoffWarped, 2*mOrder)));
868  break;
869  case kHighPass: // highpass
870  if (FreqWarped/CutoffWarped > fOverflowThresh)
871  Magn = 1;
872  else
873  Magn = sqrt (pow (FreqWarped/CutoffWarped, 2*mOrder) / (1 + pow (FreqWarped/CutoffWarped, 2*mOrder)));
874  break;
875  }
876  break;
877 
878  case kChebyshevTypeI: // Chebyshev Type 1
879  double eps; eps = sqrt(pow (10.0, wxMax(0.001, mRipple)/10.0) - 1);
880  switch (mFilterSubtype)
881  {
882  case 0: // lowpass
883  default:
884  Magn = sqrt (1 / (1 + square(eps) * square(ChebyPoly(mOrder, FreqWarped/CutoffWarped))));
885  break;
886  case 1:
887  Magn = sqrt (1 / (1 + square(eps) * square(ChebyPoly(mOrder, CutoffWarped/FreqWarped))));
888  break;
889  }
890  break;
891 
892  case kChebyshevTypeII: // Chebyshev Type 2
893  eps = 1 / sqrt(pow (10.0, wxMax(0.001, mStopbandRipple)/10.0) - 1);
894  switch (mFilterSubtype)
895  {
896  case kLowPass: // lowpass
897  default:
898  Magn = sqrt (1 / (1 + 1 / (square(eps) * square(ChebyPoly(mOrder, CutoffWarped/FreqWarped)))));
899  break;
900  case kHighPass:
901  Magn = sqrt (1 / (1 + 1 / (square(eps) * square(ChebyPoly(mOrder, FreqWarped/CutoffWarped)))));
902  break;
903  }
904  break;
905  }
906 
907  return Magn;
908 }
