65 /* Non-linear amplifier with soft distortion curve. */
66 float CubicAmplifier( float input )
67 {
68  float output, temp;
69  if( input < 0.0 )
70  {
71  temp = input + 1.0f;
72  output = (temp * temp * temp) - 1.0f;
73  }
74  else
75  {
76  temp = input - 1.0f;
77  output = (temp * temp * temp) + 1.0f;
78  }
79 
80  return output;
81 }
