235  IDiscMaster *pDiscMaster;

250  }
251 
252  CoUninitialize();
253  }
254

258  return pbErrBurnFailed;

445 /* Return a human-readable error string for the last operating system
446  specific error (NOT human readable strings for the PortBurn error
447  codes). Caller should dispose of the returned string using free(). */
448 char *PortBurn_v1_LastError(void *handle)
449 {
450  PBHandlev1 *h = (PBHandlev1 *) handle;
451  HRESULT hr;
452  LPTSTR windowsErrorString = NULL;
453  char *errorString = NULL;
454  int len;
455 
456  if (h == NULL) {
457  return NULL;
458  }
459 
460  /* Have Windows allocate a buffer for us and format the error
461  message in windowsErrorString */
462  hr = h->hres;
463  if (HRESULT_FACILITY(hr) == FACILITY_WINDOWS) {
464  hr = HRESULT_CODE(hr);
465  }
466 
467  len = FormatMessage(FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM,
468  NULL, hr, MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
469  (LPTSTR) &windowsErrorString, 0, NULL);
470  if (windowsErrorString == NULL) {
471  windowsErrorString = (LPTSTR) LocalAlloc(LPTR, 128 * sizeof(TCHAR));
472  if (windowsErrorString == NULL) {
473  return NULL;
474  }
475 
476  len = _stprintf_s(windowsErrorString, 128, L"HRESULT = %08x", hr);
477  }
478 
479  /* Convert the string */
480  errorString = (char *) malloc(len + 1);
481  if (errorString != NULL) {
482  errorString[0] = '\0';
483  WideCharToMultiByte(CP_ACP, 0, windowsErrorString, len, errorString, len, NULL, NULL);
484  errorString[len] = '\0';
485  }
486 
487  LocalFree(windowsErrorString);
488 
489  return errorString;
490 }

492 /* Get the number of devices capable of burning audio CDs.
493  If the result is N, then calls to GetDeviceName and OpenDevice
494  are guaranteed to be valid for indices from 0 up to N-1, until
495  the next time you call GetNumDevices. At that point, the list of
496  devices will be rescanned, and may be different. */
497 int PortBurn_v1_GetNumDevices(void *handle)
498 {
499  PBHandlev1 *h = (PBHandlev1 *) handle;
500  IDiscRecorder *pDiscRecorder;
501  ULONG fetched;
502  int count = 0;
503 
504  if (h == NULL) {
505  return 0;
506  }
507 
508  if (h->pEnumDiscRecorders != NULL) {
509  h->pEnumDiscRecorders->Release();
510  h->pEnumDiscRecorders = NULL;
511  }
512 
513  h->hres = h->pDiscMaster->EnumDiscRecorders(&h->pEnumDiscRecorders);
514  if (FAILED(h->hres)) {
515  return 0;
516  }
517 
518  while (h->pEnumDiscRecorders->Next(1, &pDiscRecorder, &fetched) == S_OK) {
519  pDiscRecorder->Release();
520  count++;
521  }
522 
523  return count;
524 }

526 /* Get the name of the device with a given index. Only valid
527  after a call to GetNumDevices. */
528 char *PortBurn_v1_GetDeviceName(void *handle, int index)
529 {
530  PBHandlev1 *h = (PBHandlev1 *) handle;
531  IDiscRecorder *pDiscRecorder;
532  BSTR bVendor = NULL;
533  BSTR bProduct = NULL;
534  BSTR bRevision = NULL;
535  ULONG fetched;
536  TCHAR *wname;
537  char *name;
538  int len;
539 
540  if (h == NULL) {
541  return NULL;
542  }
543 
544  h->hres = S_OK;
545 
546  if (index < 0 || index >= PortBurn_v1_GetNumDevices(h)) {
547  h->hres = E_INVALIDARG;
548  return NULL;
549  }
550 
551  h->pEnumDiscRecorders->Reset();
552  if (index != 0) {
553  h->pEnumDiscRecorders->Skip(index - 1);
554  }
555 
556  h->hres = h->pEnumDiscRecorders->Next(1, &pDiscRecorder, &fetched);
557  if (FAILED(h->hres)) {
558  return NULL;
559  }
560 
561  h->hres = pDiscRecorder->GetDisplayNames(&bVendor, &bProduct, &bRevision);
562 
563  pDiscRecorder->Release();
564 
565  if (FAILED(h->hres)) {
566  return NULL;
567  }
568 
569  len = SysStringLen(bVendor) + 1 +
570  SysStringLen(bProduct) + 1 +
571  SysStringLen(bRevision);
572 
573  wname = (LPWSTR)alloca((len + 1) * sizeof(TCHAR));
574 
575  _stprintf_s(wname,
576  (len + 1),
577  _T("%s %s %s"),
578  (LPCTSTR) bVendor,
579  (LPCTSTR) bProduct,
580  (LPCTSTR) bRevision);
581 
582  SysFreeString(bVendor);
583  SysFreeString(bProduct);
584  SysFreeString(bRevision);
585 
586  name = (char *) malloc(len + 1);
587  if (name == NULL) {
588  return NULL;
589  }
590 
591  name[0] = '\0';
592  WideCharToMultiByte(CP_ACP, 0, wname, len, name, len, NULL, NULL);

595  return name;

602 {

604  ULONG fetched;
605 
606  if (h == NULL) {

610  h->hres = S_OK;
611 
612  if (h->pDiscRecorder != NULL) {
613  h->hres = E_POINTER;

615  }
616 
617  if (index < 0 || index >= PortBurn_v1_GetNumDevices(h)) {
618  h->hres = E_INVALIDARG;
