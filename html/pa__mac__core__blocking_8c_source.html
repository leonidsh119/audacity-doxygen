77 static size_t computeSampleSizeFromFormat( PaSampleFormat format )
78 {
79  switch( format & (~paNonInterleaved) ) {
80  case paFloat32: return 4;
81  case paInt32: return 4;
82  case paInt24: return 3;
83  case paInt16: return 2;
84  case paInt8: case paUInt8: return 1;
85  default: return 0;
86  }
87 }

97  case paInt24: return 4;
98  case paInt16: return 2;
99  case paInt8: case paUInt8: return 1;

125 
126  /* zeroify things */

128  /* this is redundant, but the buffers are used to check
129  if the bufffers have been initialized, so we do it explicitly. */
130  blio->inputRingBuffer.buffer = NULL;
131  blio->outputRingBuffer.buffer = NULL;
132

134  blio->ringBufferFrames = ringBufferSize;
135  blio->inputSampleFormat = inputSampleFormat;

152  /* setup ring buffers */
153 #ifdef PA_MAC_BLIO_MUTEX
154  result = PaMacCore_SetUnixError( pthread_mutex_init(&(blio->inputMutex),NULL), 0 );
155  if( result )
156  goto error;
157  result = UNIX_ERR( pthread_cond_init( &(blio->inputCond), NULL ) );

190  data );
191  assert( !err );
192  }
193 
194  result = resetBlioRingBuffers( blio );

336 /*
337  * this is the BlioCallback function. It expects to recieve a PaMacBlio Object
338  * pointer as userData.
339  *
340  */
341 int BlioCallback( const void *input, void *output, unsigned long frameCount,
342  const PaStreamCallbackTimeInfo* timeInfo,
343  PaStreamCallbackFlags statusFlags,
344  void *userData )
345 {
346  PaMacBlio *blio = (PaMacBlio*)userData;
347  long avail;
348  long toRead;
349  long toWrite;
350  long read;
351  long written;
352 
353  /* set flags returned by OS: */
354  OSAtomicOr32( statusFlags, &blio->statusFlags ) ;
355 
356  /* --- Handle Input Buffer --- */
357  if( blio->inChan ) {
358  avail = PaUtil_GetRingBufferWriteAvailable( &blio->inputRingBuffer );
359 
360  /* check for underflow */
361  if( avail < frameCount * blio->inputSampleSizeActual * blio->inChan )
362  {
363  OSAtomicOr32( paInputOverflow, &blio->statusFlags );
364  }
365  toRead = MIN( avail, frameCount * blio->inputSampleSizeActual * blio->inChan );
366 
367  /* copy the data */
368  /*printf( "reading %d\n", toRead );*/
369  read = PaUtil_WriteRingBuffer( &blio->inputRingBuffer, input, toRead );
370  assert( toRead == read );
371 #ifdef PA_MAC__BLIO_MUTEX
372  /* Priority inversion. See notes below. */
373  blioSetIsInputEmpty( blio, false );
374 #endif
375  }
376 
377 
378  /* --- Handle Output Buffer --- */
379  if( blio->outChan ) {
380  avail = PaUtil_GetRingBufferReadAvailable( &blio->outputRingBuffer );
381 
382  /* check for underflow */
383  if( avail < frameCount * blio->outputSampleSizeActual * blio->outChan )
384  OSAtomicOr32( paOutputUnderflow, &blio->statusFlags );
385 
386  toWrite = MIN( avail, frameCount * blio->outputSampleSizeActual * blio->outChan );
387 
388  if( toWrite != frameCount * blio->outputSampleSizeActual * blio->outChan )
389  bzero( ((char *)output)+toWrite,
390  frameCount * blio->outputSampleSizeActual * blio->outChan - toWrite );
391  /* copy the data */
392  /*printf( "writing %d\n", toWrite );*/
393  written = PaUtil_ReadRingBuffer( &blio->outputRingBuffer, output, toWrite );
394  assert( toWrite == written );
395 #ifdef PA_MAC__BLIO_MUTEX
396  /* We have a priority inversion here. However, we will only have to
397  wait if this was true and is now false, which means we've got
398  some room in the buffer.
399  Hopefully problems will be minimized. */
400  blioSetIsOutputFull( blio, false );
401 #endif
402  }
403 
404  return paContinue;
405 }

407 PaError ReadStream( PaStream* stream,
408  void *buffer,
409  unsigned long frames )
410 {
411  PaMacBlio *blio = & ((PaMacCoreStream*)stream) -> blio;
412  char *cbuf = (char *) buffer;
413  PaError ret = paNoError;
414  VVDBUG(("ReadStream()\n"));
415 
416  while( frames > 0 ) {
417  long avail;
418  long toRead;
419  do {
420  avail = PaUtil_GetRingBufferReadAvailable( &blio->inputRingBuffer );
421 /*
422  printf( "Read Buffer is %%%g full: %ld of %ld.\n",
423  100 * (float)avail / (float) blio->inputRingBuffer.bufferSize,
424  avail, blio->inputRingBuffer.bufferSize );
425 */
426  if( avail == 0 ) {
427 #ifdef PA_MAC_BLIO_MUTEX
428 
429  ret = UNIX_ERR( pthread_mutex_lock( &blio->inputMutex ) );
430  if( ret )
431  return ret;
432  while( blio->isInputEmpty ) {
433  ret = UNIX_ERR( pthread_cond_wait( &blio->inputCond, &blio->inputMutex ) );
434  if( ret )
435  return ret;
436  }
437  ret = UNIX_ERR( pthread_mutex_unlock( &blio->inputMutex ) );
438  if( ret )
439  return ret;
440 #else
441  Pa_Sleep( PA_MAC_BLIO_BUSY_WAIT_SLEEP_INTERVAL );
442 #endif
443  }
444  } while( avail == 0 );
445  toRead = MIN( avail, frames * blio->inputSampleSizeActual * blio->inChan );
446  toRead -= toRead % blio->inputSampleSizeActual * blio->inChan ;
447  PaUtil_ReadRingBuffer( &blio->inputRingBuffer, (void *)cbuf, toRead );
448  cbuf += toRead;
449  frames -= toRead / ( blio->inputSampleSizeActual * blio->inChan );
450 
451  if( toRead == avail ) {
452 #ifdef PA_MAC_BLIO_MUTEX
453  /* we just emptied the buffer, so we need to mark it as empty. */
454  ret = blioSetIsInputEmpty( blio, true );
455  if( ret )
456  return ret;
457  /* of course, in the meantime, the callback may have put some sats
458  in, so
459  so check for that, too, to avoid a race condition. */
460  if( PaUtil_GetRingBufferReadAvailable( &blio->inputRingBuffer ) ) {
461  blioSetIsInputEmpty( blio, false );
462  if( ret )
463  return ret;
464  }
465 #endif
466  }
467  }
468 
469  /* Report either paNoError or paInputOverflowed. */
470  /* may also want to report other errors, but this is non-standard. */
471  ret = blio->statusFlags & paInputOverflow;
472 
473  /* report underflow only once: */
474  if( ret ) {
475  OSAtomicAnd32( (uint32_t)(~paInputOverflow), &blio->statusFlags );
476  ret = paInputOverflowed;
477  }
478 
479  return ret;
480 }

528 
529 #ifdef PA_MAC_BLIO_MUTEX

531  /* we just filled up the buffer, so we need to mark it as filled. */
532  ret = blioSetIsOutputFull( blio, true );

538  blioSetIsOutputFull( blio, false );
539  if( ret )
540  return ret;
541  }
542  }
543 #endif
544  }
545 
546  /* Report either paNoError or paOutputUnderflowed. */
547  /* may also want to report other errors, but this is non-standard. */
548  ret = blio->statusFlags & paOutputUnderflow;
549 
550  /* report underflow only once: */
551  if( ret ) {
552  OSAtomicAnd32( (uint32_t)(~paOutputUnderflow), &blio->statusFlags );
553  ret = paOutputUnderflowed;
