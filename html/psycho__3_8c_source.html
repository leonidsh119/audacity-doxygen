40  This implementation is not built for speed, but is rather meant to

45 static inline FLOAT psycho_3_add_db(psycho_3_mem * mem, FLOAT a, FLOAT b)
46 {
47  /* MFC - if the difference between a and b is large (>99), then just return the largest one.
48  (about 10% of the time) - For differences between 0 and 99, return the largest value, but
49  add in a pre-calculated difference value. - the value 99 was chosen arbitarily. - maximum
50  (a-b) i've seen is 572 */
51  FLOAT fdiff;
52  int idiff;
53  fdiff = (10.0 * (a - b));
54 
55  if (fdiff > 990.0) {
56  return a;
57  }
58  if (fdiff < -990.0) {
59  return (b);
60  }
61 
62  idiff = (int) fdiff;
63  if (idiff >= 0) {
64  return (a + mem->dbtable[idiff]);
65  }
66 
67  return (b + mem->dbtable[-idiff]);
68 }

72 /* ISO11172 Sec D.1 Step 1 - Window with HANN and then perform the FFT */
73 static void psycho_3_fft(FLOAT sample[BLKSIZE], FLOAT energy[BLKSIZE])
74 {
75  FLOAT x_real[BLKSIZE];
76  int i;
77  static int init = 0;
78  static FLOAT window[FFT_SIZE];
79 
80  if (!init) { /* calculate window function for the Fourier transform */
81  FLOAT sqrt_8_over_3 = pow(8.0 / 3.0, 0.5);
82  for (i = 0; i < BLKSIZE; i++) {
83  window[i] = sqrt_8_over_3 * 0.5 * (1 - cos(2.0 * PI * i / (BLKSIZE))) / BLKSIZE;
84  }
85  init++;
86  }
87 
88  /* convolve the samples with the hann window */
89  for (i = 0; i < BLKSIZE; i++)
90  x_real[i] = (FLOAT) (sample[i] * window[i]);
91  /* do the FFT */
92  psycho_1_fft(x_real, energy, BLKSIZE);
93 }

134 /* Sect D.1 Step4b
135  A tone within the range (start -> end), must be 7.0 dB greater than
136  all it's neighbours within +/- srange. Don't count its immediate neighbours. */
137 static void psycho_3_tonal_label_range(psycho_3_mem * mem, FLOAT * power, int *tonelabel,
138  int *maxima, FLOAT * Xtm, int start, int end, int srange)
139 {
140  int j, k;
141 
142  for (k = start; k < end; k++) /* Search for all the maxima in this range */
143  if (maxima[k] == 1) {
144  tonelabel[k] = TONE; /* assume it's a TONE and then prove otherwise */
145  for (j = -srange; j <= +srange; j++) /* Check the neighbours within +/- srange */
146  if (abs(j) > 1) /* Don't count the immediate neighbours, or itself */
147  if ((power[k] - power[k + j]) < 7.0)
148  tonelabel[k] = 0; /* Not greater by 7dB, therefore not a tone */
149  if (tonelabel[k] == TONE) {
150  /* Calculate the sound pressure level for this tone by summing the adjacent
151  spectral lines Xtm[k] = 10 * log10( pow(10.0, 0.1*power[k-1]) + pow(10.0,
152  0.1*power[k]) + pow(10.0, 0.1*power[k+1]) ); */
153  FLOAT temp = psycho_3_add_db(mem, power[k - 1], power[k]);
154  Xtm[k] = psycho_3_add_db(mem, temp, power[k + 1]);
155 
156  /* *ALL* spectral lines within +/- srange are set to -inf dB So that when we do the
157  noise calculate, they are not counted */
158  for (j = -srange; j <= +srange; j++)
159  power[k + j] = DBMIN;
160  }
161  }
162 }

167  FLOAT Xtm[HBLKSIZE])

170  int maxima[HBLKSIZE];

174  tonelabel[0] = tonelabel[HBLKSIZE - 1] = 0;

190  neighbours are immediately set to -inf dB */

213 /* D.1 Step 4.c Labelling non-tonal (noise) components
214  Sum the energies in each critical band (the tone energies have been removed
215  during the tone labelling).
216  Find the "geometric mean" of these energies - i.e. find the best spot to put the
217  sum of energies within this critical band. */
218 static void psycho_3_noise_label(psycho_3_mem * mem, FLOAT power[HBLKSIZE], FLOAT energy[BLKSIZE],
219  int *tonelabel, int *noiselabel, FLOAT Xnm[HBLKSIZE])
220 {
221  int i, j;
222  int cbands = mem->cbands;
223  int *cbandindex = mem->cbandindex;
224 
225  Xnm[0] = DBMIN;
226  for (i = 0; i < cbands; i++) {
227  /* for each critical band */
228  FLOAT sum = DBMIN;
229  FLOAT esum = 0;
230  FLOAT centreweight = 0;
231  int centre;
232  for (j = cbandindex[i]; j < cbandindex[i + 1]; j++) {
233  Xnm[j] = DBMIN;
234  /* go through all the spectral lines within the critical band, adding the energies. The
235  tone energies have already been removed */
236  if (power[j] != DBMIN) {
237  /* Found a noise energy, add it to the sum */
238  sum = psycho_3_add_db(mem, power[j], sum);
239 
240  /* calculations for the geometric mean FIXME MFC Feb 2003: Would it just be easier
241  to do the *whole* of psycho_1 in the energy domain rather than in the dB domain?
242  FIXME: This is just a lazy arsed arithmetic mean. Don't know if it's really going
243  to make that much difference */
244  esum += energy[j]; /* Calculate the sum of energies */
245  centreweight += (j - cbandindex[i]) * energy[j]; /* And the energy moment */
246  }
247  }
248 
249  /* MEANX, crash on AMD64 without this hack. See
250  https://sourceforge.net/tracker/?func=detail&atid=735435&aid=1453400&group_id=136040
251  Probably a better way to do this */
252  if (sum <= DBMIN || esum < 0.00001)
253  /* If the energy sum is really small, just pretend the noise occurs in the centre
254  frequency line */
255  centre = (cbandindex[i] + cbandindex[i + 1]) / 2;
256  else {
257  /* Otherwise, work out the mean position of the noise, and put it there. */
258  centre = cbandindex[i] + (int) (centreweight / esum);
259  }
260  // /MEANX
261  Xnm[centre] = sum;
262  noiselabel[centre] = NOISE;
263  }
264 }

297 /* ISO11172 Sect D.1 Step 6
298  Calculation of individual masking thresholds
299  Work out how each of the tones&noises maskes other frequencies
300  NOTE: Only a subset of other frequencies is checked. According to the
301  standard different subbands are subsampled to different amounts.
302  See psycho_3_init and freq_subset */
303 static void psycho_3_threshold(psycho_3_mem * mem, FLOAT * LTg, int *tonelabel, FLOAT * Xtm,
304  int *noiselabel, FLOAT * Xnm, FLOAT * bark, FLOAT * ath,
305  int bit_rate, int *freq_subset)
306 {
307  int i, j, k;
308  FLOAT LTtm[SUBSIZE];
309  FLOAT LTnm[SUBSIZE];
310 
311  for (i = 0; i < SUBSIZE; i++) {
312  LTtm[i] = DBMIN;
313  LTnm[i] = DBMIN;
314  }
315  /* Loop over the entire spectrum and find every noise and tone And then with each noise/tone
316  work out how it masks the spectral lines around it */
317  for (k = 1; k < HBLKSIZE; k++) {
318  /* Find every tone */
319  if (tonelabel[k] == TONE) {
320  for (j = 0; j < SUBSIZE; j++) {
321  /* figure out how it masks the levels around it */
322  FLOAT dz = bark[freq_subset[j]] - bark[k];
323  if (dz >= -3.0 && dz < 8.0) {
324  FLOAT vf;
325  FLOAT av = -1.525 - 0.275 * bark[k] - 4.5 + Xtm[k];
326  /* masking function for lower & upper slopes */
327  if (dz < -1)
328  vf = 17 * (dz + 1) - (0.4 * Xtm[k] + 6);
329  else if (dz < 0)
330  vf = (0.4 * Xtm[k] + 6) * dz;
331  else if (dz < 1)
332  vf = (-17 * dz);
333  else
334  vf = -(dz - 1) * (17 - 0.15 * Xtm[k]) - 17;
335  LTtm[j] = psycho_3_add_db(mem, LTtm[j], av + vf);
336  }
337  }
338  }
339 
340  /* find every noise label */
341  if (noiselabel[k] == NOISE) {
342  for (j = 0; j < SUBSIZE; j++) {
343  /* figure out how it masks the levels around it */
344  FLOAT dz = bark[freq_subset[j]] - bark[k];
345  if (dz >= -3.0 && dz < 8.0) {
346  FLOAT vf;
347  FLOAT av = -1.525 - 0.175 * bark[k] - 0.5 + Xnm[k];
348  /* masking function for lower & upper slopes */
349  if (dz < -1)
350  vf = 17 * (dz + 1) - (0.4 * Xnm[k] + 6);
351  else if (dz < 0)
352  vf = (0.4 * Xnm[k] + 6) * dz;
353  else if (dz < 1)
354  vf = (-17 * dz);
355  else
356  vf = -(dz - 1) * (17 - 0.15 * Xnm[k]) - 17;
357  LTnm[j] = psycho_3_add_db(mem, LTnm[j], av + vf);
358  }
359  }
360  }
361  }
362 
363  /* ISO11172 D.1 Step 7 Calculate the global masking threhold */
364  for (i = 0; i < SUBSIZE; i++) {
365  LTg[i] = psycho_3_add_db(mem, LTnm[i], LTtm[i]);
366  if (bit_rate < 96)
367  LTg[i] = psycho_3_add_db(mem, ath[freq_subset[i]], LTg[i]);
368  else
369  LTg[i] = psycho_3_add_db(mem, ath[freq_subset[i]] - 12.0, LTg[i]);
370  }
371 }

404 static psycho_3_mem *psycho_3_init(twolame_options * glopts)
405 {
406  int i;
407  int cbase = 0; /* current base index for the bark range calculation */
408  FLOAT sfreq;
409  psycho_3_mem *mem;
410  int numlines[HBLKSIZE];
411  FLOAT cbval[HBLKSIZE];
412  int partition[HBLKSIZE];
413  int *freq_subset;
414  FLOAT *bark, *ath;
415  int cbands = 0;
416  int *cbandindex;
417 
418  mem = (psycho_3_mem *) TWOLAME_MALLOC(sizeof(psycho_3_mem));
419  mem->off[0] = mem->off[1] = 256;
420  freq_subset = mem->freq_subset;
421  bark = mem->bark;
422  ath = mem->ath;
423  cbandindex = mem->cbandindex;
424 
425  /* Initialise the tables for the adding dB */
426  psycho_3_init_add_db(mem);
427 
428  /* For each spectral line calculate the bark and the ATH (in dB) */
429  sfreq = (FLOAT) glopts->samplerate_out;
430  for (i = 1; i < HBLKSIZE; i++) {
431  FLOAT freq = i * sfreq / BLKSIZE;
432  bark[i] = ath_freq2bark(freq);
433  ath[i] = ath_db(freq, glopts->athlevel);
434  }
435 
436  { /* Work out the critical bands Starting from line 0, all lines
437  within 1 bark of the starting bark are added to the same
438  critical band. When a line is greater by 1.0 of a bark, start a
439  new critical band. */
440 
441  cbandindex[0] = 1;
442  for (i = 1; i < HBLKSIZE; i++) {
443  if ((bark[i] - bark[cbase]) > 1.0) { /* 1 critical band? 1 bark? */
444  /* this frequency line is too different from the starting line, (in terms of the
445  bark distance) so make this spectral line the first member of the next critical
446  band */
447  cbase = i; /* Start the new critical band from this frequency line */
448  cbands++;
449  cbandindex[cbands] = cbase;
450  }
451  /* partition[i] tells us which critical band the i'th frequency line is in */
452  partition[i] = cbands;
453  /* keep a count of how many frequency lines are in each partition */
454  numlines[cbands]++;
455  }
456 
457  cbands++;
458  cbandindex[cbands] = 513; /* Set the top of the last critical band */
459  mem->cbands = cbands; // make a not of the number of cbands
460 
461  /* For each crtical band calculate the average bark value cbval [central bark value] */
462  for (i = 1; i < HBLKSIZE; i++)
463  cbval[partition[i]] += bark[i]; /* sum up all the bark values */
464  for (i = 1; i < CBANDS; i++) {
465  if (numlines[i] != 0)
466  cbval[i] /= numlines[i]; /* divide by the number of values */
467  else {
468  cbval[i] = 0; /* this isn't a partition */
469  }
470  }
471  }
472 
473  {
474  /* For Step6 - For the calculation of individual masking thresholds the spectral lines are
475  subsampled i.e. no need to work out the masking for every single spectral line.
476  Depending upon which subband the calculation is for, you can skip a number of lines
477  There are 16 lines per subband -> 32 * 16 = 512 Subband 0-2 : Every line (3 * 16 = 48
478  lines) Subband 3-5 : Every Second line (3 * 16/2 = 24 lines) Subband 6-11 : Every 4th
479  line (6 * 16/4 = 24 lines) Subband 12-31 : Every 12th line (20 * 16/8 = 40 lines)
480 
481  create this subset of frequencies (freq_subset) */
482  int freq_index = 0;
483  for (i = 1; i < (3 * 16) + 1; i++)
484  freq_subset[freq_index++] = i;
485  for (; i < (6 * 16) + 1; i += 2)
486  freq_subset[freq_index++] = i;
487  for (; i < (12 * 16) + 1; i += 4)
488  freq_subset[freq_index++] = i;
489  for (; i < (32 * 16) + 1; i += 8)
490  freq_subset[freq_index++] = i;
491  }
492 
493  if (glopts->verbosity > 4) {
494  fprintf(stderr, "%i critical bands\n", cbands);
495  for (i = 0; i < cbands; i++)
496  fprintf(stderr, "cband %i spectral line index %i\n", i, cbandindex[i]);
497  fprintf(stderr, "%i Subsampled spectral lines\n", SUBSIZE);
498  for (i = 0; i < SUBSIZE; i++)
499  fprintf(stderr, "%i Spectral line %i Bark %.2f\n", i, freq_subset[i],
500  bark[freq_subset[i]]);
501  }
502 
503  return (mem);
504 }
