94 void edit_transcription(Alg_seq &seq , bool warp, FILE *outf,
95  const char *midi_filename, const char *beat_filename) {
96  int note_x = 1;
97  seq.convert_to_seconds();
98  Alg_iterator iter(&seq, false); // no note-offs
99  iter.begin();
100  Alg_event_ptr e = iter.next();
101 
102  while (e) {
103  if (e->is_note()) {
104  Alg_note_ptr n = (Alg_note_ptr) e;
105  fprintf(outf, "%d %ld %d %d ",
106  note_x++, n->chan, ROUND(n->pitch), ROUND(n->loud));
107  // now compute onset time mapped to audio time
108  double start = n->time;
109  double finish = n->time + n->dur;
110  if (warp) {
111  start = sa.map_time(start);
112  finish = sa.map_time(finish);
113  }
114  fprintf(outf, "%.3f %.3f\n", start, finish-start);
115  }
116  e = iter.next();
117  }
118  iter.end();
119  fclose(outf);
120  if (warp) {
121  // align the midi file and write out
122  sa.midi_tempo_align(seq);
123  seq.smf_write(midi_filename);
124  print_beat_map(seq, beat_filename);
125  }

162 assume audio file is file 1 and midi file is file 2

164 
165 If warp is false, simply write a transcription of the midi file.
166 
167 Every note has 6 fields separated by a space character. The fields are:
168 <sequence number> <channel> <pitch> <velocity> <onset> <duration>

170  <sequence number> is just an integer note number, e.g. 1, 2, 3, ...
171  <channel> is MIDI channel from 0 to 15
172  <pitch> is MIDI key number (60 = middle C)
