36 long pluck_parameters(double hz, double sr, double final, double dur,
37  double *stretch, double *cons, double *rho)
38 {
39  double t = PI * (hz / sr);
40  double y = fabs(cos(t));
41  /* original m4c code used ratio of initial amp to final amp in dB
42  and then converted to a ratio, e.g. you specify 60 and the
43  parameter Final is 1000.0. This is counterintuitive to me (RBD)
44  because I would expect the value to be -60dB or 0.001. That is
45  what I implemented, so to get this back into correspondence
46  with the m4c algorithm, I take the NEGATIVE log to get lf,
47  whereas m4c takes the positive log:
48  */
49  double lf = -log(final);
50  double tdecay = -lf / (hz * log(y));
51  double st;
52  long len;
53  double x;
54 
55  if (hz <= sr / MAXLENGTH) {
56  xlfail("pluck hz is too low");
57  } else if (hz >= sr / 3) {
58  xlfail("pluck hz is too high");
59  }
60  /*
61  * if desired decay time is shorter than the natural decay time,
62  * then introduce a loss factor. Otherwise, stretch note out.
63  */
64  st = hz * dur;
65  if (dur < tdecay) {
66  *rho = exp(-lf / st) / y;
67  *stretch = 0.5;
68  } else {
69  *rho = 1;
70  *stretch = 0.5 + sqrt(0.25 -
71  (1 - exp(2 * lf * (hz - sr) / (st * sr))) /
72  (2 - 2 * cos(2 * t)));
73  }
74 
75  /* delay line length is */
76  len = (int) ((sr / hz) - *stretch - 0.001);
77 
78  /* tuning constant is */
79  x = (sr / hz) - len - *stretch;
80  *cons = (1.0 - x) / (1.0 + x);
81 
82  if (len <= 1) {
83  xlfail("internal error: pluck delay line length too short");
84  }
85  return len;
86 }
