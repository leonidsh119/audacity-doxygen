725 bool EffectEqualization48x::ProcessBuffer4x(BufferInfo *bufferInfo)
726 {
727  // length must be a factor of window size for 4x processing.
728  if(bufferInfo->mBufferLength%mBlockSize)
729  return false;
730 
731  sampleCount blockCount=bufferInfo->mBufferLength/mBlockSize;
732 
733  __m128 *readBlocks[4]; // some temps so we dont destroy the vars in the struct
734  __m128 *writeBlocks[4];
735  for(int i=0;i<4;i++) {
736  readBlocks[i]=(__m128 *)bufferInfo->mBufferSouce[i];
737  writeBlocks[i]=(__m128 *)bufferInfo->mBufferDest[i];
738  }
739 
740  __m128 *swizzledBuffer128=(__m128 *)bufferInfo->mScratchBuffer;
741  __m128 *scratchBuffer=&swizzledBuffer128[mWindowSize*2];
742 
743  for(int run4x=0;run4x<blockCount;run4x++)
744  {
745  // swizzle the data to the swizzle buffer
746  __m128 *currentSwizzledBlock=&swizzledBuffer128[mWindowSize*(run4x&1)];
747  for(int i=0,j=0;j<mBlockSize;i++,j+=4) {
748  __m128 tmp0 = _mm_shuffle_ps(readBlocks[0][i], readBlocks[1][i], _MM_SHUFFLE(1,0,1,0));
749  __m128 tmp1 = _mm_shuffle_ps(readBlocks[0][i], readBlocks[1][i], _MM_SHUFFLE(3,2,3,2));
750  __m128 tmp2 = _mm_shuffle_ps(readBlocks[2][i], readBlocks[3][i], _MM_SHUFFLE(1,0,1,0));
751  __m128 tmp3 = _mm_shuffle_ps(readBlocks[2][i], readBlocks[3][i], _MM_SHUFFLE(3,2,3,2));
752  currentSwizzledBlock[j] = _mm_shuffle_ps(tmp0, tmp2, _MM_SHUFFLE(2,0,2,0));
753  currentSwizzledBlock[j+1] = _mm_shuffle_ps(tmp0, tmp2, _MM_SHUFFLE(3,1,3,1));
754  currentSwizzledBlock[j+2] = _mm_shuffle_ps(tmp1, tmp3, _MM_SHUFFLE(2,0,2,0));
755  currentSwizzledBlock[j+3] = _mm_shuffle_ps(tmp1, tmp3, _MM_SHUFFLE(3,1,3,1));
756  }
757  __m128 *thisOverrun128=&currentSwizzledBlock[mBlockSize];
758  for(int i=0;i<mFilterSize;i++)
759  thisOverrun128[i]=_mm_set1_ps(0.0);
760  Filter4x(mWindowSize, (float *)currentSwizzledBlock, (float *)scratchBuffer);
761  int writeStart=0, writeToStart=0; // note readStart is where the read data is written
762  int writeEnd=mBlockSize;
763  if(run4x) {
764  // maybe later swizzle add and write in one
765  __m128 *lastOverrun128=&swizzledBuffer128[mWindowSize*((run4x+1)&1)+mBlockSize];
766  // add and swizzle data + filter
767  for(int i=0,j=0;j<mFilterSize;i++,j+=4) {
768  __m128 tmps0 = _mm_add_ps(currentSwizzledBlock[j], lastOverrun128[j]);
769  __m128 tmps1 = _mm_add_ps(currentSwizzledBlock[j+1], lastOverrun128[j+1]);
770  __m128 tmps2 = _mm_add_ps(currentSwizzledBlock[j+2], lastOverrun128[j+2]);
771  __m128 tmps3 = _mm_add_ps(currentSwizzledBlock[j+3], lastOverrun128[j+3]);
772  __m128 tmp0 = _mm_shuffle_ps(tmps1, tmps0, _MM_SHUFFLE(0,1,0,1));
773  __m128 tmp1 = _mm_shuffle_ps(tmps1, tmps0, _MM_SHUFFLE(2,3,2,3));
774  __m128 tmp2 = _mm_shuffle_ps(tmps3, tmps2, _MM_SHUFFLE(0,1,0,1));
775  __m128 tmp3 = _mm_shuffle_ps(tmps3, tmps2, _MM_SHUFFLE(2,3,2,3));
776  writeBlocks[0][i] = _mm_shuffle_ps(tmp0, tmp2, _MM_SHUFFLE(1,3,1,3));
777  writeBlocks[1][i] = _mm_shuffle_ps(tmp0, tmp2, _MM_SHUFFLE(0,2,0,2));
778  writeBlocks[2][i] = _mm_shuffle_ps(tmp1, tmp3, _MM_SHUFFLE(1,3,1,3));
779  writeBlocks[3][i] = _mm_shuffle_ps(tmp1, tmp3, _MM_SHUFFLE(0,2,0,2));
780  }
781  writeStart=mFilterSize;
782  writeToStart=mFilterSize>>2;
783  // swizzle it back.
784  for(int i=writeToStart,j=writeStart;j<writeEnd;i++,j+=4) {
785  __m128 tmp0 = _mm_shuffle_ps(currentSwizzledBlock[j+1], currentSwizzledBlock[j], _MM_SHUFFLE(0,1,0,1));
786  __m128 tmp1 = _mm_shuffle_ps(currentSwizzledBlock[j+1], currentSwizzledBlock[j], _MM_SHUFFLE(2,3,2,3));
787  __m128 tmp2 = _mm_shuffle_ps(currentSwizzledBlock[j+3], currentSwizzledBlock[j+2], _MM_SHUFFLE(0,1,0,1));
788  __m128 tmp3 = _mm_shuffle_ps(currentSwizzledBlock[j+3], currentSwizzledBlock[j+2], _MM_SHUFFLE(2,3,2,3));
789  writeBlocks[0][i] = _mm_shuffle_ps(tmp0, tmp2, _MM_SHUFFLE(1,3,1,3));
790  writeBlocks[1][i] = _mm_shuffle_ps(tmp0, tmp2, _MM_SHUFFLE(0,2,0,2));
791  writeBlocks[2][i] = _mm_shuffle_ps(tmp1, tmp3, _MM_SHUFFLE(1,3,1,3));
792  writeBlocks[3][i] = _mm_shuffle_ps(tmp1, tmp3, _MM_SHUFFLE(0,2,0,2));
793  }
794  } else {
795  // swizzle it back. We overlap one block so we only write the first block on the first run
796  writeStart=0;
797  writeToStart=0;
798  for(int i=writeToStart,j=writeStart;j<writeEnd;i++,j+=4) {
799  __m128 tmp0 = _mm_shuffle_ps(currentSwizzledBlock[j+1], currentSwizzledBlock[j], _MM_SHUFFLE(0,1,0,1));
800  __m128 tmp2 = _mm_shuffle_ps(currentSwizzledBlock[j+3], currentSwizzledBlock[j+2], _MM_SHUFFLE(0,1,0,1));
801  writeBlocks[0][i] = _mm_shuffle_ps(tmp0, tmp2, _MM_SHUFFLE(1,3,1,3));
802  }
803  }
804  for(int i=0;i<4;i++) { // shift each block
805  readBlocks[i]+=mBlockSize>>2; // these are 128b pointers, each window is 1/4 blockSize for those
806  writeBlocks[i]+=mBlockSize>>2;
807  }
808  }
809  return true;
810 }
