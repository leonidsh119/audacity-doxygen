202 // See http://www.w3.org/TR/REC-xml for reference
203 wxString XMLWriter::XMLEsc(const wxString & s)
204 {
205  wxString result;
206  int len = s.Length();
207 
208  for(int i=0; i<len; i++) {
209  wxUChar c = s.GetChar(i);
210 
211  switch (c) {
212  case wxT('\''):
213  result += wxT("&apos;");
214  break;
215 
216  case wxT('"'):
217  result += wxT("&quot;");
218  break;
219 
220  case wxT('&'):
221  result += wxT("&amp;");
222  break;
223 
224  case wxT('<'):
225  result += wxT("&lt;");
226  break;
227 
228  case wxT('>'):
229  result += wxT("&gt;");
230  break;
231 
232  default:
233  if (!wxIsprint(c)) {
234  //ignore several characters such ase eot (0x04) and stx (0x02) because it makes expat parser bail
235  //see xmltok.c in expat checkCharRefNumber() to see how expat bails on these chars.
236  //also see wxWidgets-2.8.12/src/expat/lib/asciitab.h to see which characters are nonxml compatible
237  //post decode (we can still encode '&' and '<' with this table, but it prevents us from encoding eot)
238  //everything is compatible past ascii 0x20, so we don't check higher than this.
239  if(c> 0x1F || charXMLCompatiblity[c]!=0)
240  result += wxString::Format(wxT("&#x%04x;"), c);
241  }
242  else {
243  result += c;
244  }
245  break;
246  }
247  }
248 
249  return result;
250 }
