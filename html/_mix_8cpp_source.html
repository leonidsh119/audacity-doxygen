413 sampleCount Mixer::MixVariableRates(int *channelFlags, const WaveTrack *track,
414  sampleCount *pos, float *queue,
415  int *queueStart, int *queueLen,
416  Resample * pResample)
417 {
418  const double trackRate = track->GetRate();
419  const double initialWarp = mRate / mSpeed / trackRate;
420  const double tstep = 1.0 / trackRate;
421  int sampleSize = SAMPLE_SIZE(floatSample);
422 
423  sampleCount out = 0;
424 
425  /* time is floating point. Sample rate is integer. The number of samples
426  * has to be integer, but the multiplication gives a float result, which we
427  * round to get an integer result. TODO: is this always right or can it be
428  * off by one sometimes? Can we not get this information directly from the
429  * clip (which must know) rather than convert the time?
430  *
431  * LLL: Not at this time. While WaveClips provide methods to retrieve the
432  * start and end sample, they do the same float->sampleCount conversion
433  * to calculate the position.
434  */
435 
436  // Find the last sample
437  double endTime = track->GetEndTime();
438  double startTime = track->GetStartTime();
439  const bool backwards = (mT1 < mT0);
440  const double tEnd = backwards
441  ? std::max(startTime, mT1)
442  : std::min(endTime, mT1);
443  const sampleCount endPos = track->TimeToLongSamples(tEnd);
444  // Find the time corresponding to the start of the queue, for use with time track
445  double t = (*pos + (backwards ? *queueLen : - *queueLen)) / trackRate;
446 
447  while (out < mMaxOut) {
448  if (*queueLen < mProcessLen) {
449  // Shift pending portion to start of the buffer
450  memmove(queue, &queue[*queueStart], (*queueLen) * sampleSize);
451  *queueStart = 0;
452 
453  int getLen =
454  std::min((backwards ? *pos - endPos : endPos - *pos),
455  sampleCount(mQueueMaxLen - *queueLen));
456 
457  // Nothing to do if past end of play interval
458  if (getLen > 0) {
459  if (backwards) {
460  track->Get((samplePtr)&queue[*queueLen],
461  floatSample,
462  *pos - (getLen - 1),
463  getLen);
464 
465  track->GetEnvelopeValues(mEnvValues,
466  getLen,
467  (*pos - (getLen- 1)) / trackRate,
468  tstep);
469 
470  *pos -= getLen;
471  }
472  else {
473  track->Get((samplePtr)&queue[*queueLen],
474  floatSample,
475  *pos,
476  getLen);
477 
478  track->GetEnvelopeValues(mEnvValues,
479  getLen,
480  (*pos) / trackRate,
481  tstep);
482 
483  *pos += getLen;
484  }
485 
486  for (int i = 0; i < getLen; i++) {
487  queue[(*queueLen) + i] *= mEnvValues[i];
488  }
489 
490  if (backwards)
491  ReverseSamples((samplePtr)&queue[0], floatSample,
492  *queueLen, getLen);
493 
494  *queueLen += getLen;
495  }
496  }
497 
498  sampleCount thisProcessLen = mProcessLen;
499  bool last = (*queueLen < mProcessLen);
500  if (last) {
501  thisProcessLen = *queueLen;
502  }
503 
504  double factor = initialWarp;
505  if (mTimeTrack)
506  {
507  //TODO-MB: The end time is wrong when the resampler doesn't use all input samples,
508  // as a result of this the warp factor may be slightly wrong, so AudioIO will stop too soon
509  // or too late (resulting in missing sound or inserted silence). This can't be fixed
510  // without changing the way the resampler works, because the number of input samples that will be used
511  // is unpredictable. Maybe it can be compensated later though.
512  if (backwards)
513  factor *= mTimeTrack->ComputeWarpFactor
514  (t - (double)thisProcessLen / trackRate + tstep, t + tstep);
515  else
516  factor *= mTimeTrack->ComputeWarpFactor
517  (t, t + (double)thisProcessLen / trackRate);
518  }
519 
520  int input_used;
521  int outgen = pResample->Process(factor,
522  &queue[*queueStart],
523  thisProcessLen,
524  last,
525  &input_used,
526  &mFloatBuffer[out],
527  mMaxOut - out);
528 
529  if (outgen < 0) {
530  return 0;
531  }
532 
533  *queueStart += input_used;
534  *queueLen -= input_used;
535  out += outgen;
536  t += ((backwards ? -input_used : input_used) / trackRate);
537 
538  if (last) {
539  break;
540  }
541  }
542 
543  for (int c = 0; c < mNumChannels; c++) {
544  if (mApplyTrackGains) {
545  mGains[c] = track->GetChannelGain(c);
546  }
547  else {
548  mGains[c] = 1.0;
549  }
550  }
551 
552  MixBuffers(mNumChannels,
553  channelFlags,
554  mGains,
555  (samplePtr)mFloatBuffer,
556  mTemp,
557  out,
558  mInterleaved);
559 
560  return out;
561 }
