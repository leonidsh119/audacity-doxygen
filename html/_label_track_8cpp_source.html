318 void LabelTrack::ComputeTextPosition(const wxRect & r, int index) const
319 {
320  // xExtra is extra space
321  // between the text and the endpoints.
322  const int xExtra=mIconWidth;
323  int x = mLabels[index]->x; // left endpoint
324  int x1 = mLabels[index]->x1; // right endpoint.
325  int width = mLabels[index]->width;
326 
327  int xText; // This is where the text will end up.
328 
329  // Will the text all fit at this zoom?
330  bool bTooWideForScreen = width > (r.width-2*xExtra);
331 // bool bSimpleCentering = !bTooWideForScreen;
332  bool bSimpleCentering = false;
333 
334  //TODO (possibly):
335  // Add configurable options as to when to use simple
336  // and when complex centering.
337  //
338  // Simple centering does its best to keep the text
339  // centered between the label limits.
340  //
341  // Complex centering positions the text proportionally
342  // to how far we are through the label.
343  //
344  // If we add preferences for this, we want to be able to
345  // choose separately whether:
346  // a) Wide text labels centered simple/complex.
347  // b) Other text labels centered simple/complex.
348  //
349 
350  if( bSimpleCentering )
351  {
352  // Center text between the two end points.
353  xText = (x+x1-width)/2;
354  }
355  else
356  {
357  // Calculate xText position to make text line
358  // scroll sideways evenly as r moves right.
359 
360  // xText is a linear function of r.x.
361  // These variables are used to compute that function.
362  int rx0,rx1,xText0,xText1;
363 
364  // Since we will be using a linear function,
365  // we should blend smoothly between left and right
366  // aligned text as r, the 'viewport' moves.
367  if( bTooWideForScreen )
368  {
369  rx0=x; // when viewport at label start.
370  xText0=x+xExtra; // text aligned left.
371  rx1=x1-r.width; // when viewport end at label end
372  xText1=x1-(width+xExtra); // text aligned right.
373  }
374  else
375  {
376  // when label start + width + extra spacing at viewport end..
377  rx0=x-r.width+width+2*xExtra;
378  // ..text aligned left.
379  xText0=x+xExtra;
380  // when viewport start + width + extra spacing at label end..
381  rx1=x1-(width+2*xExtra);
382  // ..text aligned right.
383  xText1=x1-(width+xExtra);
384  }
385 
386  if( rx1 > rx0 ) // Avoid divide by zero case.
387  {
388  // Compute the blend between left and right aligned.
389 
390  // Don't use:
391  //
392  // xText = xText0 + ((xText1-xText0)*(r.x-rx0))/(rx1-rx0);
393  //
394  // The problem with the above is that it integer-oveflows at
395  // high zoom.
396 
397  // Instead use:
398  xText = xText0 + (int)((xText1-xText0)*(((float)(r.x-rx0))/(rx1-rx0)));
399  }
400  else
401  {
402  // Avoid divide by zero by reverting to
403  // simple centering.
404  //
405  // We could also fall into this case if x and x1
406  // are swapped, in which case we'll end up
407  // left aligned anyway because code later on
408  // will catch that.
409  xText = (x+x1-width)/2;
410  }
411  }
412 
413  // Is the text now appearing partly outside r?
414  bool bOffLeft = xText < r.x+xExtra;
415  bool bOffRight = xText > r.x+r.width-width-xExtra;
416 
417  // IF any part of the text is offscreen
418  // THEN we may bring it back.
419  if( bOffLeft == bOffRight )
420  {
421  //IF both sides on screen, THEN nothing to do.
422  //IF both sides off screen THEN don't do
423  //anything about it.
424  //(because if we did, you'd never get to read
425  //all the text by scrolling).
426  }
427  else if( bOffLeft != bTooWideForScreen)
428  {
429  // IF we're off on the left, OR we're
430  // too wide for the screen and off on the right
431  // (only) THEN align left.
432  xText = r.x+xExtra;
433  }
434  else
435  {
436  // We're off on the right, OR we're
437  // too wide and off on the left (only)
438  // SO align right.
439  xText =r.x+r.width-width-xExtra;
440  }
441 
442  // But if we've taken the text out from its endpoints
443  // we must move it back so that it's between the endpoints.
444 
445  // We test the left end point last because the
446  // text might not even fit between the endpoints (at this
447  // zoom factor), and in that case we'd like to position
448  // the text at the left end point.
449  if( xText > (x1-width-xExtra))
450  xText=(x1-width-xExtra);
451  if( xText < x+xExtra )
452  xText=x+xExtra;
453 
454  mLabels[index]->xText = xText;
455 }

460 void LabelTrack::ComputeLayout(const wxRect & r, const ZoomInfo &zoomInfo) const
461 {
462  int i;
463  int iRow;
464  // Rows are the 'same' height as icons or as the text,
465  // whichever is taller.
466  const int yRowHeight = wxMax(mTextHeight,mIconHeight)+3;// pixels.
467  // Extra space at end of rows.
468  // We allow space for one half icon at the start and two
469  // half icon widths for extra x for the text frame.
470  // [we don't allow half a width space for the end icon since it is
471  // allowed to be obscured by the text].
472  const int xExtra= (3 * mIconWidth)/2;
473 
474  const int nRows = wxMin((r.height / yRowHeight) + 1, MAX_NUM_ROWS);
475  // Initially none of the rows have been used.
476  // So set a value that is less than any valid value.
477  {
478  const int xStart = zoomInfo.TimeToPosition(0.0, r.x) - 100;
479  for(i=0;i<MAX_NUM_ROWS;i++)
480  xUsed[i]=xStart;
481  }
482  int nRowsUsed=0;
483 
484  for (i = 0; i < (int)mLabels.Count(); i++)
485  {
486  const int x = zoomInfo.TimeToPosition(mLabels[i]->getT0(), r.x);
487  const int x1 = zoomInfo.TimeToPosition(mLabels[i]->getT1(), r.x);
488  int y = r.y;
489 
490  mLabels[i]->x=x;
491  mLabels[i]->x1=x1;
492  mLabels[i]->y=-1;// -ve indicates nothing doing.
493  iRow=0;
494  // Our first preference is a row that ends where we start.
495  // (This is to encourage merging of adjacent label boundaries).
496  while( (iRow<nRowsUsed) && (xUsed[iRow] != x ))
497  iRow++;
498  // IF we didn't find one THEN
499  // find any row that can take a span starting at x.
500  if( iRow >= nRowsUsed )
501  {
502  iRow=0;
503  while( (iRow<nRows) && (xUsed[iRow] > x ))
504  iRow++;
505  }
506  // IF we found such a row THEN record a valid position.
507  if( iRow<nRows )
508  {
509  // Possibly update the number of rows actually used.
510  if( iRow >= nRowsUsed )
511  nRowsUsed=iRow+1;
512  // Record the position for this label
513  y= r.y + iRow * yRowHeight +(yRowHeight/2)+1;
514  mLabels[i]->y=y;
515  // On this row we have used up to max of end marker and width.
516  // Plus also allow space to show the start icon and
517  // some space for the text frame.
518  xUsed[iRow]=x+mLabels[i]->width+xExtra;
519  if( xUsed[iRow] < x1 ) xUsed[iRow]=x1;
520  ComputeTextPosition( r, i );
521  }
522  }
523 }
