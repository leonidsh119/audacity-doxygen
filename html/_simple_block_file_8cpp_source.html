161 bool SimpleBlockFile::WriteSimpleBlockFile(
162  samplePtr sampleData,
163  sampleCount sampleLen,
164  sampleFormat format,
165  void* summaryData)
166 {
167  wxFFile file(mFileName.GetFullPath(), wxT("wb"));
168  if( !file.IsOpened() ){
169  // Can't do anything else.
170  return false;
171  }
172 
173  auHeader header;
174 
175  // AU files can be either big or little endian. Which it is is
176  // determined implicitly by the byte-order of the magic 0x2e736e64
177  // (.snd). We want it to be native-endian, so we write the magic
178  // to memory and then let it write that to a file in native
179  // endianness
180  header.magic = 0x2e736e64;
181 
182  // We store the summary data at the end of the header, so the data
183  // offset is the length of the summary data plus the length of the header
184  header.dataOffset = sizeof(auHeader) + mSummaryInfo.totalSummaryBytes;
185 
186  // dataSize is optional, and we opt out
187  header.dataSize = 0xffffffff;
188 
189  switch(format) {
190  case int16Sample:
191  header.encoding = AU_SAMPLE_FORMAT_16;
192  break;
193 
194  case int24Sample:
195  header.encoding = AU_SAMPLE_FORMAT_24;
196  break;
197 
198  case floatSample:
199  header.encoding = AU_SAMPLE_FORMAT_FLOAT;
200  break;
201  }
202 
203  // TODO: don't fabricate
204  header.sampleRate = 44100;
205 
206  // BlockFiles are always mono
207  header.channels = 1;
208 
209  // Write the file
210  if (!summaryData)
211  summaryData = /*BlockFile::*/CalcSummary(sampleData, sampleLen, format); //mchinen:allowing virtual override of calc summary for ODDecodeBlockFile.
212 
213  size_t nBytesToWrite = sizeof(header);
214  size_t nBytesWritten = file.Write(&header, nBytesToWrite);
215  if (nBytesWritten != nBytesToWrite)
216  {
217  wxLogDebug(wxT("Wrote %lld bytes, expected %lld."), (long long) nBytesWritten, (long long) nBytesToWrite);
218  return false;
219  }
220 
221  nBytesToWrite = mSummaryInfo.totalSummaryBytes;
222  nBytesWritten = file.Write(summaryData, nBytesToWrite);
223  if (nBytesWritten != nBytesToWrite)
224  {
225  wxLogDebug(wxT("Wrote %lld bytes, expected %lld."), (long long) nBytesWritten, (long long) nBytesToWrite);
226  return false;
227  }
228 
229  if( format == int24Sample )
230  {
231  // we can't write the buffer directly to disk, because 24-bit samples
232  // on disk need to be packed, not padded to 32 bits like they are in
233  // memory
234  int *int24sampleData = (int*)sampleData;
235 
236  for( int i = 0; i < sampleLen; i++ )
237  {
238  nBytesToWrite = 3;
239  nBytesWritten =
240  #if wxBYTE_ORDER == wxBIG_ENDIAN
241  file.Write((char*)&int24sampleData[i] + 1, nBytesToWrite);
242  #else
243  file.Write((char*)&int24sampleData[i], nBytesToWrite);
244  #endif
245  if (nBytesWritten != nBytesToWrite)
246  {
247  wxLogDebug(wxT("Wrote %lld bytes, expected %lld."), (long long) nBytesWritten, (long long) nBytesToWrite);
248  return false;
249  }
250  }
251  }
252  else
253  {
254  // for all other sample formats we can write straight from the buffer
255  // to disk
256  nBytesToWrite = sampleLen * SAMPLE_SIZE(format);
257  nBytesWritten = file.Write(sampleData, nBytesToWrite);
258  if (nBytesWritten != nBytesToWrite)
259  {
260  wxLogDebug(wxT("Wrote %lld bytes, expected %lld."), (long long) nBytesWritten, (long long) nBytesToWrite);
261  return false;
262  }
263  }
264 
265  return true;
266 }

274  wxFFile file(mFileName.GetFullPath(), wxT("rb"));

386 int SimpleBlockFile::ReadData(samplePtr data, sampleFormat format,
387  sampleCount start, sampleCount len)
388 {
389  if (mCache.active)
390  {
391  //wxLogDebug("SimpleBlockFile::ReadData(): Data are already in cache.");
392 
393  if (len > mLen - start)
394  len = mLen - start;
395  CopySamples(
396  (samplePtr)(((char*)mCache.sampleData) +
397  start * SAMPLE_SIZE(mCache.format)),
398  mCache.format, data, format, len);
399  return len;
400  } else
401  {
402  //wxLogDebug("SimpleBlockFile::ReadData(): Reading data from disk.");
403 
404  SF_INFO info;
405  wxFile f; // will be closed when it goes out of scope
406  SNDFILE *sf = NULL;
407  {
408  Maybe<wxLogNull> silence{};
409  if (mSilentLog)
410  silence.create();
411 
412  memset(&info, 0, sizeof(info));
413 
414  if (f.Open(mFileName.GetFullPath())) {
415  // Even though there is an sf_open() that takes a filename, use the one that
416  // takes a file descriptor since wxWidgets can open a file with a Unicode name and
417  // libsndfile can't (under Windows).
418  sf = sf_open_fd(f.fd(), SFM_READ, &info, FALSE);
419  }
420 
421  if (!sf) {
422 
423  memset(data, 0, SAMPLE_SIZE(format)*len);
424 
425  mSilentLog = TRUE;
426 
427  return len;
428  }
429  }
430  mSilentLog=FALSE;
431 
432  sf_seek(sf, start, SEEK_SET);
433  SampleBuffer buffer(len, floatSample);
434 
435  int framesRead = 0;
436 
437  // If both the src and dest formats are integer formats,
438  // read integers from the file (otherwise we would be
439  // converting to float and back, which is unneccesary)
440  if (format == int16Sample &&
441  sf_subtype_is_integer(info.format)) {
442  framesRead = sf_readf_short(sf, (short *)data, len);
443  }
444  else
445  if (format == int24Sample &&
446  sf_subtype_is_integer(info.format))
447  {
448  framesRead = sf_readf_int(sf, (int *)data, len);
449 
450  // libsndfile gave us the 3 byte sample in the 3 most
451  // significant bytes -- we want it in the 3 least
452  // significant bytes.
453  int *intPtr = (int *)data;
454  for( int i = 0; i < framesRead; i++ )
455  intPtr[i] = intPtr[i] >> 8;
456  }
457  else {
458  // Otherwise, let libsndfile handle the conversion and
459  // scaling, and pass us normalized data as floats. We can
460  // then convert to whatever format we want.
461  framesRead = sf_readf_float(sf, (float *)buffer.ptr(), len);
462  CopySamples(buffer.ptr(), floatSample,
463  (samplePtr)data, format, framesRead);
464  }
465 
466  sf_close(sf);
467 
468  return framesRead;
469  }
