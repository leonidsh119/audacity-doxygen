49 /****************************************************************
50 * data structure notes: the midi stream is stored as an array
51 * of 4-byte records, each of which is either a time or midi
52 * data. Midi data always begins with a control byte (high
53 * order bit set), and it is assumed times are positive (high
54 * order bit clear), so the two are easy to distinguish
55 * IF THE COMPILER PUTS THESE BITS IN THE SAME PLACE. It looks
56 * like the high order byte of the time lines up with the last
57 * byte of a 4 byte array, so we will always set the high order
58 * bit of the last array byte when the first 3 bytes are filled
59 * with MIDI data. This is refered to as the "tag" bit.
60 * WARNING: Lattice C longs are UNSIGNED, therefore always
61 * positive. Test the high order bit with a mask.
62 ****************************************************************/
63

65 #define HIGH_BIT 0x80000000
66 #define istime(note) (!(((note)->when) & HIGH_BIT))

94 /****************************************************************************
95 * bend_filter
96 * Inputs:
97 * note_type note: the current note
98 * note_type last: the last recorded event
99 * long now: the current time
100 * Effect:
101 * remove pitch bend events in same 0.01 sec time slot
102 * Implementation:
103 * If the current event is a pitch bend that bends again
104 * in the same time slot, make it a no-op by replacing it with
105 * the time.
106 ****************************************************************************/
107 
108 private void bend_filter(note, last, now)
109 note_type note; /* current note */
110 note_type last; /* the last recorded event */
111 long now; /* the current time */
112 {
113  /* first see if there is another bend in this time
114  * slot.
115  */
116  note_type note2 = note + 1;
117  while (note2 < last) {
118  if (istime(note2) && (note2->when > now)) {
119  break; /* new time slot */
120  }
121  else if (note->n[0] == note2->n[0]) {
122  note->when = now;
123  return; /* found another bend */
124  }
125  note2++;
126  }
127 }

205 /****************************************************************************
206 * filter
207 * Inputs:
208 * note_type last: the last note recorded
209 * Effect: allow only one control change per time slot (0.01 sec)
210 * Implementation:
211 * call ctrl_filter and bend_filter to overwrite control changes with
212 * noop data (the current time is used as a noop)
213 ****************************************************************************/
214 
215 private void filter(last)
216 note_type last;
217 {
218  note_type note; /* loop control variable */
219  long now=0; /* last time seen */
220  int command; /* command pointed to by note */
221 
222  for (note = event_buff; note <= last; note++) {
223  if (istime(note)) {
224  now = note->when;
225  }
226  else {
227  command = note->n[0] & MIDI_CODE_MASK;
228 
229  if (command == MIDI_CTRL &&
230  note->n[1] == SUSTAIN) {
231  /* do nothing */;
232  }
233  else if (command == MIDI_CTRL) {
234  ctrl_filter(note, last, now);
235  }
236  else if (command == MIDI_TOUCH) {
237  bend_filter(note, last, now); /* bend and touch use the */
238  }

240  bend_filter(note, last, now);
241  }
242  }

247 /****************************************************************************
248 * getdur
249 * Inputs:
250 * int i: index of the note
251 * note_type last: pointer to the last event recorded
252 * int ped: TRUE if pedal is down at event i
253 * long now: the time at event i
254 * Outputs:
255 * returns long: the duration of note i
256 * Assumes:
257 * assumes i is a note
258 * Implementation:
259 * This is tricky because of pedal messages. The note is kept on by
260 * either the key or the pedal. Keep 2 flags, key and ped. Key is
261 * turned off when a key is released, ped goes off and on with pedal.
262 * Note ends when (1) both key and ped are FALSE, (2) key is
263 * pressed (this event will also start another note).
264 ****************************************************************************/
265 
266 private long getdur(i, last, ped, now)
267 int i;
268 note_type last;
269 int ped;
270 long now;
271 {
272  int key = TRUE; /* flag that says if note is on */
273  long start = now;
274  int chan = event_buff[i].n[0] & MIDI_CHN_MASK;
275  int pitch = event_buff[i].n[1];
276  note_type note = &(event_buff[i+1]);
277  int noteon; /* TRUE if a noteon message received on chan */
278  int keyon; /* TRUE if noteon message had non-zero velocity */
279 
280  /* search from the next event (i+1) to the end of the buffer:
281  */
282  for (; note < last; note++) {
283  if (istime(note)) {
284  now = note->when;
285  }
286  else {
287  noteon = keyon = FALSE;
288  if ((note->n[0] & MIDI_CHN_MASK) == chan) {
289  noteon = ((note->n[0] & MIDI_CODE_MASK) == MIDI_ON_NOTE) &&
290  (note->n[1] == pitch);
291  keyon = noteon && (note->n[2] != 0);
292  if ((noteon && (note->n[2] == 0)) ||
293  (((note->n[0] & MIDI_CODE_MASK) == MIDI_OFF_NOTE) &&
294  (note->n[1] == pitch))) key = FALSE;
295  if (((note->n[0] & MIDI_CODE_MASK) == MIDI_CTRL) &&
296  note->n[1] == SUSTAIN && note->n[2] == 127) ped = TRUE;
297  if (((note->n[0] & MIDI_CODE_MASK) == MIDI_CTRL) &&
298  note->n[1] == SUSTAIN && note->n[2] == 0) ped = FALSE;
299 
300  if ((!key && !ped) || keyon)
301  return(now - start);
302  }
303  }
304  }
305  return(last->when - start);
306 }

401  int last_voice = 0; /* the default adagio channel (1) */
402  int ped = FALSE; /* flag maintains state of pedal */
403  int how_many = last - event_buff;
404  long now=0; /* the time of the next event */
405 
406  if (fp == NULL) {
407  gprintf(ERROR, "internal error: output called with NULL file.\n");
408  EXIT(1);

413 
414

416  /* set the initial absolute time, all other times are relative */
417

420  if (now < 0) {

461  ped = (event_buff[i].n[2] != 0);
462  needs_voice = FALSE;
463  } else if (command == MIDI_CTRL) {
464  char c = map_ctrl(event_buff[i].n[1]);

467  } else if (command == MIDI_TOUCH) {

472  needs_voice = FALSE; /* ignore note-offs */

550 /* ENOUGH_ROOM says if we have room for 10000 events + 10000 timestamps =
551  * 20000 note_struct's, then that's "enough room" for recording a sequence.
552  * If more ram is available, it won't be used. If less is available, we'll
553  * use as much as we can get, minus "SPACE_FOR_PLAY", which leaves a little
554  * bit of spare ram in case Moxc or stdio need to allocate some space.
555  * For DOS, we limit recording space to 64K.
556  */
557 #ifdef DOS
558 #define ENOUGH_ROOM 64000L
559 #else
560 #define ENOUGH_ROOM (20000L * sizeof(union note_struct))
561 #endif

564 boolean rec_init(boolean bender)
565 {
566  size_t biggestChunk, spaceForRecord;
567 
568  debug_rec = cl_switch("debug");
569  byteorder();
570  pile_ups = 0;
571  max_pile_up = 0;
572  previous_time = (unsigned) -1L; /* this will force putting in initial timestamp */
573  fixed_octave = FALSE;
574 
575  if (max_notes == -1) { /* allocate space 1st time rec_init called */
576  biggestChunk = AVAILMEM;
577  if (biggestChunk <= SPACE_FOR_PLAY) {
578  /* not enough memory; give up */
579  return(FALSE);
580  }
581  else {
582  spaceForRecord =
583  MIN((biggestChunk - SPACE_FOR_PLAY), ENOUGH_ROOM);
584  /* leave SPACE_FOR_PLAY contiguous bytes of memory */
585  }
586  max_notes = spaceForRecord / sizeof(note_node);
587  /* gprintf(GDEBUG,"max_notes = %d\n", max_notes);*/
588  event_buff = (note_type) MALLOC(spaceForRecord);
589  if (event_buff == NULL) {
590  /* should never happen */
591  gprintf(FATAL, "Implementation error (record.c): getting memory.");
592  return FALSE;
593  }
594  }
595  next = event_buff;
596  last = event_buff + max_notes - 2; /* it is critical that last not point
597  * to the very last storage loc */
598  midi_cont(bender);
599  return((boolean)(max_notes > 10));
600  /* it would be silly to record with only room enough for 10 notes! */
601 }

617 /****************************************************************************
618 * rec_event
619 * Inputs:
620 * long time: the current time
621 * long data: midi data to record
622 * Outputs:
623 * returns FALSE if there is no more memory
624 * Effect: reads and stores any input
625 * Implementation:
626 * time stamps and midi events share the same buffer of 4-byte events
627 * save time at most once per call to rec_poll
628 * save time only if it changes
629 ****************************************************************************/
630 
631 boolean rec_event(long *data, time_type time)
632 {
633  /* can't allow negative time because sign bit distinguishes
634  * data from time: */
635  if (time < 0) time = 0;
636 
637  if (previous_time != time) {
638  next++->when = previous_time = time;
639  if (next >= last) goto overflow;
640  }
641 
642  next->when = *data;
643  next++->n[3] = MIDI_CMD_BIT; /* set tag bit */
644  if (next >= last) goto overflow;
645  return TRUE;
646 
647 overflow:
648  next = last; /* last doesn't really point to last storage */
649  gprintf(ERROR, "No more memory.\n");
650  return FALSE;
