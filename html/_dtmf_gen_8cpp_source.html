110 bool EffectDtmf::ProcessInitialize(sampleCount WXUNUSED(totalLen), ChannelNames WXUNUSED(chanMap))
111 {
112  double duration = GetDuration();
113 
114  // all dtmf sequence durations in samples from seconds
115  // MJS: Note that mDuration is in seconds but will have been quantised to the units of the TTC.
116  // If this was 'samples' and the project rate was lower than the track rate,
117  // extra samples may get created as mDuration may now be > mT1 - mT0;
118  // However we are making our best efforts at creating what was asked for.
119 
120  sampleCount nT0 = (sampleCount)floor(mT0 * mSampleRate + 0.5);
121  sampleCount nT1 = (sampleCount)floor((mT0 + duration) * mSampleRate + 0.5);
122  numSamplesSequence = nT1 - nT0; // needs to be exact number of samples selected
123 
124  //make under-estimates if anything, and then redistribute the few remaining samples
125  numSamplesTone = (sampleCount)floor(dtmfTone * mSampleRate);
126  numSamplesSilence = (sampleCount)floor(dtmfSilence * mSampleRate);
127 
128  // recalculate the sum, and spread the difference - due to approximations.
129  // Since diff should be in the order of "some" samples, a division (resulting in zero)
130  // is not sufficient, so we add the additional remaining samples in each tone/silence block,
131  // at least until available.
132  diff = numSamplesSequence - (dtmfNTones*numSamplesTone) - (dtmfNTones-1)*numSamplesSilence;
133  while (diff > 2*dtmfNTones - 1) { // more than one per thingToBeGenerated
134  // in this case, both numSamplesTone and numSamplesSilence would change, so it makes sense
135  // to recalculate diff here, otherwise just keep the value we already have
136 
137  // should always be the case that dtmfNTones>1, as if 0, we don't even start processing,
138  // and with 1 there is no difference to spread (no silence slot)...
139  wxASSERT(dtmfNTones > 1);
140  numSamplesTone += (diff/(dtmfNTones));
141  numSamplesSilence += (diff/(dtmfNTones-1));
142  diff = numSamplesSequence - (dtmfNTones*numSamplesTone) - (dtmfNTones-1)*numSamplesSilence;
143  }
144  wxASSERT(diff >= 0); // should never be negative
145 
146  curSeqPos = -1; // pointer to string in dtmfSequence
147  isTone = false;
148  numRemaining = 0;
149 
150  return true;
151 }

153 sampleCount EffectDtmf::ProcessBlock(float **WXUNUSED(inbuf), float **outbuf, sampleCount size)
154 {
155  float *buffer = outbuf[0];
156  sampleCount processed = 0;
157 
158  // for the whole dtmf sequence, we will be generating either tone or silence
159  // according to a bool value, and this might be done in small chunks of size
160  // 'block', as a single tone might sometimes be larger than the block
161  // tone and silence generally have different duration, thus two generation blocks
162  //
163  // Note: to overcome a 'clicking' noise introduced by the abrupt transition from/to
164  // silence, I added a fade in/out of 1/250th of a second (4ms). This can still be
165  // tweaked but gives excellent results at 44.1kHz: I haven't tried other freqs.
166  // A problem might be if the tone duration is very short (<10ms)... (?)
167  //
168  // One more problem is to deal with the approximations done when calculating the duration
169  // of both tone and silence: in some cases the final sum might not be same as the initial
170  // duration. So, to overcome this, we had a redistribution block up, and now we will spread
171  // the remaining samples in every bin in order to achieve the full duration: test case was
172  // to generate an 11 tone DTMF sequence, in 4 seconds, and with DutyCycle=75%: after generation
173  // you ended up with 3.999s or in other units: 3 seconds and 44097 samples.
174  //
175  while (size)
176  {
177  if (numRemaining == 0)
178  {
179  isTone = !isTone;
180 
181  if (isTone)
182  {
183  curSeqPos++;
184  numRemaining = numSamplesTone;
185  curTonePos = 0;
186  }
187  else
188  {
189  numRemaining = numSamplesSilence;
190  }
191 
192  // the statement takes care of extracting one sample from the diff bin and
193  // adding it into the current block until depletion
194  numRemaining += (diff-- > 0 ? 1 : 0);
195  }
196 
197  sampleCount len = wxMin(numRemaining, size);
198 
199  if (isTone)
200  {
201  // generate the tone and append
202  MakeDtmfTone(buffer, len, mSampleRate, dtmfSequence[curSeqPos], curTonePos, numSamplesTone, dtmfAmplitude);
203  curTonePos += len;
204  }
205  else
206  {
207  memset(buffer, 0, sizeof(float) * len);
208  }
209 
210  numRemaining -= len;
211 
212  buffer += len;
213  size -= len;
214  processed += len;
215  }
216 
217  return processed;
218 }

390 void EffectDtmf::Recalculate()
391 {
392  // remember that dtmfDutyCycle is in range (0.0-100.0)
393 
394  dtmfNTones = (int) dtmfSequence.Length();
395 
396  if (dtmfNTones==0) {
397  // no tones, all zero: don't do anything
398  // this should take care of the case where user got an empty
399  // dtmf sequence into the generator: track won't be generated
400  SetDuration(0.0);
401  dtmfTone = 0;
402  dtmfSilence = 0;
403  } else {
404  if (dtmfNTones==1) {
405  // single tone, as long as the sequence
406  dtmfTone = GetDuration();
407  dtmfSilence = 0;
408  } else {
409  // Don't be fooled by the fact that you divide the sequence into dtmfNTones:
410  // the last slot will only contain a tone, not ending with silence.
411  // Given this, the right thing to do is to divide the sequence duration
412  // by dtmfNTones tones and (dtmfNTones-1) silences each sized according to the duty
413  // cycle: original division was:
414  // slot=mDuration / (dtmfNTones*(dtmfDutyCycle/MAX_DutyCycle)+(dtmfNTones-1)*(1.0-dtmfDutyCycle/MAX_DutyCycle))
415  // which can be simplified in the one below.
416  // Then just take the part that belongs to tone or silence.
417  //
418  double slot = GetDuration() / ((double)dtmfNTones + (dtmfDutyCycle / 100.0) - 1);
419  dtmfTone = slot * (dtmfDutyCycle / 100.0); // seconds
420  dtmfSilence = slot * (1.0 - (dtmfDutyCycle / 100.0)); // seconds
421 
422  // Note that in the extremes we have:
423  // - dutyCycle=100%, this means no silence, so each tone will measure mDuration/dtmfNTones
424  // - dutyCycle=0%, this means no tones, so each silence slot will measure mDuration/(NTones-1)
425  // But we always count:
426  // - dtmfNTones tones
427  // - dtmfNTones-1 silences
428  }
429  }

432 bool EffectDtmf::MakeDtmfTone(float *buffer, sampleCount len, float fs, wxChar tone, sampleCount last, sampleCount total, float amplitude)
433 {
434 /*
435  --------------------------------------------
436  1209 Hz 1336 Hz 1477 Hz 1633 Hz
437 
438  ABC DEF
439  697 Hz 1 2 3 A
440 
441  GHI JKL MNO
442  770 Hz 4 5 6 B
443 
444  PQRS TUV WXYZ
445  852 Hz 7 8 9 C
446 
447  oper
448  941 Hz * 0 # D
449  --------------------------------------------
450  Essentially we need to generate two sin with
451  frequencies according to this table, and sum
452  them up.
453  sin wave is generated by:
454  s(n)=sin(2*pi*n*f/fs)
455 
456  We will precalculate:
457  A= 2*pi*f1/fs
458  B= 2*pi*f2/fs
459 
460  And use two switch statements to select the frequency
461 
462  Note: added support for letters, like those on the keypad
463  This support is only for lowercase letters: uppercase
464  are still considered to be the 'military'/carrier extra
465  tones.
466 */
467 
468  float f1, f2=0.0;
469  double A,B;
470 
471  // select low tone: left column
472  switch (tone) {
473  case '1': case '2': case '3': case 'A':
474  case 'a': case 'b': case 'c':
475  case 'd': case 'e': case 'f':
476  f1=697;
477  break;
478  case '4': case '5': case '6': case 'B':
479  case 'g': case 'h': case 'i':
480  case 'j': case 'k': case 'l':
481  case 'm': case 'n': case 'o':
482  f1=770;
483  break;
484  case '7': case '8': case '9': case 'C':
485  case 'p': case 'q': case 'r': case 's':
486  case 't': case 'u': case 'v':
487  case 'w': case 'x': case 'y': case 'z':
488  f1=852;
489  break;
490  case '*': case '0': case '#': case 'D':
491  f1=941;
492  break;
493  default:
494  f1=0;
495  }
496 
497  // select high tone: top row
498  switch (tone) {
499  case '1': case '4': case '7': case '*':
500  case 'g': case 'h': case 'i':
501  case 'p': case 'q': case 'r': case 's':
502  f2=1209;
503  break;
504  case '2': case '5': case '8': case '0':
505  case 'a': case 'b': case 'c':
506  case 'j': case 'k': case 'l':
507  case 't': case 'u': case 'v':
508  f2=1336;
509  break;
510  case '3': case '6': case '9': case '#':
511  case 'd': case 'e': case 'f':
512  case 'm': case 'n': case 'o':
513  case 'w': case 'x': case 'y': case 'z':
514  f2=1477;
515  break;
516  case 'A': case 'B': case 'C': case 'D':
517  f2=1633;
518  break;
519  default:
520  f2=0;
521  }
522 
523  // precalculations
524  A=B=2*M_PI/fs;
525  A*=f1;
526  B*=f2;
527 
528  // now generate the wave: 'last' is used to avoid phase errors
529  // when inside the inner for loop of the Process() function.
530  for(sampleCount i=0; i<len; i++) {
531  buffer[i]=amplitude*0.5*(sin(A*(i+last))+sin(B*(i+last)));
532  }
533 
534  // generate a fade-in of duration 1/250th of second
535  if (last==0) {
536  A=(fs/kFadeInOut);
537  for(sampleCount i=0; i<A; i++) {
538  buffer[i]*=i/A;
539  }
540  }
541 
542  // generate a fade-out of duration 1/250th of second
543  if (last==total-len) {
544  // we are at the last buffer of 'len' size, so, offset is to
545  // backup 'A' samples, from 'len'
546  A=(fs/kFadeInOut);
547  sampleCount offset=len-(sampleCount)(fs/kFadeInOut);
548  // protect against negative offset, which can occur if too a
549  // small selection is made
550  if (offset>=0) {
551  for(sampleCount i=0; i<A; i++) {
552  buffer[i+offset]*=(1-(i/A));
553  }
554  }
555  }
556  return true;
557 }
