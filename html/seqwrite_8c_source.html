104 private void write_event(seq, event, f, abs_flag)
105  seq_type seq;
106  event_type event;
107  FILE *f;
108  boolean abs_flag;
109 {
110  int voice = vc_voice(event->nvoice);
111 
112  /* process all current (and earlier) events */
113  if (is_note(event)) { /*** a note or rest ***/
114  /* if this note is not a rest, write it */
115  if (event->value != NO_PITCH &&
116  (seq_channel_mask(seq) &
117  (1 << (voice - 1)))) {
118  write_pitch(f, event->value);
119  fprintf(f, " U%ld ", event->u.note.ndur >> 8);
120  write_velocity(f, (int) (event->u.note.ndur & 0xFF));
121  write_voice(f, voice);
122  write_time(f, event, abs_flag);
123  }
124  } else { /*** a control command ***/
125  switch (vc_ctrl(event->nvoice)) {
126  case PSWITCH_CTRL:
127  case MODWHEEL_CTRL:
128  case TOUCH_CTRL:
129  case VOLUME_CTRL:
130  case BEND_CTRL:
131  fprintf(f, "%c%d ", ctrl_letter[vc_ctrl(event->nvoice)],
132  event->value);
133  write_voice(f, voice);
134  write_time(f, event, abs_flag);
135  break;
136  case PROGRAM_CTRL:
137  fprintf(f, "Z%d ", event->value + 1);
138  write_voice(f, voice);
139  write_time(f, event, abs_flag);
140  break;
141  case ESC_CTRL:
142  switch (event->value) {
143  case CALL_VALUE:
144  call_count++;
145  write_rest(f, event, abs_flag);
146  /* !call's are not written because there isn't enough
147  * information. A future version of seq should store
148  * the hash table entry instead of the address of the
149  * routine (then we could get string name of the call)
150  * and the number of arguments.
151  */
152  break;
153  case CLOCK_VALUE:
154  /* design for !clock: for absolute (absflag) files, put the
155  clocks (if any) at the end where they can be edited out
156  easily. For relative files, keep the clocks in-line.
157  On each clock, write a !tempo command inferred by the
158  clock and followed by the !clock. Because the clock
159  event comes before the actual tempo change, the chnage
160  must be delayed by a half tick except for the first one.
161  */
162  if (abs_flag) break;
163  new_tempo = (2500L << 16) / event->u.clock.ticksize;
164  if (clock_started) {
165  tempo_change_at = event->ntime + clock_half_tick;
166  } else {
167  fprintf(f, "!tempo %d\n", new_tempo);
168  fprintf(f, "!clock\n");
169  clock_started = TRUE;
170  }
171  clock_half_tick = (event->u.clock.ticksize) >> 17;
172  break;
173  case MACCTRL_VALUE:
174  fprintf(f, "~%d(%d) ", event->u.macctrl.ctrl_number,
175  event->u.macctrl.value);
176  write_voice(f, voice);
177  write_time(f, event, abs_flag);
178  break;
179  case MACRO_VALUE:
180  macro_count++;
181  write_rest(f, event, abs_flag);
182  /* macros are not written because there isn't enough
183  * information. A future version of seq should store
184  * the number of arguments in the event, or better yet,
185  * in the definition. Send complaints to RBD!
186  */
187  break;
188  case CTRLRAMP_VALUE:
189  fprintf(f, "!ramp ~%d(%d) ~%d(%d) U%d U%d ",
190  event->u.ramp.ctrl, event->u.ramp.u.ctrl.from_value,
191  event->u.ramp.ctrl, event->u.ramp.u.ctrl.to_value,
192  (int)event->u.ramp.step,
193  (int)event->u.ramp.dur);
194  write_voice(f, voice);
195  write_time(f, event, abs_flag);
196  break;
197  case DEFRAMP_VALUE:
198  deframp_count++;
199  write_rest(f, event, abs_flag);
200  /* See MACRO_VALUE above for why this isn't implemented. */
201  break;
202  case SETI_VALUE:
203  seti_count++;
204  write_rest(f, event, abs_flag);
205  /* !seti and !setv are not implemented -- a future version
206  * of seq should save more information so we can reconstruct
207  * the Adagio source command.
208  */
209  break;
210  default:
211  gprintf(TRANS, "unexpected ESC_CTRL value\n");
212  break;
213  }
214  break;
215  default:
216  gprintf(TRANS, "unexpected seq data\n");
217  break;
218  }
219  }
