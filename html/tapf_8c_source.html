1013  time_type t0_min = t0;

1021

1024  sound_unref(vardelay);
1025  snd_badsr();
1026  }

1029  interp_desc = (interp_desc << 2) + interp_style(s1, sr);
1030  interp_desc = (interp_desc << 2) + interp_style(vardelay, sr);
1031  switch (interp_desc) {
1032  case INTERP_ns: /* handled below */
1033  case INTERP_nn: susp->susp.fetch = tapf_nn_fetch; break;
1034  case INTERP_ni: susp->susp.fetch = tapf_ni_fetch; break;

1040  default: snd_badsr(); break;

1062  susp->susp.name = "tapf";
1063  susp->logically_stopped = false;
1064  susp->susp.log_stop_cnt = logical_stop_cnt_cvt(s1);
1065  susp->started = false;
1066  susp->susp.current = 0;
1067  susp->s1 = s1;
1068  susp->s1_cnt = 0;
1069  susp->vardelay = vardelay;
1070  susp->vardelay_cnt = 0;
1071  susp->vardelay_pHaSe = 0.0;
1072  susp->vardelay_pHaSe_iNcR = vardelay->sr / sr;
1073  susp->vardelay_n = 0;
