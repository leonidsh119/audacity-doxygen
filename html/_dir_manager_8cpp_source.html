142 // Behavior of RecursivelyEnumerate is tailored to our uses and not
143 // entirely straightforward. It recurs depth-first from the passed-
144 // in directory into its subdirs according to optional dirspec
145 // pattern, building a list of directories and (optionally) files
146 // in the listed order. The dirspec is not applied to
147 // subdirs of subdirs. Files in the passed-in directory will not be
148 // enumerated. Also, the passed-in directory is the last entry added
149 // to the list.
150 static int RecursivelyEnumerate(wxString dirPath,
151  wxArrayString& filePathArray, // output: all files in dirPath tree
152  wxString dirspec,
153  bool bFiles, bool bDirs,
154  int progress_count = 0,
155  int progress_bias = 0,
156  ProgressDialog* progress = NULL)
157 {
158  int count=0;
159  bool cont;
160 
161  wxDir dir(dirPath);
162  if(dir.IsOpened()){
163  wxString name;
164 
165  if (bFiles){
166  cont= dir.GetFirst(&name, dirspec, wxDIR_FILES);
167  while ( cont ){
168  wxString filepath = dirPath + wxFILE_SEP_PATH + name;
169 
170  count++;
171  filePathArray.Add(filepath);
172 
173  cont = dir.GetNext(&name);
174 
175  if (progress)
176  progress->Update(count + progress_bias,
177  progress_count);
178  }
179  }
180 
181  cont= dir.GetFirst(&name, dirspec, wxDIR_DIRS);
182  while ( cont ){
183  wxString subdirPath = dirPath + wxFILE_SEP_PATH + name;
184  count += RecursivelyEnumerate(
185  subdirPath, filePathArray, wxEmptyString,
186  bFiles, bDirs,
187  progress_count, count + progress_bias,
188  progress);
189  cont = dir.GetNext(&name);
190  }
191  }
192 
193  if (bDirs) {
194  filePathArray.Add(dirPath);
195  count++;
196  }
197 
198  return count;
199 }

768  Each toplevel directory is represented by "e" + two unique
769  hexadecimal digits, for a total possible number of 256
770  toplevels. Each toplevel contains up to 256 subdirs named
771  "d" + two hex digits. Each subdir contains 'a number' of
772  files. */
773 
774  filenum=0;
775  midnum=0;
776  topnum=0;
777 
778  // first action: if there is no available two-level directory in
779  // the available pool, try to make one
780 
781  if(dirMidPool.empty()){
782 
783  // is there a toplevel directory with space for a NEW subdir?
784 
785  if(!dirTopPool.empty()){
786 
787  // there's still a toplevel with room for a subdir
788 
789  DirHash::iterator iter = dirTopPool.begin();
790  int newcount = 0;
791  topnum = iter->first;
792 
793 
794  // search for unused midlevels; linear search adequate
795  // add 32 NEW topnum/midnum dirs full of prospective filenames to midpool
796  for(midnum=0;midnum<256;midnum++){
797  midkey=(topnum<<8)+midnum;
798  if(BalanceMidAdd(topnum,midkey)){
799  newcount++;
800  if(newcount>=32)break;
801  }
802  }
803 
804  if(dirMidPool.empty()){
805  // all the midlevels in this toplevel are in use yet the
806  // toplevel claims some are free; this implies multiple
807  // internal logic faults, but simply giving up and going
808  // into an infinite loop isn't acceptible. Just in case,
809  // for some reason, we get here, dynamite this toplevel so
810  // we don't just fail.
811 
812  // this is 'wrong', but the best we can do given that
813  // something else is also wrong. It will contain the
814  // problem so we can keep going without worry.
815  dirTopPool.erase(topnum);
816  dirTopFull[topnum]=256;
817  }
818  continue;
819  }
820  }
821 
822  if(dirMidPool.empty()){
823  // still empty, thus an absurdly large project; all dirs are
824  // full to 256/256/256; keep working, but fall back to 'big
825  // filenames' and randomized placement
826 
827  filenum = rand();
828  midnum = (int)(256.*rand()/(RAND_MAX+1.));
829  topnum = (int)(256.*rand()/(RAND_MAX+1.));
830  midkey=(topnum<<8)+midnum;
831 
832 
833  }else{
834 
835  DirHash::iterator iter = dirMidPool.begin();
836  midkey = iter->first;
837 
838  // split the retrieved 16 bit directory key into two 8 bit numbers
839  topnum = midkey >> 8;
840  midnum = midkey & 0xff;
841  filenum = (int)(4096.*rand()/(RAND_MAX+1.));
842 
843  }
844 
845  baseFileName.Printf(wxT("e%02x%02x%03x"),topnum,midnum,filenum);
846 
847  if (mBlockFileHash.find(baseFileName) == mBlockFileHash.end()){
848  // not in the hash, good.
849  if (!this->AssignFile(ret, baseFileName, true))
850  {
851  // this indicates an on-disk collision, likely due to an
852  // orphan blockfile. We should try again, but first
853  // alert the balancing info there's a phantom file here;
854  // if the directory is nearly full of orphans we neither
855  // want performance to suffer nor potentially get into an
856  // infinite loop if all possible filenames are taken by
857  // orphans (unlikely but possible)
858  BalanceFileAdd(midkey);
859 
860  }else break;
861  }
862  }
863  // FIXME: Might we get here without midkey having been set?
864  // Seemed like a possible problem in these changes in .aup directory hierarchy.
865  BalanceFileAdd(midkey);
866  return ret;
867 }
