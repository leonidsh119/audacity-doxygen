107 {
108  bool rValue = true; // return value
109 
110  sampleCount end = (sampleCount) start + len; // start, end, len refer to the selected reverse region
111 
112  // STEP 1:
113  // If a reverse selection begins and/or ends at the inside of a clip
114  // perform a split at the start and/or end of the reverse selection
115  WaveClipList::compatibility_iterator node = track->GetClipIterator();
116  while (node) {
117  WaveClip *clip = node->GetData();
118  sampleCount clipStart = clip->GetStartSample();
119  sampleCount clipEnd = clip->GetEndSample();
120  if (clipStart < start && clipEnd > start && clipEnd <= end) { // the reverse selection begins at the inside of a clip
121  double splitTime = track->LongSamplesToTime(start);
122  track->SplitAt(splitTime);
123  }
124  else if (clipStart >= start && clipStart < end && clipEnd > end) { // the reverse selection ends at the inside of a clip
125  double splitTime = track->LongSamplesToTime(end);
126  track->SplitAt(splitTime);
127  }
128  else if (clipStart < start && clipEnd > end) { // the selection begins AND ends at the inside of a clip
129  double splitTime = track->LongSamplesToTime(start);
130  track->SplitAt(splitTime);
131  splitTime = track->LongSamplesToTime(end);
132  track->SplitAt(splitTime);
133  }
134  node = node->GetNext();
135  }
136 
137  //STEP 2:
138  // Individually reverse each clip inside the selected region
139  // and apply the appropriate offset after detaching them from the track
140 
141  bool checkedFirstClip = false;
142 
143  // used in calculating the offset of clips to rearrange
144  // holds the NEW end position of the current clip
145  sampleCount currentEnd = (sampleCount)end;
146 
147  WaveClipList revClips; // holds the reversed clips
148  WaveClipList otherClips; // holds the clips that appear after the reverse selection region
149  WaveClipArray clipArray;
150  track->FillSortedClipArray(clipArray);
151  size_t i;
152  for (i=0; i < clipArray.Count(); i++) {
153 
154  WaveClip *clip = clipArray.Item(i);
155  sampleCount clipStart = clip->GetStartSample();
156  sampleCount clipEnd = clip->GetEndSample();
157 
158  if (clipStart >= start && clipEnd <= end) { // if the clip is inside the selected region
159 
160  // this is used to check if the selected region begins with a whitespace.
161  // if yes then clipStart (of the first clip) and start are not the same.
162  // adjust currentEnd accordingly and set endMerge to false
163  if(checkedFirstClip == false && clipStart > start) {
164  checkedFirstClip = true;
165  if(i > 0) {
166  if (clipArray.Item(i-1)->GetEndSample() <= start) {
167  currentEnd -= (clipStart - start);
168  }
169  }
170  else {
171  currentEnd -= (clipStart - start);
172  }
173  }
174 
175  sampleCount revStart = (clipStart >= start)? clipStart: start;
176  sampleCount revEnd = (clipEnd >= end)? end: clipEnd;
177  sampleCount revLen = (sampleCount)revEnd-revStart;
178  if (revEnd >= revStart) {
179  if(!ProcessOneClip(count, track, revStart, revLen, start, end)) // reverse the clip
180  {
181  rValue = false;
182  break;
183  }
184 
185  sampleCount clipOffsetStart = (sampleCount)(currentEnd - (clipEnd-clipStart)); // calculate the offset required
186  double offsetStartTime = track->LongSamplesToTime(clipOffsetStart);
187  if(i+1 < clipArray.Count()) // update currentEnd if there is a clip to process next
188  {
189  sampleCount nextClipStart = clipArray.Item(i+1)->GetStartSample();
190  currentEnd = (sampleCount)(currentEnd - (clipEnd - clipStart) - (nextClipStart - clipEnd));
191  }
192 
193  clip = track->RemoveAndReturnClip(clip); // detach the clip from track
194  clip->SetOffset(track->LongSamplesToTime(track->TimeToLongSamples(offsetStartTime))); // align time to a sample and set offset
195  revClips.Append(clip);
196 
197  }
198  }
199  else if (clipStart >= end) { // clip is after the selection region
200  clip = track->RemoveAndReturnClip(clip); // simply remove and append to otherClips
201  otherClips.Append(clip);
202  }
203  }
204 
205  // STEP 3: Append the clips from
206  // revClips and otherClips back to the track
207  size_t revClipsCount = revClips.GetCount();
208  for (i = 0; i < revClipsCount; i++) {
209  WaveClipList::compatibility_iterator node = revClips.Item(revClipsCount - 1 - i); // the last clip of revClips is appended to the track first
210  WaveClip *clip = node->GetData();
211  track->AddClip(clip);
212  }
213 
214  for (i = 0; i < otherClips.GetCount(); i++) {
215  WaveClipList::compatibility_iterator node = otherClips.Item(i);
216  track->AddClip(node->GetData());
217  }
218 
219  return rValue;
220 }
