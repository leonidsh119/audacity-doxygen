160  { printf ("\n\nLine %d: sf_command (SFC_SET_DITHER_ON_WRITE) returned error : %s\n\n",
161  __LINE__, sf_strerror (file)) ;
162  exit (1) ;
163  } ;

165  /* Write data to file. */
