181 void MyFLACFile::metadata_callback(const FLAC__StreamMetadata *metadata)
182 {
183  switch (metadata->type)
184  {
185  case FLAC__METADATA_TYPE_VORBIS_COMMENT:
186  for (FLAC__uint32 i = 0; i < metadata->data.vorbis_comment.num_comments; i++) {
187  mComments.Add(UTF8CTOWX((char *)metadata->data.vorbis_comment.comments[i].entry));
188  }
189  break;
190 
191  case FLAC__METADATA_TYPE_STREAMINFO:
192  mFile->mSampleRate=metadata->data.stream_info.sample_rate;
193  mFile->mNumChannels=metadata->data.stream_info.channels;
194  mFile->mBitsPerSample=metadata->data.stream_info.bits_per_sample;
195  mFile->mNumSamples=metadata->data.stream_info.total_samples;
196 
197  if (mFile->mBitsPerSample<=16) {
198  if (mFile->mFormat<int16Sample) {
199  mFile->mFormat=int16Sample;
200  }
201  } else if (mFile->mBitsPerSample<=24) {
202  if (mFile->mFormat<int24Sample) {
203  mFile->mFormat=int24Sample;
204  }
205  } else {
206  mFile->mFormat=floatSample;
207  }
208  mFile->mStreamInfoDone=true;
209  break;
210  // handle the other types we do nothing with to avoid a warning
211  case FLAC__METADATA_TYPE_PADDING: // do nothing with padding
212  case FLAC__METADATA_TYPE_APPLICATION: // no idea what to do with this
213  case FLAC__METADATA_TYPE_SEEKTABLE: // don't need a seektable here
214  case FLAC__METADATA_TYPE_CUESHEET: // convert this to labels?
215  case FLAC__METADATA_TYPE_PICTURE: // ignore pictures
216  case FLAC__METADATA_TYPE_UNDEFINED: // do nothing with this either
217  break;
218  }
219 }

243 FLAC__StreamDecoderWriteStatus MyFLACFile::write_callback(const FLAC__Frame *frame,
244  const FLAC__int32 * const buffer[])
245 {
246  short *tmp=new short[frame->header.blocksize];
247 
248  for (unsigned int chn=0; chn<mFile->mNumChannels; chn++) {
249  if (frame->header.bits_per_sample == 16) {
250  for (unsigned int s=0; s<frame->header.blocksize; s++) {
251  tmp[s]=buffer[chn][s];
252  }
253 
254  mFile->mChannels[chn]->Append((samplePtr)tmp,
255  int16Sample,
256  frame->header.blocksize);
257  }
258  else {
259  mFile->mChannels[chn]->Append((samplePtr)buffer[chn],
260  int24Sample,
261  frame->header.blocksize);
262  }
263  }
264 
265  delete [] tmp;
266 
267  mFile->mSamplesDone += frame->header.blocksize;
268 
269  mFile->mUpdateResult = mFile->mProgress->Update((wxULongLong_t) mFile->mSamplesDone, mFile->mNumSamples != 0 ? (wxULongLong_t)mFile->mNumSamples : 1);
270  if (mFile->mUpdateResult != eProgressSuccess)
271  {
272  return FLAC__STREAM_DECODER_WRITE_STATUS_ABORT;
273  }
274 
275  return FLAC__STREAM_DECODER_WRITE_STATUS_CONTINUE;

349  if(!odDecoder || !odDecoder->ReadHeader())

352  delete mDecoderTask;
353  return false;

365 
366 
367  return true;

370  bool success = mFile->set_filename(OSINPUT(mFilename));
371  if (!success) {
372  return false;
373  }
374  mFile->set_metadata_respond(FLAC__METADATA_TYPE_STREAMINFO);
375  mFile->set_metadata_respond(FLAC__METADATA_TYPE_VORBIS_COMMENT);
376  FLAC::Decoder::File::State state = mFile->init();
377  if (state != FLAC__FILE_DECODER_OK) {
378  return false;
379  }
380 #else
381  if (!mHandle.Open(mFilename, wxT("rb"))) {
382  return false;
383  }
384 
385  // Even though there is an init() method that takes a filename, use the one that
386  // takes a file handle because wxWidgets can open a file with a Unicode name and
387  // libflac can't (under Windows).
388  //
389  // Responsibility for closing the file is passed to libflac.
390  // (it happens when mFile->finish() is called)
391  bool result = mFile->init(mHandle.fp())?true:false;
392  mHandle.Detach();
393 
394  if (result != FLAC__STREAM_DECODER_INIT_STATUS_OK) {
395  return false;
396  }
397 #endif
398  mFile->process_until_end_of_metadata();
399 
400 #ifdef LEGACY_FLAC
401  state = mFile->get_state();
402  if (state != FLAC__FILE_DECODER_OK) {
403  return false;
404  }
405 #else
406  // not necessary to check state, error callback will catch errors, but here's how:
407  if (mFile->get_state() > FLAC__STREAM_DECODER_READ_FRAME) {
408  return false;
409  }
410 #endif
411 
412  if (!mFile->is_valid() || mFile->get_was_error()) {
413  // This probably is not a FLAC file at all
414  return false;
415  }
416  return true;
417 }

432 int FLACImportFileHandle::Import(TrackFactory *trackFactory,
433  Track ***outTracks,
434  int *outNumTracks,
435  Tags *tags)
436 {
437  wxASSERT(mStreamInfoDone);
438 
439  CreateProgress();
440 
441  mChannels = new WaveTrack *[mNumChannels];
442 
443  unsigned long c;
444  for (c = 0; c < mNumChannels; c++) {
445  mChannels[c] = trackFactory->NewWaveTrack(mFormat, mSampleRate);
446 
447  if (mNumChannels == 2) {
448  switch (c) {
449  case 0:
450  mChannels[c]->SetChannel(Track::LeftChannel);
451  mChannels[c]->SetLinked(true);
452  break;
453  case 1:
454  mChannels[c]->SetChannel(Track::RightChannel);
455  break;
456  }
457  }
458  else {
459  mChannels[c]->SetChannel(Track::MonoChannel);
460  }
461  }
462 
463 
464 //Start OD
465  bool useOD = false;
466 #ifdef EXPERIMENTAL_OD_FLAC
467  useOD=true;
468 #endif
469 
470  // TODO: Vigilant Sentry: Variable res unused after assignment (error code DA1)
471  // Should check the result.
472  #ifdef LEGACY_FLAC
473  bool res = (mFile->process_until_end_of_file() != 0);
474  #else
475  bool res = true;
476  if(!useOD)
477  res = (mFile->process_until_end_of_stream() != 0);
478  #endif
479  wxUnusedVar(res);
480 
481  //add the task to the ODManager
482  if(useOD)
483  {
484  sampleCount fileTotalFrames = mNumSamples;
485  sampleCount maxBlockSize = mChannels[0]->GetMaxBlockSize();
486  for (sampleCount i = 0; i < fileTotalFrames; i += maxBlockSize) {
487  sampleCount blockLen = maxBlockSize;
488  if (i + blockLen > fileTotalFrames)
489  blockLen = fileTotalFrames - i;
490 
491  for (c = 0; c < mNumChannels; c++)
492  mChannels[c]->AppendCoded(mFilename, i, blockLen, c,ODTask::eODFLAC);
493 
494  mUpdateResult = mProgress->Update(i, fileTotalFrames);
495  if (mUpdateResult != eProgressSuccess)
496  break;
497  }
498 
499  bool moreThanStereo = mNumChannels>2;
500  for (c = 0; c < mNumChannels; c++)
501  {
502  mDecoderTask->AddWaveTrack(mChannels[c]);
503  if(moreThanStereo)
504  {
505  //if we have 3 more channels, they get imported on seperate tracks, so we add individual tasks for each.
506  ODManager::Instance()->AddNewTask(mDecoderTask);
507  mDecoderTask=new ODDecodeFlacTask; //TODO: see if we need to use clone to keep the metadata.
508  }
509  }
510  //if we have mono or a linked track (stereo), we add ONE task for the one linked wave track
511  if(!moreThanStereo)
512  ODManager::Instance()->AddNewTask(mDecoderTask);
513  }
514 //END OD
515 
516 
517  if (mUpdateResult == eProgressFailed || mUpdateResult == eProgressCancelled) {
518  for(c = 0; c < mNumChannels; c++) {
519  delete mChannels[c];
520  }
521  delete[] mChannels;
522 
523  return mUpdateResult;
524  }
525 
526  *outNumTracks = mNumChannels;
527  *outTracks = new Track *[mNumChannels];
528  for (c = 0; c < mNumChannels; c++) {
529  mChannels[c]->Flush();
530  (*outTracks)[c] = mChannels[c];
531  }
532  delete[] mChannels;
533 
534  tags->Clear();
535  size_t cnt = mFile->mComments.GetCount();
536  for (c = 0; c < cnt; c++) {
537  wxString name = mFile->mComments[c].BeforeFirst(wxT('='));
538  wxString value = mFile->mComments[c].AfterFirst(wxT('='));
539  if (name.Upper() == wxT("DATE") && !tags->HasTag(TAG_YEAR)) {
540  long val;
541  if (value.Length() == 4 && value.ToLong(&val)) {
542  name = TAG_YEAR;
543  }
544  }
545  tags->SetTag(name, value);
546  }
547 
548  return mUpdateResult;
549 }
