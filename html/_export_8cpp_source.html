417 bool Exporter::ExamineTracks()
418 {
419  // Init
420  mNumSelected = 0;
421  mNumLeft = 0;
422  mNumRight = 0;
423  mNumMono = 0;
424 
425  // First analyze the selected audio, perform sanity checks, and provide
426  // information as appropriate.
427 
428  // Tally how many are right, left, mono, and make sure at
429  // least one track is selected (if selectedOnly==true)
430 
431  double earliestBegin = mT1;
432  double latestEnd = mT0;
433 
434  const TrackList *tracks = mProject->GetTracks();
435  TrackListConstIterator iter1(tracks);
436  const Track *tr = iter1.First();
437 
438  while (tr) {
439  if (tr->GetKind() == Track::Wave) {
440  if ( (tr->GetSelected() || !mSelectedOnly) && !tr->GetMute() ) { // don't count muted tracks
441  mNumSelected++;
442 
443  if (tr->GetChannel() == Track::LeftChannel) {
444  mNumLeft++;
445  }
446  else if (tr->GetChannel() == Track::RightChannel) {
447  mNumRight++;
448  }
449  else if (tr->GetChannel() == Track::MonoChannel) {
450  // It's a mono channel, but it may be panned
451  float pan = ((WaveTrack*)tr)->GetPan();
452 
453  if (pan == -1.0)
454  mNumLeft++;
455  else if (pan == 1.0)
456  mNumRight++;
457  else if (pan == 0)
458  mNumMono++;
459  else {
460  // Panned partially off-center. Mix as stereo.
461  mNumLeft++;
462  mNumRight++;
463  }
464  }
465 
466  if (tr->GetOffset() < earliestBegin) {
467  earliestBegin = tr->GetOffset();
468  }
469 
470  if (tr->GetEndTime() > latestEnd) {
471  latestEnd = tr->GetEndTime();
472  }
473 
474  }
475  }
476 
477  tr = iter1.Next();
478  }
479 
480  if (mNumSelected == 0) {
481  wxString message;
482  if(mSelectedOnly)
483  message = _("All selected audio is muted.");
484  else
485  message = _("All audio is muted.");
486  wxMessageBox(message,
487  _("Unable to export"),
488  wxOK | wxICON_INFORMATION);
489  return false;
490  }
491 
492  if (mT0 < earliestBegin)
493  mT0 = earliestBegin;
494 
495  if (mT1 > latestEnd)
496  mT1 = latestEnd;
497 
498  return true;
499 }

530  mFilename.SetName(mProject->GetName());
531  while (true) {

533  mBook = NULL;
534 
535  FileDialog fd(mProject,
536  mFileDialogTitle,

538  mFilename.GetFullName(),
539  maskString,

548 
549  mDialog->PopEventHandler();
550

552  return false;

557  return false;

561  mFilterIndex = fd.GetFilterIndex();
562

564  for (size_t i = 0; i < mPlugins.GetCount(); i++)
565  {
566  for (int j = 0; j < mPlugins[i]->GetFormatCount(); j++)
567  {
568  if (mFilterIndex == c)
569  {
570  mFormat = i;
571  mSubFormat = j;
572  }
573  c++;
574  }
575  }

577  wxString ext = mFilename.GetExt();
578  wxString defext = mPlugins[mFormat]->GetExtension(mSubFormat).Lower();
579 
580  //
581  // Check the extension - add the default if it's not there,
582  // and warn user if it's abnormal.
583  //
584  if (ext.IsEmpty()) {
585  //
586  // Make sure the user doesn't accidentally save the file
587  // as an extension with no name, like just plain ".wav".
588  //

605  {
606  continue;
607  }
608  else if (!ext.IsEmpty() && !mPlugins[mFormat]->IsExtension(ext,mSubFormat) && ext.CmpNoCase(defext)) {
609  wxString prompt;
610  prompt.Printf(_("You are about to export a %s file with the name \"%s\".\n\nNormally these files end in \".%s\", and some programs will not open files with nonstandard extensions.\n\nAre you sure you want to export the file under this name?"),
611  mPlugins[mFormat]->GetFormat(mSubFormat).c_str(),

618  if (action != wxYES) {
619  continue;

621  }

626  }

631  // Also, this can only happen for uncompressed audio.
