558  /* grab next width_x1_sample when phase goes past 1.0; */
559  /* use width_n (computed below) to avoid roundoff errors: */
560  if (susp->width_n <= 0) {
561  susp_check_term_log_samples(width, width_ptr, width_cnt);
562  susp->width_x1_sample = susp_fetch_sample(width, width_ptr, width_cnt);
563  susp->width_pHaSe -= 1.0;
564  /* width_n gets number of samples before phase exceeds 1.0: */
565  susp->width_n = (long) ((1.0 - susp->width_pHaSe) *
566  susp->output_per_width);
567  susp->recompute = true;
568  }
569  togo = min(togo, susp->width_n);
570  width_val = susp->width_x1_sample;
571  if (susp->recompute) {
572  /* susp->a0 = 1.0 + susp->gg / susp->J; */
573  double a_0_recip = susp->J / (susp->J + susp->gg);
574  susp->recompute = false;
575  susp->gg = susp->sw * sinh(log_of_2_over_2 *
576  width_val * susp->w1 / susp->sw);
577  susp->b0 = (1.0 + susp->gg * susp->J) * a_0_recip;
578  susp->b1 *= a_0_recip;
579  susp->b2 = (1.0 - susp->gg * susp->J) * a_0_recip;
580  susp->a1 *= a_0_recip;
581  susp->a2 = (susp->gg / susp->J - 1.0) * a_0_recip;
582  }
583  /* don't run past terminate time */
584  if (susp->terminate_cnt != UNKNOWN &&
585  susp->terminate_cnt <= susp->susp.current + cnt + togo) {
586  togo = susp->terminate_cnt - (susp->susp.current + cnt);
587  if (togo < 0) togo = 0; /* avoids rounding errros */
588  if (togo == 0) break;
589  }
590 
591 
592  /* don't run past logical stop time */
593  if (!susp->logically_stopped && susp->susp.log_stop_cnt != UNKNOWN) {
594  int to_stop = susp->susp.log_stop_cnt - (susp->susp.current + cnt);
595  /* break if to_stop == 0 (we're at the logical stop)
596  * AND cnt > 0 (we're not at the beginning of the
597  * output block).
598  */
599  if (to_stop < 0) to_stop = 0; /* avoids rounding errors */
600  if (to_stop < togo) {
601  if (to_stop == 0) {
602  if (cnt) {
603  togo = 0;
604  break;
605  } else /* keep togo as is: since cnt == 0, we
606  * can set the logical stop flag on this
607  * output block
608  */
609  susp->logically_stopped = true;
610  } else /* limit togo so we can start a new
611  * block at the LST
612  */
613  togo = to_stop;
614  }
615  }
616 
617  n = togo;
618  cw_reg = susp->cw;
619  b0_reg = susp->b0;
620  b1_reg = susp->b1;
621  b2_reg = susp->b2;
622  a1_reg = susp->a1;
623  a2_reg = susp->a2;
624  z1_reg = susp->z1;
625  z2_reg = susp->z2;
626  inp_period_reg = susp->inp_period;
627  input_ptr_reg = susp->input_ptr;
628  out_ptr_reg = out_ptr;
629  if (n) do { /* the inner sample computation loop */
630  double z0;
631  z0 = *input_ptr_reg++ + a1_reg*z1_reg + a2_reg*z2_reg;
632  *out_ptr_reg++ = (sample_type) (z0*b0_reg + z1_reg*b1_reg + z2_reg*b2_reg);
633  z2_reg = z1_reg; z1_reg = z0;
634  } while (--n); /* inner loop */
635 
636  susp->z1 = z1_reg;
637  susp->z2 = z2_reg;
638  /* using input_ptr_reg is a bad idea on RS/6000: */
639  susp->input_ptr += togo;
640  out_ptr += togo;
641  susp_took(input_cnt, togo);
642  susp->hz_pHaSe += togo * susp->hz_pHaSe_iNcR;
643  susp->hz_n -= togo;
644  susp->gain_pHaSe += togo * susp->gain_pHaSe_iNcR;
645  susp->gain_n -= togo;
646  susp->width_pHaSe += togo * susp->width_pHaSe_iNcR;
647  susp->width_n -= togo;
648  cnt += togo;
649  } /* outer loop */
650 
651  /* test for termination */
652  if (togo == 0 && cnt == 0) {
653  snd_list_terminate(snd_list);
654  } else {
655  snd_list->block_len = cnt;
656  susp->susp.current += cnt;
657  }
658  /* test for logical stop */
659  if (susp->logically_stopped) {
660  snd_list->logically_stopped = true;
661  } else if (susp->susp.log_stop_cnt == susp->susp.current) {
662  susp->logically_stopped = true;
663  }
664 } /* eqbandvvv_nrrr_fetch */

667 void eqbandvvv_toss_fetch(snd_susp_type a_susp, snd_list_type snd_list)
668  {
669  eqbandvvv_susp_type susp = (eqbandvvv_susp_type) a_susp;
670  time_type final_time = susp->susp.t0;
671  long n;
672 
673  /* fetch samples from input up to final_time for this block of zeros */
674  while ((round((final_time - susp->input->t0) * susp->input->sr)) >=
675  susp->input->current)
676  susp_get_samples(input, input_ptr, input_cnt);
677  /* fetch samples from hz up to final_time for this block of zeros */
678  while ((round((final_time - susp->hz->t0) * susp->hz->sr)) >=
679  susp->hz->current)
680  susp_get_samples(hz, hz_ptr, hz_cnt);
681  /* fetch samples from gain up to final_time for this block of zeros */
682  while ((round((final_time - susp->gain->t0) * susp->gain->sr)) >=
683  susp->gain->current)
684  susp_get_samples(gain, gain_ptr, gain_cnt);
685  /* fetch samples from width up to final_time for this block of zeros */
686  while ((round((final_time - susp->width->t0) * susp->width->sr)) >=
687  susp->width->current)
688  susp_get_samples(width, width_ptr, width_cnt);
689  /* convert to normal processing when we hit final_count */
690  /* we want each signal positioned at final_time */
691  n = round((final_time - susp->input->t0) * susp->input->sr -
692  (susp->input->current - susp->input_cnt));
693  susp->input_ptr += n;
694  susp_took(input_cnt, n);
695  n = round((final_time - susp->hz->t0) * susp->hz->sr -
696  (susp->hz->current - susp->hz_cnt));
697  susp->hz_ptr += n;
698  susp_took(hz_cnt, n);
699  n = round((final_time - susp->gain->t0) * susp->gain->sr -
700  (susp->gain->current - susp->gain_cnt));
701  susp->gain_ptr += n;
702  susp_took(gain_cnt, n);
703  n = round((final_time - susp->width->t0) * susp->width->sr -
704  (susp->width->current - susp->width_cnt));
705  susp->width_ptr += n;
706  susp_took(width_cnt, n);
707  susp->susp.fetch = susp->susp.keep_fetch;
708  (*(susp->susp.fetch))(a_susp, snd_list);
709 }

754 sound_type snd_make_eqbandvvv(sound_type input, sound_type hz, sound_type gain, sound_type width)
755 {
756  register eqbandvvv_susp_type susp;
757  rate_type sr = input->sr;
758  time_type t0 = min(min(min(input->t0, hz->t0), gain->t0), width->t0);
759  int interp_desc = 0;
760  sample_type scale_factor = 1.0F;
761  time_type t0_min = t0;
762  long lsc;
763  /* combine scale factors of linear inputs (INPUT) */
764  scale_factor *= input->scale;
765  input->scale = 1.0F;
766 
767  /* try to push scale_factor back to a low sr input */
768  if (input->sr < sr) { input->scale = scale_factor; scale_factor = 1.0F; }
769 
770  falloc_generic(susp, eqbandvvv_susp_node, "snd_make_eqbandvvv");
771  susp->inp_scale = input->scale;
772  susp->w1 = 0.0;
773  susp->sw = 0.0;
774  susp->cw = 0.0;
775  susp->J = 0.0;
776  susp->gg = 0.0;
777  susp->b0 = 0.0;
778  susp->b1 = 0.0;
779  susp->b2 = 0.0;
780  susp->a0 = 0.0;
781  susp->a1 = 0.0;
782  susp->a2 = 0.0;
783  susp->z1 = 0.0;
784  susp->z2 = 0.0;
785  susp->recompute = false;
786  susp->inp_period = 1.0 / input->sr;
787 
788  /* make sure no sample rate is too high */
789  if (hz->sr > sr) {
790  sound_unref(hz);
791  snd_badsr();
792  }
793  if (gain->sr > sr) {
794  sound_unref(gain);
795  snd_badsr();
796  }
797  if (width->sr > sr) {
798  sound_unref(width);
799  snd_badsr();
800  }
801 
802  /* select a susp fn based on sample rates */
803  interp_desc = (interp_desc << 2) + interp_style(input, sr);
804  interp_desc = (interp_desc << 2) + interp_style(hz, sr);
805  interp_desc = (interp_desc << 2) + interp_style(gain, sr);
806  interp_desc = (interp_desc << 2) + interp_style(width, sr);
807  switch (interp_desc) {
808  case INTERP_nnnn: /* handled below */
809  case INTERP_nnns: /* handled below */
810  case INTERP_nnsn: /* handled below */
811  case INTERP_nnss: /* handled below */
812  case INTERP_nsnn: /* handled below */
813  case INTERP_nsns: /* handled below */
814  case INTERP_nssn: /* handled below */
815  case INTERP_nsss: susp->susp.fetch = eqbandvvv_nsss_fetch; break;
816  case INTERP_niii: susp->susp.fetch = eqbandvvv_niii_fetch; break;
817  case INTERP_nrrr: susp->susp.fetch = eqbandvvv_nrrr_fetch; break;
818  default: snd_badsr(); break;
819  }
820 
821  susp->terminate_cnt = UNKNOWN;
822  /* handle unequal start times, if any */
823  if (t0 < input->t0) sound_prepend_zeros(input, t0);
824  if (t0 < hz->t0) sound_prepend_zeros(hz, t0);
825  if (t0 < gain->t0) sound_prepend_zeros(gain, t0);
826  if (t0 < width->t0) sound_prepend_zeros(width, t0);
827  /* minimum start time over all inputs: */
828  t0_min = min(input->t0, min(hz->t0, min(gain->t0, min(width->t0, t0))));
829  /* how many samples to toss before t0: */
830  susp->susp.toss_cnt = (long) ((t0 - t0_min) * sr + 0.5);
831  if (susp->susp.toss_cnt > 0) {
832  susp->susp.keep_fetch = susp->susp.fetch;
833  susp->susp.fetch = eqbandvvv_toss_fetch;
834  }
835 
836  /* initialize susp state */
837  susp->susp.free = eqbandvvv_free;
838  susp->susp.sr = sr;
839  susp->susp.t0 = t0;
840  susp->susp.mark = eqbandvvv_mark;
841  susp->susp.print_tree = eqbandvvv_print_tree;
842  susp->susp.name = "eqbandvvv";
843  susp->logically_stopped = false;
844  susp->susp.log_stop_cnt = logical_stop_cnt_cvt(input);
845  lsc = logical_stop_cnt_cvt(hz);
846  if (susp->susp.log_stop_cnt > lsc)
847  susp->susp.log_stop_cnt = lsc;
848  lsc = logical_stop_cnt_cvt(gain);
849  if (susp->susp.log_stop_cnt > lsc)
850  susp->susp.log_stop_cnt = lsc;
851  lsc = logical_stop_cnt_cvt(width);
852  if (susp->susp.log_stop_cnt > lsc)
853  susp->susp.log_stop_cnt = lsc;
854  susp->started = false;
855  susp->susp.current = 0;
856  susp->input = input;
857  susp->input_cnt = 0;
858  susp->hz = hz;
859  susp->hz_cnt = 0;
860  susp->hz_pHaSe = 0.0;
861  susp->hz_pHaSe_iNcR = hz->sr / sr;
862  susp->hz_n = 0;
863  susp->output_per_hz = sr / hz->sr;
864  susp->gain = gain;
865  susp->gain_cnt = 0;
866  susp->gain_pHaSe = 0.0;
867  susp->gain_pHaSe_iNcR = gain->sr / sr;
868  susp->gain_n = 0;
869  susp->output_per_gain = sr / gain->sr;
870  susp->width = width;
871  susp->width_cnt = 0;
872  susp->width_pHaSe = 0.0;
873  susp->width_pHaSe_iNcR = width->sr / sr;
874  susp->width_n = 0;
875  susp->output_per_width = sr / width->sr;
876  return sound_create((snd_susp_type)susp, t0, sr, scale_factor);
877 }
