108 static void
109 run(LV2_Handle instance,
110  uint32_t sample_count)
111 {
112  Fifths* self = (Fifths*)instance;
113  FifthsURIs* uris = &self->uris;
114 
115  // Struct for a 3 byte MIDI event, used for writing notes
116  typedef struct {
117  LV2_Atom_Event event;
118  uint8_t msg[3];
119  } MIDINoteEvent;
120 
121  // Initially self->out_port contains a Chunk with size set to capacity
122 
123  // Get the capacity
124  const uint32_t out_capacity = self->out_port->atom.size;
125 
126  // Write an empty Sequence header to the output
127  lv2_atom_sequence_clear(self->out_port);
128  self->out_port->atom.type = self->in_port->atom.type;
129 
130  // Read incoming events
131  LV2_ATOM_SEQUENCE_FOREACH(self->in_port, ev) {
132  if (ev->body.type == uris->midi_Event) {
133  const uint8_t* const msg = (const uint8_t*)(ev + 1);
134  switch (lv2_midi_message_type(msg)) {
135  case LV2_MIDI_MSG_NOTE_ON:
136  case LV2_MIDI_MSG_NOTE_OFF:
137  // Forward note to output
138  lv2_atom_sequence_append_event(
139  self->out_port, out_capacity, ev);
140 
141  const uint8_t note = msg[1];
142  if (note <= 127 - 7) {
143  // Make a note one 5th (7 semitones) higher than input
144  MIDINoteEvent fifth;
145 
146  // Could simply do fifth.event = *ev here instead...
147  fifth.event.time.frames = ev->time.frames; // Same time
148  fifth.event.body.type = ev->body.type; // Same type
149  fifth.event.body.size = ev->body.size; // Same size
150 
151  fifth.msg[0] = msg[0]; // Same status
152  fifth.msg[1] = msg[1] + 7; // Pitch up 7 semitones
153  fifth.msg[2] = msg[2]; // Same velocity
154 
155  // Write 5th event
156  lv2_atom_sequence_append_event(
157  self->out_port, out_capacity, &fifth.event);
158  }
159  break;
160  default:
161  // Forward all other MIDI events directly
162  lv2_atom_sequence_append_event(
163  self->out_port, out_capacity, ev);
164  break;
165  }
166  }
167  }
168 }
