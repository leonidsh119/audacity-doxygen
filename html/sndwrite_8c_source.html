109 /*
110  * We want to write as soon as space is available so that
111  * a full sound buffer can be queued up for output. This
112  * may require transferring only part of buf, so we keep
113  * track of progress and output whenever space is available.

115 void write_to_audio(snd_type player, void *buf, long buflen)
116 {
117  long rslt;

123  snd_write(player, buf, rslt);

127  }

375 sample_type sound_save_sound(LVAL s_as_lval, long n, snd_type snd,
376  char *buf, long *ntotal, snd_type player)
377 {
378  long blocklen;
379  long buflen;
380  sound_type s;
381  long debug_unit; /* print messages at intervals of this many samples */
382  long debug_count; /* next point at which to print a message */
383  sample_type max_sample = 0.0F;
384  cvtfn_type cvtfn;
385  *ntotal = 0;
386 
387  /* if snd_expr was simply a symbol, then s now points to
388  a shared sound_node. If we read samples from it, then
389  the sound bound to the symbol will be destroyed, so
390  copy it first. If snd_expr was a real expression that
391  computed a new value, then the next garbage collection
392  will reclaim the sound_node. We need to make the new
393  sound reachable by the garbage collector to that any
394  lisp data reachable from the sound do not get collected.
395  To make the sound reachable, we need to allocate a node,
396  and the GC might run, so we need to protect the OLD s
397  but then make it unreachable.
398  We will let the GC collect the sound in the end.
399  */
400  xlprot1(s_as_lval);
401  s = sound_copy(getsound(s_as_lval));
402  s_as_lval = cvsound(s); /* destroys only ref. to original */
403 
404  /* for debugging */
405 /* printing_this_sound = s;*/
406 
407 
408  debug_unit = debug_count = (long) max(snd->format.srate, 10000.0);
409 
410  cvtfn = find_cvt_to_fn(snd, buf);
411 
412 #ifdef MACINTOSH
413  if (player) {
414  gprintf(TRANS, "Playing audio: Click and hold mouse button to stop playback.\n");
415  }
416 #endif
417 
418  while (n > 0) {
419  long togo;
420  float peak;
421  sample_block_type sampblock = sound_get_next(s, &blocklen);
422  oscheck();
423 #ifdef SNAPSHOTS
424  stdputstr(".");
425  if (sound_created_flag) {
426  stdputstr("SNAPSHOT: ");
427  sound_print_tree(printing_this_sound);
428  sound_created_flag = false;
429  }
430  fflush(stdout);
431 #endif
432  if (sampblock == zero_block || blocklen == 0) {
433  break;
434  }
435  togo = min(blocklen, n);
436 
437  buflen = (*cvtfn)((void *) buf, (void *) sampblock->samples,
438  togo, s->scale, &peak);
439  if (peak > max_sample) max_sample = peak;
440 
441 #ifdef MACINTOSH
442  if (Button()) {
443  if (player) {
444  snd_reset(player);
445  }
446  gprintf(TRANS, "\n\nStopping playback...\n\n\n");
447  break;
448  }
449 #endif
450

452  if (player) write_to_audio(player, (void *) buf, buflen);

455  *ntotal += togo;

458  fflush(stdout);
459  debug_count += debug_unit;
460  }
461  }
