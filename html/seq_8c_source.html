48 /* chunk_alloc -- allocate data for a sequence */
49 /*
50  * NOTE: This assumes one chunk is already allocated.
51  * The first chunk holds shared sequence information in
52  * the info struct, and by convention this is always in
53  * the first chunk.
54  */
55 private char *chunk_alloc(seq_type seq, int size)
56 {
57  chunk_type chunk = seq->chunklist->u.info.last_chunk;
58  /* gprintf(TRANS, "chunk_alloc: seq %lx size %d\n", seq, size); */
59  if (size & 1) size++; /* make it even */
60  if (chunk->free + size >= CHUNK_SIZE) {
61  chunk_type new_chunk = chunk_create(FALSE);
62  if (!new_chunk)
63  {
64  gprintf(FATAL, "Out of memory while reading seq\n");
65  return NULL;
66  }
67  /* add new_chunk to chunk chain */
68  seq->chunklist->u.info.last_chunk = new_chunk;
69  chunk->next = new_chunk;
70  chunk = new_chunk;
71  }
72  chunk->free += size;
73  return &(chunk->u.data[chunk->free - size]);
74 }

364 /****************************************************************************
365 * insert_event
366 * Inputs:
367 * seq_type seq: where to put the event
368 * event_type event: the event to insert
369 * Effect:
370 * inserts event into the event list
371 * NOTE: it is inserted *after* previously inserted events with the same time
372 * Implementation:
373 * adagio files often contain many independent voices. Although each voice
374 * consists of events in sequence, the voices need not be inter-twined in
375 * the input file. Rather, all the events of voice 1 appear followed by all
376 * the events of voice 2, and so forth. As phase one merges these event
377 * sequences, it must make many passes over an increasingly long list of
378 * events: expensive if we always start from the beginning of the list!
379 * we can exploit the fact that each voice is sequential by starting the
380 * search for the proper point of insertion at the last event inserted.
381 * the variable "last_event" is used to remember this hint. We could
382 * also snapshot "last_event" in "ref_event" when a !tempo or !rate
383 * command occurs as another hint, but we don't.
384 ****************************************************************************/
385 
386 private void insert_event(seq, event)
387  seq_type seq;
388  register event_type event;
389 {
390  event_type *evlptr = &(seq_eventlist(seq));
391  if ((*evlptr == NULL) ||
392  (event->ntime < (*evlptr)->ntime)) {
393  /* insert at the head of the list */
394  event->next = *evlptr;
395  *evlptr = event;
396  seq->current = event;
397  } else {
398  /* insert somewhere after the head of the list
399  * do not assume: current is not NULL. Although we always leave
400  * it set, the client may access the sequence before the next
401  * insert.
402  */
403  register event_type previous;
404  register event_type insert_before;
405 
406  if (!seq->current) {
407  seq->current = seq_eventlist(seq);
408  }
409  if (event->ntime >= seq->current->ntime) {
410  /* insertion point is after current */
411  previous = seq->current;
412  insert_before = previous->next;
413  } else {
414  /* insertion point is before current; start at beginning */
415  /* assume: not inserting at very head of list; that would
416  * have been taken care of above */
417  previous = seq_events(seq);
418  insert_before = previous->next;
419  }
420 
421  while ((insert_before != NULL) &&
422  (event->ntime >= insert_before->ntime)) {
423  previous = insert_before;
424  insert_before = insert_before->next;
425  }
426  previous->next = event;
427  event->next = insert_before;
428  seq->current = event;
429  }
430 }

558 private void process_event(seq)
559  seq_type seq;
560 {
561  register event_type event;
562  if (!seq->runflag) return;
563  while ((event = seq->current) && (event->ntime <= virttime)) {
564  int voice;
565  /* process all current (and earlier) events */
566  if (is_note(event)) { /*** play a note or rest ***/
567  /* if this note is not a rest, play it and schedule an off event */
568  if (event->value != NO_PITCH &&
569  (seq_channel_mask(seq) &
570  (1 << ((voice = vc_voice(event->nvoice)) - 1)))) {
571  seq_noteon(seq, voice, event->value,
572  (int) event->u.note.ndur & 0xFF);
573  if (debug) {
574  gprintf(TRANS, "play pitch %d at %ld\n",
575  event->value, event->ntime);
576  }
577  seq_cause_noteoff(seq, (event->u.note.ndur) >> 8,
578  voice, event->value);
579 
580  }
581  } else { /*** send a control command ***/
582  int n;
583  time_type step;
584  int delta;
585  long increment;
586  int voice = vc_voice(event->nvoice);
587  ulong enabled = seq_channel_mask(seq) & (1 << (voice - 1));
588 
589  switch (vc_ctrl(event->nvoice)) {
590  case PSWITCH_CTRL:
591  if (!enabled) break;
592  seq_midi_ctrl(seq, voice, PORTASWITCH, event->value);
593  break;
594  case MODWHEEL_CTRL:
595  if (!enabled) break;
596  seq_midi_ctrl(seq, voice, MODWHEEL, event->value);
597  break;
598  case TOUCH_CTRL:
599  if (!enabled) break;
600  seq_midi_touch(seq, voice, event->value);
601  break;
602  case VOLUME_CTRL:
603  if (!enabled) break;
604  seq_midi_ctrl(seq, voice, VOLUME, event->value);
605  break;
606  case BEND_CTRL:
607  if (!enabled) break;
608  seq_midi_bend(seq, voice, (event->value << 6));
609  break;
610  case PROGRAM_CTRL:
611  if (!enabled) break;
612  seq_midi_program(seq, voice, event->value + 1);
613  break;
614  case ESC_CTRL:
615  switch (event->value) {
616  case CALL_VALUE:
617  sequence = seq;
618  (*(event->u.call.routine))(event->u.call.args);
619  break;
620  case CLOCK_VALUE:
621  clock_ticksize = event->u.clock.ticksize;
622  if (!clock_running && !suppress_midi_clock &&
623  !external_midi_clock) {
624  clock_running = TRUE;
625  midi_start();
626  clock_tick(seq, 0L);
627  }
628  break;
629  case MACCTRL_VALUE:
630  if (!enabled) break;
631  seq_midi_ctrl(seq, voice, event->u.macctrl.ctrl_number,
632  event->u.macctrl.value);
633  break;
634  case MACRO_VALUE: {
635  if (!enabled) break;
636  send_macro(event->u.macro.definition, voice,
637  event->u.macro.parameter, -1, 0,
638  event->nline);
639  break;
640  }
641  case CTRLRAMP_VALUE:
642  case DEFRAMP_VALUE: {
643  int from, to;
644  if (!enabled) break;
645 
646  step = event->u.ramp.step;
647  if (event->value == CTRLRAMP_VALUE) {
648  from = event->u.ramp.u.ctrl.from_value;
649  to = event->u.ramp.u.ctrl.to_value;
650  } else {
651  from = event->u.ramp.u.def.parameter[
652  event->u.ramp.u.def.parm_num];
653  to = event->u.ramp.u.def.to_value;
654  }
655  delta = to - from;
656  increment = delta;

757 /* send_macro -- instantiate macro and send it */
758 /*
759  * note: to support ramping, "value" is used in place of
760  * parameter["parm_num"]
761  */
762 /*private*/
763 void send_macro(ptr, voice, parameter, parm_num, value, nline)
764  register unsigned char *ptr;
765  int voice;
766  short parameter[];
767  int parm_num;
768  int value;
769  int nline;
770 {
771  register unsigned char code, *loc;
772  while ((code = *ptr++)) {
773  loc = ptr + *ptr;
774  ptr++;
775  if (code <= nmacroparms) {
776  code--;
777  *loc = (code == parm_num ? value : parameter[code]) & 0x7f;
778  } else if (code == nmacroparms + 1) {
779  /* take old high order bits and OR in 4 voice bits */
780  *loc = (*loc & 0xF0) | ((voice - 1) & 0xF);
781  } else {
782  code -= (nmacroparms + 2);
783  *loc = ((code == parm_num ? value : parameter[code]) >> 7) & 0x7F;
784  }
785  }
786  if (ptr[1] == MIDI_SYSEX) {
787  midi_exclusive(ptr + 1);
788  } else {
789  /* make sure user didn't try to send more than 3 bytes. This test
790  * could be done at sequence read time, but it's tricky because the
791  * first byte could be a parameter, so in general you need to
792  * plug the actual parameters into the message and then do the test.
793  * Currently, this is the only place parameters are plugged in.
794  */
795  if (*ptr > 3) {
796  gprintf(ERROR,
797  "Non-sysex macro longer than 3 bytes ignored, line %d.\n",
798  nline);
799  } else {
800  midi_write((int) *ptr, MIDI_PORT(voice), ptr[1], ptr[2], ptr[3]);
