403 // Process the input with 2 buffers available at a time
404 // buffer1 will be written upon return
405 // buffer2 will be passed as buffer1 on the next call
406 bool EffectCompressor::TwoBufferProcessPass1(float *buffer1, sampleCount len1, float *buffer2, sampleCount len2)
407 {
408  int i;
409 
410  // If buffers are bigger than allocated, then abort
411  // (this should never happen, but if it does, we don't want to crash)
412  if((len1 > mFollowLen) || (len2 > mFollowLen))
413  return false;
414 
415  // This makes sure that the initial value is well-chosen
416  // buffer1 == NULL on the first and only the first call
417  if (buffer1 == NULL) {
418  // Initialize the mLastLevel to the peak level in the first buffer
419  // This avoids problems with large spike events near the beginning of the track
420  mLastLevel = mThreshold;
421  for(i=0; i<len2; i++) {
422  if(mLastLevel < fabs(buffer2[i]))
423  mLastLevel = fabs(buffer2[i]);
424  }
425  }
426 
427  // buffer2 is NULL on the last and only the last call
428  if(buffer2 != NULL) {
429  Follow(buffer2, mFollow2, len2, mFollow1, len1);
430  }
431 
432  if(buffer1 != NULL) {
433  for (i = 0; i < len1; i++) {
434  buffer1[i] = DoCompression(buffer1[i], mFollow1[i]);
435  }
436  }
437 
438 
439 #if 0
440  // Copy the envelope over the track data (for debug purposes)
441  memcpy(buffer1, mFollow1, len1*sizeof(float));
442 #endif
443 
444  // Rotate the buffer pointers
445  float *tmpfloat = mFollow1;
446  mFollow1 = mFollow2;
447  mFollow2 = tmpfloat;
448 
449  return true;

464 {
465  // Recompute the RMS sum periodically to prevent accumulation of rounding errors

472 float EffectCompressor::AvgCircle(float value)
473 {
474  float level;
475 
476  // Calculate current level from root-mean-squared of
477  // circular buffer ("RMS")
478  mRMSSum -= mCircle[mCirclePos];
479  mCircle[mCirclePos] = value*value;
480  mRMSSum += mCircle[mCirclePos];
481  level = sqrt(mRMSSum/mCircleSize);
482  mCirclePos = (mCirclePos+1)%mCircleSize;
483 
484  return level;
485 }

487 void EffectCompressor::Follow(float *buffer, float *env, int len, float *previous, int previous_len)
488 {
489  /*
490 
491  "Follow"ing algorithm by Roger B. Dannenberg, taken from
492  Nyquist. His description follows. -DMM
493 
494  Description: this is a sophisticated envelope follower.
495  The input is an envelope, e.g. something produced with
496  the AVG function. The purpose of this function is to
497  generate a smooth envelope that is generally not less
498  than the input signal. In other words, we want to "ride"
499  the peaks of the signal with a smooth function. The
500  algorithm is as follows: keep a current output value
501  (called the "value"). The value is allowed to increase
502  by at most rise_factor and decrease by at most fall_factor.
503  Therefore, the next value should be between
504  value * rise_factor and value * fall_factor. If the input
505  is in this range, then the next value is simply the input.
506  If the input is less than value * fall_factor, then the
507  next value is just value * fall_factor, which will be greater
508  than the input signal. If the input is greater than value *
509  rise_factor, then we compute a rising envelope that meets
510  the input value by working bacwards in time, changing the
511  previous values to input / rise_factor, input / rise_factor^2,
512  input / rise_factor^3, etc. until this NEW envelope intersects
513  the previously computed values. There is only a limited buffer
514  in which we can work backwards, so if the NEW envelope does not
515  intersect the old one, then make yet another pass, this time
516  from the oldest buffered value forward, increasing on each
517  sample by rise_factor to produce a maximal envelope. This will
518  still be less than the input.
519 
520  The value has a lower limit of floor to make sure value has a
521  reasonable positive value from which to begin an attack.
522  */
523  int i;
524  double level,last;
525 
526  if(!mUsePeak) {
527  // Update RMS sum directly from the circle buffer
528  // to avoid accumulation of rounding errors
529  FreshenCircle();
530  }
531  // First apply a peak detect with the requested decay rate
532  last = mLastLevel;
533  for(i=0; i<len; i++) {
534  if(mUsePeak)
535  level = fabs(buffer[i]);
536  else // use RMS
537  level = AvgCircle(buffer[i]);
538  // Don't increase gain when signal is continuously below the noise floor
539  if(level < mNoiseFloor) {
540  mNoiseCounter++;
541  } else {
542  mNoiseCounter = 0;
543  }
544  if(mNoiseCounter < 100) {
545  last *= mDecayFactor;
546  if(last < mThreshold)
547  last = mThreshold;
548  if(level > last)
549  last = level;
550  }
551  env[i] = last;
552  }
553  mLastLevel = last;
554 
555  // Next do the same process in reverse direction to get the requested attack rate
556  last = mLastLevel;
557  for(i=len-1; i>=0; i--) {
558  last *= mAttackInverseFactor;
559  if(last < mThreshold)
560  last = mThreshold;
561  if(env[i] < last)
562  env[i] = last;
563  else
564  last = env[i];
565  }
566 
567  if((previous != NULL) && (previous_len > 0)) {
568  // If the previous envelope was passed, propagate the rise back until we intersect
569  for(i=previous_len-1; i>0; i--) {
570  last *= mAttackInverseFactor;
571  if(last < mThreshold)
572  last = mThreshold;
573  if(previous[i] < last)
574  previous[i] = last;
575  else // Intersected the previous envelope buffer, so we are finished
576  return;
577  }
578  // If we can't back up far enough, project the starting level forward
579  // until we intersect the desired envelope
580  last = previous[0];
581  for(i=1; i<previous_len; i++) {
582  last *= mAttackFactor;
583  if(previous[i] > last)
584  previous[i] = last;
585  else // Intersected the desired envelope, so we are finished
586  return;
587  }
588  // If we still didn't intersect, then continue ramp up into current buffer
589  for(i=0; i<len; i++) {
590  last *= mAttackFactor;
591  if(buffer[i] > last)
592  buffer[i] = last;
593  else // Finally got an intersect
594  return;
595  }
596  // If we still didn't intersect, then reset mLastLevel
597  mLastLevel = last;
598  }
599 }

601 float EffectCompressor::DoCompression(float value, double env)
602 {
603  float out;
604  if(mUsePeak) {
605  // Peak values map 1.0 to 1.0 - 'upward' compression
606  out = value * pow(1.0/env, mCompression);
607  } else {
608  // With RMS-based compression don't change values below mThreshold - 'downward' compression
609  out = value * pow(mThreshold/env, mCompression);
610  }
611 
612  // Retain the maximum value for use in the normalization pass
613  if(mMax < fabs(out))
614  mMax = fabs(out);
615 
616  return out;
617 }
