189  wxASSERT(!paramValue.IsType(wxT("null")));
190 
191  ParamValueMap::iterator iter = mParams.find(paramName);
192  if (iter == mParams.end())

195  return false;
196  }
197 
198  Validator &validator = mType.GetSignature().GetValidator(iter->first);
199  if (!validator.Validate(paramValue))

201  Error(wxT("Invalid value for parameter '")
