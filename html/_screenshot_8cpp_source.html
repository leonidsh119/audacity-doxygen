456 bool ScreenFrame::ProcessEvent(wxEvent & e)
457 {
458  int id = e.GetId();
459 
460  if (mDelayCheckBox &&
461  mDelayCheckBox->GetValue() &&
462  e.IsCommandEvent() &&
463  e.GetEventType() == wxEVT_COMMAND_BUTTON_CLICKED &&
464  id >= IdAllDelayedEvents && id <= IdLastDelayedEvent &&
465  e.GetEventObject() != NULL) {
466  ScreenFrameTimer *timer = new ScreenFrameTimer(this, e);
467  timer->Start(5000, true);
468  return true;
469  }
470 
471  if (e.IsCommandEvent() && e.GetEventObject() == NULL) {
472  e.SetEventObject(this);
473  }
474  return wxFrame::ProcessEvent(e);
475 }
