278 bool
279 PluginInputDomainAdapter::Impl::initialise(size_t channels, size_t stepSize, size_t blockSize)
280 {
281  if (m_plugin->getInputDomain() == TimeDomain) {
282 
283  m_stepSize = int(stepSize);
284  m_blockSize = int(blockSize);
285  m_channels = int(channels);
286 
287  return m_plugin->initialise(channels, stepSize, blockSize);
288  }
289 
290  if (blockSize < 2) {
291  std::cerr << "ERROR: PluginInputDomainAdapter::initialise: blocksize < 2 not supported" << std::endl;
292  return false;
293  }
294 
295  if (blockSize & (blockSize-1)) {
296  std::cerr << "ERROR: PluginInputDomainAdapter::initialise: non-power-of-two\nblocksize " << blockSize << " not supported" << std::endl;
297  return false;
298  }
299 
300  if (m_channels > 0) {
301  for (int c = 0; c < m_channels; ++c) {
302  delete[] m_freqbuf[c];
303  }
304  delete[] m_freqbuf;
305 #ifdef HAVE_FFTW3
306  if (m_plan) {
307  fftw_destroy_plan(m_plan);
308  fftw_free(m_ri);
309  fftw_free(m_cbuf);
310  m_plan = 0;
311  }
312 #else
313  delete[] m_ri;
314  delete[] m_ro;
315  delete[] m_io;
316 #endif
317  delete m_window;
318  }
319 
320  m_stepSize = int(stepSize);
321  m_blockSize = int(blockSize);
322  m_channels = int(channels);
323 
324  m_freqbuf = new float *[m_channels];
325  for (int c = 0; c < m_channels; ++c) {
326  m_freqbuf[c] = new float[m_blockSize + 2];
327  }
328 
329  m_window = new Window<double>(convertType(m_windowType), m_blockSize);
330 
331 #ifdef HAVE_FFTW3
332  m_ri = (double *)fftw_malloc(blockSize * sizeof(double));
333  m_cbuf = (fftw_complex *)fftw_malloc((blockSize/2 + 1) * sizeof(fftw_complex));
334  m_plan = fftw_plan_dft_r2c_1d(blockSize, m_ri, m_cbuf, FFTW_MEASURE);
335 #else
336  m_ri = new double[m_blockSize];
337  m_ro = new double[m_blockSize];
338  m_io = new double[m_blockSize];
339 #endif
340 
341  m_processCount = 0;
342 
343  return m_plugin->initialise(channels, stepSize, blockSize);
344 }

381 size_t
382 PluginInputDomainAdapter::Impl::makeBlockSizeAcceptable(size_t blockSize) const
383 {
384  if (blockSize < 2) {
385 
386  std::cerr << "WARNING: PluginInputDomainAdapter::initialise: blocksize < 2 not" << std::endl
387  << "supported, increasing from " << blockSize << " to 2" << std::endl;
388  blockSize = 2;
389 
390  } else if (blockSize & (blockSize-1)) {
391 
392 #ifdef HAVE_FFTW3
393  // not an issue with FFTW
394 #else
395 
396  // not a power of two, can't handle that with our built-in FFT
397  // implementation
398 
399  size_t nearest = blockSize;
400  size_t power = 0;
401  while (nearest > 1) {
402  nearest >>= 1;
403  ++power;
404  }
405  nearest = 1;
406  while (power) {
407  nearest <<= 1;
408  --power;
409  }
410 
411  if (blockSize - nearest > (nearest*2) - blockSize) {
412  nearest = nearest*2;
413  }
414 
415  std::cerr << "WARNING: PluginInputDomainAdapter::initialise: non-power-of-two\nblocksize " << blockSize << " not supported, using blocksize " << nearest << " instead" << std::endl;
416  blockSize = nearest;
417 
418 #endif
419  }
420 
421  return blockSize;
422 }

504 Plugin::FeatureSet
505 PluginInputDomainAdapter::Impl::processShiftingTimestamp(const float *const *inputBuffers,
506  RealTime timestamp)
507 {
508  if (m_method == ShiftTimestamp) {
509  timestamp = timestamp + getTimestampAdjustment();
510  }
511 
512  for (int c = 0; c < m_channels; ++c) {
513 
514  m_window->cut(inputBuffers[c], m_ri);
515 
516  for (int i = 0; i < m_blockSize/2; ++i) {
517  // FFT shift
518  double value = m_ri[i];
519  m_ri[i] = m_ri[i + m_blockSize/2];
520  m_ri[i + m_blockSize/2] = value;
521  }
522 
523 #ifdef HAVE_FFTW3
524  fftw_execute(m_plan);
525 
526  for (int i = 0; i <= m_blockSize/2; ++i) {
527  m_freqbuf[c][i * 2] = float(m_cbuf[i][0]);
528  m_freqbuf[c][i * 2 + 1] = float(m_cbuf[i][1]);
529  }
530 #else
531  fft(m_blockSize, false, m_ri, 0, m_ro, m_io);
532 
533  for (int i = 0; i <= m_blockSize/2; ++i) {
534  m_freqbuf[c][i * 2] = float(m_ro[i]);
535  m_freqbuf[c][i * 2 + 1] = float(m_io[i]);
536  }
537 #endif
538  }
539 
540  return m_plugin->process(m_freqbuf, timestamp);

543 Plugin::FeatureSet
544 PluginInputDomainAdapter::Impl::processShiftingData(const float *const *inputBuffers,
545  RealTime timestamp)
546 {
547  if (m_processCount == 0) {
548  if (!m_shiftBuffers) {
549  m_shiftBuffers = new float *[m_channels];
550  for (int c = 0; c < m_channels; ++c) {
551  m_shiftBuffers[c] = new float[m_blockSize + m_blockSize/2];
552  }
553  }
554  for (int c = 0; c < m_channels; ++c) {
555  for (int i = 0; i < m_blockSize + m_blockSize/2; ++i) {
556  m_shiftBuffers[c][i] = 0.f;
557  }
558  }
559  }
560 
561  for (int c = 0; c < m_channels; ++c) {
562  for (int i = m_stepSize; i < m_blockSize + m_blockSize/2; ++i) {
563  m_shiftBuffers[c][i - m_stepSize] = m_shiftBuffers[c][i];
564  }
565  for (int i = 0; i < m_blockSize; ++i) {
566  m_shiftBuffers[c][i + m_blockSize/2] = inputBuffers[c][i];
567  }
568  }
569 
570  for (int c = 0; c < m_channels; ++c) {
571 
572  m_window->cut(m_shiftBuffers[c], m_ri);
573 
574  for (int i = 0; i < m_blockSize/2; ++i) {
575  // FFT shift
576  double value = m_ri[i];
577  m_ri[i] = m_ri[i + m_blockSize/2];
578  m_ri[i + m_blockSize/2] = value;
579  }
580 
581 #ifdef HAVE_FFTW3
582  fftw_execute(m_plan);
583 
584  for (int i = 0; i <= m_blockSize/2; ++i) {
