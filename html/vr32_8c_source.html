497 static bool do_input_stage(rate_t * p, int stage_num, int sign, int min_stage_num)
498 {
499  int i = 0;
500  float * dest;
501  stage_t * s = &p->stages[stage_num];
502  stage_t * s1 = &p->stages[stage_num - sign];
503  float const * src = (float *)fifo_read_ptr(&s1->fifo) + HALF_FIR_LEN_2;
504  int len = shiftr(fifo_occupancy(&s1->fifo) - HALF_FIR_LEN_2 * 2, sign);
505  int already_done = fifo_occupancy(&s->fifo) - s->preload;
506  if ((len -= already_done) <= 0)
507  return false;
508  src += shiftl(already_done, sign);
509 
510  dest = fifo_reserve(&s->fifo, len);
511  if (stage_num < 0) for (; i < len; ++src)
512  dest[i++] = double_fir0(src), dest[i++] = double_fir1(src);
513  else {
514  bool should_be_fast = p->stage_inc;
515  if (!s->x_fade_len && stage_num == p->switch_stage_num) {
516  p->switch_stage_num = 0;
517  if (s->is_fast != should_be_fast) {
518  s->x_fade_len = 1 << FADE_LEN_BITS, s->is_fast = should_be_fast, ++p->xfade;
519  lsx_debug("xfade level %i, inc?=%i", stage_num, p->stage_inc);
520  }
521  }
522  if (s->x_fade_len) {
523  float const * vol1 = fade_coefs + (s->x_fade_len << 1);
524  float const * vol2 = fade_coefs + (((1 << FADE_LEN_BITS) - s->x_fade_len) << 1);
525  int n = min(len, s->x_fade_len);
526  /*lsx_debug("xfade level %i, inc?=%i len=%i n=%i", stage_num, p->stage_inc, s->x_fade_len, n);*/
527  if (should_be_fast)
528  for (; i < n; vol2 += 2, vol1 -= 2, src += 2)
529  dest[i++] = *vol1 * fast_half_fir(src) + *vol2 * half_fir(src);
530  else for (; i < n; vol2 += 2, vol1 -= 2, src += 2)
531  dest[i++] = *vol2 * fast_half_fir(src) + *vol1 * half_fir(src);
532  s->x_fade_len -= n;
533  p->xfade -= !s->x_fade_len;
534  }
535  if (stage_num < min_stage_num)
536  for (; i < len; dest[i++] = fast_half_fir(src), src += 2);
537  else for (; i < len; dest[i++] = half_fir(src), src += 2);
538  }
539  if (p->flushing > 0)
540  stage_preload(s);
541  return true;
542 }

544 static int vr_process(rate_t * p, int olen0)
545 {
546  assert(p->num_stages > 0);
547  if (p->default_io_ratio)
548  vr_set_io_ratio(p, p->default_io_ratio, 0);
549  {
550  float * output = fifo_reserve(&p->output_fifo, olen0);
551  int j, odone0 = 0, min_stage_num = p->current.stage_num;
552  int occupancy0, max_stage_num = min_stage_num;
553  if (p->fade_len) {
554  min_stage_num = min(min_stage_num, p->fadeout.stage_num);
555  max_stage_num = max(max_stage_num, p->fadeout.stage_num);
556  }
557 
558  for (j = min(min_stage_num, 0); j <= max_stage_num; ++j)
559  if (j && !do_input_stage(p, j, j < 0? -1 : 1, min_stage_num))
560  break;
561  if (p->flushing > 0)
562  p->flushing = -1;
563 
564  occupancy0 = shiftl(max(0,stage_occupancy(&p->stages[max_stage_num])), max_stage_num);
565  p->current.len = shiftr(occupancy0, p->current.stage_num);
566  p->current.input = stage_read_p(&p->stages[p->current.stage_num]);
567  if (p->fade_len) {
568  p->fadeout.len = shiftr(occupancy0, p->fadeout.stage_num);
569  p->fadeout.input = stage_read_p(&p->stages[p->fadeout.stage_num]);
570  }
571 
572  while (odone0 < olen0) {
573  int odone, odone2, olen = olen0 - odone0, stage_dif = 0, shift;
574  float buf[64 << 1];
575 
576  olen = min(olen, (int)(AL(buf) >> 1));
577  if (p->slew_len)
578  olen = min(olen, p->slew_len);
579  else if (p->new_io_ratio) {
580  set_step(&p->current, p->new_io_ratio);
581  set_step(&p->fadeout, p->new_io_ratio);
582  p->fadeout.step_step.all = p->current.step_step.all = 0;
583  p->new_io_ratio = 0;
584  }
585  if (!p->flushing && !p->fade_len && !p->xfade) {
586  if (p->current.is_d) {
587  if (INT(p->current.step) && FRAC(p->current.step))
588  stage_dif = 1, ++max_stage_num;
589  else if (!INT(p->current.step) && FRAC(p->current.step) < (1u << 31))
590  stage_dif = -1, --min_stage_num;
591  } else if (INT(p->current.step) > 1 && FRAC(p->current.step))
592  stage_dif = 1, ++max_stage_num;
593  }
594  if (stage_dif) {
595  int n = p->current.stage_num + stage_dif;
596  if (n >= p->num_stages)
597  --max_stage_num;
598  else {
599  p->stage_inc = stage_dif > 0;
600  p->fadeout = p->current;
601  p->current.stage_num += stage_dif;
602  if (!p->stage_inc)
603  p->switch_stage_num = p->current.stage_num;
604  if ((p->current.stage_num < 0 && stage_dif < 0) ||
605  (p->current.stage_num > 0 && stage_dif > 0)) {
606  stage_t * s = &p->stages[p->current.stage_num];
607  fifo_clear(&s->fifo);
608  stage_preload(s);
609  s->is_fast = false;
610  do_input_stage(p, p->current.stage_num, stage_dif, p->current.stage_num);
611  }
612  if (p->current.stage_num > 0 && stage_dif < 0) {
613  int idone = INT(p->current.at);
614  stage_t * s = &p->stages[p->current.stage_num];
615  fifo_trim_to(&s->fifo, 2 * HALF_FIR_LEN_2 + idone + (POLY_FIR_LEN_D >> 1));
616  do_input_stage(p, p->current.stage_num, 1, p->current.stage_num);
617  }
618  enter_new_stage(p, occupancy0);
619  shift = -stage_dif;
620 #define lshift(x,by) (x)=(by)>0?(x)<<(by):(x)>>-(by)
621  lshift(p->current.at.all, shift);
622  shift += p->fadeout.is_d - p->current.is_d;
623  lshift(p->current.step.all, shift);
624  lshift(p->current.step_step.all, shift);
625  p->fade_len = AL(fade_coefs) - 1;
626  lsx_debug("switch from stage %i to %i, x2 from %i to %i", p->fadeout.stage_num, p->current.stage_num, p->fadeout.is_d, p->current.is_d);
627  }
628  }
629 
630  if (p->fade_len) {
631  float const * vol1 = fade_coefs + p->fade_len;
632  float const * vol2 = fade_coefs + (iAL(fade_coefs) - 1 - p->fade_len);
633  int olen2 = (olen = min(olen, p->fade_len >> 1)) << 1;
634 
635  /* x2 is more fine-grained so may fail to produce a pair of samples
636  * where x1 would not (the x1 second sample is a zero so is always
637  * available). So do x2 first, then feed odone to the second one. */
638  memset(buf, 0, sizeof(*buf) * (size_t)olen2);
639  if (p->current.is_d && p->fadeout.is_d) {
640  odone = poly_fir_fade_d(&p->current, vol1,-1, buf, olen2);
641  odone2 = poly_fir_fade_d(&p->fadeout, vol2, 1, buf, odone);
642  } else if (p->current.is_d) {
643  odone = poly_fir_fade_d(&p->current, vol1,-1, buf, olen2);
644  odone2 = poly_fir_fade_u(&p->fadeout, vol2, 2, buf, odone);
645  } else {
646  assert(p->fadeout.is_d);
647  odone = poly_fir_fade_d(&p->fadeout, vol2, 1, buf, olen2);
648  odone2 = poly_fir_fade_u(&p->current, vol1,-2, buf, odone);
649  }
650  assert(odone == odone2);
651  (void)odone2;
652  p->fade_len -= odone;
653  if (!p->fade_len) {
654  if (p->stage_inc)
655  p->switch_stage_num = min_stage_num++;
656  else
657  --max_stage_num;
658  }
659  half_iir(&p->halfer, &output[odone0], buf, odone >>= 1);
660  }
661  else if (p->current.is_d) {
662  odone = poly_fir_d(&p->current, buf, olen << 1) >> 1;
663  half_iir(&p->halfer, &output[odone0], buf, odone);
664  }
665  else {
666  odone = poly_fir_u(&p->current, &output[odone0], olen);
667  if (p->num_stages0)
668  half_phase(&p->halfer, &output[odone0], odone);
669  }
670  odone0 += odone;
671  if (p->slew_len)
672  p->slew_len -= odone;
673  if (odone != olen)
674  break; /* Need more input. */
675  } {
676  int from = max(0, max_stage_num), to = min(0, min_stage_num);
677  int i, idone = shiftr(INT(p->current.at), from - p->current.stage_num);
678  INT(p->current.at) -= shiftl(idone, from - p->current.stage_num);
679  if (p->fade_len)
680  INT(p->fadeout.at) -= shiftl(idone, from - p->fadeout.stage_num);
681  for (i = from; i >= to; --i, idone <<= 1)
682  fifo_read(&p->stages[i].fifo, idone, NULL);
683  }
684  fifo_trim_by(&p->output_fifo, olen0 - odone0);
685  return odone0;
686  }
687 }
