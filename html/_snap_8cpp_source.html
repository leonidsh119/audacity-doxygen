207 // Find the SnapPoint nearest to time t
208 size_t SnapManager::Find(double t)
209 {
210  size_t cnt = mSnapPoints.size();
211  size_t index = Find(t, 0, cnt);
212 
213  // At this point, either index is the closest, or the next one
214  // to the right is. Keep moving to the right until we get a
215  // different value
216  size_t next = index + 1;
217  while (next + 1 < cnt && Get(next) == Get(index))
218  {
219  next++;
220  }
221 
222  // Now return whichever one is closer to time t
223  if (next < cnt && PixelDiff(t, next) < PixelDiff(t, index))
224  {
225  return next;
226  }
227 
228  return index;
229 }

231 // Helper: performs snap-to-points for Snap(). Returns true if a snap happened.
232 bool SnapManager::SnapToPoints(Track *currentTrack,
233  double t,
234  bool rightEdge,
235  double *outT)
236 {
237  *outT = t;
238 
239  size_t cnt = mSnapPoints.size();
240  if (cnt == 0)
241  {
242  return false;
243  }
244 
245  // Find the nearest SnapPoint
246  size_t index = Find(t);
247 
248  // If it's too far away, just give up now
249  if (PixelDiff(t, index) >= mPixelTolerance)
250  {
251  return false;
252  }
253 
254  // Otherwise, search left and right for all of the points
255  // within the allowed range.
256  size_t left = index;
257  size_t right = index;
258  size_t i;
259 
260  while (left > 0 && PixelDiff(t, left - 1) < mPixelTolerance)
261  {
262  left--;
263  }
264 
265  while (right < cnt - 1 && PixelDiff(t, right + 1) < mPixelTolerance)
266  {
267  right++;
268  }
269 
270  if (left == index && right == index)
271  {
272  // Awesome, there's only one point that matches!
273  *outT = Get(index);
274  return true;
275  }
276 
277  size_t indexInThisTrack = -1;
278  size_t countInThisTrack = 0;
279  for (i = left; i <= right; ++i)
280  {
281  if (mSnapPoints[i].track == currentTrack)
282  {
283  indexInThisTrack = i;
284  countInThisTrack++;
285  }
286  }
287 
288  if (countInThisTrack == 1)
289  {
290  // Cool, only one of the points is in the same track, so
291  // we'll use that one.
292  *outT = Get(indexInThisTrack);
293  return true;
294  }
295 
296  if (Get(right) - Get(left) < mEpsilon)
297  {
298  // OK, they're basically the same point
299  if (rightEdge)
300  {
301  *outT = Get(right); // Return rightmost
302  }
303  else {
304  *outT = Get(left); // Return leftmost
305  }
306  return true;
307  }
308 
309  // None of the points matched, bummer.
310  return false;
311 }

314  double t,
315  bool rightEdge,
316  double *outT,
317  bool *snappedPoint,
318  bool *snappedTime)
319 {
320  // Check to see if we need to reinitialize
321  Reinit();
322

324  *outT = t;
325  *snappedPoint = SnapToPoints(currentTrack, t, rightEdge, outT);
326 
327  // Now snap to the time grid
328  *snappedTime = false;
329  if (mSnapToTime)
330  {
331  if (*snappedPoint)

333  // Since mSnapPoints only contains points on the grid, we're done
334  *snappedTime = true;
