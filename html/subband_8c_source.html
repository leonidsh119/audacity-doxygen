38 static void create_dct_matrix(FLOAT filter[16][32])
39 {
40  register int i, k;
41 
42  for (i = 0; i < 16; i++)
43  for (k = 0; k < 32; k++) {
44  if ((filter[i][k] = 1e9 * cos((FLOAT) ((2 * i + 1) * k * PI64))) >= 0)
45  modf(filter[i][k] + 0.5, &filter[i][k]);
46  else
47  modf(filter[i][k] - 0.5, &filter[i][k]);
