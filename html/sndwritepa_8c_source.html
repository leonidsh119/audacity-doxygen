40 /* Previously, Nyquist would wrap samples that
41  * overflowed -- this produces horrible output,
42  * but makes it really easy to detect clipping,
43  * which I found helpful in my own work and good
44  * for students too since the effect is impossible
45  * to ignore. Now that Nyquist is doing IO to
46  * libraries that clip, we're going to artificially
47  * generate the wrapping here. This is floating point
48  * wrapping, so +1.0 does not wrap (it would if it
49  * were an integer since the maximum sample value for
50  * 16-bit data is a bit less than 1.) Since this is extra
51  * overhead, I'm trying to be a bit clever by using
52  * the compare to max_sample to eliminate compares
53  * for clipping in the common case.
54  *
55  * INPUTS: max_sample -- initially 0.0
56  * threshold -- initially 0.0
57  * s -- the value of the current sample
58  * x -- if s has to be wrapped, put the value here
59  */
60 #define COMPUTE_MAXIMUM_AND_WRAP(x) \
61  if (s > threshold) { \
62  if (s > max_sample) { \
63  max_sample = s; \
64  threshold = min(1.0, s); \
65  } \
66  if (s > 1.0) { \
67  s = fmod(s + 1.0, 2.0) - 1.0; \
68  (x) = s; \
69  } \
70  } else if (s < -threshold) { \
71  if (s < -max_sample) { \
72  max_sample = -s; \
73  threshold = min(1.0, -s); \
74  } \
75  if (s < -1.0) { \
76  s = -(fmod(-s + 1.0, 2.0) - 1.0); \
77  (x) = s; \
78  } \
79  }
80 // the s < -threshold case is tricky:
81 // flip the signal, do the wrap, flip again
82 // in order to pass positive values to fmod

91  max_sample = s; \

672 sample_type sound_save_array(LVAL sa, long n, SF_INFO *sf_info,
673  SNDFILE *sndfile, float *buf, long *ntotal, PaStream *audio_stream)
674 {
675  long i, chans;
676  float *float_bufp;
677  sound_state_type state;
678  double start_time = HUGE_VAL;
679  LVAL sa_copy;
680  long debug_unit; /* print messages at intervals of this many samples */
681  long debug_count; /* next point at which to print a message */
682  sample_type max_sample = 0.0F;
683  sample_type threshold = 0.0F;
684  /* cvtfn_type cvtfn; jlh */
685 
686  *ntotal = 0;
687 
688  /* THE ALGORITHM: first merge floating point samples from N channels
689  * into consecutive multi-channel frames in buf. Then, treat buf
690  * as just one channel and use one of the cvt_to_* functions to
691  * convert the data IN PLACE in the buffer (this is ok because the
692  * converted data will never take more space than the original 32-bit
693  * floats, so the converted data will not overwrite any floats before
694  * the floats are converted
695  */
696 
697  /* if snd_expr was simply a symbol, then sa now points to
698  a shared sound_node. If we read samples from it, then
699  the sounds bound to the symbol will be destroyed, so
700  copy it first. If snd_expr was a real expression that
701  computed a new value, then the next garbage collection
702  will reclaim the sound array. See also sound_save_sound()
703  */
704 
705  chans = getsize(sa);
706  if (chans > MAX_SND_CHANNELS) {
707  xlerror("sound_save: too many channels", sa);
708  free(buf);
709  sf_close(sndfile);
710  }
711  xlprot1(sa);
712  sa_copy = newvector(chans);
713  xlprot1(sa_copy);
714 
715  /* Why do we copy the array into an xlisp array instead of just
716  * the state[i] array? Because some of these sounds may reference
717  * the lisp heap. We must put the sounds in an xlisp array so that
718  * the gc will find and mark them. xlprot1(sa_copy) makes the array
719  * visible to gc.
720  */
721  for (i = 0; i < chans; i++) {
722  sound_type s = getsound(getelement(sa, i));
723  setelement(sa_copy, i, cvsound(sound_copy(s)));
724  }
725  sa = sa_copy; /* destroy original reference to allow GC */
726 
727  state = (sound_state_type) malloc(sizeof(sound_state_node) * chans);
728  for (i = 0; i < chans; i++) {
729  state[i].sound = getsound(getelement(sa, i));
730  state[i].scale = state[i].sound->scale;
731 D nyquist_printf("save scale factor %ld = %g\n", i, state[i].scale);
732  state[i].terminated = false;
733  state[i].cnt = 0; /* force a fetch */
734  start_time = min(start_time, state[i].sound->t0);
735  }
736 
737  for (i = 0; i < chans; i++) {
738  if (state[i].sound->t0 > start_time)
739  sound_prepend_zeros(state[i].sound, start_time);
740  }
741 
742  debug_unit = debug_count = (long) max(sf_info->samplerate, 10000.0);
743 
744  sound_frames = 0;
745  sound_srate = sf_info->samplerate;
746  while (n > 0) {
747  /* keep the following information for each sound:
748  has it terminated?
749  pointer to samples
750  number of samples remaining in block
751  scan to find the minimum remaining samples and
752  output that many in an inner loop. Stop outer
753  loop if all sounds have terminated
754  */
755  int terminated = true;
756  int togo = n;
757  int j;
758 
759  oscheck();
760 
761  for (i = 0; i < chans; i++) {
762  if (state[i].cnt == 0) {
763  if (sndwrite_trace) {
764  nyquist_printf("CALLING SOUND_GET_NEXT ON CHANNEL %ld (%lx)\n",
765  i, (unsigned long) state[i].sound); /* jlh 64 bit issue */
766  sound_print_tree(state[i].sound);
767  }
768  state[i].ptr = sound_get_next(state[i].sound,
769  &(state[i].cnt))->samples;
770  if (sndwrite_trace) {
771  nyquist_printf("RETURNED FROM CALL TO SOUND_GET_NEXT ON CHANNEL %ld\n", i);
772  }
773  if (state[i].ptr == zero_block->samples) {
774  state[i].terminated = true;
775  }
776  }
777  if (!state[i].terminated) terminated = false;
778  togo = min(togo, state[i].cnt);
779  }
780 
781  if (terminated) break;
782 
783  float_bufp = (float *) buf;
784  if (is_pcm(sf_info)) {
785  for (j = 0; j < togo; j++) {
786  for (i = 0; i < chans; i++) {
787  float s = (float) (*(state[i].ptr++) * (float) state[i].scale);
788  COMPUTE_MAXIMUM_AND_WRAP(s);
789  *float_bufp++ = s;
790  }
791  }
792  } else {
793  for (j = 0; j < togo; j++) {
794  for (i = 0; i < chans; i++) {
795  float s = (float) (*(state[i].ptr++) * (float) state[i].scale);
796  COMPUTE_MAXIMUM();
797  *float_bufp++ = s;
798  }
799  }
800  }
801  /* Here we have interleaved floats. Before converting to the sound
802  file format, call PortAudio to play them. */
803  if (audio_stream) {
804  PaError err = Pa_WriteStream(audio_stream, buf, togo);
805  if (err) {
806  printf("Pa_WriteStream error %d\n", err);
807  }
808  sound_frames += togo;
809  }
810  if (sndfile) sf_writef_float(sndfile, buf, togo);
811 
812  n -= togo;
813  for (i = 0; i < chans; i++) {
814  state[i].cnt -= togo;
815  }
816  *ntotal += togo;
817  if (*ntotal > debug_count) {
818  gprintf(TRANS, " %ld ", *ntotal);
819  fflush(stdout);
820  debug_count += debug_unit;
821  }
822  }
823  gprintf(TRANS, "total samples: %ld x %ld channels\n",
824  *ntotal, chans);
825 
826  /* references to sounds are shared by sa_copy and state[].
827  * here, we dispose of state[], allowing GC to do the
828  * sound_unref call that frees the sounds. (Freeing them now
829  * would be a bug.)
830  */
831  free(state);
832  xlpop();
833  return max_sample;
834 }
