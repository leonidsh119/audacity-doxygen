20 /* Notes on ascii input:
21 
22 Input is complicated because different systems have varying input models,
23 especially with regard to handling ^C. The CMT model handles ^C and ^G as
24 special characters, and these do not cause software interrupts. Also, the
25 lowest level of the CMT model does not support line editing: Every character
26 is acted upon immediately. This has two implications:
27 (1) CMT must "read ahead" looking for ^C and ^G characters. This is handled
28 by the check_aborted() procedure, which reads characters into the
29 type_ahead[] array.
30 (2) CMT must do its own line editing. This is handled by the ggets() routine.
31 
32 A number of functions support ascii input, only some of which are visible
33 to the application programmer. Let's start at the top-level and work down;
34 each of the following calls the routine below it:
35 
36 ggets() - gets a string with line editing support. This function is fairly
37 machine independent, except for some backspace-and-erase control character
38 code sequences.
39 
40 ggetchar() - gets a raw character. This function calls wait_ascii()
41  and echoes it. Note that it may return ABORT_CHAR or BREAK_CHAR.
42 
43 wait_ascii() - gets a raw character without echo and without character
44 code translation. wait_ascii() either polls get_ascii() or uses some
45 kind of system-dependent event waiting. Returns ABORT_CHAR or BREAK_CHAR
46 immediately if abort_flag has been set, regardless of whether there is
47 new ascii input.
48 
49 get_ascii() - checks to see if a character is available. (Using
50 check_aborted().)
51 The only dependency here is on the Amiga, we restart input when buffer goes
52 from full to non-full.
53 
54 check_aborted() - looks for input by calling ascii_input. If found,
55 put the input into the type_ahead[] buffer. Returns abort_flag.
56 
57 ascii_input() - lowest level of input; just gets a character if there is
58 one. Does conversion from RETURN (\r) to EOL (\n). The Amiga handles
59 this in-line directly in check_aborted().
60 
61 Here's a quick summary:
62 
63 ggets = ggetchar + line editing & string building
64 ggetchar = wait_ascii + character echo
65 wait_ascii = get_ascii + wait for character
66 get_ascii = check_aborted + pull char from buffer
67 check_aborted = ascii_input + test for ^C,^G + put in buffer
68 ascii_input = poll for char + CR->EOL conversion
69 
70 */

432 /****************************************************************************
433 * askbool
434 * Inputs:
435 * char *prompt: string to prompt for user input
436 * int deflt: TRUE or FALSE default
437 * Returns:
438 * boolean: TRUE or FALSE as entered by user
439 * Effect:
440 * prompts user for yes or no input, returns result
441 ****************************************************************************/
442 
443 int askbool(prompt, deflt)
444 char *prompt;
445 int deflt;
446 {
447 #define undefined -1
448  char defchar; /* the default answer */
449  char c; /* user input */
450  char in_string[100];
451  int result = -1; /* the result: -1 = undefined, 0 = FALSE, 1 = TRUE */
452  if (deflt) defchar = 'y';
453  else defchar = 'n';
454  while (result == undefined) {
455  gprintf(TRANS, "%s? [%c]: ", prompt, defchar);
456  ggets(in_string);
457  c = in_string[0];
458  if (islower(c)) c = toupper(c);
459  if (c == 'Y') result = TRUE;
460  else if (c == 'N') result = FALSE;
461  else if (c == EOS) result = deflt;
462  else if (abort_flag) result = deflt;
463  /* space before Please to separate from user's type-in: */
464  else gprintf(TRANS, " Please type Y or N.\n");
465  }
466  if (abort_flag == BREAK_LEVEL) {
467  abort_flag = 0;
468  result = deflt;
469  gprintf(TRANS, "\n");
470  }
471  return result;
472 }

475 /****************************************************************************
476 * fileopen
477 * Inputs:
478 * char *deflt: the default file name (e.g. from command line)
479 * char *extension: default extension
480 * char *mode: read ("r") or write ("w")
481 * char *prompt: prompt for user
482 * Returns:
483 * opened file pointer
484 * Effect:
485 * opens file, prompts for user input if necessary and warns about
486 * possible confusion. If deflt is a null string or NULL, the user will
487 * be prompted for a name. The routine loops until a file is opened.
488 * If the mode is "r", a check is made to see if the file exists
489 * with and without the extension. If both exist a warning is given.
490 * For mode "w", a check is made to see if the file will be overwritten.
491 * The extension is automatically added if the default or user-typed
492 * file has no "." At the bottom of the loop body, if no file has
493 * been opened, the user is prompted for another file name.
494 ****************************************************************************/
495 
496 char fileopen_name[100]; /* name of the opened file */
497 
498 FILE *fileopen(deflt, extension, mode, prompt)
499  char *deflt;
500  char *extension; /* default extension */
501  char *mode; /* read "r" or write "w" */
502  char *prompt; /* prompt for user */
503 {
504  char extname[100]; /* trial name with extension added */
505  FILE *fp = NULL; /* file corresponding to filename */
506  FILE *fpext; /* file corresponding to extname */
507  char *problem = NULL; /* tells user why he has to try again */
508 
509  if (!deflt) deflt = ""; /* treat NULL as the empty string */
510  strcpy(fileopen_name, deflt);
511  /* keep trying until a good file is found: */
512  while (fp == NULL) {
513  /* avoid null file names: */
514  while (strlen(fileopen_name) == 0) {
515 #ifndef MACINTOSH
516  gprintf(TRANS, "%s : ", prompt);
517  ggets(fileopen_name);
518  if (abort_flag) {
519  if (abort_flag == BREAK_LEVEL) {
520  abort_flag = 0;
521  /* type return since user didn't... */
522  gprintf(TRANS, "\n");
523  }
524  return NULL;
525  }
526 #else /* use Macintosh file dialog */
527  if (mode[0] == 'r') {
528  if (!GetReadFileName(fileopen_name)) return NULL;
529  } else if (mode[0] == 'w') {
530  if (!(GetWriteFileName(fileopen_name, prompt))) return NULL;
531  } else {
532  gprintf(ERROR, "(fileopen) internal error: bad mode\n");
533  }
534 #endif /* MACINTOSH */
535  }
536  if (mode[0] == 'r') {
537  strcpy(extname, fileopen_name);
538  strcat(extname, ".");
539  strcat(extname, extension);
540  fp = NULL;
541  fpext = NULL;
542  if (ok_to_open(fileopen_name, mode)) {
543  fp = fopen(fileopen_name, mode);
544  }
545  if (ok_to_open(extname, mode)) {
546  fpext = fopen(extname, mode);
547  }
548  if (fp != NULL && fpext != NULL) {
549  gprintf(TRANS,
550  "warning: both %s and %s exist. %s will be used.\n",
551  fileopen_name, extname, fileopen_name);
552  fclose(fpext);
553  } else if (fpext != NULL) {
554  fp = fpext;
555  strcpy(fileopen_name, extname); /* remember what was opened */
556  }
557  if (fp == NULL) problem = "Couldn't find %s.\n";
558  } else if (mode[0] == 'w') {
559  boolean added_extension = FALSE;
560 
561  /* add the extension if there is no '.' in the file name */
562  if (!strchr(fileopen_name, '.')) {
563  strcat(fileopen_name, ".");
564  strcat(fileopen_name, extension);
565  added_extension = TRUE;
566  }
567  if (TRUE
568 #ifdef MACINTOSH
569  /* file open dialog already asked user to confirm unless we're
570  * adding an extension
571  */
572  && added_extension
573 #endif
574  ) {
575  fp = NULL;
576  if (ok_to_open(fileopen_name, "r"))
577  fp = fopen(fileopen_name, "r");
578  if (fp != NULL) {
579  char question[100];
580  fclose(fp);
581  strcpy(question, "OK to overwrite ");
582  strcat(question, fileopen_name);
583  if (!askbool(question, FALSE)) {
584  fp = NULL;
585  problem = "\n";
586  goto tryagain;
587  }
588  }
589  }
590  fp = NULL;
591  if (ok_to_open(fileopen_name, mode))
592  fp = fopen(fileopen_name, mode);
593  if (fp == NULL) problem = "Couldn't create %s.\n";
594  }
595  tryagain:
596  if (fp == NULL) {
597  gprintf(TRANS, problem, fileopen_name);
598  gprintf(TRANS,"Try again.\n");
599  fileopen_name[0] = EOS;
600  }
601  }
602  return fp;
603 }
