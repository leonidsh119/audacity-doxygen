518 /*
519  * input pcm data, output (maybe) mp3 frames.
520  * This routine handles all buffering, resampling and filtering for you.
521  *
522  * return code number of bytes output in mp3buf. Can be 0
523  * -1: mp3buf was too small
524  * -2: malloc() problem
525  * -3: lame_init_params() not called
526  * -4: psycho acoustic problems
527  * -5: ogg cleanup encoding error
528  * -6: ogg frame encoding error
529  *
530  * The required mp3buf_size can be computed from num_samples,
531  * samplerate and encoding rate, but here is a worst case estimate:
532  *
533  * mp3buf_size in bytes = 1.25*num_samples + 7200
534  *
535  * I think a tighter bound could be: (mt, March 2000)
536  * MPEG1:
537  * num_samples*(bitrate/8)/samplerate + 4*1152*(bitrate/8)/samplerate + 512
538  * MPEG2:
539  * num_samples*(bitrate/8)/samplerate + 4*576*(bitrate/8)/samplerate + 256
540  *
541  * but test first if you use that!
542  *
543  * set mp3buf_size = 0 and LAME will not check if mp3buf_size is
544  * large enough.
545  *
546  * NOTE:
547  * if gfp->num_channels=2, but gfp->mode = 3 (mono), the L & R channels
548  * will be averaged into the L channel before encoding only the L channel
549  * This will overwrite the data in buffer_l[] and buffer_r[].
550  *
551 */
552 int CDECL lame_encode_buffer (
553  lame_global_flags* gfp, /* global context handle */
554  const short int buffer_l [], /* PCM data for left channel */
555  const short int buffer_r [], /* PCM data for right channel */
556  const int nsamples, /* number of samples per channel */
557  unsigned char* mp3buf, /* pointer to encoded MP3 stream */
558  const int mp3buf_size ); /* number of valid octets in this
559  stream */
