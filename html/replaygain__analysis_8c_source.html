34 /*
35  * Here's the deal. Call
36  *
37  * InitGainAnalysis ( long samplefreq );
38  *
39  * to initialize everything. Call
40  *
41  * AnalyzeSamples ( const flac_float_t* left_samples,
42  * const flac_float_t* right_samples,
43  * size_t num_samples,
44  * int num_channels );
45  *
46  * as many times as you want, with as many or as few samples as you want.
47  * If mono, pass the sample buffer in through left_samples, leave
48  * right_samples NULL, and make sure num_channels = 1.
49  *
50  * GetTitleGain()
51  *
52  * will return the recommended dB level change for all samples analyzed
53  * SINCE THE LAST TIME you called GetTitleGain() OR InitGainAnalysis().
54  *
55  * GetAlbumGain()
56  *
57  * will return the recommended dB level change for all samples analyzed
58  * since InitGainAnalysis() was called and finalized with GetTitleGain().
59  *
60  * Pseudo-code to process an album:
61  *
62  * flac_float_t l_samples [4096];
63  * flac_float_t r_samples [4096];
64  * size_t num_samples;
65  * unsigned int num_songs;
66  * unsigned int i;
67  *
68  * InitGainAnalysis ( 44100 );
69  * for ( i = 1; i <= num_songs; i++ ) {
70  * while ( ( num_samples = getSongSamples ( song[i], left_samples, right_samples ) ) > 0 )
71  * AnalyzeSamples ( left_samples, right_samples, num_samples, 2 );
72  * fprintf ("Recommended dB change for song %2d: %+6.2f dB\n", i, GetTitleGain() );
73  * }
74  * fprintf ("Recommended dB change for whole album: %+6.2f dB\n", GetAlbumGain() );
75  */

270 static void
271 filter ( const flac_float_t* input, flac_float_t* output, size_t nSamples, const flac_float_t* a, const flac_float_t* b, size_t order, unsigned downsample )
272 {
273  double y;
274  size_t i;
275  size_t k;
276 
277  const flac_float_t* input_head = input;
278  const flac_float_t* input_tail;
279 
280  flac_float_t* output_head = output;
281  flac_float_t* output_tail;
282 
283  for ( i = 0; i < nSamples; i++, input_head += downsample, ++output_head ) {
284 
285  input_tail = input_head;
286  output_tail = output_head;
287 
288  y = *input_head * b[0];
289 
290  for ( k = 1; k <= order; k++ ) {
291  input_tail -= downsample;
292  --output_tail;
293  y += *input_tail * b[k] - *output_tail * a[k];
294  }
295 
296  output[i] = (flac_float_t)y;
297  }
298 }

302 static struct ReplayGainFilter*
303 CreateGainFilter ( long samplefreq )
304 {
305  unsigned i;
306  long maxrate = 0;
307  unsigned downsample = 1;
308  struct ReplayGainFilter* gainfilter = malloc(sizeof(*gainfilter));
309 
310  if ( !gainfilter )
311  return 0;
312 
313  while (1) {
314  for ( i = 0; i < sizeof(ReplayGainFilters)/sizeof(ReplayGainFilters[0]); ++i ) {
315  if (maxrate < ReplayGainFilters[i].rate)
316  maxrate = ReplayGainFilters[i].rate;
317 
318  if ( ReplayGainFilters[i].rate == samplefreq ) {
319  *gainfilter = ReplayGainFilters[i];
320  gainfilter->downsample = downsample;
321  return gainfilter;
322  }
323  }
324 
325  if (samplefreq < maxrate)
326  break;
327 
328  while (samplefreq > maxrate) {
329  downsample *= 2;
330  samplefreq /= 2;
331  }
332  }
333 
334  free(gainfilter);
335 
336  return 0;
337 }

352 ResetSampleFrequency ( long samplefreq ) {
353  int i;
354 
355  free(replaygainfilter);

357  replaygainfilter = CreateGainFilter( samplefreq );

363  sampleWindow =

366  if ( ! ReallocateWindowBuffer(sampleWindow, &lstepbuf) ||

371  return INIT_GAIN_ANALYSIS_ERROR;

418 int
419 AnalyzeSamples ( const flac_float_t* left_samples, const flac_float_t* right_samples, size_t num_samples, int num_channels )
420 {
421  unsigned downsample = replaygainfilter->downsample;
422  const flac_float_t* curleft;
423  const flac_float_t* curright;
424  long prebufsamples;
425  long batchsamples;
426  long cursamples;
427  long cursamplepos;
428  int i;
429 
430  num_samples /= downsample;
431 
432  if ( num_samples == 0 )
433  return GAIN_ANALYSIS_OK;
434 
435  cursamplepos = 0;
436  batchsamples = num_samples;
437 
438  switch ( num_channels) {

443 
444  prebufsamples = MAX_ORDER;
445  if ((size_t) prebufsamples > num_samples)
