382 bool FFmpegImportFileHandle::InitCodecs()
383 {
384  // Allocate the array of pointers to hold stream contexts pointers
385  // Some of the allocated space may be unused (corresponds to video, subtitle, or undecodeable audio streams)
386  mScs = (streamContext**)malloc(sizeof(streamContext**)*mFormatContext->nb_streams);
387  // Fill the stream contexts
388  for (unsigned int i = 0; i < mFormatContext->nb_streams; i++)
389  {
390  if (mFormatContext->streams[i]->codec->codec_type == AVMEDIA_TYPE_AUDIO)
391  {
392  //Create a context
393  streamContext *sc = new streamContext;
394  memset(sc,0,sizeof(*sc));
395 
396  sc->m_stream = mFormatContext->streams[i];
397  sc->m_codecCtx = sc->m_stream->codec;
398 
399  AVCodec *codec = avcodec_find_decoder(sc->m_codecCtx->codec_id);
400  if (codec == NULL)
401  {
402  wxLogError(wxT("FFmpeg : avcodec_find_decoder() failed. Index[%02d], Codec[%02x - %s]"),i,sc->m_codecCtx->codec_id,sc->m_codecCtx->codec_name);
403  //FFmpeg can't decode this stream, skip it
404  delete sc;
405  continue;
406  }
407  if (codec->type != sc->m_codecCtx->codec_type)
408  {
409  wxLogError(wxT("FFmpeg : Codec type mismatch, skipping. Index[%02d], Codec[%02x - %s]"),i,sc->m_codecCtx->codec_id,sc->m_codecCtx->codec_name);
410  //Non-audio codec reported as audio? Nevertheless, we don't need THIS.
411  delete sc;
412  continue;
413  }
414 
415  if (avcodec_open2(sc->m_codecCtx, codec, NULL) < 0)
416  {
417  wxLogError(wxT("FFmpeg : avcodec_open() failed. Index[%02d], Codec[%02x - %s]"),i,sc->m_codecCtx->codec_id,sc->m_codecCtx->codec_name);
418  //Can't open decoder - skip this stream
419  delete sc;
420  continue;
421  }
422 
423  // Stream is decodeable and it is audio. Add it and its decription to the arrays
424  wxString strinfo;
425  int duration = 0;
426  if (sc->m_stream->duration > 0)
427  duration = sc->m_stream->duration * sc->m_stream->time_base.num / sc->m_stream->time_base.den;
428  else
429  duration = mFormatContext->duration / AV_TIME_BASE;
430  wxString bitrate = wxT("");
431  if (sc->m_codecCtx->bit_rate > 0)
432  bitrate.Printf(wxT("%d"),sc->m_codecCtx->bit_rate);
433  else
434  bitrate.Printf(wxT("?"));
435 
436  AVDictionaryEntry *tag = av_dict_get(sc->m_stream->metadata, "language", NULL, 0);
437  wxString lang;
438  if (tag)
439  {
440  lang.FromUTF8(tag->value);
441  }
442  strinfo.Printf(_("Index[%02x] Codec[%s], Language[%s], Bitrate[%s], Channels[%d], Duration[%d]"),sc->m_stream->id,codec->name,lang.c_str(),bitrate.c_str(),sc->m_stream->codec->channels, duration);
443  mStreamInfo->Add(strinfo);
444  mScs[mNumStreams++] = sc;
445  }
446  //for video and unknown streams do nothing
447  }
448  //It doesn't really returns false, but GetStreamCount() will return 0 if file is composed entierly of unreadable streams
449  return true;
450 }

464 int FFmpegImportFileHandle::Import(TrackFactory *trackFactory,
465  Track ***outTracks,
466  int *outNumTracks,
467  Tags *tags)
468 {
469 
470  CreateProgress();
471 
472  // Remove stream contexts which are not marked for importing and adjust mScs and mNumStreams accordingly
473  for (int i = 0; i < mNumStreams;)
474  {
475  if (!mScs[i]->m_use)
476  {
477  delete mScs[i];
478  for (int j = i; j < mNumStreams - 1; j++)
479  {
480  mScs[j] = mScs[j+1];
481  }
482  mNumStreams--;
483  }
484  else i++;
485  }
486 
487  mChannels = new WaveTrack **[mNumStreams];
488 
489  for (int s = 0; s < mNumStreams; s++)
490  {
491  switch (mScs[s]->m_stream->codec->sample_fmt)
492  {
493  case AV_SAMPLE_FMT_U8:
494  case AV_SAMPLE_FMT_S16:
495  case AV_SAMPLE_FMT_U8P:
496  case AV_SAMPLE_FMT_S16P:
497  mScs[s]->m_osamplesize = sizeof(int16_t);
498  mScs[s]->m_osamplefmt = int16Sample;
499  break;
500  default:
501  mScs[s]->m_osamplesize = sizeof(float);
502  mScs[s]->m_osamplefmt = floatSample;
503  break;
504  }
505 
506  // There is a possibility that number of channels will change over time, but we do not have WaveTracks for NEW channels. Remember the number of channels and stick to it.
507  mScs[s]->m_initialchannels = mScs[s]->m_stream->codec->channels;
508  mChannels[s] = new WaveTrack *[mScs[s]->m_stream->codec->channels];
509  int c;
510  for (c = 0; c < mScs[s]->m_stream->codec->channels; c++)
511  {
512  mChannels[s][c] = trackFactory->NewWaveTrack(mScs[s]->m_osamplefmt, mScs[s]->m_stream->codec->sample_rate);
513 
514  if (mScs[s]->m_stream->codec->channels == 2)
515  {
516  switch (c)
517  {
518  case 0:
519  mChannels[s][c]->SetChannel(Track::LeftChannel);
520  mChannels[s][c]->SetLinked(true);
521  break;
522  case 1:
523  mChannels[s][c]->SetChannel(Track::RightChannel);
524  break;
525  }
526  }
527  else
528  {
529  mChannels[s][c]->SetChannel(Track::MonoChannel);
530  }
531  }
532  }
533 
534  // Handles the start_time by creating silence. This may or may not be correct.
535  // There is a possibility that we should ignore first N milliseconds of audio instead. I do not know.
537  for (int s = 0; s < mNumStreams; s++)
538  {
539  int64_t stream_delay = 0;
540  if (mScs[s]->m_stream->start_time != int64_t(AV_NOPTS_VALUE) && mScs[s]->m_stream->start_time > 0)
541  {
542  stream_delay = mScs[s]->m_stream->start_time;
543  wxLogDebug(wxT("Stream %d start_time = %lld, that would be %f milliseconds."), s, (long long) mScs[s]->m_stream->start_time, double(mScs[s]->m_stream->start_time)/AV_TIME_BASE*1000);
544  }
545  if (stream_delay != 0)
546  {
547  for (int c = 0; c < mScs[s]->m_stream->codec->channels; c++)
548  {
549  WaveTrack *t = mChannels[s][c];
550  t->InsertSilence(0,double(stream_delay)/AV_TIME_BASE);
551  }
552  }
553  }
554  // This is the heart of the importing process
555  // The result of Import() to be returend. It will be something other than zero if user canceled or some error appears.
556  int res = eProgressSuccess;
557 
558 #ifdef EXPERIMENTAL_OD_FFMPEG
559  mUsingOD = false;
560  gPrefs->Read(wxT("/Library/FFmpegOnDemand"), &mUsingOD);
561  //at this point we know the file is good and that we have to load the number of channels in mScs[s]->m_stream->codec->channels;
562  //so for OD loading we create the tracks and releasee the modal lock after starting the ODTask.
563  if (mUsingOD) {
564  std::vector<ODDecodeFFmpegTask*> tasks;
565  //append blockfiles to each stream and add an individual ODDecodeTask for each one.
566  for (int s = 0; s < mNumStreams; s++) {
567  ODDecodeFFmpegTask* odTask=new ODDecodeFFmpegTask(mScs,mNumStreams,mChannels,mFormatContext, s);
568  odTask->CreateFileDecoder(mFilename);
569 
570  //each stream has different duration. We need to know it if seeking is to be allowed.
571  sampleCount sampleDuration = 0;
572  if (mScs[s]->m_stream->duration > 0)
573  sampleDuration = ((sampleCount)mScs[s]->m_stream->duration * mScs[s]->m_stream->time_base.num) *mScs[s]->m_stream->codec->sample_rate / mScs[s]->m_stream->time_base.den;
574  else
575  sampleDuration = ((sampleCount)mFormatContext->duration *mScs[s]->m_stream->codec->sample_rate) / AV_TIME_BASE;
576 
577  // printf(" OD duration samples %qi, sr %d, secs %d\n",sampleDuration, (int)mScs[s]->m_stream->codec->sample_rate,(int)sampleDuration/mScs[s]->m_stream->codec->sample_rate);
578 
579  //for each wavetrack within the stream add coded blockfiles
580  for (int c = 0; c < mScs[s]->m_stream->codec->channels; c++) {
581  WaveTrack *t = mChannels[s][c];
582  odTask->AddWaveTrack(t);
583 
584  sampleCount maxBlockSize = t->GetMaxBlockSize();
585  //use the maximum blockfile size to divide the sections (about 11secs per blockfile at 44.1khz)
586  for (sampleCount i = 0; i < sampleDuration; i += maxBlockSize) {
587  sampleCount blockLen = maxBlockSize;
588  if (i + blockLen > sampleDuration)
589  blockLen = sampleDuration - i;
590 
591  t->AppendCoded(mFilename, i, blockLen, c,ODTask::eODFFMPEG);
592 
593  // This only works well for single streams since we assume
594  // each stream is of the same duration and channels
595  res = mProgress->Update(i+sampleDuration*c+ sampleDuration*mScs[s]->m_stream->codec->channels*s,
596  sampleDuration*mScs[s]->m_stream->codec->channels*mNumStreams);
597  if (res != eProgressSuccess)
598  break;
599  }
600  }
601  tasks.push_back(odTask);
602  }
603  //Now we add the tasks and let them run, or DELETE them if the user cancelled
604  for(int i=0; i < (int)tasks.size(); i++) {
605  if(res==eProgressSuccess)
606  ODManager::Instance()->AddNewTask(tasks[i]);
607  else
608  {
609  delete tasks[i];
610  }
611  }
612  } else {
613 #endif
614  streamContext *sc = NULL;
615 
616  // Read next frame.
617  while ((sc = ReadNextFrame()) != NULL && (res == eProgressSuccess))
618  {
619  // ReadNextFrame returns 1 if stream is not to be imported
620  if (sc != (streamContext*)1)
621  {
622  // Decode frame until it is not possible to decode any further
623  while (sc->m_pktRemainingSiz > 0 && (res == eProgressSuccess || res == eProgressStopped))
624  {
625  if (DecodeFrame(sc,false) < 0)
626  break;
627 
628  // If something useable was decoded - write it to mChannels
629  if (sc->m_frameValid)
630  res = WriteData(sc);
631  }
632 
633  // Cleanup after frame decoding
634  if (sc->m_pktValid)
635  {
636  av_free_packet(&sc->m_pkt);
637  sc->m_pktValid = 0;
638  }
639  }
640  }
641 
642  // Flush the decoders.
643  if ((mNumStreams != 0) && (res == eProgressSuccess || res == eProgressStopped))
644  {
645  for (int i = 0; i < mNumStreams; i++)
646  {
647  if (DecodeFrame(mScs[i], true) == 0)
648  {
649  WriteData(mScs[i]);
650 
651  if (mScs[i]->m_pktValid)
652  {
653  av_free_packet(&mScs[i]->m_pkt);
654  mScs[i]->m_pktValid = 0;
655  }
656  }
657  }
658  }
659 #ifdef EXPERIMENTAL_OD_FFMPEG
660  } // else -- !mUsingOD == true
661 #endif //EXPERIMENTAL_OD_FFMPEG
662 
663  // Something bad happened - destroy everything!
664  if (res == eProgressCancelled || res == eProgressFailed)
665  {
666  for (int s = 0; s < mNumStreams; s++)
667  {
668  delete[] mChannels[s];
669  }
670  delete[] mChannels;
671 
672  return res;
673  }
674  //else if (res == 2), we just stop the decoding as if the file has ended
675 
676  *outNumTracks = 0;
677  for (int s = 0; s < mNumStreams; s++)
678  {
679  *outNumTracks += mScs[s]->m_initialchannels;
680  }
681 
682  // Create NEW tracks
683  *outTracks = new Track *[*outNumTracks];
684 
685  // Copy audio from mChannels to newly created tracks (destroying mChannels elements in process)
686  int trackindex = 0;
687  for (int s = 0; s < mNumStreams; s++)
688  {
689  for(int c = 0; c < mScs[s]->m_initialchannels; c++)
690  {
691  mChannels[s][c]->Flush();
692  (*outTracks)[trackindex++] = mChannels[s][c];
693  }
694  delete[] mChannels[s];
695  }
696  delete[] mChannels;
697 
698  // Save metadata
699  WriteMetadata(tags);
700 
701  return res;
702 }

714 int FFmpegImportFileHandle::WriteData(streamContext *sc)
715 {
716  // Find the stream index in mScs array
717  int streamid = -1;
718  for (int i = 0; i < mNumStreams; i++)
719  {
720  if (mScs[i] == sc)
721  {
722  streamid = i;
723  break;
724  }
725  }
726  // Stream is not found. This should not really happen
727  if (streamid == -1)
728  {
729  return 1;
730  }
731 
732  // Allocate the buffer to store audio.
733  int insamples = sc->m_decodedAudioSamplesValidSiz / sc->m_samplesize;
734  int nChannels = sc->m_stream->codec->channels < sc->m_initialchannels ? sc->m_stream->codec->channels : sc->m_initialchannels;
735  uint8_t **tmp = (uint8_t **) malloc(sizeof(uint8_t *) * nChannels);
736  for (int chn = 0; chn < nChannels; chn++)
737  {
738  tmp[chn] = (uint8_t *) malloc(sc->m_osamplesize * (insamples / nChannels));
739  }
740 
741  // Separate the channels and convert input sample format to 16-bit
742  uint8_t *in = sc->m_decodedAudioSamples;
743  int index = 0;
744  int pos = 0;
745  while (pos < insamples)
746  {
747  for (int chn = 0; chn < sc->m_stream->codec->channels; chn++)
748  {
749  if (chn < nChannels)
750  {
751  switch (sc->m_samplefmt)
752  {
753  case AV_SAMPLE_FMT_U8:
754  case AV_SAMPLE_FMT_U8P:
755  ((int16_t *)tmp[chn])[index] = (int16_t) (*(uint8_t *)in - 0x80) << 8;
756  break;
757 
758  case AV_SAMPLE_FMT_S16:
759  case AV_SAMPLE_FMT_S16P:
760  ((int16_t *)tmp[chn])[index] = (int16_t) *(int16_t *)in;
761  break;
762 
763  case AV_SAMPLE_FMT_S32:
764  case AV_SAMPLE_FMT_S32P:
765  ((float *)tmp[chn])[index] = (float) *(int32_t *)in * (1.0 / (1 << 31));
766  break;
767 
768  case AV_SAMPLE_FMT_FLT:
769  case AV_SAMPLE_FMT_FLTP:
770  ((float *)tmp[chn])[index] = (float) *(float *)in;
771  break;
772 
773  case AV_SAMPLE_FMT_DBL:
774  case AV_SAMPLE_FMT_DBLP:
775  ((float *)tmp[chn])[index] = (float) *(double *)in;
776  break;
777 
778  default:
779  wxLogError(wxT("Stream %d has unrecognized sample format %d."), streamid, sc->m_samplefmt);
780  for (int chn=0; chn < nChannels; chn++)
781  {
782  free(tmp[chn]);
783  }
784  free(tmp);
785  return 1;
786  break;
787  }
788  }
789  in += sc->m_samplesize;
790  pos++;
791  }
792  index++;
793  }
794 
795  // Write audio into WaveTracks
796  for (int chn=0; chn < nChannels; chn++)
797  {
798  mChannels[streamid][chn]->Append((samplePtr)tmp[chn],sc->m_osamplefmt,index);
799  free(tmp[chn]);
800  }
801 
802  free(tmp);
803 
804  // Try to update the progress indicator (and see if user wants to cancel)
805  int updateResult = eProgressSuccess;
806  int64_t filesize = avio_size(mFormatContext->pb);
807  // PTS (presentation time) is the proper way of getting current position
808  if (sc->m_pkt.pts != int64_t(AV_NOPTS_VALUE) && mFormatContext->duration != int64_t(AV_NOPTS_VALUE))
809  {
810  mProgressPos = sc->m_pkt.pts * sc->m_stream->time_base.num / sc->m_stream->time_base.den;
811  mProgressLen = (mFormatContext->duration > 0 ? mFormatContext->duration / AV_TIME_BASE: 1);
812  }
813  // When PTS is not set, use number of frames and number of current frame
814  else if (sc->m_stream->nb_frames > 0 && sc->m_codecCtx->frame_number > 0 && sc->m_codecCtx->frame_number <= sc->m_stream->nb_frames)
815  {
816  mProgressPos = sc->m_codecCtx->frame_number;
817  mProgressLen = sc->m_stream->nb_frames;
818  }
819  // When number of frames is unknown, use position in file
820  else if (filesize > 0 && sc->m_pkt.pos > 0 && sc->m_pkt.pos <= filesize)
821  {
822  mProgressPos = sc->m_pkt.pos;
823  mProgressLen = filesize;
824  }
825  updateResult = mProgress->Update(mProgressPos, mProgressLen != 0 ? mProgressLen : 1);
826 
827  return updateResult;
828 }
