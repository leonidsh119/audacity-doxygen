8 # This is a replacement for the GNU gettext "msgmerge" program, which

11 # flags but simply the def, ref, and output file names.

40 # Compute the edit-distance between str1 and str2, taking a couple
41 # of shortcuts such that it returns 999 quickly if the edit-distance
42 # is clearly not going to be less than 10 percent.
43 def edit_distance(str1, str2):
44  l1 = len(str1)
45  l2 = len(str2)
46 
47  # Exit if the difference in the string lenghts is 10% or more
48  if l1*1.0/l2 < 0.9 or l1*1.0/l2 > 1.1:
49  return 999
50 
51  # Compute a beam width of +/- 5% - the path through the matrix cannot
52  # go outside the main diagonal +/- the beam.
53  beam = int(0.5 + 0.1 * ((l1 + l2) / 2))
54 
55  # Create a 2D array
56  d = [None]*(l1+1)
57  for i in range(l1+1):
58  d[i] = [999]*(l2+1)
59 
60  # Initialize the first row and column
61  for i in range(l1+1):
62  d[i][0] = i
63  for j in range(l2+1):
64  d[0][j] = j
65 
66  # Dynamic programming
67  for i in range(1, l1+1):
68  # Quick short-circuit after 30 rows; stop if things are
69  # looking really bad
70  if i==30 and l2>=30 and d[29][29] > 20:
71  return 999
72  for j in range(max(1, i-beam), min(l2+1, i+beam+1)):
73  if str1[i-1] == str2[j-1]:
74  cost = 0
75  else:
76  cost = 1
77  d[i][j] = min(
78  d[i-1][j] + 1, # deletion
79  d[i][j-1] + 1, # insertion
80  d[i-1][j-1] + cost # substitution
81  )
82  return d[l1][l2]
