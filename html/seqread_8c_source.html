305 /* def_append -- append a byte to the current definition */
306 /*
307  * The def data structure:
308  * [code][offset][code][offset]...[0][length][data][data][data]...
309  * where code is 1:nmacroparms for %n,
310  * nmacroparms+1 for %v,
311  * nmacroparms+2:nmacroparms*2+1 for ^n
312  * and offset is the byte offset (from the offset byte) to the data
313  * where the parameter should be substituted
314  * and length is the number of data bytes
315  */
316 boolean def_append(def, nparms, data)
317  unsigned char def[];
318  int nparms;
319  int data;
320 {
321  int base = (nparms << 1) + 1; /* this byte is the length */
322  /* first parameter has to be able to reference last byte: */
323  if ((def[base])++ >= (254 - (nparms << 1))) {
324  fferror("Data too long");
325  return FALSE;
326  }
327  def[base + def[base]] = data;
328  return TRUE;
329 }

346 void def_parm(def, nparms, code)
347  unsigned char def[];
348  int nparms;
349  int code;
350 {
351  int i, j;
352  /* in order to insert a 2-byte parameter descriptor, the offsets from
353  * previous descriptors (that precede the data) need to be increased by 2:
354  */
355  for (i = 1; i < (nparms << 1); i += 2) {
356  def[i] += 2;
357  }
358  /* now i is index of length; work backwards from the last byte, moving
359  * everything up by 2 bytes to make room for the new descriptor:
360  */
361  for (j = i + def[i] + 2; j > i; j--) {
362  def[j] = def[j - 2];
363  }
364  /* now i is index of offset; insert the descriptor code (first byte)
365  * and the offset to the parameter location in the message (second byte)
366  */
367  def[i - 1] = code;
368  def[i] = def[i + 2] + 2;
369 }

581 private void dodef()
582 {
583  /* maximum def size is 256 + 9 parms * 2 + 2 = 276 */
584  unsigned char def[280];
585  char symbol[100];
586  int nparms = 0;
587  int nibcount = 0;
588  int data = 0;
589  register char c;
590 
591  linex += scan();
592 
593  if (!token[0]) fferror("Symbol expected");
594  else {
595  strcpy(symbol, token);
596  def[0] = def[1] = 0;
597  while (TRUE) {
598  linex += scan1(&line[linex]);
599  c = token[0];
600  if (!c) {
601  linex--;
602  if (nibcount & 1) {
603  fferror("Expected pairs of hex digits: one missing");
604  return;
605  }
606  break;
607  } else if (c == ' ' || c == '\t' || c == '\n') continue;
608  else if (isdigit(c)) {
609  data = (data << 4) + (c - '0');
610  nibcount++;
611  if (!(nibcount & 1)) {
612  if (!def_append(def, nparms, data))
613  return;
614  data = 0;
615  }
616  } else if ('A' <= c && c <= 'F') {
617  data = (data << 4) + (c - 'A') + 10;
618  nibcount++;
619  if (!(nibcount & 1)) {
620  if (!def_append(def, nparms, data))
621  return;
622  data = 0;
623  }
624  } else if (c == 'V') {
625  data = data << 4;
626  nibcount++;
627  /* v without a leading nibble is equivalent to 0v: */
628  if (nibcount & 1) nibcount++;
629  if (!def_append(def, nparms, data))
630  return;
631  def_parm(def, nparms++, nmacroparms+1);
632  } else if (c == '%') {
633  linex += scan1(&line[linex]);
634  c = token[0];
635  if (c < '1' || c > ('0' + nmacroparms)) {
636  fferror(parm_expected_error);
637  break;
638  }
639  if (!def_append(def, nparms, 0))
640  return;
641  def_parm(def, nparms++, c - '0');
642  } else if (c == '^') {
643  linex += scan1(&line[linex]);
644  c = token[0];
645  if (c < '1' || c > ('0' + nmacroparms)) {
646  fferror(parm_expected_error);
647  break;
648  }
649  if (!def_append(def, nparms, 0))
650  return;
651  def_parm(def, nparms++, (c - '0') + nmacroparms + 1);
652  } else { /* something unexpected here -- just exit */
653  linex--;
654  fferror("Unexpected data");
655  return;
656  }
657  }
658  insert_def(the_score, symbol, def,
659  (nparms << 1) + def[(nparms << 1) + 1] + 2);

844 /****************************************************************************
845 * dopitch
846 * Effect: parses a pitch command
847 ****************************************************************************/
848 
849 private int dopitch()
850 {
851  int p, octave=0;
852  int octflag = FALSE; /* set if octave is specified */
853  int oldfieldx = fieldx;
854 
855  p = pitchtable[token[fieldx-1]-'A'];
856  while (TRUE) {
857  if (token[fieldx] == 'S') { /* sharp */
858  p++;
859  fieldx++;
860  }
861  else if (token[fieldx] == 'N') { /* skip */
862  fieldx++;
863  }
864  else if (token[fieldx] == 'F') { /* flat */
865  p--;
866  fieldx++;
867  }
868  else if (isdigit(token[fieldx]) && !octflag) { /* octave */
869  octave = (int) scanint();
870  octflag = TRUE;
871  }
872  else break; /* none of the above */
873  }
874  if (octflag) p = (p-48) + 12 * octave; /* adjust p to given octave */
875  else { /* adjust p to note nearest the default pitch */
876  int octdiff = (p + 126 - pitch) / 12;
877  p = p + 120 - (octdiff * 12);
878  }
879  if (p > maxpitch) { /* pitch in range? */
880  int newfield = fieldx;
881  fieldx = oldfieldx;
882  fferror("Pitch too high");
883  fieldx = newfield;
884  p = maxpitch;
885  }
886  /* We really should test for end-of-field, but we don't know if we're
887  in a parameter list, so comma may or may not be legal */
888  return p;
889 }

915 private void doramp()
916 {
917  int values[2];
918  time_type stepsize = 100L; /* default 10 per second */
919  int index = 0;
920  ndurp = FALSE;
921  values[0] = values[1] = 0;
922  while (TRUE) {
923  linex += scan();
924  fieldx = 1;
925  if (nullstring(token)) {
926  break;
927  } else if (index == 2) { /* must be stepsize in dur syntax */
928  stepsize = dodur();
929  } else {
930  int ctrlx = 0;
931  static int ctrl_map[] = { -BEND_CTRL, VOLUME, -TOUCH_CTRL, MODWHEEL };
932 
933  switch (token[0]) {
934  case 'M': ctrlx++; /* modwheel */
935  case 'O': ctrlx++; /* aftertouch */
936  case 'X': ctrlx++; /* volume */
937  case 'Y': /* pitch bend */
938 
939  if (index < 2) {
940  macctrlnum[index] = ctrl_map[ctrlx];
941  macctrlparmx[index] = (int) scanint();
942  if (token[fieldx])
943  fferror("Only digits expected here");
944  macctrldef[index] = NULL;
945  } else fferror("Unexpected control");
946  break;
947  case '~':
948  if (index < 2) {
949  domacro();
950  if (token[fieldx]) fferror("Unexpected character");
951  } else fferror("Unexpected control");
952  break;
953  case 'T':
954  if (index < 2) fferror("Control expected");
955  dotime();
956  break;
957  case 'V':
958  if (index < 2) fferror("Control expected");
959  dovoice();
960  break;
961  case 'N':
962  if (index < 2) fferror("Control expected");
963  donextdur();
964  break;
965  default:
966  if (index < 2) fferror("Control expected");
967  dur = dodur();
968  break;
969  }
970  if (index == 1 && (macctrlnum[0] != macctrlnum[1] ||
971  macctrldef[0] != macctrldef[1])) {
972  fferror("Controls do not match");
973  }
974  }
975  index++;
976  }
977  if (index < 3) fferror("Expected 2 controls and step size");
978  else {
979  if (macctrldef[0]) {
980  int i, j, n;
981  n = 0;
982  i = macctrlparmx[0];
983  j = macctrlparmx[1];
984  while (n < macctrlnum[0]) {
985  if (macctrlparms[i] != macctrlparms[j]) break;
986  n++; i++; j++;
987  }
988  if (n >= macctrlnum[0]) n = 0;
989  /* Note: duration shortened to prevent overlap with next ramp */
990  insert_deframp(the_score, seqround(thetime), lineno, voice,
991  seqround(stepsize), trunc(dur) - 1, macctrldef[0], macctrlnum[0],
992  macctrlparms + macctrlparmx[0], n, macctrlparms[j]);
993  } else {
994  /* Note: duration shortened to prevent overlap with next ramp */
995  insert_ctrlramp(the_score, seqround(thetime), lineno, voice,
996  seqround(stepsize), trunc(dur) - 1,
997  macctrlnum[0], macctrlparmx[0], macctrlparmx[1]);
998  }
999  }
1000  /* advance the time only if an N field was given */
1001  if (ndurp) thetime += ntime;
1002  else thetime += dur;
1003 }

1158 /****************************************************************************
1159 * dosymdur
1160 * Effect: parses a duration (^, %, S, I, Q, H, or W) command
1161 ****************************************************************************/
1162 
1163 private time_type dosymdur()
1164 {
1165  int i, dotcnt = 0;
1166  long dotfactor;
1167  time_type result = 0;
1168 
1169  for (i = 0; i < durtable_len; i++) {
1170  if (durtable[i].symbol == token[fieldx-1]) {
1171  /* the shift right is because durs are stored doubled because
1172  * otherwise a 64th note would have the value 75/2: */
1173  result = precise(durtable[i].value) >> 1;
1174  break;
1175  }
1176  }
1177  if (i == durtable_len) {
1178  fieldx--;
1179  fferror("Duration expected: one of W, H, Q, I, S, %, or ^");
1180  return 0L;
1181  }
1182  while (token[fieldx]) {
1183  if (token[fieldx] == 'T') { /* triplet notation */
1184  result = (result * 2) / 3; /* lose a bit but avoid scale() */
1185  fieldx++;
1186  }
1187  else if (token[fieldx] == '.') { /* dotted notation */
1188  dotcnt++;
1189  fieldx++;
1190  }
1191  else if (token[fieldx] == '/') {
1192  long divisor;
1193  fieldx++;
1194  divisor = scanint();
1195  if (divisor > 0) result = result / divisor;
1196  else fferror("non-zero integer expected");
1197  }
1198  else if (isdigit(token[fieldx])) { /* numbers are multipliers */
1199  result = result * scanint();
1200  }
1201  else break;
1202  }
1203  dotfactor = 1L;
1204  for (i=1; i<=dotcnt; i++)
1205  dotfactor = dotfactor * 2;
1206  result = (2 * result) - (result / dotfactor);
1207 
1208  return scale(result, 100L, tempo); /* time in milliseconds */
1209 }
