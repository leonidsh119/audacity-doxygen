658 {
659  if (maxlen <= len) {
660  expand();
661  }
662  // Note: if the new event is the last one, the assignment
663  // events[i] = event; (below) will never execute, so just

943 bool Alg_time_map::insert_tempo(double tempo, double beat)
944 {
945  tempo = tempo / 60.0; // convert to beats per second
946  // change the tempo at the given beat until the next beat event
947  if (beat < 0) return false;
948  double time = beat_to_time(beat);
949  long i = locate_time(time);
950  if (i >= beats.len || !within(beats[i].time, time, 0.000001)) {
951  insert_beat(time, beat);
952  }
953  // now i is index of beat where tempo will change
954  if (i == beats.len - 1) {
955  last_tempo = tempo;
956  // printf("last_tempo to %g\n", last_tempo);
957  last_tempo_flag = true;
958  } else { // adjust all future beats
959  // compute the difference in beats
960  double diff = beats[i + 1].beat - beats[i].beat;
961  // convert beat difference to seconds at new tempo
962  diff = diff / tempo;
963  // figure out old time difference:
964  double old_diff = beats[i + 1].time - time;
965  // compute difference too
966  diff = diff - old_diff;
967  // apply new_diff to score and beats
968  i++;
969  while (i < beats.len) {
970  beats[i].time = beats[i].time + diff;
971  i++;
972  }
973  }
974  return true;
975 }

1059 void Alg_time_map::trim(double start, double end, bool units_are_seconds)
1060 {
1061  // extract the time map from start to end and shift to time zero
1062  // start and end are time in seconds if units_are_seconds is true
1063  int i = 0; // index into beats
1064  int start_index; // index of first breakpoint after start
1065  int count = 1;
1066  double initial_beat = start;
1067  double final_beat = end;
1068  if (units_are_seconds) {
1069  initial_beat = time_to_beat(start);
1070  final_beat = time_to_beat(end);
1071  } else {
1072  start = beat_to_time(initial_beat);
1073  end = beat_to_time(final_beat);
1074  }
1075  while (i < length() && beats[i].time < start) i++;
1076  // now i is index into beats of the first breakpoint after start
1077  // beats[0] is (0,0) and remains that way
1078  // copy beats[start_index] to beats[1], etc.
1079  // skip any beats at or near (start,initial_beat), using count
1080  // to keep track of how many entries there are
1081  start_index = i;
1082  while (i < length() && beats[i].time < end) {
1083  if (beats[i].time - start > ALG_EPS &&
1084  beats[i].beat - initial_beat > ALG_EPS) {
1085  beats[i].time = beats[i].time - start;
1086  beats[i].beat = beats[i].beat - initial_beat;
1087  beats[i - start_index + 1] = beats[i];
1088  count = count + 1;
1089  } else {
1090  start_index = start_index + 1;
1091  }
1092  i = i + 1;
1093  }
1094  // set last tempo data
1095  // we last examined beats[i-1] and copied it to
1096  // beats[i - start_index]. Next tempo should come
1097  // from beats[i] and store in beats[i - start_index + 1]
1098  // case 1: there is at least one breakpoint beyond end
1099  // => interpolate to put a breakpoint at end
1100  // case 2: no more breakpoints => set last tempo data
1101  if (i < length()) {
1102  // we know beats[i].time >= end, so case 1 applies
1103  beats[i - start_index + 1].time = end - start;
1104  beats[i - start_index + 1].beat = final_beat - initial_beat;
1105  count = count + 1;
1106  }
1107  // else we'll just use stored last tempo (if any)
1108  beats.len = count;
1109 }

2482  // A) i < len && time_sig[i].beat > start
2483  // B) i == len

2492  tsden = time_sigs[i-1].den;
