136 // This is a helper class used by IntegerValidator and FloatingPointValidator
137 // below that implements Transfer{To,From}Window() adapted to the type of the

139 //
140 // The template argument B is the name of the base class which must derive from

186  {
187  if ( m_value )
188  {
189  wxTextEntry * const control = BaseValidator::GetTextEntry();
190  if ( !control )
191  return false;

199  bool TransferFromWindow() override
200  {
201  if ( m_value )
202  {
203  wxTextEntry * const control = BaseValidator::GetTextEntry();
204  if ( !control )
205  return false;
206 
207  // If window is disabled, simply return
208  if ( !control->IsEnabled() )
209  return true;
210 
211  const wxString s(control->GetValue());
212  LongestValueType value;
213  if ( s.empty() && BaseValidator::HasFlag(NUM_VAL_ZERO_AS_BLANK) )
214  value = 0;
215  else if ( !BaseValidator::FromString(s, &value) )
216  return false;
217 
218  if ( !this->IsInRange(value) )
219  return false;
220 
221  *m_value = static_cast<ValueType>(value);
222  }
223 
224  return true;
225  }
