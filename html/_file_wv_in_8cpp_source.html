172  finished_ = true;

174  }
175 
176  StkFloat tyme = time_;

182

184  chunkPointer_ -= chunkSize_ - 1; // overlap chunks by one frame
185  if ( chunkPointer_ < 0 ) chunkPointer_ = 0;

188  chunkPointer_ += chunkSize_ - 1; // overlap chunks by one frame

190  chunkPointer_ = file_.fileSize() - chunkSize_;

195  }
