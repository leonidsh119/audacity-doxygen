179 wav_open (SF_PRIVATE *psf)
180 { WAV_PRIVATE * wpriv ;
181  int format, subformat, error, blockalign = 0, framesperblock = 0 ;
182 
183  if ((wpriv = calloc (1, sizeof (WAV_PRIVATE))) == NULL)
184  return SFE_MALLOC_FAILED ;
185  psf->container_data = wpriv ;
186 
187  wpriv->wavex_ambisonic = SF_AMBISONIC_NONE ;
188  psf->str_flags = SF_STR_ALLOW_START | SF_STR_ALLOW_END ;
189 
190  if (psf->file.mode == SFM_READ || (psf->file.mode == SFM_RDWR && psf->filelength > 0))
191  { if ((error = wav_read_header (psf, &blockalign, &framesperblock)))
192  return error ;
193  } ;
194 
195  subformat = SF_CODEC (psf->sf.format) ;
196 
197  if (psf->file.mode == SFM_WRITE || psf->file.mode == SFM_RDWR)
198  { if (psf->is_pipe)
199  return SFE_NO_PIPE_WRITE ;
200 
201  wpriv->wavex_ambisonic = SF_AMBISONIC_NONE ;
202 
203  format = SF_CONTAINER (psf->sf.format) ;
204  if (format != SF_FORMAT_WAV && format != SF_FORMAT_WAVEX)
205  return SFE_BAD_OPEN_FORMAT ;
206 
207  psf->blockwidth = psf->bytewidth * psf->sf.channels ;
208 
209  /* RIFF WAVs are little-endian, RIFX WAVs are big-endian, default to little */
210  psf->endian = SF_ENDIAN (psf->sf.format) ;
211  if (CPU_IS_BIG_ENDIAN && psf->endian == SF_ENDIAN_CPU)
212  psf->endian = SF_ENDIAN_BIG ;
213  else if (psf->endian != SF_ENDIAN_BIG)
214  psf->endian = SF_ENDIAN_LITTLE ;
215 
216  if (psf->file.mode != SFM_RDWR || psf->filelength < 44)
217  { psf->filelength = 0 ;
218  psf->datalength = 0 ;
219  psf->dataoffset = 0 ;
220  psf->sf.frames = 0 ;
221  } ;
222 
223  if (subformat == SF_FORMAT_IMA_ADPCM || subformat == SF_FORMAT_MS_ADPCM)
224  { blockalign = wav_w64_srate2blocksize (psf->sf.samplerate * psf->sf.channels) ;
225  framesperblock = -1 ; /* Corrected later. */
226  } ;
227 
228  /* By default, add the peak chunk to floating point files. Default behaviour
229  ** can be switched off using sf_command (SFC_SET_PEAK_CHUNK, SF_FALSE).
230  */
231  if (psf->file.mode == SFM_WRITE && (subformat == SF_FORMAT_FLOAT || subformat == SF_FORMAT_DOUBLE))
232  { if ((psf->peak_info = peak_info_calloc (psf->sf.channels)) == NULL)
233  return SFE_MALLOC_FAILED ;
234  psf->peak_info->peak_loc = SF_PEAK_START ;
235  } ;
236 
237  psf->write_header = wav_write_header ;
238  } ;
239 
240  psf->container_close = wav_close ;
241  psf->command = wav_command ;
242 
243  switch (subformat)
244  { case SF_FORMAT_PCM_U8 :
245  case SF_FORMAT_PCM_16 :
246  case SF_FORMAT_PCM_24 :
247  case SF_FORMAT_PCM_32 :
248  error = pcm_init (psf) ;
249  break ;
250 
251  case SF_FORMAT_ULAW :
252  error = ulaw_init (psf) ;
253  break ;
254 
255  case SF_FORMAT_ALAW :
256  error = alaw_init (psf) ;
257  break ;
258 
259  /* Lite remove start */
260  case SF_FORMAT_FLOAT :
261  error = float32_init (psf) ;
262  break ;
263 
264  case SF_FORMAT_DOUBLE :
265  error = double64_init (psf) ;
266  break ;
267 
268  case SF_FORMAT_IMA_ADPCM :
269  error = wav_w64_ima_init (psf, blockalign, framesperblock) ;
270  break ;
271 
272  case SF_FORMAT_MS_ADPCM :
273  error = wav_w64_msadpcm_init (psf, blockalign, framesperblock) ;
274  break ;
275 
276  case SF_FORMAT_G721_32 :
277  error = g72x_init (psf) ;
278  break ;
279  /* Lite remove end */
280 
281  case SF_FORMAT_GSM610 :
282  error = gsm610_init (psf) ;
283  break ;
284 
285  default : return SFE_UNIMPLEMENTED ;
286  } ;
287 
288  if (psf->file.mode == SFM_WRITE || (psf->file.mode == SFM_RDWR && psf->filelength == 0))
289  return psf->write_header (psf, SF_FALSE) ;
290 
291  return error ;

298 static int
299 wav_read_header (SF_PRIVATE *psf, int *blockalign, int *framesperblock)
300 { WAV_PRIVATE *wpriv ;
301  WAV_FMT *wav_fmt ;
302  FACT_CHUNK fact_chunk ;
303  unsigned dword = 0, marker, RIFFsize = 0, done = 0 ;
304  int parsestage = 0, error, format = 0 ;
305  char *cptr ;
306 
307  if (psf->filelength > SF_PLATFORM_S64 (0xffffffff))
308  psf_log_printf (psf, "Warning : filelength > 0xffffffff. This is bad!!!!\n") ;
309 
310  if ((wpriv = psf->container_data) == NULL)
311  return SFE_INTERNAL ;
312  wav_fmt = &wpriv->wav_fmt ;
313 
314  /* Set position to start of file to begin reading header. */
315  psf_binheader_readf (psf, "p", 0) ;
316 
317  while (! done)
318  { psf_binheader_readf (psf, "m", &marker) ;
319 
320  switch (marker)
321  { case RIFF_MARKER :
322  case RIFX_MARKER :
323  if (parsestage)
324  return SFE_WAV_NO_RIFF ;
325 
326  parsestage |= HAVE_RIFF ;
327 
328  /* RIFX signifies big-endian format for all header and data
329  ** to prevent lots of code copying here, we'll set the psf->rwf_endian
330  ** flag once here, and never specify endian-ness for all other header ops
331  */
332  if (marker == RIFF_MARKER)
333  psf->rwf_endian = SF_ENDIAN_LITTLE ;
334  else
335  psf->rwf_endian = SF_ENDIAN_BIG ;
336 
337  psf_binheader_readf (psf, "4", &RIFFsize) ;
338 
339  if (psf->fileoffset > 0 && psf->filelength > RIFFsize + 8)
340  { /* Set file length. */
341  psf->filelength = RIFFsize + 8 ;
342  if (marker == RIFF_MARKER)
343  psf_log_printf (psf, "RIFF : %u\n", RIFFsize) ;
344  else
345  psf_log_printf (psf, "RIFX : %u\n", RIFFsize) ;
346  }
347  else if (psf->filelength < RIFFsize + 2 * SIGNED_SIZEOF (dword))
348  { if (marker == RIFF_MARKER)
349  psf_log_printf (psf, "RIFF : %u (should be %D)\n", RIFFsize, psf->filelength - 2 * SIGNED_SIZEOF (dword)) ;
350  else
351  psf_log_printf (psf, "RIFX : %u (should be %D)\n", RIFFsize, psf->filelength - 2 * SIGNED_SIZEOF (dword)) ;
352 
353  RIFFsize = dword ;
354  }
355  else
356  { if (marker == RIFF_MARKER)
357  psf_log_printf (psf, "RIFF : %u\n", RIFFsize) ;
358  else
359  psf_log_printf (psf, "RIFX : %u\n", RIFFsize) ;
360  } ;
361  break ;
362 
363  case WAVE_MARKER :
364  if ((parsestage & HAVE_RIFF) != HAVE_RIFF)
365  return SFE_WAV_NO_WAVE ;
366  parsestage |= HAVE_WAVE ;
367 
368  psf_log_printf (psf, "WAVE\n") ;
369  break ;
370 
371  case fmt_MARKER :
372  if ((parsestage & (HAVE_RIFF | HAVE_WAVE)) != (HAVE_RIFF | HAVE_WAVE))
373  return SFE_WAV_NO_FMT ;
374 
375  /* If this file has a SECOND fmt chunk, I don't want to know about it. */
376  if (parsestage & HAVE_fmt)
377  break ;
378 
379  parsestage |= HAVE_fmt ;
380 
381  psf_binheader_readf (psf, "4", &dword) ;
382  psf_log_printf (psf, "fmt : %d\n", dword) ;
383 
384  if ((error = wav_w64_read_fmt_chunk (psf, dword)))
385  return error ;
386 
387  format = wav_fmt->format ;
388  break ;
389 
390  case data_MARKER :
391  if ((parsestage & (HAVE_RIFF | HAVE_WAVE | HAVE_fmt)) != (HAVE_RIFF | HAVE_WAVE | HAVE_fmt))
392  return SFE_WAV_NO_DATA ;
393 
394  if (psf->file.mode == SFM_RDWR && (parsestage & HAVE_other) != 0)
395  return SFE_RDWR_BAD_HEADER ;
396 
397  parsestage |= HAVE_data ;
398 
399  psf_binheader_readf (psf, "4", &dword) ;
400 
401  psf->datalength = dword ;
402  psf->dataoffset = psf_ftell (psf) ;
403 
404  if (psf->dataoffset > 0)
405  { if (dword == 0 && RIFFsize == 8 && psf->filelength > 44)
406  { psf_log_printf (psf, "*** Looks like a WAV file which wasn't closed properly. Fixing it.\n") ;
407  psf->datalength = psf->filelength - psf->dataoffset ;
408  } ;
409 
410  if (psf->datalength > psf->filelength - psf->dataoffset)
411  { psf_log_printf (psf, "data : %D (should be %D)\n", psf->datalength, psf->filelength - psf->dataoffset) ;
412  psf->datalength = psf->filelength - psf->dataoffset ;
413  }
414  else
415  psf_log_printf (psf, "data : %D\n", psf->datalength) ;
416 
417  /* Only set dataend if there really is data at the end. */
418  if (psf->datalength + psf->dataoffset < psf->filelength)
419  psf->dataend = psf->datalength + psf->dataoffset ;
420 
421  if (format == WAVE_FORMAT_MS_ADPCM && psf->datalength % 2)
422  { psf->datalength ++ ;
423  psf_log_printf (psf, "*** Data length odd. Increasing it by 1.\n") ;
424  } ;
425  } ;
426 
427  if (! psf->sf.seekable || psf->dataoffset < 0)
428  break ;
429 
430  /* Seek past data and continue reading header. */
431  psf_fseek (psf, psf->datalength, SEEK_CUR) ;
432 
433  if (psf_ftell (psf) != psf->datalength + psf->dataoffset)
434  psf_log_printf (psf, "*** psf_fseek past end error ***\n") ;
435  break ;
436 
437  case fact_MARKER :
438  if ((parsestage & (HAVE_RIFF | HAVE_WAVE)) != (HAVE_RIFF | HAVE_WAVE))
439  return SFE_WAV_BAD_FACT ;
440 
441  parsestage |= HAVE_fact ;
442 
443  if ((parsestage & HAVE_fmt) != HAVE_fmt)
444  psf_log_printf (psf, "*** Should have 'fmt ' chunk before 'fact'\n") ;
445 
446  psf_binheader_readf (psf, "44", &dword, & (fact_chunk.frames)) ;
447 
448  if (dword > SIGNED_SIZEOF (fact_chunk))
449  psf_binheader_readf (psf, "j", (int) (dword - SIGNED_SIZEOF (fact_chunk))) ;
450 
451  if (dword)
452  psf_log_printf (psf, "%M : %d\n", marker, dword) ;
453  else
454  psf_log_printf (psf, "%M : %d (should not be zero)\n", marker, dword) ;
455 
456  psf_log_printf (psf, " frames : %d\n", fact_chunk.frames) ;
457  break ;
458 
459  case PEAK_MARKER :
460  if ((parsestage & (HAVE_RIFF | HAVE_WAVE | HAVE_fmt)) != (HAVE_RIFF | HAVE_WAVE | HAVE_fmt))
461  return SFE_WAV_PEAK_B4_FMT ;
462 
463  parsestage |= HAVE_PEAK ;
464 
465  psf_binheader_readf (psf, "4", &dword) ;
466 
467  psf_log_printf (psf, "%M : %d\n", marker, dword) ;
468  if (dword != WAV_PEAK_CHUNK_SIZE (psf->sf.channels))
469  { psf_binheader_readf (psf, "j", dword) ;
470  psf_log_printf (psf, "*** File PEAK chunk size doesn't fit with number of channels (%d).\n", psf->sf.channels) ;
471  return SFE_WAV_BAD_PEAK ;
472  } ;
473 
474  if ((psf->peak_info = peak_info_calloc (psf->sf.channels)) == NULL)
475  return SFE_MALLOC_FAILED ;
476 
477  /* read in rest of PEAK chunk. */
478  psf_binheader_readf (psf, "44", & (psf->peak_info->version), & (psf->peak_info->timestamp)) ;
479 
480  if (psf->peak_info->version != 1)
481  psf_log_printf (psf, " version : %d *** (should be version 1)\n", psf->peak_info->version) ;
482  else
483  psf_log_printf (psf, " version : %d\n", psf->peak_info->version) ;
484 
485  psf_log_printf (psf, " time stamp : %d\n", psf->peak_info->timestamp) ;
486  psf_log_printf (psf, " Ch Position Value\n") ;
487 
488  cptr = psf->u.cbuf ;
489  for (dword = 0 ; dword < (unsigned) psf->sf.channels ; dword++)
490  { float value ;
491  unsigned int position ;
492 
493  psf_binheader_readf (psf, "f4", &value, &position) ;
494  psf->peak_info->peaks [dword].value = value ;
495  psf->peak_info->peaks [dword].position = position ;
496 
497  snprintf (cptr, sizeof (psf->u.cbuf), " %2d %-12" PRId64 " %g\n",
498  dword, psf->peak_info->peaks [dword].position, psf->peak_info->peaks [dword].value) ;
499  cptr [sizeof (psf->u.cbuf) - 1] = 0 ;
500  psf_log_printf (psf, "%s", cptr) ;
501  } ;

1012 wav_write_header (SF_PRIVATE *psf, int calc_length)
1013 { sf_count_t current ;
1014  int k, error, has_data = SF_FALSE ;
1015 
1016  current = psf_ftell (psf) ;

1018  if (current > psf->dataoffset)
1019  has_data = SF_TRUE ;
1020

1023 
1024  psf->datalength = psf->filelength - psf->dataoffset ;
1025

1027  psf->datalength -= psf->filelength - psf->dataend ;
1028 
1029  if (psf->bytewidth > 0)
1030  psf->sf.frames = psf->datalength / (psf->bytewidth * psf->sf.channels) ;
1031  } ;

1040  ** To prevent lots of code copying here, we'll set the psf->rwf_endian flag
1041  ** once here, and never specify endian-ness for all other header operations.
1042  */
1043 
1044  /* RIFF/RIFX marker, length, WAVE and 'fmt ' markers. */
1045 
1046  if (psf->endian == SF_ENDIAN_LITTLE)
1047  psf_binheader_writef (psf, "etm8", RIFF_MARKER, (psf->filelength < 8) ? 8 : psf->filelength - 8) ;
1048  else
1049  psf_binheader_writef (psf, "Etm8", RIFX_MARKER, (psf->filelength < 8) ? 8 : psf->filelength - 8) ;
