20 long Audio_reader::read_window(float *data)
21 // reads the next window of samples
22 // the first time, fill half the window with zeros and the second half
23 // with data from the file
24 // after that, shift the window by hop_size and fill the end of the window
25 // with hop_size new samples
26 // the window is actually constructed in temp_data, then copied to data.
27 // That way, the caller can apply a smoothing function to data and we'll
28 // still have a copy.
29 // the function returns false on the next call when detecting that there
30 // is no more samples,
31 // data -- the window to be returned
32 // temp_data -- since we destroy data by windowing, temp_data saves
33 // overlapping samples so we don't have to read them again
34 // samples_per_frame -- must be even, note that first window is padded
35 // half-full with zeros
36 // hop_samples -- additional samples read each time after the first window
37 {
38  int frames_read; // how many frames did we read?
39 
40  int hop = hop_samples;
41  if (reading_first_window) {
42  hop = samples_per_frame / 2; // first time we read more data
43  // zero end of temp_data, which will shift to beginning
44  memset(temp_data + hop, 0,
45  sizeof(float) * (samples_per_frame - hop));
46  reading_first_window = false;
47  }
48 
49  // before reading in new sounds, shift temp_data by hop_size
50  memmove(temp_data, temp_data + hop,
51  (samples_per_frame - hop) * sizeof(float));
52 
53  frames_read = read(temp_data + samples_per_frame - hop, hop);
54  // zero any leftovers (happens at last frame):
55  //printf("check fr %i hs %i ws %i ",frames_read,hop_size,window_size);
56  memset(temp_data + samples_per_frame - hop + frames_read, 0,
57  sizeof(float) * (hop - frames_read));
58  assert(samples_per_frame - frames_read >= 0);
59 
60  // now copy temp_data to data
61  memcpy(data, temp_data, sizeof(float) * samples_per_frame);
62 
63  if (frames_read != hop && reading_last_window == false) {
64  reading_last_window = true;

66  } else if (reading_last_window == true) {
