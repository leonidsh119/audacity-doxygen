452 void KeyConfigPrefs::OnFilterKeyDown(wxKeyEvent & e)
453 {
454  wxTextCtrl *t = (wxTextCtrl *)e.GetEventObject();
455  int keycode = e.GetKeyCode();
456 
457  // Make sure we can navigate away from the hotkey textctrl.
458  // On Linux and OSX, it an get stuck, but it doesn't hurt
459  // to do it for Windows as well.
460  if (keycode == WXK_TAB) {
461  wxNavigationKeyEvent nevent;
462  nevent.SetWindowChange(e.ControlDown());
463  nevent.SetDirection(!e.ShiftDown());
464  nevent.SetEventObject(t);
465  nevent.SetCurrentFocus(t);
466  t->GetParent()->GetEventHandler()->ProcessEvent(nevent);
467 
468  return;
469  }
470 
471  if (mViewType == ViewByKey) {
472  wxString key = KeyStringDisplay(KeyEventToKeyString(e));
473  t->SetValue(key);
474 
475  if (key != wxEmptyString) {
476  mView->SetFilter(t->GetValue());
477  }
478  }
479  else
480  {
481  if (keycode == WXK_RETURN) {
482  mFilterPending = false;
483  mView->SetFilter(t->GetValue());
484  }
485  else {
486  mFilterPending = true;
487  mFilterTimer.Start(500, wxTIMER_ONE_SHOT);
488 
489  e.Skip();
490  }
491  }

529 {

621 bool KeyConfigPrefs::Apply()
622 {
623  // On the Mac, preferences may be changed without any active
624  // projects. This means that the CommandManager isn't availabe
625  // either. So we can't attempt to save preferences, otherwise
626  // NULL ptr dereferences will happen in ShuttleGui because the
627  // radio buttons are never created. (See Populate() above.)
628  if (!GetActiveProject()) {
629  return true;
630  }
631 
632  ShuttleGui S(this, eIsSavingToPrefs);
633  PopulateOrExchange(S);
634 
635  for (size_t i = 0; i < mNames.GetCount(); i++) {
636  wxString dkey = KeyStringNormalize(mDefaultKeys[i]);
637  wxString name = wxT("/NewKeys/") + mNames[i];
638  wxString key = KeyStringNormalize(mNewKeys[i]);
639 
640  if (gPrefs->HasEntry(name)) {
641  if (key != KeyStringNormalize(gPrefs->Read(name, key))) {
642  gPrefs->Write(name, key);
643  }
644  if (key == dkey) {
645  gPrefs->DeleteEntry(name);
646  }
647  }
648  else {
649  if (key != dkey) {
650  gPrefs->Write(name, key);
651  }
652  }
653  }
654 
655  return gPrefs->Flush();
656 }
