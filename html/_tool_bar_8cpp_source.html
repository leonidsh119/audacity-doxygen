175 void ToolBarResizer::OnMotion( wxMouseEvent & event )
176 {
177  // Go ahead and set the event to propagate
178  event.Skip();
179 
180  // Retrieve the mouse position
181  wxPoint raw_pos = event.GetPosition();
182  wxPoint pos = ClientToScreen( raw_pos );
183 
184  if( event.Dragging() )
185  {
186  wxRect r = mBar->GetRect();
187  wxSize msz = mBar->GetMinSize();
188  wxSize psz = mBar->GetParent()->GetClientSize();
189 
190  // Adjust the size by the difference between the
191  // last mouse and current mouse positions.
192  r.width += ( pos.x - mResizeStart.x );
193 
194  // Constrain
195  if( r.width < msz.x )
196  {
197  // Don't allow resizing to go too small
198  r.width = msz.x;
199  }
200  else if( r.GetRight() > psz.x - 3 )
201  {
202  // Don't allow resizing to go too large
203  //
204  // The 3 magic pixels are because I'm too chicken to change the
205  // calculations in ToolDock::LayoutToolBars() even though I'm
206  // the one that set them up. :-)
207  r.SetRight( psz.x - 3 );
208  }
209  else
210  {
211  // Remember for next go round
212  mResizeStart = pos;
213  }
214 
215  // Resize the bar
216  mBar->SetSize( r.GetSize() );
217 
218  // Tell everyone we've changed sizes
219  mBar->Updated();
220 
221  // Refresh our world
222  mBar->GetParent()->Refresh();
223  mBar->GetParent()->Update();
224  }
225 }

414 void ToolBar::ReCreateButtons()
415 {
416  // SetSizer(NULL) detaches mHSizer and deletes it.
417  // Do not use Detach() here, as that attempts to detach mHSizer from itself!
418  SetSizer( NULL );
419 
420  // Get rid of any children we may have
421  DestroyChildren();
422  mGrabber = NULL;
423  mResizer = NULL;
424 
425  {
426  // Create the main sizer
427  auto ms = std::make_unique<wxBoxSizer>(wxHORIZONTAL);
428 
429  // Create the grabber and add it to the main sizer
430  mGrabber = safenew Grabber(this, mType);
431  ms->Add(mGrabber, 0, wxEXPAND | wxALIGN_LEFT | wxALIGN_TOP | wxRIGHT, 1);
432 
433  // Use a box sizer for laying out controls
434  ms->Add((mHSizer = safenew wxBoxSizer(wxHORIZONTAL)), 1, wxEXPAND);
435 
436  // (Re)Establish dock state
437  SetDocked(GetDock(), false);
438 
439  // Go add all the rest of the gadgets
440  Populate();
441 
442  // Add some space for the resize border
443  if (IsResizable())
444  {
445  // Create the resizer and add it to the main sizer
446  mResizer = safenew ToolBarResizer(this);
447  ms->Add(mResizer, 0, wxEXPAND | wxALIGN_TOP | wxLEFT, 1);
448  mResizer->SetToolTip(_("Click and drag to resize toolbar"));
449  }
450 
451  // Set the sizer
452  SetSizerAndFit(ms.release());
453  }
454 
455  // Recalculate the height to be a multiple of toolbarSingle
456  const int tbs = toolbarSingle + toolbarGap;
457  wxSize sz = GetSize();
458  sz.y = ( ( ( sz.y + tbs -1) / tbs ) * tbs ) - 1;
459 
460  // Set the true AND minimum sizes and do final layout
461  if(IsResizable())
462  {
463  sz.SetWidth(GetMinToolbarWidth());
464  // JKC we're going to allow all resizable toolbars to be resized
465  // to 1 unit high!
466  wxSize sz2 = sz;
467  sz2.y = tbs -1;
468  SetMinSize(sz2);
469  sz.SetWidth(GetInitialWidth());
470  SetSize(sz);
471  }
472  else
473  {
474  SetInitialSize(sz);
475  }
476  Layout();
477 }
