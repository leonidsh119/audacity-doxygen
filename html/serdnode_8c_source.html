130 {

137  case '/': // separator

148 SerdNode
149 serd_node_new_file_uri(const uint8_t* path,
150  const uint8_t* hostname,
151  SerdURI* out,
152  bool escape)
153 {
154  const size_t path_len = strlen((const char*)path);
155  const size_t hostname_len = hostname ? strlen((const char*)hostname) : 0;
156  const bool evil = is_windows_path(path);
157  size_t uri_len = 0;
158  uint8_t* uri = NULL;
159 
160  if (path[0] == '/' || is_windows_path(path)) {
161  uri_len = strlen("file://") + hostname_len + evil;
162  uri = (uint8_t*)malloc(uri_len + 1);
163  snprintf((char*)uri, uri_len + 1, "file://%s%s",
164  hostname ? (const char*)hostname : "",
165  evil ? "/" : "");
166  }
167 
168  SerdChunk chunk = { uri, uri_len };
169  for (size_t i = 0; i < path_len; ++i) {
170  if (evil && path[i] == '\\') {
171  serd_chunk_sink("/", 1, &chunk);
172  } else if (path[i] == '%') {
173  serd_chunk_sink("%%", 2, &chunk);
174  } else if (!escape || is_uri_path_char(path[i])) {
175  serd_chunk_sink(path + i, 1, &chunk);
176  } else {
177  char escape_str[4] = { '%', 0, 0, 0 };
178  snprintf(escape_str + 1, sizeof(escape_str) - 1, "%X", path[i]);
179  serd_chunk_sink(escape_str, 3, &chunk);
180  }
181  }
182  serd_chunk_sink_finish(&chunk);
183 
184  if (out) {
185  serd_uri_parse(chunk.buf, out);
186  }
187 
188  return serd_node_from_string(SERD_URI, chunk.buf);
189 }

219 SERD_API
220 SerdNode
221 serd_node_new_decimal(double d, unsigned frac_digits)
222 {
223  if (isnan(d) || isinf(d)) {
224  return SERD_NODE_NULL;
225  }
226 
227  const double abs_d = fabs(d);
228  const unsigned int_digits = (unsigned)fmax(1.0, ceil(log10(abs_d + 1)));
229  char* buf = (char*)calloc(int_digits + frac_digits + 3, 1);
230  SerdNode node = { (const uint8_t*)buf, 0, 0, 0, SERD_LITERAL };
231  const double int_part = floor(abs_d);
232 
233  // Point s to decimal point location
234  char* s = buf + int_digits;
235  if (d < 0.0) {
236  *buf = '-';
237  ++s;
238  }
239 
240  // Write integer part (right to left)
241  char* t = s - 1;
242  uint64_t dec = (uint64_t)int_part;
243  do {
244  *t-- = '0' + (dec % 10);
245  } while ((dec /= 10) > 0);
246 
247  *s++ = '.';
248 
249  // Write fractional part (right to left)
250  double frac_part = fabs(d - int_part);
251  if (frac_part < DBL_EPSILON) {
252  *s++ = '0';
253  node.n_bytes = node.n_chars = (s - buf);
254  } else {
255  uint64_t frac = frac_part * pow(10.0, (int)frac_digits) + 0.5;
256  s += frac_digits - 1;
257  unsigned i = 0;
258 
259  // Skip trailing zeros
260  for (; i < frac_digits - 1 && !(frac % 10); ++i, --s, frac /= 10) {}
261 
262  node.n_bytes = node.n_chars = (s - buf) + 1;
263 
264  // Write digits from last trailing zero to decimal point
265  for (; i < frac_digits; ++i) {
266  *s-- = '0' + (frac % 10);
267  frac /= 10;
268  }
269  }
270 
271  return node;
272 }

310 static inline void
311 encode_chunk(uint8_t out[4], const uint8_t in[3], size_t n_in)
312 {
313  out[0] = b64_map[in[0] >> 2];
314  out[1] = b64_map[((in[0] & 0x03) << 4) | ((in[1] & 0xF0) >> 4)];
315  out[2] = ((n_in > 1)
316  ? (b64_map[((in[1] & 0x0F) << 2) | ((in[2] & 0xC0) >> 6)])
317  : (uint8_t)'=');
318  out[3] = ((n_in > 2) ? b64_map[in[2] & 0x3F] : (uint8_t)'=');
