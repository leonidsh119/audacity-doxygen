38 int Midifile_reader::readmt(char *s, int skip)
39  /* read through the "MThd" or "MTrk" header string */
40  /* if skip == 1, we attempt to skip initial garbage. */
41 {
42  assert(strlen(s) == 4); // must be "MThd" or "MTrk"
43  int nread = 0;
44  char b[4];
45  char buff[32];
46  int c;
47  char *errmsg = "expecting ";
48 
49  retry:
50  while ( nread<4 ) {
51  c = Mf_getc();
52  if ( c == EOF ) {
53  errmsg = "EOF while expecting ";
54  goto err;
55  }
56  b[nread++] = c;
57  }
58  /* See if we found the 4 characters we're looking for */
59  if ( s[0]==b[0] && s[1]==b[1] && s[2]==b[2] && s[3]==b[3] )
60  return(0);
61  if ( skip ) {
62  /* If we are supposed to skip initial garbage, */
63  /* try again with the next character. */
64  b[0]=b[1];
65  b[1]=b[2];
66  b[2]=b[3];
67  nread = 3;
68  goto retry;
69  }
70  err:
71 #pragma warning(disable: 4996) // strcpy is safe since strings have known lengths
72  (void) strcpy(buff,errmsg);
73  (void) strcat(buff,s);
74 #pragma warning(default: 4996) // turn it back on
75  mferror(buff);
76  return(0);
77 }

117 void Midifile_reader::readtrack()
118  /* read a track chunk */
119 {
120  /* This array is indexed by the high half of a status byte. It's */
121  /* value is either the number of bytes needed (1 or 2) for a channel */
122  /* message, or 0 (meaning it's not a channel message). */
123  static int chantype[] = {
124  0, 0, 0, 0, 0, 0, 0, 0, /* 0x00 through 0x70 */
125  2, 2, 2, 2, 1, 1, 2, 0 /* 0x80 through 0xf0 */
126  };
127  long lookfor, lng;
128  int c, c1, type;
129  int sysexcontinue = 0; /* 1 if last message was an unfinished sysex */
130  int running = 0; /* 1 when running status used */
131  int status = 0; /* (possibly running) status byte */
132  int needed;
133 
134  if ( readmt("MTrk",0) == EOF )
135  return;
136 
137  Mf_toberead = read32bit();
138 
139  if (midifile_error) return;
140 
141  Mf_currtime = 0L;
142 
143  Mf_starttrack();
144 
145  while ( Mf_toberead > 0 ) {
146 
147  Mf_currtime += readvarinum(); /* delta time */
148  if (midifile_error) return;
149 
150  c = egetc();
151  if (midifile_error) return;
152 
153  if ( sysexcontinue && c != 0xf7 ) {
154  mferror("didn't find expected continuation of a sysex");
155  return;
156  }
157  if ( (c & 0x80) == 0 ) { /* running status? */
158  if ( status == 0 ) {
159  mferror("unexpected running status");
160  return;
161  }
162  running = 1;
163  } else {
164  status = c;
165  running = 0;
166  }
167 
168  needed = chantype[ (status>>4) & 0xf ];
169 
170  if ( needed ) { /* ie. is it a channel message? */
171 
172  if ( running )
173  c1 = c;
174  else {
175  c1 = egetc();
176  if (midifile_error) return;
177  }
178  chanmessage( status, c1, (needed>1) ? egetc() : 0 );
179  if (midifile_error) return;
180  continue;;
181  }
182 
183  switch ( c ) {
184 
185  case 0xff: /* meta event */
186 
187  type = egetc();
188  if (midifile_error) return;
189  /* watch out - Don't combine the next 2 statements */
190  lng = readvarinum();
191  if (midifile_error) return;
192  lookfor = Mf_toberead - lng;
193  msginit();
194 
195  while ( Mf_toberead > lookfor ) {
196  unsigned char c = egetc();
197  if (midifile_error) return;
198  msgadd(c);
199  }
200  metaevent(type);
201  break;
202 
203  case 0xf0: /* start of system exclusive */
204 
205  /* watch out - Don't combine the next 2 statements */
206  lng = readvarinum();
207  if (midifile_error) return;
208  lookfor = Mf_toberead - lng;
209  msginit();
210  msgadd(0xf0);
211 
212  while ( Mf_toberead > lookfor ) {
213  c = egetc();
214  if (midifile_error) return;
215  msgadd(c);
216  }
217  if ( c==0xf7 || Mf_nomerge==0 )
218  sysex();
219  else
220  sysexcontinue = 1; /* merge into next msg */
221  break;
222 
223  case 0xf7: /* sysex continuation or arbitrary stuff */
224 
225  /* watch out - Don't combine the next 2 statements */
226  lng = readvarinum();
227  if (midifile_error) return;
228  lookfor = Mf_toberead - lng;
229 
230  if ( ! sysexcontinue )
231  msginit();
232 
233  while ( Mf_toberead > lookfor ) {
234  c = egetc();
235  if (midifile_error) return;
236  msgadd(c);
237  }
238  if ( ! sysexcontinue ) {
239  Mf_arbitrary(msgleng(), msg());
240  }
241  else if ( c == 0xf7 ) {
242  sysex();
243  sysexcontinue = 0;
244  }
245  break;
246  default:
247 
248  badbyte(c);
249 
250  break;
251  }
252  }
253  Mf_endtrack();
254  return;
255 }

266 void Midifile_reader::metaevent(int type)
267 {
268  int leng = msgleng();
269  // made this unsigned to avoid sign extend
270  unsigned char *m = msg();
271 
272  switch ( type ) {
273  case 0x00:
274  Mf_seqnum(to16bit(m[0],m[1]));
275  break;
276  case 0x01: /* Text event */
277  case 0x02: /* Copyright notice */
278  case 0x03: /* Sequence/Track name */
279  case 0x04: /* Instrument name */
280  case 0x05: /* Lyric */
281  case 0x06: /* Marker */
282  case 0x07: /* Cue point */
283  case 0x08:
284  case 0x09:
285  case 0x0a:
286  case 0x0b:
287  case 0x0c:
288  case 0x0d:
289  case 0x0e:
290  case 0x0f:
291  /* These are all text events */
292  Mf_text(type,leng,m);
293  break;
294  case 0x20:
295  Mf_chanprefix(m[0]);
296  break;
297  case 0x21:
298  Mf_portprefix(m[0]);
299  break;
300  case 0x2f: /* End of Track */
301  Mf_eot();
302  break;
303  case 0x51: /* Set tempo */
304  Mf_tempo(to32bit(0,m[0],m[1],m[2]));
305  break;
306  case 0x54:
307  Mf_smpte(m[0],m[1],m[2],m[3],m[4]);
308  break;
309  case 0x58:
310  Mf_timesig(m[0],m[1],m[2],m[3]);
311  break;
312  case 0x59:
313  Mf_keysig(m[0],m[1]);
314  break;
315  case 0x7f:
316  Mf_sqspecific(leng,m);
317  break;
318  default:
319  Mf_metamisc(type,leng,m);
320  }
321 }
