338 // Bouncing Ball:
339 // Given the current time t, returns the x/y position of the scrolling
340 // karaoke display. For some syllable i, when t==mSyllables[i].t,
341 // it will return mSyllables[i].x for outX and 0 for outY.
342 // In-between words, outX is interpolated using smooth acceleration
343 // between the two neighboring words, and y is a positive number indicating
344 // the bouncing ball height
345 void Lyrics::GetKaraokePosition(double t,
346  int *outX, double *outY)
347 {
348  *outX = 0;
349  *outY = 0;
350 
351  if (t < mSyllables[I_FIRST_REAL_SYLLABLE].t || t > mSyllables[mSyllables.GetCount()-3].t)
352  return;
353 
354  int i0, i1, i2, i3;
355  int x0, x1, x2, x3;
356  double t0, t1, t2, t3;
357  i1 = FindSyllable(t);
358  i2 = i1 + 1;
359 
360  // Because we've padded the syllables with two dummies at the beginning
361  // and end, we know that i0...i3 will always exist. Also, we've made
362  // sure that we don't ever have two of the same time, so t2>t1 strictly.
363  //
364  // t
365  // \/
366  // time: t0 t1 t2 t3
367  // pos: x0 x1 x2 x3
368  // index: i0 i1 i2 i3
369  // vel: vel1 vel2
370 
371  i0 = i1 - 1;
372  i3 = i2 + 1;
373 
374  x0 = mSyllables[i0].x;
375  x1 = mSyllables[i1].x;
376  x2 = mSyllables[i2].x;
377  x3 = mSyllables[i3].x;
378 
379  t0 = mSyllables[i0].t;
380  t1 = mSyllables[i1].t;
381  t2 = mSyllables[i2].t;
382  t3 = mSyllables[i3].t;
383 
384  double linear_vel0 = (x1 - x0) / (t1 - t0);
385  double linear_vel1 = (x2 - x1) / (t2 - t1);
386  double linear_vel2 = (x3 - x2) / (t3 - t2);
387 
388  // average velocities
389  double v1 = (linear_vel0 + linear_vel1) / 2;
390  double v2 = (linear_vel1 + linear_vel2) / 2;
391 
392  // Solve a cubic equation f(t) = at^3 + bt^2 + ct + d
393  // which gives the position x as a function of
394  // (t - t1), by constraining f(0), f'(0), f(t2-t1), f'(t2-t1)
395  double delta_t = t2 - t1;
396  double delta_x = x2 - x1;
397  v1 *= delta_t;
398  v2 *= delta_t;
399  double a = v1 + v2 - 2*delta_x;
400  double b = 3*delta_x - 2*v1 - v2;
401  double c = v1;
402  double d = x1;
403 
404  t = (t - t1) / (t2 - t1);
405  double xx = a*t*t*t + b*t*t + c*t + d;
406 
407  // Unfortunately sometimes our cubic goes backwards. This is a quick
408  // hack to stop that from happening.
409  if (xx < x1)
410  xx = x1;
411 
412  *outX = (int)xx;
413 
414  // The y position is a simple cosine curve; the max height is a
415  // function of the time.
416  double height = t2 - t1 > 4.0? 1.0: sqrt((t2-t1)/4.0);
417  *outY = height * sin(M_PI * t);
418 }
