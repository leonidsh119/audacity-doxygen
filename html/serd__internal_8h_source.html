116 {

173 static inline size_t
174 serd_bulk_sink_write(const void* buf, size_t len, SerdBulkSink* bsink)
175 {
176  const size_t orig_len = len;
177  while (len) {
178  const size_t space = bsink->block_size - bsink->size;
179  const size_t n = MIN(space, len);
180 
181  // Write as much as possible into the remaining buffer space
182  memcpy(bsink->buf + bsink->size, buf, n);
183  bsink->size += n;
184  buf = (const uint8_t*)buf + n;
185  len -= n;
186 
187  // Flush page if buffer is full
188  if (bsink->size == bsink->block_size) {
189  bsink->sink(bsink->buf, bsink->block_size, bsink->stream);
190  bsink->size = 0;
191  }
192  }
193  return orig_len;
194 }

269 static inline bool
270 uri_is_under(const SerdURI* uri, const SerdURI* root)
271 {
272  if (!root || !root->scheme.len ||
273  !chunk_equals(&root->scheme, &uri->scheme) ||
274  !chunk_equals(&root->authority, &uri->authority)) {
275  return false;
276  }
277 
278  bool differ = false;
279  const size_t path_len = uri_path_len(uri);
280  const size_t root_len = uri_path_len(root);
281  for (size_t i = 0; i < path_len && i < root_len; ++i) {
282  if (uri_path_at(uri, i) != uri_path_at(root, i)) {
283  differ = true;
284  }
285  if (differ && uri_path_at(root, i) == '/') {
286  return false;
287  }
288  }
289 
290  return true;
291 }
