235 /* This indicates you're ready to start staging audio data for the
236  currently opened device. At this point you are committing to
237  exclusive access to the CD burner, and this is the function that
238  will fail if another program is using the device, or if there is
239  no writable CD media in the device at this point.
240  You should pass in the path to a temporary directory that has at
241  least 700 MB of free space, to stage the audio, although note that
242  not all implementations will make use of this directory. */
243 int PortBurn_StartStaging(void *handle, const char *tmpdir)
244 
245 {
246  PBHandle *h = (PBHandle *)handle;
247  CFDictionaryRef deviceStatus;
248  CFStringRef mediaState;
249  CFDictionaryRef mediaInfo;
250  CFBooleanRef blank;
251  CFBooleanRef writable;
252  CFBooleanRef reserved;
253 
254  if (!h)
255  return pbErrNoHandle;
256 
257  if (!h->device)
258  return pbErrDeviceNotOpen;
259 
260  if (h->staging)
261  return pbErrAlreadyStagingOrBurning;
262 
263  /* First we check to see that we have blank media in the drive. */
264  deviceStatus = DRDeviceCopyStatus(h->device);
265  mediaState = (CFStringRef)CFDictionaryGetValue(deviceStatus,
266  kDRDeviceMediaStateKey);
267  if (mediaState == NULL) {
268  CFRelease(deviceStatus);
269  return pbErrCannotAccessDevice;
270  }
271 
272  if (!CFEqual(mediaState, kDRDeviceMediaStateMediaPresent)) {
273  CFRelease(deviceStatus);
274  return pbErrNoMediaInDrive;
275  }
276 
277  mediaInfo = (CFDictionaryRef)CFDictionaryGetValue(deviceStatus,
278  kDRDeviceMediaInfoKey);
279  blank = (CFBooleanRef)CFDictionaryGetValue(mediaInfo,
280  kDRDeviceMediaIsBlankKey);
281 
282  if (blank == NULL || !CFBooleanGetValue(blank)) {
283  CFRelease(deviceStatus);
284  return pbErrMediaIsNotBlankAndWritable;
285  }
286 
287  writable = (CFBooleanRef)CFDictionaryGetValue(mediaInfo,
288  kDRDeviceMediaIsAppendableKey);
289 
290  if (writable == NULL || !CFBooleanGetValue(writable)) {
291  CFRelease(deviceStatus);
292  return pbErrMediaIsNotBlankAndWritable;
293  }
294 
295  reserved = (CFBooleanRef)CFDictionaryGetValue(mediaInfo,
296  kDRDeviceMediaIsReservedKey);
297  if (reserved == NULL || !CFBooleanGetValue(reserved)) {
298  CFRelease(deviceStatus);
299  return pbErrCannotReserveDevice;
300  }
301 
302  CFRelease(deviceStatus);
303 
304  h->staging = PortBurn_TempDirStaging(tmpdir);
305 
306  if (!h->staging) {
307  return pbErrCannotCreateStagingDirectory;

311  &kCFTypeArrayCallBacks);

448 int PortBurn_GetStatus(void *handle, float *out_fraction_complete)
449 {
450  PBHandle *h = (PBHandle *)handle;
451  CFDictionaryRef status;
452  CFStringRef stateRef;
453  CFNumberRef fracRef;
454  CFNumberRef trackNumRef;
455  float frac = 0.0;
456  int trackNum = 0;
457 
458  *out_fraction_complete = h->frac;
459 
460  if (!h)
461  return pbErrNoHandle;
462 
463  if (!h->burn)
464  return pbErrNotCurrentlyBurning;
465 
466  status = DRBurnCopyStatus(h->burn);
467  if (!status)
468  return pbErrCannotGetBurnStatus;
469 
470  trackNumRef = (CFNumberRef)
471  CFDictionaryGetValue(status, kDRStatusCurrentTrackKey);
472  if (trackNumRef != NULL) {
473  CFNumberGetValue(trackNumRef, kCFNumberIntType, &trackNum);
474  }
475 
476  fracRef = (CFNumberRef)
477  CFDictionaryGetValue(status, kDRStatusPercentCompleteKey);
478  if (fracRef != NULL) {
479  CFNumberGetValue(fracRef, kCFNumberFloatType, &frac);
480  }
481 
482  stateRef = (CFStringRef)
483  CFDictionaryGetValue(status, kDRStatusStateKey);
484  if (stateRef == NULL) {
485  /* Stick with the last percentage */
486  return pbSuccess;
487  }
488 
489  if (CFEqual(stateRef, kDRStatusStateNone)) {
490  /* Stick with the last percentage */
491  return pbSuccess;
492  }
493 
494  if (CFEqual(stateRef, kDRStatusStatePreparing)) {
495  /* This takes about 1 second */
496  h->frac = 0.01;
497  }
498 
499  if (CFEqual(stateRef, kDRStatusStateSessionOpen)) {
500  /* This takes about 3 seconds */
501  h->frac = 0.02;
502  }
503 
504  if (CFEqual(stateRef, kDRStatusStateTrackOpen) ||
505  CFEqual(stateRef, kDRStatusStateTrackWrite) ||
506  CFEqual(stateRef, kDRStatusStateTrackClose) ||
507  CFEqual(stateRef, kDRStatusStateSessionClose)) {
508  /* The fraction ("percentage") complete will be valid during
509  the majority of this range */
510  float newFrac = 0.0;
511  if (frac > 0.0 && frac <= 1.0)
512  newFrac = frac;
513 
514  /* Scale it to the range 0.05 - 0.99 */
515  newFrac = 0.05 + 0.94 * newFrac;
516 
517  /* Only use that value if it's larger than the previous value */
518  if (newFrac > h->frac)
519  h->frac = newFrac;
520  }
521 
522  if (CFEqual(stateRef, kDRStatusStateDone)) {
523  /* Returning a fraction complete of 1.0 means we're done! */
524  h->frac = 1.0;
525  }
526 
527  if (CFEqual(stateRef, kDRStatusStateFailed)) {
528  CFRelease(status);
529  return pbErrBurnFailed;
530  }
531 
532  *out_fraction_complete = h->frac;
533 
534  CFRelease(status);
535  return pbSuccess;
536 }
