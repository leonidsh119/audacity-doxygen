135 static void AddSourcesFromStream(int deviceIndex, const PaDeviceInfo *info, std::vector<DeviceSourceMap> *maps, PaStream *stream)
136 {
137 #ifdef USE_PORTMIXER
138  int i;
139 #endif
140  DeviceSourceMap map;
141 
142  map.sourceIndex = -1;
143  map.totalSources = 0;
144  // Only inputs have sources, so we call FillHostDeviceInfo with a 1 to indicate this
145  FillHostDeviceInfo(&map, info, deviceIndex, 1);
146 
147 #ifdef USE_PORTMIXER
148  PxMixer *portMixer = Px_OpenMixer(stream, 0);
149  if (!portMixer) {
150  maps->push_back(map);
151  return;
152  }
153 
154  //if there is only one source, we don't need to concatenate the source
155  //or enumerate, because it is something meaningless like 'master'
156  //(as opposed to 'mic in' or 'line in'), and the user doesn't have any choice.
157  //note that some devices have no input sources at all but are still valid.
158  //the behavior we do is the same for 0 and 1 source cases.
159  map.totalSources = Px_GetNumInputSources(portMixer);
160 #endif
161 
162  if (map.totalSources <= 1) {
163  map.sourceIndex = 0;
164  maps->push_back(map);
165  }
166 #ifdef USE_PORTMIXER
167  else {
168  //open up a stream with the device so portmixer can get the info out of it.
169  for (i = 0; i < map.totalSources; i++) {
170  map.sourceIndex = i;
171  map.sourceString = wxString(wxSafeConvertMB2WX(Px_GetInputSourceName(portMixer, i)));
172  maps->push_back(map);
173  }
174  }
175  Px_CloseMixer(portMixer);
176 #endif
177 }

179 static bool IsInputDeviceAMapperDevice(const PaDeviceInfo *info)
180 {
181  // For Windows only, portaudio returns the default mapper object
182  // as the first index after a NEW hostApi index is detected (true for MME and DS)
183  // this is a bit of a hack, but there's no other way to find out which device is a mapper,
184  // I've looked at string comparisons, but if the system is in a different language this breaks.
185 #ifdef __WXMSW__
186  static int lastHostApiTypeId = -1;
187  int hostApiTypeId = Pa_GetHostApiInfo(info->hostApi)->type;
188  if(hostApiTypeId != lastHostApiTypeId &&
189  (hostApiTypeId == paMME || hostApiTypeId == paDirectSound)) {
190  lastHostApiTypeId = hostApiTypeId;
191  return true;
192  }
193 #endif
194 
195  return false;
196 }

198 static void AddSources(int deviceIndex, int rate, std::vector<DeviceSourceMap> *maps, int isInput)
199 {
200  int error = 0;
201  DeviceSourceMap map;
202  const PaDeviceInfo *info = Pa_GetDeviceInfo(deviceIndex);
203 
204  // This tries to open the device with the samplerate worked out above, which
205  // will be the highest available for play and record on the device, or
206  // 44.1kHz if the info cannot be fetched.
207 
208  PaStream *stream = NULL;
209 
210  PaStreamParameters parameters;
211 
212  parameters.device = deviceIndex;
213  parameters.sampleFormat = paFloat32;
214  parameters.hostApiSpecificStreamInfo = NULL;
215  parameters.channelCount = 1;
216 
217  // If the device is for input, open a stream so we can use portmixer to query
218  // the number of inputs. We skip this for outputs because there are no 'sources'
219  // and some platforms (e.g. XP) have the same device for input and output, (while
220  // Vista/Win7 seperate these into two devices with the same names (but different
221  // portaudio indecies)
222  // Also, for mapper devices we don't want to keep any sources, so check for it here
223  if (isInput && !IsInputDeviceAMapperDevice(info)) {
224  if (info)
225  parameters.suggestedLatency = info->defaultLowInputLatency;
226  else
227  parameters.suggestedLatency = 10.0;
228 
229  error = Pa_OpenStream(&stream,
230  &parameters,
231  NULL,
232  rate, paFramesPerBufferUnspecified,
233  paClipOff | paDitherOff,
234  DummyPaStreamCallback, NULL);
235  }
236 
237  if (stream && !error) {
238  AddSourcesFromStream(deviceIndex, info, maps, stream);
239  Pa_CloseStream(stream);
240  } else {
241  map.sourceIndex = -1;
242  map.totalSources = 0;
243  FillHostDeviceInfo(&map, info, deviceIndex, isInput);
244  maps->push_back(map);
245  }
246 
247  if(error) {
248  wxLogDebug(wxT("PortAudio stream error creating device list: ") +
249  map.hostString + wxT(":") + map.deviceString + wxT(": ") +
250  wxString(wxSafeConvertMB2WX(Pa_GetErrorText((PaError)error))));
251  }
252 }

257 void DeviceManager::Rescan()
258 {
259  // get rid of the previous scan info
260  this->mInputDeviceSourceMaps.clear();
261  this->mOutputDeviceSourceMaps.clear();
262 
263  // if we are doing a second scan then restart portaudio to get NEW devices
264  if (m_inited) {
265  // check to see if there is a stream open - can happen if monitoring,
266  // but otherwise Rescan() should not be available to the user.
267  if (gAudioIO) {
268  if (gAudioIO->IsMonitoring())
269  {
270  gAudioIO->StopStream();
271  while (gAudioIO->IsBusy())
272  wxMilliSleep(100);
273  }
274  }
275 
276  // restart portaudio - this updates the device list
277  Pa_Terminate();
278  Pa_Initialize();
279  }
280 
281  int nDevices = Pa_GetDeviceCount();
282 
283  //The heirarchy for devices is Host/device/source.
284  //Some newer systems aggregate this.
285  //So we need to call port mixer for every device to get the sources
286  for (int i = 0; i < nDevices; i++) {
287  const PaDeviceInfo *info = Pa_GetDeviceInfo(i);
288  if (info->maxOutputChannels > 0) {
289  AddSources(i, info->defaultSampleRate, &mOutputDeviceSourceMaps, 0);
290  }
291 
292  if (info->maxInputChannels > 0) {
293 #ifdef __WXMSW__
294 #if !defined(EXPERIMENTAL_FULL_WASAPI)
295  if (Pa_GetHostApiInfo(info->hostApi)->type != paWASAPI ||
296  PaWasapi_IsLoopback(i) > 0)
297 #endif
298 #endif
299  AddSources(i, info->defaultSampleRate, &mInputDeviceSourceMaps, 1);
300  }
301  }
302 
303  // If this was not an initial scan update each device toolbar.
304  // Hosts may have disappeared or appeared so a complete repopulate is needed.
305  if (m_inited) {
306  DeviceToolBar *dt;
307  for (size_t i = 0; i < gAudacityProjects.GetCount(); i++) {
308  dt = gAudacityProjects[i]->GetDeviceToolBar();
309  dt->RefillCombos();
310  }
311  }
312  m_inited = true;
313 }
