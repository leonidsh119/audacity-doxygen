178 {
179  TrackList *tracks = proj->GetTracks();
180  TrackListIterator iter1(tracks);
181  Track *tr = iter1.First();
182 
183  while (tr)
184  {
185  //go over all tracks in the project
186  if (tr->GetKind() == Track::Wave)
187  {
188  //look inside our task's track list for one that matches this projects one.
189  mWaveTrackMutex.Lock();
190  for(int i=0;i<(int)mWaveTracks.size();i++)
191  {
192  if(mWaveTracks[i]==tr)
193  {
194  //if we find one, then the project is associated with us;return true
195  mWaveTrackMutex.Unlock();
196  return true;
197  }
198  }
199  mWaveTrackMutex.Unlock();
200  }
201  tr = iter1.Next();
202  }
203 
204  return false;
