139 static int
140 ogg_read_header (SF_PRIVATE *psf, int log_data)
141 {
142  OGG_PRIVATE *odata = (OGG_PRIVATE *) psf->container_data ;
143  VORBIS_PRIVATE *vdata = (VORBIS_PRIVATE *) psf->codec_data ;
144  char *buffer ;
145  int bytes ;
146  int i, nn ;
147 
148  odata->eos = 0 ;
149 
150  /* Weird stuff happens if these aren't called. */
151  ogg_stream_reset (&odata->os) ;
152  ogg_sync_reset (&odata->oy) ;
153 
154  /*
155  ** Grab some data at the head of the stream. We want the first page
156  ** (which is guaranteed to be small and only contain the Vorbis
157  ** stream initial header) We need the first page to get the stream
158  ** serialno.
159  */
160 
161  /* Expose the buffer */
162  buffer = ogg_sync_buffer (&odata->oy, 4096L) ;
163 
164  /* Grab the part of the header that has already been read. */
165  memcpy (buffer, psf->header, psf->headindex) ;
166  bytes = psf->headindex ;
167 
168  /* Submit a 4k block to libvorbis' Ogg layer */
169  bytes += psf_fread (buffer + psf->headindex, 1, 4096 - psf->headindex, psf) ;
170  ogg_sync_wrote (&odata->oy, bytes) ;
171 
172  /* Get the first page. */
173  if ((nn = ogg_sync_pageout (&odata->oy, &odata->og)) != 1)
174  {
175  /* Have we simply run out of data? If so, we're done. */
176  if (bytes < 4096)
177  return 0 ;
178 
179  /* Error case. Must not be Vorbis data */
180  psf_log_printf (psf, "Input does not appear to be an Ogg bitstream.\n") ;
181  return SFE_MALFORMED_FILE ;
182  } ;
183 
184  /*
185  ** Get the serial number and set up the rest of decode.
186  ** Serialno first ; use it to set up a logical stream.
187  */
188  ogg_stream_clear (&odata->os) ;
189  ogg_stream_init (&odata->os, ogg_page_serialno (&odata->og)) ;
190 
191  /*
192  ** This function (ogg_read_header) gets called multiple times, so the OGG
193  ** and vorbis structs have to be cleared every time we pass through to
194  ** prevent memory leaks.
195  */
196  vorbis_block_clear (&vdata->vb) ;
197  vorbis_dsp_clear (&vdata->vd) ;
198  vorbis_comment_clear (&vdata->vc) ;
199  vorbis_info_clear (&vdata->vi) ;
200 
201  /*
202  ** Extract the initial header from the first page and verify that the
203  ** Ogg bitstream is in fact Vorbis data.
204  **
205  ** I handle the initial header first instead of just having the code
206  ** read all three Vorbis headers at once because reading the initial
207  ** header is an easy way to identify a Vorbis bitstream and it's
208  ** useful to see that functionality seperated out.
209  */
210  vorbis_info_init (&vdata->vi) ;
211  vorbis_comment_init (&vdata->vc) ;
212 
213  if (ogg_stream_pagein (&odata->os, &odata->og) < 0)
214  { /* Error ; stream version mismatch perhaps. */
215  psf_log_printf (psf, "Error reading first page of Ogg bitstream data\n") ;
216  return SFE_MALFORMED_FILE ;
217  } ;
218 
219  if (ogg_stream_packetout (&odata->os, &odata->op) != 1)
220  { /* No page? must not be vorbis. */
221  psf_log_printf (psf, "Error reading initial header packet.\n") ;
222  return SFE_MALFORMED_FILE ;
223  } ;
224 
225  if (vorbis_synthesis_headerin (&vdata->vi, &vdata->vc, &odata->op) < 0)
226  { /* Error case ; not a vorbis header. */
227  psf_log_printf (psf, "This Ogg bitstream does not contain Vorbis audio data.\n") ;
228  return SFE_MALFORMED_FILE ;
229  } ;
230 
231  /*
232  ** Common Ogg metadata fields?
233  ** TITLE, VERSION, ALBUM, TRACKNUMBER, ARTIST, PERFORMER, COPYRIGHT, LICENSE,
234  ** ORGANIZATION, DESCRIPTION, GENRE, DATE, LOCATION, CONTACT, ISRC,
235  */
236 
237  if (log_data)
238  { int k ;
239 
240  for (k = 0 ; k < ARRAY_LEN (vorbis_metatypes) ; k++)
241  { char *dd ;
242 
243  dd = vorbis_comment_query (&vdata->vc, vorbis_metatypes [k].name, 0) ;
244  if (dd == NULL)
245  continue ;
246  psf_store_string (psf, vorbis_metatypes [k].id, dd) ;
247  } ;
248  } ;
249 
250  /*
251  ** At this point, we're sure we're Vorbis. We've set up the logical (Ogg)
252  ** bitstream decoder. Get the comment and codebook headers and set up the
253  ** Vorbis decoder.
254  **
255  ** The next two packets in order are the comment and codebook headers.
256  ** They're likely large and may span multiple pages. Thus we reead
257  ** and submit data until we get our two pacakets, watching that no
258  ** pages are missing. If a page is missing, error out ; losing a
259  ** header page is the only place where missing data is fatal.
260  */
261 
262  i = 0 ; /* Count of number of packets read */
263  while (i < 2)
264  { int result = ogg_sync_pageout (&odata->oy, &odata->og) ;
265  if (result == 0)
266  { /* Need more data */
267  buffer = ogg_sync_buffer (&odata->oy, 4096) ;
268  bytes = psf_fread (buffer, 1, 4096, psf) ;
269 
270  if (bytes == 0)
271  { psf_log_printf (psf, "End of file before finding all Vorbis headers!\n") ;
272  return SFE_MALFORMED_FILE ;
273  } ;
274  nn = ogg_sync_wrote (&odata->oy, bytes) ;
275  }
276  else if (result == 1)
277  { /*
278  ** Don't complain about missing or corrupt data yet. We'll
279  ** catch it at the packet output phase.
280  **
281  ** We can ignore any errors here as they'll also become apparent
282  ** at packetout.
283  */
284  nn = ogg_stream_pagein (&odata->os, &odata->og) ;
285  while (i < 2)
286  { result = ogg_stream_packetout (&odata->os, &odata->op) ;
287  if (result == 0)
288  break ;
289  if (result < 0)
290  { /* Uh oh ; data at some point was corrupted or missing!
291  ** We can't tolerate that in a header. Die. */
292  psf_log_printf (psf, "Corrupt secondary header. Exiting.\n") ;
293  return SFE_MALFORMED_FILE ;
294  } ;
295 
296  vorbis_synthesis_headerin (&vdata->vi, &vdata->vc, &odata->op) ;
297  i++ ;
298  } ;
299  } ;
300  } ;
301 
302  if (log_data)
303  { int printed_metadata_msg = 0 ;
304  int k ;
305 
306  psf_log_printf (psf, "\nBitstream is %d channel, %D Hz\n", vdata->vi.channels, vdata->vi.rate) ;
307  psf_log_printf (psf, "Encoded by: %s\n", vdata->vc.vendor) ;
308 
309  /* Throw the comments plus a few lines about the bitstream we're decoding. */
310  for (k = 0 ; k < ARRAY_LEN (vorbis_metatypes) ; k++)
311  { char *dd ;
312 
313  dd = vorbis_comment_query (&vdata->vc, vorbis_metatypes [k].name, 0) ;
314  if (dd == NULL)
315  continue ;
316 
317  if (printed_metadata_msg == 0)
318  { psf_log_printf (psf, "Metadata :\n") ;
319  printed_metadata_msg = 1 ;
320  } ;
321 
322  psf_store_string (psf, vorbis_metatypes [k].id, dd) ;
323  psf_log_printf (psf, " %-10s : %s\n", vorbis_metatypes [k].name, dd) ;
324  } ;
325 
326  psf_log_printf (psf, "End\n") ;
327  } ;
328 
329  psf->sf.samplerate = vdata->vi.rate ;
330  psf->sf.channels = vdata->vi.channels ;
331  psf->sf.format = SF_FORMAT_OGG | SF_FORMAT_VORBIS ;
332 
333  /* OK, got and parsed all three headers. Initialize the Vorbis
334  ** packet->PCM decoder.
335  ** Central decode state. */
336  vorbis_synthesis_init (&vdata->vd, &vdata->vi) ;
337 
338  /* Local state for most of the decode so multiple block decodes can
339  ** proceed in parallel. We could init multiple vorbis_block structures
340  ** for vd here. */
341  vorbis_block_init (&vdata->vd, &vdata->vb) ;
342 
343  vdata->loc = 0 ;
344 
345  return 0 ;
346 } /* ogg_read_header */

349 ogg_write_header (SF_PRIVATE *psf, int UNUSED (calc_length))
350 {
351  OGG_PRIVATE *odata = (OGG_PRIVATE *) psf->container_data ;
352  VORBIS_PRIVATE *vdata = (VORBIS_PRIVATE *) psf->codec_data ;
353  int k, ret ;
354 
355  vorbis_info_init (&vdata->vi) ;
356 
357  /* The style of encoding should be selectable here, VBR quality mode. */
358  ret = vorbis_encode_init_vbr (&vdata->vi, psf->sf.channels, psf->sf.samplerate, vdata->quality) ;
359 
360 #if 0
361  ret = vorbis_encode_init (&vdata->vi, psf->sf.channels, psf->sf.samplerate, -1, 128000, -1) ; /* average bitrate mode */
362  ret = ( vorbis_encode_setup_managed (&vdata->vi, psf->sf.channels,
363  psf->sf.samplerate, -1, 128000, -1) ||
364  vorbis_encode_ctl (&vdata->vi, OV_ECTL_RATEMANAGE_AVG, NULL) ||
365  vorbis_encode_setup_init (&vdata->vi)) ;
366 #endif
367  if (ret)
368  return SFE_BAD_OPEN_FORMAT ;
369 
370  vdata->loc = 0 ;
371 
372  /* add a comment */
373  vorbis_comment_init (&vdata->vc) ;
374 
375  vorbis_comment_add_tag (&vdata->vc, "ENCODER", "libsndfile") ;
376  for (k = 0 ; k < SF_MAX_STRINGS ; k++)
377  { const char * name ;
378 
379  if (psf->strings [k].type == 0)
380  break ;
381 
382  switch (psf->strings [k].type)
383  { case SF_STR_TITLE : name = "TITLE" ; break ;
384  case SF_STR_COPYRIGHT : name = "COPYRIGHT" ; break ;
385  case SF_STR_SOFTWARE : name = "SOFTWARE" ; break ;
386  case SF_STR_ARTIST : name = "ARTIST" ; break ;
387  case SF_STR_COMMENT : name = "COMMENT" ; break ;
388  case SF_STR_DATE : name = "DATE" ; break ;
389  case SF_STR_ALBUM : name = "ALBUM" ; break ;
390  case SF_STR_LICENSE : name = "LICENSE" ; break ;
391  case SF_STR_TRACKNUMBER : name = "TRACKNUMBER" ; break ;
392  case SF_STR_GENRE : name = "GENRE" ; break ;
393  default : continue ;
394  } ;
395 
396  vorbis_comment_add_tag (&vdata->vc, name, psf->strings [k].str) ;
397  } ;
398 
399  /* set up the analysis state and auxiliary encoding storage */
400  vorbis_analysis_init (&vdata->vd, &vdata->vi) ;
401  vorbis_block_init (&vdata->vd, &vdata->vb) ;
402 
403  /*
404  ** Set up our packet->stream encoder.
405  ** Pick a random serial number ; that way we can more likely build
406  ** chained streams just by concatenation.
407  */
408 
409  ogg_stream_init (&odata->os, psf_rand_int32 ()) ;
410 
411  /* Vorbis streams begin with three headers ; the initial header (with
412  most of the codec setup parameters) which is mandated by the Ogg
413  bitstream spec. The second header holds any comment fields. The
414  third header holds the bitstream codebook. We merely need to
415  make the headers, then pass them to libvorbis one at a time ;
416  libvorbis handles the additional Ogg bitstream constraints */
417 
418  { ogg_packet header ;
419  ogg_packet header_comm ;
420  ogg_packet header_code ;
421  int result ;
422 
423  vorbis_analysis_headerout (&vdata->vd, &vdata->vc, &header, &header_comm, &header_code) ;
424  ogg_stream_packetin (&odata->os, &header) ; /* automatically placed in its own page */
425  ogg_stream_packetin (&odata->os, &header_comm) ;
426  ogg_stream_packetin (&odata->os, &header_code) ;
427 
428  /* This ensures the actual
429  * audio data will start on a new page, as per spec
430  */
431  while ((result = ogg_stream_flush (&odata->os, &odata->og)) != 0)
432  { psf_fwrite (odata->og.header, 1, odata->og.header_len, psf) ;
433  psf_fwrite (odata->og.body, 1, odata->og.body_len, psf) ;
434  } ;
435  }
436 
437  return 0 ;
438 } /* ogg_write_header */

440 static int
441 ogg_close (SF_PRIVATE *psf)
442 {
443  OGG_PRIVATE *odata = (OGG_PRIVATE *) psf->container_data ;
444  VORBIS_PRIVATE *vdata = (VORBIS_PRIVATE *) psf->codec_data ;
445 
446  if (odata == NULL || vdata == NULL)
447  return 0 ;
448 
449  /* Clean up this logical bitstream ; before exit we shuld see if we're
450  ** followed by another [chained]. */
451 
452  if (psf->file.mode == SFM_WRITE)
453  {
454  if (psf->write_current <= 0)
455  ogg_write_header (psf, 0) ;
456 
457  vorbis_analysis_wrote (&vdata->vd, 0) ;
458  while (vorbis_analysis_blockout (&vdata->vd, &vdata->vb) == 1)
459  {
460 
461  /* analysis, assume we want to use bitrate management */
462  vorbis_analysis (&vdata->vb, NULL) ;
463  vorbis_bitrate_addblock (&vdata->vb) ;
464 
465  while (vorbis_bitrate_flushpacket (&vdata->vd, &odata->op))
466  { /* weld the packet into the bitstream */
467  ogg_stream_packetin (&odata->os, &odata->op) ;
468 
469  /* write out pages (if any) */
470  while (!odata->eos)
471  { int result = ogg_stream_pageout (&odata->os, &odata->og) ;
472  if (result == 0) break ;
473  psf_fwrite (odata->og.header, 1, odata->og.header_len, psf) ;
474  psf_fwrite (odata->og.body, 1, odata->og.body_len, psf) ;
475 
476  /* this could be set above, but for illustrative purposes, I do
477  it here (to show that vorbis does know where the stream ends) */
478 
479  if (ogg_page_eos (&odata->og)) odata->eos = 1 ;
480  }
481  }
482  }
483  }
484 
485  /* ogg_page and ogg_packet structs always point to storage in
486  libvorbis. They are never freed or manipulated directly */
487 
488  vorbis_block_clear (&vdata->vb) ;
489  vorbis_dsp_clear (&vdata->vd) ;
490  vorbis_comment_clear (&vdata->vc) ;
491  vorbis_info_clear (&vdata->vi) ; /* must be called last */
492  /* should look here to reopen if chained */
493 
494  /* OK, clean up the framer */
495  ogg_sync_clear (&odata->oy) ;
496  ogg_stream_clear (&odata->os) ;
497 
498  return 0 ;
499 } /* ogg_close */

568 ogg_command (SF_PRIVATE *psf, int command, void * data, int datasize)

570

672 static sf_count_t
673 ogg_read_sample (SF_PRIVATE *psf, void *ptr, sf_count_t lens, convert_func *transfn)
674 {
675  VORBIS_PRIVATE *vdata = psf->codec_data ;
676  OGG_PRIVATE *odata = psf->container_data ;
677  int result, len, samples, i = 0 ;
678  float **pcm ;
679 
680  len = lens / psf->sf.channels ;
681 
682  while ((samples = vorbis_synthesis_pcmout (&vdata->vd, &pcm)) > 0)
683  { if (samples > len) samples = len ;
684  i += transfn (samples, ptr, i, psf->sf.channels, pcm) ;
685  len -= samples ;
686  /* tell libvorbis how many samples we actually consumed */
687  vorbis_synthesis_read (&vdata->vd, samples) ;
688  vdata->loc += samples ;
689  if (len == 0)
690  return i ; /* Is this necessary */
691  }
692  goto start0 ; /* Jump into the nasty nest */
693  while (len > 0 && !odata->eos)
694  {
695  while (len > 0 && !odata->eos)
696  { result = ogg_sync_pageout (&odata->oy, &odata->og) ;
697  if (result == 0) break ; /* need more data */
698  if (result < 0)
699  { /* missing or corrupt data at this page position */
700  psf_log_printf (psf, "Corrupt or missing data in bitstream ; continuing...\n") ;
701  }
702  else
703  { /* can safely ignore errors at this point */
704  ogg_stream_pagein (&odata->os, &odata->og) ;
705  start0:
706  while (1)
707  { result = ogg_stream_packetout (&odata->os, &odata->op) ;
708  if (result == 0)
709  break ; /* need more data */
710  if (result < 0)
711  { /* missing or corrupt data at this page position */
712  /* no reason to complain ; already complained above */
713  }
714  else
715  { /* we have a packet. Decode it */
716  if (vorbis_synthesis (&vdata->vb, &odata->op) == 0) /* test for success! */
717  vorbis_synthesis_blockin (&vdata->vd, &vdata->vb) ;
718  /*
719  ** The **pcm variable is a multichannel float vector. In stereo, for
720  ** example, pcm [0] is left, and pcm [1] is right. samples is
721  ** the size of each channel. Convert the float values
722  ** (-1.<=range<=1.) to whatever PCM format and write it out.
723  */
724 
725  while ((samples = vorbis_synthesis_pcmout (&vdata->vd, &pcm)) > 0)
726  { if (samples>len) samples = len ;
727  i += transfn (samples, ptr, i, psf->sf.channels, pcm) ;
728  len -= samples ;
729  /* tell libvorbis how many samples we actually consumed */
730  vorbis_synthesis_read (&vdata->vd, samples) ;
731  vdata->loc += samples ;
732  if (len == 0)
733  return i ; /* Is this necessary */
734  } ;
735  }
736  }
737  if (ogg_page_eos (&odata->og)) odata->eos = 1 ;
738  }
739  }
740  if (!odata->eos)
741  { char *buffer ;
742  int bytes ;
743  buffer = ogg_sync_buffer (&odata->oy, 4096) ;
744  bytes = psf_fread (buffer, 1, 4096, psf) ;
745  ogg_sync_wrote (&odata->oy, bytes) ;
746  if (bytes == 0) odata->eos = 1 ;
747  }
748  }

777 {
