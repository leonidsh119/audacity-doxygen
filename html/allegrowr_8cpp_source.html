15 // Note about precision: %g prints 6 significant digits. For 1ms precision,
16 // the maximum magnitude is 999.999, i.e. 1000s < 17minutes. For anything
17 // over 1000s, time in seconds will be printed with 10ms precision, which
18 // is not good. Therefore, times and durations are printed as %.4d, which
19 // gives 100us precision.
20 // The following define allows you to change this decision:

22 #define TIMPREC 4

51 Alg_event_ptr Alg_seq::write_track_name(ostream &file, int n,
52  Alg_events &events)
53 // write #track <n> <trackname-or-sequencename>
54 // if we write the name on the "#track" line, then we do *not* want
55 // to write again as an update: "-seqnames:"Jordu", so if we do
56 // find a name and write it, return a pointer to it so the track
57 // writer knows what update (if any) to skip
58 {
59  Alg_event_ptr e = NULL; // e is the result, default is NULL
60  file << "#track " << n;
61  const char *attr = symbol_table.insert_string(
62  n == 0 ? "seqnames" : "tracknames");
63  // search for name in events with timestamp of 0
64  for (int i = 0; i < events.length(); i++) {
65  Alg_event_ptr ue = events[i];
66  if (ue->time > 0) break;
67  if (ue->is_update()) {
68  Alg_update_ptr u = (Alg_update_ptr) ue;
69  if (u->parameter.attr == attr) {
70  file << " " << u->parameter.s;
71  e = ue; // return the update event we found
72  break;
73  }
74  }
75  }
76  file << endl; // end of line containing #track [<name>]
77  return e; // return parameter event with name if one was found
78 }
