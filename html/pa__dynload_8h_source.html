72 /* A little explanation of what's going on here.
73  *
74  * Only one source file should define PADL_DEFINE_POINTERS before including the header which
75  * defines the functions. This will cause the compiler to dump all of the function pointers
76  * to a single object file and prevent duplicate symbol definitions during link.
77  *
78  * The PADL_FUNC_WITH_RETURN and PADL_FUNC_NO_RETURN macros do two things each:
79  * 1) Define or reference the variable that contains the actual function pointer
80  * 2) Define an inline function to pass control to the real function
81  *
82  * Since the macros redefine the real functions of the same name, the compiler will make
83  * sure that the definitions are the same. If not, it will complain. For this to occur,
84  * the functions MUST be defined in an extern "C" block otherwise the compiler just thinks the
85  * functions are being overloaded.
86  *
87  * The compiler should optimize away the inline function since it just passes control to the real
88  * function and we should wind up with about the same function call we had before, only now it is
89  * safer due to the validation.
90  *
91  * The PADL_FUNC_WITH_RETURN takes 4 arguments:
92  * 1) The return type <---|
93  * 2) The function name | Taken from the real funciton prototype
94  * 3) The function arguments <---|
95  * 4) The argument list to pass to the real function
96  *
97  * The PADL_FUNC_NO_RETURN takes 3 arguments:
98  * 1) The function name <---| Taken from the FFmpeg funciton prototype
99  * 2) The function arguments <---|
100  * 3) The argument list to pass to the real function
101  *
102  * The PADL_FINDSYMBOL macro is responsible for retrieving the address of the real function
103  * and storing that address in the function pointer variable.
104  */
105 #if defined(PADL_DEFINE_POINTERS)
106  #define FFX
107 #else
108  #define FFX extern
109 #endif
110 
111 #define PADL_FUNC_WITH_RETURN(r, n, a, p) \
112  FFX r (*paDynFunc_ ## n ## _fp) a; \
113  PADL_INLINE r n a \
114  { \
115  return paDynFunc_ ## n ## _fp p; \

118 #define PADL_FUNC_NO_RETURN(n, a, p) \
119  FFX void (*paDynFunc_ ## n ## _fp) a; \
