145 const char *find_in_xlisp_path(const char *fname)
146 {
147  const char *paths = return_xlisp_path();
148  if (!paths)
149  return NULL;
150 
151  while (paths && *paths) {
152  FILE *fp;
153  const char *start;
154  int len;
155 
156  /* skip over separator */
157  while (*paths == os_sepchar || *paths == ';') paths++;
158 
159  /* find next directory */
160  start = paths;
161  while (*paths && (*paths != os_sepchar && *paths != ';'))
162  paths++;
163 
164  if (g_xlptemp) {
165  free(g_xlptemp);
166  g_xlptemp = NULL;
167  }
168 
169  len = paths - start;
170  g_xlptemp = malloc(len + strlen(fname) + 10);
171  memcpy(g_xlptemp, start, len);
172 
173  if (len == 0)
174  continue;
175 
176  /* add "/" if needed */
177  if (g_xlptemp[len-1] != os_pathchar)
178  g_xlptemp[len++] = os_pathchar;
179 
180  /* append the file name */
181  memcpy(&g_xlptemp[len], fname, strlen(fname));
182  len += strlen(fname);
183  g_xlptemp[len] = 0;
184 
185 
186  /* printf("Attempting to open %s, start is %s\n", g_xlptemp, start); */
187  fp = osaopen(g_xlptemp, "r");
188  if (!fp) {
189  /* try appending the .lsp extension */
190  if (needsextension(g_xlptemp)) {
191  strcat(g_xlptemp, ".lsp");
192  fp = osaopen(g_xlptemp, "r");
193  if (!fp) {
194  g_xlptemp[strlen(g_xlptemp) - 4] = 0; /* remove .lsp */
195  }
196  }
197  }
198  if (fp) {
199  fclose(fp);
200 
201  #ifdef MACINTOSH
202  /* We found the file ok, call setup_preferences to create
203  * XLisp Preferences file (this only happens if previous
204  * attempt to find the file failed
205  */
206  setup_preferences(g_xlptemp);
207  #endif
208 
209  return g_xlptemp;
210  }
211  }
212 
213  /* It wasn't found */
214  return NULL;
215 }
