96 static int open_mixer(PxDev *dev, int cmd)
97 {
98  char name[sizeof(MIXER_NAME_BASE) + 1];
99  int mask;
100  int i;
101  int id = 0;
102 
103  for (i = strlen(dev->name) - 1; i >= 0; i++) {
104  if (!isdigit(dev->name[i])) {
105  break;
106  }
107  }
108 
109  id = atoi(&dev->name[i + 1]);
110  if (id < 0 || id >= MIXER_COUNT_MAX) {
111  return -1;
112  }
113 
114  strcpy(name, MIXER_NAME_BASE);
115  if (id == 0)
116  name[strlen(MIXER_NAME_BASE)] = 0;
117  else
118  name[strlen(MIXER_NAME_BASE)] = '0' + (id - 1);
119 
120  do {
121  dev->fd = open(name, O_RDWR);
122  if (dev->fd < 0) {
123  break;
124  }
125 
126  if (ioctl(dev->fd, cmd, &mask) == -1) {
127  break;
128  }
129 
130  dev->num = 0;
131 
132  for (i = 0; i < SOUND_MIXER_NRDEVICES; i++) {
133  if (mask & (1 << i)) {
134  dev->chans[dev->num++] = i;
135  }
136  }
137  return TRUE;
138 
139  } while (FALSE);
140 
141  if (dev->fd >= 0) {
142  close(dev->fd);
143  dev->fd = -1;
144  }
145 
146  return FALSE;
147 }

188 static PxVolume get_volume(int fd, int channel)
189 {
190  int vol;
191  int stereo;
192 
193  if (ioctl(fd, SOUND_MIXER_READ_STEREODEVS, &stereo) == 0)
194  stereo = ((stereo & (1 << channel)) != 0);
195  else
196  stereo = 0;
197 
198  if (ioctl(fd, MIXER_READ(channel), &vol) == -1)
199  return 0.0;
200 
201  if (stereo)
202  return ((vol & 0xFF)/200.0) + (((vol>>8) & 0xFF)/200.0);
203  else
204  return (vol & 0xFF)/100.0;
205 }

459 static int get_current_input_source(px_mixer *Px)
460 {
461  PxInfo *info = (PxInfo *)Px->info;
462  int mask;
463  int i;
464 
465  /* Note that there may be more than one in OSS; we pick

467  if (info->capture.fd >= 0) {
468  if (ioctl(info->capture.fd, SOUND_MIXER_READ_RECSRC, &mask) == -1) {
469  return -1; /* none / error */
470  }
471 
472  for (i = 0; i < info->capture.num; i++) {
473  if (mask & (1 << info->capture.chans[i])) {
474  return i;

476  }
477  }
478 
479  return -1; /* none */
