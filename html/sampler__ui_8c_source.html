147 static void
148 port_event(LV2UI_Handle handle,
149  uint32_t port_index,
150  uint32_t buffer_size,
151  uint32_t format,
152  const void* buffer)
153 {
154  SamplerUI* ui = (SamplerUI*)handle;
155  if (format == ui->uris.atom_eventTransfer) {
156  const LV2_Atom* atom = (const LV2_Atom*)buffer;
157  if (atom->type == ui->uris.atom_Blank) {
158  const LV2_Atom_Object* obj = (const LV2_Atom_Object*)atom;
159  const LV2_Atom* file_uri = read_set_file(&ui->uris, obj);
160  if (!file_uri) {
161  fprintf(stderr, "Unknown message sent to UI.\n");
162  return;
163  }
164 
165  const char* uri = (const char*)LV2_ATOM_BODY_CONST(file_uri);
166  gtk_label_set_text(GTK_LABEL(ui->label), uri);
167  } else {
168  fprintf(stderr, "Unknown message type.\n");
169  }
170  } else {
171  fprintf(stderr, "Unknown format.\n");
172  }
173 }
