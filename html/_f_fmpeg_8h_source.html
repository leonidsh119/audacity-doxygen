436  // A little explanation of what's going on here.
437  //
438  // The FFmpeg function pointers used to be defined in the FFmpegLibs class and all calls would
439  // be done via the global class pointer FFmpegLibsInst. This was fine as long as the prototypes
440  // defined in the class matched the actual functions in the FFmpeg libraries. There was no
441  // compile time validation to prevent invalid function calls. So, what follows is one way of
442  // getting the extra validation.
443  //
444  // Only one source file should define DEFINE_FFMPEG_POINTERS before including ffmpeg.h. This
445  // will cause the compiler to dump all of the function pointers to a single object file and
446  // prevent duplicate symbol definitions during link. This is currently done in ffmpeg.cpp.
447  //
448  // The FFMPEG_FUNCTION_WITH_RETURN and FFMPEG_FUNCTION_NO_RETURN macros do two things each:
449  // 1) Define or reference the variable that contains the actual function pointer
450  // 2) Define an inline function to pass control to the real function
451  //
452  // Since the macros redefine the real ffmpeg functions of the same name, the compiler will
453  // make sure that the definitions are the same. If not, it will complain. For this to occur,
454  // the functions MUST be defined in an extern "C" block otherwise the compiler just thinks the
455  // functions are being overloaded.
456  //
457  // The compiler should optimize away the inline function since it just passes control to the real
458  // function and we should wind up with about the same function call we had before, only now it is
459  // safer due to the validation.
460  //
461  // The FFMPEG_FUNCTION_WITH_RETURN takes 4 arguments:
462  // 1) The return type <---|
463  // 2) The function name | Taken from the FFmpeg funciton prototype
464  // 3) The function arguments <---|
465  // 4) The argument list to pass to the real function
466  //
467  // The FFMPEG_FUNCTION_NO_RETURN takes 3 arguments:
468  // 1) The function name <---| Taken from the FFmpeg funciton prototype
469  // 2) The function arguments <---|
470  // 3) The argument list to pass to the real function
471  //
472  // The FFMPEG_INITDYN macro is responsible for retrieving the address of the real function
473  // and storing that address in the function pointer variable. It will emit an error to the
474  // currently define log destination and return to the calling function.
475  //
476 #if defined(DEFINE_FFMPEG_POINTERS)
477  #define FFX
478 #else
479  #define FFX extern
480 #endif
481 
482 #define FFMPEG_FUNCTION_WITH_RETURN(r, n, a, p) \
483  extern "C" \
484  { \
485  FFX r (*n ## _fp) a; \
486  inline r n a \
487  { \
488  return n ## _fp p; \
489  } \
490  }
491 
492 #define FFMPEG_FUNCTION_NO_RETURN(n, a, p) \
493  extern "C" \
494  { \
495  FFX void (*n ## _fp) a; \
496  inline void n a \
497  { \
498  n ## _fp p; \
499  } \
500  }
501 
502 #define FFMPEG_INITDYN(w, f) \
503  { \
504  wxLogNull off; \
505  *(void**)&f ## _fp = (void*)w->GetSymbol(wxT(#f)); \
506  } \
507  if (f ## _fp == NULL) \
508  { \
509  wxLogError(wxT("Failed to load symbol ") wxT(#f)); \
510  return false; \
511  }
512 
513 #define FFMPEG_INITALT(w, f, x, a) \
514  { \
515  wxLogNull off; \
516  *(void**)&f ## _fp = (void*)w->GetSymbol(wxT(#f)); \
517  } \
518  if (f ## _fp == NULL) \
519  { \
520  { \
521  wxLogNull off; \
522  *(void**)&f ## _fp = (void*)x->GetSymbol(wxT(#a)); \
523  } \
524  if (f ## _fp == NULL) \
525  { \
526  wxLogError(wxT("Failed to load symbol ") wxT(#f)); \
527  return false; \
528  } \
529  }
