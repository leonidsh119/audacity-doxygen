40 bool OpenProjectCommand::Apply(CommandExecutionContext context)

42  wxString fileName = GetString(wxT("Filename"));
43  bool addToHistory = GetBool(wxT("AddToHistory"));
44  wxString oldFileName = context.GetProject()->GetFileName();
45  if(fileName == wxEmptyString)
