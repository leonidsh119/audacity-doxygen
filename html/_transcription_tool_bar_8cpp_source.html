386 void TranscriptionToolBar::GetSamples(WaveTrack *t, sampleCount *s0, sampleCount *slen)
387 {
388  // GetSamples attempts to translate the start and end selection markers into sample indices
389  // These selection numbers are doubles.
390 
391  AudacityProject *p = GetActiveProject();
392  if (!p) {
393  return;
394  }
395 
396  //First, get the current selection. It is part of the mViewInfo, which is
397  //part of the project
398 
399  double start = p->GetSel0();
400  double end = p->GetSel1();
401 
402  sampleCount ss0 = sampleCount( (start - t->GetOffset()) * t->GetRate() );
403  sampleCount ss1 = sampleCount( (end - t->GetOffset()) * t->GetRate() );
404 
405  if (start < t->GetOffset()) {
406  ss0 = 0;
407  }
408 
409 #if 0

413  }
414 #endif
415 
416  if (ss1 < ss0) {
417  ss1 = ss0;
418  }
419 
420  *s0 = ss0;
421  *slen = ss1 - ss0;

425 void TranscriptionToolBar::PlayAtSpeed(bool looped, bool cutPreview)
426 {
427  // Can't do anything without an active project
428  AudacityProject * p = GetActiveProject();
429  if (!p) {
430  return;
431  }
432 
433  // Create a TimeTrack if we haven't done so already
434  if (!mTimeTrack) {
435  mTimeTrack = p->GetTrackFactory()->NewTimeTrack();
436  if (!mTimeTrack) {
437  return;
438  }
439  }
440 
441  // Pop up the button
442  SetButton(false, mButtons[TTB_PlaySpeed]);
443 
444  // If IO is busy, abort immediately
445  if (gAudioIO->IsBusy()) {
446  p->GetControlToolBar()->StopPlaying();
447  }
448 
449  // Set the speed range
450  //mTimeTrack->SetRangeUpper((double)mPlaySpeed / 100.0);
451  //mTimeTrack->SetRangeLower((double)mPlaySpeed / 100.0);
452  mTimeTrack->GetEnvelope()->Flatten((double)mPlaySpeed / 100.0);
453 
454  // Get the current play region
455  double playRegionStart, playRegionEnd;
456  p->GetPlayRegion(&playRegionStart, &playRegionEnd);
457 
458  // Start playing
459  if (playRegionStart >= 0) {
460 // playRegionEnd = playRegionStart + (playRegionEnd-playRegionStart)* 100.0/mPlaySpeed;
461 #ifdef EXPERIMENTAL_MIDI_OUT
462  gAudioIO->SetMidiPlaySpeed(mPlaySpeed);
463 #endif
464  AudioIOStartStreamOptions options(p->GetDefaultPlayOptions());
465  options.playLooped = looped;
466  options.timeTrack = mTimeTrack;
467  p->GetControlToolBar()->PlayPlayRegion
468  (SelectedRegion(playRegionStart, playRegionEnd),
469  options,
470  cutPreview);
471  }
472 }

768 void TranscriptionToolBar::OnAutomateSelection(wxCommandEvent & WXUNUSED(event))
769 {
770 
771 
772  //If IO is busy, abort immediately
773  if (gAudioIO->IsBusy())
774  {
775  SetButton(false,mButtons[TTB_EndOff]);
776  return;
777  }
778 
779  wxBusyCursor busy;
780 
781  mVk->AdjustThreshold(GetSensitivity());
782  AudacityProject *p = GetActiveProject();
783  TrackList *tl = p->GetTracks();
784  TrackListIterator iter(tl);
785 
786  Track *t = iter.First(); //Make a track
787  if(t)
788  {
789  sampleCount start,len;
790  GetSamples((WaveTrack*)t, &start,&len);
791 
792  //Adjust length to end if selection is null
793  if(len == 0)
794  {
795  len = start;
796  start = 0;
797  }
798  int lastlen = 0;
799  sampleCount newStart, newEnd;
800  double newStartPos, newEndPos;
801 
802 
803  //This is the minumum word size in samples (.05 is 50 ms)
804  int minWordSize = (int)(((WaveTrack*)t)->GetRate() * .05);
805 
806  //Continue until we have processed the entire
807  //region, or we are making no progress.
808  while(len > 0 && lastlen != len)
809  {
810 
811  lastlen = len;
812 
813  newStart = mVk->OnForward(*(WaveTrack*)t,start,len);
814 
815  //JKC: If no start found then don't add any labels.
816  if( newStart==start)
817  break;
818 
819  //Adjust len by the NEW start position
820  len -= (newStart - start);
821 
822  //Adjust len by the minimum word size
823  len -= minWordSize;
824 
825 
826 
827  //OK, now we have found a NEW starting point. A 'word' should be at least
828  //50 ms long, so jump ahead minWordSize
829 
830  newEnd = mVk->OffForward(*(WaveTrack*)t,newStart+minWordSize, len);
831 
832  //If newEnd didn't move, we should give up, because
833  // there isn't another end before the end of the selection.
834  if(newEnd == (newStart + minWordSize))
835  break;
836 
837 
838  //Adjust len by the NEW word end
839  len -= (newEnd - newStart);
840 
841  //Calculate the start and end of the words, in seconds
842  newStartPos = newStart / ((WaveTrack*)t)->GetRate();
843  newEndPos = newEnd / ((WaveTrack*)t)->GetRate();
844 
845 
846  //Increment
847  start = newEnd;
848 
849  p->DoAddLabel(SelectedRegion(newStartPos, newEndPos));
850  p->RedrawProject();
851  }
852  SetButton(false, mButtons[TTB_AutomateSelection]);
853  }
