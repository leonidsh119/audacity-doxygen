72 // Usual C-version of the filter routine for stereo sound
73 uint FIRFilter::evaluateFilterStereo(SAMPLETYPE *dest, const SAMPLETYPE *src, uint numSamples) const
74 {
75  uint i, j, end;
76  LONG_SAMPLETYPE suml, sumr;
77 #ifdef SOUNDTOUCH_FLOAT_SAMPLES
78  // when using floating point samples, use a scaler instead of a divider
79  // because division is much slower operation than multiplying.
80  double dScaler = 1.0 / (double)resultDivider;
81 #endif
82 
83  assert(length != 0);
84  assert(src != NULL);
85  assert(dest != NULL);
86  assert(filterCoeffs != NULL);
87 
88  end = 2 * (numSamples - length);
89 
90  for (j = 0; j < end; j += 2)
91  {
92  const SAMPLETYPE *ptr;
93 
94  suml = sumr = 0;
95  ptr = src + j;
96 
97  for (i = 0; i < length; i += 4)
98  {
99  // loop is unrolled by factor of 4 here for efficiency
100  suml += ptr[2 * i + 0] * filterCoeffs[i + 0] +
101  ptr[2 * i + 2] * filterCoeffs[i + 1] +
102  ptr[2 * i + 4] * filterCoeffs[i + 2] +
103  ptr[2 * i + 6] * filterCoeffs[i + 3];
104  sumr += ptr[2 * i + 1] * filterCoeffs[i + 0] +
105  ptr[2 * i + 3] * filterCoeffs[i + 1] +
106  ptr[2 * i + 5] * filterCoeffs[i + 2] +
107  ptr[2 * i + 7] * filterCoeffs[i + 3];
108  }
109 
110 #ifdef SOUNDTOUCH_INTEGER_SAMPLES
111  suml >>= resultDivFactor;
112  sumr >>= resultDivFactor;
113  // saturate to 16 bit integer limits
114  suml = (suml < -32768) ? -32768 : (suml > 32767) ? 32767 : suml;
115  // saturate to 16 bit integer limits
116  sumr = (sumr < -32768) ? -32768 : (sumr > 32767) ? 32767 : sumr;
117 #else
118  suml *= dScaler;
119  sumr *= dScaler;
120 #endif // SOUNDTOUCH_INTEGER_SAMPLES
121  dest[j] = (SAMPLETYPE)suml;
122  dest[j + 1] = (SAMPLETYPE)sumr;
123  }
124  return numSamples - length;
125 }

130 // Usual C-version of the filter routine for mono sound
131 uint FIRFilter::evaluateFilterMono(SAMPLETYPE *dest, const SAMPLETYPE *src, uint numSamples) const
132 {
133  uint i, j, end;
134  LONG_SAMPLETYPE sum;
135 #ifdef SOUNDTOUCH_FLOAT_SAMPLES
136  // when using floating point samples, use a scaler instead of a divider
137  // because division is much slower operation than multiplying.
138  double dScaler = 1.0 / (double)resultDivider;
139 #endif
140 
141 
142  assert(length != 0);
143 
144  end = numSamples - length;
145  for (j = 0; j < end; j ++)
146  {
147  sum = 0;
148  for (i = 0; i < length; i += 4)
149  {
150  // loop is unrolled by factor of 4 here for efficiency
151  sum += src[i + 0] * filterCoeffs[i + 0] +
152  src[i + 1] * filterCoeffs[i + 1] +
153  src[i + 2] * filterCoeffs[i + 2] +
154  src[i + 3] * filterCoeffs[i + 3];
155  }
156 #ifdef SOUNDTOUCH_INTEGER_SAMPLES
157  sum >>= resultDivFactor;
158  // saturate to 16 bit integer limits
159  sum = (sum < -32768) ? -32768 : (sum > 32767) ? 32767 : sum;
160 #else
161  sum *= dScaler;
162 #endif // SOUNDTOUCH_INTEGER_SAMPLES
163  dest[j] = (SAMPLETYPE)sum;
164  src ++;
165  }
166  return end;
167 }

170 // Set filter coeffiecients and length.

172 // Throws an exception if filter length isn't divisible by 8
173 void FIRFilter::setCoefficients(const SAMPLETYPE *coeffs, uint newLength, uint uResultDivFactor)
174 {
175  assert(newLength > 0);
176  if (newLength % 8) ST_THROW_RT_ERROR("FIR filter length not divisible by 8");
177 
178  lengthDiv8 = newLength / 8;
179  length = lengthDiv8 * 8;
180  assert(length == newLength);
181 
182  resultDivFactor = uResultDivFactor;
183  resultDivider = (SAMPLETYPE)::pow(2.0, (int)resultDivFactor);
184 
185  delete[] filterCoeffs;
186  filterCoeffs = new SAMPLETYPE[length];
187  memcpy(filterCoeffs, coeffs, length * sizeof(SAMPLETYPE));
188 }

202 uint FIRFilter::evaluate(SAMPLETYPE *dest, const SAMPLETYPE *src, uint numSamples, uint numChannels) const
203 {
204  assert(numChannels == 1 || numChannels == 2);
205 
206  assert(length > 0);
207  assert(lengthDiv8 * 8 == length);
208  if (numSamples < length) return 0;
209  if (numChannels == 2)

211  return evaluateFilterStereo(dest, src, numSamples);
