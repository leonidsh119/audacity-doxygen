1070 int Sequence::FindBlock(sampleCount pos) const
1071 {
1072  wxASSERT(pos >= 0 && pos < mNumSamples);
1073 
1074  if (pos == 0)
1075  return 0;
1076 
1077  int numBlocks = mBlock.size();
1078 
1079  sampleCount lo = 0, loSamples = 0;
1080  sampleCount hi = numBlocks, hiSamples = mNumSamples;
1081  sampleCount guess;
1082 
1083  while (true) {
1084  //this is not a binary search, but a
1085  //dictionary search where we guess something smarter than the binary division
1086  //of the unsearched area, since samples are usually proportional to block file number.
1087  const double frac = double(pos - loSamples) / (hiSamples - loSamples);
1088  guess = std::min(hi - 1, lo + sampleCount(frac * (hi - lo)));
1089  const SeqBlock &block = mBlock[guess];
1090 
1091  wxASSERT(block.f->GetLength() > 0);
1092  wxASSERT(lo <= guess && guess < hi && lo < hi);
1093 
1094  if (pos < block.start) {
1095  wxASSERT(lo != guess);
1096  hi = guess;
1097  hiSamples = block.start;
1098  }
1099  else {
1100  const sampleCount nextStart = block.start + block.f->GetLength();
1101  if (pos < nextStart)
1102  break;
1103  else {
1104  wxASSERT(guess < hi - 1);
1105  lo = guess + 1;
1106  loSamples = nextStart;
1107  }
1108  }
1109  }
1110 
1111  const int rval = guess;
1112  wxASSERT(rval >= 0 && rval < numBlocks &&
1113  pos >= mBlock[rval].start &&
1114  pos < mBlock[rval].start + mBlock[rval].f->GetLength());
1115 
1116  return rval;

1303 bool Sequence::GetWaveDisplay(float *min, float *max, float *rms, int* bl,
1304  int len, const sampleCount *where)
1305 {
1306  const sampleCount s0 = std::max(sampleCount(0), where[0]);
1307  if (s0 >= mNumSamples)
1308  // None of the samples asked for are in range. Abandon.
1309  return false;
1310 
1311  // In case where[len - 1] == where[len], raise the limit by one,
1312  // so we load at least one pixel for column len - 1
1313  // ... unless the mNumSamples ceiling applies, and then there are other defenses
1314  const sampleCount s1 =
1315  std::min(mNumSamples, std::max(1 + where[len - 1], where[len]));
1316  float *temp = new float[mMaxSamples];
1317 
1318  int pixel = 0;
1319 
1320  sampleCount srcX = s0;
1321  sampleCount nextSrcX = 0;
1322  int lastRmsDenom = 0;
1323  int lastDivisor = 0;
1324  sampleCount whereNow = std::min(s1 - 1, where[0]);
1325  sampleCount whereNext = 0;
1326  // Loop over block files, opening and reading and closing each
1327  // not more than once
1328  unsigned nBlocks = mBlock.size();
1329  const unsigned int block0 = FindBlock(s0);
1330  for (unsigned int b = block0; b < nBlocks; ++b) {
1331  if (b > block0)
1332  srcX = nextSrcX;
1333  if (srcX >= s1)
1334  break;
1335 
1336  // Find the range of sample values for this block that
1337  // are in the display.
1338  SeqBlock &seqBlock = mBlock[b];
1339  const sampleCount start = seqBlock.start;
1340  nextSrcX = std::min(s1, start + seqBlock.f->GetLength());
1341 
1342  // The column for pixel p covers samples from
1343  // where[p] up to but excluding where[p + 1].
1344 
1345  // Find the range of pixels covered by the current block file
1346  // (Their starting samples covered by it, to be exact)
1347  int nextPixel;
1348  if (nextSrcX >= s1)
1349  // last pass
1350  nextPixel = len;
1351  else {
1352  nextPixel = pixel;
1353  // Taking min with s1 - 1, here and elsewhere, is another defense
1354  // to be sure the last pixel column gets at least one sample
1355  while (nextPixel < len &&
1356  (whereNext = std::min(s1 - 1, where[nextPixel])) < nextSrcX)
1357  ++nextPixel;
1358  }
1359  if (nextPixel == pixel)
1360  // The entire block's samples fall within one pixel column.
1361  // Either it's a rare odd block at the end, or else,
1362  // we must be really zoomed out!
1363  // Omit the entire block's contents from min/max/rms
1364  // calculation, which is not correct, but correctness might not
1365  // be worth the compute time if this happens every pixel
1366  // column. -- PRL
1367  continue;
1368  if (nextPixel == len)
1369  whereNext = s1;
1370 
1371  // Decide the summary level
1372  const double samplesPerPixel =
1373  double(whereNext - whereNow) / (nextPixel - pixel);
1374  const int divisor =
1375  (samplesPerPixel >= 65536) ? 65536
1376  : (samplesPerPixel >= 256) ? 256
