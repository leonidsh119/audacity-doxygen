1400  map[1] = ChannelNameFrontLeft;

1550 bool Effect::ProcessTrack(int count,
1551  ChannelNames map,
1552  WaveTrack *left,
1553  WaveTrack *right,
1554  sampleCount leftStart,
1555  sampleCount rightStart,
1556  sampleCount len)
1557 {
1558  bool rc = true;
1559 
1560  // Give the plugin a chance to initialize
1561  if (!ProcessInitialize(len, map))
1562  {
1563  return false;
1564  }
1565 
1566  // For each input block of samples, we pass it to the effect along with a
1567  // variable output location. This output location is simply a pointer into a
1568  // much larger buffer. This reduces the number of calls required to add the
1569  // samples to the output track.
1570  //
1571  // Upon return from the effect, the output samples are "moved to the left" by
1572  // the number of samples in the current latency setting, effectively removing any
1573  // delay introduced by the effect.
1574  //
1575  // At the same time the total number of delayed samples are gathered and when
1576  // there is no further input data to process, the loop continues to call the
1577  // effect with an empty input buffer until the effect has had a chance to
1578  // return all of the remaining delayed samples.
1579  sampleCount inLeftPos = leftStart;
1580  sampleCount inRightPos = rightStart;
1581  sampleCount outLeftPos = leftStart;
1582  sampleCount outRightPos = rightStart;
1583 
1584  sampleCount inputRemaining = len;
1585  sampleCount delayRemaining = 0;
1586  sampleCount curBlockSize = 0;
1587  sampleCount curDelay = 0;
1588 
1589  sampleCount inputBufferCnt = 0;
1590  sampleCount outputBufferCnt = 0;
1591  bool cleared = false;
1592 
1593  int chans = wxMin(mNumAudioOut, mNumChannels);
1594 
1595  WaveTrack *genLeft = NULL;
1596  WaveTrack *genRight = NULL;
1597  sampleCount genLength = 0;
1598  bool isGenerator = GetType() == EffectTypeGenerate;
1599  bool isProcessor = GetType() == EffectTypeProcess;
1600  if (isGenerator)
1601  {
1602  double genDur;
1603  if (mIsPreview) {
1604  gPrefs->Read(wxT("/AudioIO/EffectsPreviewLen"), &genDur, 6.0);
1605  genDur = wxMin(mDuration, CalcPreviewInputLength(genDur));
1606  }
1607  else {
1608  genDur = mDuration;
1609  }
1610 
1611  genLength = left->GetRate() * genDur;
1612  delayRemaining = genLength;
1613  cleared = true;
1614 
1615  // Create temporary tracks
1616  genLeft = mFactory->NewWaveTrack(left->GetSampleFormat(), left->GetRate());
1617  if (right)
1618  {
1619  genRight = mFactory->NewWaveTrack(right->GetSampleFormat(), right->GetRate());
1620  }
1621  }
1622 
1623  // Call the effect until we run out of input or delayed samples
1624  while (inputRemaining || delayRemaining)
1625  {
1626  // Still working on the input samples
1627  if (inputRemaining)
1628  {
1629  // Need to refill the input buffers
1630  if (inputBufferCnt == 0)
1631  {
1632  // Calculate the number of samples to get
1633  inputBufferCnt = mBufferSize;
1634  if (inputBufferCnt > inputRemaining)
1635  {
1636  inputBufferCnt = inputRemaining;
1637  }
1638 
1639  // Fill the input buffers
1640  left->Get((samplePtr) mInBuffer[0], floatSample, inLeftPos, inputBufferCnt);
1641  if (right)
1642  {
1643  right->Get((samplePtr) mInBuffer[1], floatSample, inRightPos, inputBufferCnt);
1644  }
1645 
1646  // Reset the input buffer positions
1647  for (int i = 0; i < mNumChannels; i++)
1648  {
1649  mInBufPos[i] = mInBuffer[i];
1650  }
1651  }
1652 
1653  // Calculate the number of samples to process
1654  curBlockSize = mBlockSize;
1655  if (curBlockSize > inputRemaining)
1656  {
1657  // We've reached the last block...set current block size to what's left
1658  curBlockSize = inputRemaining;
1659  inputRemaining = 0;
1660 
1661  // Clear the remainder of the buffers so that a full block can be passed
1662  // to the effect
1663  sampleCount cnt = mBlockSize - curBlockSize;
1664  for (int i = 0; i < mNumChannels; i++)
1665  {
1666  for (int j = 0 ; j < cnt; j++)
1667  {
1668  mInBufPos[i][j + curBlockSize] = 0.0;
1669  }
1670  }
1671 
1672  // Might be able to use up some of the delayed samples
1673  if (delayRemaining)
1674  {
1675  // Don't use more than needed
1676  if (delayRemaining < cnt)
1677  {
1678  cnt = delayRemaining;
1679  }
1680  delayRemaining -= cnt;
1681  curBlockSize += cnt;
1682  }
1683  }
1684  }
1685  // We've exhausted the input samples and are now working on the delay
1686  else if (delayRemaining)
1687  {
1688  // Calculate the number of samples to process
1689  curBlockSize = mBlockSize;
1690  if (curBlockSize > delayRemaining)
1691  {
1692  curBlockSize = delayRemaining;
1693  }
1694  delayRemaining -= curBlockSize;
1695 
1696  // From this point on, we only want to feed zeros to the plugin
1697  if (!cleared)
1698  {
1699  // Reset the input buffer positions
1700  for (int i = 0; i < mNumChannels; i++)
1701  {
1702  mInBufPos[i] = mInBuffer[i];
1703 
1704  // And clear
1705  for (int j = 0; j < mBlockSize; j++)
1706  {
1707  mInBuffer[i][j] = 0.0;
1708  }
1709  }
1710  cleared = true;
1711  }

2503  {
2504  clientIn[i] += cnt;
2505  }
2506 
2507  for (int i = 0 ; i < mNumAudioOut; i++)
2508  {
2509  clientOut[i] += cnt;
2510  }
2511  }

2514  processor++;
