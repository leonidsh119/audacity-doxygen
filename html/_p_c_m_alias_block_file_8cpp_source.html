74 int PCMAliasBlockFile::ReadData(samplePtr data, sampleFormat format,
75  sampleCount start, sampleCount len)
76 {
77  SF_INFO info;
78 
79  if(!mAliasedFileName.IsOk()){ // intentionally silenced
80  memset(data,0,SAMPLE_SIZE(format)*len);
81  return len;
82  }
83 
84  wxFile f; // will be closed when it goes out of scope
85  SNDFILE *sf = NULL;
86  {
87  Maybe<wxLogNull> silence{};
88  if (mSilentAliasLog)
89  silence.create();
90 
91  memset(&info, 0, sizeof(info));
92 
93  if (f.Exists(mAliasedFileName.GetFullPath())) { // Don't use Open if file does not exits
94  if (f.Open(mAliasedFileName.GetFullPath())) {
95  // Even though there is an sf_open() that takes a filename, use the one that
96  // takes a file descriptor since wxWidgets can open a file with a Unicode name and
97  // libsndfile can't (under Windows).
98  ODManager::LockLibSndFileMutex();
99  sf = sf_open_fd(f.fd(), SFM_READ, &info, FALSE);
100  ODManager::UnlockLibSndFileMutex();
101  }
102  }
103 
104  if (!sf){
105  memset(data, 0, SAMPLE_SIZE(format)*len);
106  silence.reset();
107  mSilentAliasLog = TRUE;
108 
109  // Set a marker to display an error message for the silence
110  if (!wxGetApp().ShouldShowMissingAliasedFileWarning())
111  wxGetApp().MarkAliasedFilesMissingWarning(this);
112  return len;
113  }
114  }
115  mSilentAliasLog=FALSE;
116 
117  ODManager::LockLibSndFileMutex();
118  sf_seek(sf, mAliasStart + start, SEEK_SET);
119  ODManager::UnlockLibSndFileMutex();
120  SampleBuffer buffer(len * info.channels, floatSample);
121 
122  int framesRead = 0;
123 
124  if (format == int16Sample &&
125  !sf_subtype_more_than_16_bits(info.format)) {
126  // Special case: if the file is in 16-bit (or less) format,
127  // and the calling method wants 16-bit data, go ahead and
128  // read 16-bit data directly. This is a pretty common
129  // case, as most audio files are 16-bit.
130  ODManager::LockLibSndFileMutex();
131  framesRead = sf_readf_short(sf, (short *)buffer.ptr(), len);
132  ODManager::UnlockLibSndFileMutex();
133  for (int i = 0; i < framesRead; i++)
134  ((short *)data)[i] =
135  ((short *)buffer.ptr())[(info.channels * i) + mAliasChannel];
136  }
137  else {
138  // Otherwise, let libsndfile handle the conversion and
139  // scaling, and pass us normalized data as floats. We can
140  // then convert to whatever format we want.
141  ODManager::LockLibSndFileMutex();
142  framesRead = sf_readf_float(sf, (float *)buffer.ptr(), len);
143  ODManager::UnlockLibSndFileMutex();
144  float *bufferPtr = &((float *)buffer.ptr())[mAliasChannel];
145  CopySamples((samplePtr)bufferPtr, floatSample,
146  (samplePtr)data, format,
147  framesRead, true, info.channels);
148  }
149 
150  ODManager::LockLibSndFileMutex();
151  sf_close(sf);
152  ODManager::UnlockLibSndFileMutex();
153  return framesRead;
154 }

189 BlockFile *PCMAliasBlockFile::BuildFromXML(DirManager &dm, const wxChar **attrs)
190 {
191  wxFileName summaryFileName;
192  wxFileName aliasFileName;
193  int aliasStart=0, aliasLen=0, aliasChannel=0;
194  float min = 0.0f, max = 0.0f, rms = 0.0f;
195  double dblValue;
196  long nValue;
197 
198  while(*attrs)
199  {
200  const wxChar *attr = *attrs++;
201  const wxChar *value = *attrs++;
202  if (!value)
203  break;
204 
205  const wxString strValue = value;
206  if (!wxStricmp(attr, wxT("summaryfile")) &&
207  // Can't use XMLValueChecker::IsGoodFileName here, but do part of its test.
208  XMLValueChecker::IsGoodFileString(strValue) &&
209  (strValue.Length() + 1 + dm.GetProjectDataDir().Length() <= PLATFORM_MAX_PATH))
210  {
211  if (!dm.AssignFile(summaryFileName, strValue, false))
212  // Make sure summaryFileName is back to uninitialized state so we can detect problem later.
213  summaryFileName.Clear();
214  }
215  else if (!wxStricmp(attr, wxT("aliasfile")))
216  {
217  if (XMLValueChecker::IsGoodPathName(strValue))
218  aliasFileName.Assign(strValue);
219  else if (XMLValueChecker::IsGoodFileName(strValue, dm.GetProjectDataDir()))
220  // Allow fallback of looking for the file name, located in the data directory.
221  aliasFileName.Assign(dm.GetProjectDataDir(), strValue);
222  else if (XMLValueChecker::IsGoodPathString(strValue))
223  // If the aliased file is missing, we failed XMLValueChecker::IsGoodPathName()
224  // and XMLValueChecker::IsGoodFileName, because both do existence tests,
225  // but we want to keep the reference to the missing file because it's a good path string.
226  aliasFileName.Assign(strValue);
227  }
228  else if (XMLValueChecker::IsGoodInt(strValue) && strValue.ToLong(&nValue))
229  { // integer parameters
230  if (!wxStricmp(attr, wxT("aliasstart")) && (nValue >= 0))
231  aliasStart = nValue;
232  else if (!wxStricmp(attr, wxT("aliaslen")) && (nValue >= 0))
233  aliasLen = nValue;
234  else if (!wxStricmp(attr, wxT("aliaschannel")) && XMLValueChecker::IsValidChannel(aliasChannel))
235  aliasChannel = nValue;
236  else if (!wxStricmp(attr, wxT("min")))
237  min = nValue;
238  else if (!wxStricmp(attr, wxT("max")))
239  max = nValue;
240  else if (!wxStricmp(attr, wxT("rms")) && (nValue >= 0))
241  rms = nValue;
242  }
243  // mchinen: the min/max can be (are?) doubles as well, so handle those cases.
244  // Vaughan: The code to which I added the XMLValueChecker checks
245  // used wxAtoi to convert the string to an int.
246  // So it's possible some prior project formats used ints (?), so am keeping
247  // those above, but yes, we need to handle floats.
248  else if (XMLValueChecker::IsGoodString(strValue) && Internat::CompatibleToDouble(strValue, &dblValue))
249  { // double parameters
250  if (!wxStricmp(attr, wxT("min")))
251  min = dblValue;
252  else if (!wxStricmp(attr, wxT("max")))
253  max = dblValue;
254  else if (!wxStricmp(attr, wxT("rms")) && (dblValue >= 0.0))
255  rms = dblValue;
256  }
257  }
258 
259  return new PCMAliasBlockFile(summaryFileName, aliasFileName,
260  aliasStart, aliasLen, aliasChannel,
261  min, max, rms);
262 }
