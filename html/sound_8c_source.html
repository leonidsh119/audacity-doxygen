94 /* compute-phase - given a phase in radians, a wavetable specified as
95  * the nominal pitch (in half steps), the table length, and the sample
96  * rate, compute the sample number corresponding to the phase. This
97  * routine makes it easy to initialize the table pointer at the beginning
98  * of various oscillator implementations in Nyquist. Note that the table
99  * may represent several periods, in which case phase 360 is not the same
100  * as 0. Also note that the phase increment is also computed and returned
101  * through incr_ptr.
102  */
103 double compute_phase(phase, key, n, srate, new_srate, freq, incr_ptr)
104  double phase; /* phase in degrees (depends on ANGLEBASE) */
105  double key; /* the semitone number of the table played at srate */
106  long n; /* number of samples */
107  double srate; /* the sample rate of the table */
108  double new_srate; /* sample rate of the result */
109  double freq; /* the desired frequency */
110  double *incr_ptr; /* the sample increment */
111 {
112  double period = 1.0 / step_to_hz(key);
113 
114  /* convert phase to sample units */
115  phase = srate * period * (phase / (double) ANGLEBASE);
116  /* phase is now in sample units; if phase is less than zero, then increase
117  it by some number of sLength's to make it positive:
118  */
119  if (phase < 0)
120  phase += (((int) ((-phase) / n)) + 1) * n;
121 
122  /* if phase is longer than the sample length, wrap it by subtracting the
123  integer part of the division by sLength:
124  */
125  if (phase > n)
126  phase -= ((int) (phase / n)) * n;
127 
128  /* Now figure the phase increment: to reproduce original pitch
129  required incr = srate / new_srate. To get the new frequency,
130  scale by freq / nominal_freq = freq * period:
131  */
132  *incr_ptr = (srate / new_srate) * freq * period;
133  return phase;
134 }

298 /* sound_prepend_zeros -- modify sound_type so that it starts at t0 */
299 /*
300  * assumes t0 is earlier than snd->t0, so the sound should return zeros
301  * until snd->t0 is reached, after which we revert to normal computation.
302  * When we return, the new snd->t0 will be t0, meaning that the first
303  * sample returned will be at time t0.
304  * NOTE: t0 may not be an exact multiple of samples earlier than snd->t0,
305  * but Nyquist allows any sound to be shifted by +/- 0.5 samples in
306  * order to achieve alignment. Since sound_prepend_zeros can be called
307  * many times on the same sound_type, there is a chance that rounding
308  * errors could accumulate. My first solution was to return with
309  * snd->t0 computed exactly and not reflecting any fractional sample
310  * shift of the signal, but this caused problems for the caller: a
311  * fractional sample shift at a low sample rate could correspond to
312  * many client samples,fooling the client into thinking that some
313  * initial samples should be discarded (or else requiring the client
314  * to be pretty smart). The solution used here is to return to the
315  * client with snd->t0 exactly equal to t0, but to save snd->true_t0
316  * equal to the time of the first sample with no sound shifting. This
317  * time is used for any future sound_prepend_zeros operations so that
318  * any accumulated rounding errors are due only to floating point
319  * precision and not to accumulated fractional sample shifts of snd.
320  */
321 void sound_prepend_zeros(sound_type snd, time_type t0)
322 {
323  long n;
324 
325  /* first, see if we're already prepending some zeros */
326  if (snd->get_next != SND_get_zeros) {
327 /* nyquist_printf("sound_prepend_zeros 1: snd->t0 %g t0 %g\n", snd->t0, t0); */
328 
329  /* if not, then initialize some fields that support prepending */
330  snd->prepend_cnt = 0;
331  snd->true_t0 = snd->t0;
332 
333  /* save old get_next and plug in special get_next function */
334  snd->after_prepend = snd->get_next;
335  snd->get_next = SND_get_zeros;
336  }
337 
338  n = (long) (((snd->true_t0 - t0) * snd->sr) + 0.5); /* how many samples to prepend */
339 
340  /* add to prepend_cnt so first sample will correspond to new t0 */
341  snd->prepend_cnt += n;
342  /* compute the true t0 which corresponds to the time of first sample */
343  snd->true_t0 -= (n / snd->sr);
344  /* make caller happy by claiming the sound now starts at exactly t0;
345  * this is always true within 0.5 samples as allowed by Nyquist. */
346  snd->t0 = t0;
347 /* nyquist_printf("sound_prepend_zeros: snd %p true_t0 %g sr %g n %d\n",
348  snd, snd->true_t0, snd->sr, n);*/
349 }

646 {

648  int cnt; /* how many samples to flush */
649  sample_block_type sampblock = NULL;
650  long blocklen;

652 
653  /* changed true_t0 to just t0 based on comment that true_t0 is only

663  /* now flush cnt samples */
664  while (cnt >= 0) {
665  sampblock = sound_get_next(s, &blocklen);
666  cnt -= blocklen;
667  if (sampblock == zero_block) {
668  sound_unref(s);
669  return 0.0;
670  }
671  }
672  /* -blocklen <= cnt <= -1 */
673 
674  /* get next 2 samples and interpolate */
675  x1 = sampblock->samples[blocklen + cnt];
676  if (cnt == -1) {
677  sampblock = sound_get_next(s, &blocklen);
678  cnt -= blocklen;
679  }
680  x2 = sampblock->samples[blocklen + cnt + 1];
681  sound_unref(s); /* free the reader */
682 
683  return (x1 + exact_cnt * (x2 - x1)) * s->scale;

687 /* snd_sref_inverse -- find time point corresponding to some value */
688 
689 double snd_sref_inverse(sound_type s, double val)
690 {
691  double exact_cnt; /* how many fractional samples to scan */
692  int i;
693  sample_block_type sampblock;
694  long blocklen;
695  sample_type x1, x2; /* interpolate between these samples */
696 
697  if (val < 0) {
698  xlcerror("return 0", "negative value", cvflonum(val));
699  return 0.0;
700  }
701  s = sound_copy(s); /* don't modify s, create new reader */
702 
703  x1 = 0.0F;
704  /* now flush cnt samples */
705  while (true) {
706  sampblock = sound_get_next(s, &blocklen);
707  x2 = sampblock->samples[blocklen - 1];
708  if (x2 >= val) break;
709  x1 = x2;
710  if (sampblock == zero_block) {
711  xlcerror("return 0", "too large, no inverse", cvflonum(val));
712  sound_unref(s);
713  return 0.0;
714  }
715  }
716  /* x1 = last sample of previous block,
717  sampblock contains a value larger than val
718  blocklen is the length of sampblock */
719 
720  /* search for first element exceeding val - could
721  * use binary search, but maximum block size places
722  * an upper bound on how bad this can get and we
723  * search for the right block linearly anyway.
724  */
725  for (i = 0; i < blocklen && sampblock->samples[i] <= val; i++) ;
726 
727  /* now i is index of element exceeding val */
728  if (i > 1) x1 = sampblock->samples[i - 1];
729  x2 = sampblock->samples[i];
730 
731  /* now interpolate to get fractional part */
732  if (x2 == x1) exact_cnt = 0;
733  else exact_cnt = (val - x1) / (x2 - x1);
734 
735  /* and add the sample count of x1 */
736  exact_cnt += (s->current - blocklen) + (i - 1);
737 
738  /* negative counts are possible because the first x1 is at
739  * sample -1, so force the location to be at least 0
740  */
741  if (exact_cnt < 0) exact_cnt = 0;
742 
743  /* compute time = t0 + count / samplerate; */
744  exact_cnt = s->t0 + exact_cnt / s->sr;
745 
746  sound_unref(s); /* free the reader */
747  return exact_cnt;
748 }

758 /* snd_xform -- return a sound with transformations applied */
759 /*
760  * The "logical" sound starts at snd->time and runs until some
761  * as yet unknown termination time. (There is also a possibly
762  * as yet unknown logical stop time that is irrelevant here.)
763  * The sound is clipped (zero) until snd->t0 and after snd->stop,
764  * the latter being a sample count, not a time_type.
765  * So, the "physical" sound starts at snd->t0 and runs for up to
766  * snd->stop samples (or less if the sound terminates beforehand).
767  *
768  * The snd_xform procedure operates at the "logical" level, shifting
769  * the sound from its snd->time to time. The sound is stretched as
770  * a result of setting the sample rate to sr. It is then (further)
771  * clipped between start_time and stop_time. If initial samples
772  * are clipped, the sound is shifted again so that it still starts
773  * at time. The sound is then scaled by scale.
774  *
775  * To support clipping of initial samples, the "physical" start time
776  * t0 is set to when the first unclipped sample will be returned, but
777  * the number of samples to clip is saved as a negative count. The
778  * fetch routine SND_flush is installed to flush the clipped samples
779  * at the time of the first fetch. SND_get_first is then installed
780  * for future fetches.
781  *
782  * An empty (zero) sound will be returned if all samples are clipped.
783  *
784  */
785 sound_type snd_xform(sound_type snd,
786  rate_type sr,
787  time_type time,
788  time_type start_time,
789  time_type stop_time,
790  promoted_sample_type scale)
791 {
792  long start_cnt, stop_cnt; /* clipping samples (sample 0 at new t0) */
793 
794  /* start_cnt should reflect max of where the sound starts (t0)
795  * and the new start_time.
796  */
797  if (start_time == MIN_START_TIME) {
798  start_cnt = 0;
799  } else {
800  double new_start_cnt = ((start_time - time) * sr) + 0.5;
801  start_cnt = ((new_start_cnt > 0) ? (long) new_start_cnt : 0);
802  }
803  /* if (start_cnt < -(snd->current)) start_cnt = -(snd->current); */
804 
805  /* stop_cnt should reflect min of the new stop_time and the previous
806  * snd->stop.
807  */
808  if (stop_time == MAX_STOP_TIME) {
809  stop_cnt = MAX_STOP;
810  } else {
811  double new_stop_cnt = ((stop_time - time) * sr) + 0.5;
812  if (new_stop_cnt < MAX_STOP) {
813  stop_cnt = (long) new_stop_cnt;
814  } else {
815  errputstr("Warning: stop count overflow in snd_xform\n");
816  stop_cnt = MAX_STOP;
817  }
818  }
819 
820  if (stop_cnt > snd->stop) {
821  stop_cnt = snd->stop;
822  }
823 
824  if (stop_cnt < 0 || start_cnt >= stop_cnt) {
825  snd = sound_create(NULL, time, sr, 1.0);
826  /* sound_create goes ahead and allocates a snd_list node, so
827  * we need to free it.
828  * Calling snd_list_unref here seems like the right thing, but
829  * it assumes too much structure is in place. ffree_snd_list
830  * is simpler and more direct:
831  */
832  ffree_snd_list(snd->list, "snd_xform");
833  snd->list = zero_snd_list;
834  nyquist_printf("snd_xform: (stop_time < t0 or start >= stop) "
835  "-> zero sound = %p\n", snd);
836 
837  } else {
838  snd = sound_copy(snd);
839  snd->t0 = time;
840  if (start_cnt) {
841  snd->current -= start_cnt; /* indicate flush with negative num. */
842  /* the following code assumes that SND_get_first is the
843  routine to be called to get the first samples from this
844  sound. We're going to replace it with SND_flush. First,
845  make sure that the assumption is correct:
846  */
847  if ((snd->get_next != SND_get_first) &&
848  (snd->get_next != SND_flush)) {
849  errputstr("snd_xform: SND_get_first expected\n");
850  EXIT(1);
851  }
852  /* this will flush -current samples and revert to SND_get_first */
853  snd->get_next = SND_flush;
854  stop_cnt -= start_cnt;
855  }
856  snd->stop = stop_cnt;
857  snd->sr = sr;
858  snd->scale *= (float) scale;
859  }
860  return snd;
861 }

864 /* SND_flush -- the get_next function for flushing clipped samples */
865 /*
866  * this only gets called once: it flushes -current samples (a
867  * non-real-time operation) and installs SND_get_next to return
868  * blocks normally from then on.
869  */
870 sample_block_type SND_flush(sound_type snd, long * cnt)
871 {
872  long mycnt;
873  sample_block_type block = SND_get_first(snd, &mycnt);
874  /* changed from < to <= because we want to read at least the first sample */
875  while (snd->current <= 0) {
876  block = SND_get_next(snd, &mycnt);
877  }
878  /* at this point, we've read to and including the block with
879  * the first samples we want to return. If the block boundary
880  * is in the right place, we can do a minimal fixup and return:
881  */
882  if (snd->current == snd->list->block_len) {
883  *cnt = snd->current; /* == snd->list->block_len */
884  /* snd->get_next = SND_get_next; -- done by SND_get_first */
885  return block;
886  } else /* snd->current < snd->list->block_len */ {
887  long i;
888  sample_block_values_type from_ptr;
889  /* we have to return a partial block */
890  /* NOTE: if we had been smart, we would have had SND_get_next
891  * return a pointer to samples rather than a pointer to the
892  * block, which has a reference count. Since the caller
893  * expects a pointer to a reference count, we have to copy
894  * snd->current samples to a new block
895  */
896  snd_list_type snd_list = snd_list_create((snd_susp_type) snd->list->u.next);
897  snd_list->u.next->refcnt++;
898  falloc_sample_block(snd_list->block, "SND_flush");
899  /* now copy samples */
900  from_ptr = block->samples + snd->list->block_len - snd->current;
901  for (i = 0; i < snd->current; i++) {
902  snd_list->block->samples[i] = from_ptr[i];
903  }
904  snd_list_unref(snd->list);
905  snd->list = snd_list;
906  *cnt = snd->current;
907  return snd_list->block;
908  }
909 }

943 /****************************************************************************
944 * SND_get_next
945 * Inputs:
946 * sound_type snd: The iterator whose next block is to be computed
947 * int * cnt: Place to put count of samples returned
948 * Result: snd_list_type
949 * Pointer to the sample block computed ---------------------------+
950 * Effect: |
951 * force suspension to compute next block of samples |
952 * |
953 * Here's the protocol for using this and related functions: |
954 * Every client (sample reader) has a private sound_type (an iterator), |
955 * and the sound_type's 'list' field points to a header (of type |
956 * snd_list_type). The header in turn points to a block of samples. |
957 * |
958 * +---------------------------------------+
959 * |
960 * |
961 * | sample_block_type
962 * (snd) V +---+--+--+--+--+--+--+-...-+--+
963 * sound_type: snd_list_type +-->|ref| | | | |//|//| |//|
964 * +---------+ +----------+ | +---+--+--+--+--+--+--+-...-+--+
965 * | list +------->| block +--+ ^
966 * +---------+ +----------+ :
967 * | t0 | | block_len|....................:
968 * +---------+ +----------+
969 * | sr | | refcnt |
970 * +---------+ +-+--------+
971 * | current | | next +---->... Note: the union u
972 * +---------+ |u|........| snd_list_type points to only one
973 * | rate | | | susp +---->... of the indicated
974 * +---------+ +-+--------+ susp_type types
975 * | scalse | |log_stop |
976 * +---------+ +----------+
977 * | lsc |
978 * +---------+
979 * |get_next +-----> SND_get_next()
980 * +---------+
981 *
982 * The sound_type keeps track of where the next sample block will
983 * come from. The field 'current' is the number of the first sample of
984 * the next block to be returned, where sample numbers start
985 * at zero. The normal fetch procedure is this one, although special
986 * cases may generate special block generators, e.g., CONST does not need
987 * to allocate and refill a block and can reuse the same block over and
988 * over again, so it may have its own fetch procedure. This is the
989 * general fetch procedure, which assumes that the generator function
990 * actually produces a slightly different value for each sample.
991 *
992 * The distinguishing characteristic of whether the 'u' field is to be
993 * interpreted as 'next', a link to the next list element, or 'susp', a
994 * reference to the suspension for generating a new sample block, is
995 * whether the 'block' parameter is NULL or not. If it is NULL, then
996 * u.susp tells how to generate the block; if it is not NULL, u.next is
997 * a pointer to the next sound block in the list.
998 *
999 * When the 'block' pointer is NULL, we create a block of samples, and
1000 * create a new sound list element which follows it which has a NULL
1001 * 'block' pointer; the 'u' field of the current list element is now
1002 * interpreted as 'u.next'.
1003 *
1004 * The client calls SND_get_next to get a pointer to a block of samples.
1005 * The count of samples generated is returned via a ref parameter, and
1006 * SND_get_next will not be called again until this set is exhausted.
1007 *
1008 * The next time SND_get_next is called, it knows that the sample block
1009 * has been exhausted. It releases its reference to the block (and if
1010 * that was the last reference, frees the block to the block allocation
1011 * pool), allocates a new block from the block pool, and proceeds to
1012 * fill it with samples.
1013 *
1014 * Note that as an optimization, if the refcnt field goes to 0 it
1015 * could immediately re-use the block without freeing back to the block
1016 * pool and reallocating it.
1017 *
1018 * Because of the way we handle sound sample blocks, the sound sample blocks
1019 * themselves are ref-counted, so freeing the snd_list_type may not free
1020 * the sample block it references. At the level of this procedure, that
1021 * is transparently handled by the snd_list_unref function.
1022 *
1023 * Logical stop:
1024 *
1025 * Logical stop is handled by several mechanisms. The /intrinsic/ logical
1026 * stop is an immutable property of the signal, and is determined by the
1027 * specification in the algorithm description file. When it is encountered,
1028 * the 'logically_stopped' flag of the snd_list_node is set.
1029 * The generators guarantee that the first time this is encountered, it
1030 * will always be constructed so that the first sample of the block it
1031 * references is the logical stop time.
1032 *
1033 * In addition, the client may have set the /explicit logical stop time/ of
1034 * the iterator (e.g., in nyquist, the (set-logical-stop sound time) call copies
1035 * the sound, altering its logical stop). The logical stop time, when set
1036 * in this way, causes the logical_stop_cnt ('lsc' in the above diagram)
1037 * to be set to the count of the last sample to be generated before the
1038 * <logical stop time. This will guarantee that the sound will indicate that
1039 * it has reached its logical stop time when the indicated sample is
1040 * generated.
1041 ****************************************************************************/

1046  * the first block, plug in SND_get_next for successive blocks
1047  */
1048 sample_block_type SND_get_first(sound_type snd, long * cnt)
1049 {
1050  register snd_list_type snd_list = snd->list;
1051  /*
1052  * If there is not a block of samples, we need to generate one.
1053  */
1054  if (snd_list->block == NULL) {
1055  /*
1056  * Call the 'fetch' method for this sound_type to generate
1057  * a new block of samples.
1058  */
1059  snd_susp_type susp = snd_list->u.susp;
1060 
1061  snd_list->u.next = snd_list_create(susp);
1062  snd_list->block = internal_zero_block;
1063  /* nyquist_printf("SND_get_first: susp->fetch %p\n",
1064  susp->fetch); */
1065  assert(susp->log_stop_cnt == UNKNOWN || susp->log_stop_cnt >= 0);
1066  (*(susp->fetch))(susp, snd_list);
1067 #ifdef GC_DEBUG
1068  snd_list_debug(snd_list, "SND_get_first");
1069 #endif

1071  snd_list, snd_list->block, snd_list->block_len); */
1072  }
1073  if ((snd->logical_stop_cnt == UNKNOWN) && snd_list->logically_stopped) {
1074  /* nyquist_printf("SND_get_first/next: snd %p logically stopped at %d\n",
1075  snd, snd->current); */
1076  snd->logical_stop_cnt = snd->current;
1077  }

1080  if (snd->current + snd_list->block_len > snd->stop) {

1082  if (snd->current == snd->stop) {

1090  snd->list->block = snd_list->block;

1092  snd_list_unref(snd_list);
1093  }
1094  snd_list = snd->list; /* used below to return block ptr */
1095  }

1103  dbg_mem_print("block info:", snd_list->block);
1104 #endif

1107  stdputstr("using something like: (SETF X (SEQ (SOUND X) ...))\n");
1108  stdputstr("Nyquist aborts from non-recoverable error\n");
1109  abort();
1110  }
1111  snd->current += snd_list->block_len; /* count how many we read */
1112  snd->get_next = SND_get_next;
1113  return snd_list->block;
1114 }
