378 /* mark - mark all accessible nodes */
379 void mark(LVAL ptr)
380 {
381  register LVAL this,prev,tmp;
382  int type,i,n;
383 
384  /* initialize */
385  prev = NIL;
386  this = ptr;
387 
388  /* mark this list */
389  for (;;) {
390 
391  /* descend as far as we can */
392  while (!(this->n_flags & MARK))
393 
394  /* check cons and symbol nodes */
395  if (((type = ntype(this))) == CONS || type == USTREAM) {
396  if ((tmp = car(this))) {
397  this->n_flags |= MARK|LEFT;
398  rplaca(this,prev);
399  }
400  else if ((tmp = cdr(this))) {
401  this->n_flags |= MARK;
402  rplacd(this,prev);
403  }
404  else { /* both sides nil */
405  this->n_flags |= MARK;
406  break;
407  }
408  prev = this; /* step down the branch */
409  this = tmp;
410  }
411 
412  /* mark other node types */
413  else {
414  this->n_flags |= MARK;
415  switch (type) {
416  case SYMBOL:
417  case OBJECT:
418  case VECTOR:
419  case CLOSURE:
420  for (i = 0, n = getsize(this); --n >= 0; ++i)
421  if ((tmp = getelement(this,i)))
422  mark(tmp);
423  break;
424  case EXTERN:
425  if (getdesc(this)->mark_meth) { (*(getdesc(this)->mark_meth))(getinst(this));
426  }
427  }
428  break;
429  }
430 
431  /* backup to a point where we can continue descending */
432  for (;;)
433 
434  /* make sure there is a previous node */
435  if (prev) {
436  if (prev->n_flags & LEFT) { /* came from left side */
437  prev->n_flags &= ~LEFT;
438  tmp = car(prev);
439  rplaca(prev,this);
440  if ((this = cdr(prev))) {
441  rplacd(prev,tmp);
442  break;
443  }
444  }
445  else { /* came from right side */
446  tmp = cdr(prev);
447  rplacd(prev,this);
448  }
449  this = prev; /* step back up the branch */
450  prev = tmp;
451  }
452 
453  /* no previous node, must be done */
454  else
455  return;
456  }
457 }

459 /* sweep - sweep all unmarked nodes and add them to the free list */
460 LOCAL void sweep(void)
461 {
462  SEGMENT *seg;
463  LVAL p;
464  int n;
465 
466  /* empty the free list */
467  fnodes = NIL;
468  nfree = 0L;
469 
470  /* add all unmarked nodes */
471  for (seg = segs; seg; seg = seg->sg_next) {
472  if (seg == fixseg) /* don't sweep the fixnum segment */
473  continue;
474  else if (seg == charseg) /* don't sweep the character segment */
475  continue;
476  p = &seg->sg_nodes[0];
477  for (n = seg->sg_size; --n >= 0; ++p) {
478 #ifdef DEBUG_MEM
479  if (xldmem_trace &&
480  ntype(p) == EXTERN &&
481  xldmem_trace == getinst(p)) {
482  printf("sweep: EXTERN node %lx is %smarked, points to %lx\n",
483  p, (p->n_flags & MARK ? "" : "un"), getinst(p));
484  }
485 #endif
486  if (!(p->n_flags & MARK)) {
487  switch (ntype(p)) {
488  case STRING:
489  if (getstring(p) != NULL) {
490  total -= (long)getslength(p);
491  free(getstring(p));
492  }
493  break;
494  case STREAM:
495  if (getfile(p))
496  osclose(getfile(p));
497  break;
498  case SYMBOL:
499  case OBJECT:
500  case VECTOR:
501  case CLOSURE:
502  if (p->n_vsize) {
503  total -= (long) (p->n_vsize * sizeof(LVAL));
504  free((void *) p->n_vdata);
505  }
506  break;
507  case EXTERN:
508  /* printf("GC about to free %x\n", p);
509  * fflush(stdout);
510  */
511  if (getdesc(p)) { (*(getdesc(p)->free_meth))(getinst(p));
512  }
513  break;
514  }
515  p->n_type = FREE_NODE;
516  rplaca(p,NIL);
517  rplacd(p,fnodes);
518  fnodes = p;
519  nfree += 1L;
520  }
521  else
522  p->n_flags &= ~MARK;
523  }
524  }
525 }
