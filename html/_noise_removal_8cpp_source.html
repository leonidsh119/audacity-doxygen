205 bool EffectNoiseRemoval::Process()
206 {
207  Initialize();
208 
209  // This same code will both remove noise and profile it,
210  // depending on 'mDoProfile'
211  this->CopyInputTracks(); // Set up mOutputTracks.
212  bool bGoodResult = true;
213 
214  SelectedTrackListOfKindIterator iter(Track::Wave, mOutputTracks);
215  WaveTrack *track = (WaveTrack *) iter.First();
216  int count = 0;
217  while (track) {
218  double trackStart = track->GetStartTime();
219  double trackEnd = track->GetEndTime();
220  double t0 = mT0 < trackStart? trackStart: mT0;
221  double t1 = mT1 > trackEnd? trackEnd: mT1;
222 
223  if (t1 > t0) {
224  sampleCount start = track->TimeToLongSamples(t0);
225  sampleCount end = track->TimeToLongSamples(t1);
226  sampleCount len = (sampleCount)(end - start);
227 
228  if (!ProcessOne(count, track, start, len)) {
229  Cleanup();
230  bGoodResult = false;
231  break;
232  }
233  }
234  track = (WaveTrack *) iter.Next();
235  count++;
236  }
237 
238  if (bGoodResult && mDoProfile) {
239  mHasProfile = true;
240  mDoProfile = false;
241  }
242 
243  if (bGoodResult)
244  Cleanup();
245  this->ReplaceProcessedTracks(bGoodResult);
246  return bGoodResult;
247 }

249 void EffectNoiseRemoval::ApplyFreqSmoothing(float *spec)
250 {
251  float *tmp = new float[mSpectrumSize];
252  int i, j, j0, j1;
253 
254  for(i = 0; i < mSpectrumSize; i++) {
255  j0 = wxMax(0, i - mFreqSmoothingBins);
256  j1 = wxMin(mSpectrumSize-1, i + mFreqSmoothingBins);
257  tmp[i] = 0.0;
258  for(j = j0; j <= j1; j++) {
259  tmp[i] += spec[j];
260  }
261  tmp[i] /= (j1 - j0 + 1);
262  }
263 
264  for(i = 0; i < mSpectrumSize; i++)
265  spec[i] = tmp[i];
266 
267  delete[] tmp;
268 }

270 void EffectNoiseRemoval::Initialize()
271 {
272  int i;
273 
274  mSampleRate = mProjectRate;
275  mFreqSmoothingBins = (int)(mFreqSmoothingHz * mWindowSize / mSampleRate);
276  mAttackDecayBlocks = 1 +
277  (int)(mAttackDecayTime * mSampleRate / (mWindowSize / 2));
278  mNoiseAttenFactor = DB_TO_LINEAR(mNoiseGain);
279  mOneBlockAttackDecay = DB_TO_LINEAR(mNoiseGain / mAttackDecayBlocks);
280  // Applies to power, divide by 10:
281  mSensitivityFactor = pow(10.0, mSensitivity/10.0);
282  mMinSignalBlocks =
283  (int)(mMinSignalTime * mSampleRate / (mWindowSize / 2));
284  if( mMinSignalBlocks < 1 )
285  mMinSignalBlocks = 1;
286  mHistoryLen = (2 * mAttackDecayBlocks) - 1;
287 
288  if (mHistoryLen < mMinSignalBlocks)
289  mHistoryLen = mMinSignalBlocks;
290 
291  mSpectrums = new float*[mHistoryLen];
292  mGains = new float*[mHistoryLen];
293  mRealFFTs = new float*[mHistoryLen];
294  mImagFFTs = new float*[mHistoryLen];
295  for(i = 0; i < mHistoryLen; i++) {
296  mSpectrums[i] = new float[mSpectrumSize];
297  mGains[i] = new float[mSpectrumSize];
298  mRealFFTs[i] = new float[mSpectrumSize];
299  mImagFFTs[i] = new float[mSpectrumSize];
300  }
301 
302  // Initialize the FFT
303  hFFT = InitializeFFT(mWindowSize);
304 
305  mFFTBuffer = new float[mWindowSize];
306  mInWaveBuffer = new float[mWindowSize];
307  mWindow = new float[mWindowSize];
308  mOutOverlapBuffer = new float[mWindowSize];
309 
310  // Create a Hanning window function
311  for(i=0; i<mWindowSize; i++)
312  mWindow[i] = 0.5 - 0.5 * cos((2.0*M_PI*i) / mWindowSize);
313 
314  if (mDoProfile) {
315  for (i = 0; i < mSpectrumSize; i++)
316  mNoiseThreshold[i] = float(0);
317  }

369 {

371 
372  while(len && mOutSampleCount < mInSampleCount) {
373  int avail = wxMin(len, mWindowSize - mInputPos);

397 void EffectNoiseRemoval::FillFirstHistoryWindow()
398 {
399  int i;
400 
401  for(i=0; i < mWindowSize; i++)
402  mFFTBuffer[i] = mInWaveBuffer[i];
403  RealFFTf(mFFTBuffer, hFFT);
404  for(i = 1; i < (mSpectrumSize-1); i++) {
405  mRealFFTs[0][i] = mFFTBuffer[hFFT->BitReversed[i] ];
406  mImagFFTs[0][i] = mFFTBuffer[hFFT->BitReversed[i]+1];
407  mSpectrums[0][i] = mRealFFTs[0][i]*mRealFFTs[0][i] + mImagFFTs[0][i]*mImagFFTs[0][i];
408  mGains[0][i] = mNoiseAttenFactor;
409  }
410  // DC and Fs/2 bins need to be handled specially
411  mSpectrums[0][0] = mFFTBuffer[0]*mFFTBuffer[0];
412  mSpectrums[0][mSpectrumSize-1] = mFFTBuffer[1]*mFFTBuffer[1];
413  mGains[0][0] = mNoiseAttenFactor;
414  mGains[0][mSpectrumSize-1] = mNoiseAttenFactor;
415 }

417 void EffectNoiseRemoval::RotateHistoryWindows()
418 {
419  int last = mHistoryLen - 1;
420  int i;
421 
422  // Remember the last window so we can reuse it
423  float *lastSpectrum = mSpectrums[last];
424  float *lastGain = mGains[last];
425  float *lastRealFFT = mRealFFTs[last];
426  float *lastImagFFT = mImagFFTs[last];
427 
428  // Rotate each window forward
429  for(i = last; i >= 1; i--) {
430  mSpectrums[i] = mSpectrums[i-1];
431  mGains[i] = mGains[i-1];
432  mRealFFTs[i] = mRealFFTs[i-1];
433  mImagFFTs[i] = mImagFFTs[i-1];
434  }
435 
436  // Reuse the last buffers as the NEW first window
437  mSpectrums[0] = lastSpectrum;
438  mGains[0] = lastGain;
439  mRealFFTs[0] = lastRealFFT;
440  mImagFFTs[0] = lastImagFFT;
441 }

463 void EffectNoiseRemoval::GetProfile()
464 {
465  // The noise threshold for each frequency is the maximum
466  // level achieved at that frequency for a minimum of
467  // mMinSignalBlocks blocks in a row - the max of a min.
468 
469  int start = mHistoryLen - mMinSignalBlocks;
470  int finish = mHistoryLen;
471  int i, j;
472 
473  for (j = 0; j < mSpectrumSize; j++) {
474  float min = mSpectrums[start][j];
475  for (i = start+1; i < finish; i++) {
476  if (mSpectrums[i][j] < min)
477  min = mSpectrums[i][j];
478  }
479  if (min > mNoiseThreshold[j])
480  mNoiseThreshold[j] = min;
481  }
482 
483  mOutSampleCount += mWindowSize / 2; // what is this for? Not used when we are getting the profile?
484 }

486 void EffectNoiseRemoval::RemoveNoise()
487 {
488  int center = mHistoryLen / 2;
489  int start = center - mMinSignalBlocks/2;
490  int finish = start + mMinSignalBlocks;
491  int i, j;
492 
493  // Raise the gain for elements in the center of the sliding history
494  for (j = 0; j < mSpectrumSize; j++) {
495  float min = mSpectrums[start][j];
496  for (i = start+1; i < finish; i++) {
497  if (mSpectrums[i][j] < min)
498  min = mSpectrums[i][j];
499  }
500  if (min > mSensitivityFactor * mNoiseThreshold[j] && mGains[center][j] < 1.0) {
501  if (mbLeaveNoise) mGains[center][j] = 0.0;
502  else mGains[center][j] = 1.0;
503  } else {
504  if (mbLeaveNoise) mGains[center][j] = 1.0;
505  }
506  }
507 
508  // Decay the gain in both directions;
509  // note that mOneBlockAttackDecay is less than 1.0
510  // of linear attenuation per block
511  for (j = 0; j < mSpectrumSize; j++) {
512  for (i = center + 1; i < mHistoryLen; i++) {
513  if (mGains[i][j] < mGains[i - 1][j] * mOneBlockAttackDecay)
514  mGains[i][j] = mGains[i - 1][j] * mOneBlockAttackDecay;
515  if (mGains[i][j] < mNoiseAttenFactor)
516  mGains[i][j] = mNoiseAttenFactor;
517  }
518  for (i = center - 1; i >= 0; i--) {
519  if (mGains[i][j] < mGains[i + 1][j] * mOneBlockAttackDecay)
520  mGains[i][j] = mGains[i + 1][j] * mOneBlockAttackDecay;
521  if (mGains[i][j] < mNoiseAttenFactor)
522  mGains[i][j] = mNoiseAttenFactor;
523  }
524  }
525 
526 
527  // Apply frequency smoothing to output gain
528  int out = mHistoryLen - 1; // end of the queue
529 
530  ApplyFreqSmoothing(mGains[out]);
531 
532  // Apply gain to FFT
533  for (j = 0; j < (mSpectrumSize-1); j++) {
534  mFFTBuffer[j*2 ] = mRealFFTs[out][j] * mGains[out][j];
535  mFFTBuffer[j*2+1] = mImagFFTs[out][j] * mGains[out][j];
536  }
537  // The Fs/2 component is stored as the imaginary part of the DC component
538  mFFTBuffer[1] = mRealFFTs[out][mSpectrumSize-1] * mGains[out][mSpectrumSize-1];
539 
540  // Invert the FFT into the output buffer
541  InverseRealFFTf(mFFTBuffer, hFFT);
542 
543  // Overlap-add
544  for(j = 0; j < (mSpectrumSize-1); j++) {
545  mOutOverlapBuffer[j*2 ] += mFFTBuffer[hFFT->BitReversed[j] ] * mWindow[j*2 ];
546  mOutOverlapBuffer[j*2+1] += mFFTBuffer[hFFT->BitReversed[j]+1] * mWindow[j*2+1];
547  }
548 
549  // Output the first half of the overlap buffer, they're done -
550  // and then shift the next half over.
551  if (mOutSampleCount >= 0) { // ...but not if it's the first half-window
552  mOutputTrack->Append((samplePtr)mOutOverlapBuffer, floatSample,
553  mWindowSize / 2);
554  }
555  mOutSampleCount += mWindowSize / 2;
556  for(j = 0; j < mWindowSize / 2; j++) {
557  mOutOverlapBuffer[j] = mOutOverlapBuffer[j + (mWindowSize / 2)];
558  mOutOverlapBuffer[j + (mWindowSize / 2)] = 0.0;
559  }
560 }

562 bool EffectNoiseRemoval::ProcessOne(int count, WaveTrack * track,
563  sampleCount start, sampleCount len)
564 {
565  if (track == NULL)
566  return false;
567 
568  StartNewTrack();
569 
570  if (!mDoProfile)
571  mOutputTrack = mFactory->NewWaveTrack(track->GetSampleFormat(),
572  track->GetRate());
573 
574  sampleCount bufferSize = track->GetMaxBlockSize();
575  float *buffer = new float[bufferSize];
576 
577  bool bLoopSuccess = true;
578  sampleCount blockSize;
579  sampleCount samplePos = start;
580  while (samplePos < start + len) {
581  //Get a blockSize of samples (smaller than the size of the buffer)
582  blockSize = track->GetBestBlockSize(samplePos);
583 
584  //Adjust the block size if it is the final block in the track
585  if (samplePos + blockSize > start + len)
586  blockSize = start + len - samplePos;
587 
588  //Get the samples from the track and put them in the buffer
589  track->Get((samplePtr)buffer, floatSample, samplePos, blockSize);
590 
591  mInSampleCount += blockSize;
592  ProcessSamples(blockSize, buffer);
593 
594  samplePos += blockSize;
595 
596  // Update the Progress meter
597  if (TrackProgress(count, (samplePos - start) / (double)len)) {
598  bLoopSuccess = false;
599  break;
600  }
601  }
602 
603  FinishTrack();
604  delete [] buffer;
605 
606  if (!mDoProfile) {
607  // Flush the output WaveTrack (since it's buffered)
608  mOutputTrack->Flush();
609 
610  // Take the output track and insert it in place of the original
611  // sample data (as operated on -- this may not match mT0/mT1)
612  if (bLoopSuccess) {
613  double t0 = mOutputTrack->LongSamplesToTime(start);
614  double tLen = mOutputTrack->LongSamplesToTime(len);
615  // Filtering effects always end up with more data than they started with. Delete this 'tail'.
616  mOutputTrack->HandleClear(tLen, mOutputTrack->GetEndTime(), false, false);
617  bool bResult = track->ClearAndPaste(t0, t0 + tLen, mOutputTrack, true, false);
618  wxASSERT(bResult); // TO DO: Actually handle this.
619  }
620 
621  // Delete the outputTrack now that its data is inserted in place
622  delete mOutputTrack;
623  mOutputTrack = NULL;
624  }
625 
626  return bLoopSuccess;
627 }
