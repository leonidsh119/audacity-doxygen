536 int

538 { int subformat, endian ;
539 
540  subformat = SF_CODEC (info->format) ;
541  endian = SF_ENDIAN (info->format) ;
542 
543  /* This is the place where each file format can check if the suppiled
544  ** SF_INFO struct is valid.
545  ** Return 0 on failure, 1 ons success.
546  */
547 
548  if (info->channels < 1 || info->channels > SF_MAX_CHANNELS)
549  return 0 ;
550 
551  if (info->samplerate < 0)
552  return 0 ;
553 
554  switch (SF_CONTAINER (info->format))
555  { case SF_FORMAT_WAV :
556  case SF_FORMAT_WAVEX :
557  /* WAV now allows both endian, RIFF or RIFX (little or big respectively) */
558  if (subformat == SF_FORMAT_PCM_U8 || subformat == SF_FORMAT_PCM_16)
559  return 1 ;
560  if (subformat == SF_FORMAT_PCM_24 || subformat == SF_FORMAT_PCM_32)
561  return 1 ;
562  if ((subformat == SF_FORMAT_IMA_ADPCM || subformat == SF_FORMAT_MS_ADPCM) && info->channels <= 2)
563  return 1 ;
564  if (subformat == SF_FORMAT_GSM610 && info->channels == 1)
565  return 1 ;
566  if (subformat == SF_FORMAT_G721_32 && info->channels == 1)
567  return 1 ;
568  if (subformat == SF_FORMAT_ULAW || subformat == SF_FORMAT_ALAW)
569  return 1 ;
570  if (subformat == SF_FORMAT_FLOAT || subformat == SF_FORMAT_DOUBLE)
571  return 1 ;
572  break ;
573 
574  case SF_FORMAT_AIFF :
575  /* AIFF does allow both endian-nesses for PCM data.*/
576  if (subformat == SF_FORMAT_PCM_16 || subformat == SF_FORMAT_PCM_24 || subformat == SF_FORMAT_PCM_32)
577  return 1 ;
578  /* Other encodings. Check for endian-ness. */
579  if (endian == SF_ENDIAN_LITTLE || endian == SF_ENDIAN_CPU)
580  return 0 ;
581  if (subformat == SF_FORMAT_PCM_U8 || subformat == SF_FORMAT_PCM_S8)
582  return 1 ;
583  if (subformat == SF_FORMAT_FLOAT || subformat == SF_FORMAT_DOUBLE)
584  return 1 ;
585  if (subformat == SF_FORMAT_ULAW || subformat == SF_FORMAT_ALAW)
586  return 1 ;
587  if ((subformat == SF_FORMAT_DWVW_12 || subformat == SF_FORMAT_DWVW_16 ||
588  subformat == SF_FORMAT_DWVW_24) && info-> channels == 1)
589  return 1 ;
590  if (subformat == SF_FORMAT_GSM610 && info->channels == 1)
591  return 1 ;
592  if (subformat == SF_FORMAT_IMA_ADPCM && (info->channels == 1 || info->channels == 2))
593  return 1 ;
594  break ;
595 
596  case SF_FORMAT_AU :
597  if (subformat == SF_FORMAT_PCM_S8 || subformat == SF_FORMAT_PCM_16)
598  return 1 ;
599  if (subformat == SF_FORMAT_PCM_24 || subformat == SF_FORMAT_PCM_32)
600  return 1 ;
601  if (subformat == SF_FORMAT_ULAW || subformat == SF_FORMAT_ALAW)
602  return 1 ;
603  if (subformat == SF_FORMAT_FLOAT || subformat == SF_FORMAT_DOUBLE)
604  return 1 ;
605  if (subformat == SF_FORMAT_G721_32 && info->channels == 1)
606  return 1 ;
607  if (subformat == SF_FORMAT_G723_24 && info->channels == 1)
608  return 1 ;
609  if (subformat == SF_FORMAT_G723_40 && info->channels == 1)
610  return 1 ;
611  break ;
612 
613  case SF_FORMAT_CAF :
614  if (subformat == SF_FORMAT_PCM_S8 || subformat == SF_FORMAT_PCM_16)
615  return 1 ;
616  if (subformat == SF_FORMAT_PCM_24 || subformat == SF_FORMAT_PCM_32)
617  return 1 ;
618  if (subformat == SF_FORMAT_ULAW || subformat == SF_FORMAT_ALAW)
619  return 1 ;
620  if (subformat == SF_FORMAT_FLOAT || subformat == SF_FORMAT_DOUBLE)
621  return 1 ;
622  break ;
623 
624  case SF_FORMAT_RAW :
625  if (subformat == SF_FORMAT_PCM_U8 || subformat == SF_FORMAT_PCM_S8 || subformat == SF_FORMAT_PCM_16)
626  return 1 ;
627  if (subformat == SF_FORMAT_PCM_24 || subformat == SF_FORMAT_PCM_32)
628  return 1 ;
629  if (subformat == SF_FORMAT_FLOAT || subformat == SF_FORMAT_DOUBLE)
630  return 1 ;
631  if (subformat == SF_FORMAT_ALAW || subformat == SF_FORMAT_ULAW)
632  return 1 ;

637  return 1 ;
638  if (subformat == SF_FORMAT_VOX_ADPCM && info->channels == 1)
639  return 1 ;
640  break ;
641 
642  case SF_FORMAT_PAF :
643  if (subformat == SF_FORMAT_PCM_S8 || subformat == SF_FORMAT_PCM_16)
644  return 1 ;
645  if (subformat == SF_FORMAT_PCM_24 || subformat == SF_FORMAT_PCM_32)
646  return 1 ;
647  break ;
648 
649  case SF_FORMAT_SVX :
650  /* SVX only supports writing mono SVX files. */
651  if (info->channels > 1)
652  return 0 ;
653  /* Always big endian. */
654  if (endian == SF_ENDIAN_LITTLE || endian == SF_ENDIAN_CPU)
655  return 0 ;
656 
657  if (subformat == SF_FORMAT_PCM_S8 || subformat == SF_FORMAT_PCM_16)
658  return 1 ;
659  break ;

661  case SF_FORMAT_NIST :

679  if (endian == SF_ENDIAN_BIG || endian == SF_ENDIAN_CPU)
680  return 0 ;
681  if (subformat == SF_FORMAT_PCM_U8 || subformat == SF_FORMAT_PCM_16)
682  return 1 ;
683  if (subformat == SF_FORMAT_ULAW || subformat == SF_FORMAT_ALAW)
684  return 1 ;

687  case SF_FORMAT_W64 :
688  /* W64 is strictly little endian. */
689  if (endian == SF_ENDIAN_BIG || endian == SF_ENDIAN_CPU)
690  return 0 ;
691  if (subformat == SF_FORMAT_PCM_U8 || subformat == SF_FORMAT_PCM_16)
692  return 1 ;
693  if (subformat == SF_FORMAT_PCM_24 || subformat == SF_FORMAT_PCM_32)
694  return 1 ;
695  if ((subformat == SF_FORMAT_IMA_ADPCM || subformat == SF_FORMAT_MS_ADPCM) && info->channels <= 2)
696  return 1 ;
697  if (subformat == SF_FORMAT_GSM610 && info->channels == 1)
698  return 1 ;
699  if (subformat == SF_FORMAT_ULAW || subformat == SF_FORMAT_ALAW)
700  return 1 ;
701  if (subformat == SF_FORMAT_FLOAT || subformat == SF_FORMAT_DOUBLE)
702  return 1 ;
703  break ;
704 
705  case SF_FORMAT_MAT4 :
706  if (subformat == SF_FORMAT_PCM_16 || subformat == SF_FORMAT_PCM_32)
707  return 1 ;
708  if (subformat == SF_FORMAT_FLOAT || subformat == SF_FORMAT_DOUBLE)
709  return 1 ;
710  break ;

712  case SF_FORMAT_MAT5 :
713  if (subformat == SF_FORMAT_PCM_U8 || subformat == SF_FORMAT_PCM_16 || subformat == SF_FORMAT_PCM_32)

715  if (subformat == SF_FORMAT_FLOAT || subformat == SF_FORMAT_DOUBLE)

720  if (subformat == SF_FORMAT_PCM_S8 || subformat == SF_FORMAT_PCM_16 || subformat == SF_FORMAT_PCM_32)
721  return 1 ;
722  break ;
723 
724  case SF_FORMAT_XI :
725  if (info->channels != 1)
726  return 0 ;
727  if (subformat == SF_FORMAT_DPCM_8 || subformat == SF_FORMAT_DPCM_16)
728  return 1 ;
729  break ;
730 
731  case SF_FORMAT_HTK :
732  /* HTK is strictly big endian. */
733  if (endian == SF_ENDIAN_LITTLE || endian == SF_ENDIAN_CPU)
734  return 0 ;
735  if (info->channels != 1)
736  return 0 ;
737  if (subformat == SF_FORMAT_PCM_16)
738  return 1 ;
739  break ;

741  case SF_FORMAT_SDS :
742  /* SDS is strictly big endian. */
743  if (endian == SF_ENDIAN_LITTLE || endian == SF_ENDIAN_CPU)
744  return 0 ;
745  if (info->channels != 1)
746  return 0 ;
747  if (subformat == SF_FORMAT_PCM_S8 || subformat == SF_FORMAT_PCM_16 || subformat == SF_FORMAT_PCM_24)
748  return 1 ;
749  break ;

751  case SF_FORMAT_AVR :
752  /* SDS is strictly big endian. */
753  if (endian == SF_ENDIAN_LITTLE || endian == SF_ENDIAN_CPU)
754  return 0 ;
755  if (info->channels > 2)
756  return 0 ;
757  if (subformat == SF_FORMAT_PCM_U8 || subformat == SF_FORMAT_PCM_S8 || subformat == SF_FORMAT_PCM_16)
758  return 1 ;
759  break ;
760 
761  case SF_FORMAT_FLAC :
762  /* FLAC can't do more than 8 channels. */
763  if (info->channels > 8)
764  return 0 ;
765  if (subformat == SF_FORMAT_PCM_S8 || subformat == SF_FORMAT_PCM_16 || subformat == SF_FORMAT_PCM_24)
766  return 1 ;
767  break ;

774  return 1 ;
775  break ;

777  case SF_FORMAT_WVE :

779  if (endian == SF_ENDIAN_BIG || endian == SF_ENDIAN_CPU)
780  return 0 ;
781  if (info->channels > 1)
782  return 0 ;
783  if (subformat == SF_FORMAT_ALAW)
784  return 1 ;
785  break ;
786 
787  case SF_FORMAT_OGG :
788  if (subformat == SF_FORMAT_VORBIS)
789  return 1 ;
790  break ;
791 
792  case SF_FORMAT_MPC2K :
793  /* MPC2000 is strictly little endian. */
794  if (endian == SF_ENDIAN_BIG || endian == SF_ENDIAN_CPU)
795  return 0 ;
796  if (info->channels > 2)
797  return 0 ;
798  if (subformat == SF_FORMAT_PCM_16)
799  return 1 ;
800  break ;
801 
802  case SF_FORMAT_RF64 :
803  if (endian == SF_ENDIAN_BIG || endian == SF_ENDIAN_CPU)
804  return 0 ;
805  if (subformat == SF_FORMAT_PCM_U8 || subformat == SF_FORMAT_PCM_16)
806  return 1 ;
807  if (subformat == SF_FORMAT_PCM_24 || subformat == SF_FORMAT_PCM_32)

810  return 1 ;

873  return psf_get_format_major (data) ;
874 
875  case SFC_GET_FORMAT_SUBTYPE_COUNT :
876  if (data == NULL || datasize != SIGNED_SIZEOF (int))
877  return (sf_errno = SFE_BAD_COMMAND_PARAM) ;
878  *((int*) data) = psf_get_format_subtype_count () ;
879  return 0 ;
880 
881  case SFC_GET_FORMAT_SUBTYPE :
882  if (data == NULL || datasize != SIGNED_SIZEOF (SF_FORMAT_INFO))
883  return (sf_errno = SFE_BAD_COMMAND_PARAM) ;
884  return psf_get_format_subtype (data) ;
885 
886  case SFC_GET_FORMAT_INFO :
887  if (data == NULL || datasize != SIGNED_SIZEOF (SF_FORMAT_INFO))
888  return (sf_errno = SFE_BAD_COMMAND_PARAM) ;
889  return psf_get_format_info (data) ;
890  } ;
891

894  return (sf_errno = SFE_BAD_COMMAND_PARAM) ;

896  return strlen (data) ;

900 
901  switch (command)
902  { case SFC_SET_NORM_FLOAT :
903  old_value = psf->norm_float ;
904  psf->norm_float = (datasize) ? SF_TRUE : SF_FALSE ;
905  return old_value ;
906

908  if (data == NULL || datasize != SIGNED_SIZEOF (SF_INFO))
909  return (sf_errno = SFE_BAD_COMMAND_PARAM) ;
910  memcpy (data, &psf->sf, sizeof (SF_INFO)) ;
911  break ;
912

914  old_value = psf->norm_double ;
915  psf->norm_double = (datasize) ? SF_TRUE : SF_FALSE ;
916  return old_value ;
917 
918  case SFC_GET_NORM_FLOAT :
919  return psf->norm_float ;
920 
921  case SFC_GET_NORM_DOUBLE :
922  return psf->norm_double ;
923 
924  case SFC_SET_SCALE_FLOAT_INT_READ :
925  old_value = psf->float_int_mult ;
926 
927  psf->float_int_mult = (datasize != 0) ? SF_TRUE : SF_FALSE ;
928  if (psf->float_int_mult && psf->float_max < 0.0)
929  psf->float_max = psf_calc_signal_max (psf, SF_FALSE) ;
930  return old_value ;
931

933  old_value = psf->scale_int_float ;

943  case SF_FORMAT_CAF :
944  case SF_FORMAT_WAV :
945  case SF_FORMAT_WAVEX :

953

955  if (format != SF_FORMAT_FLOAT && format != SF_FORMAT_DOUBLE)
956  return SF_FALSE ;

958  } ;

961  return SF_FALSE ;

963  if (psf->have_written)

969  { free (psf->peak_info) ;

971  }
972  else if (psf->peak_info == NULL)
973  { psf->peak_info = peak_info_calloc (psf->sf.channels) ;

979  psf->write_header (psf, SF_TRUE) ;
980  return datasize ;

983  return SF_FALSE ;
984 
985  case SFC_GET_LOG_INFO :
986  if (data == NULL)
987  return SFE_BAD_COMMAND_PARAM ;
988  snprintf (data, datasize, "%s", psf->logbuffer) ;
989  break ;
990 
991  case SFC_CALC_SIGNAL_MAX :
992  if (data == NULL || datasize != sizeof (double))
993  return (psf->error = SFE_BAD_COMMAND_PARAM) ;
994  *((double*) data) = psf_calc_signal_max (psf, SF_FALSE) ;
995  break ;
996 
997  case SFC_CALC_NORM_SIGNAL_MAX :
998  if (data == NULL || datasize != sizeof (double))
999  return (psf->error = SFE_BAD_COMMAND_PARAM) ;
1000  *((double*) data) = psf_calc_signal_max (psf, SF_TRUE) ;
1001  break ;
1002 
1003  case SFC_CALC_MAX_ALL_CHANNELS :
1004  if (data == NULL || datasize != SIGNED_SIZEOF (double) * psf->sf.channels)
1005  return (psf->error = SFE_BAD_COMMAND_PARAM) ;
1006  return psf_calc_max_all_channels (psf, (double*) data, SF_FALSE) ;
1007 
1008  case SFC_CALC_NORM_MAX_ALL_CHANNELS :
1009  if (data == NULL || datasize != SIGNED_SIZEOF (double) * psf->sf.channels)
1010  return (psf->error = SFE_BAD_COMMAND_PARAM) ;
1011  return psf_calc_max_all_channels (psf, (double*) data, SF_TRUE) ;
1012 
1013  case SFC_GET_SIGNAL_MAX :
1014  if (data == NULL || datasize != sizeof (double))
1015  { psf->error = SFE_BAD_COMMAND_PARAM ;
1016  return SF_FALSE ;
1017  } ;
1018  return psf_get_signal_max (psf, (double *) data) ;
1019 
1020  case SFC_GET_MAX_ALL_CHANNELS :
1021  if (data == NULL || datasize != SIGNED_SIZEOF (double) * psf->sf.channels)
1022  { psf->error = SFE_BAD_COMMAND_PARAM ;
1023  return SF_FALSE ;
1024  } ;
1025  return psf_get_max_all_channels (psf, (double*) data) ;
1026 
1027  case SFC_UPDATE_HEADER_NOW :
1028  if (psf->write_header)
1029  psf->write_header (psf, SF_TRUE) ;
1030  break ;
1031 
1032  case SFC_SET_UPDATE_HEADER_AUTO :
1033  psf->auto_header = datasize ? SF_TRUE : SF_FALSE ;
1034  return psf->auto_header ;
1035  break ;
1036 
1037  case SFC_SET_ADD_DITHER_ON_WRITE :
1038  case SFC_SET_ADD_DITHER_ON_READ :
1039  /*
1040  ** FIXME !
1041  ** These are obsolete. Just return.
1042  ** Remove some time after version 1.0.8.
1043  */
1044  break ;
1045 
1046  case SFC_SET_DITHER_ON_WRITE :
1047  if (data == NULL || datasize != SIGNED_SIZEOF (SF_DITHER_INFO))
1048  return (psf->error = SFE_BAD_COMMAND_PARAM) ;
1049  memcpy (&psf->write_dither, data, sizeof (psf->write_dither)) ;
1050  if (psf->file.mode == SFM_WRITE || psf->file.mode == SFM_RDWR)
1051  dither_init (psf, SFM_WRITE) ;
1052  break ;
1053 
1054  case SFC_SET_DITHER_ON_READ :
1055  if (data == NULL || datasize != SIGNED_SIZEOF (SF_DITHER_INFO))
1056  return (psf->error = SFE_BAD_COMMAND_PARAM) ;
1057  memcpy (&psf->read_dither, data, sizeof (psf->read_dither)) ;
1058  if (psf->file.mode == SFM_READ || psf->file.mode == SFM_RDWR)
1059  dither_init (psf, SFM_READ) ;
1060  break ;
1061 
1062  case SFC_FILE_TRUNCATE :
1063  if (psf->file.mode != SFM_WRITE && psf->file.mode != SFM_RDWR)
1064  return SF_TRUE ;
1065  if (datasize != sizeof (sf_count_t))
1066  return SF_TRUE ;
1067  if (data == NULL || datasize != sizeof (sf_count_t))
1068  { psf->error = SFE_BAD_COMMAND_PARAM ;
1069  return SF_FALSE ;
1070  }
1071  else
1072  { sf_count_t position ;
1073 
1074  position = *((sf_count_t*) data) ;
1075 
1076  if (sf_seek (sndfile, position, SEEK_SET) != position)
1077  return SF_TRUE ;
1078

1273 sf_count_t
1274 sf_seek (SNDFILE *sndfile, sf_count_t offset, int whence)
1275 { SF_PRIVATE *psf ;
1276  sf_count_t seek_from_start = 0, retval ;
1277 
1278  VALIDATE_SNDFILE_AND_ASSIGN_PSF (sndfile, psf, 1) ;
1279 
1280  if (! psf->sf.seekable)
1281  { psf->error = SFE_NOT_SEEKABLE ;
1282  return PSF_SEEK_ERROR ;
1283  } ;
1284 
1285  /* If the whence parameter has a mode ORed in, check to see that
1286  ** it makes sense.
1287  */
1288  if (((whence & SFM_MASK) == SFM_WRITE && psf->file.mode == SFM_READ) ||
1289  ((whence & SFM_MASK) == SFM_READ && psf->file.mode == SFM_WRITE))
1290  { psf->error = SFE_WRONG_SEEK ;
1291  return PSF_SEEK_ERROR ;
1292  } ;
1293 
1294  /* Convert all SEEK_CUR and SEEK_END into seek_from_start to be
1295  ** used with SEEK_SET.
1296  */
1297  switch (whence)
1298  { /* The SEEK_SET behaviour is independant of mode. */
1299  case SEEK_SET :
1300  case SEEK_SET | SFM_READ :
1301  case SEEK_SET | SFM_WRITE :
1302  case SEEK_SET | SFM_RDWR :
1303  seek_from_start = offset ;
1304  break ;
1305 
1306  /* The SEEK_CUR is a little more tricky. */
1307  case SEEK_CUR :
1308  if (offset == 0)
1309  { if (psf->file.mode == SFM_READ)
1310  return psf->read_current ;
1311  if (psf->file.mode == SFM_WRITE)
1312  return psf->write_current ;
1313  } ;
1314  if (psf->file.mode == SFM_READ)
1315  seek_from_start = psf->read_current + offset ;
1316  else if (psf->file.mode == SFM_WRITE || psf->file.mode == SFM_RDWR)
1317  seek_from_start = psf->write_current + offset ;
1318  else
1319  psf->error = SFE_AMBIGUOUS_SEEK ;
1320  break ;
1321 
1322  case SEEK_CUR | SFM_READ :
1323  if (offset == 0)
1324  return psf->read_current ;
1325  seek_from_start = psf->read_current + offset ;
1326  break ;
1327 
1328  case SEEK_CUR | SFM_WRITE :
1329  if (offset == 0)
1330  return psf->write_current ;
1331  seek_from_start = psf->write_current + offset ;
1332  break ;
1333 
1334  /* The SEEK_END */
1335  case SEEK_END :
1336  case SEEK_END | SFM_READ :
1337  case SEEK_END | SFM_WRITE :
1338  seek_from_start = psf->sf.frames + offset ;
1339  break ;
1340 
1341  default :
1342  psf->error = SFE_BAD_SEEK ;
1343  break ;
1344  } ;
1345 
1346  if (psf->error)
1347  return PSF_SEEK_ERROR ;
1348 
1349  if (psf->file.mode == SFM_RDWR || psf->file.mode == SFM_WRITE)
1350  { if (seek_from_start < 0)
1351  { psf->error = SFE_BAD_SEEK ;
1352  return PSF_SEEK_ERROR ;
1353  } ;
1354  }
1355  else if (seek_from_start < 0 || seek_from_start > psf->sf.frames)
1356  { psf->error = SFE_BAD_SEEK ;
1357  return PSF_SEEK_ERROR ;
1358  } ;
1359 
1360  if (psf->seek)
1361  { int new_mode = (whence & SFM_MASK) ? (whence & SFM_MASK) : psf->file.mode ;
1362 
1363  retval = psf->seek (psf, new_mode, seek_from_start) ;
1364 
1365  switch (new_mode)
1366  { case SFM_READ :
1367  psf->read_current = retval ;
1368  break ;
1369  case SFM_WRITE :
1370  psf->write_current = retval ;
1371  break ;
1372  case SFM_RDWR :
1373  psf->read_current = retval ;
1374  psf->write_current = retval ;
1375  new_mode = SFM_READ ;
1376  break ;
1377  } ;
1378 
1379  psf->last_op = new_mode ;
1380 
1381  return retval ;
1382  } ;
1383 
1384  psf->error = SFE_AMBIGUOUS_SEEK ;
1385  return PSF_SEEK_ERROR ;
1386 } /* sf_seek */
