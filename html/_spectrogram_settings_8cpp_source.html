185 bool SpectrogramSettings::Validate(bool quiet)
186 {
187  if (!quiet &&
188  maxFreq < 100) {
189  wxMessageBox(_("Maximum frequency must be 100 Hz or above"));
190  return false;
191  }
192  else
193  maxFreq = std::max(100, maxFreq);
194 
195  if (!quiet &&
196  minFreq < 0) {
197  wxMessageBox(_("Minimum frequency must be at least 0 Hz"));
198  return false;
199  }
200  else
201  minFreq = std::max(0, minFreq);
202 
203  if (!quiet &&
204  maxFreq <= minFreq) {
205  wxMessageBox(_("Minimum frequency must be less than maximum frequency"));
206  return false;
207  }
208  else
209  maxFreq = std::max(1 + minFreq, maxFreq);
210 
211  if (!quiet &&
212  range <= 0) {
213  wxMessageBox(_("The range must be at least 1 dB"));
214  return false;
215  }
216  else
217  range = std::max(1, range);
218 
219  if (!quiet &&
220  frequencyGain < 0) {
221  wxMessageBox(_("The frequency gain cannot be negative"));
222  return false;
223  }
224  else if (!quiet &&
225  frequencyGain > 60) {
226  wxMessageBox(_("The frequency gain must be no more than 60 dB/dec"));
227  return false;
228  }
229  else
230  frequencyGain =
231  std::max(0, std::min(60, frequencyGain));
232 
233  // The rest are controlled by drop-down menus so they can't go wrong
234  // in the Preferences dialog, but we also come here after reading fom saved
235  // preference files, which could be or from future versions. Validate quietly.
236  windowType =
237  std::max(0, std::min(NumWindowFuncs() - 1, windowType));
238  scaleType =
239  ScaleType(std::max(0,
240  std::min(int(SpectrogramSettings::stNumScaleTypes) - 1,
241  int(scaleType))));
242  algorithm = Algorithm(
243  std::max(0, std::min(int(algNumAlgorithms) - 1, int(algorithm)))
244  );
245  ConvertToEnumeratedWindowSizes();
246  ConvertToActualWindowSizes();
247 
248  return true;
249 }

374  enum { WINDOW, TWINDOW, DWINDOW };
375  void RecreateWindow(
376  float *&window, int which, int fftLen,
377  int padding, int windowType, int windowSize, double &scale)
378  {
379  if (window != NULL)
380  delete[] window;
381  // Create the requested window function
382  window = new float[fftLen];
383  int ii;
384 
385  const bool extra = padding > 0;
386  wxASSERT(windowSize % 2 == 0);
387  if (extra)
388  // For windows that do not go to 0 at the edges, this improves symmetry
389  ++windowSize;
390  const int endOfWindow = padding + windowSize;
391  // Left and right padding
392  for (ii = 0; ii < padding; ++ii) {
393  window[ii] = 0.0;
394  window[fftLen - ii - 1] = 0.0;
395  }
396  // Default rectangular window in the middle
397  for (; ii < endOfWindow; ++ii)
398  window[ii] = 1.0;
399  // Overwrite middle as needed
400  switch (which) {
401  case WINDOW:
402  NewWindowFunc(windowType, windowSize, extra, window + padding);
403  break;
404  // Future, reassignment
405  case TWINDOW:
406  NewWindowFunc(windowType, windowSize, extra, window + padding);
407  for (int ii = padding, multiplier = -windowSize / 2; ii < endOfWindow; ++ii, ++multiplier)
408  window[ii] *= multiplier;
409  break;
410  case DWINDOW:
411  DerivativeOfWindowFunc(windowType, windowSize, extra, window + padding);
412  break;
413  default:
414  wxASSERT(false);
415  }
416  // Scale the window function to give 0dB spectrum for 0dB sine tone
417  if (which == WINDOW) {
418  scale = 0.0;
419  for (ii = padding; ii < endOfWindow; ++ii)
420  scale += window[ii];
421  if (scale > 0)
422  scale = 2.0 / scale;
423  }
424  for (ii = padding; ii < endOfWindow; ++ii)
425  window[ii] *= scale;
426  }
427 }

450 void SpectrogramSettings::ConvertToEnumeratedWindowSizes()
451 {
452  unsigned size;
453  int logarithm;
454 
455  logarithm = -LogMinWindowSize;
456  size = unsigned(windowSize);
457  while (size > 1)
458  size >>= 1, ++logarithm;
459  windowSize = std::max(0, std::min(NumWindowSizes - 1, logarithm));
460 
461 #ifdef EXPERIMENTAL_ZERO_PADDED_SPECTROGRAMS
462  // Choices for zero padding begin at 1
463  logarithm = 0;
464  size = unsigned(zeroPaddingFactor);
465  while (zeroPaddingFactor > 1)
466  zeroPaddingFactor >>= 1, ++logarithm;
467  zeroPaddingFactor = std::max(0,
468  std::min(LogMaxWindowSize - (windowSize + LogMinWindowSize),
469  logarithm
470  ));
471 #endif
472 }
