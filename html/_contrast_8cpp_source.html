63  float rms = float(0.0);

65 
66  AudacityProject *p = GetActiveProject();
67  SelectedTrackListOfKindIterator iter(Track::Wave, p->GetTracks());
68  WaveTrack *t = (WaveTrack *) iter.First();

74  wxMessageDialog m(NULL, _("You can only measure one track at a time."), _("Error"), wxOK);
75  m.ShowModal();
76  return false;
77  }
78 
79  if(mT0 > mT1)
80  {
81  wxMessageDialog m(NULL, _("Start time after end time!\nPlease enter reasonable times."), _("Error"), wxOK);
82  m.ShowModal();
83  return false;
84  }
85 
86  // FIXME: See bugs 1361 and 1362
87  if(mT0 < t->GetStartTime())
88  mT0 = t->GetStartTime();
89  if(mT1 > t->GetEndTime())
90  mT1 = t->GetEndTime();
91 
92  wxLogDebug(wxT("mT0 = %.20f mT1 = %.10f"),mT0 ,mT1);
93 
94  // TODO: Let's use the actual selected samples
95  sampleCount SelT0 = t->TimeToLongSamples(mT0);
96  sampleCount SelT1 = t->TimeToLongSamples(mT1);
97 
98  if(SelT0 > SelT1)
99  {
100  // FIXME: Bad error message. This should never happen.
101  wxMessageDialog m(NULL, _("Times are not reasonable!\nPlease enter reasonable times."), _("Error"), wxOK);
102  m.ShowModal();
103  return false;
104  }
105  if(SelT0 == SelT1)
106  {
107  wxMessageDialog m(NULL, _("Nothing to measure.\nPlease select a section of a track."), _("Error"), wxOK);
108  m.ShowModal();
109  return false;
110  }
111 
112  ((WaveTrack *)t)->GetRMS(&rms, mT0, mT1);
113  t = (WaveTrack *) iter.Next();
114  }
115 
116  if(numberSelecteTracks == 0) {
117  wxMessageDialog m(NULL, _("Please select an audio track."), _("Error"), wxOK);
118  m.ShowModal();
119  return false;
120  }
121

443  mBackgroundRMSText->ChangeValue(wxString::Format(wxT("")));

447  /* i18n-hint: i.e. difference in loudness at the moment. */
448  mDiffText->SetName(_("Current difference"));
449  float diffdB = foregrounddB - backgrounddB;
450  if(diffdB > WCAG2_PASS) {
451  mPassFailText->ChangeValue(_("WCAG2 Pass"));
452  }
453  else {
454  mPassFailText->ChangeValue(_("WCAG2 Fail"));
455  }

459  mDiffText->ChangeValue(wxString::Format(_("indeterminate")));
460  }
461  else {
462  if( fabs(diffdB) != std::numeric_limits<float>::infinity() ) {
463  mDiffText->ChangeValue(wxString::Format(_("%.1f dB Average RMS"), diffdB));
464  }
465  else {
466  mDiffText->ChangeValue(wxString::Format(_("infinite dB difference")));
467  }
468  }
469  }
470  else {
471  mPassFailText->SetName(wxT(""));
