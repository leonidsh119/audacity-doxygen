339  }

344  "does not match result count %d\n",
345  TUP_FMT_ARGS(subpat), count, num_sub_results);
346  }
347 
348  last_subject = id[0];
349  }
350  fprintf(stderr, "OK\n\n");
351  sord_iter_free(iter);
352 
353  return ret;
