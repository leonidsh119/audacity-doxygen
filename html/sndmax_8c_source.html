30  /* BE CAREFUL - DO NOT ALLOW GC TO RUN WHILE LVAL IS UNPROTECTED */
31  if (exttypep(s_as_lval, a_sound)) {
32  /* if snd_expr was simply a symbol, then s now points to
33  a shared sound_node. If we read samples from it, then
34  the sound bound to the symbol will be destroyed, so
35  copy it first. If snd_expr was a real expression that
36  computed a new value, then the next garbage collection
37  will reclaim the sound_node. We need to make the new
38  sound reachable by the garbage collector to that any
39  lisp data reachable from the sound do not get collected.
40  To make the sound reachable, we need to allocate a node,
41  and the GC might run, so we need to protect the OLD s
42  but then make it unreachable.
43  We will let the GC collect the sound in the end.
44  */
45  xlprot1(s_as_lval);

47  s_as_lval = cvsound(s); /* destroys only ref. to original */
