255 bool EffectAutoDuck::Process()
256 {
257  sampleCount i;
258 
259  if (GetNumWaveTracks() == 0 || !mControlTrack)
260  return false;
261 
262  bool cancel = false;
263 
264  sampleCount start =
265  mControlTrack->TimeToLongSamples(mT0 + mOuterFadeDownLen);
266  sampleCount end =
267  mControlTrack->TimeToLongSamples(mT1 - mOuterFadeUpLen);
268 
269  if (end <= start)
270  return false;
271 
272  // the minimum number of samples we have to wait until the maximum
273  // pause has been exceeded
274  double maxPause = mMaximumPause;
275 
276  // We don't fade in until we have time enough to actually fade out again
277  if (maxPause < mOuterFadeDownLen + mOuterFadeUpLen)
278  maxPause = mOuterFadeDownLen + mOuterFadeUpLen;
279 
280  sampleCount minSamplesPause =
281  mControlTrack->TimeToLongSamples(maxPause);
282 
283  double threshold = DB_TO_LINEAR(mThresholdDb);
284 
285  // adjust the threshold so we can compare it to the rmsSum value
286  threshold = threshold * threshold * kRMSWindowSize;
287 
288  int rmsPos = 0;
289  float rmsSum = 0;
290  float *rmsWindow = new float[kRMSWindowSize];
291  for (i = 0; i < kRMSWindowSize; i++)
292  rmsWindow[i] = 0;
293 
294  float *buf = new float[kBufSize];
295 
296  bool inDuckRegion = false;
297 
298  // initialize the following two variables to prevent compiler warning
299  double duckRegionStart = 0;
300  sampleCount curSamplesPause = 0;
301 
302  // to make the progress bar appear more natural, we first look for all
303  // duck regions and apply them all at once afterwards
304  AutoDuckRegionArray regions;
305  sampleCount pos = start;
306 
307  while (pos < end)
308  {
309  sampleCount len = end - pos;
310  if (len > kBufSize)
311  len = kBufSize;
312 
313  mControlTrack->Get((samplePtr)buf, floatSample, pos, (sampleCount)len);
314 
315  for (i = pos; i < pos + len; i++)
316  {
317  rmsSum -= rmsWindow[rmsPos];
318  rmsWindow[rmsPos] = buf[i - pos] * buf[i - pos];
319  rmsSum += rmsWindow[rmsPos];
320  rmsPos = (rmsPos + 1) % kRMSWindowSize;
321 
322  bool thresholdExceeded = rmsSum > threshold;
323 
324  if (thresholdExceeded)
325  {
326  // everytime the threshold is exceeded, reset our count for
327  // the number of pause samples
328  curSamplesPause = 0;
329 
330  if (!inDuckRegion)
331  {
332  // the threshold has been exceeded for the first time, so
333  // let the duck region begin here
334  inDuckRegion = true;
335  duckRegionStart = mControlTrack->LongSamplesToTime(i);
336  }
337  }
338 
339  if (!thresholdExceeded && inDuckRegion)
340  {
341  // the threshold has not been exceeded and we are in a duck
342  // region, but only fade in if the maximum pause has been
343  // exceeded
344  curSamplesPause += 1;
345 
346  if (curSamplesPause >= minSamplesPause)
347  {
348  // do the actual duck fade and reset all values
349  double duckRegionEnd =
350  mControlTrack->LongSamplesToTime(i - curSamplesPause);
351 
352  regions.Add(AutoDuckRegion(
353  duckRegionStart - mOuterFadeDownLen,
354  duckRegionEnd + mOuterFadeUpLen));
355 
356  inDuckRegion = false;
357  }
358  }
359  }
360 
361  pos += len;
362 
363  if (TotalProgress( ((double)(pos-start)) / (end-start) /
364  (GetNumWaveTracks() + 1) ))
365  {
366  cancel = true;
367  break;
368  }
369  }
370 
371  // apply last duck fade, if any
372  if (inDuckRegion)
373  {
374  double duckRegionEnd =
375  mControlTrack->LongSamplesToTime(end - curSamplesPause);
376  regions.Add(AutoDuckRegion(
377  duckRegionStart - mOuterFadeDownLen,
378  duckRegionEnd + mOuterFadeUpLen));
379  }
380 
381  delete[] buf;
382  delete[] rmsWindow;
383 
384  if (!cancel)
385  {
386  CopyInputTracks(); // Set up mOutputTracks.
387  SelectedTrackListOfKindIterator iter(Track::Wave, mOutputTracks);
388  Track *iterTrack = iter.First();
389 
390  int trackNumber = 0;
391 
392  while (iterTrack)
393  {
394  wxASSERT(iterTrack->GetKind() == Track::Wave);
395 
396  WaveTrack* t = (WaveTrack*)iterTrack;
397 
398  for (i = 0; i < (int)regions.GetCount(); i++)
399  {
400  const AutoDuckRegion& region = regions[i];
401  if (ApplyDuckFade(trackNumber, t, region.t0, region.t1))
402  {
403  cancel = true;
404  break;
405  }
406  }
407 
408  if (cancel)
409  break;
410 
411  iterTrack = iter.Next();
412  trackNumber++;
413  }
414  }
415 
416  ReplaceProcessedTracks(!cancel);
417  return !cancel;
418 }

512 // this currently does an exponential fade
513 bool EffectAutoDuck::ApplyDuckFade(int trackNumber, WaveTrack* t,
514  double t0, double t1)
515 {
516  bool cancel = false;
517 
518  sampleCount start = t->TimeToLongSamples(t0);
519  sampleCount end = t->TimeToLongSamples(t1);
520 
521  float *buf = new float[kBufSize];
522  sampleCount pos = start;
523 
524  int fadeDownSamples = t->TimeToLongSamples(
525  mOuterFadeDownLen + mInnerFadeDownLen);
526  if (fadeDownSamples < 1)
527  fadeDownSamples = 1;
528 
529  int fadeUpSamples = t->TimeToLongSamples(
530  mOuterFadeUpLen + mInnerFadeUpLen);
531  if (fadeUpSamples < 1)
532  fadeUpSamples = 1;
533 
534  float fadeDownStep = mDuckAmountDb / fadeDownSamples;
535  float fadeUpStep = mDuckAmountDb / fadeUpSamples;
536 
537  while (pos < end)
538  {
539  sampleCount len = end - pos;
540  if (len > kBufSize)
541  len = kBufSize;
542 
543  t->Get((samplePtr)buf, floatSample, pos, len);
544 
545  for (sampleCount i = pos; i < pos + len; i++)
546  {
547  float gainDown = fadeDownStep * (i - start);
548  float gainUp = fadeUpStep * (end - i);;
549 
550  float gain;
551  if (gainDown > gainUp)
552  gain = gainDown;
553  else
554  gain = gainUp;
555  if (gain < mDuckAmountDb)
556  gain = mDuckAmountDb;
557 
558  buf[i - pos] *= DB_TO_LINEAR(gain);
559  }
560 
561  t->Set((samplePtr)buf, floatSample, pos, len);
562 
563  pos += len;
564 
565  float curTime = t->LongSamplesToTime(pos);
566  float fractionFinished = (curTime - mT0) / (mT1 - mT0);
567  if (TotalProgress( (trackNumber + 1 + fractionFinished) /
568  (GetNumWaveTracks() + 1) ))
569  {
570  cancel = true;
571  break;
572  }
573  }
574 
575  delete[] buf;
576  return cancel;
577 }
