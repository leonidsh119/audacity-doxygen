92  GCO DllGetClassObject;
93  IClassFactory *pcf = NULL;
94  IKsPropertySet *pps = NULL;
95  HRESULT hr;
96  ULONG bytes;
97  LPGUID guidIn;
98  LPGUID guidOut;
99  UINT deviceIn = -1;
100  UINT deviceOut = -1;
101  int ret = FALSE;
102 
103  guidIn = PaWinDS_GetStreamInputGUID(Px->pa_stream);
104  guidOut = PaWinDS_GetStreamOutputGUID(Px->pa_stream);
105 
106  do {
107  hDsound = LoadLibraryA("dsound.dll");
108  if (hDsound == NULL) {
109  break;
110  }
111 
112  DllGetClassObject = (GCO) GetProcAddress(hDsound, "DllGetClassObject");
113  if (DllGetClassObject == NULL) {
114  break;
115  }
116 
117  hr = DllGetClassObject(&CLSID_DirectSoundPrivate,
118  &IID_IClassFactory,
119  (void **)(&pcf));
120  if (hr || pcf == NULL) {
121  break;
122  }
123 
124  hr = IClassFactory_CreateInstance(pcf,
125  NULL,
126  &IID_IKsPropertySet,
127  (void **)(&pps));
128  if (hr || pps == NULL) {
129  break;
130  }
131 
132  if (guidIn) {
133  memset(&desc, 0, sizeof(desc));
134  memcpy(&desc.DeviceId, guidIn, sizeof(desc.DeviceId));
135 
136  desc.WaveDeviceId = kImpossibleWaveID;
137  desc.DataFlow = DIRECTSOUNDDEVICE_DATAFLOW_CAPTURE;
138 
139  hr = IKsPropertySet_Get(pps,
140  &DSPROPSETID_DirectSoundDevice,
141  DSPROPERTY_DIRECTSOUNDDEVICE_DESCRIPTION,
142  NULL,
143  0,
144  &desc,
145  sizeof(desc),
146  &bytes);
147  if (FAILED(hr) || kImpossibleWaveID == desc.WaveDeviceId) {
148  break;
149  }
150  deviceIn = desc.WaveDeviceId;
151  }
152 
153  if (guidOut) {
154  memset(&desc, 0, sizeof(desc));
155  memcpy(&desc.DeviceId, guidOut, sizeof(desc.DeviceId));
156 
157  desc.WaveDeviceId = kImpossibleWaveID;
158  desc.DataFlow = DIRECTSOUNDDEVICE_DATAFLOW_RENDER;
159 
160  hr = IKsPropertySet_Get(pps,
161  &DSPROPSETID_DirectSoundDevice,
162  DSPROPERTY_DIRECTSOUNDDEVICE_DESCRIPTION,
163  NULL,
164  0,
165  &desc,
166  sizeof(desc),
167  &bytes);
168  if (FAILED(hr) || kImpossibleWaveID == desc.WaveDeviceId) {
169  break;
170  }
171  deviceOut = desc.WaveDeviceId;
172  }
173 
174  if (open_mixers(Px, deviceIn, deviceOut))
175  {
176  ret = TRUE;
177  }
178  } while( FALSE );
179 
180  if (pps) {
181  IUnknown_Release(pps);
182  }
183 
184  if (pcf) {
185  IUnknown_Release(pcf);
186  }
187

189  // so this probably doesn't do anything until Pa_Terminate is called.
190  if (hDsound != INVALID_HANDLE_VALUE) {
191  FreeLibrary(hDsound);
192  }
193 
194  return ret;
195 }
