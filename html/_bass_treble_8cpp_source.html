118 bool EffectBassTreble::ProcessInitialize(sampleCount WXUNUSED(totalLen), ChannelNames WXUNUSED(chanMap))
119 {
120  if (GetPass() == 1)
121  {
122  const float slope = 0.4f; // same slope for both filters
123  const double hzBass = 250.0f;
124  const double hzTreble = 4000.0f;
125 
126  //(re)initialise filter parameters
127  xn1Bass=xn2Bass=yn1Bass=yn2Bass=0;
128  xn1Treble=xn2Treble=yn1Treble=yn2Treble=0;
129 
130  // Compute coefficents of the low shelf biquand IIR filter

144 sampleCount EffectBassTreble::ProcessBlock(float **inBlock, float **outBlock, sampleCount blockLen)
145 {
146  float *ibuf = inBlock[0];
147  float *obuf = outBlock[0];
148 
149  if (GetPass() == 1)
150  {
151  for (sampleCount i = 0; i < blockLen; i++)
152  {
153  obuf[i] = DoFilter(ibuf[i]) / mPreGain;
154  }
155  }
156  else
157  {
158  float gain = DB_TO_LINEAR(dB_level) / mMax;
159  for (sampleCount i = 0; i < blockLen; i++)
160  {
161  // Normalize to specified level
162  obuf[i] = ibuf[i] * (mPreGain * gain);
163  }
164  }
165 
166  return blockLen;
167 }

231 bool EffectBassTreble::InitPass1()
232 {
233  mMax = 0.0;
234 
235  // Integer format tracks require headroom to avoid clipping
236  // when saved between passes (bug 619)
237 
238  if (mbNormalize) // don't need to calculate this if only doing one pass.
239  {
240  // Up to (gain + 6dB) headroom required for treble boost (experimental).
241  mPreGain = (dB_treble > 0) ? (dB_treble + 6.0) : 0.0;
242  if (dB_bass >= 0)
243  {
244  mPreGain = (mPreGain > dB_bass) ? mPreGain : dB_bass;
245  }
246  else
247  {
248  // Up to 6 dB headroom reaquired for bass cut (experimental)
249  mPreGain = (mPreGain > 6.0) ? mPreGain : 6.0;
250  }
251  mPreGain = (exp(log(10.0) * mPreGain / 20)); // to linear
252  }
253  else
254  {
255  mPreGain = 1.0; // Unity gain
256  }
257 
258  return true;
259 }

354 void EffectBassTreble::Coefficents(double hz, float slope, double gain, int type,
355  float& a0, float& a1, float& a2,
356  float& b0, float& b1, float& b2)
357 {
358  double w = 2 * M_PI * hz / mSampleRate;
359  double a = exp(log(10.0) * gain / 40);
360  double b = sqrt((a * a + 1) / slope - (pow((a - 1), 2)));
361 
362  if (type == kBass)
363  {
364  b0 = a * ((a + 1) - (a - 1) * cos(w) + b * sin(w));
365  b1 = 2 * a * ((a - 1) - (a + 1) * cos(w));
366  b2 = a * ((a + 1) - (a - 1) * cos(w) - b * sin(w));
367  a0 = ((a + 1) + (a - 1) * cos(w) + b * sin(w));
368  a1 = -2 * ((a - 1) + (a + 1) * cos(w));
369  a2 = (a + 1) + (a - 1) * cos(w) - b * sin(w);
370  }
371  else //assumed kTreble
372  {
373  b0 = a * ((a + 1) + (a - 1) * cos(w) + b * sin(w));
374  b1 = -2 * a * ((a - 1) + (a + 1) * cos(w));
375  b2 = a * ((a + 1) + (a - 1) * cos(w) - b * sin(w));
376  a0 = ((a + 1) - (a - 1) * cos(w) + b * sin(w));
377  a1 = 2 * ((a - 1) - (a + 1) * cos(w));
378  a2 = (a + 1) - (a - 1) * cos(w) - b * sin(w);
379  }
380 }

382 float EffectBassTreble::DoFilter(float in)
383 {
384  // Bass filter
385  float out = (b0Bass * in + b1Bass * xn1Bass + b2Bass * xn2Bass -
386  a1Bass * yn1Bass - a2Bass * yn2Bass) / a0Bass;
387  xn2Bass = xn1Bass;
388  xn1Bass = in;
389  yn2Bass = yn1Bass;
390  yn1Bass = out;
391 
392  // Treble filter
393  in = out;
394  out = (b0Treble * in + b1Treble * xn1Treble + b2Treble * xn2Treble -
395  a1Treble * yn1Treble - a2Treble * yn2Treble) / a0Treble;
396  xn2Treble = xn1Treble;
397  xn1Treble = in;
398  yn2Treble = yn1Treble;
399  yn1Treble = out;
400 
401  // Retain the maximum value for use in the normalization pass
402  if(mMax < fabs(out))
403  mMax = fabs(out);
404  return out;
405 }
