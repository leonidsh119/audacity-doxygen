49 bool EffectSoundTouch::Process()
50 {
51  // Assumes that mSoundTouch has already been initialized
52  // by the subclass for subclass-specific parameters. The
53  // time warper should also be set.
54 
55  // Check if this effect will alter the selection length; if so, we need
56  // to operate on sync-lock selected tracks.
57  bool mustSync = true;
58  if (mT1 == GetTimeWarper()->Warp(mT1)) {
59  mustSync = false;
60  }
61 
62  //Iterate over each track
63  // Needs Track::All for sync-lock grouping.
64  this->CopyInputTracks(Track::All);
65  bool bGoodResult = true;
66 
67  TrackListIterator iter(mOutputTracks);
68  Track* t;
69  mCurTrackNum = 0;
70  m_maxNewLength = 0.0;
71 
72  t = iter.First();
73  while (t != NULL) {
74  if (t->GetKind() == Track::Label &&
75  (t->GetSelected() || (mustSync && t->IsSyncLockSelected())) )
76  {
77  if (!ProcessLabelTrack(t))
78  {
79  bGoodResult = false;
80  break;
81  }
82  }
83 #ifdef USE_MIDI
84  else if (t->GetKind() == Track::Note &&
85  (t->GetSelected() || (mustSync && t->IsSyncLockSelected())))
86  {
87  if (!ProcessNoteTrack(t))
88  {
89  bGoodResult = false;
90  break;
91  }
92  }
93 #endif
94  else if (t->GetKind() == Track::Wave && t->GetSelected())
95  {
96  WaveTrack* leftTrack = (WaveTrack*)t;
97  //Get start and end times from track
98  mCurT0 = leftTrack->GetStartTime();
99  mCurT1 = leftTrack->GetEndTime();
100 
101  //Set the current bounds to whichever left marker is
102  //greater and whichever right marker is less
103  mCurT0 = wxMax(mT0, mCurT0);
104  mCurT1 = wxMin(mT1, mCurT1);
105 
106  // Process only if the right marker is to the right of the left marker
107  if (mCurT1 > mCurT0) {
108  sampleCount start, end;
109 
110  if (leftTrack->GetLinked()) {
111  double t;
112  WaveTrack* rightTrack = (WaveTrack*)(iter.Next());
113 
114  //Adjust bounds by the right tracks markers
115  t = rightTrack->GetStartTime();
116  t = wxMax(mT0, t);
117  mCurT0 = wxMin(mCurT0, t);
118  t = rightTrack->GetEndTime();
119  t = wxMin(mT1, t);
120  mCurT1 = wxMax(mCurT1, t);
121 
122  //Transform the marker timepoints to samples
123  start = leftTrack->TimeToLongSamples(mCurT0);
124  end = leftTrack->TimeToLongSamples(mCurT1);
125 
126  //Inform soundtouch there's 2 channels
127  mSoundTouch->setChannels(2);
128 
129  //ProcessStereo() (implemented below) processes a stereo track
130  if (!ProcessStereo(leftTrack, rightTrack, start, end))
131  {
132  bGoodResult = false;
133  break;
134  }
135  mCurTrackNum++; // Increment for rightTrack, too.
136  } else {
137  //Transform the marker timepoints to samples
138  start = leftTrack->TimeToLongSamples(mCurT0);
139  end = leftTrack->TimeToLongSamples(mCurT1);
140 
141  //Inform soundtouch there's a single channel
142  mSoundTouch->setChannels(1);
143 
144  //ProcessOne() (implemented below) processes a single track
145  if (!ProcessOne(leftTrack, start, end))
146  {
147  bGoodResult = false;
148  break;
149  }
150  }
151  }
152  mCurTrackNum++;
153  }
154  else if (mustSync && t->IsSyncLockSelected()) {
155  t->SyncLockAdjust(mT1, GetTimeWarper()->Warp(mT1));
156  }
157 
158  //Iterate to the next track
159  t = iter.Next();
160  }
161 
162  if (bGoodResult)
163  ReplaceProcessedTracks(bGoodResult);
164 
165  delete mSoundTouch;
166  mSoundTouch = NULL;
167 
168 // mT0 = mCurT0;
169 // mT1 = mCurT0 + m_maxNewLength; // Update selection.
170 
171  return bGoodResult;
172 }

174 //ProcessOne() takes a track, transforms it to bunch of buffer-blocks,
175 //and executes ProcessSoundTouch on these blocks
176 bool EffectSoundTouch::ProcessOne(WaveTrack *track,
177  sampleCount start, sampleCount end)
178 {
179  WaveTrack *outputTrack;
180  sampleCount s;
181 
182  mSoundTouch->setSampleRate((unsigned int)(track->GetRate()+0.5));
183 
184  outputTrack = mFactory->NewWaveTrack(track->GetSampleFormat(), track->GetRate());
185 
186  //Get the length of the buffer (as double). len is
187  //used simple to calculate a progress meter, so it is easier
188  //to make it a double now than it is to do it later
189  double len = (double)(end - start);
190 
191  //Initiate a processing buffer. This buffer will (most likely)
192  //be shorter than the length of the track being processed.
193  float *buffer = new float[track->GetMaxBlockSize()];
194 
195  //Go through the track one buffer at a time. s counts which
196  //sample the current buffer starts at.
197  s = start;
198  while (s < end) {
199  //Get a block of samples (smaller than the size of the buffer)
200  sampleCount block = track->GetBestBlockSize(s);
201 
202  //Adjust the block size if it is the final block in the track
203  if (s + block > end)
204  block = end - s;
205 
206  //Get the samples from the track and put them in the buffer
207  track->Get((samplePtr) buffer, floatSample, s, block);
208 
209  //Add samples to SoundTouch
210  mSoundTouch->putSamples(buffer, block);
211 
212  //Get back samples from SoundTouch
213  unsigned int outputCount = mSoundTouch->numSamples();
214  if (outputCount > 0) {
215  float *buffer2 = new float[outputCount];
216  mSoundTouch->receiveSamples(buffer2, outputCount);
217  outputTrack->Append((samplePtr)buffer2, floatSample, outputCount);
218  delete[] buffer2;
219  }
220 
221  //Increment s one blockfull of samples
222  s += block;
223 
224  //Update the Progress meter
225  if (TrackProgress(mCurTrackNum, (s - start) / len))
226  return false;
227  }
228 
229  // Tell SoundTouch to finish processing any remaining samples
230  mSoundTouch->flush(); // this should only be used for changeTempo - it dumps data otherwise with pRateTransposer->clear();
231 
232  unsigned int outputCount = mSoundTouch->numSamples();
233  if (outputCount > 0) {
234  float *buffer2 = new float[outputCount];
235  mSoundTouch->receiveSamples(buffer2, outputCount);
236  outputTrack->Append((samplePtr)buffer2, floatSample, outputCount);
237  delete[] buffer2;
238  }
239 
240  // Flush the output WaveTrack (since it's buffered, too)
241  outputTrack->Flush();
242 
243  // Clean up the buffer
244  delete[]buffer;
245 
246  // Take the output track and insert it in place of the original
247  // sample data
248  track->ClearAndPaste(mCurT0, mCurT1, outputTrack, true, false, GetTimeWarper());
249 
250  double newLength = outputTrack->GetEndTime();
251  m_maxNewLength = wxMax(m_maxNewLength, newLength);
252 
253  // Delete the outputTrack now that its data is inserted in place
254  delete outputTrack;
255 
256  //Return true because the effect processing succeeded.
257  return true;
258 }

260 bool EffectSoundTouch::ProcessStereo(WaveTrack* leftTrack, WaveTrack* rightTrack,
261  sampleCount start, sampleCount end)
262 {
263  mSoundTouch->setSampleRate((unsigned int)(leftTrack->GetRate()+0.5));
264 
265  WaveTrack* outputLeftTrack = mFactory->NewWaveTrack(leftTrack->GetSampleFormat(),
266  leftTrack->GetRate());
267  WaveTrack* outputRightTrack = mFactory->NewWaveTrack(rightTrack->GetSampleFormat(),
268  rightTrack->GetRate());
269 
270  //Get the length of the buffer (as double). len is
271  //used simple to calculate a progress meter, so it is easier
272  //to make it a double now than it is to do it later
273  double len = (double)(end - start);
274 
275  //Initiate a processing buffer. This buffer will (most likely)
276  //be shorter than the length of the track being processed.
277  // Make soundTouchBuffer twice as big as MaxBlockSize for each channel,
278  // because Soundtouch wants them interleaved, i.e., each
279  // Soundtouch sample is left-right pair.
280  sampleCount maxBlockSize = leftTrack->GetMaxBlockSize();
281  float* leftBuffer = new float[maxBlockSize];
282  float* rightBuffer = new float[maxBlockSize];
283  float* soundTouchBuffer = new float[maxBlockSize * 2];
284 
285  // Go through the track one stereo buffer at a time.
286  // sourceSampleCount counts the sample at which the current buffer starts,
287  // per channel.
288  sampleCount sourceSampleCount = start;
289  while (sourceSampleCount < end) {
290  //Get a block of samples (smaller than the size of the buffer)
291  sampleCount blockSize = leftTrack->GetBestBlockSize(sourceSampleCount);
292 
293  //Adjust the block size if it is the final block in the track
294  if (sourceSampleCount + blockSize > end)
295  blockSize = end - sourceSampleCount;
296 
297  // Get the samples from the tracks and put them in the buffers.
298  leftTrack->Get((samplePtr)(leftBuffer), floatSample, sourceSampleCount, blockSize);
299  rightTrack->Get((samplePtr)(rightBuffer), floatSample, sourceSampleCount, blockSize);
300 
301  // Interleave into soundTouchBuffer.
302  for (int index = 0; index < blockSize; index++) {
303  soundTouchBuffer[index*2] = leftBuffer[index];
304  soundTouchBuffer[(index*2)+1] = rightBuffer[index];
305  }
306 
307  //Add samples to SoundTouch
308  mSoundTouch->putSamples(soundTouchBuffer, blockSize);
309 
310  //Get back samples from SoundTouch
311  unsigned int outputCount = mSoundTouch->numSamples();
312  if (outputCount > 0)
313  this->ProcessStereoResults(outputCount, outputLeftTrack, outputRightTrack);
314 
315  //Increment sourceSampleCount one blockfull of samples
316  sourceSampleCount += blockSize;
317 
318  //Update the Progress meter
319  // mCurTrackNum is left track. Include right track.
320  int nWhichTrack = mCurTrackNum;
321  double frac = (sourceSampleCount - start) / len;
322  if (frac < 0.5)
323  frac *= 2.0; // Show twice as far for each track, because we're doing 2 at once.
324  else
325  {
326  nWhichTrack++;
327  frac -= 0.5;
328  frac *= 2.0; // Show twice as far for each track, because we're doing 2 at once.
329  }
330  if (TrackProgress(nWhichTrack, frac))
331  return false;
332  }
333 
334  // Tell SoundTouch to finish processing any remaining samples
335  mSoundTouch->flush();
336 
337  unsigned int outputCount = mSoundTouch->numSamples();
338  if (outputCount > 0)
339  this->ProcessStereoResults(outputCount, outputLeftTrack, outputRightTrack);
340 
341  // Flush the output WaveTracks (since they're buffered, too)
342  outputLeftTrack->Flush();
343  outputRightTrack->Flush();
344 
345  // Clean up the buffers.
346  delete [] leftBuffer;
347  delete [] rightBuffer;
348  delete [] soundTouchBuffer;
349 
350  // Take the output tracks and insert in place of the original
351  // sample data.
352  leftTrack->ClearAndPaste(mCurT0, mCurT1, outputLeftTrack, true, false, GetTimeWarper());
353  rightTrack->ClearAndPaste(mCurT0, mCurT1, outputRightTrack, true, false, GetTimeWarper());
354 
355  // Track the longest result length
356  double newLength = outputLeftTrack->GetEndTime();
357  m_maxNewLength = wxMax(m_maxNewLength, newLength);
358  newLength = outputRightTrack->GetEndTime();
359  m_maxNewLength = wxMax(m_maxNewLength, newLength);
360 
361  // Delete the outputTracks now that their data are inserted in place.
362  delete outputLeftTrack;
363  delete outputRightTrack;
364 
365  //Return true because the effect processing succeeded.
366  return true;
367 }

369 bool EffectSoundTouch::ProcessStereoResults(const unsigned int outputCount,
370  WaveTrack* outputLeftTrack,
371  WaveTrack* outputRightTrack)
372 {
373  float* outputSoundTouchBuffer = new float[outputCount*2];
374  mSoundTouch->receiveSamples(outputSoundTouchBuffer, outputCount);
375 
376  // Dis-interleave outputSoundTouchBuffer into separate track buffers.
377  float* outputLeftBuffer = new float[outputCount];
378  float* outputRightBuffer = new float[outputCount];
379  for (unsigned int index = 0; index < outputCount; index++)
380  {
381  outputLeftBuffer[index] = outputSoundTouchBuffer[index*2];
382  outputRightBuffer[index] = outputSoundTouchBuffer[(index*2)+1];
383  }
384 
385  outputLeftTrack->Append((samplePtr)outputLeftBuffer, floatSample, outputCount);
386  outputRightTrack->Append((samplePtr)outputRightBuffer, floatSample, outputCount);
387 
388  delete[] outputSoundTouchBuffer;
389  delete[] outputLeftBuffer;
390  delete[] outputRightBuffer;
391 
392  return true;
