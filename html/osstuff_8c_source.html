379 int ostgetc()
380 {
381 /*
382  * NOTE: lbuf[] accumulates characters as they are typed
383  * lpos[] is the column position of the characters
384  * lcount is the number of characters in lbuf
385  * lposition is current position
386  * lindex is index of next char to output
387  * line_edit is true iff we're inputing characters
388  *
389  */
390  int ch;
391 
392  while (line_edit) {
393  ch = rawtchar();
394  if (ch == EOF) xlisp_wrapup();
395  oscheck(); /* in case user typed ^C */
396  /* assume for now we should add the character */
397  lbuf[lcount] = ch;
398  lpos[lcount] = lposition;
399  lcount++;
400  lposition++;
401 
402  /* now do all the special character processing */
403  switch (ch) {
404  case '\001': /* take out non-printing character */
405  lcount--;
406  lposition--;
407  mark_audio_time();
408  break;
409  case '\n':
410  lposition = 0;
411  end_of_line_edit();
412  if (echo_enabled) {
413  osaputc('\r', stdout);
414  osaputc(ch, stdout);
415  }
416  break;
417  /* delete key generates: 1b, 5b, 33, 7E
418  which is: ESC, [, 3, ~ */
419  case '\010': /* backspace */
420  case '\177': /* delete */
421  lcount--; /* take out backspace or delete char */
422  lposition--;
423  if (lcount) {
424  lcount--;
425  while (lposition > lpos[lcount]) {
426  if (echo_enabled) {
427  putchar('\010');
428  putchar(' ');
429  putchar('\010');
430  }
431  lposition--;
432  }
433  }
434  break;
435  case '\025': /* control-u */
436  lcount--;
437  lposition--;
438  if (lcount) {
439  while (lposition > lpos[0]) {
440  if (echo_enabled) {
441  putchar('\010');
442  putchar(' ');
443  putchar('\010');
444  }
445  lposition--;
446  }
447  lcount = 0;
448  }
449  break;
450 
451  /* note that control-z never reaches here */
452  case '\003': /* control-c */
453  xltoplevel();
454  lcount = 0;
455  break;
456  case '\007': /* control-g */
457  lcount--; /* take out non-printing char */
458  lposition--;
459  xlcleanup();
460  lcount = 0;
461  break;
462  case '\016':
463  lcount--; /* take out non-printing char */
464  lposition--;
465  hidden_msg(); /* process hidden msg chars */
466  break;
467  case '\020': /* control-p */
468  lcount--; /* take out non-printing char */
469  lposition--;
470  xlcontinue();
471  lcount = 0;
472  break;
473  case '\002':
474  lcount--; /* take out non-printing char */
475  lposition--;
476  xflush(); /* control-b */
477  xlbreak("BREAK",s_unbound);
478  break;
479  case '\005': /* control-e */
480  lcount--; /* take out non-printing char */
481  lposition--;
482  echo_enabled = TRUE;
483  break;
484  case '\006': /* control-f */
485  lcount--; /* take out non-printing char */
486  lposition--;
487  echo_enabled = FALSE;
488  break;
489  case '\024': /* control-t */
490  lcount--; /* take out non-printing char */
491  lposition--;
492  xinfo();
493  lcount = 0;
494  break;
495 
496  case '\t': /* TAB */
497  lposition--; /* undo the increment above */
498  do {
499  lposition++;
500  if (echo_enabled) osaputc(' ', stdout);
501  } while (lposition & 7);
502  break;
503  default:
504  if (echo_enabled) osaputc(ch, stdout);
505  break;
506  }
507  // avoid line buffer overflow here:
508  if (lposition > LBSIZE - 10) {
509  // buffer is about to overflow, so write newline and
510  // feed chars to XLISP
511  if (echo_enabled) {
512  osaputc('\r', stdout);
513  osaputc('\n', stdout);
514  }
515  lposition = 0;
516  end_of_line_edit();
517  }
518  }
519  if (lindex + 1 >= lcount) {
520  lcount = 0;
521  line_edit = TRUE;
522  }
523  ch = lbuf[lindex++];
524  /* printf("-%c-", ch); */
525  if (echo_enabled) fflush(stdout);
526  return ch;
527 }

562 /* hidden_msg - process a "hidden message"
563  *
564  * NOTE: a "hidden message" is a sequence of characters starting
565  * with '\016' and ending with '\021'. These are designed to allow
566  * a graphical interface, namely jNyqIDE, to control sliders in
567  * real-time (during synthesis). The character sequences are hidden
568  * meaning they are not echoed and they are not interpreted as LISP.
569  *
570  * This function assumes that '\016' has been received already.
571  */
572 LOCAL void hidden_msg()
573 {
574 #define MSGBUF_MAX 64
575  char msgbuf[MSGBUF_MAX];
576  int msgbufx = 0;
577  char type_char = rawtchar();
578  char ch;
579  // message is terminated by '\021'
580  while ((ch = term_getchar()) != '\021' && ch != EOF &&
581  msgbufx < MSGBUF_MAX - 1) {
582  msgbuf[msgbufx++] = ch;
583  }
584  msgbuf[msgbufx++] = 0;
585  // printf("hidden message: %s, len %ld\n", msgbuf, (long) strlen(msgbuf));
586  if (msgbufx < MSGBUF_MAX) {
587  if (type_char == 'S') { // slider change message
588  int index;
589  float value;
590  if (sscanf(msgbuf, "%d %g", &index, &value) == 2) {
591  set_slider(index, value);
592  }
593  }
594  } /* other hidden messages could be parsed here */
595 }

598 /* oscheck - check for control characters during execution */
599 /*
600  * NOTE: to support type-ahead, unused characters are put
601  * into a queue to be removed by ostgetc
602  */
603 void oscheck(void)
604 {
605  int ch;
606 
607 #if OSC
608  if (nosc_enabled) nosc_poll();
609 #endif
610 
611  if (ctc) { /* control-c */
612  /* printf("[oscheck: control-c detected]"); */
613  ctc=FALSE; ctcreset();
614  xflush(); xltoplevel(); return;
615  }
616 
617  if ((ch = xcheck())) {
618  switch (ch) {
619  case BREAK_CHAR: /* control-b */
620  /* printf("BREAK_CHAR\n"); */
621  xflush(); xlbreak("BREAK",s_unbound); break;
622  case '\024': /* control-t */
623  /* printf("control-t\n"); */
624  xinfo(); break;
625  case '\025': /* control-u */
626  /* printf("control-u\n"); */
627  xcleanup();
628  case '\016': { /* begin hidden message */
629  /* printf("hidden msg\n"); */
630  hidden_msg();
631  break;
632  }
633  case '\001': /* control-a -- mark audio time */
634  mark_audio_time(); break;
635  case -1: /* EOF - lost connection, so die */
636  xlisp_wrapup();
637  break;
638  case -2: /* no character was ready */
639  break;
640  default:
641  /* printf("Got %d\n", ch); */
642 #ifndef READ_LINE
643  /* printf("+%c+", ch); */
644  typeahead[typeahead_tail++] = ch;
645  typeahead_tail &= (typeahead_max - 1);
646  if (typeahead_tail == typeahead_head) {
647  oserror("Input buffer overflow\n");
648  }
649 #endif
650  break;
651  }
652  }
653 
654  run_time++;
655  // when compute-bound, run_time is incremented by 10000 in about 15s, so
656  // that's about 700 Hz. We want to flush any output at about 2Hz, so
657  // we'll pick 400 as a round number.
658  // It's 2014, and now I'm seeing 3000 Hz. That's very high, so I
659  // changed SAMPLE to get this down to about 66Hz. Using % 30 to get
660  // 2Hz flush rate.
661  if (run_time % 30 == 0) {
662  fflush(stdout);
663  if (run_time_limit > 0 && run_time > run_time_limit) {
664  xlfatal("Run time limit exceeded");
665  }
666  if (memory_limit > 0 &&
667  npools * MAXPOOLSIZE + table_memory + total >
668  memory_limit * 1000000) {
669  xlfatal("Memory limit exceeded");
670  }
671  }
672 }
