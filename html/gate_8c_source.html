45 /* Overview:
46 This operation generates an exponential rise and decay suitable for
47 implementing a noise gate. The decay starts when the signal drops
48 below threshold and stays there for longer than lookahead.
49 Decay continues until the value reaches floor, at which point the
50 decay stops and the value is held constant. Either during the decay
51 or after the floor is reached, if the signal goes above threshold,
52 then the output value will rise to 1.0 (0dB) at the point the
53 signal crosses the threshold. Again, lookahead is used, so the rise
54 actually starts before the signal crosses the threshold. The rise
55 rate is constant and set so that a rise from floor to 0dB occurs
56 in the specified risetime. Similarly, the fall rate is constant
57 such that a fall from 0dB to the floor takes falltime.
58 
59 Rather than looking ahead, the output actually lags the input by
60 lookahead. The caller should advance the time of the input signal
61 in order to get a correct output signal, and this will be taken
62 care of in Lisp code.
63 
64 The implementation is a finite-state machine that simultaneously
65 computes the value and scans ahead for threshold crossings. Time
66 points, remembered as sample counts are saved in variables:
67  on_count -- the time at which the rise should complete
68  off_count -- the time at which the fall should begin
69  rise_factor -- multiply by this to get exponential rise
70  fall_factor -- multiply by this to get exponential fall
71  rise_time -- number of samples for a full rise
72  fall_time -- number of samples for a full fall
73  floor -- the lowest value to output
74  threshold -- compare the signal s to this value
75  start_rise -- the sample count at which a rise begins
76  delay_len -- number of samples to look ahead, length of buffer
77  state -- the current state of finite state machine
78  (see the individual 'case' statements for description of states)
79  value -- the current output value
80 
81 computing fall_factor:
82  factor ^ (sample_rate * time) == floor
83  log(factor) * sample_rate * time == log(floor)
84  log(factor) == log(floor) / (sample_rate * time)
85  factor == exp(log(floor) / (sample_rate * time))
86 
87 */

89 void compute_start_rise(gate_susp_type susp)
90 {
91  /* to compute when to start rise to achieve 0dB at on_count:
92  By similar triangles:
93  truncated rise time truncated fall time
94  ------------------- == -------------------
95  full rise time full fall time
96  when you enter ST_FALL, set start_fall = now
97  then if (on_count - start_fall) < (rise_time + fall_time)
98  then start rise at
99  on_time - rise_time * (on_count-start_fall)/(rise_time+fall_time)
100  */
101  long total = (long) (susp->rise_time + susp->fall_time);
102  if ((susp->on_count - susp->start_fall) < total) {
103  susp->start_rise = (long) (susp->on_count -
104  (susp->rise_time * susp->on_count - susp->start_fall) / total);
105  } else susp->start_rise = (long) (susp->on_count - susp->rise_time);
106 }
