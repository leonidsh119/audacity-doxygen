91 wxString Internat::ToDisplayString(double numberToConvert,
92  int digitsAfterDecimalPoint /* = -1 */)
93 {
94  wxString decSep = wxString(GetDecimalSeparator());
95  wxString result;
96 
97  if (digitsAfterDecimalPoint == -1)
98  {
99  result.Printf(wxT("%f"), numberToConvert);
100 
101  // Not all libcs respect the decimal separator, so always convert
102  // any dots found to the decimal separator.
103  result.Replace(wxT("."), decSep);
104 
105  if (result.Find(decSep) != -1)
106  {
107  // Strip trailing zeros, but leave one, and decimal separator.
108  int pos = result.Length() - 1;
109  while ((pos > 1) &&
110  (result.GetChar(pos) == wxT('0')) &&
111  (result.GetChar(pos - 1) != decSep))
112  pos--;
113  // (Previous code removed all of them and decimal separator.)
114  // if (result.GetChar(pos) == decSep)
115  // pos--; // strip point before empty fractional part
116  result = result.Left(pos+1);
117  }
118  }
119  else
120  {
121  wxString format;
122  format.Printf(wxT("%%.%if"), digitsAfterDecimalPoint);
123  result.Printf(format, numberToConvert);

126  // any dots found to the decimal separator
127  result.Replace(wxT("."), decSep);
