124 wxArrayString VampEffectsModule::FindPlugins(PluginManagerInterface & WXUNUSED(pm))
125 {
126  wxArrayString names;
127 
128  PluginLoader *loader = PluginLoader::getInstance();
129 
130  PluginLoader::PluginKeyList keys = loader->listPlugins();
131 
132  for (PluginLoader::PluginKeyList::iterator i = keys.begin(); i != keys.end(); ++i)
133  {
134  Plugin *vp = PluginLoader::getInstance()->loadPlugin(*i, 48000); // rate doesn't matter here
135  if (!vp)
136  {
137  continue;
138  }
139 
140  // We limit the listed plugin outputs to those whose results can
141  // readily be displayed in an Audacity label track.
142  //
143  // - Any output whose features have no values (time instants only),
144  // with or without duration, is fine
145  //
146  // - Any output whose features have more than one value, or an
147  // unknown or variable number of values, is right out
148  //
149  // - Any output whose features have exactly one value, with
150  // variable sample rate or with duration, should be OK --
151  // this implies a sparse feature, of which the time and/or
152  // duration are significant aspects worth displaying
153  //
154  // - An output whose features have exactly one value, with
155  // fixed sample rate and no duration, cannot be usefully
156  // displayed -- the value is the only significant piece of
157  // data there and we have no good value plot
158 
159  Plugin::OutputList outputs = vp->getOutputDescriptors();
160 
161  int output = 0;
162 
163  for (Plugin::OutputList::iterator j = outputs.begin(); j != outputs.end(); ++j)
164  {
165  if (j->sampleType == Plugin::OutputDescriptor::FixedSampleRate ||
166  j->sampleType == Plugin::OutputDescriptor::OneSamplePerStep ||
167  !j->hasFixedBinCount ||
168  (j->hasFixedBinCount && j->binCount > 1))
169  {
170  // All of these qualities disqualify (see notes above)
171 
172  ++output;
173  continue;
174  }
175 
176  wxString name = wxString::FromUTF8(vp->getName().c_str());
177 
178  if (outputs.size() > 1)
179  {
180  // This is not the plugin's only output.
181  // Use "plugin name: output name" as the effect name,
182  // unless the output name is the same as the plugin name
183  wxString outputName = wxString::FromUTF8(j->name.c_str());
184  if (outputName != name)
185  {
186  name = wxString::Format(wxT("%s: %s"),
187  name.c_str(), outputName.c_str());
188  }
189  }
190 
191  wxString path = wxString::FromUTF8(i->c_str()) + wxT("/") + name;
192  names.Add(path);
193 
194  ++output;
195  }
196 
197  delete vp;
198  }
199 
200  return names;
201 }
