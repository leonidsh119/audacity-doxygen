98 wxTopLevelWindow *ScreenshotCommand::GetFrontWindow(AudacityProject *project)
99 {
100  wxWindow *front = NULL;
101  wxWindow *proj = wxGetTopLevelParent(project);
102 
103  // This is kind of an odd hack. There's no method to enumerate all
104  // possible windows, so we search the whole screen for any windows
105  // that are not this one and not the given Audacity project and
106  // if we find anything, we assume that's the dialog the user wants
107  // to capture.
108 
109  int width, height, x, y;
110  wxDisplaySize(&width, &height);
111  for (x = 0; x < width; x += 50) {
112  for (y = 0; y < height; y += 50) {
113  wxWindow *win = wxFindWindowAtPoint(wxPoint(x, y));
114  if (win) {
115  win = wxGetTopLevelParent(win);
116  if (win != mIgnore && win != proj) {
117  front = win;
118  break;
119  }
120  }
121  }
122  }
123 
124  if (!front || !front->IsTopLevel()) {
125  return (wxTopLevelWindow *)proj;
126  }
127 
128  return (wxTopLevelWindow *)front;
129 }

155 void ScreenshotCommand::Capture(const wxString &filename,
156  wxWindow *window,
157  int x, int y, int width, int height,
158  bool bg)
159 {
160  if (window) {
161  if (window->IsTopLevel()) {
162  window->Raise();
163  }
164  else {
165  wxGetTopLevelParent(window)->Raise();
166  }
167  }
168 
169  Yield();
170 
171  int screenW, screenH;
172  wxDisplaySize(&screenW, &screenH);
173  wxBitmap full(screenW, screenH);
174 
175  wxScreenDC screenDC;
176  wxMemoryDC fullDC;
177 
178 #if defined(__WXMAC__) && !wxCHECK_VERSION(3, 0, 0)
179  full = DoGetAsBitmap(NULL);
180 #else
181  // We grab the whole screen image since there seems to be a problem with
182  // using non-zero source coordinates on OSX. (as of wx2.8.9)
183  fullDC.SelectObject(full);
184  fullDC.Blit(0, 0, screenW, screenH, &screenDC, 0, 0);
185  fullDC.SelectObject(wxNullBitmap);
186 #endif
187 
188  wxRect r(x, y, width, height);
189 
190  // Ensure within bounds (x/y are negative on Windows when maximized)
191  r.Intersect(wxRect(0, 0, screenW, screenH));
192 
193  // Convert to screen coordinates if needed
194  if (window && window->GetParent() && !window->IsTopLevel()) {
195  r.SetPosition(window->GetParent()->ClientToScreen(r.GetPosition()));
196  }
197 
198  // Extract the actual image
199  wxBitmap part = full.GetSubBitmap(r);
200 
201  // Add a background
202  if (bg && mBackground) {
203  wxRect b = GetBackgroundRect();
204 
205  wxBitmap back(width + b.width, height + b.height);
206  fullDC.SelectObject(back);
207 
208  fullDC.SetBackground(wxBrush(mBackColor, wxSOLID));
209  fullDC.Clear();
210 
211  fullDC.DrawBitmap(part, b.x, b.y);
212  fullDC.SelectObject(wxNullBitmap);
213 
214  part = back;
215  }
216 
217  // Save the final image
218  wxImage image = part.ConvertToImage();
219  if (image.SaveFile(filename)) {
220  mOutput->Status(_("Saved ") + filename);
221  }
222  else {
223  mOutput->Error(_("Error trying to save file: ") + filename);
224  }
225 
226  ::wxBell();
227 }
