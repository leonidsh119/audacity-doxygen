58 static SuilWrapper*
59 open_wrapper(SuilHost* host,
60  const char* container_type_uri,
61  const char* ui_type_uri,
62  LV2_Feature*** features,
63  unsigned n_features)
64 {
65  if (!container_type_uri || !strcmp(container_type_uri, ui_type_uri)) {
66  return NULL;
67  }
68 
69  const char* module_name = NULL;
70  if (!strcmp(container_type_uri, QT4_UI_URI)
71  && !strcmp(ui_type_uri, GTK2_UI_URI)) {
72  module_name = "suil_gtk2_in_qt4";
73  } else if (!strcmp(container_type_uri, GTK2_UI_URI)
74  && !strcmp(ui_type_uri, QT4_UI_URI)) {
75  module_name = "suil_qt4_in_gtk2";
76  } else if (!strcmp(container_type_uri, GTK2_UI_URI)
77  && !strcmp(ui_type_uri, X11_UI_URI)) {
78  module_name = "suil_x11_in_gtk2";
79  } else if (!strcmp(container_type_uri, GTK2_UI_URI)
80  && !strcmp(ui_type_uri, WIN_UI_URI)) {
81  module_name = "suil_win_in_gtk2";
82  } else if (!strcmp(container_type_uri, QT4_UI_URI)
83  && !strcmp(ui_type_uri, X11_UI_URI)) {
84  module_name = "suil_x11_in_qt4";
85  }
86 
87  if (!module_name) {
88  SUIL_ERRORF("Unable to wrap UI type <%s> as type <%s>\n",
89  ui_type_uri, container_type_uri);
90  return NULL;
91  }
92 
93  const char* const env_dir = getenv("SUIL_MODULE_DIR");
94  const char* const mod_dir = env_dir ? env_dir : SUIL_MODULE_DIR;
95 
96  const size_t path_len = strlen(mod_dir)
97  + strlen(SUIL_DIR_SEP SUIL_MODULE_PREFIX SUIL_MODULE_EXT)
98  + strlen(module_name)
99  + 2;
100 
101  char* const path = (char*)calloc(path_len, 1);
102  snprintf(path, path_len, "%s%s%s%s%s",
103  mod_dir, SUIL_DIR_SEP,
104  SUIL_MODULE_PREFIX, module_name, SUIL_MODULE_EXT);
105 
106  // Open wrap module
107  dlerror();
108  void* lib = dlopen(path, RTLD_NOW);
109  if (!lib) {
110  SUIL_ERRORF("Unable to open wrap module %s (%s)\n", path, dlerror());
111  return NULL;
112  }
113 
114  SuilWrapperNewFunc wrapper_new = (SuilWrapperNewFunc)suil_dlfunc(
115  lib, "suil_wrapper_new");
116 
117  SuilWrapper* wrapper = wrapper_new
118  ? wrapper_new(host,
119  container_type_uri,
120  ui_type_uri,
121  features,
122  n_features)
123  : NULL;
124 
125  if (!wrapper) {
126  SUIL_ERRORF("Corrupt module %s\n", path);
127  dlclose(lib);
128  free(path);
129  return NULL;
130  }
131 
132  free(path);
133 
134  wrapper->lib = lib;
135 
136  return wrapper;
137 }

140 SuilInstance*
141 suil_instance_new(SuilHost* host,
142  SuilController controller,
143  const char* container_type_uri,
144  const char* plugin_uri,
145  const char* ui_uri,
146  const char* ui_type_uri,
147  const char* ui_bundle_path,
148  const char* ui_binary_path,
149  const LV2_Feature* const* features)
150 {
151  // Open UI library
152  dlerror();
153  void* lib = dlopen(ui_binary_path, RTLD_NOW);
154  if (!lib) {
155  SUIL_ERRORF("Unable to open UI library %s (%s)\n",
156  ui_binary_path, dlerror());
157  return NULL;
158  }
159 
160  // Get discovery function
161  LV2UI_DescriptorFunction df = (LV2UI_DescriptorFunction)
162  suil_dlfunc(lib, "lv2ui_descriptor");
163  if (!df) {
164  SUIL_ERRORF("Broken LV2 UI %s (no lv2ui_descriptor symbol found)\n",
165  ui_binary_path);
166  dlclose(lib);
167  return NULL;
168  }
169 
170  // Get UI descriptor
171  const LV2UI_Descriptor* descriptor = NULL;
172  for (uint32_t i = 0; true; ++i) {
173  const LV2UI_Descriptor* ld = df(i);
174  if (!ld) {
175  break;
176  } else if (!strcmp(ld->URI, ui_uri)) {
177  descriptor = ld;
178  break;
179  }
180  }
181  if (!descriptor) {
182  SUIL_ERRORF("Failed to find descriptor for <%s> in %s\n",
183  ui_uri, ui_binary_path);
184  dlclose(lib);
185  return NULL;
186  }
187 
188  // Create SuilInstance
189  SuilInstance* instance = (SuilInstance*)calloc(1, sizeof(SuilInstance));
190  if (!instance) {
191  SUIL_ERRORF("Failed to allocate memory for <%s> instance\n", ui_uri);
192  dlclose(lib);
193  return NULL;
194  }
195 
196  instance->lib_handle = lib;
197  instance->descriptor = descriptor;
198 
199  // Make UI features array
200  instance->features = (LV2_Feature**)malloc(sizeof(LV2_Feature**));
201  instance->features[0] = NULL;
202 
203  // Copy user provided features
204  const LV2_Feature* const* fi = features;
205  unsigned n_features = 0;
206  while (fi && *fi) {
207  const LV2_Feature* f = *fi++;
208  suil_add_feature(&instance->features, &n_features, f->URI, f->data);
209  }
210 
211  // Add additional features implemented by SuilHost functions
212  if (host->index_func) {
213  instance->port_map.handle = controller;
214  instance->port_map.port_index = host->index_func;
215  suil_add_feature(&instance->features, &n_features,
216  LV2_UI__portMap, &instance->port_map);
217  }
218  if (host->subscribe_func && host->unsubscribe_func) {
219  instance->port_subscribe.handle = controller;
220  instance->port_subscribe.subscribe = host->subscribe_func;
221  instance->port_subscribe.unsubscribe = host->unsubscribe_func;
222  suil_add_feature(&instance->features, &n_features,
223  LV2_UI__portSubscribe, &instance->port_subscribe);
224  }
225  if (host->touch_func) {
226  instance->touch.handle = controller;
227  instance->touch.touch = host->touch_func;
228  suil_add_feature(&instance->features, &n_features,
229  LV2_UI__touch, &instance->touch);
230  }
231 
232  // Open wrapper (this may add additional features)
233  instance->wrapper = open_wrapper(
234  host, container_type_uri, ui_type_uri, &instance->features, n_features);
235 
236  // Instantiate UI
237  instance->handle = descriptor->instantiate(
238  descriptor,
239  plugin_uri,
240  ui_bundle_path,
241  host->write_func,
242  controller,
243  &instance->ui_widget,
244  (const LV2_Feature* const*)instance->features);
245 
246  // Failed to instantiate UI
247  if (!instance->handle) {
248  SUIL_ERRORF("Failed to instantiate UI <%s> in %s\n",
249  ui_uri, ui_binary_path);
250  suil_instance_free(instance);
251  return NULL;
252  }
253 
254  if (instance->wrapper) {
255  if (instance->wrapper->wrap(instance->wrapper, instance)) {
256  SUIL_ERRORF("Failed to wrap UI <%s> in type <%s>\n",
257  ui_uri, container_type_uri);
258  suil_instance_free(instance);
259  return NULL;
260  }
261  } else {
262  instance->host_widget = instance->ui_widget;
263  }
264 
265  return instance;
266 }

270 suil_instance_free(SuilInstance* instance)
271 {
272  if (instance) {
273  for (unsigned i = 0; instance->features[i]; ++i) {
274  free(instance->features[i]);
275  }
276  free(instance->features);
277 
278  // Call wrapper free function to destroy widgets and drop references
279  if (instance->wrapper && instance->wrapper->free) {
280  instance->wrapper->free(instance->wrapper);
281  }
282 
283  // Call cleanup to destroy UI (if it still exists at this point)
284  if (instance->handle) {
285  instance->descriptor->cleanup(instance->handle);
286  }
287 
288  // Close libraries and free everything
289  if (instance->wrapper) {
290  dlclose(instance->wrapper->lib);
291  free(instance->wrapper);
292  }
293  dlclose(instance->lib_handle);
294  free(instance);
295  }
