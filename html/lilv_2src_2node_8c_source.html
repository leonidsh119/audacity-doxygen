52 LilvNode*
53 lilv_node_new(LilvWorld* world, LilvNodeType type, const char* str)
54 {
55  LilvNode* val = (LilvNode*)malloc(sizeof(LilvNode));
56  val->world = world;
57  val->type = type;
58 
59  const uint8_t* ustr = (const uint8_t*)str;
60  switch (type) {
61  case LILV_VALUE_URI:
62  val->node = sord_new_uri(world->world, ustr);
63  break;
64  case LILV_VALUE_BLANK:
65  val->node = sord_new_blank(world->world, ustr);
66  break;
67  case LILV_VALUE_STRING:
68  val->node = sord_new_literal(world->world, NULL, ustr, NULL);
69  break;
70  case LILV_VALUE_INT:
71  val->node = sord_new_literal(
72  world->world, world->uris.xsd_integer, ustr, NULL);
73  break;
74  case LILV_VALUE_FLOAT:
75  val->node = sord_new_literal(
76  world->world, world->uris.xsd_decimal, ustr, NULL);
77  break;
78  case LILV_VALUE_BOOL:
79  val->node = sord_new_literal(
80  world->world, world->uris.xsd_boolean, ustr, NULL);
81  break;
82  case LILV_VALUE_BLOB:
83  val->node = sord_new_literal(
84  world->world, world->uris.xsd_base64Binary, ustr, NULL);
85  break;
86  }
87 
88  if (!val->node) {
89  free(val);
90  return NULL;
91  }
92 
93  return val;
94 }

97 LilvNode*
98 lilv_node_new_from_node(LilvWorld* world, const SordNode* node)
99 {
100  if (!node) {
101  return NULL;
102  }
103 
104  LilvNode* result = NULL;
105  SordNode* datatype_uri = NULL;
106  LilvNodeType type = LILV_VALUE_STRING;
107  size_t len = 0;
108 
109  switch (sord_node_get_type(node)) {
110  case SORD_URI:
111  result = (LilvNode*)malloc(sizeof(LilvNode));
112  result->world = (LilvWorld*)world;
113  result->type = LILV_VALUE_URI;
114  result->node = sord_node_copy(node);
115  break;
116  case SORD_BLANK:
117  result = (LilvNode*)malloc(sizeof(LilvNode));
118  result->world = (LilvWorld*)world;
119  result->type = LILV_VALUE_BLANK;
120  result->node = sord_node_copy(node);
121  break;
122  case SORD_LITERAL:
123  datatype_uri = sord_node_get_datatype(node);
124  if (datatype_uri) {
125  if (sord_node_equals(datatype_uri, world->uris.xsd_boolean))
126  type = LILV_VALUE_BOOL;
127  else if (sord_node_equals(datatype_uri, world->uris.xsd_decimal)
128  || sord_node_equals(datatype_uri, world->uris.xsd_double))
129  type = LILV_VALUE_FLOAT;
130  else if (sord_node_equals(datatype_uri, world->uris.xsd_integer))
131  type = LILV_VALUE_INT;
132  else if (sord_node_equals(datatype_uri,
133  world->uris.xsd_base64Binary))
134  type = LILV_VALUE_BLOB;
135  else
136  LILV_ERRORF("Unknown datatype `%s'\n",
137  sord_node_get_string(datatype_uri));
138  }
139  result = lilv_node_new(
140  world, type, (const char*)sord_node_get_string_counted(node, &len));
141  lilv_node_set_numerics_from_string(result, len);
142  break;
143  default:
144  assert(false);
145  }
146 
147  return result;
148 }

228 LILV_API bool
229 lilv_node_equals(const LilvNode* value, const LilvNode* other)
230 {
231  if (value == NULL && other == NULL)
232  return true;
233  else if (value == NULL || other == NULL)
234  return false;
235  else if (value->type != other->type)
236  return false;
237 
238  switch (value->type) {
239  case LILV_VALUE_URI:
240  case LILV_VALUE_BLANK:
241  case LILV_VALUE_STRING:
242  case LILV_VALUE_BLOB:
243  return sord_node_equals(value->node, other->node);
244  case LILV_VALUE_INT:
245  return (value->val.int_val == other->val.int_val);
246  case LILV_VALUE_FLOAT:
247  return (value->val.float_val == other->val.float_val);
248  case LILV_VALUE_BOOL:
249  return (value->val.bool_val == other->val.bool_val);
250  }
251 
252  return false; /* shouldn't get here */
253 }
