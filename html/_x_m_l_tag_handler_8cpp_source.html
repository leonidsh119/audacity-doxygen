109 bool XMLValueChecker::IsGoodInt(const wxString & strInt)
110 {
111  if (!IsGoodString(strInt))
112  return false;
113 
114  // Check that the value won't overflow.
115  // Signed long: -2,147,483,648 to +2,147,483,647, i.e., -2^31 to 2^31-1
116  // We're strict about disallowing spaces and commas, and requiring minus sign to be first char for negative.
117  const size_t lenMAXABS = strlen("2147483647");
118  const size_t lenStrInt = strInt.Length();
119 
120  if (lenStrInt > (lenMAXABS + 1))
121  return false;
122  else if ((lenStrInt == (lenMAXABS + 1)) && (strInt[0] == '-'))
123  {
124  const int digitsMAXABS[] = {2, 1, 4, 7, 4, 8, 3, 6, 4, 9};
125  unsigned int i;
126  for (i = 0; i < lenMAXABS; i++)
127  if (strInt[i+1] < '0' || strInt[i+1] > '9')
128  return false; // not a digit
129 
130  for (i = 0; i < lenMAXABS; i++)
131  if (strInt[i+1] - '0' < digitsMAXABS[i])
132  return true; // number is small enough
133  return false;
134  }
135  else if (lenStrInt == lenMAXABS)
136  {
137  const int digitsMAXABS[] = {2, 1, 4, 7, 4, 8, 3, 6, 4, 8};
138  unsigned int i;
139  for (i = 0; i < lenMAXABS; i++)
