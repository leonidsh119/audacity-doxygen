95 sampleCount EffectNoise::ProcessBlock(float **WXUNUSED(inbuf), float **outbuf, sampleCount size)
96 {
97  float *buffer = outbuf[0];
98 
99  float white;
100  float amplitude;
101  float div = ((float) RAND_MAX) / 2.0f;
102 
103  switch (mType)
104  {
105  default:
106  case kWhite: // white
107  for (sampleCount i = 0; i < size; i++)
108  {
109  buffer[i] = mAmp * ((rand() / div) - 1.0f);
110  }
111  break;
112 
113  case kPink: // pink
114  // based on Paul Kellet's "instrumentation grade" algorithm.
115 
116  // 0.129f is an experimental normalization factor.
117  amplitude = mAmp * 0.129f;
118  for (sampleCount i = 0; i < size; i++)
119  {
120  white = (rand() / div) - 1.0f;
121  buf0 = 0.99886f * buf0 + 0.0555179f * white;
122  buf1 = 0.99332f * buf1 + 0.0750759f * white;
123  buf2 = 0.96900f * buf2 + 0.1538520f * white;
124  buf3 = 0.86650f * buf3 + 0.3104856f * white;
125  buf4 = 0.55000f * buf4 + 0.5329522f * white;
126  buf5 = -0.7616f * buf5 - 0.0168980f * white;
127  buffer[i] = amplitude *
128  (buf0 + buf1 + buf2 + buf3 + buf4 + buf5 + buf6 + white * 0.5362);
129  buf6 = white * 0.115926;
130  }
131  break;
132 
133  case kBrownian: // Brownian
134  //float leakage=0.997; // experimental value at 44.1kHz
135  //double scaling = 0.05; // experimental value at 44.1kHz
136  // min and max protect against instability at extreme sample rates.
137  float leakage = ((mSampleRate - 144.0) / mSampleRate < 0.9999)
138  ? (mSampleRate - 144.0) / mSampleRate
139  : 0.9999f;
140 
141  float scaling = (9.0 / sqrt(mSampleRate) > 0.01)
142  ? 9.0 / sqrt(mSampleRate)
143  : 0.01f;
144 
145  for (sampleCount i = 0; i < size; i++)
146  {
147  white = (rand() / div) - 1.0f;
148  z = leakage * y + white * scaling;
149  y = fabs(z) > 1.0
150  ? leakage * y - white * scaling
151  : z;
152  buffer[i] = mAmp * y;
153  }
154  break;
155  }
156 
157  return size;
158 }
