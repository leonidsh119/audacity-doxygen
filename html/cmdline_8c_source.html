1 /* cmdline.c -- command line parsing routines */
2 /* Copyright 1989 Carnegie Mellon University */
3 /*
4  * This module is designed to allow various modules to scan (and rescan)
5  * the command line for applicable arguments. The goal is to hide as
6  * much information about switches and their names as possible so that
7  * switches become more consistent across applications and so that the
8  * author of an application need not do a lot of work to provide numerous
9  * options. Instead, each module scans the command line for its own
10  * arguments.
11  *
12  * Command lines are of the following form:
13  * command -s1 -s2 opt2 -s3 arg1 arg2 -s4 opt4 arg3
14  * Note that there are three kinds of command line parameters:
15  * (1) A Switch is a "-" followed by a name, e.g. "-s1"
16  * (2) An Option is a Switch followed by a space and name, e.g. "-s2 opt2"
17  * (3) An Argument is a name by itself, e.g. "arg1"
18  * Note also that a switch followed by an argument looks just like an
19  * option, so a list of valid option names is necessary to disambiguate.
20  *
21  * Long names are good for readability, but single character abbreviations
22  * are nice for experienced users. cmdline.c allows single character
23  * abbreviations provided that they are unambiguous. These are
24  * recognized with no extra work by the programmer. If an
25  * isolated '?' is encountered in the command line, then all of
26  * the options and switches are printed as help and for debugging.
27  *
28  * Given that we must tell this module about option names and switch
29  * names, how should we do it? We can't wait until modules are
30  * initialized, since often modules want to read the command line
31  * at initialization time. In the original implementation, the
32  * main program was supposed to provide names for the whole program,
33  * but this violates modularity: when an option is added to a module,
34  * the main program has to be modified too. This is a real pain when
35  * different machines support different options and you want to have
36  * a single machine-independent main program. The solution is to
37  * have the main program import strings describing the options and
38  * switches used by each module. These are passed into cmdline.c
39  * before initialization of other modules is begun.
40  *
41  * A main program that uses cmdline.c should do the following:
42  * call cl_syntax(s) for each module's option/switch string.
43  * The string s should have the following format:
44  * "opt1<o>description;opt2<o>description;...;switch1<s>description;..."
45  * where opt1 and opt2 are option names (without the preceding "-"), and
46  * switch1 is a switch name. The <o> and <s> indicate whether the
47  * name is an option or a switch. The descriptions are arbitrary strings
48  * (without semicolons) that are printed out for the user when "?"
49  * is typed on the command line.
50  *
51  * After calling cl_syntax, main() should call
52  * cl_init(argv, argc)
53  * cl_init will report an error (to STDERR) if it finds any illegal
54  * switch or option names in argv, and help will be printed if "?"
55  * is found in argv. If cl_init returns false, then the user has been
56  * given an error message or help, and main should probably exit.
57  *
58  * Afterward, switches, options, and arguments can be accessed by
59  * calling cl_switch, cl_option, and cl_arg. If cl_switch or cl_option
60  * is called with a switch name that was not mentioned in the call to
61  * cl_init, an error will result. This indicates that the application
62  * author omitted a valid switch or option name when calling cl_init.
63  * This is an error because the full set of names is needed for error
64  * checking and to distinguish arguments from options.
65  *
66  */

138 void cl_help()
139 {
140  register int i, j;
141  int count = 0; /* see if there are any switches or flags */
142 
143  for (i = 0; i < n_syntax; i++) {
144  register char *ptr = syntax[i];
145  register char c = *ptr++;
146  while (c != EOS) {
147  while (c != EOS && !(isalnum(c))) c = *ptr++;
148  if (c != EOS) {
149  count++;
150  gprintf(TRANS, "-");
151  j = 1;
152  while (c != EOS && c != '<') {
153  gprintf(TRANS, "%c", c);
154  c = *ptr++;
155  j++;
156  }
157  if (c != EOS) {
158  c = *ptr++;
159  if (c == 'o') {
160  gprintf(TRANS, " xxx");
161  j += 4;
162  }
163  }
164  /* attempt to tab */
165  do {
166  gprintf(TRANS, " ");
167  } while (j++ < 16);
168  while (c != EOS && c != '>') c = *ptr++;
169  if (c != EOS) c = *ptr++;
170  while (c != EOS && c != ';') {
171  gprintf(TRANS, "%c", c);
172  c = *ptr++;
173  }
174  gprintf(TRANS, "\n");
175  }
176  }
177  }
178  if (!count) gprintf(TRANS, "No switches or options exist.\n");
179 }

246 /****************************************************************
247 * cl_search
248 * Inputs:
249 * char *name: name of field, must be non-null if opt_sw == cl_INIT
250 * int opt_sw: option, switch, init, or arg
251 * int n: argument number (if opt_sw is cl_ARG)
252 * Result:
253 * returns pointer to option value/switch if one exists, otherwise null
254 * Implementation:
255 * parse the command line until name or arg is found
256 * see if the option is followed by a string that does
257 * not start with "-"
258 *****************************************************************/
259 
260 private char *cl_search(name, opt_sw, n)
261  char *name;
262  int opt_sw;
263  int n; /* if opt_sw is cl_ARG, n > 0 tells which one */
264 {
265  register int i = 1; /* index into command line */
266  boolean abbr;
267  boolean result = TRUE;
268 
269  ready_check();
270 
271  /* parse command line: */
272  while (i < argc) {
273  register char *arg = argv[i];
274  /* arguments that start with '-' should be quoted and quotes must
275  be removed by the application
276  */
277  if (*arg == '-') {
278  int arg_type = find_string(arg + 1, &abbr);
279  if (arg_type == cl_OPT) {
280  i += 1; /* skip name and option */
281  /* don't look for '-' because the option might be a
282  * negative number
283  */
284  if (i >= argc /* || *arg == '-' */) {
285  if (opt_sw == cl_INIT) {
286  gprintf(ERROR, "missing argument after %s\n", arg);
287  result = FALSE;
288  }
289  } else if (opt_sw == cl_OPT &&
290  (strcmp(arg + 1, name) == 0 ||
291  (abbr && *(arg + 1) == name[0]))) {
292  return argv[i];
293  }
294  } else if (arg_type == cl_SW) {
295  if (opt_sw == cl_SW &&
296  (strcmp(arg + 1, name) == 0 ||
297  (abbr && *(arg + 1) == name[0])))
298  return arg;
299  } else if (opt_sw == cl_INIT) {
300  gprintf(ERROR, "invalid switch: %s\n", arg);
301  result = FALSE;
302  }
303  } else if (opt_sw == cl_ARG) {
304  if (n == 1) return arg;
305  n--;
306  }
307  i++; /* skip to next field */
308  }
309  if (opt_sw == cl_INIT) {
310  /* return name or NULL to represent TRUE or FALSE */
311  return (result ? name : NULL);
312  }
313  return NULL;
314 }

357 /****************************************************************
358 * find_string
359 * Inputs:
360 * char *s: string to find, terminated by any non-alphanumeric
361 * boolean *abbr: set TRUE if s is an abbreviation, otherwise false
362 * Effect:
363 * Looks for s in syntax strings
364 * Returns:
365 * 0 = FALSE = not found, 1 = cl_OPT = option, 2 = cl_SW = switch
366 *****************************************************************/
367 
368 private int find_string(s, abbr)
369  char *s;
370  boolean *abbr;
371 {
372  int found_it = FALSE;
373  int i;
374  *abbr = FALSE;
375  for (i = 0; i < n_syntax; i++) { /* loop through strings */
376  register char *syntax_ptr = syntax[i];
377  while (*syntax_ptr != EOS) {
378  register char *s_ptr = s;
379  while (*syntax_ptr != EOS &&
380  !(isalnum(*syntax_ptr))) syntax_ptr++;
381  while (*s_ptr != EOS && (*s_ptr++ == *syntax_ptr))
382  syntax_ptr++; /* only increment if there's a match */
383  if (!(isalnum(*s_ptr)) && *syntax_ptr == '<') {
384  syntax_ptr++; /* advance to the type field */
385  if (*syntax_ptr == 's') return cl_SW;
386  if (*syntax_ptr != 'o')
387  gprintf(ERROR,
388  "(internal error) bad cl_syntax string: %s\n",
389  syntax[i]);
390  return cl_OPT;
391  }
392  /* no match, so go to next */
393  while (*syntax_ptr != ';' && *syntax_ptr != EOS) syntax_ptr++;
394  if (*syntax_ptr == ';') syntax_ptr++;
395  }
396  }
397 
398  /* no match, maybe there is a single character match */
399  if (s[0] == EOS || s[1] != EOS) return FALSE;
400 
401  for (i = 0; i < n_syntax; i++) { /* loop through strings */
402  char *syntax_ptr = syntax[i];
403  while (*syntax_ptr != EOS) {
404  while (*syntax_ptr != EOS &&
405  !(isalnum(*syntax_ptr))) syntax_ptr++;
406  if (s[0] == *syntax_ptr) {
407  if (found_it) return FALSE; /* ambiguous */
408  /* else, find the type */
409  while (*syntax_ptr != '<' && *syntax_ptr != EOS)
410  syntax_ptr++;
411  syntax_ptr++;
412  if (*syntax_ptr == 's') found_it = cl_SW;
413  else if (*syntax_ptr == 'o') found_it = cl_OPT;
414  else return FALSE; /* error in string syntax */
415  }
416  /* no match, so go to next */
417  while (*syntax_ptr != ';' && *syntax_ptr != EOS) syntax_ptr++;
418  if (*syntax_ptr == ';') syntax_ptr++;
419  }
420  }
421  if (found_it) *abbr = TRUE;
422  return found_it;
423 }
