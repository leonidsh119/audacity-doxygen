65 PaError PaWinUtil_CoInitialize( PaHostApiTypeId hostApiType, PaWinUtilComInitializationResult *comInitializationResult )
66 {
67  HRESULT hr;
68 
69  comInitializationResult->state = PAWINUTIL_COM_NOT_INITIALIZED;
70 
71  /*
72  If COM is already initialized CoInitialize will either return
73  FALSE, or RPC_E_CHANGED_MODE if it was initialised in a different
74  threading mode. In either case we shouldn't consider it an error
75  but we need to be careful to not call CoUninitialize() if
76  RPC_E_CHANGED_MODE was returned.
77  */
78 
79  hr = CoInitialize(0); /* use legacy-safe equivalent to CoInitializeEx(NULL, COINIT_APARTMENTTHREADED) */
80  if( FAILED(hr) && hr != RPC_E_CHANGED_MODE )
81  {
82  PA_DEBUG(("CoInitialize(0) failed. hr=%d\n", hr));
83 
84  if( hr == E_OUTOFMEMORY )
85  return paInsufficientMemory;
86 
87  {
88  char *lpMsgBuf;
89  FormatMessage(
90  FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM,
91  NULL,
92  hr,
93  MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
94  (LPTSTR) &lpMsgBuf,
95  0,
96  NULL
97  );
98  PaUtil_SetLastHostErrorInfo( hostApiType, hr, lpMsgBuf );
99  LocalFree( lpMsgBuf );
100  }
101 
102  return paUnanticipatedHostError;
103  }
104 
105  if( hr != RPC_E_CHANGED_MODE )
106  {
107  comInitializationResult->state = PAWINUTIL_COM_INITIALIZED;
108 
109  /*
110  Memorize calling thread id and report warning on Uninitialize if
111  calling thread is different as CoInitialize must match CoUninitialize
112  in the same thread.
113  */
114  comInitializationResult->initializingThreadId = GetCurrentThreadId();
115  }
116 
117  return paNoError;
118 }

121 void PaWinUtil_CoUninitialize( PaHostApiTypeId hostApiType, PaWinUtilComInitializationResult *comInitializationResult )
122 {
123  if( comInitializationResult->state != PAWINUTIL_COM_NOT_INITIALIZED
124  && comInitializationResult->state != PAWINUTIL_COM_INITIALIZED ){
125 
126  PA_DEBUG(("ERROR: PaWinUtil_CoUninitialize called without calling PaWinUtil_CoInitialize\n"));
127  }
128 
129  if( comInitializationResult->state == PAWINUTIL_COM_INITIALIZED )
130  {
131  DWORD currentThreadId = GetCurrentThreadId();
132  if( comInitializationResult->initializingThreadId != currentThreadId )
133  {
134  PA_DEBUG(("ERROR: failed PaWinUtil_CoUninitialize calling thread[%d] does not match initializing thread[%d]\n",
135  currentThreadId, comInitializationResult->initializingThreadId));
136  }
137  else
138  {
139  CoUninitialize();
140 
141  comInitializationResult->state = PAWINUTIL_COM_NOT_INITIALIZED;
142  }
143  }
