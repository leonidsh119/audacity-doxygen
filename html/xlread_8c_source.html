292 int readone(LVAL fptr, LVAL *pval)
293 {
294  LVAL val,type;
295  int ch;
296 
297 #ifdef DEBUG_INPUT
298  if (debug_input_fp) {
299  int c = getc(debug_input_fp);
300  ungetc(c, debug_input_fp);
301  }
302 #endif

304  if ((ch = xlgetc(fptr)) == EOF)
305  return (EOF);
306 
307  /* handle white space */
308  if ((type = tentry(ch)) == k_wspace)
309  return (FALSE);
310 
311  /* handle symbol constituents */
312  else if (type == k_const) {
313  xlungetc(fptr,ch);
314  *pval = psymbol(fptr);
315  return (TRUE);
316  }
317

320  xlungetc(fptr,ch);
321  *pval = psymbol(fptr);
322  return (TRUE);
323  }
324 
325  /* handle read macros */
326  else if (consp(type)) {
327  if ((val = callmacro(fptr,ch)) && consp(val)) {
328  *pval = car(val);
329  return (TRUE);
330  }
331  else
332  return (FALSE);
333  }
334 
335  /* handle illegal characters */
336  else {
337  xlerror("illegal character",cvfixnum((FIXTYPE)ch));
338  /* this point will never be reached because xlerror() does a
339  _longjmp(). The return is added to avoid false positive

607 /* pnumber - parse a number */
608 LOCAL LVAL pnumber(LVAL fptr, int radix)
609 {
610  int digit,ch;
611  long num;
612 
613  for (num = 0L; (ch = xlgetc(fptr)) != EOF; ) {
614  if (islower(ch)) ch = toupper(ch);
615  if (!('0' <= ch && ch <= '9') && !('A' <= ch && ch <= 'F'))
616  break;

905  p++;
906

909  p++, dr++;
910  }

913  if ((dl == 0 && dr == 0) || *p)

917  if (pval) {
