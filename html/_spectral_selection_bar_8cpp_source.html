230 void SpectralSelectionBar::ModifySpectralSelection(bool done)
231 {
232  const double nyq = mListener->SSBL_GetRate() / 2.0;
233 
234  double bottom, top;
235  if (mbCenterAndWidth) {
236  mCenter = mCenterCtrl->GetValue();
237  mWidth = mWidthCtrl->GetValue();
238  if ((mCenter < 0 || mWidth < 0) &&
239  (mLow >= 0 || mHigh >= 0))
240  // Transition from defined spectral selection to undefined
241  bottom = top = SelectedRegion::UndefinedFrequency;
242  else if (mCenter < 0 && mWidth < 0)
243  bottom = top = SelectedRegion::UndefinedFrequency;
244  else {
245  if (mCenter < 0) {
246  mWidth = log(std::min(nyq, exp(mWidth)));
247  // Choose arbitrary center for the width
248  mCenter = sqrt(nyq);
249  }
250  else if (mWidth < 0) {
251  mCenter = std::max(1.0, std::min(nyq, mCenter));
252  // Choose arbitrary width for the center
253  const double ratio = std::min(mCenter, nyq / mCenter);
254  mWidth = log(ratio * ratio);
255  }
256  else {
257  mCenter = std::max(1.0, std::min(nyq, mCenter));
258  double ratio = std::min(mCenter, nyq / mCenter);
259  mWidth = std::min(2 * log(ratio), mWidth);
260  }
261 
262  const double ratio = exp(mWidth / 2);
263  bottom = mCenter / ratio, top = mCenter * ratio;
264  }
265  }
266  else {
267  bottom = mLowCtrl->GetValue();
268  top = mHighCtrl->GetValue();
269 
270  if (bottom >= 0)
271  bottom = std::min(nyq, bottom);
272  else
273  bottom = SelectedRegion::UndefinedFrequency;
274 
275  if (top >= 0)
276  top = std::min(nyq, top);
277  else
278  top = SelectedRegion::UndefinedFrequency;
279  }
280 
281  mLow = bottom;
282  mHigh = top;
283  SetBounds();
284 
285  // Notify project and track panel, which may change
286  // the values again, and call back to us in SetFrequencies()
287  mListener->SSBL_ModifySpectralSelection(bottom, top, done);
288 }

313 void SpectralSelectionBar::OnUpdate(wxCommandEvent &evt)
314 {
315  int index = evt.GetInt();
316  wxWindow *w = FindFocus();
317  bool centerFocus = (w && w == mCenterCtrl);
318  bool widthFocus = (w && w == mWidthCtrl);
319  bool lowFocus = (w && w == mLowCtrl);
320  bool highFocus = (w && w == mHighCtrl);
321 
322  evt.Skip(false);
323 
324  // Save formats before recreating the controls so they resize properly
325  wxEventType type = evt.GetEventType();
326  if (type == EVT_FREQUENCYTEXTCTRL_UPDATED) {
327  NumericTextCtrl *frequencyCtrl = (mbCenterAndWidth ? mCenterCtrl : mLowCtrl);
328  wxString frequencyFormatName = frequencyCtrl->GetBuiltinName(index);
329  mListener->SSBL_SetFrequencySelectionFormatName(frequencyFormatName);
330  }
331  else if (mbCenterAndWidth &&
332  type == EVT_BANDWIDTHTEXTCTRL_UPDATED) {
333  wxString bandwidthFormatName = mWidthCtrl->GetBuiltinName(index);
334  mListener->SSBL_SetBandwidthSelectionFormatName(bandwidthFormatName);
335  }
336 
337  // ToolBar::ReCreateButtons() will get rid of our sizers and controls
338  // so reset pointers first.
339  mCenterCtrl = mWidthCtrl = NULL;
340  mLowCtrl = mHighCtrl = NULL;
341 
342  ToolBar::ReCreateButtons();
343  ValuesToControls();
344 
345 
346  if (centerFocus) {
347  mCenterCtrl->SetFocus();
348  }
349  else if (widthFocus) {
350  mWidthCtrl->SetFocus();
351  }
352  else if (lowFocus) {
353  mLowCtrl->SetFocus();
354  }
355  else if (highFocus) {
356  mHighCtrl->SetFocus();
357  }

359  Updated();
