345 // Deduce m_FromFrequency from the samples at the beginning of
346 // the selection. Then set some other params accordingly.
347 void EffectChangePitch::DeduceFrequencies()
348 {
349  // As a neat trick, attempt to get the frequency of the note at the
350  // beginning of the selection.
351  SelectedTrackListOfKindIterator iter(Track::Wave, mTracks);
352  WaveTrack *track = (WaveTrack *) iter.First();
353  if (track) {
354  double rate = track->GetRate();
355 
356  // Auto-size window -- high sample rates require larger windowSize.
357  // Aim for around 2048 samples at 44.1 kHz (good down to about 100 Hz).
358  // To detect single notes, analysis period should be about 0.2 seconds.
359  // windowSize must be a power of 2.
360  int windowSize = wxRound(pow(2.0, floor((log(rate / 20.0)/log(2.0)) + 0.5)));
361  // windowSize < 256 too inaccurate
362  windowSize = (windowSize > 256)? windowSize : 256;
363 
364  // we want about 0.2 seconds to catch the first note.
365  // number of windows rounded to nearest integer >= 1.
366  int numWindows = wxRound((double)(rate / (5.0f * windowSize)));
367  numWindows = (numWindows > 0)? numWindows : 1;
368 
369  double trackStart = track->GetStartTime();
370  double t0 = mT0 < trackStart? trackStart: mT0;
371  sampleCount start = track->TimeToLongSamples(t0);
372 
373  int analyzeSize = windowSize * numWindows;
374  float * buffer;
375  buffer = new float[analyzeSize];
376 
377  float * freq;
378  freq = new float[windowSize/2];
379 
380  float * freqa;
381  freqa = new float[windowSize/2];
382 
383  int i, j, argmax;
384  int lag;
385 
386  for(j=0; j<windowSize/2; j++)
387  freqa[j] = 0;
388 
389  track->Get((samplePtr) buffer, floatSample, start, analyzeSize);
390  for(i=0; i<numWindows; i++) {
391  ComputeSpectrum(buffer+i*windowSize, windowSize,
392  windowSize, rate, freq, true);
393  for(j=0; j<windowSize/2; j++)
394  freqa[j] += freq[j];
395  }
396  argmax=0;
397  for(j=1; j<windowSize/2; j++)
398  if (freqa[j] > freqa[argmax])
399  argmax = j;
400 
401  delete [] freq;
402  delete [] freqa;
403  delete [] buffer;
404 
405  lag = (windowSize/2 - 1) - argmax;
406  m_dStartFrequency = rate / lag;
407  }
408 
409  double dFromMIDInote = FreqToMIDInote(m_dStartFrequency);
410  double dToMIDInote = dFromMIDInote + m_dSemitonesChange;
411  m_nFromPitch = PitchIndex(dFromMIDInote);
412  m_nFromOctave = PitchOctave(dFromMIDInote);
413  m_nToPitch = PitchIndex(dToMIDInote);
414  m_nToOctave = PitchOctave(dToMIDInote);
415 
416  m_FromFrequency = m_dStartFrequency;
417  Calc_PercentChange();
418  Calc_ToFrequency();
419 }
