482  time_type t0_min = t0;
483  /* combine scale factors of linear inputs (INPUT) */
484  scale_factor *= input->scale;
485  input->scale = 1.0F;
486 
487  /* try to push scale_factor back to a low sr input */
488  if (input->sr < sr) { input->scale = scale_factor; scale_factor = 1.0F; }
489 
490  falloc_generic(susp, alpassvc_susp_node, "snd_make_alpassvc");
491  susp->delay_scale_factor = (float) (input->sr * delaysnd->scale);
492  susp->feedback = feedback;
493  susp->buflen = max(2, (long) (input->sr * maxdelay + 2.5));
494  susp->delaybuf = (sample_type *) calloc (susp->buflen + 1, sizeof(sample_type));
495  susp->delayptr = susp->delaybuf;
496  susp->endptr = susp->delaybuf + susp->buflen;
497 
498  /* make sure no sample rate is too high */
499  if (delaysnd->sr > sr) {
500  sound_unref(delaysnd);
501  snd_badsr();
502  }
503 
504  /* select a susp fn based on sample rates */
505  interp_desc = (interp_desc << 2) + interp_style(input, sr);
506  interp_desc = (interp_desc << 2) + interp_style(delaysnd, sr);
507  switch (interp_desc) {
508  case INTERP_ns: /* handled below */
509  case INTERP_nn: susp->susp.fetch = alpassvc_nn_fetch; break;
510  case INTERP_ni: susp->susp.fetch = alpassvc_ni_fetch; break;
511  case INTERP_nr: susp->susp.fetch = alpassvc_nr_fetch; break;
512  default: snd_badsr(); break;
513  }
514 
515  susp->terminate_cnt = UNKNOWN;
516  /* handle unequal start times, if any */
517  if (t0 < input->t0) sound_prepend_zeros(input, t0);
518  if (t0 < delaysnd->t0) sound_prepend_zeros(delaysnd, t0);
519  /* minimum start time over all inputs: */
520  t0_min = min(input->t0, min(delaysnd->t0, t0));
521  /* how many samples to toss before t0: */
522  susp->susp.toss_cnt = (long) ((t0 - t0_min) * sr + 0.5);
523  if (susp->susp.toss_cnt > 0) {
524  susp->susp.keep_fetch = susp->susp.fetch;
525  susp->susp.fetch = alpassvc_toss_fetch;
526  }
527 
528  /* initialize susp state */
529  susp->susp.free = alpassvc_free;
530  susp->susp.sr = sr;
531  susp->susp.t0 = t0;
532  susp->susp.mark = alpassvc_mark;
533  susp->susp.print_tree = alpassvc_print_tree;
534  susp->susp.name = "alpassvc";
535  susp->susp.log_stop_cnt = UNKNOWN;
536  susp->started = false;
537  susp->susp.current = 0;
538  susp->input = input;
539  susp->input_cnt = 0;
540  susp->delaysnd = delaysnd;
541  susp->delaysnd_cnt = 0;
542  susp->delaysnd_pHaSe = 0.0;
543  susp->delaysnd_pHaSe_iNcR = delaysnd->sr / sr;
544  susp->delaysnd_n = 0;

546  return sound_create((snd_susp_type)susp, t0, sr, scale_factor);
547 }
