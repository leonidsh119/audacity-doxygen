150 static void
151 run(LV2_Handle instance, uint32_t sample_count)
152 {
153  Midigate* self = (Midigate*)instance;
154  uint32_t offset = 0;
155 
156  LV2_ATOM_SEQUENCE_FOREACH(self->control, ev) {
157  if (ev->body.type == self->uris.midi_MidiEvent) {
158  const uint8_t* const msg = (const uint8_t*)(ev + 1);
159  switch (lv2_midi_message_type(msg)) {
160  case LV2_MIDI_MSG_NOTE_ON:
161  ++self->n_active_notes;
162  break;
163  case LV2_MIDI_MSG_NOTE_OFF:
164  --self->n_active_notes;
165  break;
166  case LV2_MIDI_MSG_PGM_CHANGE:
167  if (msg[1] == 0 || msg[1] == 1) {
168  self->program = msg[1];
169  }
170  break;
171  default: break;
172  }
173  }
174 
175  write_output(self, offset, ev->time.frames - offset);
176  offset = (uint32_t)ev->time.frames;
177  }
178 
179  write_output(self, offset, sample_count - offset);
180 }
