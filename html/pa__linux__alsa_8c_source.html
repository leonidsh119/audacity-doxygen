835 static PaError GropeDevice( snd_pcm_t* pcm, int isPlug, StreamDirection mode, int openBlocking,
836  PaAlsaDeviceInfo* devInfo )
837 {
838  PaError result = paNoError;
839  snd_pcm_hw_params_t *hwParams;
840  snd_pcm_uframes_t alsaBufferFrames, alsaPeriodFrames;
841  unsigned int minChans, maxChans;
842  int* minChannels, * maxChannels;
843  double * defaultLowLatency, * defaultHighLatency, * defaultSampleRate =
844  &devInfo->baseDeviceInfo.defaultSampleRate;
845  double defaultSr = *defaultSampleRate;
846  int dir;
847 
848  assert( pcm );
849 
850  PA_DEBUG(( "%s: collecting info ..\n", __FUNCTION__ ));
851 
852  if( StreamDirection_In == mode )
853  {
854  minChannels = &devInfo->minInputChannels;
855  maxChannels = &devInfo->baseDeviceInfo.maxInputChannels;
856  defaultLowLatency = &devInfo->baseDeviceInfo.defaultLowInputLatency;
857  defaultHighLatency = &devInfo->baseDeviceInfo.defaultHighInputLatency;
858  }
859  else
860  {
861  minChannels = &devInfo->minOutputChannels;
862  maxChannels = &devInfo->baseDeviceInfo.maxOutputChannels;
863  defaultLowLatency = &devInfo->baseDeviceInfo.defaultLowOutputLatency;
864  defaultHighLatency = &devInfo->baseDeviceInfo.defaultHighOutputLatency;
865  }
866 
867  ENSURE_( alsa_snd_pcm_nonblock( pcm, 0 ), paUnanticipatedHostError );
868 
869  alsa_snd_pcm_hw_params_alloca( &hwParams );
870  alsa_snd_pcm_hw_params_any( pcm, hwParams );
871 
872  if( defaultSr >= 0 )
873  {
874  /* Could be that the device opened in one mode supports samplerates that the other mode wont have,
875  * so try again .. */
876  if( SetApproximateSampleRate( pcm, hwParams, defaultSr ) < 0 )
877  {
878  defaultSr = -1.;
879  alsa_snd_pcm_hw_params_any( pcm, hwParams ); /* Clear any params (rate) that might have been set */
880  PA_DEBUG(( "%s: Original default samplerate failed, trying again ..\n", __FUNCTION__ ));
881  }
882  }
883 
884  if( defaultSr < 0. ) /* Default sample rate not set */
885  {
886  unsigned int sampleRate = 44100; /* Will contain approximate rate returned by alsa-lib */
887 
888  /* Don't allow rate resampling when probing for the default rate (but ignore if this call fails) */
889  alsa_snd_pcm_hw_params_set_rate_resample( pcm, hwParams, 0 );
890  if( alsa_snd_pcm_hw_params_set_rate_near( pcm, hwParams, &sampleRate, NULL ) < 0 )
891  {
892  result = paUnanticipatedHostError;
893  goto error;
894  }
895  ENSURE_( GetExactSampleRate( hwParams, &defaultSr ), paUnanticipatedHostError );
896  }
897 
898  ENSURE_( alsa_snd_pcm_hw_params_get_channels_min( hwParams, &minChans ), paUnanticipatedHostError );
899  ENSURE_( alsa_snd_pcm_hw_params_get_channels_max( hwParams, &maxChans ), paUnanticipatedHostError );
900  assert( maxChans <= INT_MAX );
901  assert( maxChans > 0 ); /* Weird linking issue could cause wrong version of ALSA symbols to be called,
902  resulting in zeroed values */
903 
904  /* XXX: Limit to sensible number (ALSA plugins accept a crazy amount of channels)? */
905  if( isPlug && maxChans > 128 )
906  {
907  maxChans = 128;
908  PA_DEBUG(( "%s: Limiting number of plugin channels to %u\n", __FUNCTION__, maxChans ));
909  }
910 
911  /* TWEAKME:
912  * Giving values for default min and max latency is not straightforward.
913  * * for low latency, we want to give the lowest value that will work reliably.
914  * This varies based on the sound card, kernel, CPU, etc. Better to give
915  * sub-optimal latency than to give a number too low and cause dropouts.
916  * * for high latency we want to give a large enough value that dropouts are basically impossible.
917  * This doesn't really require as much tweaking, since providing too large a number will
918  * just cause us to select the nearest setting that will work at stream config time.
919  */
920  /* Try low latency values, (sometimes the buffer & period that result are larger) */
921  alsaBufferFrames = 512;
922  alsaPeriodFrames = 128;
923  ENSURE_( alsa_snd_pcm_hw_params_set_buffer_size_near( pcm, hwParams, &alsaBufferFrames ), paUnanticipatedHostError );
924  ENSURE_( alsa_snd_pcm_hw_params_set_period_size_near( pcm, hwParams, &alsaPeriodFrames, &dir ), paUnanticipatedHostError );
925  *defaultLowLatency = (double) (alsaBufferFrames - alsaPeriodFrames) / defaultSr;
926 
927  /* Base the high latency case on values four times larger */
928  alsaBufferFrames = 2048;
929  alsaPeriodFrames = 512;
930  /* Have to reset hwParams, to set new buffer size; need to also set sample rate again */
931  ENSURE_( alsa_snd_pcm_hw_params_any( pcm, hwParams ), paUnanticipatedHostError );
932  ENSURE_( SetApproximateSampleRate( pcm, hwParams, defaultSr ), paUnanticipatedHostError );
933  ENSURE_( alsa_snd_pcm_hw_params_set_buffer_size_near( pcm, hwParams, &alsaBufferFrames ), paUnanticipatedHostError );
934  ENSURE_( alsa_snd_pcm_hw_params_set_period_size_near( pcm, hwParams, &alsaPeriodFrames, &dir ), paUnanticipatedHostError );
935  *defaultHighLatency = (double) (alsaBufferFrames - alsaPeriodFrames) / defaultSr;
936 
937  *minChannels = (int)minChans;
938  *maxChannels = (int)maxChans;
939  *defaultSampleRate = defaultSr;
940 
941 end:
942  alsa_snd_pcm_close( pcm );
943  return result;
944 
945 error:
946  goto end;
947 }

1082 /* Skip past parts at the beginning of a (pcm) info name that are already in the card name, to avoid duplication */
1083 static char *SkipCardDetailsInName( char *infoSkipName, char *cardRefName )
1084 {
1085  char *lastSpacePosn = infoSkipName;
1086 
1087  /* Skip matching chars; but only in chunks separated by ' ' (not part words etc), so track lastSpacePosn */
1088  while( *cardRefName )
1089  {
1090  while( *infoSkipName && *cardRefName && *infoSkipName == *cardRefName)
1091  {
1092  infoSkipName++;
1093  cardRefName++;
1094  if( *infoSkipName == ' ' || *infoSkipName == '\0' )
1095  lastSpacePosn = infoSkipName;
1096  }
1097  infoSkipName = lastSpacePosn;
1098  /* Look for another chunk; post-increment means ends pointing to next char */
1099  while( *cardRefName && ( *cardRefName++ != ' ' ));
1100  }
1101  if( *infoSkipName == '\0' )
1102  return "-"; /* The 2 names were identical; instead of a nul-string, return a marker string */
1103 
1104  /* Now want to move to the first char after any spaces */
1105  while( *lastSpacePosn && *lastSpacePosn == ' ' )
1106  lastSpacePosn++;
1107  /* Skip a single separator char if present in the remaining pcm name; (pa will add its own) */
1108  if(( *lastSpacePosn == '-' || *lastSpacePosn == ':' ) && *(lastSpacePosn + 1) == ' ' )
1109  lastSpacePosn += 2;
1110 
1111  return lastSpacePosn;
1112 }

1747 static PaError TestParameters( const PaUtilHostApiRepresentation *hostApi, const PaStreamParameters *parameters,
1748  double sampleRate, StreamDirection streamDir )
1749 {
1750  PaError result = paNoError;
1751  snd_pcm_t *pcm = NULL;
1752  PaSampleFormat availableFormats;
1753  /* We are able to adapt to a number of channels less than what the device supports */
1754  unsigned int numHostChannels;
1755  PaSampleFormat hostFormat;
1756  snd_pcm_hw_params_t *hwParams;
1757  alsa_snd_pcm_hw_params_alloca( &hwParams );
1758 
1759  if( !parameters->hostApiSpecificStreamInfo )
1760  {
1761  const PaAlsaDeviceInfo *devInfo = GetDeviceInfo( hostApi, parameters->device );
1762  numHostChannels = PA_MAX( parameters->channelCount, StreamDirection_In == streamDir ?
1763  devInfo->minInputChannels : devInfo->minOutputChannels );
1764  }
1765  else
1766  numHostChannels = parameters->channelCount;
1767

1769 
1770  alsa_snd_pcm_hw_params_any( pcm, hwParams );
1771 
1772  if( SetApproximateSampleRate( pcm, hwParams, sampleRate ) < 0 )
1773  {
1774  result = paInvalidSampleRate;
1775  goto error;
1776  }

1779  {
1780  result = paInvalidChannelCount;
1781  goto error;
1782  }
1783

1785  availableFormats = GetAvailableFormats( pcm );
1786  PA_ENSURE( hostFormat = PaUtil_SelectClosestAvailableFormat( availableFormats, parameters->sampleFormat ) );
1787 
1788  /* Some specific hardware (reported: Audio8 DJ) can fail with assertion during this step. */
1789  ENSURE_( alsa_snd_pcm_hw_params_set_format( pcm, hwParams, Pa2AlsaFormat( hostFormat ) ), paUnanticipatedHostError );
1790 
1791  {
1792  /* It happens that this call fails because the device is busy */
1793  int ret = 0;
1794  if( ( ret = alsa_snd_pcm_hw_params( pcm, hwParams ) ) < 0 )
1795  {
1796  if( -EINVAL == ret )
1797  {
1798  /* Don't know what to return here */
1799  result = paBadIODeviceCombination;
1800  goto error;
1801  }
1802  else if( -EBUSY == ret )
1803  {
1804  result = paDeviceUnavailable;

1806  }
1807  else
1808  {
1809  result = paUnanticipatedHostError;
1810  }
1811 
1812  ENSURE_( ret, result );
1813  }
1814  }
1815 
1816 end:
1817  if( pcm )
1818  {
1819  alsa_snd_pcm_close( pcm );
1820  }
1821  return result;
1822 
1823 error:
1824  goto end;
1825 }

1873  const PaStreamParameters *params, StreamDirection streamDir, int callbackMode )

1958 static PaError PaAlsaStreamComponent_InitialConfigure( PaAlsaStreamComponent *self, const PaStreamParameters *params,
1959  int primeBuffers, snd_pcm_hw_params_t *hwParams, double *sampleRate )
1960 {
1961  /* Configuration consists of setting all of ALSA's parameters.
1962  * These parameters come in two flavors: hardware parameters
1963  * and software paramters. Hardware parameters will affect
1964  * the way the device is initialized, software parameters
1965  * affect the way ALSA interacts with me, the user-level client.
1966  */
1967 
1968  PaError result = paNoError;
1969  snd_pcm_access_t accessMode, alternateAccessMode;
1970  int dir = 0;
1971  snd_pcm_t *pcm = self->pcm;
1972  double sr = *sampleRate;
1973  unsigned int minPeriods = 2;
1974 
1975  /* self->framesPerPeriod = framesPerHostBuffer; */
1976 
1977  /* ... fill up the configuration space with all possibile
1978  * combinations of parameters this device will accept */
1979  ENSURE_( alsa_snd_pcm_hw_params_any( pcm, hwParams ), paUnanticipatedHostError );
1980 
1981  ENSURE_( alsa_snd_pcm_hw_params_set_periods_integer( pcm, hwParams ), paUnanticipatedHostError );
1982  /* I think there should be at least 2 periods (even though ALSA doesn't appear to enforce this) */
1983  dir = 0;
1984  ENSURE_( alsa_snd_pcm_hw_params_set_periods_min( pcm, hwParams, &minPeriods, &dir ), paUnanticipatedHostError );
1985 
1986  if( self->userInterleaved )
1987  {
1988  accessMode = SND_PCM_ACCESS_MMAP_INTERLEAVED;
1989  alternateAccessMode = SND_PCM_ACCESS_MMAP_NONINTERLEAVED;
1990 
1991  /* test if MMAP supported */
1992  self->canMmap = alsa_snd_pcm_hw_params_test_access( pcm, hwParams, accessMode ) >= 0 ||
1993  alsa_snd_pcm_hw_params_test_access( pcm, hwParams, alternateAccessMode ) >= 0;
1994 
1995  PA_DEBUG(( "%s: device MMAP SND_PCM_ACCESS_MMAP_INTERLEAVED: %s\n", __FUNCTION__, ( alsa_snd_pcm_hw_params_test_access( pcm, hwParams, accessMode ) >= 0 ? "YES" : "NO" ) ));
1996  PA_DEBUG(( "%s: device MMAP SND_PCM_ACCESS_MMAP_NONINTERLEAVED: %s\n", __FUNCTION__, ( alsa_snd_pcm_hw_params_test_access( pcm, hwParams, alternateAccessMode ) >= 0 ? "YES" : "NO" ) ));
1997 
1998  if( !self->canMmap )
1999  {
2000  accessMode = SND_PCM_ACCESS_RW_INTERLEAVED;
2001  alternateAccessMode = SND_PCM_ACCESS_RW_NONINTERLEAVED;
2002  }
2003  }
2004  else
2005  {
2006  accessMode = SND_PCM_ACCESS_MMAP_NONINTERLEAVED;
2007  alternateAccessMode = SND_PCM_ACCESS_MMAP_INTERLEAVED;
2008 
2009  /* test if MMAP supported */
2010  self->canMmap = alsa_snd_pcm_hw_params_test_access( pcm, hwParams, accessMode ) >= 0 ||
2011  alsa_snd_pcm_hw_params_test_access( pcm, hwParams, alternateAccessMode ) >= 0;
2012 
2013  PA_DEBUG((" %s: device MMAP SND_PCM_ACCESS_MMAP_NONINTERLEAVED: %s\n", __FUNCTION__, ( alsa_snd_pcm_hw_params_test_access( pcm, hwParams, accessMode ) >= 0 ? "YES" : "NO" ) ));
2014  PA_DEBUG(( "%s: device MMAP SND_PCM_ACCESS_MMAP_INTERLEAVED: %s\n", __FUNCTION__, ( alsa_snd_pcm_hw_params_test_access( pcm, hwParams, alternateAccessMode ) >= 0 ? "YES" : "NO" ) ));
2015 
2016  if( !self->canMmap )
2017  {
2018  accessMode = SND_PCM_ACCESS_RW_NONINTERLEAVED;
2019  alternateAccessMode = SND_PCM_ACCESS_RW_INTERLEAVED;
2020  }
2021  }
2022 
2023  PA_DEBUG(( "%s: device can MMAP: %s\n", __FUNCTION__, ( self->canMmap ? "YES" : "NO" ) ));
2024 
2025  /* If requested access mode fails, try alternate mode */
2026  if( alsa_snd_pcm_hw_params_set_access( pcm, hwParams, accessMode ) < 0 )
2027  {
2028  int err = 0;
2029  if( ( err = alsa_snd_pcm_hw_params_set_access( pcm, hwParams, alternateAccessMode )) < 0 )
2030  {
2031  result = paUnanticipatedHostError;
2032  PaUtil_SetLastHostErrorInfo( paALSA, err, alsa_snd_strerror( err ) );
2033  goto error;
2034  }
2035  /* Flip mode */
2036  self->hostInterleaved = !self->userInterleaved;
2037  }
2038 
2039  /* Some specific hardware (reported: Audio8 DJ) can fail with assertion during this step. */
2040  ENSURE_( alsa_snd_pcm_hw_params_set_format( pcm, hwParams, self->nativeFormat ), paUnanticipatedHostError );
2041 
2042  if( ( result = SetApproximateSampleRate( pcm, hwParams, sr )) != paUnanticipatedHostError )
2043  {
2044  ENSURE_( GetExactSampleRate( hwParams, &sr ), paUnanticipatedHostError );
2045  if( result == paInvalidSampleRate ) /* From the SetApproximateSampleRate() call above */
2046  { /* The sample rate was returned as 'out of tolerance' of the one requested */
2047  PA_DEBUG(( "%s: Wanted %.3f, closest sample rate was %.3f\n", __FUNCTION__, sampleRate, sr ));
2048  PA_ENSURE( paInvalidSampleRate );
2049  }
2050  }
2051  else
2052  {
2053  PA_ENSURE( paUnanticipatedHostError );
2054  }
2055 
2056  ENSURE_( alsa_snd_pcm_hw_params_set_channels( pcm, hwParams, self->numHostChannels ), paInvalidChannelCount );
2057 
2058  *sampleRate = sr;
2059 
2060 end:
2061  return result;
2062 
2063 error:
2064  /* No particular action */
2065  goto end;
