37 void ComputeLegacySummaryInfo(wxFileName fileName,
38  int summaryLen,
39  sampleFormat format,
40  SummaryInfo *info,
41  bool noRMS,bool Silent,
42  float *min, float *max, float *rms)
43 {
44  int fields = 3; /* min, max, rms */
45 
46  if (noRMS)
47  fields = 2;
48 
49  info->fields = fields;
50  info->format = format;
51  info->bytesPerFrame =
52  SAMPLE_SIZE(info->format) * fields;
53  info->totalSummaryBytes = summaryLen;
54  info->offset64K = 20; /* legacy header tag len */
55  info->frames64K = (summaryLen-20) /
56  (info->bytesPerFrame * 256);
57  info->offset256 = info->offset64K +
58  (info->frames64K * info->bytesPerFrame);
59  info->frames256 =
60  (summaryLen - 20 -
61  (info->frames64K * info->bytesPerFrame)) /
62  info->bytesPerFrame;
63 
64  //
65  // Compute the min, max, and RMS of the block from the
66  // 64K summary data
67  //
68 
69  float *summary = new float[info->frames64K * fields];
70  SampleBuffer data(info->frames64K * fields,
71  info->format);
72 
73  int read;
74  {
75  Maybe<wxLogNull> silence{};
76  wxFFile summaryFile(fileName.GetFullPath(), wxT("rb"));
77  if (Silent)
78  silence.create();
79 
80  if (!summaryFile.IsOpened()) {
81  wxLogWarning(wxT("Unable to access summary file %s; substituting silence for remainder of session"),
82  fileName.GetFullPath().c_str());
83 
84  read = info->frames64K * info->bytesPerFrame;
85  memset(data.ptr(), 0, read);
86  }
87  else{
88  summaryFile.Seek(info->offset64K);
89  read = summaryFile.Read(data.ptr(),
90  info->frames64K *
91  info->bytesPerFrame);
92  }
93  }
94 
95  int count = read / info->bytesPerFrame;
96 
97  CopySamples(data.ptr(), info->format,
98  (samplePtr)summary, floatSample, count);
99 
100  (*min) = FLT_MAX;
101  (*max) = FLT_MIN;
102  float sumsq = 0;
103 
104  for(int i=0; i<count; i++) {
105  if (summary[fields*i] < (*min))
106  (*min) = summary[fields*i];
107  if (summary[fields*i+1] > (*max))
108  (*max) = summary[fields*i+1];
109  if (fields >= 3)
110  sumsq += summary[fields*i+2]*summary[fields*i+2];
111  }
112  if (fields >= 3)
113  (*rms) = sqrt(sumsq / count);
114  else
115  (*rms) = 0;
116 
117  delete[] summary;
118 }

186 int LegacyBlockFile::ReadData(samplePtr data, sampleFormat format,
187  sampleCount start, sampleCount len)
188 {
189  SF_INFO info;
190 
191  memset(&info, 0, sizeof(info));
192 
193  switch(mFormat) {
194  case int16Sample:
195  info.format =
196  SF_FORMAT_RAW | SF_FORMAT_PCM_16 | SF_ENDIAN_CPU;
197  break;
198  default:
199  case floatSample:
200  info.format =
201  SF_FORMAT_RAW | SF_FORMAT_FLOAT | SF_ENDIAN_CPU;
202  break;
203  case int24Sample:
204  info.format = SF_FORMAT_RAW | SF_FORMAT_PCM_32 | SF_ENDIAN_CPU;
205  break;
206  }
207  info.samplerate = 44100; // Doesn't matter
208  info.channels = 1;
209  info.frames = mLen + (mSummaryInfo.totalSummaryBytes /
210  SAMPLE_SIZE(mFormat));
211 
212  wxFile f; // will be closed when it goes out of scope
213  SNDFILE *sf = NULL;
214 
215  if (f.Open(mFileName.GetFullPath())) {
216  // Even though there is an sf_open() that takes a filename, use the one that
217  // takes a file descriptor since wxWidgets can open a file with a Unicode name and
218  // libsndfile can't (under Windows).
219  sf = sf_open_fd(f.fd(), SFM_READ, &info, FALSE);
220  }
221 
222  {
223  Maybe<wxLogNull> silence{};
224  if (mSilentLog)
225  silence.create();
226 
227  if (!sf){
228 
229  memset(data, 0, SAMPLE_SIZE(format)*len);
230 
231  mSilentLog = TRUE;
232 
233  return len;
234  }
235  }
236  mSilentLog=FALSE;
237 
238  sf_count_t seekstart = start +
239  (mSummaryInfo.totalSummaryBytes / SAMPLE_SIZE(mFormat));
240  sf_seek(sf, seekstart , SEEK_SET);
241 
242  SampleBuffer buffer(len, floatSample);
243  int framesRead = 0;
244 
245  // If both the src and dest formats are integer formats,
246  // read integers from the file (otherwise we would be
247  // converting to float and back, which is unneccesary)
248  if (format == int16Sample &&
249  sf_subtype_is_integer(info.format)) {
250  framesRead = sf_readf_short(sf, (short *)data, len);
251  }else if (format == int24Sample &&
252  sf_subtype_is_integer(info.format)) {
253  framesRead = sf_readf_int(sf, (int *)data, len);
254 
255  // libsndfile gave us the 3 byte sample in the 3 most
256  // significant bytes -- we want it in the 3 least
257  // significant bytes.
258  int *intPtr = (int *)data;
259  for( int i = 0; i < framesRead; i++ )
260  intPtr[i] = intPtr[i] >> 8;
261  } else {
262  // Otherwise, let libsndfile handle the conversion and
263  // scaling, and pass us normalized data as floats. We can
264  // then convert to whatever format we want.
265  framesRead = sf_readf_float(sf, (float *)buffer.ptr(), len);
266  CopySamples(buffer.ptr(), floatSample,
267  (samplePtr)data, format, framesRead);
268  }
269 
270  sf_close(sf);
271 
272  return framesRead;
273 }
