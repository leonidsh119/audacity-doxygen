107 PaWinWaveFormatChannelMask PaWin_DefaultChannelMask( int numChannels )
108 {
109  switch( numChannels ){
110  case 1:
111  return PAWIN_SPEAKER_MONO;
112  case 2:
113  return PAWIN_SPEAKER_STEREO;
114  case 3:
115  return PAWIN_SPEAKER_FRONT_LEFT | PAWIN_SPEAKER_FRONT_CENTER | PAWIN_SPEAKER_FRONT_RIGHT;
116  case 4:
117  return PAWIN_SPEAKER_QUAD;
118  case 5:
119  return PAWIN_SPEAKER_QUAD | PAWIN_SPEAKER_FRONT_CENTER;
120  case 6:
121  /* The meaning of the PAWIN_SPEAKER_5POINT1 flag has changed over time:
122  http://msdn2.microsoft.com/en-us/library/aa474707.aspx
123  We use PAWIN_SPEAKER_5POINT1 (not PAWIN_SPEAKER_5POINT1_SURROUND)
124  because on some cards (eg Audigy) PAWIN_SPEAKER_5POINT1_SURROUND
125  results in a virtual mixdown placing the rear output in the
126  front _and_ rear speakers.
127  */
128  return PAWIN_SPEAKER_5POINT1;
129  /* case 7: */
130  case 8:
131  /* RoBi: PAWIN_SPEAKER_7POINT1_SURROUND fits normal surround sound setups better than PAWIN_SPEAKER_7POINT1, f.i. NVidia HDMI Audio
132  output is silent on channels 5&6 with NVidia drivers, and channel 7&8 with Micrsoft HD Audio driver using PAWIN_SPEAKER_7POINT1.
133  With PAWIN_SPEAKER_7POINT1_SURROUND both setups work OK. */
134  return PAWIN_SPEAKER_7POINT1_SURROUND;
135  }
136 
137  /* Apparently some Audigy drivers will output silence
138  if the direct-out constant (0) is used. So this is not ideal.
139 
140  RoBi 2012-12-19: Also, NVidia driver seem to output garbage instead. Again not very ideal.
141  */
142  return PAWIN_SPEAKER_DIRECTOUT;
143 
144  /* Note that Alec Rogers proposed the following as an alternate method to
145  generate the default channel mask, however it doesn't seem to be an improvement
146  over the above, since some drivers will matrix outputs mapping to non-present
147  speakers accross multiple physical speakers.
148 
149  if(nChannels==1) {
150  pwfFormat->dwChannelMask = SPEAKER_FRONT_CENTER;
151  }
152  else {
153  pwfFormat->dwChannelMask = 0;
154  for(i=0; i<nChannels; i++)
155  pwfFormat->dwChannelMask = (pwfFormat->dwChannelMask << 1) | 0x1;
156  }
157  */
