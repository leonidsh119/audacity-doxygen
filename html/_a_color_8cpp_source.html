161  dc.SetPen(wxPen(wxT("MEDIUM GREY"), 0, wxSOLID));
162 
163  // this seems to be closer than what Windows does than wxINVERT although
164  // I'm still not sure if it's correct
165  dc.SetLogicalFunction(wxAND_REVERSE);
166 
167  wxCoord z;
168  for ( z = x1 + 1; z < x2; z += 2 )
169  dc.DrawPoint(z, y1);
170 
171  wxCoord shift = z == x2 ? 0 : 1;
172  for ( z = y1 + shift; z < y2; z += 2 )
173  dc.DrawPoint(x2, z);
174 
175  shift = z == y2 ? 0 : 1;
176  for ( z = x2 - shift; z > x1; z -= 2 )
177  dc.DrawPoint(z, y2);
178 
179  shift = z == x1 ? 0 : 1;

581 void AColor::PreComputeGradient() {
582  {
583  if (!gradient_inited) {
584  gradient_inited = 1;
585 
586  for (int selected = 0; selected < ColorGradientTotal; selected++)
587  for (int grayscale = 0; grayscale <= 1; grayscale++) {
588  float r, g, b;
589 
590  int i;
591  for (i=0; i<gradientSteps; i++) {
592  float value = float(i)/gradientSteps;
593 
594  if (grayscale) {
595  r = g = b = 0.84 - 0.84 * value;
596  } else {
597  const int gsteps = 4;
598  float gradient[gsteps + 1][3] = {
599  {float(0.75), float(0.75), float(0.75)}, // lt gray
600  {float(0.30), float(0.60), float(1.00)}, // lt blue
601  {float(0.90), float(0.10), float(0.90)}, // violet
602  {float(1.00), float(0.00), float(0.00)}, // red
603  {float(1.00), float(1.00), float(1.00)} // white
604  };
605 
606  int left = int (value * gsteps);
607  int right = (left == gsteps ? gsteps : left + 1);
608 
609  float rweight = (value * gsteps) - left;
610  float lweight = 1.0 - rweight;
611 
612  r = (gradient[left][0] * lweight) + (gradient[right][0] * rweight);
613  g = (gradient[left][1] * lweight) + (gradient[right][1] * rweight);
614  b = (gradient[left][2] * lweight) + (gradient[right][2] * rweight);
615  }
616 
617  switch (selected) {
618  case ColorGradientUnselected:
619  // not dimmed
620  break;
621 
622  case ColorGradientTimeAndFrequencySelected:
623  if( !grayscale )
624  {
625  float temp;
626  temp = r;
627  r = g;
628  g = b;
629  b = temp;
630  break;
631  }
632  // else fall through to SAME grayscale colour as normal selection.
633  // The white lines show it up clearly enough.
634 
635  case ColorGradientTimeSelected:
636  // partly dimmed
637  r *= 0.75f;
638  g *= 0.75f;
639  b *= 0.75f;
640  break;
641 
642 
643  // For now edge colour is just black (or white if grey-scale)
644  // Later we might invert or something else funky.
645  case ColorGradientEdge:
646  // fully dimmed
647  r = 1.0f * grayscale;
648  g = 1.0f * grayscale;
649  b = 1.0f * grayscale;
650  break;
651  }
652  gradient_pre[selected][grayscale][i][0] = (unsigned char) (255 * r);
653  gradient_pre[selected][grayscale][i][1] = (unsigned char) (255 * g);
654  gradient_pre[selected][grayscale][i][2] = (unsigned char) (255 * b);
655  }
656  }
657  }
658  }
