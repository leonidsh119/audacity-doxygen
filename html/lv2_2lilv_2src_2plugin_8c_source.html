163

165  lilv_node_free(prototype);
166  }

221 static void
222 lilv_plugin_load_ports_if_necessary(const LilvPlugin* const_p)
223 {
224  LilvPlugin* p = (LilvPlugin*)const_p;
225  if (!p->loaded)
226  lilv_plugin_load(p);
227 
228  if (!p->ports) {
229  p->ports = (LilvPort**)malloc(sizeof(LilvPort*));
230  p->ports[0] = NULL;
231 
232  SordIter* ports = lilv_world_query_internal(
233  p->world,
234  p->plugin_uri->node,
235  p->world->uris.lv2_port,
236  NULL);
237 
238  FOREACH_MATCH(ports) {
239  const SordNode* port = sord_iter_get_node(ports, SORD_OBJECT);
240  LilvNode* index = lilv_plugin_get_unique(
241  p, port, p->world->uris.lv2_index);
242  LilvNode* symbol = lilv_plugin_get_unique(
243  p, port, p->world->uris.lv2_symbol);
244 
245  bool error = false;
246  if (!lilv_node_is_string(symbol) ||
247  !is_symbol((const char*)sord_node_get_string(symbol->node))) {
248  LILV_ERRORF("Plugin <%s> port symbol `%s' is invalid\n",
249  lilv_node_as_uri(p->plugin_uri),
250  lilv_node_as_string(symbol));
251  error = true;
252  goto done;
253  }
254 
255  if (!lilv_node_is_int(index)) {
256  LILV_ERRORF("Plugin <%s> port index is not an integer\n",
257  lilv_node_as_uri(p->plugin_uri));
258  error = true;
259  goto done;
260  }
261 
262  uint32_t this_index = lilv_node_as_int(index);
263  LilvPort* this_port = NULL;
264  if (p->num_ports > this_index) {
265  this_port = p->ports[this_index];
266  } else {
267  p->ports = (LilvPort**)realloc(
268  p->ports, (this_index + 1) * sizeof(LilvPort*));
269  memset(p->ports + p->num_ports, '\0',
270  (this_index - p->num_ports) * sizeof(LilvPort*));
271  p->num_ports = this_index + 1;
272  }
273 
274  // Havn't seen this port yet, add it to array
275  if (!this_port) {
276  this_port = lilv_port_new(p->world,
277  port,
278  this_index,
279  lilv_node_as_string(symbol));
280  p->ports[this_index] = this_port;
281  }
282 
283  SordIter* types = lilv_world_query_internal(
284  p->world, port, p->world->uris.rdf_a, NULL);
285  FOREACH_MATCH(types) {
286  const SordNode* type = sord_iter_get_node(types, SORD_OBJECT);
287  if (sord_node_get_type(type) == SORD_URI) {
288  zix_tree_insert(
289  (ZixTree*)this_port->classes,
290  lilv_node_new_from_node(p->world, type), NULL);
291  } else {
292  LILV_WARNF("Plugin <%s> port type is not a URI\n",
293  lilv_node_as_uri(p->plugin_uri));
294  }
295  }
296  sord_iter_free(types);
297 
298  done:
299  lilv_node_free(symbol);
300  lilv_node_free(index);
301  if (error) { // Invalid plugin
302  lilv_plugin_free_ports(p);
303  break;
304  }
305  }
306  sord_iter_free(ports);
307 
308  // Check sanity
309  for (uint32_t i = 0; i < p->num_ports; ++i) {
310  if (!p->ports[i]) {
311  LILV_ERRORF("Plugin <%s> is missing port %d/%d\n",
312  lilv_node_as_uri(p->plugin_uri), i, p->num_ports);
313  lilv_plugin_free_ports(p);
314  break;
315  }
316  }
317  }
318 }
