59  // Iterate through all Modules listed in prefs.
60  // Get their names and values.
61  gPrefs->SetPath( wxT("Module/") );
62  bool bCont = gPrefs->GetFirstEntry(str, dummy);
63  while ( bCont ) {
64  int iStatus;
65  gPrefs->Read( str, &iStatus, kModuleDisabled );
66  wxString fname;
67  gPrefs->Read( wxString( wxT("/ModulePath/") ) + str, &fname, wxEmptyString );
68  if( fname != wxEmptyString && wxFileExists( fname ) ){
69  if( iStatus > kModuleNew ){
70  iStatus = kModuleNew;
71  gPrefs->Write( str, iStatus );
72  }
73  //wxLogDebug( wxT("Entry: %s Value: %i"), str.c_str(), iStatus );
74  mModules.Add( str );
75  mStatuses.Add( iStatus );
76  mPaths.Add( fname );
77  }
78  bCont = gPrefs->GetNextEntry(str, dummy);
79  }
