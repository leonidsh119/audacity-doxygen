151 double EffectTimeScale::CalcPreviewInputLength(double previewLength)
152 {
153  double inputLength = Effect::GetDuration();
154  if(inputLength == 0.0) {
155  return 0.0;
156  } else {
157  double rateStart = PercentChangeToRatio(m_RatePercentChangeStart);
158  double rateEnd = PercentChangeToRatio(m_RatePercentChangeEnd);
159  double tOut = previewLength/inputLength;
160  double t = EffectSBSMS::getInvertedStretchedTime(rateStart,rateEnd,slideTypeRate,tOut);
161  return t * inputLength;
162  }
163 }

173 bool EffectTimeScale::Process()
174 {
175  double pitchStart = PercentChangeToRatio(m_PitchPercentChangeStart);
176  double pitchEnd = PercentChangeToRatio(m_PitchPercentChangeEnd);
177  double rateStart = PercentChangeToRatio(m_RatePercentChangeStart);
178  double rateEnd = PercentChangeToRatio(m_RatePercentChangeEnd);
179 
180  if(bPreview) {
181  double t = (mT1-mT0) / previewSelectedDuration;
182  rateEnd = EffectSBSMS::getRate(rateStart,rateEnd,slideTypeRate,t);
183  pitchEnd = EffectSBSMS::getRate(pitchStart,pitchEnd,slideTypePitch,t);
184  }
185 
186  EffectSBSMS::setParameters(rateStart,rateEnd,pitchStart,pitchEnd,slideTypeRate,slideTypePitch,false,false,false);
187  return EffectSBSMS::Process();
188 }
