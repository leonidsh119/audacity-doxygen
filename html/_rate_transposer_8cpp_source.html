212 // Transposes up the sample rate, causing the observed playback 'rate' of the
213 // sound to decrease
214 void RateTransposer::upsample(const SAMPLETYPE *src, uint nSamples)
215 {
216  uint count, sizeTemp, num;
217 
218  // If the parameter 'uRate' value is smaller than 'SCALE', first transpose
219  // the samples and then apply the anti-alias filter to remove aliasing.
220 
221  // First check that there's enough room in 'storeBuffer'
222  // (+16 is to reserve some slack in the destination buffer)
223  sizeTemp = (uint)((float)nSamples / fRate + 16.0f);
224 
225  // Transpose the samples, store the result into the end of "storeBuffer"
226  count = transpose(storeBuffer.ptrEnd(sizeTemp), src, nSamples);
227  storeBuffer.putSamples(count);
228 
229  // Apply the anti-alias filter to samples in "store output", output the
230  // result to "dest"
231  num = storeBuffer.numSamples();
232  count = pAAFilter->evaluate(outputBuffer.ptrEnd(num),
233  storeBuffer.ptrBegin(), num, (uint)numChannels);
234  outputBuffer.putSamples(count);
235 
236  // Remove the processed samples from "storeBuffer"
237  storeBuffer.receiveSamples(count);
238 }

241 // Transposes down the sample rate, causing the observed playback 'rate' of the
242 // sound to increase
243 void RateTransposer::downsample(const SAMPLETYPE *src, uint nSamples)
244 {
245  uint count, sizeTemp;
246 
247  // If the parameter 'uRate' value is larger than 'SCALE', first apply the
248  // anti-alias filter to remove high frequencies (prevent them from folding
249  // over the lover frequencies), then transpose.
250 
251  // Add the new samples to the end of the storeBuffer
252  storeBuffer.putSamples(src, nSamples);
253 
254  // Anti-alias filter the samples to prevent folding and output the filtered
255  // data to tempBuffer. Note : because of the FIR filter length, the
256  // filtering routine takes in 'filter_length' more samples than it outputs.
257  assert(tempBuffer.isEmpty());
258  sizeTemp = storeBuffer.numSamples();
259 
260  count = pAAFilter->evaluate(tempBuffer.ptrEnd(sizeTemp),
261  storeBuffer.ptrBegin(), sizeTemp, (uint)numChannels);
262 
263  if (count == 0) return;
264 
265  // Remove the filtered samples from 'storeBuffer'
266  storeBuffer.receiveSamples(count);
267 
268  // Transpose the samples (+16 is to reserve some slack in the destination buffer)
269  sizeTemp = (uint)((float)nSamples / fRate + 16.0f);
270  count = transpose(outputBuffer.ptrEnd(sizeTemp), tempBuffer.ptrBegin(), count);
271  outputBuffer.putSamples(count);
272 }

275 // Transposes sample rate by applying anti-alias filter to prevent folding.
276 // Returns amount of samples returned in the "dest" buffer.
277 // The maximum amount of samples that can be returned at a time is set by
278 // the 'set_returnBuffer_size' function.
279 void RateTransposer::processSamples(const SAMPLETYPE *src, uint nSamples)
280 {
281  uint count;
282  uint sizeReq;
283 
284  if (nSamples == 0) return;
285  assert(pAAFilter);
286 
287  // If anti-alias filter is turned off, simply transpose without applying
288  // the filter
289  if (bUseAAFilter == FALSE)
290  {
291  sizeReq = (uint)((float)nSamples / fRate + 1.0f);
292  count = transpose(outputBuffer.ptrEnd(sizeReq), src, nSamples);
293  outputBuffer.putSamples(count);
294  return;
295  }
296 
297  // Transpose with anti-alias filter
298  if (fRate < 1.0f)
299  {
300  upsample(src, nSamples);
301  }
302  else
303  {
304  downsample(src, nSamples);
305  }
306 }

393 // Transposes the sample rate of the given samples using linear interpolation.
394 // 'Mono' version of the routine. Returns the number of samples returned in
395 // the "dest" buffer
396 uint RateTransposerInteger::transposeMono(SAMPLETYPE *dest, const SAMPLETYPE *src, uint nSamples)
397 {
398  unsigned int i, used;
399  LONG_SAMPLETYPE temp, vol1;
400 
401  if (nSamples == 0) return 0; // no samples, no work
402 
403  used = 0;
404  i = 0;
405 
406  // Process the last sample saved from the previous call first...
407  while (iSlopeCount <= SCALE)
408  {
409  vol1 = (LONG_SAMPLETYPE)(SCALE - iSlopeCount);
410  temp = vol1 * sPrevSampleL + iSlopeCount * src[0];
411  dest[i] = (SAMPLETYPE)(temp / SCALE);
412  i++;
413  iSlopeCount += iRate;
414  }
415  // now always (iSlopeCount > SCALE)
416  iSlopeCount -= SCALE;
417 
418  while (1)
419  {
420  while (iSlopeCount > SCALE)
421  {
422  iSlopeCount -= SCALE;
423  used ++;
424  if (used >= nSamples - 1) goto end;
425  }
426  vol1 = (LONG_SAMPLETYPE)(SCALE - iSlopeCount);
427  temp = src[used] * vol1 + iSlopeCount * src[used + 1];
428  dest[i] = (SAMPLETYPE)(temp / SCALE);
429 
430  i++;
431  iSlopeCount += iRate;
432  }
433 end:
434  // Store the last sample for the next round
435  sPrevSampleL = src[nSamples - 1];
436 
437  return i;
438 }

444 uint RateTransposerInteger::transposeStereo(SAMPLETYPE *dest, const SAMPLETYPE *src, uint nSamples)

475  }

477  vol1 = (LONG_SAMPLETYPE)(SCALE - iSlopeCount);
