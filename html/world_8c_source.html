354 lilv_world_add_plugin(LilvWorld* world,

386  ++((LilvDynManifest*)dynmanifest)->refs;
387  }
388 #endif
389 
390  // Add all plugin data files (rdfs:seeAlso)
391  SordIter* files = sord_search(world->model,
392  plugin_node,
393  world->uris.rdfs_seeAlso,
394  NULL,
395  NULL);
396  FOREACH_MATCH(files) {

398  zix_tree_insert((ZixTree*)plugin->data_uris,

400  NULL);

402  sord_iter_free(files);
403 
404  // Add plugin to world plugin sequence
405  zix_tree_insert((ZixTree*)world->plugins, plugin, NULL);

423 static void
424 lilv_world_load_dyn_manifest(LilvWorld* world,
425  SordNode* bundle_node,
426  const LilvNode* manifest)
427 {
428 #ifdef LILV_DYN_MANIFEST
429  if (!world->opt.dyn_manifest) {
430  return;
431  }
432 
433  typedef void* LV2_Dyn_Manifest_Handle;
434  LV2_Dyn_Manifest_Handle handle = NULL;
435 
436  // ?dman a dynman:DynManifest
437  SordIter* dmanifests = sord_search(world->model,
438  NULL,
439  world->uris.rdf_a,
440  world->uris.dman_DynManifest,
441  bundle_node);
442  FOREACH_MATCH(dmanifests) {
443  const SordNode* dmanifest = sord_iter_get_node(dmanifests, SORD_SUBJECT);
444 
445  // ?dman lv2:binary ?binary
446  SordIter* binaries = sord_search(world->model,
447  dmanifest,
448  world->uris.lv2_binary,
449  NULL,
450  bundle_node);
451  if (sord_iter_end(binaries)) {
452  sord_iter_free(binaries);
453  LILV_ERRORF("Dynamic manifest in <%s> has no binaries, ignored\n",
454  sord_node_get_string(bundle_node));
455  continue;
456  }
457 
458  // Get binary path
459  const SordNode* binary = sord_iter_get_node(binaries, SORD_OBJECT);
460  const uint8_t* lib_uri = sord_node_get_string(binary);
461  const char* lib_path = lilv_uri_to_path((const char*)lib_uri);
462  if (!lib_path) {
463  LILV_ERROR("No dynamic manifest library path\n");
464  sord_iter_free(binaries);
465  continue;
466  }
467 
468  // Open library
469  void* lib = dlopen(lib_path, RTLD_LAZY);
470  if (!lib) {
471  LILV_ERRORF("Failed to open dynmanifest library `%s'\n", lib_path);
472  sord_iter_free(binaries);
473  continue;
474  }
475 
476  // Open dynamic manifest
477  typedef int (*OpenFunc)(LV2_Dyn_Manifest_Handle*,
478  const LV2_Feature *const *);
479  OpenFunc dmopen = (OpenFunc)lilv_dlfunc(lib, "lv2_dyn_manifest_open");
480  if (!dmopen || dmopen(&handle, &dman_features)) {
481  LILV_ERRORF("No `lv2_dyn_manifest_open' in `%s'\n", lib_path);
482  sord_iter_free(binaries);
483  dlclose(lib);
484  continue;
485  }
486 
487  // Get subjects (the data that would be in manifest.ttl)
488  typedef int (*GetSubjectsFunc)(LV2_Dyn_Manifest_Handle, FILE*);
489  GetSubjectsFunc get_subjects_func = (GetSubjectsFunc)lilv_dlfunc(
490  lib, "lv2_dyn_manifest_get_subjects");
491  if (!get_subjects_func) {
492  LILV_ERRORF("No `lv2_dyn_manifest_get_subjects' in `%s'\n",
493  lib_path);
494  sord_iter_free(binaries);
495  dlclose(lib);
496  continue;
497  }
498 
499  LilvDynManifest* desc = malloc(sizeof(LilvDynManifest));
500  desc->bundle = lilv_node_new_from_node(world, bundle_node);
501  desc->lib = lib;
502  desc->handle = handle;
503  desc->refs = 0;
504 
505  // Generate data file
506  FILE* fd = tmpfile();
507  get_subjects_func(handle, fd);
508  rewind(fd);
509 
510  // Parse generated data file
511  const SerdNode* base = sord_node_to_serd_node(dmanifest);
512  SerdEnv* env = serd_env_new(base);
513  SerdReader* reader = sord_new_reader(
514  world->model, env, SERD_TURTLE, sord_node_copy(dmanifest));
515  serd_reader_add_blank_prefix(reader,
516  lilv_world_blank_node_prefix(world));
517  serd_reader_read_file_handle(reader, fd,
518  (const uint8_t*)"(dyn-manifest)");
519  serd_reader_free(reader);
520  serd_env_free(env);
521 
522  // Close (and automatically delete) temporary data file
523  fclose(fd);
524 
525  // ?plugin a lv2:Plugin
526  SordIter* plug_results = sord_search(
527  world->model,
528  NULL,
529  world->uris.rdf_a,
530  world->uris.lv2_Plugin,
531  dmanifest);
532  FOREACH_MATCH(plug_results) {
533  const SordNode* plug = sord_iter_get_node(plug_results, SORD_SUBJECT);
534  lilv_world_add_plugin(world, plug, manifest, desc, bundle_node);
535  }
536  sord_iter_free(plug_results);
537 
538  sord_iter_free(binaries);
539  }
540  sord_iter_free(dmanifests);
541 #endif // LILV_DYN_MANIFEST
542 }

584  lilv_world_add_plugin(world, plug, manifest, NULL, bundle_node);
585  }

592  NULL,
