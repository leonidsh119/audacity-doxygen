103 bool Module::Load()
104 {
105  if (mLib->IsLoaded()) {
106  if (mDispatch) {
107  return true;
108  }
109  return false;
110  }
111 
112  if (!mLib->Load(mName, wxDL_LAZY)) {
113  return false;
114  }
115 
116  // Check version string matches. (For now, they must match exactly)
117  tVersionFn versionFn = (tVersionFn)(mLib->GetSymbol(wxT(versionFnName)));
118  if (versionFn == NULL){
119  wxString ShortName = wxFileName( mName ).GetName();
120  wxMessageBox(wxString::Format(_("The module %s does not provide a version string.\nIt will not be loaded."), ShortName.c_str()), _("Module Unsuitable"));
121  wxLogMessage(wxString::Format(_("The module %s does not provide a version string. It will not be loaded."), mName.c_str()));
122  mLib->Unload();
123  return false;
124  }
125 
126  wxString moduleVersion = versionFn();
127  if( !moduleVersion.IsSameAs(AUDACITY_VERSION_STRING)) {
128  wxString ShortName = wxFileName( mName ).GetName();
129  wxMessageBox(wxString::Format(_("The module %s is matched with Audacity version %s.\n\nIt will not be loaded."), ShortName.c_str(), moduleVersion.c_str()), _("Module Unsuitable"));
130  wxLogMessage(wxString::Format(_("The module %s is matched with Audacity version %s. It will not be loaded."), mName.c_str(), moduleVersion.c_str()));
131  mLib->Unload();
132  return false;
133  }
134 
135  mDispatch = (fnModuleDispatch) mLib->GetSymbol(wxT(ModuleDispatchName));
136  if (!mDispatch) {
137  // Module does not provide a dispatch function...
138  // That can be OK, as long as we never try to call it.
139  return true;
140  }
141 
142  // However if we do have it and it does not work,
143  // then the module is bad.
144  bool res = ((mDispatch(ModuleInitialize))!=0);
145  if (res) {
146  return true;
147  }
148 
149  mDispatch = NULL;
150  return false;
151 }
