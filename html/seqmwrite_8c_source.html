276 private void smfw_process_event(seq)
277  seq_type seq;
278 {
279  register event_type event;
280  if (!seq->runflag) return;
281  while ((event = seq->current) && (event->ntime <= virttime)) {
282  unsigned int voice;
283  if ((vc_voice(event->nvoice) == smfw_seq.track) || /*if on current track*/
284  (((vc_voice(event->nvoice) - 16) == smfw_seq.track)
285  && (smfw_seq.track > 0)) ||
286  /* acknowledge clock change on all tracks*/
287  (event->value == CLOCK_VALUE && vc_ctrl(event->nvoice) == ESC_CTRL)) {
288 
289  /* process all current (and earlier) events */
290  if (is_note(event)) { /*** play a note or rest ***/
291  /* if this note is not a rest, play it and schedule an off event */
292  if (event->value != NO_PITCH &&
293  (seq_channel_mask(seq) &
294  (1 << ((voice = vc_voice(event->nvoice)) - 1)))) {
295  seq_noteon(seq, voice, (0xFF & event->value),
296  (int) (event->u.note.ndur & 0xFF));
297 
298  seq_cause_noteoff(seq, (event->u.note.ndur) >> 8,
299  voice, (0xFF & event->value));
300  }
301  } else { /*** send a control command ***/
302  int n;
303  time_type step;
304  int delta;
305  int increment;
306  int voice = vc_voice(event->nvoice);
307  ulong enabled = seq_channel_mask(seq) & (1 << (voice - 1));
308 
309  switch (vc_ctrl(event->nvoice)) {
310  case PSWITCH_CTRL:
311  if (!enabled) break;
312  if(debug) gprintf(TRANS, "porta %d (time:%ld)... ", event->value, virttime);
313  seq_midi_ctrl(seq, voice, PORTASWITCH, 0xFF & event->value);
314  break;
315  case MODWHEEL_CTRL:
316  if (!enabled) break;
317  if(debug) gprintf(TRANS, "modw %d (time:%ld)...", event->value, virttime);
318  seq_midi_ctrl(seq, voice, MODWHEEL, 0xFF & event->value);
319  break;
320  case TOUCH_CTRL:
321  if (!enabled) break;
322  if(debug) gprintf(TRANS, "touch %d (time:%ld)... ", event->value, virttime);
323  seq_midi_touch(seq, voice, 0xFF & event->value);
324  break;
325  case VOLUME_CTRL:
326  if (!enabled) break;
327  if(debug) gprintf(TRANS, "ftvol %d (time:%ld)...", event->value, virttime);
328  seq_midi_ctrl(seq, voice, VOLUME, 0xFF & event->value);
329  break;
330  case BEND_CTRL:
331  if (!enabled) break;
332  if(debug) gprintf(TRANS, "bend %d (time:%ld)... ", event->value, virttime);
333  seq_midi_bend(seq, voice, event->value);
334  break;
335  case PROGRAM_CTRL:
336  if (!enabled) break;
337  if(debug) gprintf(TRANS, "prog %d (time:%ld)\n", event->value, virttime);
338  smfw_deltatime();
339  putc(MIDI_CH_PROGRAM | (voice - 1), smfw_seq.outfile);
340  putc(0xFF & event->value, smfw_seq.outfile);
341  break;
342  case ESC_CTRL:
343  switch (event->value) {
344  time_type this_event;
345  case CALL_VALUE: /*called routine will write to midifile in execution */
346  sequence = seq;
347  (*(event->u.call.routine))(event->u.call.args);
348  break;
349  case CLOCK_VALUE:
350  clock_ticksize = event->u.clock.ticksize;
351  if(debug) gprintf(TRANS, "clockevent! ticksize: %lu (time:%ld)\n",
352  clock_ticksize, virttime);
353 
354  if (virttime > 0) { /* any clock before this is already recorded in the header */
355  if (smfw_seq.track == 0) { /* record clock event on tempo track = 0 */
356  /* cause clock write in half a newtick, because it was written .5 tick early*/
357  cause((delay_type) (clock_ticksize >> 17), smfw_clock_event,
358  last_tick_size, clock_ticksize);
359  last_tick_size = clock_ticksize; /*set new ticksize*/
360  } else { /*not on tempo track*/
361  this_event = ((virttime - last_clock_event) *
362  ((2500L << 16) / last_tick_size)) / 100;
363  if(debug) gprintf(TRANS, "track != 0: Lastev: %ld Thisev: %ld NewLast: %ld\n",
364  last_event, this_event, this_event - last_event);
365  last_event = 0L - (this_event - last_event);
366  last_clock_event = virttime;
367  /*last_event is negative, so will be ADDED to next this_event*/
368 
369  last_tick_size = clock_ticksize;
370  }
371  } else if (debug) gprintf(TRANS, "IGNORED\n");/* if virttime <= 0 */
372  break;
373  case MACCTRL_VALUE:
374  if (!enabled) break;
375  if (debug) gprintf(TRANS, "MACCTRL %d: %d (time:%ld)\n",
376  event->u.macctrl.ctrl_number, event->u.macctrl.value, virttime);
377  smfw_deltatime();
378  putc(MIDI_CTRL | (voice - 1), smfw_seq.outfile);
379  putc(0xFF & event->u.macctrl.ctrl_number, smfw_seq.outfile);
380  putc(0xFF & event->u.macctrl.value, smfw_seq.outfile);
381  break;
382  case MACRO_VALUE:
383  if (!enabled) break;
384  if (debug) gprintf(TRANS, "MACRO sent to...\n");
385  smfw_send_macro(event->u.macro.definition,
386  voice, event->u.macro.parameter, -1, 0);
387  break;
388  case CTRLRAMP_VALUE:
389  case DEFRAMP_VALUE: {
390  int from, to;
391  if (!enabled) break;
392  step = event->u.ramp.step;
393  if (event->value == CTRLRAMP_VALUE) {
394  if(debug) gprintf(TRANS, "CTRLRAMP (time:%ld)...", virttime);
395  from = event->u.ramp.u.ctrl.from_value;
396  to = event->u.ramp.u.ctrl.to_value;
397  } else {
398  if (debug) gprintf(TRANS, "DEFRAMP (time:%ld)...", virttime);
399  from = event->u.ramp.u.def.parameter[
400  event->u.ramp.u.def.parm_num];
401  to = event->u.ramp.u.def.to_value;
402  }
403  delta = to - from;
404  increment = delta;
405  if (delta < 0) delta = -delta;
406  /* RBD - Note: Step is always non-zero */
407  n = event->u.ramp.dur / step;
408  increment = (increment << 8) / n;
409  smfw_ramp_event(seq, event, from << 8, to << 8,
410  increment, step, n);
411  seq->noteoff_count++;
412  break;
413  }
414  case SETI_VALUE:
415  seti_counter++; /*will be printed after writing is completed*/
416  *(event->u.seti.int_to_set) = event->u.seti.value;
417  break;
418  default:
419  gprintf(TRANS, "unexpected ESC_CTRL value\n");
420  break;
421  }
422  break;
423  default:
424  gprintf(TRANS, "unexpected seq data\n");
425  break;
426  }
427  }
428  }
429  seq->current = event->next;
430  }
431  if (seq->current) { /* if there is an event: delay, and then process again */
432  cause((delay_type)(event->ntime - virttime), smfw_process_event, seq);
433  }
434 }
