161 void ToolDock::LayoutToolBars()
162 {
163  wxRect stack[ ToolBarCount + 1 ];
164  wxPoint cpos, lpos;
165  ToolBar *lt = NULL;
166  int ndx, stkcnt = 0;
167  int width, height;
168 
169  // Get size of our parent since we haven't been sized yet
170  GetParent()->GetClientSize( &width, &height );
171  width -= toolbarGap;
172  height -= toolbarGap;
173 
174  // Get the number of docked toolbars and take a quick exit
175  // if we don't have any
176  int cnt = mDockedBars.GetCount();
177  if( cnt == 0 )
178  {
179  SetMinSize( wxSize( width, toolbarGap ) );
180  return;
181  }
182 
183  // Set initial stack entry to maximum size
184  stack[ 0 ].SetX( toolbarGap );
185  stack[ 0 ].SetY( toolbarGap );
186  stack[ 0 ].SetWidth( width );
187  stack[ 0 ].SetHeight( height );
188 
189  // Process all docked and visible toolbars
190  for( ndx = 0; ndx < cnt; ndx++ )
191  {
192  // Cache toolbar pointer
193  ToolBar *ct = (ToolBar *)mDockedBars[ ndx ];
194 
195  // Get and cache the toolbar sizes
196  wxSize sz = ct->GetSize();
197  int tw = sz.GetWidth() + toolbarGap;
198  int th = sz.GetHeight() + toolbarGap;
199 
200  // Will this one fit in remaining horizontal space?
201  if( ( tw > stack[ stkcnt ].GetWidth() ) ||
202  ( th > stack[ stkcnt ].GetHeight() ) )
203  {
204  // Destack entries until one is found in which this bar
205  // will fit or until we run out of stacked entries
206  while( stkcnt > 0 )
207  {
208  stkcnt--;
209 
210  // Get out if it will fit
211  if( ( tw <= stack[ stkcnt ].GetWidth() ) &&
212  ( th <= stack[ stkcnt ].GetHeight() ) )
213  {
214  break;
215  }
216  }
217  }
218 
219  // The current stack entry position is where the bar
220  // will be placed.
221  cpos = stack[ stkcnt ].GetPosition();
222 
223  // We'll be using at least a portion of this stack entry, so
224  // adjust the location and size. It is possible that these
225  // will become zero if this entry and the toolbar have the
226  // same height. This is what we want as it will be destacked
227  // in the next iteration.
228  stack[ stkcnt ].SetY( stack[ stkcnt ].GetY() + th );
229  stack[ stkcnt ].SetHeight( stack[ stkcnt ].GetHeight() - th );
230 
231  // Calc the next possible horizontal location.
232  int x = cpos.x + tw;
233 
234  // Add a NEW stack entry
235  stkcnt++;
236  stack[ stkcnt ].SetX( x );
237  stack[ stkcnt ].SetY( cpos.y );
238  stack[ stkcnt ].SetWidth( width - x );
239  stack[ stkcnt ].SetHeight( th );
240 
241  // Position the previous toolbar
242  if( ndx > 0 )
243  {
244  // Keep the tab order in order
245  ct->MoveAfterInTabOrder( lt );
246 
247  // Place the last toolbar
248  lt->SetPosition( wxPoint( lpos.x, lpos.y ) );
249  }
250 
251  // Place the final toolbar
252  if( ndx == cnt - 1 )
253  {
254  ct->SetPosition( wxPoint( cpos.x, cpos.y ) );
255  }
256 
257  // Remember for next iteration
258  lt = ct;
259  lpos = cpos;
260  }
261 
262  // Set the final size of the dock window
263  SetMinSize( wxSize( -1, stack[ 0 ].GetY() ) );
264 
265  // Clean things up
266  Refresh( false );
267 }
