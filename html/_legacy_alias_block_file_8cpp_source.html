85 {
86  wxFileName summaryFileName;
87  wxFileName aliasFileName;

89  int summaryLen=0;
90  bool noRMS = false;
91  long nValue;
92 
93  while(*attrs)
94  {
95  const wxChar *attr = *attrs++;
96  const wxChar *value = *attrs++;
97  if (!value)
98  break;
99 
100  const wxString strValue = value;
101  if (!wxStricmp(attr, wxT("name")) && XMLValueChecker::IsGoodFileName(strValue, projDir))
102  //v Should this be
103  // dm.AssignFile(summaryFileName, strValue, false);
104  // as in PCMAliasBlockFile::BuildFromXML? Test with an old project.
105  summaryFileName.Assign(projDir, strValue, wxT(""));
106  else if ( !wxStricmp(attr, wxT("aliaspath")) )
107  {
108  if (XMLValueChecker::IsGoodPathName(strValue))
109  aliasFileName.Assign(strValue);
110  else if (XMLValueChecker::IsGoodFileName(strValue, projDir))
111  // Allow fallback of looking for the file name, located in the data directory.
112  aliasFileName.Assign(projDir, strValue);
113  else if (XMLValueChecker::IsGoodPathString(strValue))
114  // If the aliased file is missing, we failed XMLValueChecker::IsGoodPathName()
115  // and XMLValueChecker::IsGoodFileName, because both do existence tests,
116  // but we want to keep the reference to the missing file because it's a good path string.
117  aliasFileName.Assign(strValue);
118  }
119  else if (XMLValueChecker::IsGoodInt(strValue) && strValue.ToLong(&nValue))
120  { // integer parameters
121  if (!wxStricmp(attr, wxT("aliasstart")) && (nValue >= 0))
122  aliasStart = nValue;
123  else if (!wxStricmp(attr, wxT("aliaslen")) && (nValue >= 0))
124  aliasLen = nValue;
125  else if (!wxStricmp(attr, wxT("aliaschannel")) && XMLValueChecker::IsValidChannel(aliasChannel))
126  aliasChannel = nValue;
127  else if (!wxStricmp(attr, wxT("summarylen")) && (nValue > 0))
128  summaryLen = nValue;
129  else if (!wxStricmp(attr, wxT("norms")))
130  noRMS = (nValue != 0);
131  }
132  }
133 
134  return new LegacyAliasBlockFile(summaryFileName, aliasFileName,
