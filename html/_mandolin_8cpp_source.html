125 StkFloat Mandolin :: computeSample()
126 {
127  StkFloat temp = 0.0;
128  if ( !waveDone_ ) {
129  // Scale the pluck excitation with comb
130  // filtering for the duration of the file.
131  temp = soundfile_[mic_]->tick() * pluckAmplitude_;
132  temp = temp - combDelay_.tick(temp);
133  waveDone_ = soundfile_[mic_]->isFinished();
134  }
135 
136  // Damping hack to help avoid overflow on re-plucking.
137  if ( dampTime_ >=0 ) {
138  dampTime_ -= 1;
139  // Calculate 1st delay filtered reflection plus pluck excitation.
140  lastOutput_ = delayLine_.tick( filter_.tick( temp + (delayLine_.lastOut() * 0.7) ) );
141  // Calculate 2nd delay just like the 1st.
142  lastOutput_ += delayLine2_.tick( filter2_.tick( temp + (delayLine2_.lastOut() * 0.7) ) );
143  }
144  else { // No damping hack after 1 period.
145  // Calculate 1st delay filtered reflection plus pluck excitation.
146  lastOutput_ = delayLine_.tick( filter_.tick( temp + (delayLine_.lastOut() * loopGain_) ) );
147  // Calculate 2nd delay just like the 1st.
148  lastOutput_ += delayLine2_.tick( filter2_.tick( temp + (delayLine2_.lastOut() * loopGain_) ) );
149  }
150 
151  lastOutput_ *= 0.3;
152  return lastOutput_;
153 }

172  this->setPluckPosition( norm );
173  else if (number == __SK_StringDamping_) // 11
