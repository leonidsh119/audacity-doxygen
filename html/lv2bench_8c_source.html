65 static double
66 bench(const LilvPlugin* p, uint32_t sample_count, uint32_t block_size)
67 {
68  URITable uri_table;
69  uri_table_init(&uri_table);
70 
71  LV2_URID_Map map = { &uri_table, uri_table_map };
72  LV2_Feature map_feature = { LV2_URID_MAP_URI, &map };
73  LV2_URID_Unmap unmap = { &uri_table, uri_table_unmap };
74  LV2_Feature unmap_feature = { LV2_URID_UNMAP_URI, &unmap };
75  const LV2_Feature* features[] = { &map_feature, &unmap_feature, NULL };
76 
77  float* const buf = (float*)calloc(block_size * 2, sizeof(float));
78  float* const in = buf;
79  float* const out = buf + block_size;
80  if (!buf) {
81  fprintf(stderr, "Out of memory\n");
82  return 0.0;
83  }
84 
85  LV2_Atom_Sequence seq = {
86  { sizeof(LV2_Atom_Sequence_Body),
87  uri_table_map(&uri_table, LV2_ATOM__Sequence) },
88  { 0, 0 } };
89 
90  const char* uri = lilv_node_as_string(lilv_plugin_get_uri(p));
91  LilvNodes* required = lilv_plugin_get_required_features(p);
92  LILV_FOREACH(nodes, i, required) {
93  const LilvNode* feature = lilv_nodes_get(required, i);
94  if (!lilv_node_equals(feature, urid_map)) {
95  fprintf(stderr, "<%s> requires feature <%s>, skipping\n",
96  uri, lilv_node_as_uri(feature));
97  free(buf);
98  return 0.0;
99  }
100  }
101 
102  LilvInstance* instance = lilv_plugin_instantiate(p, 48000.0, features);
103  if (!instance) {
104  fprintf(stderr, "Failed to instantiate <%s>\n",
105  lilv_node_as_uri(lilv_plugin_get_uri(p)));
106  free(buf);
107  return 0.0;
108  }
109 
110  float* controls = (float*)calloc(
111  lilv_plugin_get_num_ports(p), sizeof(float));
112  lilv_plugin_get_port_ranges_float(p, NULL, NULL, controls);
113 
114  const uint32_t n_ports = lilv_plugin_get_num_ports(p);
115  for (uint32_t index = 0; index < n_ports; ++index) {
116  const LilvPort* port = lilv_plugin_get_port_by_index(p, index);
117  if (lilv_port_is_a(p, port, lv2_ControlPort)) {
118  lilv_instance_connect_port(instance, index, &controls[index]);
119  } else if (lilv_port_is_a(p, port, lv2_AudioPort) ||
120  lilv_port_is_a(p, port, lv2_CVPort)) {
121  if (lilv_port_is_a(p, port, lv2_InputPort)) {
122  lilv_instance_connect_port(instance, index, in);
123  } else if (lilv_port_is_a(p, port, lv2_OutputPort)) {
124  lilv_instance_connect_port(instance, index, out);
125  } else {
126  fprintf(stderr, "<%s> port %d neither input nor output, skipping\n",
127  uri, index);
128  lilv_instance_free(instance);
129  free(buf);
130  free(controls);
131  return 0.0;
132  }
133  } else if (lilv_port_is_a(p, port, atom_AtomPort)) {
134  lilv_instance_connect_port(instance, index, &seq);
135  } else {
136  fprintf(stderr, "<%s> port %d has unknown type, skipping\n",
137  uri, index);
138  lilv_instance_free(instance);
139  free(buf);
140  free(controls);
141  return 0.0;
142  }
143  }
144 
145  lilv_instance_activate(instance);
146 
147  struct timespec ts = bench_start();
148  for (uint32_t i = 0; i < (sample_count / block_size); ++i) {
149  lilv_instance_run(instance, block_size);
150  }
151  const double elapsed = bench_end(&ts);
152 
153  lilv_instance_deactivate(instance);
154  lilv_instance_free(instance);
155 
156  uri_table_destroy(&uri_table);
157 
158  if (full_output) {
159  printf("%d %d ", block_size, sample_count);
160  }
161  printf("%lf %s\n", elapsed, uri);
162 
163  free(buf);
164  free(controls);
165  return elapsed;
166 }

168 int
169 main(int argc, char** argv)
170 {

172  uint32_t sample_count = (1 << 19);

208

213 
214  lilv_node_free(urid_map);
