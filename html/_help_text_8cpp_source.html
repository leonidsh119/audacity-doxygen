76 static wxString LinkExpand( const wxString & Text )
77 {
78  wxString Temp = Text;
79  int i,j,k;
80  while( (i=Temp.First( wxT("[[") ))!= wxNOT_FOUND )
81  {
82  wxString Key = Temp.Mid(i+2);
83  j = Key.First( wxT("|") );
84  if( j==wxNOT_FOUND )
85  return Temp;
86  wxString LinkText = Key.Mid( j+1);
87  k = LinkText.First( wxT("]]") );
88  if( k==wxNOT_FOUND )
89  return Temp;
90  Key = Key.Mid( 0, j );
91  LinkText = LinkText.Mid( 0, k );
92 
93  wxString Replacement;
94  if( Key.StartsWith( wxT("wiki:") ))
95  {
96  Replacement = WikiLink( Key.Mid( 5 ), LinkText );
97  }
98  else if( Key.StartsWith( wxT("file:") ))
99  {
100  Replacement = FileLink( Key.Mid( 5 ), LinkText );
101  }
102  else if( Key.StartsWith( wxT("http:") ))
103  {
104  Replacement = HttpLink( Key.Mid( 5 ), LinkText );
105  }
106  else
107  {
108  Replacement = Link( Key, LinkText );
109  }
110 
111  Temp = Temp.Mid( 0, i ) + Replacement + Temp.Mid( i + j + k + 5 );// 5 for the [[|]]
112  }
113  return Temp;
114 }
