168 /* falloc_gc -- return empty pools to the system */
169 /*
170  * Algorithm: for each pool, move all free sample blocks
171  * (on the sample_block_free list) to tlist. If tlist
172  * has ALL of the blocks in the pool (determined by
173  * byte counts), the pool is returned to the heap.
174  */
175 void falloc_gc()
176 {
177  CQUE *lp = NULL;
178  CQUE *cp;
179  CQUE *np;
180  CQUE *tlist = NULL;
181 
182  /* Scan all allocated pools */
183  for (cp = pools; cp; lp = cp, cp = np) {
184  char *str = ((char *)cp) + POOL_HEAD_SIZE;
185  char *end = str + MAXSPOOLSIZE;
186  long tsiz = end - str;
187  long csiz = 0;
188  CQUE *tsave = NULL;
189  CQUE *ln = NULL;
190  CQUE *cn;
191  CQUE *nn;
192 
193  /* Save pointer to next pool */
194  np = cp->qnext;
195 
196  /* Remember head of temp free list */
197  tsave = tlist;
198 
199  /* Scan all nodes on the free list */
200  for (cn = sample_block_free; cn; ln = cn, cn = nn) {
201 
202  /* Get next node */
203  nn = cn->qnext;
204 
205  /* Count it if the node belongs to this pool */
206  if (cn >= (CQUE *) str && cn <= (CQUE *) end) {
207  csiz += round_size(sizeof(sample_block_node));
208 
209  Qenter(tlist, cn);
210 
211  /* Unlink the node */
212  if (cn == sample_block_free) {
213  sample_block_free = nn;
214  cn = NULL;
215  }
216  else {
217  ln->qnext = nn;
218  cn = ln;
219  }
220  }
221  }
222 
223  /* The pool had inuse nodes */
224  if (csiz != tsiz) {
225  continue;
226  }
227 
228  /* Remove the nodes from the temp free list */
229  tlist = tsave;
230 
231  /* Maintain stats */
232  sample_block_total -= (tsiz / round_size(sizeof(sample_block_node)));
233  npools--;
234 
235  /* If this is the active pool, then reset current pointers */
236  if (spoolp >= str && spoolp <= end) {
237  spoolp = NULL;
238  spoolend = NULL;
239  }
240 
241  /* Release the pool to the system */
242  free(cp);
243

245  if (cp == pools) {
246  pools = np;
247  cp = NULL;
248  }
249  else {

251  * cp == pools, so code above is executed. Before the for-loop
252  * iterates, pools == np (assigned above), and cp == NULL. The
253  * for-loop update (lp=cp,cp=np) produces lp == NULL, cp == pools.
254  * Since cp == pools, this else branch will not be taken.
255  * The other path to this code is via the "continue" above. In that
256  * case, the update (lp=cp,cp=np) makes lp a valid pointer or else
257  * the loop exits.
258  * The assert(lp) is here to possibly make static analyzers happy.
259  */
260  assert(lp);
261  lp->qnext = np;
262  cp = lp;
263  }
264  }
