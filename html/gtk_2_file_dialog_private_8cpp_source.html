178 static GtkWidget* find_widget(GtkWidget* parent, const gchar* name, int depth)
179 {
180  // printf("%*.*c%s\n", depth, depth, ' ', gtk_widget_get_name(parent));
181 
182  GtkWidget *widget = NULL;
183  if (g_strcasecmp(gtk_widget_get_name(parent), name) == 0)
184  {
185  return parent;
186  }
187 
188  if (GTK_IS_BIN(parent))
189  {
190  return find_widget(gtk_bin_get_child(GTK_BIN(parent)), name, depth + 1);
191  }
192 
193  if (GTK_IS_CONTAINER(parent))
194  {
195  GList *list = gtk_container_get_children(GTK_CONTAINER(parent));
196  for (GList *node = list; node; node = node->next)
197  {
198  widget = find_widget(GTK_WIDGET(node->data), name, depth + 1);
199  if (widget)
200  {
201  break;
202  }
203  }
204  g_list_free(list);
205  }
206 
207  return widget;
208 }
