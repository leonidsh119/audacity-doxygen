134  }

138  MinVer = (VerNum & 0x00FF0000) >> 16;
139  VerStage = (VerNum & 0x0000FF00) >> 8;

153  break;

156  break;
157  default:
158  VerStage = '?';
159  break;
160  }
161

163  {
164  sprintf(VerStr,"%X.%X%c%X",
165  MajVer, MinVer>>4, VerStage, VerRev);
166  }
167  else
168  {
169  sprintf(VerStr,"%X.%X.%X%c%X",
170  MajVer, MinVer >> 4, MinVer & 0x0F, VerStage, VerRev);

447 pascal short CMTreader(MIDIPacket *ThePacketPtr, long TheRefCon)
448 {
449  /* Set up our A5 world. */
450  long SysA5 = SetA5(TheRefCon);
451  short RetVal = midiMorePacket, i, j;
452  unsigned char *mm_data = ThePacketPtr->data;
453  register byte data1 = mm_data[1];
454 
455  if (midi_read_lock) {
456  /* Don't want to read packet now, get it later */
457  /* DOES THIS REALLY WORK? WHAT WILL CAUSE AN INTERRUPT
458  * TO OCCUR LATER? THIS IS ONLY USED BY midi_flush, IS
459  * BASED ON THE MidiArp CODE FROM APPLE, AND IS UNTESTED - RBD
460  */
461  RetVal = midiKeepPacket;
462  goto alldone;
463  }
464 
465  /* see if Packet is an error message */
466  if (((ThePacketPtr->flags & midiTypeMask) == midiMgrType) &&
467  *((short *) (&(ThePacketPtr->data))) < midiMaxErr) {
468  set_error(MIDIMGRERR);
469  goto alldone;
470  }
471 
472  /* filter out control changes */
473  if (ctrlFilter) {
474  register int hibits = *mm_data & 0xF0;
475  if (hibits == 0xD0 || /* Chan Pressure */
476  hibits == 0xE0 || /* Pitch Bend */
477  hibits == 0xA0 || /* Poly Pressure */
478  ((hibits == 0xB0) && /* Control change (don't count switches) */
479  ((data1 < 64) || (data1 > 121)))) {
480  /* CONTROL MESSAGE HAS BEEN FILTERED */
481  goto alldone;
482  }
483  } else if (realFilter) {
484  register int hibits = *mm_data & 0xF0;
485  if (hibits >= 0xF8) goto alldone;
486  }
487 
488 
489  /* if not a continuation, copy the data into cmt_data */
490  /* The logic to detect a non-continued
491  * packet or a first packet is: "flags bit 1 is clear".
492  */
493  if ((((ThePacketPtr->flags & midiContMask) == midiNoCont)) &&
494  (*mm_data != MIDI_SYSEX)) {
495  register byte *cmt_data = ((byte *) buff) + bufftail;
496  *((long *) cmt_data) = *((long *) mm_data);
497 
498  bufftail = (bufftail + 4) & BUFF_MASK;
499  if (bufftail == buffhead) {
500  /* filled buffer faster than client emptied it */
501  set_error(BUFFOVFL);
502  }
503  }
504 
505  /* see if we have a sysex message to copy to buffer */
506  if (xbuff && !exclFilter &&
507  ((ThePacketPtr->flags & midiContMask) || *mm_data == MIDI_SYSEX)) {
508  int i;
509  register byte *x_data = xbuff + xbuftail;
510 
511  /* iterate over data in message */
512  /* NOTE: in the previous implementation, I thought Sysex messages were
513  * always starting at the beginning of the buffer, but that didn't work.
514  * This implementation assumes nothing -- it is slower because of additional
515  * testing and parsing inside the loop, but seems to work.
516  */
517  for (i = ThePacketPtr->len - 6; i > 0; i--) {
518  if (xbuf_flush) { /* we're searching for beginning of message */
519  if (*mm_data == MIDI_SYSEX) {
520  xbuf_flush = false;
521  sysex_insert(MIDI_SYSEX);
522  }
523  } else { /* we're scanning to the end of the message */
524  if (*mm_data == MIDI_SYSEX) { /* found it, insert proper EOX */
525  sysex_insert(MIDI_EOX);
526  sysex_insert(MIDI_SYSEX);
527  } else if (*mm_data == MIDI_EOX) { /* found it */
528  sysex_insert(MIDI_EOX);
529  xbuf_flush = true;
530  } else sysex_insert(*mm_data);
531 
532  }
533  mm_data++;
534  }
535  }
536 alldone:
537 
538  /* Restore the systems A5 world. */
539  SetA5(SysA5);
540 
541  return(RetVal);
542 }
