583  int argc;
584 
585  /* protect some pointers */
586  xlstkcheck(3);
587  xlsave(fun);
588  xlsave(lists);
589  xlsave(val);
590

592  fun = xlgetarg();
593  lists = xlgalist();
594 
595  /* initialize the result list */
596  val = (valflag ? NIL : lists);
597 
598  /* build a list of argument lists */
599  for (lists = last = consa(lists); moreargs(); last = cdr(last))
600  rplacd(last,cons(xlgalist(),NIL));
601

603  for (;;) {
604  /* build an argument list from the sublists */
605  newfp = xlsp;
606  pusharg(cvfixnum((FIXTYPE)(newfp - xlfp)));
607  pusharg(fun);

611  pusharg(carflag ? car(y) : y);
612  rplaca(x,cdr(y));
613  ++argc;
614  }
615

619  break;
620  }
621

628  else val = p;
629  last = p;

631  else
