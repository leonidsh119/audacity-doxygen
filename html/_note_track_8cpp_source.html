192 void NoteTrack::WarpAndTransposeNotes(double t0, double t1,
193  const TimeWarper &warper,
194  double semitones)
195 {
196  // Since this is a duplicate and duplicates convert mSeq to
197  // a text string for saving as XML, we probably have to
198  // duplicate again to get back an mSeq
199  NoteTrack *nt = this;
200  double offset = nt->GetOffset(); // track is shifted this amount
201  if (!mSeq) { // replace saveme with an (unserialized) duplicate
202  nt = (NoteTrack *) this->Duplicate();
203  wxASSERT(!mSeq && nt->mSeq && !nt->mSerializationBuffer);
204  // swap mSeq and Buffer between this and nt
205  nt->mSerializationBuffer = mSerializationBuffer;
206  nt->mSerializationLength = mSerializationLength;
207  mSerializationBuffer = NULL;
208  mSerializationLength = 0;
209  mSeq = nt->mSeq;
210  nt->mSeq = NULL;
211  delete nt; // delete the duplicate
212  }
213  mSeq->convert_to_seconds(); // make sure time units are right
214  t1 -= offset; // adjust time range to compensate for track offset
215  t0 -= offset;
216  if (t1 > mSeq->get_dur()) { // make sure t0, t1 are within sequence
217  t1 = mSeq->get_dur();
218  if (t0 >= t1) return;
219  }
220  Alg_iterator iter(mSeq, false);
221  iter.begin();
222  Alg_event_ptr event;
223  while (0 != (event = iter.next()) && event->time < t1) {
224  if (event->is_note() && event->time >= t0 &&
225  // Allegro data structure does not restrict channels to 16.
226  // Since there is not way to select more than 16 channels,
227  // map all channel numbers mod 16. This will have no effect
228  // on MIDI files, but it will allow users to at least select
229  // all channels on non-MIDI event sequence data.
230  IsVisibleChan(event->chan % 16)) {
231  event->set_pitch(event->get_pitch() + semitones);
232  }
233  }
234  iter.end();
235  // now, use warper to warp the tempo map
236  mSeq->convert_to_beats(); // beats remain the same
237  Alg_time_map_ptr map = mSeq->get_time_map();
238  map->insert_beat(t0, map->time_to_beat(t0));
239  map->insert_beat(t1, map->time_to_beat(t1));
240  int i, len = map->length();
241  for (i = 0; i < len; i++) {
242  Alg_beat &beat = map->beats[i];
243  beat.time = warper.Warp(beat.time + offset) - offset;
244  }
245  // about to redisplay, so might as well convert back to time now
246  mSeq->convert_to_seconds();
247 }

524 {
525  // Paste inserts src at time t. If src has a positive offset,
526  // the offset is treated as silence which is also inserted. If
527  // the offset is negative, the offset is ignored and the ENTIRE
528  // src is inserted (otherwise, we would either lose data from
529  // src by not inserting things at negative times, or inserting
530  // things at negative times could overlap things already in

534  if (src == NULL || src->GetKind() != Track::Note)

554 // Call this function to manipulate the underlying sequence data. This is
555 // NOT the function that handles horizontal dragging.
556 bool NoteTrack::Shift(double t) // t is always seconds
557 {
558  if (t > 0) {
559  // insert an even number of measures
560  mSeq->convert_to_beats();
561  // get initial tempo
562  double tempo = mSeq->get_tempo(0.0);
563  double beats_per_measure = mSeq->get_bar_len(0.0);
564  int m = ROUND(t * tempo / beats_per_measure);
565  // need at least 1 measure, so if we rounded down to zero, fix it
566  if (m == 0) m = 1;
567  // compute NEW tempo so that m measures at new tempo take t seconds
568  tempo = beats_per_measure * m / t; // in beats per second
569  mSeq->insert_silence(0.0, beats_per_measure * m);
570  mSeq->set_tempo(tempo * 60.0 /* bpm */, 0.0, beats_per_measure * m);
571  mSeq->write("afterShift.gro");
572  } else if (t < 0) {
573  mSeq->convert_to_seconds();
574  mSeq->clear(0, t, true);
575  } else { // offset is zero, no modifications
576  return false;
577  }
578  return true;
579 }

608 Alg_seq_ptr NoteTrack::MakeExportableSeq()
609 {
610  double offset = GetOffset();
611  if (offset == 0)
612  return mSeq;
613  // make a copy, deleting events that are shifted before time 0
614  double start = -offset;
615  if (start < 0) start = 0;
616  // notes that begin before "start" are not included even if they
617  // extend past "start" (because "all" parameter is set to false)
618  Alg_seq_ptr seq = mSeq->copy(start, mSeq->get_dur() - start, false);
619  if (offset > 0) {
620  // swap seq and mSeq so that Shift operates on the NEW copy
621  Alg_seq_ptr old_seq = mSeq;
622  mSeq = seq;
623  Shift(offset);
624  seq = mSeq; // undo the swap
625  mSeq = old_seq;
626 #ifdef OLD_CODE
627  // now shift events by offset. This must be done with an integer
628  // number of measures, so first, find the beats-per-measure
629  double beats_per_measure = 4.0;
630  Alg_time_sig_ptr tsp = NULL;
631  if (seq->time_sig.length() > 0 && seq->time_sig[0].beat < ALG_EPS) {
632  // there is an initial time signature
633  tsp = &(seq->time_sig[0]);
634  beats_per_measure = (tsp->num * 4) / tsp->den;
635  }
636  // also need the initial tempo
637  double bps = ALG_DEFAULT_BPM / 60;
638  Alg_time_map_ptr map = seq->get_time_map();
639  Alg_beat_ptr bp = &(map->beats[0]);
640  if (bp->time < ALG_EPS) { // tempo change at time 0
641  if (map->beats.len > 1) { // compute slope to get tempo
642  bps = (map->beats[1].beat - map->beats[0].beat) /
643  (map->beats[1].time - map->beats[0].time);
644  } else if (seq->get_time_map()->last_tempo_flag) {
645  bps = seq->get_time_map()->last_tempo;
646  }
647  }
648  // find closest number of measures to fit in the gap
649  // number of measures is offset / measure_time
650  double measure_time = beats_per_measure / bps; // seconds per measure
651  int n = ROUND(offset / measure_time);
652  if (n == 0) n = 1;
653  // we will insert n measures. Compute the desired duration of each.
654  measure_time = offset / n;
655  bps = beats_per_measure / measure_time;
656  // insert integer multiple of measures at beginning
657  seq->convert_to_beats();
658  seq->insert_silence(0, beats_per_measure * n);
659  // make sure time signature at 0 is correct
660  if (tsp) {
661  seq->set_time_sig(0, tsp->num, tsp->den);
662  }
663  // adjust tempo to match offset
664  seq->set_tempo(bps * 60.0, 0, beats_per_measure * n);
665 #endif
666  } else {
667  // if offset is negative, it might not be a multiple of beats, but
668  // we want to preserve the relative positions of measures. I.e. we
669  // should shift barlines and time signatures as well as notes.
670  // Insert a time signature at the first bar-line if necessary.
671 
672  // Translate start from seconds to beats and call it beat:
673  double beat = mSeq->get_time_map()->time_to_beat(start);
674  // Find the time signature in mSeq in effect at start (beat):
675  int i = mSeq->time_sig.find_beat(beat);
676  // i is where you would insert a NEW time sig at beat,
677  // Case 1: beat coincides with a time sig at i. Time signature
678  // at beat means that there is a barline at beat, so when beat
679  // is shifted to 0, the relative barline positions are preserved
680  if (mSeq->time_sig.length() > 0 &&
681  within(beat, mSeq->time_sig[i].beat, ALG_EPS)) {
682  // beat coincides with time signature change, so offset must
683  // be a multiple of beats
684  /* do nothing */ ;
685  // Case 2: there is no time signature before beat.
686  } else if (i == 0 && (mSeq->time_sig.length() == 0 ||
687  mSeq->time_sig[i].beat > beat)) {
688  // If beat does not fall on an implied barline, we need to
689  // insert a time signature.
690  double measures = beat / 4.0;
691  double imeasures = ROUND(measures);
692  if (!within(measures, imeasures, ALG_EPS)) {
693  double bar_offset = (int(measures) + 1) * 4.0 - beat;
694  seq->set_time_sig(bar_offset, 4, 4);
695  }
696  // This case should never be true because if i == 0, either there
697  // are no time signatures before beat (Case 2),
698  // or there is one time signature at beat (Case 1)
699  } else if (i == 0) {
700  /* do nothing (might be good to assert(false)) */ ;
701  // Case 3: i-1 must be the effective time sig position
702  } else {
703  i -= 1; // index the time signature in effect at beat
704  Alg_time_sig_ptr tsp = &(mSeq->time_sig[i]);
705  double beats_per_measure = (tsp->num * 4) / tsp->den;
706  double measures = (beat - tsp->beat) / beats_per_measure;
707  int imeasures = ROUND(measures);
708  if (!within(measures, imeasures, ALG_EPS)) {
709  // beat is not on a measure, so we need to insert a time sig
710  // to force a bar line at the first measure location after
711  // beat
712  double bar = tsp->beat + beats_per_measure * (int(measures) + 1);
713  double bar_offset = bar - beat;
714  // insert NEW time signature at bar_offset in new sequence
715  // It will have the same time signature, but the position will
716  // force a barline to match the barlines in mSeq
717  seq->set_time_sig(bar_offset, tsp->num, tsp->den);
718  }
719  // else beat coincides with a barline, so no need for an extra
720  // time signature to force barline alignment
721  }
722  }
723  return seq;
724 }

807 void NoteTrack::WriteXML(XMLWriter &xmlFile)
808 {
809  std::ostringstream data;
810  // Normally, Duplicate is called in pairs -- once to put NoteTrack
811  // on the Undo stack, and again to move from the Undo stack to an
812  // "active" editable state. For efficiency, we do not do a "real"
813  // Duplicate followed by serialization into a binary blob. Instead,
814  // we combine the Duplicate with serialization or unserialization.
815  // Serialization and Unserialization happen on alternate calls to
816  // Duplicate and (usually) produce the right results at the right
817  // time.
818  // It turns out that this optimized Duplicate is a little too
819  // clever. There is at least one case where a track can be duplicated
820  // and then AutoSave'd. (E.g. do an "Insert Silence" effect on a
821  // NoteTrack.) In this case, mSeq will be NULL. To avoid a crash
822  // and perform WriteXML, we may need to restore NoteTracks from binary
823  // blobs to regular data structures (with an Alg_seq member).
824  NoteTrack *saveme = this;
825  if (!mSeq) { // replace saveme with an (unserialized) duplicate
826  saveme = (NoteTrack *) this->Duplicate();
827  assert(saveme->mSeq);
828  }
829  saveme->mSeq->write(data, true);
830  xmlFile.StartTag(wxT("notetrack"));
831  xmlFile.WriteAttr(wxT("name"), saveme->mName);
832  xmlFile.WriteAttr(wxT("offset"), saveme->GetOffset());
833  xmlFile.WriteAttr(wxT("visiblechannels"), saveme->mVisibleChannels);
834  xmlFile.WriteAttr(wxT("height"), saveme->GetActualHeight());
835  xmlFile.WriteAttr(wxT("minimized"), saveme->GetMinimized());
836  xmlFile.WriteAttr(wxT("isSelected"), this->GetSelected());
837 
838 #ifdef EXPERIMENTAL_MIDI_OUT
839  xmlFile.WriteAttr(wxT("velocity"), (double) saveme->mGain);
840 #endif
841  xmlFile.WriteAttr(wxT("bottomnote"), saveme->mBottomNote);
842  xmlFile.WriteAttr(wxT("data"), wxString(data.str().c_str(), wxConvUTF8));
843  xmlFile.EndTag(wxT("notetrack"));
844  if (this != saveme) {
845  delete saveme; // DELETE the duplicate
846  }
847 }

880 void NoteTrack::ZoomTo(int start, int end)
881 {
882  wxRect trackRect(0, GetY(), 1, GetHeight());
883  PrepareIPitchToY(trackRect);
884  int topPitch = YToIPitch(start);
885  int botPitch = YToIPitch(end);
886  if (topPitch < botPitch) { // swap
887  int temp = topPitch; topPitch = botPitch; botPitch = temp;
888  }
889  if (topPitch == botPitch) { // can't divide by zero, do something else
890  Zoom(start, 1);
891  return;
892  }
893  int trialPitchHeight = trackRect.height / (topPitch - botPitch);
894  if (trialPitchHeight > 25) { // keep mPitchHeight in bounds [1...25]
895  trialPitchHeight = 25;
896  } else if (trialPitchHeight == 0) {
897  trialPitchHeight = 1;
898  }
899  Zoom((start + end) / 2, trialPitchHeight - mPitchHeight);
900 }
