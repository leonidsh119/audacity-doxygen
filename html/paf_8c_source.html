105 int
106 paf_open (SF_PRIVATE *psf)
107 { int subformat, error, endian ;
108 
109  psf->dataoffset = PAF_HEADER_LENGTH ;
110 
111  if (psf->file.mode == SFM_READ || (psf->file.mode == SFM_RDWR && psf->filelength > 0))
112  { if ((error = paf_read_header (psf)))
113  return error ;
114  } ;
115 
116  subformat = SF_CODEC (psf->sf.format) ;
117 
118  if (psf->file.mode == SFM_WRITE || psf->file.mode == SFM_RDWR)
119  { if ((SF_CONTAINER (psf->sf.format)) != SF_FORMAT_PAF)
120  return SFE_BAD_OPEN_FORMAT ;
121 
122  endian = SF_ENDIAN (psf->sf.format) ;
123 
124  /* PAF is by default big endian. */
125  psf->endian = SF_ENDIAN_BIG ;
126 
127  if (endian == SF_ENDIAN_LITTLE || (CPU_IS_LITTLE_ENDIAN && (endian == SF_ENDIAN_CPU)))
128  psf->endian = SF_ENDIAN_LITTLE ;
129 
130  if ((error = paf_write_header (psf, SF_FALSE)))
131  return error ;
132 
133  psf->write_header = paf_write_header ;
134  } ;
135 
136  switch (subformat)
137  { case SF_FORMAT_PCM_S8 :
138  psf->bytewidth = 1 ;
139  error = pcm_init (psf) ;
140  break ;
141 
142  case SF_FORMAT_PCM_16 :
143  psf->bytewidth = 2 ;
144  error = pcm_init (psf) ;
145  break ;
146 
147  case SF_FORMAT_PCM_24 :
148  /* No bytewidth because of whacky 24 bit encoding. */
149  error = paf24_init (psf) ;
150  break ;
151 
152  default : return SFE_PAF_UNKNOWN_FORMAT ;
153  } ;
154 
155  return error ;
156 } /* paf_open */

162 paf_read_header (SF_PRIVATE *psf)
163 { PAF_FMT paf_fmt ;
164  int marker ;
165 
166  memset (&paf_fmt, 0, sizeof (paf_fmt)) ;
167  psf_binheader_readf (psf, "pm", 0, &marker) ;
168 
169  psf_log_printf (psf, "Signature : '%M'\n", marker) ;
170 
171  if (marker == PAF_MARKER)
172  { psf_binheader_readf (psf, "E444444", &(paf_fmt.version), &(paf_fmt.endianness),
173  &(paf_fmt.samplerate), &(paf_fmt.format), &(paf_fmt.channels), &(paf_fmt.source)) ;
174  }
175  else if (marker == FAP_MARKER)
176  { psf_binheader_readf (psf, "e444444", &(paf_fmt.version), &(paf_fmt.endianness),
177  &(paf_fmt.samplerate), &(paf_fmt.format), &(paf_fmt.channels), &(paf_fmt.source)) ;
178  }
179  else
180  return SFE_PAF_NO_MARKER ;
181 
182  psf_log_printf (psf, "Version : %d\n", paf_fmt.version) ;
183 
184  if (paf_fmt.version != 0)
185  { psf_log_printf (psf, "*** Bad version number. should be zero.\n") ;
186  return SFE_PAF_VERSION ;
187  } ;
188 
189  psf_log_printf (psf, "Sample Rate : %d\n", paf_fmt.samplerate) ;
190  psf_log_printf (psf, "Channels : %d\n", paf_fmt.channels) ;
191 
192  psf_log_printf (psf, "Endianness : %d => ", paf_fmt.endianness) ;
193  if (paf_fmt.endianness)
194  { psf_log_printf (psf, "Little\n", paf_fmt.endianness) ;
195  psf->endian = SF_ENDIAN_LITTLE ;
196  }
197  else
198  { psf_log_printf (psf, "Big\n", paf_fmt.endianness) ;
199  psf->endian = SF_ENDIAN_BIG ;
200  } ;
201 
202  if (psf->filelength < PAF_HEADER_LENGTH)
203  return SFE_PAF_SHORT_HEADER ;
204 
205  psf->datalength = psf->filelength - psf->dataoffset ;
206 
207  psf_binheader_readf (psf, "p", (int) psf->dataoffset) ;
208 
209  psf->sf.samplerate = paf_fmt.samplerate ;
210  psf->sf.channels = paf_fmt.channels ;
211 
212  /* Only fill in type major. */
213  psf->sf.format = SF_FORMAT_PAF ;
214 
215  psf_log_printf (psf, "Format : %d => ", paf_fmt.format) ;
216 
217  /* PAF is by default big endian. */
218  psf->sf.format |= paf_fmt.endianness ? SF_ENDIAN_LITTLE : SF_ENDIAN_BIG ;
219 
220  switch (paf_fmt.format)
221  { case PAF_PCM_S8 :
222  psf_log_printf (psf, "8 bit linear PCM\n") ;
223  psf->bytewidth = 1 ;
224 
225  psf->sf.format |= SF_FORMAT_PCM_S8 ;
226 
227  psf->blockwidth = psf->bytewidth * psf->sf.channels ;
228  psf->sf.frames = psf->datalength / psf->blockwidth ;
229  break ;
230 
231  case PAF_PCM_16 :
232  psf_log_printf (psf, "16 bit linear PCM\n") ;
233  psf->bytewidth = 2 ;
234 
235  psf->sf.format |= SF_FORMAT_PCM_16 ;
236 
237  psf->blockwidth = psf->bytewidth * psf->sf.channels ;
238  psf->sf.frames = psf->datalength / psf->blockwidth ;
239  break ;
240 
241  case PAF_PCM_24 :
242  psf_log_printf (psf, "24 bit linear PCM\n") ;
243  psf->bytewidth = 3 ;
244 
245  psf->sf.format |= SF_FORMAT_PCM_24 ;
246 
247  psf->blockwidth = 0 ;
248  psf->sf.frames = PAF24_SAMPLES_PER_BLOCK * psf->datalength /
249  (PAF24_BLOCK_SIZE * psf->sf.channels) ;
250  break ;
251 
252  default : psf_log_printf (psf, "Unknown\n") ;
253  return SFE_PAF_UNKNOWN_FORMAT ;
254  break ;
255  } ;
256 
257  psf_log_printf (psf, "Source : %d => ", paf_fmt.source) ;
258 
259  switch (paf_fmt.source)
260  { case 1 : psf_log_printf (psf, "Analog Recording\n") ;
261  break ;
262  case 2 : psf_log_printf (psf, "Digital Transfer\n") ;
263  break ;
264  case 3 : psf_log_printf (psf, "Multi-track Mixdown\n") ;
265  break ;
266  case 5 : psf_log_printf (psf, "Audio Resulting From DSP Processing\n") ;
267  break ;
268  default : psf_log_printf (psf, "Unknown\n") ;
269  break ;
270  } ;
271 
272  return 0 ;

275 static int
276 paf_write_header (SF_PRIVATE *psf, int UNUSED (calc_length))
277 { int paf_format ;
278 
279  /* PAF header already written so no need to re-write. */
280  if (psf_ftell (psf) >= PAF_HEADER_LENGTH)
281  return 0 ;
282 
283  psf->dataoffset = PAF_HEADER_LENGTH ;
284 
285  switch (SF_CODEC (psf->sf.format))
286  { case SF_FORMAT_PCM_S8 :
287  paf_format = PAF_PCM_S8 ;
288  break ;
289 
290  case SF_FORMAT_PCM_16 :
291  paf_format = PAF_PCM_16 ;
292  break ;
293 
294  case SF_FORMAT_PCM_24 :
295  paf_format = PAF_PCM_24 ;
296  break ;
297 
298  default : return SFE_PAF_UNKNOWN_FORMAT ;
299  } ;
300 
301  /* Reset the current header length to zero. */
302  psf->header [0] = 0 ;
303  psf->headindex = 0 ;
304 
305  if (psf->endian == SF_ENDIAN_BIG)
306  { /* Marker, version, endianness, samplerate */
307  psf_binheader_writef (psf, "Em444", PAF_MARKER, 0, 0, psf->sf.samplerate) ;
308  /* format, channels, source */
309  psf_binheader_writef (psf, "E444", paf_format, psf->sf.channels, 0) ;
310  }
311  else if (psf->endian == SF_ENDIAN_LITTLE)
312  { /* Marker, version, endianness, samplerate */
313  psf_binheader_writef (psf, "em444", FAP_MARKER, 0, 1, psf->sf.samplerate) ;
314  /* format, channels, source */
315  psf_binheader_writef (psf, "e444", paf_format, psf->sf.channels, 0) ;
316  } ;
317 
318  /* Zero fill to dataoffset. */
319  psf_binheader_writef (psf, "z", (size_t) (psf->dataoffset - psf->headindex)) ;
320 
321  psf_fwrite (psf->header, psf->headindex, 1, psf) ;
322 
323  return psf->error ;
324 } /* paf_write_header */

326 /*===============================================================================
327 ** 24 bit PAF files have a really weird encoding.
328 ** For a mono file, 10 samples (each being 3 bytes) are packed into a 32 byte
329 ** block. The 8 ints in this 32 byte block are then endian swapped (as ints)
330 ** if necessary before being written to disk.
331 ** For a stereo file, blocks of 10 samples from the same channel are encoded
332 ** into 32 bytes as for the mono case. The 32 byte blocks are then interleaved
333 ** on disk.
334 ** Reading has to reverse the above process :-).
335 ** Weird!!!
336 **
337 ** The code below attempts to gain efficiency while maintaining readability.
338 */

346 paf24_init (SF_PRIVATE *psf)
347 { PAF24_PRIVATE *ppaf24 ;
348  int paf24size ;
349 
350  paf24size = sizeof (PAF24_PRIVATE) + psf->sf.channels *
351  (PAF24_BLOCK_SIZE + PAF24_SAMPLES_PER_BLOCK * sizeof (int)) ;
352

354  ** Not exatly sure why this needs to be here but the tests
355  ** fail without it.
356  */
357  psf->last_op = 0 ;
358 
359  if (! (psf->codec_data = malloc (paf24size)))
360  return SFE_MALLOC_FAILED ;
361 
362  ppaf24 = (PAF24_PRIVATE*) psf->codec_data ;
363  memset (ppaf24, 0, paf24size) ;
364 
365  ppaf24->channels = psf->sf.channels ;
366  ppaf24->samples = ppaf24->data ;
367  ppaf24->block = (unsigned char*) (ppaf24->data + PAF24_SAMPLES_PER_BLOCK * ppaf24->channels) ;
368

370  ppaf24->samplesperblock = PAF24_SAMPLES_PER_BLOCK ;
371 
372  if (psf->file.mode == SFM_READ || psf->file.mode == SFM_RDWR)
373  { paf24_read_block (psf, ppaf24) ; /* Read first block. */
374 
375  psf->read_short = paf24_read_s ;
376  psf->read_int = paf24_read_i ;
377  psf->read_float = paf24_read_f ;
378  psf->read_double = paf24_read_d ;
379  } ;
380 
381  if (psf->file.mode == SFM_WRITE || psf->file.mode == SFM_RDWR)
382  { psf->write_short = paf24_write_s ;
383  psf->write_int = paf24_write_i ;
384  psf->write_float = paf24_write_f ;
385  psf->write_double = paf24_write_d ;
386  } ;
387 
388  psf->seek = paf24_seek ;
389  psf->container_close = paf24_close ;
390 
391  psf->filelength = psf_get_filelen (psf) ;
392  psf->datalength = psf->filelength - psf->dataoffset ;
393 
394  if (psf->datalength % PAF24_BLOCK_SIZE)
395  { if (psf->file.mode == SFM_READ)
396  psf_log_printf (psf, "*** Warning : file seems to be truncated.\n") ;
397  ppaf24->max_blocks = psf->datalength / ppaf24->blocksize + 1 ;
398  }
399  else
400  ppaf24->max_blocks = psf->datalength / ppaf24->blocksize ;
401 
402  ppaf24->read_block = 0 ;
403  if (psf->file.mode == SFM_RDWR)
404  ppaf24->write_block = ppaf24->max_blocks ;
405  else
406  ppaf24->write_block = 0 ;
407 
408  psf->sf.frames = ppaf24->samplesperblock * ppaf24->max_blocks ;
409  ppaf24->sample_count = psf->sf.frames ;
410 
411  return 0 ;
412 } /* paf24_init */

414 static sf_count_t
415 paf24_seek (SF_PRIVATE *psf, int mode, sf_count_t offset)
416 { PAF24_PRIVATE *ppaf24 ;
417  int newblock, newsample ;
418 
419  if (psf->codec_data == NULL)
420  { psf->error = SFE_INTERNAL ;
421  return PSF_SEEK_ERROR ;
422  } ;
423 
424  ppaf24 = (PAF24_PRIVATE*) psf->codec_data ;
425 
426  if (mode == SFM_READ && ppaf24->write_count > 0)
427  paf24_write_block (psf, ppaf24) ;
428 
429  newblock = offset / ppaf24->samplesperblock ;
430  newsample = offset % ppaf24->samplesperblock ;
431 
432  switch (mode)
433  { case SFM_READ :
434  if (psf->last_op == SFM_WRITE && ppaf24->write_count)
435  paf24_write_block (psf, ppaf24) ;
436 
437  psf_fseek (psf, psf->dataoffset + newblock * ppaf24->blocksize, SEEK_SET) ;
438  ppaf24->read_block = newblock ;
439  paf24_read_block (psf, ppaf24) ;
440  ppaf24->read_count = newsample ;
441  break ;
442 
443  case SFM_WRITE :
444  if (offset > ppaf24->sample_count)
445  { psf->error = SFE_BAD_SEEK ;
446  return PSF_SEEK_ERROR ;
447  } ;
448 
449  if (psf->last_op == SFM_WRITE && ppaf24->write_count)
450  paf24_write_block (psf, ppaf24) ;
451 
452  psf_fseek (psf, psf->dataoffset + newblock * ppaf24->blocksize, SEEK_SET) ;
453  ppaf24->write_block = newblock ;
454  paf24_read_block (psf, ppaf24) ;
455  ppaf24->write_count = newsample ;
456  break ;
457 
458  default :
459  psf->error = SFE_BAD_SEEK ;
460  return PSF_SEEK_ERROR ;
461  } ;
462 
463  return newblock * ppaf24->samplesperblock + newsample ;
464 } /* paf24_seek */

485 static int
486 paf24_read_block (SF_PRIVATE *psf, PAF24_PRIVATE *ppaf24)
487 { int k, channel ;
488  unsigned char *cptr ;
489 
490  ppaf24->read_block ++ ;
491  ppaf24->read_count = 0 ;
492 
493  if (ppaf24->read_block * ppaf24->samplesperblock > ppaf24->sample_count)
494  { memset (ppaf24->samples, 0, ppaf24->samplesperblock * ppaf24->channels) ;
495  return 1 ;
496  } ;
497 
498  /* Read the block. */
499  if ((k = psf_fread (ppaf24->block, 1, ppaf24->blocksize, psf)) != ppaf24->blocksize)
500  psf_log_printf (psf, "*** Warning : short read (%d != %d).\n", k, ppaf24->blocksize) ;
501 
502 
503  if (CPU_IS_LITTLE_ENDIAN)
504  { /* Do endian swapping if necessary. */
505  if (psf->endian == SF_ENDIAN_BIG)
506  endswap_int_array (ppaf24->data, 8 * ppaf24->channels) ;
507 
508  /* Unpack block. */
509  for (k = 0 ; k < PAF24_SAMPLES_PER_BLOCK * ppaf24->channels ; k++)
510  { channel = k % ppaf24->channels ;
511  cptr = ppaf24->block + PAF24_BLOCK_SIZE * channel + 3 * (k / ppaf24->channels) ;
512  ppaf24->samples [k] = (cptr [0] << 8) | (cptr [1] << 16) | (cptr [2] << 24) ;
513  } ;
514  }
515  else
516  { /* Do endian swapping if necessary. */
517  if (psf->endian == SF_ENDIAN_BIG)
518  endswap_int_array (ppaf24->data, 8 * ppaf24->channels) ;
519 
520  /* Unpack block. */
521  for (k = 0 ; k < PAF24_SAMPLES_PER_BLOCK * ppaf24->channels ; k++)
522  { channel = k % ppaf24->channels ;
523  cptr = ppaf24->block + PAF24_BLOCK_SIZE * channel + 3 * (k / ppaf24->channels) ;
524  ppaf24->samples [k] = (cptr [0] << 8) | (cptr [1] << 16) | (cptr [2] << 24) ;
525  } ;
526  } ;
527 
528  return 1 ;
529 } /* paf24_read_block */

531 static int
532 paf24_read (SF_PRIVATE *psf, PAF24_PRIVATE *ppaf24, int *ptr, int len)
533 { int count, total = 0 ;
534 
535  while (total < len)
536  { if (ppaf24->read_block * ppaf24->samplesperblock >= ppaf24->sample_count)
537  { memset (&(ptr [total]), 0, (len - total) * sizeof (int)) ;
538  return total ;
539  } ;
540 
541  if (ppaf24->read_count >= ppaf24->samplesperblock)
542  paf24_read_block (psf, ppaf24) ;
543 
544  count = (ppaf24->samplesperblock - ppaf24->read_count) * ppaf24->channels ;
545  count = (len - total > count) ? count : len - total ;
546 
547  memcpy (&(ptr [total]), &(ppaf24->samples [ppaf24->read_count * ppaf24->channels]), count * sizeof (int)) ;
548  total += count ;
549  ppaf24->read_count += count / ppaf24->channels ;
550  } ;
551 
552  return total ;
553 } /* paf24_read */

650 static int
651 paf24_write_block (SF_PRIVATE *psf, PAF24_PRIVATE *ppaf24)
652 { int k, nextsample, channel ;
653  unsigned char *cptr ;
654 
655  /* First pack block. */
656 
657  if (CPU_IS_LITTLE_ENDIAN)
658  { for (k = 0 ; k < PAF24_SAMPLES_PER_BLOCK * ppaf24->channels ; k++)
659  { channel = k % ppaf24->channels ;
660  cptr = ppaf24->block + PAF24_BLOCK_SIZE * channel + 3 * (k / ppaf24->channels) ;
661  nextsample = ppaf24->samples [k] >> 8 ;
662  cptr [0] = nextsample ;
663  cptr [1] = nextsample >> 8 ;
664  cptr [2] = nextsample >> 16 ;
665  } ;
666 
667  /* Do endian swapping if necessary. */
668  if (psf->endian == SF_ENDIAN_BIG)
669  endswap_int_array (ppaf24->data, 8 * ppaf24->channels) ;
670  }
671  else if (CPU_IS_BIG_ENDIAN)
672  { /* This is correct. */
673  for (k = 0 ; k < PAF24_SAMPLES_PER_BLOCK * ppaf24->channels ; k++)
674  { channel = k % ppaf24->channels ;
675  cptr = ppaf24->block + PAF24_BLOCK_SIZE * channel + 3 * (k / ppaf24->channels) ;
676  nextsample = ppaf24->samples [k] >> 8 ;
677  cptr [0] = nextsample ;
678  cptr [1] = nextsample >> 8 ;
679  cptr [2] = nextsample >> 16 ;
680  } ;
681  if (psf->endian == SF_ENDIAN_BIG)
682  endswap_int_array (ppaf24->data, 8 * ppaf24->channels) ;
683  } ;
684 
685  /* Write block to disk. */
686  if ((k = psf_fwrite (ppaf24->block, 1, ppaf24->blocksize, psf)) != ppaf24->blocksize)
687  psf_log_printf (psf, "*** Warning : short write (%d != %d).\n", k, ppaf24->blocksize) ;
688 
689  if (ppaf24->sample_count < ppaf24->write_block * ppaf24->samplesperblock + ppaf24->write_count)
690  ppaf24->sample_count = ppaf24->write_block * ppaf24->samplesperblock + ppaf24->write_count ;
691 
692  if (ppaf24->write_count == ppaf24->samplesperblock)
693  { ppaf24->write_block ++ ;
694  ppaf24->write_count = 0 ;
695  } ;
696 
697  return 1 ;
698 } /* paf24_write_block */

702 { int count, total = 0 ;
703 
704  while (total < len)
705  { count = (ppaf24->samplesperblock - ppaf24->write_count) * ppaf24->channels ;
706 
707  if (count > len - total)
708  count = len - total ;
709 
710  memcpy (&(ppaf24->samples [ppaf24->write_count * ppaf24->channels]), &(ptr [total]), count * sizeof (int)) ;
711  total += count ;

714  if (ppaf24->write_count >= ppaf24->samplesperblock)
715  paf24_write_block (psf, ppaf24) ;
716  } ;
