189 static PmError alsa_write_byte(PmInternal *midi, unsigned char byte,
190  PmTimestamp timestamp)
191 {
192  alsa_descriptor_type desc = (alsa_descriptor_type) midi->descriptor;
193  snd_seq_event_t ev;
194  int err;
195 
196  snd_seq_ev_clear(&ev);
197  if (snd_midi_event_encode_byte(desc->parser, byte, &ev) == 1) {
198  snd_seq_ev_set_dest(&ev, desc->client, desc->port);
199  snd_seq_ev_set_source(&ev, desc->this_port);
200  if (midi->latency > 0) {
201  /* compute relative time of event = timestamp - now + latency */
202  PmTimestamp now = (midi->time_proc ?
203  midi->time_proc(midi->time_info) :
204  Pt_Time(NULL));
205  int when = timestamp;
206  /* if timestamp is zero, send immediately */
207  /* otherwise compute time delay and use delay if positive */
208  if (when == 0) when = now;
209  when = (when - now) + midi->latency;
210  if (when < 0) when = 0;
211  VERBOSE printf("timestamp %d now %d latency %d, ",
212  (int) timestamp, (int) now, midi->latency);
213  VERBOSE printf("scheduling event after %d\n", when);
214  /* message is sent in relative ticks, where 1 tick = 1 ms */
215  snd_seq_ev_schedule_tick(&ev, queue, 1, when);
216  /* NOTE: for cases where the user does not supply a time function,
217  we could optimize the code by not starting Pt_Time and using
218  the alsa tick time instead. I didn't do this because it would
219  entail changing the queue management to start the queue tick
220  count when PortMidi is initialized and keep it running until
221  PortMidi is terminated. (This should be simple, but it's not
222  how the code works now.) -RBD */
223  } else { /* send event out without queueing */
224  VERBOSE printf("direct\n");
225  /* ev.queue = SND_SEQ_QUEUE_DIRECT;
226  ev.dest.client = SND_SEQ_ADDRESS_SUBSCRIBERS; */
227  snd_seq_ev_set_direct(&ev);
228  }
229  VERBOSE printf("sending event\n");
230  err = snd_seq_event_output(seq, &ev);
231  if (err < 0) {
232  desc->error = err;
233  return pmHostError;
234  }
235  }
236  return pmNoError;

483 static void handle_event(snd_seq_event_t *ev)
484 {
485  int device_id = ev->dest.port;
486  PmInternal *midi = descriptors[device_id].internalDescriptor;
487  // There is a race condition when closing a device and
488  // continuing to poll other open devices. The closed device may
489  // have outstanding events from before the close operation.
490  if (!midi) {
491  return;
492  }
493  PmEvent pm_ev;
494  PmTimeProcPtr time_proc = midi->time_proc;
495  PmTimestamp timestamp;
496 
497  /* time stamp should be in ticks, using our queue where 1 tick = 1ms */
498  assert((ev->flags & SND_SEQ_TIME_STAMP_MASK) == SND_SEQ_TIME_STAMP_TICK);
499 
500  /* if no time_proc, just return "native" ticks (ms) */
501  if (time_proc == NULL) {
502  timestamp = ev->time.tick;
503  } else { /* translate time to time_proc basis */
504  snd_seq_queue_status_t *queue_status;
505  snd_seq_queue_status_alloca(&queue_status);
506  snd_seq_get_queue_status(seq, queue, queue_status);
507  /* return (now - alsa_now) + alsa_timestamp */
508  timestamp = (*time_proc)(midi->time_info) + ev->time.tick -
509  snd_seq_queue_status_get_tick_time(queue_status);
510  }
511  pm_ev.timestamp = timestamp;
512  switch (ev->type) {
513  case SND_SEQ_EVENT_NOTEON:
514  pm_ev.message = Pm_Message(0x90 | ev->data.note.channel,
515  ev->data.note.note & 0x7f,
516  ev->data.note.velocity & 0x7f);
517  pm_read_short(midi, &pm_ev);
518  break;
519  case SND_SEQ_EVENT_NOTEOFF:
520  pm_ev.message = Pm_Message(0x80 | ev->data.note.channel,
521  ev->data.note.note & 0x7f,
522  ev->data.note.velocity & 0x7f);
523  pm_read_short(midi, &pm_ev);
524  break;
525  case SND_SEQ_EVENT_KEYPRESS:
526  pm_ev.message = Pm_Message(0xa0 | ev->data.note.channel,
527  ev->data.note.note & 0x7f,
528  ev->data.note.velocity & 0x7f);
529  pm_read_short(midi, &pm_ev);
530  break;
531  case SND_SEQ_EVENT_CONTROLLER:
532  pm_ev.message = Pm_Message(0xb0 | ev->data.note.channel,
533  ev->data.control.param & 0x7f,
534  ev->data.control.value & 0x7f);
535  pm_read_short(midi, &pm_ev);
536  break;
537  case SND_SEQ_EVENT_PGMCHANGE:
538  pm_ev.message = Pm_Message(0xc0 | ev->data.note.channel,
539  ev->data.control.value & 0x7f, 0);
540  pm_read_short(midi, &pm_ev);
541  break;
542  case SND_SEQ_EVENT_CHANPRESS:
543  pm_ev.message = Pm_Message(0xd0 | ev->data.note.channel,
544  ev->data.control.value & 0x7f, 0);
545  pm_read_short(midi, &pm_ev);
546  break;
547  case SND_SEQ_EVENT_PITCHBEND:
548  pm_ev.message = Pm_Message(0xe0 | ev->data.note.channel,
549  (ev->data.control.value + 0x2000) & 0x7f,
550  ((ev->data.control.value + 0x2000) >> 7) & 0x7f);
551  pm_read_short(midi, &pm_ev);
552  break;
553  case SND_SEQ_EVENT_CONTROL14:
554  if (ev->data.control.param < 0x20) {
555  pm_ev.message = Pm_Message(0xb0 | ev->data.note.channel,
556  ev->data.control.param,
557  (ev->data.control.value >> 7) & 0x7f);
558  pm_read_short(midi, &pm_ev);
559  pm_ev.message = Pm_Message(0xb0 | ev->data.note.channel,
560  ev->data.control.param + 0x20,
561  ev->data.control.value & 0x7f);
562  pm_read_short(midi, &pm_ev);
563  } else {
564  pm_ev.message = Pm_Message(0xb0 | ev->data.note.channel,
565  ev->data.control.param & 0x7f,
566  ev->data.control.value & 0x7f);
567 
568  pm_read_short(midi, &pm_ev);
569  }
570  break;
571  case SND_SEQ_EVENT_SONGPOS:
572  pm_ev.message = Pm_Message(0xf2,
573  ev->data.control.value & 0x7f,
574  (ev->data.control.value >> 7) & 0x7f);
575  pm_read_short(midi, &pm_ev);
576  break;
577  case SND_SEQ_EVENT_SONGSEL:
578  pm_ev.message = Pm_Message(0xf3,
579  ev->data.control.value & 0x7f, 0);
580  pm_read_short(midi, &pm_ev);
581  break;
582  case SND_SEQ_EVENT_QFRAME:
583  pm_ev.message = Pm_Message(0xf1,
584  ev->data.control.value & 0x7f, 0);

625 static PmError alsa_poll(PmInternal *midi)
626 {
627  snd_seq_event_t *ev;
628  /* expensive check for input data, gets data from device: */
629  while (snd_seq_event_input_pending(seq, TRUE) > 0) {
630  /* cheap check on local input buffer */
631  while (snd_seq_event_input_pending(seq, FALSE) > 0) {
632  /* check for and ignore errors, e.g. input overflow */
633  /* note: if there's overflow, this should be reported
634  * all the way through to client. Since input from all
635  * devices is merged, we need to find all input devices
636  * and set all to the overflow state.
637  * NOTE: this assumes every input is ALSA based.
638  */
639  int rslt = snd_seq_event_input(seq, &ev);
640  if (rslt >= 0) {
641  handle_event(ev);
642  } else if (rslt == -ENOSPC) {
643  int i;
644  for (i = 0; i < pm_descriptor_index; i++) {
645  if (descriptors[i].pub.input) {
646  PmInternal *midi = (PmInternal *)
647  descriptors[i].internalDescriptor;
648  /* careful, device may not be open! */
649  if (midi) Pm_SetOverflow(midi->queue);
650  }
651  }
652  }
653  }
654  }
655  return pmNoError;
656 }

723 PmError pm_linuxalsa_init( void )
724 {
725  int err;
726  snd_seq_client_info_t *cinfo;
727  snd_seq_port_info_t *pinfo;
728  unsigned int caps;
729 
730  /* Previously, the last parameter was SND_SEQ_NONBLOCK, but this
731  * would cause messages to be dropped if the ALSA buffer fills up.
732  * The correct behavior is for writes to block until there is
733  * room to send all the data. The client should normally allocate
734  * a large enough buffer to avoid blocking on output.
735  * Now that blocking is enabled, the seq_event_input() will block
736  * if there is no input data. This is not what we want, so must
737  * call seq_event_input_pending() to avoid blocking.
738  */
739  err = snd_seq_open(&seq, "default", SND_SEQ_OPEN_DUPLEX, 0);
740  if (err < 0) return err;
741 
742  snd_seq_client_info_alloca(&cinfo);
743  snd_seq_port_info_alloca(&pinfo);
744 
745  snd_seq_client_info_set_client(cinfo, -1);
746  while (snd_seq_query_next_client(seq, cinfo) == 0) {
747  snd_seq_port_info_set_client(pinfo, snd_seq_client_info_get_client(cinfo));
748  snd_seq_port_info_set_port(pinfo, -1);
749  while (snd_seq_query_next_port(seq, pinfo) == 0) {
750  if (snd_seq_port_info_get_client(pinfo) == SND_SEQ_CLIENT_SYSTEM)
751  continue; /* ignore Timer and Announce ports on client 0 */
752  caps = snd_seq_port_info_get_capability(pinfo);
753  if (!(caps & (SND_SEQ_PORT_CAP_SUBS_READ | SND_SEQ_PORT_CAP_SUBS_WRITE)))
754  continue; /* ignore if you cannot read or write port */
755  if (caps & SND_SEQ_PORT_CAP_SUBS_WRITE) {
756  if (pm_default_output_device_id == -1)
757  pm_default_output_device_id = pm_descriptor_index;
758  pm_add_device("ALSA",
759  pm_strdup(snd_seq_port_info_get_name(pinfo)),
760  FALSE,
761  MAKE_DESCRIPTOR(snd_seq_port_info_get_client(pinfo),
762  snd_seq_port_info_get_port(pinfo)),
763  &pm_linuxalsa_out_dictionary);
764  }
765  if (caps & SND_SEQ_PORT_CAP_SUBS_READ) {
766  if (pm_default_input_device_id == -1)
767  pm_default_input_device_id = pm_descriptor_index;
768  pm_add_device("ALSA",
769  pm_strdup(snd_seq_port_info_get_name(pinfo)),
770  TRUE,
771  MAKE_DESCRIPTOR(snd_seq_port_info_get_client(pinfo),
772  snd_seq_port_info_get_port(pinfo)),
773  &pm_linuxalsa_in_dictionary);
774  }
775  }
776  }
777  return pmNoError;
778 }
