234 bool TimerRecordDialog::RunWaitDialog()
235 {
236  AudacityProject* pProject = GetActiveProject();
237  int updateResult = eProgressSuccess;
238 
239  if (m_DateTime_Start > wxDateTime::UNow())
240  updateResult = this->WaitForStart();
241 
242  if (updateResult != eProgressSuccess)
243  {
244  // Don't proceed, but don't treat it as canceled recording. User just canceled waiting.
245  return true;
246  }
247  else
248  {
249  // Record for specified time.
250  pProject->OnRecord();
251  bool bIsRecording = true;
252 
253  wxString strMsg =
254  _("Recording start") + (wxString)wxT(":\t\t")
255  + GetDisplayDate(m_DateTime_Start) + wxT("\n") + _("Recording end")
256  + wxT(":\t\t") + GetDisplayDate(m_DateTime_End) + wxT("\n")
257  + _("Duration") + wxT(":\t\t") + m_TimeSpan_Duration.Format();
258 
259  TimerProgressDialog
260  progress(m_TimeSpan_Duration.GetMilliseconds().GetValue(),
261  _("Audacity Timer Record Progress"),
262  strMsg,
263  pdlgHideCancelButton);
264 
265  // Make sure that start and end time are updated, so we always get the full
266  // duration, even if there's some delay getting here.
267  wxTimerEvent dummyTimerEvent;
268  this->OnTimer(dummyTimerEvent);
269 
270  // Loop for progress display during recording.
271  while (bIsRecording && (updateResult == eProgressSuccess))
272  {
273  wxMilliSleep(kTimerInterval);
274  updateResult = progress.Update();
275  bIsRecording = (wxDateTime::UNow() <= m_DateTime_End); // Call UNow() again for extra accuracy...
276  }
277  }
278 
279  // Must do this AFTER the timer project dialog has been deleted to ensure the application
280  // responds to the AUDIOIO events...see not about bug #334 in the ProgressDialog constructor.
281  pProject->OnStop();
282 
283  // Let the caller handle cancellation or failure from recording progress.
284  if (updateResult == eProgressCancelled || updateResult == eProgressFailed)
285  return false;
286 
287  // Success, so let's automatically save it, for safety's sake.
288  // If user hadn't saved it before, they'll see the Save As dialog.
289  // If user had saved it before, it will safely be saved, automatically.
290  pProject->Save();
291 
292  return true;
293 }

295 wxString TimerRecordDialog::GetDisplayDate( wxDateTime & dt )
296 {
297 #if defined(__WXMSW__)
298  // On Windows, wxWidgets uses the system date control and it displays the
299  // date based on the Windows locale selected by the user. But, wxDateTime
300  // using the strftime function to return the formatted date. Since the
301  // default locale for the Windows CRT environment is "C", the dates come
302  // back in a different format.
303  //
304  // So, we make direct Windows calls to format the date like it the date
305  // control.
306  //
307  // (Most of this taken from src/msw/datectrl.cpp)
308 
309  const wxDateTime::Tm tm(dt.GetTm());
310  SYSTEMTIME st;
311  wxString s;
312  int len;
313 
314  st.wYear = (WXWORD)tm.year;
315  st.wMonth = (WXWORD)(tm.mon - wxDateTime::Jan + 1);
316  st.wDay = tm.mday;
317  st.wDayOfWeek = st.wMinute = st.wSecond = st.wMilliseconds = 0;
318 
319  len = ::GetDateFormat(LOCALE_USER_DEFAULT,
320  DATE_SHORTDATE,
321  &st,
322  NULL,
323  NULL,
324  0);
325  if (len > 0) {
326  len = ::GetDateFormat(LOCALE_USER_DEFAULT,
327  DATE_SHORTDATE,
328  &st,
329  NULL,
330  wxStringBuffer(s, len),
331  len);
332  if (len > 0) {
333  s += wxT(" ") + dt.FormatTime();
334  return s;
335  }
336  }
337 #endif
338 
339  // Use default formatting
340 wxPrintf(wxT("%s\n"), dt.Format().c_str());
341  return dt.FormatDate() + wxT(" ") + dt.FormatTime();
342 }
