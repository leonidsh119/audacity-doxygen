158 sampleCount EffectToneGen::ProcessBlock(float **WXUNUSED(inBlock), float **outBlock, sampleCount blockLen)
159 {
160  float *buffer = outBlock[0];
161  double throwaway = 0; //passed to modf but never used
162  sampleCount i;
163  double f = 0.0;
164  double a, b;
165  int k;
166 
167  double frequencyQuantum;
168  double BlendedFrequency;
169  double BlendedAmplitude;
170  double BlendedLogFrequency = 0.0;
171 
172  // calculate delta, and reposition from where we left
173  double amplitudeQuantum = (mAmplitude[1] - mAmplitude[0]) / mSampleCnt;
174  BlendedAmplitude = mAmplitude[0] + amplitudeQuantum * mSample;
175 
176  // precalculations:
177  double pre2PI = 2.0 * M_PI;
178  double pre4divPI = 4.0 / M_PI;
179 
180  // initial setup should calculate deltas
181  if (mInterpolation == kLogarithmic)
182  {
183  // this for log interpolation
184  mLogFrequency[0] = log10(mFrequency[0]);
185  mLogFrequency[1] = log10(mFrequency[1]);
186  // calculate delta, and reposition from where we left
187  frequencyQuantum = (mLogFrequency[1] - mLogFrequency[0]) / mSampleCnt;
188  BlendedLogFrequency = mLogFrequency[0] + frequencyQuantum * mSample;
189  BlendedFrequency = pow(10.0, BlendedLogFrequency);
190  }
191  else
192  {
193  // this for regular case, linear interpolation
194  frequencyQuantum = (mFrequency[1] - mFrequency[0]) / mSampleCnt;
195  BlendedFrequency = mFrequency[0] + frequencyQuantum * mSample;
196  }
197 
198  // synth loop
199  for (i = 0; i < blockLen; i++)
200  {
201  switch (mWaveform)
202  {
203  case kSine:
204  f = sin(pre2PI * mPositionInCycles / mSampleRate);
205  break;
206  case kSquare:
207  f = (modf(mPositionInCycles / mSampleRate, &throwaway) < 0.5) ? 1.0 : -1.0;
208  break;
209  case kSawtooth:
210  f = (2.0 * modf(mPositionInCycles / mSampleRate + 0.5, &throwaway)) - 1.0;
211  break;
212  case kSquareNoAlias: // Good down to 110Hz @ 44100Hz sampling.
213  //do fundamental (k=1) outside loop
214  b = (1.0 + cos((pre2PI * BlendedFrequency) / mSampleRate)) / pre4divPI; //scaling
215  f = pre4divPI * sin(pre2PI * mPositionInCycles / mSampleRate);
216  for (k = 3; (k < 200) && (k * BlendedFrequency < mSampleRate / 2.0); k += 2)
217  {
218  //Hanning Window in freq domain
219  a = 1.0 + cos((pre2PI * k * BlendedFrequency) / mSampleRate);
220  //calc harmonic, apply window, scale to amplitude of fundamental
221  f += a * sin(pre2PI * mPositionInCycles / mSampleRate * k) / (b * k);
222  }
223  }
224  // insert value in buffer
225  buffer[i] = (float) (BlendedAmplitude * f);
226  // update freq,amplitude
227  mPositionInCycles += BlendedFrequency;
228  BlendedAmplitude += amplitudeQuantum;
229  if (mInterpolation == kLogarithmic)
230  {
231  BlendedLogFrequency += frequencyQuantum;
232  BlendedFrequency = pow(10.0, BlendedLogFrequency);
233  }
234  else
235  {
236  BlendedFrequency += frequencyQuantum;
237  }
238  }
239 
240  // update external placeholder
241  mSample += blockLen;
242 
243  return blockLen;
244 }
