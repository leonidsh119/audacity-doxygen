47 bool EffectTwoPassSimpleMono::ProcessPass()
48 {
49  //Iterate over each track
50  SelectedTrackListOfKindIterator iter(Track::Wave, mOutputTracks);
51  WaveTrack *track = (WaveTrack *) iter.First();
52  mCurTrackNum = 0;
53  while (track) {
54  //Get start and end times from track
55  double trackStart = track->GetStartTime();
56  double trackEnd = track->GetEndTime();
57 
58  //Set the current bounds to whichever left marker is
59  //greater and whichever right marker is less:
60  mCurT0 = mT0 < trackStart? trackStart: mT0;
61  mCurT1 = mT1 > trackEnd? trackEnd: mT1;
62 
63  // Process only if the right marker is to the right of the left marker
64  if (mCurT1 > mCurT0) {
65 
66  //Transform the marker timepoints to samples
67  sampleCount start = track->TimeToLongSamples(mCurT0);
68  sampleCount end = track->TimeToLongSamples(mCurT1);
69 
70  //Get the track rate and samples
71  mCurRate = track->GetRate();
72  mCurChannel = track->GetChannel();
73 
74  //NewTrackPass1/2() returns true by default
75  bool ret;
76  if (mPass == 0)
77  ret = NewTrackPass1();
78  else
79  ret = NewTrackPass2();
80  if (!ret)
81  return false;
82 
83  //ProcessOne() (implemented below) processes a single track
84  if (!ProcessOne(track, start, end))
85  return false;
86  }
87 
88  //Iterate to the next track
89  track = (WaveTrack *) iter.Next();
90  mCurTrackNum++;
91  }

97 //ProcessOne() takes a track, transforms it to bunch of buffer-blocks,
98 //and executes ProcessSimpleMono on these blocks
99 bool EffectTwoPassSimpleMono::ProcessOne(WaveTrack * track,
100  sampleCount start, sampleCount end)
101 {
102  bool ret;
103  sampleCount s, samples1, samples2, tmpcount;
104  float *tmpfloat;
105 
106  //Get the length of the buffer (as double). len is
107  //used simple to calculate a progress meter, so it is easier
108  //to make it a double now than it is to do it later
109  double len = (double)(end - start);
110  sampleCount maxblock = track->GetMaxBlockSize();
111 
112  //Initiate a processing buffer. This buffer will (most likely)
113  //be shorter than the length of the track being processed.
114  float *buffer1 = new float[maxblock];
115  float *buffer2 = new float[maxblock];
116  samples1 = track->GetBestBlockSize(start);
117  if(start + samples1 > end)
118  samples1 = end - start;
119 
120  if(samples1 > maxblock)
121  samples1 = maxblock;
122 
123  //Get the samples from the track and put them in the buffer
124  track->Get((samplePtr) buffer1, floatSample, start, samples1);
125 
126  // Process the first buffer with a NULL previous buffer
127  if (mPass == 0)
128  ret = TwoBufferProcessPass1(NULL, 0, buffer1, samples1);
129  else
130  ret = TwoBufferProcessPass2(NULL, 0, buffer1, samples1);
131  if (!ret) {
132  delete[]buffer1;
133  delete[]buffer2;
134 
135  //Return false because the effect failed.
136  return false;
137  }
138 
139  //Go through the track one buffer at a time. s counts which
140  //sample the current buffer starts at.
141  s = start + samples1;
142  while (s < end) {
143  //Get a block of samples (smaller than the size of the buffer)
144  samples2 = track->GetBestBlockSize(s);
145 
146  if(samples2 > maxblock)
147  samples2 = maxblock;
148 
149  //Adjust the block size if it is the final block in the track
150  if (s + samples2 > end)
151  samples2 = end - s;
152 
153  //Get the samples from the track and put them in the buffer
154  track->Get((samplePtr) buffer2, floatSample, s, samples2);
155 
156  //Process the buffer. If it fails, clean up and exit.
157  if (mPass == 0)
158  ret = TwoBufferProcessPass1(buffer1, samples1, buffer2, samples2);
159  else
160  ret = TwoBufferProcessPass2(buffer1, samples1, buffer2, samples2);
161  if (!ret) {
162  delete[]buffer1;
163  delete[]buffer2;
164  //Return false because the effect failed.
165  return false;
166  }
167 
168  //Processing succeeded. copy the newly-changed samples back
169  //onto the track.
170  track->Set((samplePtr) buffer1, floatSample, s-samples1, samples1);
171 
172  //Increment s one blockfull of samples
173  s += samples2;
174 
175  //Update the Progress meter
176  if (mSecondPassDisabled)
177  ret = TotalProgress((mCurTrackNum + (s-start)/len) / GetNumWaveTracks());
178  else
179  ret = TotalProgress((mCurTrackNum + (s-start)/len + GetNumWaveTracks()*mPass)/ (GetNumWaveTracks()*2));
180  if (ret) {
181  delete[]buffer1;
182  delete[]buffer2;
183  //Return false because the effect failed.
184  return false;
185  }
186 
187  // Rotate the buffers
188  tmpfloat = buffer1;
189  buffer1 = buffer2;
190  buffer2 = tmpfloat;
191 
192  tmpcount = samples1;
193  samples1 = samples2;
194  samples2 = tmpcount;
195  }
196 
197  // Send the last buffer with a NULL pointer for the current buffer
198  if (mPass == 0)
199  ret = TwoBufferProcessPass1(buffer1, samples1, NULL, 0);
200  else
201  ret = TwoBufferProcessPass2(buffer1, samples1, NULL, 0);
202 
203  if (!ret) {
204  delete[]buffer1;
205  delete[]buffer2;
206  //Return false because the effect failed.
207  return false;
208  }
209 
210  //Processing succeeded. copy the newly-changed samples back
211  //onto the track.
212  track->Set((samplePtr) buffer1, floatSample, s-samples1, samples1);
213 
214  //Clean up the buffer
215  delete[]buffer1;
216  delete[]buffer2;
217 
218  //Return true because the effect processing succeeded.
219  return true;
220 }
