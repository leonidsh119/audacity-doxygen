143 static LV2_Handle
144 instantiate(const LV2_Descriptor* descriptor,
145  double rate,
146  const char* path,
147  const LV2_Feature* const* features)
148 {
149  Metro* self = (Metro*)calloc(1, sizeof(Metro));
150  if (!self) {
151  return NULL;
152  }
153 
154  // Scan host features for URID map
155  LV2_URID_Map* map = NULL;
156  for (int i = 0; features[i]; ++i) {
157  if (!strcmp(features[i]->URI, LV2_URID_URI "#map")) {
158  map = (LV2_URID_Map*)features[i]->data;
159  }
160  }
161  if (!map) {
162  fprintf(stderr, "Host does not support urid:map.\n");
163  free(self);
164  return NULL;
165  }
166 
167  // Map URIS
168  MetroURIs* const uris = &self->uris;
169  self->map = map;
170  uris->atom_Blank = map->map(map->handle, LV2_ATOM__Blank);
171  uris->atom_Float = map->map(map->handle, LV2_ATOM__Float);
172  uris->atom_Object = map->map(map->handle, LV2_ATOM__Object);
173  uris->atom_Path = map->map(map->handle, LV2_ATOM__Path);
174  uris->atom_Resource = map->map(map->handle, LV2_ATOM__Resource);
175  uris->atom_Sequence = map->map(map->handle, LV2_ATOM__Sequence);
176  uris->time_Position = map->map(map->handle, LV2_TIME__Position);
177  uris->time_barBeat = map->map(map->handle, LV2_TIME__barBeat);
178  uris->time_beatsPerMinute = map->map(map->handle, LV2_TIME__beatsPerMinute);
179  uris->time_speed = map->map(map->handle, LV2_TIME__speed);
180 
181  // Initialise instance fields
182  self->rate = rate;
183  self->bpm = 120.0f;
184  self->attack_len = (uint32_t)(attack_s * rate);
185  self->decay_len = (uint32_t)(decay_s * rate);
186  self->state = STATE_OFF;
187 
188  // Generate one cycle of a sine wave at the desired frequency
189  const double freq = 440.0 * 2.0;
190  const double amp = 0.5;
191  self->wave_len = (uint32_t)(rate / freq);
192  self->wave = (float*)malloc(self->wave_len * sizeof(float));
193  for (uint32_t i = 0; i < self->wave_len; ++i) {
194  self->wave[i] = (float)(sin(i * 2 * M_PI * freq / rate) * amp);
195  }
196 
197  return (LV2_Handle)self;
198 }

210 static void
211 play(Metro* self, uint32_t begin, uint32_t end)
212 {
213  float* const output = self->ports.output;
214  const uint32_t frames_per_beat = 60.0f / self->bpm * self->rate;
215 
216  if (self->speed == 0.0f) {
217  memset(output, 0, (end - begin) * sizeof(float));
218  return;
219  }
220 
221  for (uint32_t i = begin; i < end; ++i) {
222  switch (self->state) {
223  case STATE_ATTACK:
224  // Amplitude increases from 0..1 until attack_len
225  output[i] = self->wave[self->wave_offset] *
226  self->elapsed_len / (float)self->attack_len;
227  if (self->elapsed_len >= self->attack_len) {
228  self->state = STATE_DECAY;
229  }
230  break;
231  case STATE_DECAY:
232  // Amplitude decreases from 1..0 until attack_len + decay_len
233  output[i] = 0.0f;
234  output[i] = self->wave[self->wave_offset] *
235  (1 - ((self->elapsed_len - self->attack_len) /
236  (float)self->decay_len));
237  if (self->elapsed_len >= self->attack_len + self->decay_len) {
238  self->state = STATE_OFF;
239  }
240  break;
241  case STATE_OFF:
242  output[i] = 0.0f;
243  }
244 
245  // We continuously play the sine wave regardless of envelope
246  self->wave_offset = (self->wave_offset + 1) % self->wave_len;
247 
248  // Update elapsed time and start attack if necessary
249  if (++self->elapsed_len == frames_per_beat) {
250  self->state = STATE_ATTACK;
251  self->elapsed_len = 0;
252  }
253  }
254 }

260 static void
261 update_position(Metro* self, const LV2_Atom_Object* obj)
262 {
263  const MetroURIs* uris = &self->uris;
264 
265  // Received new transport position/speed
266  LV2_Atom *beat = NULL, *bpm = NULL, *speed = NULL;
267  lv2_atom_object_get(obj,
268  uris->time_barBeat, &beat,
269  uris->time_beatsPerMinute, &bpm,
270  uris->time_speed, &speed,
271  NULL);
272  if (bpm && bpm->type == uris->atom_Float) {
273  // Tempo changed, update BPM
274  self->bpm = ((LV2_Atom_Float*)bpm)->body;
275  }
276  if (speed && speed->type == uris->atom_Float) {
277  // Speed changed, e.g. 0 (stop) to 1 (play)
278  self->speed = ((LV2_Atom_Float*)speed)->body;
279  }
280  if (beat && beat->type == uris->atom_Float) {
281  // Received a beat position, synchronise
282  // This hard sync may cause clicks, a real plugin would be more graceful
283  const float frames_per_beat = 60.0f / self->bpm * self->rate;
284  const float bar_beats = ((LV2_Atom_Float*)beat)->body;
285  const float beat_beats = bar_beats - floorf(bar_beats);
286  self->elapsed_len = beat_beats * frames_per_beat;
287  if (self->elapsed_len < self->attack_len) {
288  self->state = STATE_ATTACK;
289  } else if (self->elapsed_len < self->attack_len + self->decay_len) {
290  self->state = STATE_DECAY;
291  } else {
292  self->state = STATE_OFF;
293  }
294  }

297 static void
298 run(LV2_Handle instance, uint32_t sample_count)
299 {
300  Metro* self = (Metro*)instance;
301  const MetroURIs* uris = &self->uris;
302 
303  // Work forwards in time frame by frame, handling events as we go
304  const LV2_Atom_Sequence* in = self->ports.control;
305  uint32_t last_t = 0;
306  for (const LV2_Atom_Event* ev = lv2_atom_sequence_begin(&in->body);
307  !lv2_atom_sequence_is_end(&in->body, in->atom.size, ev);
308  ev = lv2_atom_sequence_next(ev)) {
309 
310  // Play the click for the time slice from last_t until now
311  play(self, last_t, ev->time.frames);
312 
313  // Check if this event is an Object
314  // (or deprecated Blank to tolerate old hosts)
315  if (ev->body.type == uris->atom_Object ||
316  ev->body.type == uris->atom_Blank) {
317  const LV2_Atom_Object* obj = (const LV2_Atom_Object*)&ev->body;
318  if (obj->body.otype == uris->time_Position) {
319  // Received position information, update
320  update_position(self, obj);
321  }
322  }
323 
324  // Update time for next iteration and move to next event
325  last_t = ev->time.frames;
326  }
327 
328  // Play for remainder of cycle
329  play(self, last_t, sample_count);
330 }
