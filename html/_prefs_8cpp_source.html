74 static void CopyEntry(wxString path, wxConfigBase *src, wxConfigBase *dst, wxString entry)
75 {
76  switch(src->GetEntryType(entry)) {
77  case wxConfigBase::Type_Unknown:
78  case wxConfigBase::Type_String: {
79  wxString value = src->Read(entry, wxT(""));
80  dst->Write(path + entry, value);
81  break;
82  }
83  case wxConfigBase::Type_Boolean: {
84  bool value = false;
85  src->Read(entry, &value, value);
86  dst->Write(path + entry, value);
87  break;
88  }
89  case wxConfigBase::Type_Integer: {
90  long value = false;
91  src->Read(entry, &value, value);
92  dst->Write(path + entry, value);
93  break;
94  }
95  case wxConfigBase::Type_Float: {
96  double value = false;
97  src->Read(entry, &value, value);
98  dst->Write(path + entry, value);
99  break;
100  }

104 // Recursive routine to copy all groups and entries from one wxConfig object to another
105 static void CopyEntriesRecursive(wxString path, wxConfigBase *src, wxConfigBase *dst)
106 {
107  wxString entryName;
108  long entryIndex;
109  bool entryKeepGoing;
110 
111  entryKeepGoing = src->GetFirstEntry(entryName, entryIndex);
112  while (entryKeepGoing) {
113  CopyEntry(path, src, dst, entryName);
114  entryKeepGoing = src->GetNextEntry(entryName, entryIndex);
115  }
116 
117  wxString groupName;
118  long groupIndex;
119  bool groupKeepGoing;
120 
121  groupKeepGoing = src->GetFirstGroup(groupName, groupIndex);
122  while (groupKeepGoing) {
123  wxString subPath = path+groupName+wxT("/");
124  src->SetPath(subPath);
125  CopyEntriesRecursive(subPath, src, dst);
126  src->SetPath(path);
127  groupKeepGoing = src->GetNextGroup(groupName, groupIndex);
128  }
129 }
