95  Status(msg);
96 
97  long errorCount = 0;

99 
100  // Initialize buffers for track data to be analyzed
101  int buffSize = min(mTrack0->GetMaxBlockSize(), mTrack1->GetMaxBlockSize());
102  float *buff0 = new float[buffSize];
103  float *buff1 = new float[buffSize];
104 
105  // Compare tracks block by block
106  long s0 = mTrack0->TimeToLongSamples(mT0);
107  long s1 = mTrack0->TimeToLongSamples(mT1);
108  long position = s0;
109  long length = s1 - s0;
110  while (position < s1)
111  {
112  // Get a block of data into the buffers
113  sampleCount block = mTrack0->GetBestBlockSize(position);
114  if (position + block > s1)
115  {
116  block = s1 - position;
117  }
118  mTrack0->Get((samplePtr)buff0, floatSample, position, block);
119  mTrack1->Get((samplePtr)buff1, floatSample, position, block);
120 
121  int buffPos = 0;
122  for (buffPos = 0; buffPos < block; ++buffPos)
123  {
124  if (CompareSample(buff0[buffPos], buff1[buffPos]) > errorThreshold)
125  {
126  ++errorCount;
127  }
128  }
129 
130  position += block;
131  Progress((position - mT0) / length);
132  }
133 
134  delete [] buff0;
135  delete [] buff1;
136 
137  // Output the results
138  double errorSeconds = mTrack0->LongSamplesToTime(errorCount);
139  Status(wxString::Format(wxT("%li"), errorCount));
140  Status(wxString::Format(wxT("%.4f"), errorSeconds));
141  Status(wxString::Format(wxT("Finished comparison: %li samples (%.3f seconds) exceeded the error threshold of %f."), errorCount, errorSeconds, errorThreshold));
142  return true;
