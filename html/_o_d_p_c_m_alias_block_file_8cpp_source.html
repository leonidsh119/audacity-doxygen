290 BlockFile *ODPCMAliasBlockFile::BuildFromXML(DirManager &dm, const wxChar **attrs)
291 {
292  wxFileName summaryFileName;
293  wxFileName aliasFileName;
294  sampleCount aliasStart=0, aliasLen=0;
295  int aliasChannel=0;
296  long nValue;
297 
298  while(*attrs)
299  {
300  const wxChar *attr = *attrs++;
301  const wxChar *value = *attrs++;
302  if (!value)
303  break;
304 
305  const wxString strValue = value;
306  if (!wxStricmp(attr, wxT("summaryfile")) &&
307  // Can't use XMLValueChecker::IsGoodFileName here, but do part of its test.
308  XMLValueChecker::IsGoodFileString(strValue) &&
309  (strValue.Length() + 1 + dm.GetProjectDataDir().Length() <= PLATFORM_MAX_PATH))
310  {
311  if (!dm.AssignFile(summaryFileName, strValue, false))
312  // Make sure summaryFileName is back to uninitialized state so we can detect problem later.
313  summaryFileName.Clear();
314  }
315  else if( !wxStricmp(attr, wxT("aliasfile")) )
316  {
317  if (XMLValueChecker::IsGoodPathName(strValue))
318  aliasFileName.Assign(strValue);
319  else if (XMLValueChecker::IsGoodFileName(strValue, dm.GetProjectDataDir()))
320  // Allow fallback of looking for the file name, located in the data directory.
321  aliasFileName.Assign(dm.GetProjectDataDir(), strValue);
322  else if (XMLValueChecker::IsGoodPathString(strValue))
323  // If the aliased file is missing, we failed XMLValueChecker::IsGoodPathName()
324  // and XMLValueChecker::IsGoodFileName, because both do existence tests,
325  // but we want to keep the reference to the missing file because it's a good path string.
326  aliasFileName.Assign(strValue);
327  }
328  else if (XMLValueChecker::IsGoodInt(strValue) && strValue.ToLong(&nValue))
329  { // integer parameters
330  if (!wxStricmp(attr, wxT("aliasstart")) && (nValue >= 0))
331  aliasStart = nValue;
332  else if (!wxStricmp(attr, wxT("aliaslen")) && (nValue >= 0))
333  aliasLen = nValue;
334  else if (!wxStricmp(attr, wxT("aliaschannel")) && XMLValueChecker::IsValidChannel(aliasChannel))
335  aliasChannel = nValue;
336  }
337  }
338 
339  return new ODPCMAliasBlockFile(summaryFileName, aliasFileName,
340  aliasStart, aliasLen, aliasChannel,
341  0,0,0, false);

462 void *ODPCMAliasBlockFile::CalcSummary(samplePtr buffer, sampleCount len,
463  sampleFormat format)
464 {
465  char* localFullSummary = new char[mSummaryInfo.totalSummaryBytes];
466 
467  memcpy(localFullSummary, aheaderTag, aheaderTagLen);
468 
469  float *summary64K = (float *)(localFullSummary + mSummaryInfo.offset64K);
470  float *summary256 = (float *)(localFullSummary + mSummaryInfo.offset256);
471 
472  float *fbuffer;
473 
474  //mchinen: think we can hack this - don't allocate and copy if we don't need to.,
475  if(format==floatSample)
476  {
477  fbuffer = (float*)buffer;
478  }
479  else
480  {
481  fbuffer = new float[len];
482  CopySamples(buffer, format,
483  (samplePtr)fbuffer, floatSample, len);
484  }
485  sampleCount sumLen;
486  sampleCount i, j, jcount;
487 
488  float min, max;
489  float sumsq;
490 
491  // Recalc 256 summaries
492  sumLen = (len + 255) / 256;
493 
494 
495  for (i = 0; i < sumLen; i++) {
496  min = fbuffer[i * 256];
497  max = fbuffer[i * 256];
498  sumsq = ((float)min) * ((float)min);
499  jcount = 256;
500  if (i * 256 + jcount > len)
501  jcount = len - i * 256;
502  for (j = 1; j < jcount; j++) {
503  float f1 = fbuffer[i * 256 + j];
504  sumsq += ((float)f1) * ((float)f1);
505  if (f1 < min)
506  min = f1;
507  else if (f1 > max)
508  max = f1;
509  }
510 
511  float rms = (float)sqrt(sumsq / jcount);
512 
513  summary256[i * 3] = min;
514  summary256[i * 3 + 1] = max;
515  summary256[i * 3 + 2] = rms;
516  }
517 
518  for (i = sumLen; i < mSummaryInfo.frames256; i++) {
519  // filling in the remaining bits with non-harming/contributing values
520  summary256[i * 3] = FLT_MAX; // min
521  summary256[i * 3 + 1] = -FLT_MAX; // max
522  summary256[i * 3 + 2] = 0.0f; // rms
523  }
524 
525  // Recalc 64K summaries
526  sumLen = (len + 65535) / 65536;
527 
528  for (i = 0; i < sumLen; i++) {
529  min = summary256[3 * i * 256];
530  max = summary256[3 * i * 256 + 1];
531  sumsq = (float)summary256[3 * i * 256 + 2];
532  sumsq *= sumsq;
533 
534  for (j = 1; j < 256; j++) {
535  if (summary256[3 * (i * 256 + j)] < min)
536  min = summary256[3 * (i * 256 + j)];
537  if (summary256[3 * (i * 256 + j) + 1] > max)
538  max = summary256[3 * (i * 256 + j) + 1];
539  float r1 = summary256[3 * (i * 256 + j) + 2];
540  sumsq += r1*r1;
541  }
542 
543  float rms = (float)sqrt(sumsq / 256);
544 
545  summary64K[i * 3] = min;
546  summary64K[i * 3 + 1] = max;
547  summary64K[i * 3 + 2] = rms;
548  }
549  for (i = sumLen; i < mSummaryInfo.frames64K; i++) {
550  summary64K[i * 3] = 0.0f;
551  summary64K[i * 3 + 1] = 0.0f;
552  summary64K[i * 3 + 2] = 0.0f;
553  }
554 
555  // Recalc block-level summary
556  min = summary64K[0];
557  max = summary64K[1];
558  sumsq = (float)summary64K[2];
559  sumsq *= sumsq;
560 
561  for (i = 1; i < sumLen; i++) {
562  if (summary64K[3*i] < min)
563  min = summary64K[3*i];
564  if (summary64K[3*i+1] > max)
565  max = summary64K[3*i+1];
566  float r1 = (float)summary64K[3*i+2];
567  sumsq += (r1*r1);
568  }
569 
570  mMin = min;
571  mMax = max;
572  mRMS = sqrt(sumsq / sumLen);
573 
574 
575  //if we've used the float sample..
576  if(format!=floatSample)
577  {
578  delete[] fbuffer;
579  }
580  return localFullSummary;
581 }

594 int ODPCMAliasBlockFile::ReadData(samplePtr data, sampleFormat format,
595  sampleCount start, sampleCount len)
596 {
597 
598  LockRead();
599 
600  SF_INFO info;
601 
602  if(!mAliasedFileName.IsOk()){ // intentionally silenced
603  memset(data,0,SAMPLE_SIZE(format)*len);
604  UnlockRead();
605 
606  return len;
607  }
608 
609  memset(&info, 0, sizeof(info));
610 
611  wxString aliasPath = mAliasedFileName.GetFullPath();
612  //there are thread-unsafe crashes here - not sure why. sf_open may be called on the same file
613  //from different threads, but this seems okay, unless it is implemented strangely..
614  static ODLock sfMutex;
615 
616  wxFile f; // will be closed when it goes out of scope
617  SNDFILE *sf = NULL;
618 
619  if (f.Exists(aliasPath) && f.Open(aliasPath)) {
620  // Even though there is an sf_open() that takes a filename, use the one that
621  // takes a file descriptor since wxWidgets can open a file with a Unicode name and
622  // libsndfile can't (under Windows).
623  ODManager::LockLibSndFileMutex();
624  sf = sf_open_fd(f.fd(), SFM_READ, &info, FALSE);
625  ODManager::UnlockLibSndFileMutex();
626  }
627 
628  if (!sf){
629 
630  memset(data,0,SAMPLE_SIZE(format)*len);
631 
632  mSilentAliasLog=TRUE;
633  // Set a marker to display an error message
634  if (!wxGetApp().ShouldShowMissingAliasedFileWarning())
635  wxGetApp().MarkAliasedFilesMissingWarning(this);
636 
637  UnlockRead();
638  return len;
639  }
640 
641  mSilentAliasLog=FALSE;
642 
643  ODManager::LockLibSndFileMutex();
644  sf_seek(sf, mAliasStart + start, SEEK_SET);
645  ODManager::UnlockLibSndFileMutex();
646 
647  SampleBuffer buffer(len * info.channels, floatSample);
648 
649  int framesRead = 0;
650 
651  if (format == int16Sample &&
652  !sf_subtype_more_than_16_bits(info.format)) {
653  // Special case: if the file is in 16-bit (or less) format,
654  // and the calling method wants 16-bit data, go ahead and
655  // read 16-bit data directly. This is a pretty common
656  // case, as most audio files are 16-bit.
657  ODManager::LockLibSndFileMutex();
658  framesRead = sf_readf_short(sf, (short *)buffer.ptr(), len);
659  ODManager::UnlockLibSndFileMutex();
660 
661  for (int i = 0; i < framesRead; i++)
662  ((short *)data)[i] =
663  ((short *)buffer.ptr())[(info.channels * i) + mAliasChannel];
664  }
665  else {
666  // Otherwise, let libsndfile handle the conversion and
667  // scaling, and pass us normalized data as floats. We can
668  // then convert to whatever format we want.
669  ODManager::LockLibSndFileMutex();
670  framesRead = sf_readf_float(sf, (float *)buffer.ptr(), len);
671  ODManager::UnlockLibSndFileMutex();
672  float *bufferPtr = &((float *)buffer.ptr())[mAliasChannel];
673  CopySamples((samplePtr)bufferPtr, floatSample,
674  (samplePtr)data, format,
675  framesRead, true, info.channels);
676  }
677 
678  ODManager::LockLibSndFileMutex();
679  sf_close(sf);
680  ODManager::UnlockLibSndFileMutex();
681 
682  UnlockRead();
683  return framesRead;
684 }
