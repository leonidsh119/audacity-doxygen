132 static bool
133 is_descendant_of(SordModel* model,
134  const URIs* uris,
135  const SordNode* child,
136  const SordNode* parent,
137  const SordNode* pred)
138 {
139  if (!child) {
140  return false;
141  } else if (sord_node_equals(child, parent) ||
142  sord_ask(model, child, uris->owl_equivalentClass, parent, NULL)) {
143  return true;
144  }
145 
146  SordIter* i = sord_search(model, child, pred, NULL, NULL);
147  for (; !sord_iter_end(i); sord_iter_next(i)) {
148  const SordNode* o = sord_iter_get_node(i, SORD_OBJECT);
149  if (sord_node_equals(child, o)) {
150  continue; // Weird class is explicitly a descendent of itself
151  }
152  if (is_descendant_of(model, uris, o, parent, pred)) {
153  sord_iter_free(i);
154  return true;
155  }
156  }
157  sord_iter_free(i);
158 
159  return false;
160 }

162 static bool
163 regexp_match(const uint8_t* pat, const char* str)
164 {
165 #ifdef HAVE_PCRE
166  // Append a $ to the pattern so we only match if the entire string matches
167  const size_t len = strlen((const char*)pat);
168  char* const regx = malloc(len + 2);
169  memcpy(regx, pat, len);
170  regx[len] = '$';
171  regx[len + 1] = '\0';
172 
173  const char* err;
174  int erroffset;
175  pcre* re = pcre_compile(regx, PCRE_ANCHORED, &err, &erroffset, NULL);
176  free(regx);
177  if (!re) {
178  fprintf(stderr, "Error in pattern `%s' at offset %d (%s)\n",
179  pat, erroffset, err);
180  return false;
181  }
182 
183  const bool ret = pcre_exec(re, NULL, str, strlen(str), 0, 0, NULL, 0) >= 0;
184  pcre_free(re);
185  return ret;
186 #endif // HAVE_PCRE
187  return true;
188 }

190 static bool
191 check_restriction(SordModel* model,
192  const URIs* uris,
193  const SordNode* literal,
194  const SordNode* type,
195  const SordNode* restriction)
196 {
197  size_t len = 0;
198  const char* str = (const char*)sord_node_get_string_counted(literal, &len);
199  ++n_restrictions;
200 
201  // Check xsd:pattern
202  SordIter* p = sord_search(model, restriction, uris->xsd_pattern, 0, 0);
203  if (p) {
204  const SordNode* pat = sord_iter_get_node(p, SORD_OBJECT);
205  const bool good = regexp_match(sord_node_get_string(pat), str);
206  if (!good) {
207  fprintf(stderr, "`%s' does not match <%s> pattern `%s'\n",
208  sord_node_get_string(literal),
209  sord_node_get_string(type),
210  sord_node_get_string(pat));
211  }
212 
213  sord_iter_free(p);
214  return good;
215  }
216 
217  /* We'll do some comparison tricks for xsd:decimal types, where
218  lexicographical comparison would be incorrect. Note that if the
219  literal's type is a descendant of xsd:decimal, we'll end up checking it
220  against the xsd:decimal pattern so there's no need to validate digits
221  here. At worst we'll get a false positive but it will fail later. */
222  const bool is_decimal = is_descendant_of(
223  model, uris, type, uris->xsd_decimal, uris->owl_onDatatype);
224 
225  // Check xsd:minInclusive
226  SordIter* l = sord_search(model, restriction, uris->xsd_minInclusive, 0, 0);
227  if (l) {
228  const SordNode* lower = sord_iter_get_node(l, SORD_OBJECT);
229  size_t lower_len = 0;
230  const char* lower_str = (const char*)sord_node_get_string_counted(lower, &lower_len);
231  bool good = false;
232  if (!is_decimal || len == lower_len) {
233  // Not decimal, or equal lengths, strcmp
234  good = (strcmp(str, lower_str) >= 0);
235  } else {
236  // Decimal with different length, only good if longer than the min
237  good = (len > lower_len);
238  }
239  if (!good) {
240  fprintf(stderr, "`%s' is not >= <%s> minimum `%s'\n",
241  sord_node_get_string(literal),
242  sord_node_get_string(type),
243  sord_node_get_string(lower));
244  }
245 
246  sord_iter_free(l);
247  return good;
248  }
249 
250  // Check xsd:maxInclusive
251  SordIter* u = sord_search(model, restriction, uris->xsd_maxInclusive, 0, 0);
252  if (u) {
253  const SordNode* upper = sord_iter_get_node(u, SORD_OBJECT);
254  size_t upper_len = 0;
255  const char* upper_str = (const char*)sord_node_get_string_counted(upper, &upper_len);
256  bool good = false;
257  if (!is_decimal || len == upper_len) {
258  // Not decimal, or equal lengths, strcmp
259  good = (strcmp(str, upper_str) <= 0);
260  } else {
261  // Decimal with different length, only good if shorter than the max
262  good = (len < upper_len);
263  }
264  if (!good) {
265  fprintf(stderr, "`%s' is not <= <%s> maximum `%s'\n",
266  sord_node_get_string(literal),
267  sord_node_get_string(type),
268  sord_node_get_string(upper));
269  }
270 
271  sord_iter_free(u);
272  return good;
273  }
274 
275  --n_restrictions;
276  return true; // Unknown restriction, be quietly tolerant
277 }

279 static bool
280 literal_is_valid(SordModel* model,
281  const URIs* uris,
282  const SordNode* literal,
283  const SordNode* type)
284 {
285  if (!type) {
286  return true;
287  }
288 
289  // Find restrictions list
290  SordIter* rs = sord_search(model, type, uris->owl_withRestrictions, 0, 0);
291  if (sord_iter_end(rs)) {
292  return true; // No restrictions
293  }
294 
295  // Walk list, checking each restriction
296  const SordNode* head = sord_iter_get_node(rs, SORD_OBJECT);
297  while (head) {
298  SordIter* f = sord_search(model, head, uris->rdf_first, 0, 0);
299  if (!f) {
300  break; // Reached end of restrictions list without failure
301  }
302 
303  // Check this restriction
304  const bool good = check_restriction(
305  model, uris, literal, type, sord_iter_get_node(f, SORD_OBJECT));
306  sord_iter_free(f);
307 
308  if (!good) {
309  sord_iter_free(rs);
310  return false; // Failed, literal is invalid
311  }
312 
313  // Seek to next list node
314  SordIter* n = sord_search(model, head, uris->rdf_rest, 0, 0);
315  head = n ? sord_iter_get_node(n, SORD_OBJECT) : NULL;
316  sord_iter_free(n);
317  }
318 
319  sord_iter_free(rs);
320 
321  SordIter* s = sord_search(model, type, uris->owl_onDatatype, 0, 0);
322  if (s) {
323  const SordNode* super = sord_iter_get_node(s, SORD_OBJECT);
324  const bool good = literal_is_valid(model, uris, literal, super);
325  sord_iter_free(s);
326  return good; // Match iff literal also matches supertype
327  }
328 
329  return true; // Matches top level type
330 }

332 static bool
333 check_type(SordModel* model,
334  URIs* uris,
335  const SordNode* node,
336  const SordNode* type)
337 {
338  if (sord_node_equals(type, uris->rdfs_Resource) ||
339  sord_node_equals(type, uris->owl_Thing)) {
340  return true;
341  }
342 
343  if (sord_node_get_type(node) == SORD_LITERAL) {
344  if (sord_node_equals(type, uris->rdfs_Literal) ||
345  sord_node_equals(type, uris->xsd_string)) {
346  return true;
347  } else {
348  return literal_is_valid(model, uris, node, type);
349  }
350  } else if (sord_node_get_type(node) == SORD_URI) {
351  if (sord_node_equals(type, uris->foaf_Document)) {
352  return true; // Questionable...
353  } else if (is_descendant_of(
354  model, uris,
355  type, uris->xsd_anyURI, uris->owl_onDatatype)) {
356  /* Type is any URI and this is a URI, so pass. Restrictions on
357  anyURI subtypes are not currently checked (very uncommon). */
358  return true; // Type is anyURI, and this is a URI
359  } else {
360  SordIter* t = sord_search(model, node, uris->rdf_type, NULL, NULL);
361  for (; !sord_iter_end(t); sord_iter_next(t)) {
362  if (is_descendant_of(model, uris,
363  sord_iter_get_node(t, SORD_OBJECT),
364  type,
365  uris->rdfs_subClassOf)) {
366  sord_iter_free(t);
367  return true;
368  }
369  }
370  sord_iter_free(t);
371  return false;
372  }
373  } else {
374  return true; // Blanks often lack explicit types, ignore
375  }
376 
377  return false;
378 }
