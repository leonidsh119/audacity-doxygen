124  }
125  }

242  break;

246  break;
247  }

349 /* cviptr - convert a pointer on input */
350 LOCAL LVAL cviptr(OFFTYPE o)
351 {
352  OFFTYPE off = (OFFTYPE)2;
353  SEGMENT *seg;
354 
355  /* check for nil */
356  if (o == (OFFTYPE)0)
357  return ((LVAL)o);
358 
359  /* compute a pointer for this offset */
360  for (seg = segs; seg != NULL; seg = seg->sg_next) {
361  if (o >= off && o < off + (OFFTYPE)(seg->sg_size << 1))
362  return (seg->sg_nodes + ((int)(o - off) >> 1));
363  off += (OFFTYPE)(seg->sg_size << 1);
364  }
365 
366  /* create new segments if necessary */
367  for (;;) {
368 
369  /* create the next segment */
370  if ((seg = newsegment(anodes)) == NULL)
371  xlfatal("insufficient memory - segment");
372 
373  /* check to see if the offset is in this segment */
374  if (o >= off && o < off + (OFFTYPE)(seg->sg_size << 1))
375  return (seg->sg_nodes + ((int)(o - off) >> 1));
376  off += (OFFTYPE)(seg->sg_size << 1);
377  }

380 /* cvoptr - convert a pointer on output */
381 LOCAL OFFTYPE cvoptr(LVAL p)
382 {
383  OFFTYPE off = (OFFTYPE)2;
384  SEGMENT *seg;
385 
386  /* check for nil and small fixnums */
387  if (p == NIL)
388  return ((OFFTYPE)p);
389 
390  /* compute an offset for this pointer */
391  for (seg = segs; seg != NULL; seg = seg->sg_next) {
392  if (CVPTR(p) >= CVPTR(&seg->sg_nodes[0]) &&
393  CVPTR(p) < CVPTR(&seg->sg_nodes[0] + seg->sg_size))
394  return (off + (OFFTYPE)((p - seg->sg_nodes) << 1));
395  off += (OFFTYPE)(seg->sg_size << 1);
396  }
397

400  /* this point will never be reached because xlerror() does a
401  _longjmp(). The return is added to avoid false positive
402  error messages from static analyzers and compilers */
403  return ((OFFTYPE)NIL);
