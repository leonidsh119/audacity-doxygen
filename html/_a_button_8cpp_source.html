114 void AButton::Listener::OnKeyUp(wxKeyEvent & event)
115 {
116  event.Skip();
117 
118  if (!mButton->IsDown())
119  {
120  int idx = 0;
121  // Ignore the event, consult key states. One modifier key might
122  // have gone up but another remained down.
123  // Note that CMD (or CTRL) takes precedence over Shift if both are down
124  // and alternates are defined for both
125  // see also AButton::OnMouseEvent()
126  if (wxGetKeyState(WXK_CONTROL) && mButton->HasAlternateImages(2))
127  idx = 2;
128  else if (wxGetKeyState(WXK_SHIFT) && mButton->HasAlternateImages(1))
129  idx = 1;
130 
131  // Turn e.g. the "Play" button into a "Loop" button
132  // or "Cut Preview" button
133  mButton->SetAlternateIdx(idx);
134  }
135 }

379 void AButton::OnMouseEvent(wxMouseEvent & event)
380 {
381  wxSize clientSize = GetClientSize();
382  AButtonState prevState = GetState();
383 
384  if (event.Entering())
385  mCursorIsInWindow = true;
386  else if (event.Leaving())
387  mCursorIsInWindow = false;
388  else
389  mCursorIsInWindow =
390  (event.m_x >= 0 && event.m_y >= 0 &&
391  event.m_x < clientSize.x && event.m_y < clientSize.y);
392 
393  if (!mButtonIsDown)
394  {
395  // Note that CMD (or CTRL) takes precedence over Shift if both are down
396  // see also AButton::Listener::OnKeyUp()
397  if (event.CmdDown() && HasAlternateImages(2))
398  mAlternateIdx = 2;
399  else if (event.ShiftDown() && HasAlternateImages(1))
400  mAlternateIdx = 1;
401  else
402  mAlternateIdx = 0;
403  }
404 
405  if (mEnabled && event.IsButton()) {
406  if (event.ButtonIsDown(wxMOUSE_BTN_ANY)) {
407  mIsClicking = true;
408  if( !HasCapture() )
409  CaptureMouse();
410  }
411  else if (mIsClicking) {
412  mIsClicking = false;
413 
414  if (HasCapture())
415  ReleaseMouse();
416 
417  if (mCursorIsInWindow && (mToggle || !mButtonIsDown)) {
418  if (mToggle)
419  mButtonIsDown = !mButtonIsDown;
420  else
421  mButtonIsDown = true;
422 
423  mWasShiftDown = event.ShiftDown();
424  mWasControlDown = event.ControlDown();
425 
426  Click();
427  }
428  }
429  }
430 
431  // Only redraw and change tooltips if the state has changed.
432  AButtonState newState = GetState();
433 
434  if (newState != prevState) {
435  Refresh(false);
436 
437  if (mCursorIsInWindow) {
438  #if wxUSE_TOOLTIPS // Not available in wxX11
439  // Display the tooltip in the status bar
440  wxToolTip * pTip = this->GetToolTip();
441  if( pTip ) {
442  wxString tipText = pTip->GetTip();
443  if (!mEnabled)
444  tipText += _(" (disabled)");
445  GetActiveProject()->TP_DisplayStatusMessage(tipText);
446  }
447  #endif
448  }
449  else {
450  GetActiveProject()->TP_DisplayStatusMessage(wxT(""));
451  }
452  }
453 }

466 {
467  switch( event.GetKeyCode() )
