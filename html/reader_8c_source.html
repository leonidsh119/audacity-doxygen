131 {
132  assert(peek_byte(reader) == byte);
133  switch (byte) {
134  case '\0': reader->eof = true; break;
135  case '\n': ++reader->cur.line; reader->cur.col = 0; break;
136  default: ++reader->cur.col;

283 // Read UCHAR escape, initial \ is already eaten by caller
284 static inline bool
285 read_UCHAR(SerdReader* reader, Ref dest, uint32_t* char_code)
286 {
287  const uint8_t b = peek_byte(reader);
288  unsigned length = 0;
289  switch (b) {
290  case 'U':
291  length = 8;
292  break;
293  case 'u':
294  length = 4;
295  break;
296  default:
297  return false;
298  }
299  eat_byte_safe(reader, b);
300 
301  uint8_t buf[9] = { 0, 0, 0, 0, 0, 0, 0, 0, 0 };
302  for (unsigned i = 0; i < length; ++i) {
303  if (!(buf[i] = read_HEX(reader))) {
304  return false;
305  }
306  }
307 
308  uint32_t code;
309  sscanf((const char*)buf, "%X", &code);
310 
311  unsigned size = 0;
312  if (code < 0x00000080) {
313  size = 1;
314  } else if (code < 0x00000800) {
315  size = 2;
316  } else if (code < 0x00010000) {
317  size = 3;
318  } else if (code < 0x00110000) {
319  size = 4;
320  } else {
321  r_err(reader, SERD_ERR_BAD_SYNTAX,
322  "unicode character 0x%X out of range\n", code);
323  push_replacement(reader, dest);
324  *char_code = 0xFFFD;
325  return true;
326  }
327 
328  // Build output in buf
329  // (Note # of bytes = # of leading 1 bits in first byte)
330  uint32_t c = code;
331  switch (size) {
332  case 4:
333  buf[3] = 0x80 | (uint8_t)(c & 0x3F);
334  c >>= 6;
335  c |= (16 << 12); // set bit 4
336  case 3:
337  buf[2] = 0x80 | (uint8_t)(c & 0x3F);
338  c >>= 6;
339  c |= (32 << 6); // set bit 5
340  case 2:
341  buf[1] = 0x80 | (uint8_t)(c & 0x3F);
342  c >>= 6;
343  c |= 0xC0; // set bits 6 and 7
344  case 1:
345  buf[0] = (uint8_t)c;
346  }
347 
348  for (unsigned i = 0; i < size; ++i) {
349  push_byte(reader, dest, buf[i]);
350  }
351  *char_code = code;
352  return true;
353 }

406 static SerdStatus
407 read_utf8_character(SerdReader* reader, Ref dest, uint8_t c)
408 {
409  unsigned size = 1;
410  if ((c & 0xE0) == 0xC0) { // Starts with `110'
411  size = 2;
412  } else if ((c & 0xF0) == 0xE0) { // Starts with `1110'
413  size = 3;
414  } else if ((c & 0xF8) == 0xF0) { // Starts with `11110'
415  size = 4;
416  } else {
417  return bad_char(reader, dest, "invalid UTF-8 start 0x%X\n", c);
418  }
419 
420  char bytes[4];
421  bytes[0] = c;
422 
423  // Check character validity
424  for (unsigned i = 1; i < size; ++i) {
425  if (((bytes[i] = peek_byte(reader)) & 0x80) == 0) {
426  return bad_char(reader, dest, "invalid UTF-8 continuation 0x%X\n",
427  bytes[i]);
428  }
429  eat_byte_safe(reader, bytes[i]);
430  }
431 
432  // Emit character
433  for (unsigned i = 0; i < size; ++i) {
434  push_byte(reader, dest, bytes[i]);
435  }
436  return SERD_SUCCESS;
437 }

690  case '6': case '7': case '8': case '9': case ':': case '_':
691  push_byte(reader, dest, eat_byte_safe(reader, c));
692  break;
693  default:
694  if ((st = read_PLX(reader, dest)) > SERD_FAILURE) {
695  return st;
696  } else if (st != SERD_SUCCESS && !read_PN_CHARS_BASE(reader, dest)) {
697  return SERD_FAILURE;
698  }
699  }
700 
701  while ((c = peek_byte(reader))) { // Middle: (PN_CHARS | '.' | ';')*
702  if (c == '.' || c == ':') {
703  push_byte(reader, dest, eat_byte_safe(reader, c));
704  } else if ((st = read_PLX(reader, dest)) > SERD_FAILURE) {
705  return st;
706  } else if (st != SERD_SUCCESS && !read_PN_CHARS(reader, dest)) {
707  break;
708  }
709  }
710 
711  SerdNode* const n = deref(reader, dest);
712  if (n->buf[n->n_bytes - 1] == '.') {
713  // Ate trailing dot, pop it from stack/node and inform caller
714  --n->n_bytes;
715  serd_stack_pop(&reader->stack, 1);
716  *ate_dot = true;
717  }
718 
719  return SERD_SUCCESS;
720 }

774 static Ref

843 static bool
844 read_number(SerdReader* reader, Ref* dest, Ref* datatype, bool* ate_dot)
845 {
846  #define XSD_DECIMAL NS_XSD "decimal"
847  #define XSD_DOUBLE NS_XSD "double"
848  #define XSD_INTEGER NS_XSD "integer"
849  Ref ref = push_node(reader, SERD_LITERAL, "", 0);
850  uint8_t c = peek_byte(reader);
851  bool has_decimal = false;
852  if (c == '-' || c == '+') {
853  push_byte(reader, ref, eat_byte_safe(reader, c));
854  }
855  if ((c = peek_byte(reader)) == '.') {
856  has_decimal = true;
857  // decimal case 2 (e.g. '.0' or `-.0' or `+.0')
858  push_byte(reader, ref, eat_byte_safe(reader, c));
859  TRY_THROW(read_0_9(reader, ref, true));
860  } else {
861  // all other cases ::= ( '-' | '+' ) [0-9]+ ( . )? ( [0-9]+ )? ...
862  TRY_THROW(is_digit(c));
863  read_0_9(reader, ref, true);
864  if ((c = peek_byte(reader)) == '.') {
865  has_decimal = true;
866 
867  // Annoyingly, dot can be end of statement, so tentatively eat
868  eat_byte_safe(reader, c);
869  c = peek_byte(reader);
870  if (!is_digit(c) && c != 'e' && c != 'E') {
871  *dest = ref;
872  *ate_dot = true; // Force caller to deal with stupid grammar
873  return true; // Next byte is not a number character, done
874  }
875 
876  push_byte(reader, ref, '.');
877  read_0_9(reader, ref, false);
878  }
879  }
880  c = peek_byte(reader);
881  if (c == 'e' || c == 'E') {
882  // double
883  push_byte(reader, ref, eat_byte_safe(reader, c));
884  switch ((c = peek_byte(reader))) {
885  case '+': case '-':
886  push_byte(reader, ref, eat_byte_safe(reader, c));
887  default: break;
888  }
889  TRY_THROW(read_0_9(reader, ref, true));
890  *datatype = push_node(reader, SERD_URI,
891  XSD_DOUBLE, sizeof(XSD_DOUBLE) - 1);
892  } else if (has_decimal) {
893  *datatype = push_node(reader, SERD_URI,
894  XSD_DECIMAL, sizeof(XSD_DECIMAL) - 1);
895  } else {
896  *datatype = push_node(reader, SERD_URI,
897  XSD_INTEGER, sizeof(XSD_INTEGER) - 1);
898  }
899  *dest = ref;
900  return true;
901 except:
902  pop_node(reader, *datatype);
903  pop_node(reader, ref);
904  return false;
905 }

1108 // Recurses, calling statement_sink for every statement encountered.
1109 // Leaves stack in original calling state (i.e. pops everything it pushes).
1110 static bool
1111 read_object(SerdReader* reader, ReadContext ctx, bool* ate_dot)
1112 {
1113  static const char* const XSD_BOOLEAN = NS_XSD "boolean";
1114  static const size_t XSD_BOOLEAN_LEN = 40;
1115 
1116 #ifndef NDEBUG
1117  const size_t orig_stack_size = reader->stack.size;
1118 #endif
1119 
1120  bool ret = false;
1121  bool emit = (ctx.subject != 0);
1122  SerdNode* node = NULL;
1123  Ref o = 0;
1124  Ref datatype = 0;
1125  Ref lang = 0;
1126  uint32_t flags = 0;
1127  const uint8_t c = peek_byte(reader);
1128  switch (c) {
1129  case '\0':
1130  case ')':
1131  return false;
1132  case '[': case '(':
1133  emit = false;
1134  // fall through
1135  case '_':
1136  TRY_THROW(ret = read_blank(reader, ctx, false, &o));
1137  break;
1138  case '<': case ':':
1139  TRY_THROW(ret = read_iri(reader, &o, ate_dot));
1140  break;
1141  case '+': case '-': case '.': case '0': case '1': case '2': case '3':
1142  case '4': case '5': case '6': case '7': case '8': case '9':
1143  TRY_THROW(ret = read_number(reader, &o, &datatype, ate_dot));
1144  break;
1145  case '\"':
1146  case '\'':
1147  TRY_THROW(ret = read_literal(reader, &o, &datatype, &lang, &flags, ate_dot));
1148  break;
1149  default:
1150  /* Either a boolean literal, or a qname. Read the prefix first, and if
1151  it is in fact a "true" or "false" literal, produce that instead.
1152  */
1153  node = deref(reader, o = push_node(reader, SERD_CURIE, "", 0));
1154  while (read_PN_CHARS_BASE(reader, o)) {}
1155  if ((node->n_bytes == 4 && !memcmp(node->buf, "true", 4)) ||
1156  (node->n_bytes == 5 && !memcmp(node->buf, "false", 5))) {
1157  node->type = SERD_LITERAL;
1158  datatype = push_node(
1159  reader, SERD_URI, XSD_BOOLEAN, XSD_BOOLEAN_LEN);
1160  ret = true;
1161  } else if (read_PN_PREFIX_tail(reader, o) > SERD_FAILURE) {
1162  ret = false;
1163  } else {
1164  ret = read_PrefixedName(reader, o, false, ate_dot);
1165  }
1166  }
1167 
1168  if (ret && emit) {
1169  deref(reader, o)->flags = flags;
1170  ret = emit_statement(reader, ctx, o, datatype, lang);
1171  }
1172 
1173 except:
1174  pop_node(reader, lang);
1175  pop_node(reader, datatype);
1176  pop_node(reader, o);
1177 #ifndef NDEBUG
1178  assert(reader->stack.size == orig_stack_size);
1179 #endif
1180  return ret;
1181 }

1236 static bool
1237 read_collection(SerdReader* reader, ReadContext ctx, Ref* dest)
1238 {

1242  if (ctx.subject) {

1244  *ctx.flags |= (end ? 0 : SERD_LIST_O_BEGIN);
1245  TRY_RET(emit_statement(reader, ctx, *dest, 0, 0));
1246  *ctx.flags |= SERD_LIST_CONT;

1249  }

1252  return end_collection(reader, ctx, 0, 0, true);
1253  }

1258  Ref n2 = 0;

1261

1263  while (!(end = peek_delim(reader, ')'))) {
1264  // _:node rdf:first object
1265  ctx.predicate = reader->rdf_first;

1268  return end_collection(reader, ctx, n1, n2, false);
1269  }
1270

1274  if (!rest) {

1276  } else {
1277  set_blank_id(reader, rest, genid_size(reader));
1278  }
1279  }

1285  (end ? reader->rdf_nil : rest), 0, 0));
1286 
1287  ctx.subject = rest; // _:node = _:rest
1288  rest = node; // _:rest = (old)_:node
1289  node = ctx.subject; // invariant
1290  }
1291
