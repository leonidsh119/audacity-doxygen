110 char*
111 lilv_get_lang(void)
112 {
113  const char* const env_lang = getenv("LANG");
114  if (!env_lang || !strcmp(env_lang, "")
115  || !strcmp(env_lang, "C") || !strcmp(env_lang, "POSIX")) {
116  return NULL;
117  }
118 
119  const size_t env_lang_len = strlen(env_lang);
120  char* const lang = (char*)malloc(env_lang_len + 1);
121  for (size_t i = 0; i < env_lang_len + 1; ++i) {
122  if (env_lang[i] == '_') {
123  lang[i] = '-'; // Convert _ to -
124  } else if (env_lang[i] >= 'A' && env_lang[i] <= 'Z') {
125  lang[i] = env_lang[i] + ('a' - 'A'); // Convert to lowercase
126  } else if (env_lang[i] >= 'a' && env_lang[i] <= 'z') {
127  lang[i] = env_lang[i]; // Lowercase letter, copy verbatim
128  } else if (env_lang[i] >= '0' && env_lang[i] <= '9') {
129  lang[i] = env_lang[i]; // Digit, copy verbatim
130  } else if (env_lang[i] == '\0' || env_lang[i] == '.') {
131  // End, or start of suffix (e.g. en_CA.utf-8), finished
132  lang[i] = '\0';
133  break;
134  } else {
135  LILV_ERRORF("Illegal LANG `%s' ignored\n", env_lang);
136  free(lang);
137  return NULL;
138  }
139  }
140 
141  return lang;
142 }

171 {
172 #ifdef _WIN32
173  char* out = (char*)malloc(MAX_PATH);
174  ExpandEnvironmentStrings(path, out, MAX_PATH);
175 #else
176  char* out = NULL;
177  size_t len = 0;
178 
179  const char* start = path; // Start of current chunk to copy
180  for (const char* s = path; *s;) {
181  if (*s == '$') {
182  // Hit $ (variable reference, e.g. $VAR_NAME)
183  for (const char* t = s + 1; ; ++t) {
184  if (!*t || (!isupper(*t) && !isdigit(*t) && *t != '_')) {
185  // Append preceding chunk
186  out = strappend(out, &len, start, s - start);
187 
188  // Append variable value (or $VAR_NAME if not found)
189  char* var = (char*)calloc(t - s, 1);
190  memcpy(var, s + 1, t - s - 1);
191  out = append_var(out, &len, var);
192  free(var);
193 
194  // Continue after variable reference
195  start = s = t;
196  break;
197  }
198  }
199  } else if (*s == '~' && (*(s + 1) == '/' || !*(s + 1))) {
200  // Hit ~ before slash or end of string (home directory reference)
201  out = strappend(out, &len, start, s - start);
202  out = append_var(out, &len, "HOME");
203  start = ++s;
204  } else {
205  ++s;
206  }
207  }
208 
209  if (*start) {
210  out = strappend(out, &len, start, strlen(start));
211  }
212 #endif
213

441 char*
442 lilv_path_relative_to(const char* path, const char* base)
443 {
444  const size_t path_len = strlen(path);
445  const size_t base_len = strlen(base);
446  const size_t min_len = (path_len < base_len) ? path_len : base_len;
447 
448  // Find the last separator common to both paths
449  size_t last_shared_sep = 0;
450  for (size_t i = 0; i < min_len && path[i] == base[i]; ++i) {
451  if (lilv_is_dir_sep(path[i])) {
452  last_shared_sep = i;
453  }
454  }
455 
456  if (last_shared_sep == 0) {
457  // No common components, return path
458  return lilv_strdup(path);
459  }
460 
461  // Find the number of up references ("..") required
462  size_t up = 0;
463  for (size_t i = last_shared_sep + 1; i < base_len; ++i) {
464  if (lilv_is_dir_sep(base[i])) {
465  ++up;
466  }
467  }
468 
469  // Write up references
470  const size_t suffix_len = path_len - last_shared_sep;
471  char* rel = (char*)calloc(1, suffix_len + (up * 3) + 1);
472  for (size_t i = 0; i < up; ++i) {
473  memcpy(rel + (i * 3), "../", 3);
474  }
475 
476  // Write suffix
477  memcpy(rel + (up * 3), path + last_shared_sep + 1, suffix_len);
478  return rel;
479 }
