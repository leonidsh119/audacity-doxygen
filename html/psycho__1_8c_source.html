131 static inline FLOAT add_db(psycho_1_mem * mem, FLOAT a, FLOAT b)
132 {
133  /* MFC - if the difference between a and b is large (>99), then just return the largest one.
134  (about 10% of the time) - For differences between 0 and 99, return the largest value, but
135  add in a pre-calculated difference value. - the value 99 was chosen arbitarily. - maximum
136  (a-b) i've seen is 572 */
137  FLOAT fdiff;
138  int idiff;
139  fdiff = (10.0 * (a - b));
140 
141  if (fdiff > 990.0) {
142  return a;
143  }
144  if (fdiff < -990.0) {
145  return (b);
146  }
147 
148  idiff = (int) fdiff;
149  if (idiff >= 0) {
150  return (a + mem->dbtable[idiff]);
151  }
152 
153  return (b + mem->dbtable[-idiff]);
154 }

157 * Window the samples then,
158 * Fast Fourier transform of the input samples.

164 static void psycho_1_hann_fft_pickmax(FLOAT sample[FFT_SIZE], mask power[HAN_SIZE],
165  FLOAT spike[SBLIMIT], FLOAT energy[FFT_SIZE])
166 {
167  FLOAT x_real[FFT_SIZE];
168  register int i, j;
169  register FLOAT sqrt_8_over_3;
170  static int init = 0;
171  static FLOAT window[FFT_SIZE];
172  FLOAT sum;
173 
174  if (!init) {
175  /* calculate window function for the Fourier transform */
176  /* These values need only be initiliased once, regardless of the caller */
177  sqrt_8_over_3 = pow(8.0 / 3.0, 0.5);
178  for (i = 0; i < FFT_SIZE; i++) {
179  /* Hann window formula */
180  window[i] = sqrt_8_over_3 * 0.5 * (1 - cos(2.0 * PI * i / (FFT_SIZE))) / FFT_SIZE;
181  }
182  init = 1;
183  }
184  for (i = 0; i < FFT_SIZE; i++)
185  x_real[i] = (FLOAT) (sample[i] * window[i]);
186 
187  psycho_1_fft(x_real, energy, FFT_SIZE);
188 
189  for (i = 0; i < HAN_SIZE; i++) { /* calculate power density spectrum */
190  if (energy[i] < 1E-20)
191  power[i].x = -200.0 + POWERNORM;
192  else
193  power[i].x = 10 * log10(energy[i]) + POWERNORM;
194  power[i].next = STOP;
195  power[i].type = FALSE;
196  }
197 
198  /* Calculate the sum of spectral component in each subband from bound 4-16 */
199 
200 #define CF 1073741824 /* pow(10, 0.1*POWERNORM) */
201 #define DBM 1E-20 /* pow(10.0, 0.1*DBMIN */
202  for (i = 0; i < HAN_SIZE; spike[i >> 4] = 10.0 * log10(sum), i += 16) {
203  for (j = 0, sum = DBM; j < 16; j++)
204  sum += CF * energy[i + j];
205  }
206 }

215 static void psycho_1_tonal_label(psycho_1_mem * mem, int *tone)
216 /* this function extracts (tonal) sinusoidals from the spectrum */
217 {
218  int i, j, last = LAST, first, run, last_but_one = LAST; /* dpwe */
219  FLOAT max;
220  mask *power = mem->power;
221 
222  *tone = LAST;
223  for (i = 2; i < HAN_SIZE - 12; i++) {
224  if (power[i].x > power[i - 1].x && power[i].x >= power[i + 1].x) {
225  power[i].type = TONE;
226  power[i].next = LAST;
227  if (last != LAST)
228  power[last].next = i;
229  else
230  first = *tone = i;
231  last = i;
232  }
233  }
234  last = LAST;
235  first = *tone;
236  *tone = LAST;
237  while ((first != LAST) && (first != STOP)) { /* the conditions for the tonal */
238  if (first < 3 || first > 500)
239  run = 0; /* otherwise k+/-j will be out of bounds */
240  else if (first < 63)
241  run = 2; /* components in layer II, which */
242  else if (first < 127)
243  run = 3; /* are the boundaries for calc. */
244  else if (first < 255)
245  run = 6; /* the tonal components */
246  else
247  run = 12;
248  max = power[first].x - 7; /* after calculation of tonal */
249  for (j = 2; j <= run; j++) /* components, set to local max */
250  if (max < power[first - j].x || max < power[first + j].x) {
251  power[first].type = FALSE;
252  break;
253  }
254  if (power[first].type == TONE) { /* extract tonal components */
255  int help = first;
256  if (*tone == LAST)
257  *tone = first;
258  while ((power[help].next != LAST) && (power[help].next - first) <= run)
259  help = power[help].next;
260  help = power[help].next;
261  power[first].next = help;
262  if ((first - last) <= run) {
263  if (last_but_one != LAST)
264  power[last_but_one].next = first;
265  }
266  if (first > 1 && first < 500) { /* calculate the sum of the */
267  FLOAT tmp; /* powers of the components */
268  tmp = add_db(mem, power[first - 1].x, power[first + 1].x);
269  power[first].x = add_db(mem, power[first].x, tmp);
270  }
271  for (j = 1; j <= run; j++) {
272  power[first - j].x = power[first + j].x = DBMIN;
273  power[first - j].next = power[first + j].next = STOP;
274  power[first - j].type = power[first + j].type = FALSE;
275  }
276  last_but_one = last;
277  last = first;
278  first = power[first].next;
279  } else {
280  int ll;
281  if (last == LAST); /* *tone = power[first].next; dpwe */
282  else
283  power[last].next = power[first].next;
284  ll = first;
285  first = power[first].next;
286  power[ll].next = STOP;
287  }
288  }
289 }

299 static void psycho_1_noise_label(psycho_1_mem * mem, int *noise, FLOAT energy[FFT_SIZE])
300 {
301  int i, j, centre, last = LAST;
302  FLOAT index, weight, sum;
303  int crit_band = mem->crit_band;
304  int *cbound = mem->cbound;
305  mask *power = mem->power;
306  /* calculate the remaining spectral */
307  for (i = 0; i < crit_band - 1; i++) { /* lines for non-tonal components */
308  for (j = cbound[i], weight = 0.0, sum = DBMIN; j < cbound[i + 1]; j++) {
309  if (power[j].type != TONE) {
310  if (power[j].x != DBMIN) {
311  sum = add_db(mem, power[j].x, sum);
312  /* Weight is used in finding the geometric mean of the noise energy within a
313  subband */
314  weight += CF * energy[j] * (FLOAT) (j - cbound[i]) / (FLOAT) (cbound[i + 1] - cbound[i]); /* correction
315  */
316  power[j].x = DBMIN;
317  }
318  } /* check to see if the spectral line is low dB, and if */
319  } /* so replace the center of the critical band, which is */
320  /* the center freq. of the noise component */
321 
322  if (sum <= DBMIN)
323  centre = (cbound[i + 1] + cbound[i]) / 2;
324  else {
325  /* fprintf(stderr, "%i [%f %f] -", count++,weight/pow(10.0,0.1*sum),
326  weight*pow(10.0,-0.1*sum)); */
327  index = weight * pow(10.0, -0.1 * sum);
328  centre = cbound[i] + (int) (index * (FLOAT) (cbound[i + 1] - cbound[i]));
329  }
330 
331 
332  /* locate next non-tonal component until finished; */
333  /* add to list of non-tonal components */
334 
335  /* Masahiro Iwadare's fix for infinite looping problem? */
336  if (power[centre].type == TONE) {
337  if (power[centre + 1].type == TONE) {
338  centre++;
339  } else
340  centre--;
341  }
342 
343  if (last == LAST)
344  *noise = centre;
345  else {
346  power[centre].next = LAST;
347  power[last].next = centre;
348  }
349  power[centre].x = sum;
350  power[centre].type = NOISE;
351  last = centre;
352  }
353 }

430 /* mainly just changed the way range checking was done MFC Nov 1999 */
431 static void psycho_1_threshold(psycho_1_mem * mem, int *tone, int *noise, int bit_rate)
432 {
433  int sub_size = mem->sub_size;
434  mask *power = mem->power;
435  g_thres *ltg = mem->ltg;
436  int k, t;
437  FLOAT dz, tmps, vf;
438 
439  for (k = 1; k < sub_size; k++) {
440  ltg[k].x = DBMIN;
441  t = *tone; /* calculate individual masking threshold for */
442  while ((t != LAST) && (t != STOP)) { /* components in order to find the global */
443  dz = ltg[k].bark - ltg[power[t].map].bark; /* distance of bark value */
444  if (dz >= -3.0 && dz < 8.0) {
445  tmps = -1.525 - 0.275 * ltg[power[t].map].bark - 4.5 + power[t].x;
446  /* masking function for lower & upper slopes */
447  if (dz < -1)
448  vf = 17 * (dz + 1) - (0.4 * power[t].x + 6);
449  else if (dz < 0)
450  vf = (0.4 * power[t].x + 6) * dz;
451  else if (dz < 1)
452  vf = (-17 * dz);
453  else
454  vf = -(dz - 1) * (17 - 0.15 * power[t].x) - 17;
455  ltg[k].x = add_db(mem, ltg[k].x, tmps + vf);
456  }
457  t = power[t].next;
458  }
459 
460  t = *noise; /* calculate individual masking threshold */
461  while ((t != LAST) && (t != STOP)) { /* for non-tonal components to find LTG */
462  dz = ltg[k].bark - ltg[power[t].map].bark; /* distance of bark value */
463  if (dz >= -3.0 && dz < 8.0) {
464  tmps = -1.525 - 0.175 * ltg[power[t].map].bark - 0.5 + power[t].x;
465  /* masking function for lower & upper slopes */
466  if (dz < -1)
467  vf = 17 * (dz + 1) - (0.4 * power[t].x + 6);
468  else if (dz < 0)
469  vf = (0.4 * power[t].x + 6) * dz;
470  else if (dz < 1)
471  vf = (-17 * dz);
472  else
473  vf = -(dz - 1) * (17 - 0.15 * power[t].x) - 17;
474  ltg[k].x = add_db(mem, ltg[k].x, tmps + vf);
475  }
476  t = power[t].next;
477  }
478  if (bit_rate < 96)
479  ltg[k].x = add_db(mem, ltg[k].hear, ltg[k].x);
480  else
481  ltg[k].x = add_db(mem, ltg[k].hear - 12.0, ltg[k].x);
482  }
