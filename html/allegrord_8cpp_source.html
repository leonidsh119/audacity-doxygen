423 {

442  // special case: entire string is "-", interpret as -1
443  return -1;
444  }

686 bool Alg_reader::parse_val(Alg_parameter_ptr param, string &s, int i)
687 {
688  int len = (int) s.length();
689  if (i >= len) {
690  return false;
691  }
692  if (s[i] == '"') {
693  if (!check_type('s', param)) {
694  return false;
695  }
696  // note: (len - i) includes 2 quote characters but no EOS character
697  // so total memory to allocate is (len - i) - 1
698  char *r = new char[(len - i) - 1];
699  strncpy(r, s.c_str() + i + 1, (len - i) - 2);
700  r[(len - i) - 2] = 0; // terminate the string
701  param->s = r;
702  } else if (s[i] == '\'') {
703  if (!check_type('a', param)) {
704  return false;
705  }
706  string r = s.substr(i + 1, len - i - 2);
707  param->a = symbol_table.insert_string(r.c_str());
708  } else if (param->attr_type() == 'l') {
709  if (streql(s.c_str() + i, "true") ||
710  streql(s.c_str() + i, "t")) {
711  param->l = true;
712  } else if (streql(s.c_str() + i, "false") ||
713  streql(s.c_str() + i, "nil")) {
714  param->l = false;
715  } else return false;
716  } else if (isdigit(s[i]) || s[i] == '-' || s[i] == '.') {
717  int pos = i;
718  bool period = false;
719  int sign = 1;
720  if (s[pos] == '-') {
721  sign = -1;
722  pos++;
723  }
724  while (pos < len) {
725  if (isdigit(s[pos])) {
726  ;
727  } else if (!period && s[pos] == '.') {
728  period = true;
729  } else {
730  parse_error(s, pos, "Unexpected char in number");
731  return false;
732  }
733  pos = pos + 1;
734  }
735  string r = s.substr(i, len - i);
736  if (period) {
737  if (!check_type('r', param)) {
738  return false;
739  }
740  param->r = atof(r.c_str());
741  } else {
742  if (param->attr_type() == 'r') {
743  param->r = atoi(r.c_str());
744  } else if (!check_type('i', param)) {
745  return false;
746  } else {
747  param->i = atoi(r.c_str());
748  }
749  }
750  } else {
751  parse_error(s, i, "invalid value");
752  return false;
753  }
754  return true;
755 }
