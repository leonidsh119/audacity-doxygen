159 bool EffectAmplify::Init()
160 {
161  mPeak = 0.0;
162 
163  SelectedTrackListOfKindIterator iter(Track::Wave, mTracks);
164 
165  for (Track *t = iter.First(); t; t = iter.Next())
166  {
167  float min, max;
168  ((WaveTrack *)t)->GetMinMax(&min, &max, mT0, mT1);
169  float newpeak = (fabs(min) > fabs(max) ? fabs(min) : fabs(max));
170 
171  if (newpeak > mPeak)
172  {
173  mPeak = newpeak;
174  }
175  }
176 
177  return true;
