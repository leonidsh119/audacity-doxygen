161 void DevicePrefs::OnHost(wxCommandEvent & e)
162 {
163  // Bail if we have no hosts
164  if (mHostNames.size() < 1)
165  return;
166 
167  // Find the index for the host API selected
168  int index = -1;
169  wxString apiName = mHostNames[mHost->GetCurrentSelection()];
170  int nHosts = Pa_GetHostApiCount();
171  for (int i = 0; i < nHosts; ++i) {
172  wxString name = wxSafeConvertMB2WX(Pa_GetHostApiInfo(i)->name);
173  if (name == apiName) {
174  index = i;
175  break;
176  }
177  }
178  // We should always find the host!
179  if (index < 0) {
180  wxLogDebug(wxT("DevicePrefs::OnHost(): API index not found"));
181  return;
182  }
183 
184  int nDevices = Pa_GetDeviceCount();
185 
186  if (nDevices == 0) {
187  mHost->Clear();
188  mHost->Append(_("No audio interfaces"), (void *) NULL);
189  mHost->SetSelection(0);
190  }
191 
192  const std::vector<DeviceSourceMap> &inMaps = DeviceManager::Instance()->GetInputDeviceMaps();
193  const std::vector<DeviceSourceMap> &outMaps = DeviceManager::Instance()->GetOutputDeviceMaps();
194 
195  wxArrayString playnames;
196  wxArrayString recordnames;
197  size_t i;
198  int devindex; /* temp variable to hold the numeric ID of each device in turn */
199  wxString device;
200  wxString recDevice;
201 
202  recDevice = mRecordDevice;
203  if (this->mRecordSource != wxT(""))
204  recDevice += wxT(": ") + mRecordSource;
205 
206  mRecord->Clear();
207  for (i = 0; i < inMaps.size(); i++) {
208  if (index == inMaps[i].hostIndex) {
209  device = MakeDeviceSourceString(&inMaps[i]);
210  devindex = mRecord->Append(device);
211  // We need to const cast here because SetClientData is a wx function
212  // It is okay beause the original variable is non-const.
213  mRecord->SetClientData(devindex, const_cast<DeviceSourceMap *>(&inMaps[i]));
214  if (device == recDevice) { /* if this is the default device, select it */
215  mRecord->SetSelection(devindex);
216  }
217  }
218  }
219 
220  mPlay->Clear();
221  for (i = 0; i < outMaps.size(); i++) {
222  if (index == outMaps[i].hostIndex) {
223  device = MakeDeviceSourceString(&outMaps[i]);
224  devindex = mPlay->Append(device);
225  mPlay->SetClientData(devindex, const_cast<DeviceSourceMap *>(&outMaps[i]));
226  if (device == mPlayDevice) { /* if this is the default device, select it */
227  mPlay->SetSelection(devindex);
228  }
229  }
230  }
231 
232  /* deal with not having any devices at all */
233  if (mPlay->GetCount() == 0) {
234  playnames.Add(_("No devices found"));
235  mPlay->Append(playnames[0], (void *) NULL);
236  mPlay->SetSelection(0);
237  }
238  if (mRecord->GetCount() == 0) {
239  recordnames.Add(_("No devices found"));
240  mRecord->Append(recordnames[0], (void *) NULL);
241  mRecord->SetSelection(0);
242  }
243 
244  /* what if we have no device selected? we should choose the default on
245  * this API, as defined by PortAudio. We then fall back to using 0 only if
246  * that fails */
247  if (mPlay->GetCount() && mPlay->GetSelection() == wxNOT_FOUND) {
248  DeviceSourceMap *defaultMap = DeviceManager::Instance()->GetDefaultOutputDevice(index);
249  if (defaultMap)
250  mPlay->SetStringSelection(MakeDeviceSourceString(defaultMap));
251 
252  if (mPlay->GetSelection() == wxNOT_FOUND) {
253  mPlay->SetSelection(0);
254  }
255  }
256 
257  if (mRecord->GetCount() && mRecord->GetSelection() == wxNOT_FOUND) {
258  DeviceSourceMap *defaultMap = DeviceManager::Instance()->GetDefaultInputDevice(index);
259  if (defaultMap)
260  mRecord->SetStringSelection(MakeDeviceSourceString(defaultMap));
261 
262  if (mPlay->GetSelection() == wxNOT_FOUND) {
263  mPlay->SetSelection(0);
264  }
265  }
266 
267  ShuttleGui S(this, eIsCreating);
268  S.SetSizeHints(mPlay, mPlay->GetStrings());
269  S.SetSizeHints(mRecord, mRecord->GetStrings());
270  OnDevice(e);
271 }

273 void DevicePrefs::OnDevice(wxCommandEvent & WXUNUSED(event))
274 {
275  int ndx = mRecord->GetCurrentSelection();
276  if (ndx == wxNOT_FOUND) {
277  ndx = 0;
278  }
279 
280  int sel = mChannels->GetSelection();
281  int cnt = 0;
282 
283  DeviceSourceMap *inMap = (DeviceSourceMap *) mRecord->GetClientData(ndx);
284  if (inMap != NULL) {
285  cnt = inMap->numChannels;
286  }
287 
288  if (sel != wxNOT_FOUND) {
289  mRecordChannels = sel + 1;
290  }
291 
292  mChannels->Clear();
293 
294  // Mimic old behavior
295  if (cnt <= 0) {
296  cnt = 16;
297  }
298 
299  // Place an artifical limit on the number of channels to prevent an
300  // outrageous number. I don't know if this is really necessary, but
301  // it doesn't hurt.
302  if (cnt > 256) {
303  cnt = 256;
304  }
305 
306  wxArrayString channelnames;
307 
308  // Channel counts, mono, stereo etc...
309  for (int i = 0; i < cnt; i++) {
310  wxString name;
311 
312  if (i == 0) {
313  name = _("1 (Mono)");
314  }
315  else if (i == 1) {
316  name = _("2 (Stereo)");
317  }
318  else {
319  name = wxString::Format(wxT("%d"), i + 1);
320  }
321 
322  channelnames.Add(name);
323  int index = mChannels->Append(name);
324  if (i == mRecordChannels - 1) {
325  mChannels->SetSelection(index);
326  }
327  }
328 
329  if (mChannels->GetCount() && mChannels->GetCurrentSelection() == wxNOT_FOUND) {
330  mChannels->SetSelection(0);
331  }
332 
333  ShuttleGui S(this, eIsCreating);
334  S.SetSizeHints(mChannels, channelnames);
335  Layout();
336 }

338 bool DevicePrefs::Apply()
339 {
340  ShuttleGui S(this, eIsSavingToPrefs);
341  PopulateOrExchange(S);
342  DeviceSourceMap *map = NULL;
343 
344  if (mPlay->GetCount() > 0) {
345  map = (DeviceSourceMap *) mPlay->GetClientData(
346  mPlay->GetSelection());
347  }
348  if (map) {
349  gPrefs->Write(wxT("/AudioIO/PlaybackDevice"), map->deviceString);
350  }
351 
352  map = NULL;
353  if (mRecord->GetCount() > 0) {
354  map = (DeviceSourceMap *) mRecord->GetClientData(mRecord->GetSelection());
355  }
356  if (map) {
357  gPrefs->Write(wxT("/AudioIO/RecordingDevice"),
358  map->deviceString);
359  gPrefs->Write(wxT("/AudioIO/RecordingSourceIndex"),
360  map->sourceIndex);
361  if (map->totalSources >= 1) {
362  gPrefs->Write(wxT("/AudioIO/RecordingSource"),
363  map->sourceString);
364  } else {
365  gPrefs->Write(wxT("/AudioIO/RecordingSource"),
366  wxT(""));
367  }
368  gPrefs->Write(wxT("/AudioIO/RecordChannels"),
369  mChannels->GetSelection() + 1);
370  }
371 
372  return true;
373 }
