31 //
32 // AudacityLogger class
33 //
34 // Two reasons for this class instead of the wxLogWindow class (or any WX GUI logging class)
35 //
36 // 1) If wxLogWindow is used and initialized before the Mac's "root" window, then
37 // Audacity may crash when terminating. It's not fully understood why this occurs
38 // but it probably has to do with the order of deletion. However, deferring the
39 // creation of the log window until it is actually shown circumvents the problem.
40 // 2) By providing an Audacity specific logging class, it can be made thread-safe and,
41 // as such, can be used by the ever growing threading within Audacity.
42 //
43 enum
44 {
45  LoggerID_Save = wxID_HIGHEST + 1,
46  LoggerID_Clear,
47  LoggerID_Close
48 };
