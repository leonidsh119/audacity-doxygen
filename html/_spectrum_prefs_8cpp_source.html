115 void SpectrumPrefs::PopulatePaddingChoices(int windowSize)
116 {
117 #ifdef EXPERIMENTAL_ZERO_PADDED_SPECTROGRAMS
118  mZeroPaddingChoice = 1;
119 
120  // The choice of window size restricts the choice of padding.
121  // So the padding menu might grow or shrink.
122 
123  // If pPaddingSizeControl is NULL, we have not yet tied the choice control.
124  // If it is not NULL, we rebuild the control by hand.
125  // I don't yet know an easier way to do this with ShuttleGUI functions.
126  // PRL
127  wxChoice *const pPaddingSizeControl =
128  static_cast<wxChoice*>(wxWindow::FindWindowById(ID_PADDING_SIZE, this));
129 
130  if (pPaddingSizeControl) {
131  mZeroPaddingChoice = pPaddingSizeControl->GetSelection();
132  pPaddingSizeControl->Clear();
133  }
134 
135  int padding = 1;
136  int numChoices = 0;
137  const int maxWindowSize = 1 << (SpectrogramSettings::LogMaxWindowSize);
138  while (windowSize <= maxWindowSize) {
139  const wxString numeral = wxString::Format(wxT("%d"), padding);
140  mZeroPaddingChoices.Add(numeral);
141  if (pPaddingSizeControl)
142  pPaddingSizeControl->Append(numeral);
143  windowSize <<= 1;
144  padding <<= 1;
145  ++numChoices;
146  }
147 
148  mZeroPaddingChoice = std::min(mZeroPaddingChoice, numChoices - 1);
149 
150  if (pPaddingSizeControl)
151  pPaddingSizeControl->SetSelection(mZeroPaddingChoice);
152 #endif
153 }

379

382 
383  if (mWt) {
384  if (mDefaulted) {
385  mWt->SetSpectrogramSettings(NULL);
386  // ... and so that the vertical scale also defaults:
387  mWt->SetSpectrumBounds(-1, -1);
388  if (partner) {
389  partner->SetSpectrogramSettings(NULL);
390  partner->SetSpectrumBounds(-1, -1);
391  }
392  }
393  else {
394  SpectrogramSettings *pSettings =
395  &mWt->GetIndependentSpectrogramSettings();
396  mWt->SetSpectrumBounds(mTempSettings.minFreq, mTempSettings.maxFreq);
397  *pSettings = mTempSettings;
398  if (partner) {
399  pSettings = &partner->GetIndependentSpectrogramSettings();
400  partner->SetSpectrumBounds(mTempSettings.minFreq, mTempSettings.maxFreq);
401  *pSettings = mTempSettings;
402  }
403  }
404  }
405 
406  if (!mWt || mDefaulted) {
407  SpectrogramSettings *const pSettings = &SpectrogramSettings::defaults();
408  *pSettings = mTempSettings;
409  pSettings->SavePrefs();
410  }
411  mTempSettings.ConvertToEnumeratedWindowSizes();
412 
413  if (mWt && isOpenPage) {
414  mWt->SetDisplay(WaveTrack::Spectrum);
415  if (partner)
416  partner->SetDisplay(WaveTrack::Spectrum);
417  }
418 
419  if (isOpenPage) {
420  TrackPanel *const tp = ::GetActiveProject()->GetTrackPanel();
421  tp->UpdateVRulers();
422  tp->Refresh(false);
423  }
424
