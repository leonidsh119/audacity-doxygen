115 /* This struct contains all the fields of interest om the DWD header, but does not
116 ** do so in the same order and layout as the actual file, header.
117 ** No assumptions are made about the packing of this struct.
118 */

125 static int
126 dwd_read_header (SF_PRIVATE *psf)
127 { DWD_HEADER dwdh ;
128 
129  memset (psf->u.cbuf, 0, sizeof (psf->u.cbuf)) ;
130  /* Set position to start of file to begin reading header. */
131  psf_binheader_readf (psf, "pb", 0, psf->u.cbuf, DWD_IDENTIFIER_LEN) ;
132 
133  if (memcmp (psf->u.cbuf, DWD_IDENTIFIER, DWD_IDENTIFIER_LEN) != 0)
134  return SFE_DWD_NO_DWD ;
135 
136  psf_log_printf (psf, "Read only : DiamondWare Digitized (.dwd)\n", psf->u.cbuf) ;
137 
138  psf_binheader_readf (psf, "11", &dwdh.major, &dwdh.minor) ;
139  psf_binheader_readf (psf, "e4j1", &dwdh.id, 1, &dwdh.compression) ;
140  psf_binheader_readf (psf, "e211", &dwdh.srate, &dwdh.channels, &dwdh.bitwidth) ;
141  psf_binheader_readf (psf, "e24", &dwdh.maxval, &dwdh.datalen) ;
142  psf_binheader_readf (psf, "e44", &dwdh.frames, &dwdh.offset) ;
143 
144  psf_log_printf (psf, " Version Major : %d\n Version Minor : %d\n Unique ID : %08X\n",
145  dwdh.major, dwdh.minor, dwdh.id) ;
146  psf_log_printf (psf, " Compression : %d => ", dwdh.compression) ;
147 
148  if (dwdh.compression != 0)
149  { psf_log_printf (psf, "Unsupported compression\n") ;
150  return SFE_DWD_COMPRESSION ;
151  }
152  else
153  psf_log_printf (psf, "None\n") ;
154 
155  psf_log_printf (psf, " Sample Rate : %d\n Channels : %d\n"
156  " Bit Width : %d\n",
157  dwdh.srate, dwdh.channels, dwdh.bitwidth) ;
158 
159  switch (dwdh.bitwidth)
160  { case 8 :
161  psf->sf.format = SF_FORMAT_DWD | SF_FORMAT_PCM_S8 ;
162  psf->bytewidth = 1 ;
163  break ;
164 
165  case 16 :
166  psf->sf.format = SF_FORMAT_DWD | SF_FORMAT_PCM_16 ;
167  psf->bytewidth = 2 ;
168  break ;
169 
170  default :
171  psf_log_printf (psf, "*** Bad bit width %d\n", dwdh.bitwidth) ;
172  return SFE_DWD_BAND_BIT_WIDTH ;
173  } ;
174 
175  if (psf->filelength != dwdh.offset + dwdh.datalen)
176  { psf_log_printf (psf, " Data Length : %d (should be %D)\n", dwdh.datalen, psf->filelength - dwdh.offset) ;
177  dwdh.datalen = (unsigned int) (psf->filelength - dwdh.offset) ;
178  }
179  else
180  psf_log_printf (psf, " Data Length : %d\n", dwdh.datalen) ;
181 
182  psf_log_printf (psf, " Max Value : %d\n", dwdh.maxval) ;
183  psf_log_printf (psf, " Frames : %d\n", dwdh.frames) ;
184  psf_log_printf (psf, " Data Offset : %d\n", dwdh.offset) ;
185 
186  psf->datalength = dwdh.datalen ;
187  psf->dataoffset = dwdh.offset ;
188 
189  psf->endian = SF_ENDIAN_LITTLE ;
190 
191  psf->sf.samplerate = dwdh.srate ;
192  psf->sf.channels = dwdh.channels ;
193  psf->sf.sections = 1 ;
194 
195  return pcm_init (psf) ;
196 } /* dwd_read_header */
