106 bool BatchCommands::ReadChain(const wxString & chain)
107 {
108  // Clear any previous chain
109  ResetChain();
110 
111  // Build the filename
112  wxFileName name(FileNames::ChainDir(), chain, wxT("txt"));
113 
114  // Set the file name
115  wxTextFile tf(name.GetFullPath());
116 
117  // Open and check
118  tf.Open();
119  if (!tf.IsOpened()) {
120  // wxTextFile will display any errors
121  return false;
122  }
123 
124  // Load commands from the file
125  int lines = tf.GetLineCount();
126  if (lines > 0) {
127  for (int i = 0; i < lines; i++) {
128 
129  // Find the command name terminator...ingore line if not found
130  int splitAt = tf[i].Find(wxT(':'));
131  if (splitAt < 0) {
132  continue;
133  }
134 
135  // Parse and clean
136  wxString cmd = tf[i].Left(splitAt).Strip(wxString::both);
137  wxString parm = tf[i].Mid(splitAt + 1).Strip(wxString::trailing);
138 
139  // Backward compatibility for old Chain scripts
140  // Please comment the version of audacity these are introduced in, so
141  // that old ones can easily be removed once users have had a chance to
142  // migrate
143  if (cmd == wxT("SaveMP3_56k_before"))
144  cmd = wxT("ExportMP3_56k_before");
145  else if (cmd == wxT("SaveMP3_56k_after"))
146  cmd = wxT("ExportMP3_56k_after");
147  else if (cmd == wxT("ExportFlac"))
148  cmd = wxT("ExportFLAC");
149  else if (cmd == wxT("ExportMp3"))
150  cmd = wxT("ExportMP3");
151  else if (cmd == wxT("ExportWav"))
152  cmd = wxT("ExportWAV");
153  else if (cmd == wxT("Compressor") && (parm.find(wxT("DecayTime")) != parm.npos))
154  parm.Replace(wxT("DecayTime"), wxT("ReleaseTime"), NULL); // 2.0.6
155 
156  // Add to lists
157  mCommandChain.Add(cmd);
158  mParamsChain.Add(parm);
159  }
160  }
161 
162  // Done with the file
163  tf.Close();
164 
165  return true;
166 }
