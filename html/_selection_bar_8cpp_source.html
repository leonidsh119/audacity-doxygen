414 void SelectionBar::ValuesToControls()
415 {
416  mLeftTime->SetValue(mStart);
417 
418  if (mRightEndButton->GetValue())
419  mRightTime->SetValue(mEnd);
420  else
421  { // mRightTime is the length.
422  // Be sure to take into account the sub-sample offset.
423  // See TimeToLongSamples and LongSamplesToTime but here at the project rate.
424  double t = (sampleCount)floor(mEnd * mRate + 0.5);
425  t -= (sampleCount)floor(mStart * mRate + 0.5);
426  t /= mRate;
427  mRightTime->SetValue(t);
428  }
429 
430  mAudioTime->SetValue(mAudio);
431 }
