97 bool EffectRepeat::Process()
98 {
99  // Set up mOutputTracks.
100  // This effect needs Track::All for sync-lock grouping.
101  CopyInputTracks(Track::All);
102 
103  int nTrack = 0;
104  bool bGoodResult = true;
105  double maxDestLen = 0.0; // used to change selection to generated bit
106 
107  TrackListIterator iter(mOutputTracks);
108 
109  for (Track *t = iter.First(); t && bGoodResult; t = iter.Next())
110  {
111  if (t->GetKind() == Track::Label)
112  {
113  if (t->GetSelected() || t->IsSyncLockSelected())
114  {
115  LabelTrack* track = (LabelTrack*)t;
116 
117  if (!track->Repeat(mT0, mT1, repeatCount))
118  {
119  bGoodResult = false;
120  break;
121  }
122  }
123  }
124  else if (t->GetKind() == Track::Wave && t->GetSelected())
125  {
126  WaveTrack* track = (WaveTrack*)t;
127 
128  sampleCount start = track->TimeToLongSamples(mT0);
129  sampleCount end = track->TimeToLongSamples(mT1);
130  sampleCount len = (sampleCount)(end - start);
131  double tLen = track->LongSamplesToTime(len);
132  double tc = mT0 + tLen;
133 
134  if (len <= 0)
135  {
136  continue;
137  }
138 
139  Track *dest;
140  track->Copy(mT0, mT1, &dest);
141  for(int j=0; j<repeatCount; j++)
142  {
143  if (!track->Paste(tc, dest) ||
144  TrackProgress(nTrack, j / repeatCount)) // TrackProgress returns true on Cancel.
145  {
146  bGoodResult = false;
147  break;
148  }
149  tc += tLen;
150  }
151  if (tc > maxDestLen)
152  maxDestLen = tc;
153  delete dest;
154  nTrack++;
155  }
156  else if (t->IsSyncLockSelected())
157  {
158  t->SyncLockAdjust(mT1, mT1 + (mT1 - mT0) * repeatCount);
159  }
160  }
161 
162  if (bGoodResult)
163  {
164  // Select the NEW bits + original bit
165  mT1 = maxDestLen;
166  }
167 
168  ReplaceProcessedTracks(bGoodResult);
169  return bGoodResult;
170 }
