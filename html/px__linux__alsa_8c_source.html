86 static int open_mixer(PxDev *dev, int card, int playback)
87 {
88  do {
89  snd_mixer_selem_id_t *sid;
90  snd_mixer_elem_t *elem;
91  char name[256];
92  int err;
93  int i;
94 
95  snd_mixer_selem_id_alloca(&sid);
96 
97  sprintf(name, "hw:%d", card);
98 
99  dev->card = card;
100  dev->handle = NULL;
101  dev->playback = playback;
102  dev->source = -1;
103 
104  err = snd_mixer_open(&dev->handle, 0);
105  if (err < 0) {
106  break;
107  }
108 
109  err = snd_mixer_attach(dev->handle, name);
110  if (err < 0) {
111  break;
112  }
113 
114  err = snd_mixer_selem_register(dev->handle, NULL, NULL);
115  if (err < 0) {
116  break;
117  }
118 
119  err = snd_mixer_load(dev->handle);
120  if (err < 0) {
121  break;
122  }
123 
124  for (elem = snd_mixer_first_elem(dev->handle);
125  elem != NULL;
126  elem = snd_mixer_elem_next(elem))
127  {
128  if (playback) {
129  if (snd_mixer_selem_has_common_volume(elem) ||
130  snd_mixer_selem_has_playback_volume(elem)) {
131  dev->numselems++;
132  }
133  }
134  else {
135  if (snd_mixer_selem_get_capture_group(elem) >= 0) {
136  dev->numselems++;
137  }
138  else if (snd_mixer_selem_is_enum_capture(elem)) {
139  int retval = snd_mixer_selem_get_enum_items(elem);
140 
141  if (retval > 0)
142  dev->numselems += retval;
143  }
144  }
145  }
146 
147  dev->selems = calloc(dev->numselems, sizeof(PxSelem));
148  if (dev->selems == NULL) {
149  break;
150  }
151 
152 
153  i = 0;
154  for (elem = snd_mixer_first_elem(dev->handle);
155  elem != NULL;
156  elem = snd_mixer_elem_next(elem))
157  {
158  snd_mixer_elem_t *vol;
159  int ndx;
160 
161  if (playback) {
162  if (snd_mixer_selem_has_common_volume(elem) ||
163  snd_mixer_selem_has_playback_volume(elem)) {
164  snprintf(name,
165  sizeof(name),
166  "%s:%d",
167  snd_mixer_selem_get_name(elem),
168  snd_mixer_selem_get_index(elem));
169 
170  dev->selems[i].elem = elem;
171  dev->selems[i].index = snd_mixer_selem_get_index(elem);
172  dev->selems[i].name = strdup(name);
173  if (!dev->selems[i].name) {
174  break;
175  }
176  i++;
177  }
178  continue;
179  }
180 
181  snd_mixer_selem_id_set_name(sid, "Capture");
182  snd_mixer_selem_id_set_index(sid, snd_mixer_selem_get_index(elem));
183  vol = snd_mixer_find_selem(dev->handle, sid);
184 
185  if (snd_mixer_selem_get_capture_group(elem) >= 0) {
186  snprintf(name,
187  sizeof(name),
188  "%s:%d",
189  snd_mixer_selem_get_name(elem),
190  snd_mixer_selem_get_index(elem));
191 
192  dev->selems[i].vol = vol;
193  dev->selems[i].elem = elem;
194  dev->selems[i].index = snd_mixer_selem_get_index(elem);
195  dev->selems[i].name = strdup(name);
196  if (!dev->selems[i].name) {
197  break;
198  }
199  i++;
200  }
201  else if (snd_mixer_selem_is_enum_capture(elem)) {
202  int j;
203  int cnt = snd_mixer_selem_get_enum_items(elem);
204 
205  if (cnt < 0)
206  continue;
207 
208  for (j = 0; j < cnt; j++) {
209  char iname[256];
210  snd_mixer_selem_get_enum_item_name(elem, (unsigned int) j, sizeof(iname), iname);
211  snprintf(name,
212  sizeof(name),
213  "%s:%d",
214  iname,
215  snd_mixer_selem_get_index(elem));
216 
217  dev->selems[i].vol = vol;
218  dev->selems[i].item = j;
219  dev->selems[i].elem = elem;
220  dev->selems[i].index = snd_mixer_selem_get_index(elem);
221  dev->selems[i].name = strdup(name);
222  if (!dev->selems[i].name) {
223  break;
224  }
225  i++;
226  }
227  }
228  }
229 
230  if (i != dev->numselems) {
231  break;
232  }
233 
234  if (playback) {
235  return TRUE;
236  }
237 
238  for (i = 0; i < dev->numselems; i++) {
239  elem = dev->selems[i].elem;
240  if (snd_mixer_selem_get_capture_group(elem) >= 0) {
241  int sw = 0;
242  if (snd_mixer_selem_get_capture_switch(elem, SND_MIXER_SCHN_FRONT_LEFT, &sw) < 0) {
243  continue;
244  }
245  if (!sw) {
246  continue;
247  }
248  dev->source = i;
249  break;
250  }
251  else if (snd_mixer_selem_is_enum_capture(elem)) {
252  unsigned int src;
253  if (snd_mixer_selem_get_enum_item(elem, SND_MIXER_SCHN_FRONT_LEFT, &src) < 0) {
254  continue;
255  }
256  if (src == dev->selems[i].item) {
257  dev->source = i;
258  break;
259  }
260  }
261  }
262 
263  if (dev->source == -1) {
264  dev->source = 0;
265  }
266 
267  return TRUE;
268 
269  } while (FALSE);
270 
271  if (dev->selems) {
272  int i;
273  for (i = 0; i < dev->numselems; i++) {
274  if (dev->selems[i].name) {
275  free(dev->selems[i].name);
276  }
277  }
278  free(dev->selems);
279  dev->selems = NULL;
280  }
281 
282  if (dev->handle) {
283  snd_mixer_close(dev->handle);
284  dev->handle = NULL;
285  }
286 
287  return FALSE;
288 }

410 static PxVolume get_volume_indexed(PxDev *dev, int i)
411 {
412  snd_mixer_elem_t *elem;
413  long vol, min, max;
414 
415  if (!dev->handle) {
416  return 0.0;
417  }
418 
419  if (i < 0 || i > dev->numselems) {
420  return 0.0;
421  }
422 
423  elem = dev->selems[i].elem;
424  if (dev->playback) {
425  snd_mixer_selem_get_playback_volume_range(elem, &min, &max);
426  if (snd_mixer_selem_has_playback_channel(elem, SND_MIXER_SCHN_FRONT_LEFT)) {
427  snd_mixer_selem_get_playback_volume(elem, SND_MIXER_SCHN_FRONT_LEFT, &vol);
428  return (PxVolume) vol / (max - min);
429  }
430  }
431  else {
432  snd_mixer_selem_get_capture_volume_range(elem, &min, &max);
433  if (snd_mixer_selem_has_capture_channel(elem, SND_MIXER_SCHN_FRONT_LEFT)) {
434  snd_mixer_selem_get_capture_volume(elem, SND_MIXER_SCHN_FRONT_LEFT, &vol);
435  return (PxVolume) vol / (max - min);
436  }
437  }
438 
439  return 0.0;

460 {

462  long vol, min, max;
463  int j;
464 
465  if (!dev->handle) {
466  return;
467  }
468 
469  if (i < 0 || i > dev->numselems) {

472
