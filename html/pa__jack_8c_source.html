574 static PaError BlockingWriteStream( PaStream* s, const void *data, unsigned long numFrames )
575 {
576  PaError result = paNoError;
577  PaJackStream *stream = (PaJackStream *)s;
578  long bytesWritten;
579  char *p = (char *) data;
580  long numBytes = stream->bytesPerFrame * numFrames;
581  while( numBytes > 0 )
582  {
583  bytesWritten = PaUtil_WriteRingBuffer( &stream->outFIFO, p, numBytes );
584  numBytes -= bytesWritten;
585  p += bytesWritten;
586  if( numBytes > 0 )
587  {
588  /* we use the following algorithm:
589  * (1) write data
590  * (2) if some data didn't fit into the ringbuffer, set data_available to 0
591  * to indicate to the audio that if space becomes available, we want to know
592  * (3) retry to write data (because it might be that between (1) and (2)
593  * new space in the buffer became available)
594  * (4) if this failed, we are sure that the buffer is really empty and
595  * we will definitely receive a notification when it becomes available
596  * thus we can safely sleep
597  *
598  * if the algorithm bailed out in step (3) before, it leaks a count of 1
599  * on the semaphore; however, it doesn't matter, because if we block in (4),
600  * we also do it in a loop
601  */
602  if( stream->data_available )
603  stream->data_available = 0;
604  else
605  PaJack_WaitForStreamDataToBecomeAvailable( stream );
606  }
607  }
608 
609  return result;
610 }

652 static PaError BuildDeviceList( PaJackHostApiRepresentation *jackApi )
653 {
654  /* Utility macros for the repetitive process of allocating memory */
655 
656  /* JACK has no concept of a device. To JACK, there are clients
657  * which have an arbitrary number of ports. To make this
658  * intelligible to PortAudio clients, we will group each JACK client
659  * into a device, and make each port of that client a channel */
660 
661  PaError result = paNoError;
662  PaUtilHostApiRepresentation *commonApi = &jackApi->commonHostApiRep;
663 
664  const char **jack_ports = NULL;
665  char **client_names = NULL;
666  char *regex_pattern = NULL;
667  int port_index, client_index, i;
668  double globalSampleRate;
669  unsigned long numClients = 0, numPorts = 0;
670  char *tmp_client_name = NULL;
671 
672  commonApi->info.defaultInputDevice = paNoDevice;
673  commonApi->info.defaultOutputDevice = paNoDevice;
674  commonApi->info.deviceCount = 0;
675 
676  /* since we are rebuilding the list of devices, free all memory
677  * associated with the previous list */
678  PaUtil_FreeAllAllocations( jackApi->deviceInfoMemory );
679 
680  regex_pattern = PaUtil_GroupAllocateMemory( jackApi->deviceInfoMemory, jack_client_name_size() + 3 );
681  tmp_client_name = PaUtil_GroupAllocateMemory( jackApi->deviceInfoMemory, jack_client_name_size() );
682 
683  /* We can only retrieve the list of clients indirectly, by first
684  * asking for a list of all ports, then parsing the port names
685  * according to the client_name:port_name convention (which is
686  * enforced by jackd)
687  * A: If jack_get_ports returns NULL, there's nothing for us to do */
688  UNLESS( (jack_ports = jack_get_ports( jackApi->jack_client, "^[^:]*", JACK_PORT_TYPE_FILTER, 0 )) && jack_ports[0], paNoError );
689  /* Find number of ports */
690  while( jack_ports[numPorts] )
691  ++numPorts;
692  /* At least there will be one port per client :) */
693  UNLESS( client_names = PaUtil_GroupAllocateMemory( jackApi->deviceInfoMemory, numPorts *
694  sizeof (char *) ), paInsufficientMemory );
695 
696  /* Build a list of clients from the list of ports */
697  for( numClients = 0, port_index = 0; jack_ports[port_index] != NULL; port_index++ )
698  {
699  int client_seen = FALSE;
700  const char *port = jack_ports[port_index];
701  const char *colon;
702 
703  /* extract the client name from the port name, using a regex
704  * that parses the clientname:portname syntax */
705  UNLESS( colon = strchr(port, ':'), paInternalError );
706  assert(colon - port < jack_client_name_size());
707  memcpy( tmp_client_name, port, colon - port );
708  tmp_client_name[colon - port] = '\0';
709 
710  /* do we know about this port's client yet? */
711  for( i = 0; i < numClients; i++ )
712  {
713  if( strcmp( tmp_client_name, client_names[i] ) == 0 )
714  client_seen = TRUE;
715  }
716 
717  if (client_seen)
718  continue; /* A: Nothing to see here, move along */
719 
720  UNLESS( client_names[numClients] = (char*)PaUtil_GroupAllocateMemory( jackApi->deviceInfoMemory,
721  strlen(tmp_client_name) + 1), paInsufficientMemory );
722 
723  /* The alsa_pcm client should go in spot 0. If this
724  * is the alsa_pcm client AND we are NOT about to put
725  * it in spot 0 put it in spot 0 and move whatever
726  * was already in spot 0 to the end. */
727  if( strcmp( "alsa_pcm", tmp_client_name ) == 0 && numClients > 0 )
728  {
729  /* alsa_pcm goes in spot 0 */
730  strcpy( client_names[ numClients ], client_names[0] );
731  strcpy( client_names[0], tmp_client_name );
732  }
733  else
734  {
735  /* put the new client at the end of the client list */
736  strcpy( client_names[ numClients ], tmp_client_name );
737  }
738  ++numClients;
739  }
740 
741  /* Now we have a list of clients, which will become the list of
742  * PortAudio devices. */
743 
744  /* there is one global sample rate all clients must conform to */
745 
746  globalSampleRate = jack_get_sample_rate( jackApi->jack_client );
747  UNLESS( commonApi->deviceInfos = (PaDeviceInfo**)PaUtil_GroupAllocateMemory( jackApi->deviceInfoMemory,
748  sizeof(PaDeviceInfo*) * numClients ), paInsufficientMemory );
749 
750  assert( commonApi->info.deviceCount == 0 );
751 
752  /* Create a PaDeviceInfo structure for every client */
753  for( client_index = 0; client_index < numClients; client_index++ )
754  {
755  PaDeviceInfo *curDevInfo;
756  const char **clientPorts = NULL;
757 
758  UNLESS( curDevInfo = (PaDeviceInfo*)PaUtil_GroupAllocateMemory( jackApi->deviceInfoMemory,
759  sizeof(PaDeviceInfo) ), paInsufficientMemory );
760  UNLESS( curDevInfo->name = (char*)PaUtil_GroupAllocateMemory( jackApi->deviceInfoMemory,
761  strlen(client_names[client_index]) + 1 ), paInsufficientMemory );
762  strcpy( (char *)curDevInfo->name, client_names[client_index] );
763 
764  curDevInfo->structVersion = 2;
765  curDevInfo->hostApi = jackApi->hostApiIndex;
766 
767  /* JACK is very inflexible: there is one sample rate the whole
768  * system must run at, and all clients must speak IEEE float. */
769  curDevInfo->defaultSampleRate = globalSampleRate;
770 
771  /* To determine how many input and output channels are available,
772  * we re-query jackd with more specific parameters. */
773 
774  sprintf( regex_pattern, "%s:.*", client_names[client_index] );
775 
776  /* ... what are your output ports (that we could input from)? */
777  clientPorts = jack_get_ports( jackApi->jack_client, regex_pattern,
778  JACK_PORT_TYPE_FILTER, JackPortIsOutput);
779  curDevInfo->maxInputChannels = 0;
780  curDevInfo->defaultLowInputLatency = 0.;
781  curDevInfo->defaultHighInputLatency = 0.;
782  if( clientPorts )
783  {
784  jack_port_t *p = jack_port_by_name( jackApi->jack_client, clientPorts[0] );
785  curDevInfo->defaultLowInputLatency = curDevInfo->defaultHighInputLatency =
786  jack_port_get_latency( p ) / globalSampleRate;
787 
788  for( i = 0; clientPorts[i] != NULL; i++)
789  {
790  /* The number of ports returned is the number of output channels.
791  * We don't care what they are, we just care how many */
792  curDevInfo->maxInputChannels++;
793  }
794  jack_free(clientPorts);
795  }
796 
797  /* ... what are your input ports (that we could output to)? */
798  clientPorts = jack_get_ports( jackApi->jack_client, regex_pattern,
799  JACK_PORT_TYPE_FILTER, JackPortIsInput);
800  curDevInfo->maxOutputChannels = 0;
801  curDevInfo->defaultLowOutputLatency = 0.;
802  curDevInfo->defaultHighOutputLatency = 0.;
803  if( clientPorts )
804  {
805  jack_port_t *p = jack_port_by_name( jackApi->jack_client, clientPorts[0] );
806  curDevInfo->defaultLowOutputLatency = curDevInfo->defaultHighOutputLatency =
807  jack_port_get_latency( p ) / globalSampleRate;
808 
809  for( i = 0; clientPorts[i] != NULL; i++)
810  {
811  /* The number of ports returned is the number of input channels.
812  * We don't care what they are, we just care how many */
813  curDevInfo->maxOutputChannels++;
814  }
815  jack_free(clientPorts);
816  }
817 
818  /* Add this client to the list of devices */
819  commonApi->deviceInfos[client_index] = curDevInfo;
820  ++commonApi->info.deviceCount;
821  if( commonApi->info.defaultInputDevice == paNoDevice && curDevInfo->maxInputChannels > 0 )
822  commonApi->info.defaultInputDevice = client_index;
823  if( commonApi->info.defaultOutputDevice == paNoDevice && curDevInfo->maxOutputChannels > 0 )
824  commonApi->info.defaultOutputDevice = client_index;
825  }
826 
827 error:
828  jack_free( jack_ports );
829  return result;
830 }
