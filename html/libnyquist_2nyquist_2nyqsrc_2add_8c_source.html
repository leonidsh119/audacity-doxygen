8 /* DOCUMENTATION:
9  Most DSP modules in Nyquist select a single fetch routine
10 and use it until the signal terminates. The ADD operation
11 instead can use a number of different fetch routines in sequence.
12 This allows ADD to do the most efficient computation, such as
13 simply copying pointers when only one input signal is defined
14 (the other is zero.)
15  Here's what the functions assume and do:
16 add_s1_s2_nn_fetch: both arguments (s1, s2) have signals; add
17  them.
18 add_s1_nn_fetch: only s1 is active, so pass along pointers if
19  possible. Revert to add_s1_s2_nn_fetch when s2 becomes active.
20 add_s2_nn_fetch: symetric with add_s1_nn_fetch.
21 add_zero_fill_nn_fetch: fill in when one input has terminated and
22  the other hasn't begun.
23 
24 An important optimization (we think) is the ability to collapse
25 ADD operations. When one operand goes to zero, the ADD just
26 passes along pointers to blocks from the other operand. In some
27 cases, we can just splice out the ADD suspension and link
28 directly to the suspension of the second operand.
29 
30 Doing this requires that there be no scale factors, so ADD does
31 not deal with scaling. If an operand comes in with a scale
32 factor, ADD will create a rescaling of the operand.
33 */

823 sound_type snd_make_add(sound_type s1, sound_type s2)
824 {
825  register add_susp_type susp;
826  rate_type sr = MAX(s1->sr, s2->sr);
827  time_type t0 = MIN(s1->t0, s2->t0);
828  int interp_desc = 0;
829  double sample_offset;
830 
831  /* sort commutative signals: (S1 S2) */
832  snd_sort_2(&s1, &s2, sr);
833 
834  falloc_generic(susp, add_susp_node, "snd_make_add");
835 
836  /* select a susp fn based on sample rates */
837  interp_desc = (interp_style(s1, sr) << 2) + interp_style(s2, sr);
838  switch (interp_desc) {
839  case INTERP_nn:
840  case INTERP_ns:
841  case INTERP_ss:
842  /* eliminate scale factor on s1 if any */
843  if (((interp_desc >> INTERP_SHIFT) & INTERP_MASK) == INTERP_s) {
844  /* stdputstr("add: prescaling s1\n");*/
845  s1 = snd_make_normalize(s1);
846  }
847  /* eliminate scale factor on s2 if any */
848  if ((interp_desc & INTERP_MASK) == INTERP_s) {
849  /* stdputstr("add: prescaling s2\n"); */
850  s2 = snd_make_normalize(s2);
851  }
852  sample_offset = (s2->t0 - s1->t0) * sr;
853  if (sample_offset >= 0.5) { /* s1 starts first */
854  susp->susp.fetch = add_s1_nn_fetch;
855 D stdputstr("snd_make_add: add_s1_nn_fetch installed\n");
856  } else if (sample_offset < -0.5) { /* s2 starts first */
857  susp->susp.fetch = add_s2_nn_fetch;
858 D stdputstr("snd_make_add: add_s2_nn_fetch installed\n");
859  } else { /* equal start times */
860  susp->susp.fetch = add_s1_s2_nn_fetch;
861 D stdputstr("snd_make_add: add_s1_s2_nn_fetch installed\n");
862  }
863  break;
864  case INTERP_ni:
865  case INTERP_nr:
866  errputstr("add: can't interpolate!\n");
867  EXIT(1);
868  default:
869  errputstr("add: can't add these operands!\n");
870  EXIT(1);
871  }
872 
873  susp->terminate_cnt = UNKNOWN;
874  susp->terminate_bits = 0; /* bits for s1 and s2 termination */
875  susp->logical_stop_bits = 0; /* bits for s1 and s2 logical stop */
876 
877  /* initialize susp state */
878  susp->susp.free = add_free;
879  susp->susp.sr = sr;
880  susp->susp.t0 = t0;
881  susp->susp.mark = add_mark;
882  susp->susp.print_tree = add_print_tree;
883  susp->susp.name = "add";
884  susp->logically_stopped = false;
885  susp->susp.log_stop_cnt = UNKNOWN;
886  susp->started = false;
887  susp->susp.current = 0;
888  susp->s1 = s1;
889  susp->s1_cnt = 0;
890  susp->s2 = s2;
891  susp->s2_cnt = 0;
892 #ifdef UPSAMPLECODE
893  susp->susp.s2_phase = 0.0;
894  susp->susp.s2_phase_incr = s2->sr / sr;
895  susp->susp.output_per_s2 = sr / s2->sr;
896 #endif
897  return sound_create((snd_susp_type)susp, t0, sr, 1.0);
898 }
