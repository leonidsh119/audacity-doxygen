106 // sorting is quite subtle due to rounding
107 // For example, suppose times from a MIDI file are exact, but in
108 // decimal round to TW0.4167 Q0.3333. Since the time in whole notes
109 // rounded up, this note will start late. Even though the duration
110 // rounded down, the amount is 1/4 as much because units are quarter
111 // notes. Therefore, the total roundup is 0.0001 beats. This is
112 // enough to cause the note to sort later in the queue, perhaps
113 // coming after a new note-on on the same pitch, and resulting in
114 // a turning on-off, on-off into on, on, off, off if data is moved
115 // to Allegro (ascii) format with rounding and then back to SMF.
116 //
117 // The solution here is to consider things that round to the same
118 // tick to be simultaneous. Then, be sure to deal with note-offs
119 // before note-ons. We're going to do that by using event_queue
120 // times that are rounded to the nearest tick time. Except note-offs
121 // are going to go in with times that are 1/4 tick earlier so they
122 // get scheduled first, but still end up on the same tick.
123 //
124 event_queue* push(event_queue *queue, event_queue *event)
125 {
126  // printf("push: %.6g, %c, %d\n", event->time, event->type, event->index);
127  if (queue == NULL) {
128  event->next = NULL;
129  return event;
130  }
131 
132  event_queue *marker1 = NULL;
133  event_queue *marker2 = queue;
134  while (marker2 != NULL && marker2->time <= event->time) {
135  marker1 = marker2;
136  marker2 = marker2->next;
137  }
138  event->next = marker2;
139  if (marker1 != NULL) {
140  marker1->next=event;
141  return queue;
142  } else return event;
143 }

414 void Alg_smf_write::write_track(int i)
415 {
416  int j = 0; // note index
417  Alg_events &notes = seq->track_list[i];
418  event_queue *pending = NULL;
419  if (notes.length() > 0) {
420  pending = new event_queue('n', TICK_TIME(notes[j]->time, 0), 0, NULL);
421  }
422  if (i == 0) { // track 0 may have tempo and timesig info
423  if (seq->get_time_map()->last_tempo_flag || seq->get_time_map()->beats.len > 0) {
424  pending = push(pending, new event_queue('c', 0.0, 0, NULL));
425  }
426  if (seq->time_sig.length() > 0) {
427  pending = push(pending, new event_queue('s',
428  TICK_TIME(seq->time_sig[0].beat, 0), 0, NULL));
429  }
430  }
431  while (pending) {
432  event_queue *current = pending;
433  pending = pending->next;
434  if (current->type == 'n') {
435  Alg_note_ptr n = (Alg_note_ptr) notes[current->index];
436  if (n->is_note()) {
437  write_note(n, true);
438  pending = push(pending, new event_queue('o',
439  TICK_TIME(n->time + n->dur, -1), current->index, NULL));
440  } else if (n->is_update()) {
441  Alg_update_ptr u = (Alg_update_ptr) n;
442  write_update(u);
443  }
444  int next = current->index + 1;
445  if (next < notes.length()) {
446  current->time = TICK_TIME(notes[next]->time, 0);
447  current->index = next;
448  pending = push(pending, current);
449  }
450  } else if (current->type == 'o') { //note-off
451  Alg_note_ptr n = (Alg_note_ptr) notes[current->index];
452  write_note(n, false);
453  delete current;
454  } else if (current->type == 'c') { // tempo change
455  write_tempo_change(current->index);
456  current->index++; // -R
457  if (current->index < seq->get_time_map()->beats.len) {
458  current->time =
459  TICK_TIME(seq->get_time_map()->
460  beats[current->index].beat, 0);
461  pending = push(pending, current);
462  } else {
463  delete current;
464  }
465  } else if (current->type == 's') { // time sig
466  write_time_signature(current->index);
467  current->index++;
468  if (current->index < seq->time_sig.length()) {
469  current->time =
470  TICK_TIME(seq->time_sig[current->index].beat, 0);
471  pending = push(pending, current);
472  } else {

474  }
475  }
