254 /* this structure combines the RIFF header, the format chunk, and the data
255  * chunk header */
256 struct wav_header {
257  /* RIFF header */
258  char riffID[4]; /* "RIFF" */
259  wxUint32 lenAfterRiff; /* basically the file len - 8, or samples len + 36 */
260  char riffType[4]; /* "WAVE" */
261 
262  /* format chunk */
263  char fmtID[4]; /* "fmt " */
264  wxUint32 formatChunkLen; /* (format chunk len - first two fields) 16 in our case */
265  wxUint16 formatTag; /* 1 for PCM */
266  wxUint16 channels;
267  wxUint32 sampleRate;
268  wxUint32 avgBytesPerSec; /* sampleRate * blockAlign */
269  wxUint16 blockAlign; /* bitsPerSample * channels (assume bps % 8 = 0) */
270  wxUint16 bitsPerSample;
271 
272  /* data chunk header */
273  char dataID[4]; /* "data" */
274  wxUint32 dataLen; /* length of all samples in bytes */
275 };

314 int ExportCL::Export(AudacityProject *project,
315  int channels,
316  const wxString &fName,
317  bool selectionOnly,
318  double t0,
319  double t1,
320  MixerSpec *mixerSpec,
321  const Tags *WXUNUSED(metadata),
322  int WXUNUSED(subformat))
323 {
324  wxString output;
325  wxString cmd;
326  bool show;
327  long rc;
328 
329  // Retrieve settings
330  gPrefs->Read(wxT("/FileFormats/ExternalProgramShowOutput"), &show, false);
331  cmd = gPrefs->Read(wxT("/FileFormats/ExternalProgramExportCommand"), wxT("lame - \"%f.mp3\""));
332  cmd.Replace(wxT("%f"), fName);
333 
334 #if defined(__WXMSW__)
335  // Give Windows a chance at finding lame command in the default location.
336  wxString paths[] = {wxT("HKEY_LOCAL_MACHINE\\Software\\Lame for Audacity"),
337  wxT("HKEY_LOCAL_MACHINE\\Software\\FFmpeg for Audacity")};
338  wxString opath;
339  wxString npath;
340  wxRegKey reg;
341 
342  wxGetEnv(wxT("PATH"), &opath);
343  npath = opath;
344 
345  for (int i = 0; i < WXSIZEOF(paths); i++) {
346  reg.SetName(paths[i]);
347 
348  if (reg.Exists()) {
349  wxString ipath;
350  reg.QueryValue(wxT("InstallPath"), ipath);
351  if (!ipath.IsEmpty()) {
352  npath += wxPATH_SEP + ipath;
353  }
354  }
355  }
356 
357  wxSetEnv(wxT("PATH"),npath.c_str());
358 #endif
359 
360  // Kick off the command
361  ExportCLProcess process(&output);
362  rc = wxExecute(cmd, wxEXEC_ASYNC, &process);
363 
364 #if defined(__WXMSW__)
365  if (!opath.IsEmpty()) {
366  wxSetEnv(wxT("PATH"),opath.c_str());
367  }
368 #endif
369 
370  if (!rc) {
371  wxMessageBox(wxString::Format(_("Cannot export audio to %s"),
372  fName.c_str()));
373  process.Detach();
374  process.CloseOutput();
375 
376  return false;
377  }
378 
379  // Turn off logging to prevent broken pipe messages
380  wxLogNull nolog;
381 
382  // establish parameters
383  int rate = lrint(project->GetRate());
384  sampleCount maxBlockLen = 44100 * 5;
385  unsigned long totalSamples = lrint((t1 - t0) * rate);
386  unsigned long sampleBytes = totalSamples * channels * SAMPLE_SIZE(int16Sample);
387 
388  // fill up the wav header
389  wav_header header;
390  header.riffID[0] = 'R';
391  header.riffID[1] = 'I';
392  header.riffID[2] = 'F';
393  header.riffID[3] = 'F';
394  header.lenAfterRiff = wxUINT32_SWAP_ON_BE(sampleBytes + 36);
395  header.riffType[0] = 'W';
396  header.riffType[1] = 'A';
397  header.riffType[2] = 'V';
398  header.riffType[3] = 'E';
399 
400  header.fmtID[0] = 'f';
401  header.fmtID[1] = 'm';
402  header.fmtID[2] = 't';
403  header.fmtID[3] = ' ';
404  header.formatChunkLen = wxUINT32_SWAP_ON_BE(16);
405  header.formatTag = wxUINT16_SWAP_ON_BE(1);
406  header.channels = wxUINT16_SWAP_ON_BE(channels);
407  header.sampleRate = wxUINT32_SWAP_ON_BE(rate);
408  header.bitsPerSample = wxUINT16_SWAP_ON_BE(SAMPLE_SIZE(int16Sample) * 8);
409  header.blockAlign = wxUINT16_SWAP_ON_BE(header.bitsPerSample * header.channels / 8);
410  header.avgBytesPerSec = wxUINT32_SWAP_ON_BE(header.sampleRate * header.blockAlign);
411  header.dataID[0] = 'd';
412  header.dataID[1] = 'a';
413  header.dataID[2] = 't';
414  header.dataID[3] = 'a';
415  header.dataLen = wxUINT32_SWAP_ON_BE(sampleBytes);
416 
417  // write the header
418  wxOutputStream *os = process.GetOutputStream();
419  os->Write(&header, sizeof(wav_header));
420 
421  // Mix 'em up
422  const TrackList *tracks = project->GetTracks();
423  const WaveTrackConstArray waveTracks =
424  tracks->GetWaveTrackConstArray(selectionOnly, false);
425  Mixer *mixer = CreateMixer(
426  waveTracks,
427  tracks->GetTimeTrack(),
428  t0,
429  t1,
430  channels,
431  maxBlockLen,
432  true,
433  rate,
434  int16Sample,
435  true,
436  mixerSpec);
437 
438  size_t numBytes = 0;
439  samplePtr mixed = NULL;
440  int updateResult = eProgressSuccess;
441 
442  {
443  // Prepare the progress display
444  ProgressDialog progress(_("Export"),
445  selectionOnly ?
446  _("Exporting the selected audio using command-line encoder") :
447  _("Exporting the entire project using command-line encoder"));
448 
449  // Start piping the mixed data to the command
450  while (updateResult == eProgressSuccess && process.IsActive() && os->IsOk()) {
451  // Capture any stdout and stderr from the command
452  Drain(process.GetInputStream(), &output);
453  Drain(process.GetErrorStream(), &output);
454 
455  // Need to mix another block
456  if (numBytes == 0) {
457  sampleCount numSamples = mixer->Process(maxBlockLen);
458  if (numSamples == 0) {
459  break;
460  }
461 
462  mixed = mixer->GetBuffer();
463  numBytes = numSamples * channels;
464 
465  // Byte-swapping is neccesary on big-endian machines, since
466  // WAV files are little-endian
467 #if wxBYTE_ORDER == wxBIG_ENDIAN
468  wxUint16 *buffer = (wxUint16 *) mixed;
469  for (int i = 0; i < numBytes; i++) {
470  buffer[i] = wxUINT16_SWAP_ON_BE(buffer[i]);
471  }
472 #endif
473  numBytes *= SAMPLE_SIZE(int16Sample);
474  }
475 
476  // Don't write too much at once...pipes may not be able to handle it
477  size_t bytes = wxMin(numBytes, 4096);
478  numBytes -= bytes;
479 
480  while (bytes > 0) {
481  os->Write(mixed, bytes);
482  if (!os->IsOk()) {
483  break;
484  }
485  bytes -= os->LastWrite();
486  mixed += os->LastWrite();
487  }
488 
489  // Update the progress display
490  updateResult = progress.Update(mixer->MixGetCurrentTime() - t0, t1 - t0);
491  }
492  // Done with the progress display
493  }
494 
495  // Should make the process die
496  process.CloseOutput();
497 
498  // Wait for process to terminate
499  while (process.IsActive()) {
500  wxMilliSleep(10);
501  wxTheApp->Yield();
502  }
503 
504  // Display output on error or if the user wants to see it
505  if (process.GetStatus() != 0 || show) {
506  // TODO use ShowInfoDialog() instead.
507  wxDialog dlg(NULL,
508  wxID_ANY,
509  wxString(_("Command Output")),
510  wxDefaultPosition,
511  wxSize(600, 400),
512  wxDEFAULT_DIALOG_STYLE | wxRESIZE_BORDER);
513  dlg.SetName(dlg.GetTitle());
514 
515  ShuttleGui S(&dlg, eIsCreating);
516  S.AddTextWindow(cmd + wxT("\n\n") + output);
517  S.StartHorizontalLay(wxALIGN_CENTER, false);
518  {
519  S.Id(wxID_OK).AddButton(_("&OK"))->SetDefault();
520  }
521  dlg.GetSizer()->AddSpacer(5);
522  dlg.Layout();
523  dlg.SetMinSize(dlg.GetSize());
524  dlg.Center();
525 
526  dlg.ShowModal();
527  }
528 
529  // Clean up
530  delete mixer;
531 
532  return updateResult;
533 }
