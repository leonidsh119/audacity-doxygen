375 sampleCount EffectPhaser::InstanceProcess(EffectPhaserState & data, float **inBlock, float **outBlock, sampleCount blockLen)
376 {
377  float *ibuf = inBlock[0];
378  float *obuf = outBlock[0];
379 
380  for (int j = data.laststages; j < mStages; j++)
381  {
382  data.old[j] = 0;
383  }
384  data.laststages = mStages;
385 
386  data.lfoskip = mFreq * 2 * M_PI / data.samplerate;
387  data.phase = mPhase * M_PI / 180;
388  data.outgain = DB_TO_LINEAR(mOutGain);
389 
390  for (sampleCount i = 0; i < blockLen; i++)
391  {
392  double in = ibuf[i];
393 
394  double m = in + data.fbout * mFeedback / 101; // Feedback must be less than 100% to avoid infinite gain.
395 
396  if (((data.skipcount++) % lfoskipsamples) == 0)
397  {
398  //compute sine between 0 and 1
399  data.gain = (1.0 + cos(data.skipcount * data.lfoskip + data.phase)) / 2.0;
400 
401  // change lfo shape
402  data.gain = expm1(data.gain * phaserlfoshape) / expm1(phaserlfoshape);
403 
404  // attenuate the lfo
405  data.gain = 1.0 - data.gain / 255.0 * mDepth;
406  }
407 
408  // phasing routine
409  for (int j = 0; j < mStages; j++)
410  {
411  double tmp = data.old[j];
412  data.old[j] = data.gain * tmp + m;
413  m = tmp - data.gain * data.old[j];
414  }
415  data.fbout = m;
416 
417  obuf[i] = (float) (data.outgain * (m * mDryWet + in * (255 - mDryWet)) / 255);
418  }
419 
420  return blockLen;
421 }
