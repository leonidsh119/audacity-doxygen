222 void EditToolBar::OnButton(wxCommandEvent &event)
223 {
224  AudacityProject *p = GetActiveProject();
225  if (!p) return;
226 
227  bool busy = gAudioIO->IsBusy();
228  int id = event.GetId();
229 
230  switch (id) {
231  case ETBCutID:
232  if (!busy) p->OnCut();
233  break;
234  case ETBCopyID:
235  if (!busy) p->OnCopy();
236  break;
237  case ETBPasteID:
238  if (!busy) p->OnPaste();
239  break;
240  case ETBTrimID:
241  if (!busy) p->OnTrim();
242  break;
243  case ETBSilenceID:
244  if (!busy) p->OnSilence();
245  break;
246  case ETBUndoID:
247  if (!busy) p->OnUndo();
248  break;
249  case ETBRedoID:
250  if (!busy) p->OnRedo();
251  break;
252 #ifdef EXPERIMENTAL_SYNC_LOCK
253  case ETBSyncLockID:
254  p->OnSyncLock();
255  return;//avoiding the call to SetButton()
256 #endif
257  case ETBZoomInID:
258  p->OnZoomIn();
259  break;
260  case ETBZoomOutID:
261  p->OnZoomOut();
262  break;
263 
264 #if 0 // Disabled for version 1.2.0 since it doesn't work quite right...
265  case ETBZoomToggleID:
266  p->OnZoomToggle();
267  break;
268 #endif
269 
270  case ETBZoomSelID:
271  p->OnZoomSel();
272  break;
273  case ETBZoomFitID:
274  p->OnZoomFit();
275  break;
276 #if defined(EXPERIMENTAL_EFFECTS_RACK)
277  case ETBEffectsID:
278  EffectManager::Get().ShowRack();
279  break;
280 #endif
281  }
282 
283  SetButton(false, mButtons[id]);

286 void EditToolBar::EnableDisableButtons()
287 {
288  AudacityProject *p = GetActiveProject();
289  if (!p) return;
290 
291  // Is anything selected?
292  bool selection = false;
293  TrackListIterator iter(p->GetTracks());
294  for (Track *t = iter.First(); t; t = iter.Next())
295  if (t->GetSelected()) {
296  selection = true;
297  break;
298  }
299  selection &= (p->GetSel0() < p->GetSel1());
300 
301  mButtons[ETBCutID]->SetEnabled(selection);
302  mButtons[ETBCopyID]->SetEnabled(selection);
303  mButtons[ETBTrimID]->SetEnabled(selection);
304  mButtons[ETBSilenceID]->SetEnabled(selection);
305 
306  mButtons[ETBUndoID]->SetEnabled(p->GetUndoManager()->UndoAvailable());
307  mButtons[ETBRedoID]->SetEnabled(p->GetUndoManager()->RedoAvailable());
308 
309  bool tracks = (!p->GetTracks()->IsEmpty());
310 
311  mButtons[ETBZoomInID]->SetEnabled(tracks && (p->ZoomInAvailable()));
312  mButtons[ETBZoomOutID]->SetEnabled(tracks && (p->ZoomOutAvailable()) );
313 
314  #if 0 // Disabled for version 1.2.0 since it doesn't work quite right...
315  mButtons[ETBZoomToggleID]->SetEnabled(tracks);
316  #endif
317 
318  mButtons[ETBZoomSelID]->SetEnabled(selection);
319  mButtons[ETBZoomFitID]->SetEnabled(tracks);
320 
321  mButtons[ETBPasteID]->SetEnabled(p->Clipboard());
322 
323 #ifdef EXPERIMENTAL_SYNC_LOCK
324  bool bSyncLockTracks;
325  gPrefs->Read(wxT("/GUI/SyncLockTracks"), &bSyncLockTracks, false);
326 
327  if (bSyncLockTracks)
328  mButtons[ETBSyncLockID]->PushDown();
329  else
330  mButtons[ETBSyncLockID]->PopUp();
331 #endif
332 }
