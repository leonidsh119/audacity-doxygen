192 void LibraryPrefs::OnFFmpegFindButton(wxCommandEvent & WXUNUSED(event))
193 {
194 #ifdef USE_FFMPEG
195  FFmpegLibs* FFmpegLibsInst = PickFFmpegLibs();
196  bool showerrs =
197 #if defined(__WXDEBUG__)
198  true;
199 #else
200  false;
201 #endif
202 
203  FFmpegLibsInst->FreeLibs();
204  // Load the libs ('true' means that all errors will be shown)
205  bool locate = !LoadFFmpeg(showerrs);
206 
207  // Libs are fine, don't show "locate" dialog unless user really wants it
208  if (!locate) {
209  int response = wxMessageBox(_("Audacity has automatically detected valid FFmpeg libraries.\nDo you still want to locate them manually?"),
210  wxT("Success"),
211  wxCENTRE | wxYES_NO | wxNO_DEFAULT |wxICON_QUESTION);
212  if (response == wxYES) {
213  locate = true;
214  }
215  }
216 
217  if (locate) {
218  // Show "Locate FFmpeg" dialog
219  FFmpegLibsInst->FindLibs(this);
220  FFmpegLibsInst->FreeLibs();
221  LoadFFmpeg(showerrs);
222  }
223  SetFFmpegVersionText();
224 
225  DropFFmpegLibs();
226 #endif
227 }
