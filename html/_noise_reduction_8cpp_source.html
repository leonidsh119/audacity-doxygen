650  SelectedTrackListOfKindIterator &iter, double mT0, double mT1)
651 {

653  WaveTrack *track = (WaveTrack *) iter.First();
654  while (track) {
655  if (track->GetRate() != mSampleRate) {
656  if (mDoProfile)
657  ::wxMessageBox(_("All noise profile data must have the same sample rate."));

664  double trackEnd = track->GetEndTime();
665  double t0 = std::max(trackStart, mT0);
666  double t1 = std::min(trackEnd, mT1);
667 
668  if (t1 > t0) {

670  sampleCount end = track->TimeToLongSamples(t1);
671  sampleCount len = (sampleCount)(end - start);

683  ::wxMessageBox(_("Selected noise profile is too short."));
684  return false;

691 void EffectNoiseReduction::Worker::ApplyFreqSmoothing(FloatVector &gains)
692 {
693  // Given an array of gain mutipliers, average them
694  // GEOMETRICALLY. Don't multiply and take nth root --
695  // that may quickly cause underflows. Instead, average the logs.
696 
697  if (mFreqSmoothingBins == 0)
698  return;
699 
700  {
701  float *pScratch = &mFreqSmoothingScratch[0];
702  std::fill(pScratch, pScratch + mSpectrumSize, 0.0f);
703  }
704 
705  for (int ii = 0; ii < mSpectrumSize; ++ii)
706  gains[ii] = log(gains[ii]);
707 
708  for (int ii = 0; ii < mSpectrumSize; ++ii) {
709  const int j0 = std::max(0, ii - mFreqSmoothingBins);
710  const int j1 = std::min(mSpectrumSize - 1, ii + mFreqSmoothingBins);
711  for(int jj = j0; jj <= j1; ++jj) {
712  mFreqSmoothingScratch[ii] += gains[jj];
713  }
714  mFreqSmoothingScratch[ii] /= (j1 - j0 + 1);
715  }
716 
717  for (int ii = 0; ii < mSpectrumSize; ++ii)
718  gains[ii] = exp(mFreqSmoothingScratch[ii]);
719 }

751 , mNewSensitivity(settings.mNewSensitivity * log(10.0))
752 
753 , mInSampleCount(0)
754 , mOutStepCount(0)
755 , mInWavePos(0)
756 {
757 #ifdef EXPERIMENTAL_SPECTRAL_EDITING
758  {
759  const double bin = mSampleRate / mWindowSize;
760  if (f0 >= 0.0 )
761  mBinLow = floor(f0 / bin);
762  if (f1 >= 0.0)
763  mBinHigh = ceil(f1 / bin);
764  }
765 #endif
766 
767  const double noiseGain = -settings.mNoiseGain;
768  const int nAttackBlocks = 1 + (int)(settings.mAttackTime * sampleRate / mStepSize);
769  const int nReleaseBlocks = 1 + (int)(settings.mReleaseTime * sampleRate / mStepSize);
770  // Applies to amplitudes, divide by 20:
771  mNoiseAttenFactor = DB_TO_LINEAR(noiseGain);
772  // Apply to gain factors which apply to amplitudes, divide by 20:
773  mOneBlockAttack = DB_TO_LINEAR(noiseGain / nAttackBlocks);
774  mOneBlockRelease = DB_TO_LINEAR(noiseGain / nReleaseBlocks);
775  // Applies to power, divide by 10:
776  mOldSensitivityFactor = pow(10.0, settings.mOldSensitivity / 10.0);
777 
778  mNWindowsToExamine = (mMethod == DM_OLD_METHOD)
779  ? std::max(2, (int)(minSignalTime * sampleRate / mStepSize))
780  : 1 + mStepsPerWindow;
781 
782  mCenter = mNWindowsToExamine / 2;
783  wxASSERT(mCenter >= 1); // release depends on this assumption
784 
785  if (mDoProfile)
786 #ifdef OLD_METHOD_AVAILABLE
787  mHistoryLen = mNWindowsToExamine;
788 #else
789  mHistoryLen = 1;
790 #endif
791  else {
792  // Allow long enough queue for sufficient inspection of the middle
793  // and for attack processing
794  // See ReduceNoise()
795  mHistoryLen = std::max(mNWindowsToExamine, mCenter + nAttackBlocks);
796  }
797 
798  mQueue.resize(mHistoryLen);
799  for (int ii = 0; ii < mHistoryLen; ++ii)
800  mQueue[ii] = new Record(mSpectrumSize);
801 
802  // Create windows
803 
804  const double constantTerm =
805  windowTypesInfo[settings.mWindowTypes].productConstantTerm;
806 
807  // One or the other window must by multiplied by this to correct for
808  // overlap. Must scale down as steps get smaller, and overlaps larger.
809  const double multiplier = 1.0 / (constantTerm * mStepsPerWindow);
810 
811  // Create the analysis window
812  switch (settings.mWindowTypes) {
813  case WT_RECTANGULAR_HANN:
814  break;
815  default:
816  {
817  const bool rectangularOut =
818  settings.mWindowTypes == WT_HAMMING_RECTANGULAR ||
819  settings.mWindowTypes == WT_HANN_RECTANGULAR;
820  const double m =
821  rectangularOut ? multiplier : 1;
822  const double *const coefficients =
823  windowTypesInfo[settings.mWindowTypes].inCoefficients;
824  const double c0 = coefficients[0];
825  const double c1 = coefficients[1];
826  const double c2 = coefficients[2];
827  mInWindow.resize(mWindowSize);
828  for (int ii = 0; ii < mWindowSize; ++ii)
829  mInWindow[ii] = m *
830  (c0 + c1 * cos((2.0*M_PI*ii) / mWindowSize)
831  + c2 * cos((4.0*M_PI*ii) / mWindowSize));
832  }
833  break;
834  }
835 
836  if (!mDoProfile) {
837  // Create the synthesis window
838  switch (settings.mWindowTypes) {
839  case WT_HANN_RECTANGULAR:
840  case WT_HAMMING_RECTANGULAR:
841  break;
842  case WT_HAMMING_INV_HAMMING:
843  {
844  mOutWindow.resize(mWindowSize);
845  for (int ii = 0; ii < mWindowSize; ++ii)
846  mOutWindow[ii] = multiplier / mInWindow[ii];
847  }
848  break;
849  default:
850  {
851  const double *const coefficients =
852  windowTypesInfo[settings.mWindowTypes].outCoefficients;
853  const double c0 = coefficients[0];
854  const double c1 = coefficients[1];
855  const double c2 = coefficients[2];
856  mOutWindow.resize(mWindowSize);
857  for (int ii = 0; ii < mWindowSize; ++ii)
858  mOutWindow[ii] = multiplier *
859  (c0 + c1 * cos((2.0*M_PI*ii) / mWindowSize)
860  + c2 * cos((4.0*M_PI*ii) / mWindowSize));
861  }
862  break;
863  }
864  }

942 void EffectNoiseReduction::Worker::FillFirstHistoryWindow()
943 {
944  // Transform samples to frequency domain, windowed as needed
945  if (mInWindow.size() > 0)
946  for (int ii = 0; ii < mWindowSize; ++ii)
947  mFFTBuffer[ii] = mInWaveBuffer[ii] * mInWindow[ii];
948  else

950  RealFFTf(&mFFTBuffer[0], hFFT);
951 
952  Record &record = *mQueue[0];
953 
954  // Store real and imaginary parts for later inverse FFT, and compute
955  // power
956  {
957  float *pReal = &record.mRealFFTs[1];
958  float *pImag = &record.mImagFFTs[1];
959  float *pPower = &record.mSpectrums[1];
960  int *pBitReversed = &hFFT->BitReversed[1];
961  const int last = mSpectrumSize - 1;
962  for (int ii = 1; ii < last; ++ii) {
963  const int kk = *pBitReversed++;
964  const float realPart = *pReal++ = mFFTBuffer[kk];
965  const float imagPart = *pImag++ = mFFTBuffer[kk + 1];
966  *pPower++ = realPart * realPart + imagPart * imagPart;
967  }
968  // DC and Fs/2 bins need to be handled specially
969  const float dc = mFFTBuffer[0];
970  record.mRealFFTs[0] = dc;
971  record.mSpectrums[0] = dc*dc;
972 
973  const float nyquist = mFFTBuffer[1];
974  record.mImagFFTs[0] = nyquist; // For Fs/2, not really imaginary
975  record.mSpectrums[last] = nyquist * nyquist;
976  }
977 
978  if (mNoiseReductionChoice != NRC_ISOLATE_NOISE)
979  {
980  // Default all gains to the reduction factor,
981  // until we decide to raise some of them later
982  float *pGain = &record.mGains[0];
983  std::fill(pGain, pGain + mSpectrumSize, mNoiseAttenFactor);
984  }

1067 // Return true iff the given band of the "center" window looks like noise.
1068 // Examine the band in a few neighboring windows to decide.
1069 inline
1070 bool EffectNoiseReduction::Worker::Classify(const Statistics &statistics, int band)
1071 {
1072  switch (mMethod) {
1073 #ifdef OLD_METHOD_AVAILABLE
1074  case DM_OLD_METHOD:
1075  {
1076  float min = mQueue[0]->mSpectrums[band];
1077  for (int ii = 1; ii < mNWindowsToExamine; ++ii)
1078  min = std::min(min, mQueue[ii]->mSpectrums[band]);
1079  return min <= mOldSensitivityFactor * statistics.mNoiseThreshold[band];
1080  }
1081 #endif
1082  // New methods suppose an exponential distribution of power values
1083  // in the noise; NEW sensitivity is meant to be log of probability
1084  // that noise strays above the threshold. Call that probability
1085  // 1 - F. The quantile function of an exponential distribution is
1086  // log (1 - F) * mean. Thus simply multiply mean by sensitivity
1087  // to get the threshold.
1088  case DM_MEDIAN:
1089  // This method examines the window and all windows
1090  // that partly overlap it, and takes a median, to
1091  // avoid being fooled by up and down excursions into
1092  // either the mistake of classifying noise as not noise
1093  // (leaving a musical noise chime), or the opposite
1094  // (distorting the signal with a drop out).
1095  if (mNWindowsToExamine == 3)
1096  // No different from second greatest.
1097  goto secondGreatest;
1098  else if (mNWindowsToExamine == 5)
1099  {
1100  float greatest = 0.0, second = 0.0, third = 0.0;
1101  for (int ii = 0; ii < mNWindowsToExamine; ++ii) {
1102  const float power = mQueue[ii]->mSpectrums[band];
1103  if (power >= greatest)
1104  third = second, second = greatest, greatest = power;
1105  else if (power >= second)
1106  third = second, second = power;
1107  else if (power >= third)
1108  third = power;
1109  }
1110  return third <= mNewSensitivity * statistics.mMeans[band];
1111  }
1112  else {
1113  wxASSERT(false);
1114  return true;
1115  }
1116  secondGreatest:
1117  case DM_SECOND_GREATEST:
1118  {
1119  // This method just throws out the high outlier. It
1120  // should be less prone to distortions and more prone to
1121  // chimes.
1122  float greatest = 0.0, second = 0.0;
1123  for (int ii = 0; ii < mNWindowsToExamine; ++ii) {
1124  const float power = mQueue[ii]->mSpectrums[band];
1125  if (power >= greatest)
1126  second = greatest, greatest = power;
1127  else if (power >= second)
1128  second = power;
1129  }
1130  return second <= mNewSensitivity * statistics.mMeans[band];
1131  }
1132  default:
1133  wxASSERT(false);
1134  return true;
1135  }

1138 void EffectNoiseReduction::Worker::ReduceNoise
1139 (const Statistics &statistics, WaveTrack *outputTrack)
1140 {
1141  // Raise the gain for elements in the center of the sliding history
1142  // or, if isolating noise, zero out the non-noise
1143  {
1144  float *pGain = &mQueue[mCenter]->mGains[0];
1145  if (mNoiseReductionChoice == NRC_ISOLATE_NOISE) {
1146  // All above or below the selected frequency range is non-noise
1147  std::fill(pGain, pGain + mBinLow, 0.0f);
1148  std::fill(pGain + mBinHigh, pGain + mSpectrumSize, 0.0f);
1149  pGain += mBinLow;
1150  for (int jj = mBinLow; jj < mBinHigh; ++jj) {
1151  const bool isNoise = Classify(statistics, jj);
1152  *pGain++ = isNoise ? 1.0 : 0.0;
1153  }
1154  }
1155  else {
1156  // All above or below the selected frequency range is non-noise
1157  std::fill(pGain, pGain + mBinLow, 1.0f);
1158  std::fill(pGain + mBinHigh, pGain + mSpectrumSize, 1.0f);
1159  pGain += mBinLow;
1160  for (int jj = mBinLow; jj < mBinHigh; ++jj) {
1161  const bool isNoise = Classify(statistics, jj);
1162  if (!isNoise)
1163  *pGain = 1.0;
1164  ++pGain;
1165  }
1166  }
1167  }
1168 
1169  if (mNoiseReductionChoice != NRC_ISOLATE_NOISE)
1170  {
1171  // In each direction, define an exponential decay of gain from the
1172  // center; make actual gains the maximum of mNoiseAttenFactor, and
1173  // the decay curve, and their prior values.
1174 
1175  // First, the attack, which goes backward in time, which is,
1176  // toward higher indices in the queue.
1177  for (int jj = 0; jj < mSpectrumSize; ++jj) {
1178  for (int ii = mCenter + 1; ii < mHistoryLen; ++ii) {
1179  const float minimum =
1180  std::max(mNoiseAttenFactor,
1181  mQueue[ii - 1]->mGains[jj] * mOneBlockAttack);
1182  float &gain = mQueue[ii]->mGains[jj];
1183  if (gain < minimum)
1184  gain = minimum;
1185  else
1186  // We can stop now, our attack curve is intersecting
1187  // the decay curve of some window previously processed.
1188  break;
1189  }
1190  }
1191 
1192  // Now, release. We need only look one window ahead. This part will
1193  // be visited again when we examine the next window, and
1194  // carry the decay further.
1195  {
1196  float *pNextGain = &mQueue[mCenter - 1]->mGains[0];
1197  const float *pThisGain = &mQueue[mCenter]->mGains[0];
1198  for (int nn = mSpectrumSize; nn--;) {
1199  *pNextGain =
1200  std::max(*pNextGain,
1201  std::max(mNoiseAttenFactor,
1202  *pThisGain++ * mOneBlockRelease));
1203  ++pNextGain;
1204  }
1205  }
1206  }
1207 
1208 
1209  if (mOutStepCount >= -(mStepsPerWindow - 1)) {
1210  Record &record = *mQueue[mHistoryLen - 1]; // end of the queue
1211  const int last = mSpectrumSize - 1;
1212 
1213  if (mNoiseReductionChoice != NRC_ISOLATE_NOISE)
1214  // Apply frequency smoothing to output gain
1215  // Gains are not less than mNoiseAttenFactor
1216  ApplyFreqSmoothing(record.mGains);
1217 
1218  // Apply gain to FFT
1219  {
1220  const float *pGain = &record.mGains[1];
1221  const float *pReal = &record.mRealFFTs[1];
1222  const float *pImag = &record.mImagFFTs[1];
1223  float *pBuffer = &mFFTBuffer[2];
1224  int nn = mSpectrumSize - 2;
1225  if (mNoiseReductionChoice == NRC_LEAVE_RESIDUE) {
1226  for (; nn--;) {
1227  // Subtract the gain we would otherwise apply from 1, and
1228  // negate that to flip the phase.
1229  const double gain = *pGain++ - 1.0;
1230  *pBuffer++ = *pReal++ * gain;
1231  *pBuffer++ = *pImag++ * gain;
1232  }
1233  mFFTBuffer[0] = record.mRealFFTs[0] * (record.mGains[0] - 1.0);
1234  // The Fs/2 component is stored as the imaginary part of the DC component
1235  mFFTBuffer[1] = record.mImagFFTs[0] * (record.mGains[last] - 1.0);
1236  }
1237  else {
1238  for (; nn--;) {
1239  const double gain = *pGain++;
1240  *pBuffer++ = *pReal++ * gain;
1241  *pBuffer++ = *pImag++ * gain;
1242  }
1243  mFFTBuffer[0] = record.mRealFFTs[0] * record.mGains[0];
1244  // The Fs/2 component is stored as the imaginary part of the DC component
1245  mFFTBuffer[1] = record.mImagFFTs[0] * record.mGains[last];
1246  }
1247  }
1248 
1249  // Invert the FFT into the output buffer
1250  InverseRealFFTf(&mFFTBuffer[0], hFFT);
1251 
1252  // Overlap-add
1253  if (mOutWindow.size() > 0) {
1254  float *pOut = &mOutOverlapBuffer[0];
1255  float *pWindow = &mOutWindow[0];
1256  int *pBitReversed = &hFFT->BitReversed[0];
1257  for (int jj = 0; jj < last; ++jj) {
1258  int kk = *pBitReversed++;
1259  *pOut++ += mFFTBuffer[kk] * (*pWindow++);
1260  *pOut++ += mFFTBuffer[kk + 1] * (*pWindow++);
1261  }
1262  }
1263  else {
1264  float *pOut = &mOutOverlapBuffer[0];
1265  int *pBitReversed = &hFFT->BitReversed[0];
1266  for (int jj = 0; jj < last; ++jj) {
1267  int kk = *pBitReversed++;
1268  *pOut++ += mFFTBuffer[kk];
1269  *pOut++ += mFFTBuffer[kk + 1];
1270  }
1271  }
1272 
1273  float *buffer = &mOutOverlapBuffer[0];
1274  if (mOutStepCount >= 0) {
1275  // Output the first portion of the overlap buffer, they're done
1276  outputTrack->Append((samplePtr)buffer, floatSample, mStepSize);
1277  }
1278 
1279  // Shift the remainder over.
1280  memmove(buffer, buffer + mStepSize, sizeof(float)*(mWindowSize - mStepSize));
1281  std::fill(buffer + mWindowSize - mStepSize, buffer + mWindowSize, 0.0f);
1282  }
1283 }

1285 bool EffectNoiseReduction::Worker::ProcessOne
1286 (EffectNoiseReduction &effect, Statistics &statistics, TrackFactory &factory,
1287  int count, WaveTrack * track, sampleCount start, sampleCount len)
1288 {
1289  if (track == NULL)
1290  return false;
1291 
1292  StartNewTrack();
1293 
1294  std::unique_ptr<WaveTrack> outputTrack(
1295  mDoProfile ? NULL
1296  : factory.NewWaveTrack(track->GetSampleFormat(), track->GetRate()));
1297 
1298  sampleCount bufferSize = track->GetMaxBlockSize();
1299  FloatVector buffer(bufferSize);
1300 
1301  bool bLoopSuccess = true;
1302  sampleCount blockSize;
1303  sampleCount samplePos = start;
1304  while (bLoopSuccess && samplePos < start + len) {
1305  //Get a blockSize of samples (smaller than the size of the buffer)
1306  blockSize = std::min(start + len - samplePos, track->GetBestBlockSize(samplePos));
1307

1309  track->Get((samplePtr)&buffer[0], floatSample, samplePos, blockSize);
1310  samplePos += blockSize;

1316  bLoopSuccess =
1317  !effect.TrackProgress(count, (samplePos - start) / (double)len);
1318  }
1319

1321  if (mDoProfile)
1322  FinishTrackStatistics(statistics);
1323  else
1324  FinishTrack(statistics, &*outputTrack);

1329  outputTrack->Flush();
1330

1332  // sample data (as operated on -- this may not match mT0/mT1)

1334  double tLen = outputTrack->LongSamplesToTime(len);
1335  // Filtering effects always end up with more data than they started with. Delete this 'tail'.
1336  outputTrack->HandleClear(tLen, outputTrack->GetEndTime(), false, false);
1337  bool bResult = track->ClearAndPaste(t0, t0 + tLen, &*outputTrack, true, false);
1338  wxASSERT(bResult); // TO DO: Actually handle this.
1339  wxUnusedVar(bResult);
1340  }
1341

1343 }
