181  int FilterEvent(wxEvent& event)
182  {
183  // Quickly bail if this isn't something we want.
184  wxEventType type = event.GetEventType();
185  if (type != wxEVT_CHAR_HOOK && type != wxEVT_KEY_UP)
186  {
187  return Event_Skip;
188  }
189 
190  // We must have a project since we will be working with the Command Manager
191  // and capture handler, both of which are (currently) tied to individual projects.
192  //
193  // Shouldn't they be tied to the application instead???
194  AudacityProject *project = GetActiveProject();
195  if (!project || !project->IsEnabled())
196  {
197  return Event_Skip;
198  }
199 
200  // Make a copy of the event and (possibly) make it look like a key down
201  // event.
202  wxKeyEvent key = (wxKeyEvent &) event;
203  if (type == wxEVT_CHAR_HOOK)
204  {
205  key.SetEventType(wxEVT_KEY_DOWN);
206  }
207 
208  // Give the capture handler first dibs at the event.
209  wxWindow *handler = project->GetKeyboardCaptureHandler();
210  if (handler && HandleCapture(handler, key))
211  {
212  return Event_Processed;
213  }
214 
215  // Capture handler didn't want it, so ask the Command Manager.
216  CommandManager *manager = project->GetCommandManager();
217  if (manager && manager->FilterKeyEvent(project, key))
218  {
219  return Event_Processed;
220  }
221 
222  // Give it back to WX for normal processing.
223  return Event_Skip;
224  }

228  // Returns true if the event was captured and processed
229  bool HandleCapture(wxWindow *target, const wxKeyEvent & event)
230  {
231  if (wxGetTopLevelParent(target) != wxGetTopLevelParent(wxWindow::FindFocus()))
232  {
233  return false;
234  }
235  wxEvtHandler *handler = target->GetEventHandler();
236 
237  // We make a copy of the event because the capture handler may modify it.
238  wxKeyEvent temp = event;
239 
240 #if defined(__WXGTK__)
241  // wxGTK uses the control and alt modifiers to represent ALTGR,
242  // so remove it as it might confuse the capture handlers.
243  if (temp.GetModifiers() == (wxMOD_CONTROL | wxMOD_ALT))
244  {
245  temp.SetControlDown(false);
246  temp.SetAltDown(false);
247  }
248 #endif
249 
250  // Ask the capture handler if the key down/up event is something they it
251  // might be interested in handling.
252  wxCommandEvent e(EVT_CAPTURE_KEY);
253  e.SetEventObject(&temp);
254  e.StopPropagation();
255  if (!handler->ProcessEvent(e))
256  {
257  return false;
258  }
259 
260  // Now, let the handler process the normal key event.
261  bool keyDown = temp.GetEventType() == wxEVT_KEY_DOWN;
262  temp.WasProcessed();
263  temp.StopPropagation();
264  wxEventProcessInHandlerOnly onlyDown(temp, handler);
265  bool processed = handler->ProcessEvent(temp);
266 
267  // Don't go any further if the capture handler didn't process
268  // the key down event.
269  if (!processed && keyDown)
270  {
271  return false;
272  }
273 
274  // At this point the capture handler has either processed a key down event
275  // or we're dealing with a key up event.
276  //
277  // So, only generate the char events for key down events.
278  if (keyDown)
279  {
280  wxString chars = GetUnicodeString(temp);

597  pos = 0;
598

608  }

610  item = NULL;
611  }
612 
613  if (item == NULL) {
614  return;
615  }
616 
617  if (item->IsSubMenu()) {
618  menu = item->GetSubMenu();
619  item = NULL;
620  continue;
621  }
622 
623  if (ndx + 1 != cnt) {
624  return;
625  }
626  }
627 
628  CommandListEntry *entry = NewIdentifier(name, label_in, menu, callback, false, 0, 0);
629  int ID = entry->id;
630  wxString label = GetLabel(entry);
631 
632  if (checkmark >= 0) {
633  menu->InsertCheckItem(pos, ID, label);
634  menu->Check(ID, checkmark != 0);
635  }
636  else {

638  }

640  mbSeparatorAllowed = true;
