142 bool TimeTrack::HandleXMLTag(const wxChar *tag, const wxChar **attrs)
143 {
144  if (!wxStrcmp(tag, wxT("timetrack"))) {
145  mRescaleXMLValues = true; // will be set to false if upper/lower is found
146  long nValue;
147  while(*attrs) {
148  const wxChar *attr = *attrs++;
149  const wxChar *value = *attrs++;
150 
151  if (!value)
152  break;
153 
154  const wxString strValue = value;
155  if (!wxStrcmp(attr, wxT("name")) && XMLValueChecker::IsGoodString(strValue))
156  mName = strValue;
157  else if (!wxStrcmp(attr, wxT("height")) &&
158  XMLValueChecker::IsGoodInt(strValue) && strValue.ToLong(&nValue))
159  mHeight = nValue;
160  else if (!wxStrcmp(attr, wxT("minimized")) &&
161  XMLValueChecker::IsGoodInt(strValue) && strValue.ToLong(&nValue))
162  mMinimized = (nValue != 0);
163  else if (!wxStrcmp(attr, wxT("rangelower")))
164  {
165  mRangeLower = Internat::CompatibleToDouble(value);
166  mRescaleXMLValues = false;
167  }
168  else if (!wxStrcmp(attr, wxT("rangeupper")))
169  {
170  mRangeUpper = Internat::CompatibleToDouble(value);
171  mRescaleXMLValues = false;
172  }
173  else if (!wxStrcmp(attr, wxT("displaylog")) &&
174  XMLValueChecker::IsGoodInt(strValue) && strValue.ToLong(&nValue))
175  {
176  SetDisplayLog(nValue != 0);
177  //TODO-MB: This causes a graphical glitch, TrackPanel should probably be Refresh()ed after loading.
178  // I don't know where to do this though.
179  }
180  else if (!wxStrcmp(attr, wxT("interpolatelog")) &&
181  XMLValueChecker::IsGoodInt(strValue) && strValue.ToLong(&nValue))
182  {
183  SetInterpolateLog(nValue != 0);
184  }
185 
186  } // while
187  if(mRescaleXMLValues)
188  mEnvelope->SetRange(0.0, 1.0); // this will be restored to the actual range later
189  return true;
190  }
191 
192  return false;
193 }

258  mRuler->Draw(dc, this);

261  GetEnvelope()->GetValues(envValues, mid.width, 0, zoomInfo);
262 
263  dc.SetPen(AColor::envelopePen);
264 
265  double logLower = log(std::max(1.0e-7, mRangeLower)), logUpper = log(std::max(1.0e-7, mRangeUpper));
266  for (int x = 0; x < mid.width; x++)
267  {
268  double y;
269  if(mDisplayLog)
270  y = (double)mid.height * (logUpper - log(envValues[x])) / (logUpper - logLower);

273  int thisy = r.y + (int)y;
274  AColor::Line(dc, mid.x + x, thisy - 1, mid.x + x, thisy+2);
