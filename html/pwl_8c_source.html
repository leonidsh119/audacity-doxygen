38 /*
39  * returns true if it is time to terminate

41  * Note: compute_lvl gets called in the outer loop to skip over
42  * a breakpoint pair before starting the compute_incr loop, which
43  * searches for a breakpoint that is some number of samples in the

61 boolean compute_incr(pwl_susp_type susp, long *n, long cur)
62 {
63  double target;
64  while (*n == 0) {
65  *n = getfixnum(car(susp->bpt_ptr)) - cur;
66  /* if there is a 2nd element of the pair, get the target */
67  if (cdr(susp->bpt_ptr))

69  else target = 0.0;
70  if (*n > 0) susp->incr = (target - susp->lvl) / *n;
71  else if (compute_lvl(susp)) return true;
