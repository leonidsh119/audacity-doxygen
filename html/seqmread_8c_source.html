91 void seq_read_smf(seq, fp)
92  seq_type seq;
93  FILE *fp;
94 {
95  F = fp;
96  initfuncs();
97  sysex_id = 0; /* sysex in seq has to correspond to a symbol */
98  the_score = seq; /* current sequence is a global within this module */
99  if (!seq) return;
100  the_tempomap = tempomap_create();
101  /* insert an initial clock to correspond to the default midifile tempo
102  (tempomap_create creates a corresponding initial entry in the tempomap)
103  (see smf_tempo for explanation of the scale() call)
104  */
105  initial_clock = insert_clock(the_score, 0L, 0, 500L << 16);
106  /* scale(24 * 500000, 1 << 16, 24000) */
107  if (!the_tempomap) return;

111  * close it (in fact XLISP insists on closing it as a side effect of
112  * garbage collection.
113  */

115  seq_notecount(seq), seq_ctrlcount(seq));

128 /* gio_time -- get the time in millisec for Adagio */
129 /*
130  * Since Adagio times are (in their precise form) 1/256 ms, we want
131  * a similar time for midifiles, whose natural unit would be microseconds.
132  * We'll shift the microsecond time by 2 to get 1/250 ms = 4 us units
133  * and convert using the scale function when necessary.
134  * Real time is the time of the last tempo change (last_tempo_time)
135  * which is in 4us units + elapsed time.
136  * Elapsed time is the elapsed beats times the beat duration.
137  * Elapsed beats is Mf_currtime - last_tempo_beat.
138  * Beat duration is the specified tempo / division, where specified tempo
139  * is in microseconds, and division is parts per quarternote.
140  */
141 unsigned long divisions = 24L;
142 
143 time_type gio_time()
144 {
145  return (tempomap_lookup(the_tempomap, Mf_currtime) + 125L) / 250L;
146 }

173 {
174  snding_type snding;

193 void smf_noteoff(chan,pitch,vol)
194 {
195  snding_type *snding_ptr;
196  register snding_type snding;
197 /* prtime();
198  gprintf(TRANS, "Note off, chan=%d pitch=%d vol=%d\n",chan+1,pitch,vol);
199 */ /* search for the snding record */
200  for (snding_ptr = &snding_list;
201  (snding = *snding_ptr) &&
202  ((snding->pitch != pitch) || (snding->channel != chan));
203  snding_ptr = &(snding->next)) /* printf("* search *\n") */;
204  if (!snding) {
205  gprintf(TRANS, "Note off %d, channel %d ignored: no note on\n",
206  pitch, chan + 1);
207  } else {
208  event_type event = snding->event_ptr;
209  event->u.note.ndur += (gio_time() - event->ntime) << 8;
210  /* free the snding record */
211  *snding_ptr = snding->next;
212  snding_free(snding);
213  }

272 int leng;
273 char *mess;
274 {
275  char symb[10];
276  def_type defn;
277  int i;
278  sprintf(symb, "X%d", sysex_id++);
279  if (leng > 255) {
280  gprintf(TRANS, "sysex too long (%d bytes), ignored\n", leng - 2);
281  return;
282  }

286  */
287  defn = insert_def(the_score, symb, (unsigned char *) mess, leng + 2);
288  /* now fix up the definition by inserting the prefix bytes: */

291  defn->definition[0] = 0;
292  defn->definition[1] = leng;
293  insert_macro(the_score, gio_time(), 0, defn, 1, 0, NULL);

354 /* smf_tempo -- handle a midifile tempo change */
355 /*
356  * NOTE: if divisions is positive, it gives time units per quarter, and
357  * tempo is microsec per division. The product is microsec per quarter.
358  * To convert to ticksize (parameter to insert_clock), we divide by 24*1000
359  * to get units of millisec and 24ths of quarter notes. insert_clock
360  * expects this to have a 16 bit fractional part.
361  */
362 void smf_tempo(tempo)
363 long tempo;
364 {
365  time_type ctime = gio_time();
366  long ticksize = scale(tempo, 1024L, 375L);
367 /* (tempo / 24000) << 16; microsec/clock converted to ms/quarter, shifted 16*/
368 
369 /* prtime();
370  gprintf(TRANS, "Tempo, microseconds-per-MIDI-quarter-note = %ld\n",tempo);
371 */
372  tempomap_insert(the_tempomap, Mf_currtime, tempo / divisions);
373  if (ctime == 0) {
374  /* we already have a clock event at t=0 -> fix it */
375  initial_clock->u.clock.ticksize = ticksize;
376  } else { /* we need a new one */
377  /* NOTE: after the first clock, insert clock events 1/2 tick early
378  to make sure ticksize is set before clock_tick() wakes up and
379  reads it.
380  */
381  insert_clock(the_score, ctime - (prev_ticksize >> 17), 0, ticksize);
382  prev_ticksize = ticksize;
383  }
384 }
