270 /* If you think of the result as a complex pow(2,M2) by pow(2,M-1) matrix */
271 /* then the first column contains the positive wavenumber spectra of DC frequency */
272 /* followed by the positive wavenumber spectra of the nyquest frequency */
273 /* since these are two positive wavenumber spectra the first complex value */
274 /* of each is really the real values for the zero and nyquest wavenumber packed together */
275 /* All other columns contain the positive and negative wavenumber spectra of a positive frequency */
276 /* See rspect2dprod for multiplying two of these spectra together- ex. for fast convolution */
277 /* INPUTS */
278 /* *data = input data array */
279 /* M2 = log2 of fft size number of rows in */
280 /* M = log2 of fft size number of columns in */
281 /* OUTPUTS */
282 /* *data = output data array */

284 if((M2>0)&&(M>0)){
285  rffts(data, M, POW2(M2));

288  xpose(Array2d[M2]+POW2(M2)*2, 2, Array2d[M2], POW2(M2), POW2(M2), 2);

371 // When multiplying a pair of 2d spectra from rfft2d care must be taken to multiply the
372 // four real values seperately from the complex ones. This routine does it correctly.
373 // the result can be stored in-place over one of the inputs
374 /* *data1 = input data array first spectra */
375 /* *data2 = input data array second spectra */
376 /* N2 = fft size number of rows into rfft2d for both data1 and data2 */
377 /* N1 = fft size number of columns into rfft2d for both data1 and data2 */
378 long N = N2 * N1/2; // number of "complex" numbers in spectra
379 
380 if( (N2 > 1) && (N1>1)){
381  outdata[0] = data1[0] * data2[0]; // multiply the zero freq, zero wavenumber values
382  outdata[1] = data1[1] * data2[1]; // multiply the zero freq, nyquest wavenumber values
383 
384  cvprod(data1 + 2, data2 + 2, outdata + 2, N/2-1);
385 
386  outdata[N] = data1[N] * data2[N]; // multiply the nyquest freq, zero wavenumber values
387  outdata[N+1] = data1[N+1] * data2[N+1]; // multiply the nyquest freq, nyquest wavenumber values
388 
389  cvprod(data1 + N+2, data2 + N+2, outdata + N+2, N/2-1);
390 }

392  N = N2 * N1; // one of these is a 1
393  if(N>1){

395  outdata[1] = data1[1] * data2[1]; // multiply the nyquest freq values
396  cvprod(data1 + 2, data2 + 2, outdata + 2, N/2-1); // multiply the other positive freq values
