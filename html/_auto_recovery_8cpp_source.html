306 bool RecordingRecoveryHandler::HandleXMLTag(const wxChar *tag,
307  const wxChar **attrs)
308 {
309  if (wxStrcmp(tag, wxT("simpleblockfile")) == 0)
310  {
311  // Check if we have a valid channel and numchannels
312  if (mChannel < 0 || mNumChannels < 0 || mChannel >= mNumChannels)
313  {
314  // This should only happen if there is a bug
315  wxASSERT(false);
316  return false;
317  }
318 
319  WaveTrackArray tracks = mProject->GetTracks()->GetWaveTrackArray(false);
320  int index = FindTrack();
321  // We need to find the track and sequence where the blockfile belongs
322 
323  if (index < 0 || index >= (int)tracks.size())
324  {
325  // This should only happen if there is a bug
326  wxASSERT(false);
327  return false;
328  }
329 
330  WaveTrack* track = tracks[index];
331  WaveClip* clip = track->NewestOrNewClip();
332  Sequence* seq = clip->GetSequence();
333 
334  // Load the blockfile from the XML
335  DirManager* dirManager = mProject->GetDirManager();
336  dirManager->SetLoadingFormat(seq->GetSampleFormat());
337 
338  BlockArray array;
339  array.resize(1);
340  dirManager->SetLoadingTarget(&array, 0);
341  BlockFile *& blockFile = array[0].f;
342 
343  if (!dirManager->HandleXMLTag(tag, attrs) || !blockFile)
344  {
345  // This should only happen if there is a bug
346  wxASSERT(false);
347  return false;
348  }
349 
350  seq->AppendBlockFile(blockFile);
351  clip->UpdateEnvelopeTrackLen();
352 
353  } else if (wxStrcmp(tag, wxT("recordingrecovery")) == 0)
354  {
355  mAutoSaveIdent = 0;
356 
357  // loop through attrs, which is a null-terminated list of
358  // attribute-value pairs
359  long nValue;
360  while(*attrs)
361  {
362  const wxChar *attr = *attrs++;
363  const wxChar *value = *attrs++;
364 
365  if (!value)
366  break;
367 
368  const wxString strValue = value;
369  //this channels value does not correspond to WaveTrack::Left/Right/Mono, but which channel of the recording device
370  //it came from, and thus we can't use XMLValueChecker::IsValidChannel on it. Rather we compare to the next attribute value.
371  if (wxStrcmp(attr, wxT("channel")) == 0)
372  {
373  if (!XMLValueChecker::IsGoodInt(strValue) || !strValue.ToLong(&nValue) || nValue < 0)
374  return false;
375  mChannel = nValue;
376  }
377  else if (wxStrcmp(attr, wxT("numchannels")) == 0)
378  {
379  if (!XMLValueChecker::IsGoodInt(strValue) || !strValue.ToLong(&nValue) ||
380  (nValue < 1))
381  return false;
382  if(mChannel >= nValue )
383  return false;
384  mNumChannels = nValue;
385  }
386  else if (wxStrcmp(attr, wxT("id")) == 0)
387  {
388  if (!XMLValueChecker::IsGoodInt(strValue) || !strValue.ToLong(&nValue) ||
389  (nValue < 1))
390  return false;
391  mAutoSaveIdent = nValue;
392  }
393 
394  }
395  }
396 
397  return true;

435 // Simple "binary xml" format used exclusively for autosave files.
436 //
437 // It is not intended to transport these files across platform architectures,
438 // so endianness is not a concern.
439 //
440 // It is not intended that the user view or modify the file.
441 //
442 // It IS intended that very little work be done during auto save, so numbers
443 // and strings are written in their native format. They will be converted
444 // during recovery.
445 //
446 // The file has 3 main sections:
447 //
448 // ident literal "<?xml autosave>"
449 // name dictionary dictionary of all names used in the document
450 // data fields the "encoded" XML document
451 //
452 // If a subtree is added, it will be preceeded with FT_Push to tell the decoder
453 // to preserve the active dictionary. The decoder will then restore the
454 // dictionary when an FT_Pop is encountered. Nesting is unlimited.
455 //
456 // To save space, each name (attribute or element) encountered is stored in
457 // the name dictionary and replaced with the assigned 2-byte identifier.
458 //
459 // All strings are in native unicode format, 2-byte or 4-byte.
460 //
461 // All "lengths" are 2-byte signed, so are limited to 32767 bytes long.
462 
463 enum FieldTypes
464 {
465  FT_StartTag, // type, ID, name
466  FT_EndTag, // type, ID, name
467  FT_String, // type, ID, name, string length, string
468  FT_Int, // type, ID, value
469  FT_Bool, // type, ID, value
470  FT_Long, // type, ID, value
471  FT_LongLong, // type, ID, value
472  FT_SizeT, // type, ID, value
473  FT_Float, // type, ID, value
474  FT_Double, // type, ID, value
475  FT_Data, // type, string length, string
476  FT_Raw, // type, string length, string
477  FT_Push, // type only
478  FT_Pop, // type only
479  FT_Name // type, name length, name
480 };

683 bool AutoSaveFile::Decode(const wxString & fileName)
684 {
685  char ident[sizeof(AutoSaveIdent)];
686  size_t len = strlen(AutoSaveIdent);
687 
688  wxFileName fn(fileName);
689  wxFFile file;
690 
691  if (!file.Open(fn.GetFullPath(), wxT("rb")))
692  {
693  return false;
694  }
695 
696  if (file.Read(&ident, len) != len || strncmp(ident, AutoSaveIdent, len) != 0)
697  {
698  // It could be that the file has already been decoded or that it is one
699  // from 2.1.0 or earlier. In the latter case, we need to ensure the
700  // closing </project> tag is preset.
701 
702  // Close the file so we can reopen it in read/write mode
703  file.Close();
704 
705  // Add </project> tag, if necessary
706  if (!file.Open(fn.GetFullPath(), wxT("r+b")))
707  {
708  // Really shouldn't happen, but let the caller deal with it
709  return false;
710  }
711 
712  // Read the last 16 bytes of the file and check if they contain
713  // "</project>" somewhere.
714  const int bufsize = 16;
715  char buf[bufsize + 1];
716  if (file.SeekEnd(-bufsize))
717  {
718  if (file.Read(buf, bufsize) == bufsize)
719  {
720  buf[bufsize] = 0;
721  if (strstr(buf, "</project>") == 0)
722  {
723  // End of file does not contain closing </project> tag, so add it
724  if (file.Seek(0, wxFromEnd))
725  {
726  strcpy(buf, "</project>\n");
727  file.Write(buf, strlen(buf));
728  }
729  }
730  }
731  }
732 
733  file.Close();
734 
735  return true;
736  }
737 
738  len = file.Length() - len;
739  char *buf = new char[len];
740 
741  if (file.Read(buf, len) != len)
742  {
743  delete[] buf;
744  return false;
745  }
746 
747  wxMemoryInputStream in(buf, len);
748 
749  file.Close();
750 
751  // Decode to a temporary file to preserve the orignal.
752  wxString tempName = fn.CreateTempFileName(fn.GetFullPath());
753  bool opened = false;
754 
755  XMLFileWriter out;
756 
757  try
758  {
759  out.Open(tempName, wxT("wb"));
760  opened = out.IsOpened();
761  }
762  catch (const XMLFileWriterException&)
763  {
764  }
765 
766  if (!opened)
767  {
768  delete[] buf;
769 
770  wxRemoveFile(tempName);
771 
772  return false;
773  }
774 
775  mIds.clear();
776 
777  while (!in.Eof() && !out.Error())
778  {
779  short id;
780 
781  switch (in.GetC())
782  {
783  case FT_Push:
784  {
785  mIdStack.Add(mIds);
786  mIds.clear();
787  }
788  break;
789
