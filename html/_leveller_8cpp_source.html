266 void EffectLeveller::CalcLevellerFactors()
267 {
268  gLimit[1] = mDbSilenceThreshold;
269  int prev = 0;
270  double addOnValue = 0.0;
271  double prevLimit = 0.0;
272  double limit = gLimit[0];
273  gAddOnValue[0] = addOnValue;
274  double adjFactor = gAdjFactor[0];
275  double upperAdjLimit = gLimit[0] * adjFactor;
276  double prevAdjLimit = upperAdjLimit;
277  gAdjLimit[0] = upperAdjLimit;
278 
279  for (int f = 1; f < LEVELER_FACTORS; ++f) {
280  prev = f - 1;
281  adjFactor = gAdjFactor[f];
282  prevLimit = gLimit[prev];
283  limit = gLimit[f];
284  prevAdjLimit = gAdjLimit[prev];
285  addOnValue = prevAdjLimit - (adjFactor * prevLimit);
286 
287  gAddOnValue[f] = addOnValue;
288  gAdjLimit[f] = (adjFactor * limit) + addOnValue;
289  }

292 float EffectLeveller::LevelOneFrame(float frameInBuffer)
293 {
294  float curFrame;
295  float fabsCurFrame;
296  float curSign;
297 
298  curFrame = frameInBuffer;
299  if (curFrame < 0.0) {
300  curSign = -1.0;
301  }
302  else {
303  curSign = 1.0;
304  }
305  fabsCurFrame = (float)fabs(curFrame);
306 
307  for (int f = 0; f < LEVELER_FACTORS; ++f) {
308  if (fabsCurFrame <= gLimit[f]) {
309  curFrame *= (float)gAdjFactor[f];
310  curFrame += (float)(gAddOnValue[f] * curSign);
311  return curFrame;
312  }
313  }
314  return (float)0.99;
315 }
