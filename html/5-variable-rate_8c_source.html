4 /* Example 5: Variable-rate resampling (N.B. experimental). A test signal
5  * (held in a buffer) is resampled over a wide range of octaves. Resampled
6  * data is sent to stdout as raw, float32 samples. Choices of 2 test-signals
7  * and of 2 ways of varying the sample-rate are combined in a command-line
8  * option:
9  *
10  * Usage: ./5-variable-rate [0|1|2|3]
11  */
12 
13 #include <soxr.h>
14 #include "examples-common.h"

16 #define OCTAVES 5 /* Resampling range. ± */
17 #define OLEN 16 /* Output length in seconds. */
18 #define FS 44100 /* Output sampling rate in Hz. */
19 
20 /* For output pos in [0,1], returns an ioratio in the 2^±OCTAVES range: */
21 static double ioratio(double pos, int fm)
22 {
23  if (fm) /* fm: non-0 for a fast-changing ioratio, 0 for a slow sweep. */
24  pos = .5 - cos(pos * 2 * M_PI) * .4 + sin(pos * OLEN * 20 * M_PI) * .05;
25  return pow(2, 2 * OCTAVES * pos - OCTAVES);
26 }

28 int main(int argc, char *arg[])
29 {
30  int opt = argc <= 1? 2 : (atoi(arg[1]) & 3), saw = opt & 1, fm = opt & 2;
31  float ibuf[10 << OCTAVES], obuf[AL(ibuf)];
32  int i, wl = 2 << OCTAVES;
33  size_t ilen = AL(ibuf), need_input = 1;
34  size_t odone, total_odone, total_olen = OLEN * FS;
35  size_t olen1 = fm? 10 : AL(obuf); /* Small block-len if fast-changing ratio */
36  soxr_error_t error;
37 
38  /* When creating a var-rate resampler, q_spec must be set as follows: */
39  soxr_quality_spec_t q_spec = soxr_quality_spec(SOXR_HQ, SOXR_VR);
40 
41  /* The ratio of the given input rate and output rates must equate to the
42  * maximum I/O ratio that will be used: */
43  soxr_t soxr = soxr_create(1 << OCTAVES, 1, 1, &error, NULL, &q_spec, NULL);
44 
45  if (!error) {
46  USE_STD_STDIO;
47 
48  /* Generate input signal, sine or saw, with wave-length = wl: */
49  for (i = 0; i < (int)ilen; ++i)
50  ibuf[i] = (float)(saw? (i%wl)/(wl-1.)-.5 : .9 * sin(2 * M_PI * i / wl));
51 
52  /* Set the initial resampling ratio (N.B. 3rd parameter = 0): */
53  soxr_set_io_ratio(soxr, ioratio(0, fm), 0);
54 
55  /* Resample in blocks of size olen1: */
56  for (total_odone = 0; !error && total_odone < total_olen;) {
57 
58  /* The last block might be shorter: */
59  size_t block_len = min(olen1, total_olen - total_odone);
60 
61  /* Determine the position in [0,1] of the end of the current block: */
62  double pos = (double)(total_odone + block_len) / (double)total_olen;
63 
64  /* Calculate an ioratio for this position and instruct the resampler to
65  * move smoothly to the new value, over the course of outputting the next
66  * 'block_len' samples (or give 0 for an instant change instead): */
67  soxr_set_io_ratio(soxr, ioratio(pos, fm), block_len);
68 
69  /* Output the block of samples, supplying input samples as needed: */
70  do {

72  error = soxr_process(soxr, ibuf, len, NULL, obuf, block_len, &odone);
73  fwrite(obuf, sizeof(float), odone, stdout);
74 
75  /* Update counters for the current block and for the total length: */
76  block_len -= odone;
77  total_odone += odone;
78 
79  /* If soxr_process did not provide the complete block, we must call it
80  * again, supplying more input samples: */
81  need_input = block_len != 0;
82 
83  } while (need_input && !error);
84 
85  /* Now that the block for the current ioratio is complete, go back
86  * round the main `for' loop in order to process the next block. */
87  }
88  soxr_delete(soxr);
89  }
90  /* Diagnostics: */
91  fprintf(stderr, "%-26s %s; I/O: %s\n", arg[0],

93  return error || errno;
