149 static int
150 readmt(s,skip) /* read through the "MThd" or "MTrk" header string */
151 char *s;
152 int skip; /* if 1, we attempt to skip initial garbage. */
153 {
154  int nread = 0;
155  char b[4];
156  char buff[32];
157  int c;
158  char *errmsg = "expecting ";
159 
160  retry:
161  while ( nread<4 ) {
162  c = (*Mf_getc)();
163  if ( c == EOF ) {
164  errmsg = "EOF while expecting ";
165  goto err;
166  }
167  b[nread++] = c;
168  }
169  /* See if we found the 4 characters we're looking for */
170  if ( s[0]==b[0] && s[1]==b[1] && s[2]==b[2] && s[3]==b[3] )
171  return(0);
172  if ( skip ) {
173  /* If we are supposed to skip initial garbage, */
174  /* try again with the next character. */
175  b[0]=b[1];
176  b[1]=b[2];
177  b[2]=b[3];
178  nread = 3;
179  goto retry;
180  }
181  err:
182  (void) strcpy(buff,errmsg);
183  (void) strcat(buff,s);
184  mferror(buff);
185  return(0);
186 }

227 static void
228 readtrack() /* read a track chunk */
229 {
230  /* This array is indexed by the high half of a status byte. It's */
231  /* value is either the number of bytes needed (1 or 2) for a channel */
232  /* message, or 0 (meaning it's not a channel message). */
233  static int chantype[] = {
234  0, 0, 0, 0, 0, 0, 0, 0, /* 0x00 through 0x70 */
235  2, 2, 2, 2, 1, 1, 2, 0 /* 0x80 through 0xf0 */
236  };
237  long lookfor, lng;
238  int c, c1, type;
239  int sysexcontinue = 0; /* 1 if last message was an unfinished sysex */
240  int running = 0; /* 1 when running status used */
241  int status = 0; /* (possibly running) status byte */
242  int needed;
243 
244  if ( readmt("MTrk",0) == EOF )
245  return;
246 
247  Mf_toberead = read32bit();
248 
249  if (midifile_error) return;
250 
251  Mf_currtime = 0L;
252 
253  if ( Mf_starttrack ){
254 
255  (*Mf_starttrack)();
256  }
257  while ( Mf_toberead > 0 ) {
258 
259  Mf_currtime += readvarinum(); /* delta time */
260  if (midifile_error) return;
261 
262  c = egetc(); if (midifile_error) return;
263 
264  if ( sysexcontinue && c != 0xf7 ) {
265  mferror("didn't find expected continuation of a sysex");
266  return;
267  }
268  if ( (c & 0x80) == 0 ) { /* running status? */
269  if ( status == 0 ) {
270  mferror("unexpected running status");
271  return;
272  }
273  running = 1;
274  }
275  else {
276  status = c;
277  running = 0;
278  }
279 
280  needed = chantype[ (status>>4) & 0xf ];
281 
282  if ( needed ) { /* ie. is it a channel message? */
283 
284  if ( running )
285  c1 = c;
286  else {
287  c1 = egetc();
288  if (midifile_error) return;
289  }
290  chanmessage( status, c1, (needed>1) ? egetc() : 0 );
291  if (midifile_error) return;
292  continue;;
293  }
294 
295  switch ( c ) {
296 
297  case 0xff: /* meta event */
298 
299  type = egetc();
300  if (midifile_error) return;
301  /* watch out - Don't combine the next 2 statements */
302  lng = readvarinum();
303  if (midifile_error) return;
304  lookfor = Mf_toberead - lng;
305  msginit();
306 
307  while ( Mf_toberead > lookfor ) {
308  char c = egetc();
309  if (midifile_error) return;
310  msgadd(c);
311  }
312  metaevent(type);
313  break;
314 
315  case 0xf0: /* start of system exclusive */
316 
317  /* watch out - Don't combine the next 2 statements */
318  lng = readvarinum();
319  if (midifile_error) return;
320  lookfor = Mf_toberead - lng;
321  msginit();
322  msgadd(0xf0);
323 
324  while ( Mf_toberead > lookfor ) {
325  c = egetc();
326  if (midifile_error) return;
327  msgadd(c);
328  }
329  if ( c==0xf7 || Mf_nomerge==0 )
330  sysex();
331  else
332  sysexcontinue = 1; /* merge into next msg */
333  break;
334 
335  case 0xf7: /* sysex continuation or arbitrary stuff */
336 
337  /* watch out - Don't combine the next 2 statements */
338  lng = readvarinum();
339  if (midifile_error) return;
340  lookfor = Mf_toberead - lng;
341 
342  if ( ! sysexcontinue )
343  msginit();
344 
345  while ( Mf_toberead > lookfor ) {
346  c = egetc();
347  if (midifile_error) return;
348  msgadd(c);
349  }
350  if ( ! sysexcontinue ) {
351  if ( Mf_arbitrary )
352  (*Mf_arbitrary)(msgleng(),msg());
353  }
354  else if ( c == 0xf7 ) {
355  sysex();
356  sysexcontinue = 0;
357  }
358  break;
359  default:
360 
361  badbyte(c);
362 
363  break;
364  }
365  }
366  if ( Mf_endtrack )
367  (*Mf_endtrack)();
368  return;
369 }

403  case 0x0c:
404  case 0x0d:
405  case 0x0e:
