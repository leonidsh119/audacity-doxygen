79 void find_full_path(const char *filename, char *fullname)
80 {
81  if (full_name(filename)) {
82  strncpy(fullname, filename, STRMAX);
83  fullname[STRMAX - 1] = 0;
84  return;
85  }
86  if (!getcwd(fullname, STRMAX)) {
87  /* something is really wrong. Pretend we found a
88  cwd that will not match anything */
89  goto error;
90  }
91  /* see if we need a separator (probably) */
92  int len = strlen(fullname);
93  if (!file_sep(fullname[len - 1])) {
94  fullname[len++] = '/';
95  if (len >= STRMAX) goto error;
96  }
97  /* append filename to fullname */
98  strncpy(fullname + len, filename, STRMAX - len);
99  fullname[STRMAX - 1] = 0; /* just in case of overflow */
100  /* if windows, replace \ with / to simplify the rest */
101  char *loc = fullname;
102  if (os_pathchar != '/') {
103  while ((loc = strchr(loc, os_pathchar))) {
104  *loc = '/';
105  }
106  }
107  /* strip out .. and . */
108  while ((loc = strstr(fullname, "/.."))) {
109  /* back up to separator */
110  if (loc == fullname) goto error;
111  char *loc2 = loc - 1;
112  while (*loc2 != '/') {
113  loc2--;
114  if (loc2 <= fullname) goto error;
115  }
116  /* now loc2 points to /parent/.., and loc points to /.. */
117  /* copy from beyond /.. to loc2 */
118  memmove(loc2, loc, strlen(loc) + 1);
119  }
120  return;
121  error:
122  strcpy(fullname, "//////");
123  return;
124 }

148 static int in_tree(const char *fullname, char *secure_read_path)
149 {
150  /* fullname is in a tree if a path in secure_read_path
151  is a prefix of fullname
152  Algorithm: extract each path, check for prefix
153  */
154  char path[STRMAX];
155  while (secure_read_path && *secure_read_path) {
156  /* skip over separator */
157  while (*secure_read_path == os_sepchar ||
158  *secure_read_path == ';') secure_read_path++;
159  /* find next directory and copy into path*/
160  path[0] = 0;
161  int i = 0;
162  while (*secure_read_path && (*secure_read_path != os_sepchar &&
163  *secure_read_path != ';')) {
164  path[i++] = *secure_read_path++;

166  path[i] = 0;
167  /* see if directory is a prefix of fullname */
168  if (strstr(fullname, path) == fullname) {
169  return TRUE;
170  }
171  }
