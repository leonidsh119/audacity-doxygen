117 
118  long invalEnd = (sampleEnd - start*rate)/samplesPerPixel +1; //we should cover the end..
119 
120  //if they are both off the cache boundary in the same direction, the cache is missed,
121  //so we are safe, and don't need to track this one.

442 void findCorrection(const std::vector<sampleCount> &oldWhere, int oldLen, int newLen,
443  double t0, double rate, double samplesPerPixel,
444  int &oldX0, double &correction)
445 {
446  // Mitigate the accumulation of location errors
447  // in copies of copies of ... of caches.
448  // Look at the loop that populates "where" below to understand this.
449 
450  // Find the sample position that is the origin in the old cache.
451  const double oldWhere0 = oldWhere[1] - samplesPerPixel;
452  const double oldWhereLast = oldWhere0 + oldLen * samplesPerPixel;
453  // Find the length in samples of the old cache.
454  const double denom = oldWhereLast - oldWhere0;
455 
456  // What sample would go in where[0] with no correction?
457  const double guessWhere0 = t0 * rate;
458 
459  if ( // Skip if old and NEW are disjoint:
460  oldWhereLast <= guessWhere0 ||
461  guessWhere0 + newLen * samplesPerPixel <= oldWhere0 ||
462  // Skip unless denom rounds off to at least 1.
463  denom < 0.5)
464  {
465  // The computation of oldX0 in the other branch
466  // may underflow and the assertion would be violated.
467  oldX0 = oldLen;
468  correction = 0.0;
469  }
470  else
471  {
472  // What integer position in the old cache array does that map to?
473  // (even if it is out of bounds)
474  oldX0 = floor(0.5 + oldLen * (guessWhere0 - oldWhere0) / denom);
475  // What sample count would the old cache have put there?
476  const double where0 = oldWhere0 + double(oldX0) * samplesPerPixel;
477  // What correction is needed to align the NEW cache with the old?
478  const double correction0 = where0 - guessWhere0;
479  correction = std::max(-samplesPerPixel, std::min(samplesPerPixel, correction0));
480  wxASSERT(correction == correction0);
481  }
482 }

505  double pixelsPerSecond, bool &isLoadingOD) const
506 {
507  const bool allocated = (display.where != 0);
508 
509  const int numPixels = display.width;
510 
511  int p0 = 0; // least column requiring computation
512  int p1 = numPixels; // greatest column requiring computation, plus one
513 
514  float *min;
515  float *max;
516  float *rms;
517  int *bl;
518  std::vector<sampleCount> *pWhere;
519 
520  if (allocated) {
521  // assume ownWhere is filled.
522  min = &display.min[0];
523  max = &display.max[0];
524  rms = &display.rms[0];
525  bl = &display.bl[0];
526  pWhere = &display.ownWhere;
527  }
528  else {
529  // Lock the list of invalid regions
530  ODLocker locker(mWaveCacheMutex);
531 
532  const double tstep = 1.0 / pixelsPerSecond;
533  const double samplesPerPixel = mRate * tstep;
534 
535  // Make a tolerant comparison of the pps values in this wise:
536  // accumulated difference of times over the number of pixels is less than
537  // a sample period.
538  const bool ppsMatch = mWaveCache &&
539  (fabs(tstep - 1.0 / mWaveCache->pps) * numPixels < (1.0 / mRate));
540 
541  const bool match =
542  mWaveCache &&
543  ppsMatch &&
544  mWaveCache->len > 0 &&
545  mWaveCache->dirty == mDirty;
546 
547  if (match &&
548  mWaveCache->start == t0 &&
549  mWaveCache->len >= numPixels) {
550  mWaveCache->LoadInvalidRegions(mSequence, true);
551  mWaveCache->ClearInvalidRegions();
552 
553  // Satisfy the request completely from the cache
554  display.min = &mWaveCache->min[0];
555  display.max = &mWaveCache->max[0];
556  display.rms = &mWaveCache->rms[0];
557  display.bl = &mWaveCache->bl[0];
558  display.where = &mWaveCache->where[0];
559  isLoadingOD = mWaveCache->numODPixels > 0;
560  return true;
561  }
562

564  mWaveCache = 0;
565 
566  int oldX0 = 0;
567  double correction = 0.0;
568  int copyBegin = 0, copyEnd = 0;
569  if (match) {
570  findCorrection(oldCache->where, oldCache->len, numPixels,
571  t0, mRate, samplesPerPixel,
572  oldX0, correction);
573  // Remember our first pixel maps to oldX0 in the old cache,
574  // possibly out of bounds.
575  // For what range of pixels can data be copied?
576  copyBegin = std::min(numPixels, std::max(0, -oldX0));
577  copyEnd = std::min(numPixels,
578  copyBegin + oldCache->len - std::max(0, oldX0)
579  );
580  }
581  if (!(copyEnd > copyBegin))
582  oldCache.reset(0);
583 
584  mWaveCache = new WaveCache(numPixels, pixelsPerSecond, mRate, t0, mDirty);
585  min = &mWaveCache->min[0];
586  max = &mWaveCache->max[0];
587  rms = &mWaveCache->rms[0];
588  bl = &mWaveCache->bl[0];
589  pWhere = &mWaveCache->where;
590 
591  fillWhere(*pWhere, numPixels, 0.0, correction,
592  t0, mRate, samplesPerPixel);
593 
594  // The range of pixels we must fetch from the Sequence:
595  p0 = (copyBegin > 0) ? 0 : copyEnd;
596  p1 = (copyEnd >= numPixels) ? copyBegin : numPixels;
597 
598  // Optimization: if the old cache is good and overlaps
599  // with the current one, re-use as much of the cache as
600  // possible
601 
602  if (oldCache) {
603 
604  //TODO: only load inval regions if
605  //necessary. (usually is the case, so no rush.)
606  //also, we should be updating the NEW cache, but here we are patching the old one up.
607  oldCache->LoadInvalidRegions(mSequence, false);
608  oldCache->ClearInvalidRegions();

611  const int length = copyEnd - copyBegin;
612  const size_t sizeFloats = length * sizeof(float);
613  const int srcIdx = copyBegin + oldX0;
614  memcpy(&min[copyBegin], &oldCache->min[srcIdx], sizeFloats);
615  memcpy(&max[copyBegin], &oldCache->max[srcIdx], sizeFloats);
616  memcpy(&rms[copyBegin], &oldCache->rms[srcIdx], sizeFloats);
617  memcpy(&bl[copyBegin], &oldCache->bl[srcIdx], length * sizeof(int));
618  }
619  }
620 
621  if (p1 > p0) {
622  // Cache was not used or did not satisfy the whole request
623  std::vector<sampleCount> &where = *pWhere;
624 
625  /* handle values in the append buffer */
626 
627  int numSamples = mSequence->GetNumSamples();
628  int a;
629 
630  // Not all of the required columns might be in the sequence.
631  // Some might be in the append buffer.
632  for (a = p0; a < p1; ++a) {
633  if (where[a + 1] > numSamples)
634  break;
635  }
636 
637  // Handle the columns that land in the append buffer.
638  //compute the values that are outside the overlap from scratch.
639  if (a < p1) {
640  int i;
641 
642  sampleFormat seqFormat = mSequence->GetSampleFormat();
643  bool didUpdate = false;
644  for(i=a; i<p1; i++) {
645  sampleCount left;
646  left = where[i] - numSamples;
647  sampleCount right;
648  right = where[i + 1] - numSamples;
649 
650  //wxCriticalSectionLocker locker(mAppendCriticalSection);
651 
652  if (left < 0)
653  left = 0;
654  if (right > mAppendBufferLen)
655  right = mAppendBufferLen;
656 
657  if (right > left) {
658  float *b;
659  sampleCount len = right-left;
660  sampleCount j;
661 
662  if (seqFormat == floatSample)
663  b = &((float *)mAppendBuffer.ptr())[left];
664  else {
665  b = new float[len];
666  CopySamples(mAppendBuffer.ptr() + left*SAMPLE_SIZE(seqFormat),
667  seqFormat,
668  (samplePtr)b, floatSample, len);
669  }
670

672  {
673  const float val = b[0];
674  theMax = theMin = val;
675  sumsq = val * val;
676  }
677  for(j=1; j<len; j++) {
678  const float val = b[j];
679  theMax = std::max(theMax, val);
680  theMin = std::min(theMin, val);
681  sumsq += val * val;
682  }
683

685  max[i] = theMax;
686  rms[i] = (float)sqrt(sumsq / len);
687  bl[i] = 1; //for now just fake it.
688

690  delete[] b;
691 
692  didUpdate=true;
693  }
694  }
695 
696  // Shrink the right end of the range to fetch from Sequence
697  if(didUpdate)
698  p1 = a;
699  }
700

702  // from the sequence
703  if (p1 > p0) {

705  &max[p0],
706  &rms[p0],

713  }
714  }

716

718  if (!allocated) {
719  // Now report the results
720  display.min = min;
721  display.max = max;

723  display.bl = bl;
724  display.where = &(*pWhere)[0];

740  (int fftLen, double rate, int frequencyGain, std::vector<float> &gainFactors)
741 {
742  if (frequencyGain > 0) {
743  // Compute a frequency-dependent gain factor
744  // scaled such that 1000 Hz gets a gain of 0dB
745

747  const double factor = ((double)rate / (double)fftLen) / 1000.0;
748

752  gainFactors.push_back(frequencyGain*log10(factor));
753  for (sampleCount x = 1; x < half; x++) {
754  gainFactors.push_back(frequencyGain*log10(factor * x));
755  }
756  }
757 }

782 bool SpecCache::CalculateOneSpectrum
783  (const SpectrogramSettings &settings,
784  WaveTrackCache &waveTrackCache,
785  int xx, sampleCount numSamples,
786  double offset, double rate, double pixelsPerSecond,
787  int lowerBoundX, int upperBoundX,
788  const std::vector<float> &gainFactors,
789  float *scratch)
790 {
791  bool result = false;
792  const bool reassignment =
793  (settings.algorithm == SpectrogramSettings::algReassignment);
794  const int windowSize = settings.windowSize;
795 
796  sampleCount start;
797  if (xx < 0)
798  start = where[0] + xx * (rate / pixelsPerSecond);
799  else if (xx > len)
800  start = where[len] + (xx - len) * (rate / pixelsPerSecond);
801  else
802  start = where[xx];
803 
804  const bool autocorrelation =
805  settings.algorithm == SpectrogramSettings::algPitchEAC;
806  const int zeroPaddingFactor = (autocorrelation ? 1 : settings.zeroPaddingFactor);
807  const int padding = (windowSize * (zeroPaddingFactor - 1)) / 2;
808  const int fftLen = windowSize * zeroPaddingFactor;
809  const int half = fftLen / 2;
810 
811  if (start <= 0 || start >= numSamples) {
812  if (xx >= 0 && xx < len) {
813  // Pixel column is out of bounds of the clip! Should not happen.
814  float *const results = &freq[half * xx];
815  std::fill(results, results + half, 0.0f);
816  }
817  }
818  else {
819  // We can avoid copying memory when ComputeSpectrum is used below
820  bool copy = !autocorrelation || (padding > 0) || reassignment;
821  float *useBuffer = 0;
822  float *adj = scratch + padding;
823 
824  {
825  sampleCount myLen = windowSize;
826  // Take a window of the track centered at this sample.
827  start -= windowSize >> 1;
828  if (start < 0) {
829  // Near the start of the clip, pad left with zeroes as needed.
830  for (sampleCount ii = start; ii < 0; ++ii)
831  *adj++ = 0;
832  myLen += start;
833  start = 0;
834  copy = true;
835  }
836 
837  if (start + myLen > numSamples) {
838  // Near the end of the clip, pad right with zeroes as needed.
839  int newlen = numSamples - start;
840  for (sampleCount ii = newlen; ii < (sampleCount)myLen; ++ii)
841  adj[ii] = 0;
842  myLen = newlen;
843  copy = true;
844  }
845 
846  if (myLen > 0) {
847  useBuffer = (float*)(waveTrackCache.Get(floatSample,
848  floor(0.5 + start + offset * rate), myLen));
849  if (copy)
850  memcpy(adj, useBuffer, myLen * sizeof(float));
851  }
852  }
853 
854  if (copy)
855  useBuffer = scratch;
856 
857 #ifdef EXPERIMENTAL_USE_REALFFTF
858  if (autocorrelation) {
859  float *const results = &freq[half * xx];
860  // This function does not mutate useBuffer
861  ComputeSpectrum(useBuffer, windowSize, windowSize,
862  rate, results,
863  autocorrelation, settings.windowType);
864  }
865  else if (reassignment) {
866  static const double epsilon = 1e-16;
867  const HFFT hFFT = settings.hFFT;
868 
869  float *const scratch2 = scratch + fftLen;
870  std::copy(scratch, scratch2, scratch2);
871 
872  float *const scratch3 = scratch + 2 * fftLen;
873  std::copy(scratch, scratch2, scratch3);
874 
875  {
876  const float *const window = settings.window;
877  for (int ii = 0; ii < fftLen; ++ii)
878  scratch[ii] *= window[ii];
879  RealFFTf(scratch, hFFT);
880  }
881 
882  {
883  const float *const dWindow = settings.dWindow;
884  for (int ii = 0; ii < fftLen; ++ii)
885  scratch2[ii] *= dWindow[ii];
886  RealFFTf(scratch2, hFFT);
887  }
888 
889  {
890  const float *const tWindow = settings.tWindow;
891  for (int ii = 0; ii < fftLen; ++ii)
892  scratch3[ii] *= tWindow[ii];
893  RealFFTf(scratch3, hFFT);
894  }
895 
896  for (int ii = 0; ii < hFFT->Points; ++ii) {
897  const int index = hFFT->BitReversed[ii];
898  const float
899  denomRe = scratch[index],
900  denomIm = ii == 0 ? 0 : scratch[index + 1];
901  const double power = denomRe * denomRe + denomIm * denomIm;
902  if (power < epsilon)
903  // Avoid dividing by near-zero below
904  continue;
905 
906  double freqCorrection;
907  {
908  const double multiplier = -fftLen / (2.0f * M_PI);
909  const float
910  numRe = scratch2[index],
911  numIm = ii == 0 ? 0 : scratch2[index + 1];
912  // Find complex quotient --
913  // Which means, multiply numerator by conjugate of denominator,
914  // then divide by norm squared of denominator --
915  // Then just take its imaginary part.
916  const double
917  quotIm = (-numRe * denomIm + numIm * denomRe) / power;
918  // With appropriate multiplier, that becomes the correction of
919  // the frequency bin.
920  freqCorrection = multiplier * quotIm;
921  }
922 
923  const int bin = int(ii + freqCorrection + 0.5f);
924  if (bin >= 0 && bin < hFFT->Points) {
925  double timeCorrection;
926  {
927  const float
928  numRe = scratch3[index],
929  numIm = ii == 0 ? 0 : scratch3[index + 1];
930  // Find another complex quotient --
931  // Then just take its real part.
932  // The result has sample interval as unit.
933  timeCorrection =
934  (numRe * denomRe + numIm * denomIm) / power;
935  }
936 
937  int correctedX = (floor(0.5 + xx + timeCorrection * pixelsPerSecond / rate));
938  if (correctedX >= lowerBoundX && correctedX < upperBoundX)
939  result = true,
940  freq[half * correctedX + bin] += power;
941  }
942  }
943  }
944  else {
945  float *const results = &freq[half * xx];
946 
947  // Do the FFT. Note that useBuffer is multiplied by the window,
948  // and the window is initialized with leading and trailing zeroes
949  // when there is padding. Therefore we did not need to reinitialize
950  // the part of useBuffer in the padding zones.
951 
952  // This function mutates useBuffer
953  ComputeSpectrumUsingRealFFTf
954  (useBuffer, settings.hFFT, settings.window, fftLen, results);
955  if (!gainFactors.empty()) {
956  // Apply a frequency-dependant gain factor
957  for (int ii = 0; ii < half; ++ii)
958  results[ii] += gainFactors[ii];
959  }
960  }
961 #else // EXPERIMENTAL_USE_REALFFTF
962  // This function does not mutate scratch
963  ComputeSpectrum(scratch, windowSize, windowSize,
964  rate, results,
965  autocorrelation, settings.windowType);
966 #endif // EXPERIMENTAL_USE_REALFFTF
967  }
968  return result;
969 }

971 void SpecCache::Populate
972  (const SpectrogramSettings &settings, WaveTrackCache &waveTrackCache,
973  int copyBegin, int copyEnd, int numPixels,
974  sampleCount numSamples,
975  double offset, double rate, double pixelsPerSecond)
976 {
977 #ifdef EXPERIMENTAL_USE_REALFFTF
978  settings.CacheWindows();
979 #endif
980 
981  const int &frequencyGain = settings.frequencyGain;
982  const int &windowSize = settings.windowSize;
983  const bool autocorrelation =
984  settings.algorithm == SpectrogramSettings::algPitchEAC;
985  const bool reassignment =
986  settings.algorithm == SpectrogramSettings::algReassignment;
987 #ifdef EXPERIMENTAL_ZERO_PADDED_SPECTROGRAMS
988  const int &zeroPaddingFactor = autocorrelation ? 1 : settings.zeroPaddingFactor;
989 #else
990  const int zeroPaddingFactor = 1;
991 #endif
992 
993  // FFT length may be longer than the window of samples that affect results
994  // because of zero padding done for increased frequency resolution
995  const int fftLen = windowSize * zeroPaddingFactor;
996  const int half = fftLen / 2;
997 
998  const size_t bufferSize = fftLen;
999 
1000  std::vector<float> buffer(reassignment ? 3 * bufferSize : bufferSize);
1001 
1002  std::vector<float> gainFactors;
1003  if (!autocorrelation)
1004  ComputeSpectrogramGainFactors(fftLen, rate, frequencyGain, gainFactors);
1005 
1006  // Loop over the ranges before and after the copied portion and compute anew.
1007  // One of the ranges may be empty.
1008  for (int jj = 0; jj < 2; ++jj) {
1009  const int lowerBoundX = jj == 0 ? 0 : copyEnd;
1010  const int upperBoundX = jj == 0 ? copyBegin : numPixels;
1011  for (sampleCount xx = lowerBoundX; xx < upperBoundX; ++xx)
1012  CalculateOneSpectrum(
1013  settings, waveTrackCache, xx, numSamples,
1014  offset, rate, pixelsPerSecond,
1015  lowerBoundX, upperBoundX,
1016  gainFactors, &buffer[0]);
1017 
1018  if (reassignment) {
1019  // Need to look beyond the edges of the range to accumulate more
1020  // time reassignments.
1021  // I'm not sure what's a good stopping criterion?
1022  sampleCount xx = lowerBoundX;
1023  const double pixelsPerSample = pixelsPerSecond / rate;
1024  const int limit = std::min(int(0.5 + fftLen * pixelsPerSample), 100);
1025  for (int ii = 0; ii < limit; ++ii)
1026  {
1027  const bool result =
1028  CalculateOneSpectrum(
1029  settings, waveTrackCache, --xx, numSamples,
1030  offset, rate, pixelsPerSecond,
1031  lowerBoundX, upperBoundX,
1032  gainFactors, &buffer[0]);
1033  if (!result)
1034  break;
1035  }
1036 
1037  xx = upperBoundX;
1038  for (int ii = 0; ii < limit; ++ii)
1039  {
1040  const bool result =
1041  CalculateOneSpectrum(
1042  settings, waveTrackCache, xx++, numSamples,
1043  offset, rate, pixelsPerSecond,
1044  lowerBoundX, upperBoundX,
1045  gainFactors, &buffer[0]);
1046  if (!result)
1047  break;
1048  }
1049 
1050  // Now Convert to dB terms. Do this only after accumulating
1051  // power values, which may cross columns with the time correction.
1052  for (sampleCount xx = lowerBoundX; xx < upperBoundX; ++xx) {
1053  float *const results = &freq[half * xx];
1054  const HFFT hFFT = settings.hFFT;
1055  for (int ii = 0; ii < hFFT->Points; ++ii) {
1056  float &power = results[ii];
1057  if (power <= 0)
1058  power = -160.0;
1059  else
1060  power = 10.0*log10f(power);
1061  }
1062  if (!gainFactors.empty()) {
1063  // Apply a frequency-dependant gain factor
1064  for (int ii = 0; ii < half; ++ii)
1065  results[ii] += gainFactors[ii];
1066  }
1067  }
1068  }
1069  }
1070 }

1073  const float *& spectrogram, const sampleCount *& where,
1074  int numPixels,

1076 {

1078  const SpectrogramSettings &settings = track->GetSpectrogramSettings();
1079  const bool autocorrelation =
1080  settings.algorithm == SpectrogramSettings::algPitchEAC;
1081  const int &frequencyGain = settings.frequencyGain;
1082  const int &windowSize = settings.windowSize;
1083  const int &windowType = settings.windowType;
1084 #ifdef EXPERIMENTAL_ZERO_PADDED_SPECTROGRAMS
1085  const int &zeroPaddingFactor = autocorrelation ? 1 : settings.zeroPaddingFactor;
1086 #else
1087  const int zeroPaddingFactor = 1;
1088 #endif
1089 
1090  // FFT length may be longer than the window of samples that affect results
1091  // because of zero padding done for increased frequency resolution
1092  const int fftLen = windowSize * zeroPaddingFactor;
1093  const int half = fftLen / 2;

1096  mSpecCache &&
1097  mSpecCache->len > 0 &&
1098  mSpecCache->Matches
1099  (mDirty, pixelsPerSecond, settings, mRate);
1100 
1101  if (match &&
1102  mSpecCache->start == t0 &&
1103  mSpecCache->len >= numPixels) {
1104  spectrogram = &mSpecCache->freq[0];
1105  where = &mSpecCache->where[0];
1106  return false; //hit cache completely
1107  }
1108 
1109  if (settings.algorithm == SpectrogramSettings::algReassignment)
1110  // Caching is not implemented for reassignment, unless for
1111  // a complete hit, because of the complications of time reassignment
1112  match = false;
1113 
1114  std::unique_ptr<SpecCache> oldCache(mSpecCache);
1115  mSpecCache = 0;
1116 
1117  const double tstep = 1.0 / pixelsPerSecond;
1118  const double samplesPerPixel = mRate * tstep;
1119

1121  double correction = 0.0;
1122 
1123  int copyBegin = 0, copyEnd = 0;
1124  if (match) {
1125  findCorrection(oldCache->where, oldCache->len, numPixels,
1126  t0, mRate, samplesPerPixel,
1127  oldX0, correction);
1128  // Remember our first pixel maps to oldX0 in the old cache,
1129  // possibly out of bounds.
1130  // For what range of pixels can data be copied?
1131  copyBegin = std::min(numPixels, std::max(0, -oldX0));
1132  copyEnd = std::min(numPixels,
1133  copyBegin + oldCache->len - std::max(0, oldX0)
1134  );
1135  }
1136 
1137  if (!(copyEnd > copyBegin))
1138  oldCache.reset(0);
1139 
1140  mSpecCache = new SpecCache(
1141  numPixels, settings.algorithm, pixelsPerSecond, t0,
1142  windowType, windowSize, zeroPaddingFactor, frequencyGain);
1143 
1144  // purposely offset the display 1/2 sample to the left (as compared
1145  // to waveform display) to properly center response of the FFT
1146  fillWhere(mSpecCache->where, numPixels, 0.5, correction,
1147  t0, mRate, samplesPerPixel);
1148 
1149  // Optimization: if the old cache is good and overlaps
1150  // with the current one, re-use as much of the cache as
1151  // possible
1152  if (oldCache) {
1153  memcpy(&mSpecCache->freq[half * copyBegin],
1154  &oldCache->freq[half * (copyBegin + oldX0)],
1155  half * (copyEnd - copyBegin) * sizeof(float));
1156  }
1157 
1158  mSpecCache->Populate
1159  (settings, waveTrackCache, copyBegin, copyEnd, numPixels,
1160  mSequence->GetNumSamples(),
1161  mOffset, mRate, pixelsPerSecond);
1162 
1163  mSpecCache->dirty = mDirty;
1164  spectrogram = &mSpecCache->freq[0];
1165  where = &mSpecCache->where[0];
1166  return true;
