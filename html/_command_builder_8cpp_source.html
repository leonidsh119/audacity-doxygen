90  const wxString &cmdParamsArg)
91 {
92  // Stage 1: create a Command object of the right type
93 
94  CommandMessageTarget *scriptOutput = ScriptCommandRelay::GetResponseTarget();
95  CommandOutputTarget *output
96  = new CommandOutputTarget(new NullProgressTarget(),
97  scriptOutput,
98  scriptOutput);
99 
100  CommandType *factory = CommandDirectory::Get()->LookUp(cmdName);
101 
102  if (factory == NULL)
103  {
104  // Fall back to hoping the Batch Command system can handle it
105  CommandType *type = CommandDirectory::Get()->LookUp(wxT("BatchCommand"));
106  wxASSERT(type != NULL);
107  mCommand = type->Create(output);
108  mCommand->SetParameter(wxT("CommandName"), cmdName);
109  mCommand->SetParameter(wxT("ParamString"), cmdParamsArg);
110  Success(new ApplyAndSendResponse(mCommand));
111  return;
112  }
113 
114  CommandSignature &signature = factory->GetSignature();
115  mCommand = factory->Create(output);
116 
117  // Stage 2: set the parameters
118 
119  ShuttleCli shuttle;
120  shuttle.mParams = cmdParamsArg;
121  shuttle.mbStoreInClient = true;
122 
123  ParamValueMap::const_iterator iter;
124  ParamValueMap params = signature.GetDefaults();
125 
126  for (iter = params.begin(); iter != params.end(); ++iter)
127  {
128  wxString paramString;
129  if (shuttle.TransferString(iter->first, paramString, wxT("")))
130  {
131  if (!mCommand->SetParameter(iter->first, paramString))
132  {
133  Failure();
134  return;
135  }
136  }
137  }
138 
139  // Check for unrecognised parameters
140 
141  wxString cmdParams(cmdParamsArg);
142

144  {
145  cmdParams.Trim(true);
146  cmdParams.Trim(false);
147  int splitAt = cmdParams.Find(wxT('='));
148  if (splitAt < 0 && cmdParams != wxEmptyString)

154  if (params.find(paramName) == params.end())

157  return;

171 void CommandBuilder::BuildCommand(const wxString &cmdStringArg)
172 {
173  wxString cmdString(cmdStringArg);
174 
175  // Find the command name terminator... If there is more than one word and
176  // no terminator, the command is badly formed
177  cmdString.Trim(true); cmdString.Trim(false);
178  int splitAt = cmdString.Find(wxT(':'));
179  if (splitAt < 0 && cmdString.Find(wxT(' ')) >= 0) {
180  mError = wxT("Command is missing ':'");
181  ScriptCommandRelay::SendResponse(wxT("\n"));
182  mValid = false;
183  return;
184  }
185 
186  wxString cmdName = cmdString.Left(splitAt);
187  wxString cmdParams = cmdString.Mid(splitAt+1);
188  cmdName.Trim(true);
189  cmdParams.Trim(false);
190 
191  BuildCommand(cmdName, cmdParams);
