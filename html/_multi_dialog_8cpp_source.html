159 int ShowMultiDialog(const wxString &message,
160  const wxString &title,
161  const wxChar **buttons, const wxString &boxMsg, bool log)
162 {
163  wxWindow * pParent = wxGetApp().GetTopWindow();
164 
165  // We want a parent we can display over, so don't make it a parent if top
166  // window is a STAY_ON_TOP.
167  if (pParent) {
168  if ((pParent->GetWindowStyle() & wxSTAY_ON_TOP) == wxSTAY_ON_TOP)
169  pParent = NULL;
170  }
171  MultiDialog dlog(pParent,
172  message, title, buttons, boxMsg, log);
173  // If dialog does not have a parent, cannot be centred on it.
174  if (pParent != NULL)
175  dlog.CentreOnParent();
176  else {
177  dlog.CenterOnScreen();
178  // and after centring move the dialog left by the size of the dialog.
179  // Likely to help if we have the splash screen visible, or if
180  // we're spanning two equally sized monitors.
181  // Unlikely to make things worse.
182  wxSize Size = dlog.GetSize();
183  Size.SetHeight( 10 );
184  wxPoint Pos = dlog.GetPosition() -Size;
185  dlog.Move(Pos);
186  }
187  return dlog.ShowModal();
188 }
