50 /* xlinit - xlisp initialization routine */
51 void xlinit(void)
52 {
53  /* initialize xlisp (must be in this order) */
54  xlminit(); /* initialize xldmem.c */
55  xldinit(); /* initialize xldbug.c */
56 
57  /* finish initializing */
58 #ifdef SAVERESTORE
59  if (!xlirestore("xlisp.wks"))
60 #endif
61  {
62  initwks();
63  }
64  /* note: localinit creates type descriptors, but these
65  may be necessary for xlirestore to work. On the other
66  hand, localinit creates atoms, so it needs obarray to
67  be in place, and obarray is created by initwks. We may
68  have a circular dependency to break before xlirestore
69  will work
70  */
71  localinit(); /* initialize lisp extensions */
72 }

74 /* initwks - build an initial workspace */
75 LOCAL void initwks(void)
76 {
77  FUNDEF *p;
78  int i;
79 
80  xlsinit(); /* initialize xlsym.c */
81  xlsymbols();/* enter all symbols used by the interpreter */
82  xlrinit(); /* initialize xlread.c */
83  xloinit(); /* initialize xlobj.c */
84 
85  /* setup defaults */
86  setvalue(s_evalhook,NIL); /* no evalhook function */
87  setvalue(s_applyhook,NIL); /* no applyhook function */
88  setvalue(s_tracelist,NIL); /* no functions being traced */
89  setvalue(s_tracenable,NIL); /* traceback disabled */
90  setvalue(s_tlimit,NIL); /* trace limit infinite */
91  setvalue(s_breakenable,NIL); /* don't enter break loop on errors */
92  setvalue(s_loadingfiles,NIL); /* not loading any files initially */
93  setvalue(s_profile,NIL); /* don't do profiling */
94  setvalue(s_gcflag,NIL); /* don't show gc information */
95  setvalue(s_gchook,NIL); /* no gc hook active */
96  setvalue(s_ifmt,cvstring(IFMT)); /* integer print format */
97  setvalue(s_ffmt,cvstring("%g")); /* float print format */
98  setvalue(s_printcase,k_upcase); /* upper case output of symbols */
99 
100  /* install the built-in functions and special forms */
101  for (i = 0, p = funtab; p->fd_subr != NULL; ++i, ++p)
102  if (p->fd_name)
103  xlsubr(p->fd_name,p->fd_type,p->fd_subr,i);
104 
105  /* add some synonyms */
106  setfunction(xlenter("NOT"),getfunction(xlenter("NULL")));
107  setfunction(xlenter("FIRST"),getfunction(xlenter("CAR")));
108  setfunction(xlenter("SECOND"),getfunction(xlenter("CADR")));
109  setfunction(xlenter("THIRD"),getfunction(xlenter("CADDR")));
110  setfunction(xlenter("FOURTH"),getfunction(xlenter("CADDDR")));
111  setfunction(xlenter("REST"),getfunction(xlenter("CDR")));
112 }
