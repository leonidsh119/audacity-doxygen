256  mGrid->SetCellValue(i, Col_Etime,

260  // and write event code to update them from controls
261  }

323  newTrack->SetName(name);

677 void LabelDialog::OnCellChange(wxGridEvent &event)
678 {

681 
682  // Guard against recursion which can happen when a change to the "NEW label" row
683  // is made. When InsertRow() is done in TransferDataToWindow(), checks are made
684  // within wxGrid to see if the edit control is active and since it hasn't yet
685  // been marked inactive on the first time through here, we get entered again.

703  break;
704

707  break;
708 
709  case Col_Etime:
710  OnChangeEtime(event, row, rd);
711  break;
712  }
713 
714  // Done...no need for protection anymore
715  guard = false;
716
