43 wxImage *ChangeImageColour(wxImage * srcImage,
44  wxColour & srcColour,
45  wxColour & dstColour)
46 {
47  // This function takes a source image, which it assumes to
48  // be grayscale, and smoothly changes the overall color
49  // to the specified color, and returns the result as a
50  // NEW image. This works well for grayscale 3D images.
51  // Audacity uses this routines to make the buttons
52  // (skip-start, play, stop, record, skip-end) adapt to
53  // the color scheme of the user.
54 
55  unsigned char *src = srcImage->GetData();
56  int width = srcImage->GetWidth();
57  int height = srcImage->GetHeight();
58 
59  wxImage * dstImage = new wxImage(width, height);
60  unsigned char *dst = dstImage->GetData();
61 
62 
63  //Get the source color
64  int srcVal[3], srcOpp[3];
65  srcVal[0] = srcColour.Red();
66  srcVal[1] = srcColour.Green();
67  srcVal[2] = srcColour.Blue();
68 
69  int dstVal[3], dstOpp[3];
70  dstVal[0] = dstColour.Red();
71  dstVal[1] = dstColour.Green();
72  dstVal[2] = dstColour.Blue();
73 
74  int i;
75  for (i = 0; i < 3; i++) {
76  srcOpp[i] = 256 - srcVal[i]; // avoid zero!
77  dstOpp[i] = 255 - dstVal[i];
78  }
79 
80  int c = 0;
81  for (i = 0; i < width * height * 3; i++) {
82  int s = (int) *src;
83 
84  if (s >= srcVal[c])
85  *dst++ = dstVal[c] + dstOpp[c] * (s - srcVal[c]) / srcOpp[c];
86 
87  else
88  *dst++ = dstVal[c] * s / srcVal[c];
89  src++;
90  c = (c + 1) % 3;
91  }
92 
93  return dstImage;
94 }

101 wxImage *OverlayImage(wxImage * background, wxImage * foreground,
102  wxImage * mask, int xoff, int yoff)
103 {
104  unsigned char *bg = background->GetData();
105  unsigned char *fg = foreground->GetData();
106  unsigned char *mk = mask->GetData();
107 
108  int bgWidth = background->GetWidth();
109  int bgHeight = background->GetHeight();
110  int fgWidth = foreground->GetWidth();
111  int fgHeight = foreground->GetHeight();
112  int mkWidth = mask->GetWidth();
113  int mkHeight = mask->GetHeight();
114 
115 
116  //Now, determine the dimensions of the images to be masked together
117  //on top of the background. This should be equal to the area of the
118  //smaller of the foreground and the mask, as long as it is
119  //within the area of the background, given the offset.
120 
121  //Make sure the foreground size is no bigger than the mask
122  int wCutoff = (fgWidth < mkWidth) ? fgWidth : mkWidth;
123  int hCutoff = (fgHeight < mkHeight) ? fgHeight : mkHeight;
124 
125 
126  // If the masked foreground + offset is bigger than the background, masking
127  // should only occur within these bounds of the foreground image
128  wCutoff = (bgWidth - xoff > wCutoff) ? wCutoff : bgWidth - xoff;
129  hCutoff = (bgHeight - yoff > hCutoff) ? hCutoff : bgHeight - yoff;
130 
131 
132  //Make a NEW image the size of the background
133  wxImage * dstImage = new wxImage(bgWidth, bgHeight);
134  unsigned char *dst = dstImage->GetData();
135  memcpy(dst, bg, bgWidth * bgHeight * 3);
136 
137 
138  // Go through the foreground image bit by bit and mask it on to the
139  // background, at an offset of xoff,yoff.
140  // BUT...Don't go beyond the size of the background image,
141  // the foreground image, or the mask
142  int x, y;
143  for (y = 0; y < hCutoff; y++) {
144 
145  unsigned char *bkp = bg + 3 * ((y + yoff) * bgWidth + xoff);
146  unsigned char *dstp = dst + 3 * ((y + yoff) * bgWidth + xoff);
147 
148  for (x = 0; x < wCutoff; x++) {
149 
150  int value = mk[3 * (y * mkWidth + x)];
151  int opp = 255 - value;
152 
153  for (int c = 0; c < 3; c++)
154  dstp[x * 3 + c] =
155  ((bkp[x * 3 + c] * opp) +
156  (fg[3 * (y * fgWidth + x) + c] * value)) / 255;
157  }
158  }
159  return dstImage;
