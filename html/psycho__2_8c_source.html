154  for (i = 0; i < BLKSIZE; i++)

156  /* reset states used in unpredictability measure */

162  mem->lthr[1][i] = 60802371420160.0;
163  }
164  /*****************************************************************************
165  * Initialization: Compute the following constants for use later *
166  * partition[HBLKSIZE] = the partition number associated with each *
167  * frequency line *
168  * cbval[CBANDS] = the center (average) bark value of each *
169  * partition *
170  * numlines[CBANDS] = the number of frequency lines in each partition *
171  * tmn[CBANDS] = tone masking noise *
172  *****************************************************************************/
173  /* compute fft frequency multiplicand */
174  freq_mult = (FLOAT) sfreq / (FLOAT) BLKSIZE;
175 
176  /* calculate fft frequency, then bval of each line (use fthr[] as tmp storage) */
177  for (i = 0; i < HBLKSIZE; i++) {
178  temp1 = i * freq_mult;
179  j = 1;
180  while (temp1 > crit_band[j])
181  j++;
182  fthr[i] = j - 1 + (temp1 - crit_band[j - 1]) / (crit_band[j] - crit_band[j - 1]);
183  }
184  partition[0] = 0;
185  /* temp2 is the counter of the number of frequency lines in each partition */
186  itemp2 = 1;
187  cbval[0] = fthr[0];
188  bval_lo = fthr[0];
189  for (i = 1; i < HBLKSIZE; i++) {
190  if ((fthr[i] - bval_lo) > 0.33) {
191  partition[i] = partition[i - 1] + 1;
192  cbval[partition[i - 1]] = cbval[partition[i - 1]] / itemp2;
193  cbval[partition[i]] = fthr[i];
194  bval_lo = fthr[i];
195  numlines[partition[i - 1]] = itemp2;
196  itemp2 = 1;
197  } else {
198  partition[i] = partition[i - 1];
199  cbval[partition[i]] += fthr[i];
200  itemp2++;
201  }
202  }
203  numlines[partition[i - 1]] = itemp2;
204  cbval[partition[i - 1]] = cbval[partition[i - 1]] / itemp2;
205 
206  /************************************************************************
207  * Now compute the spreading function, s[j][i], the value of the spread-*
208  * ing function, centered at band j, for band i, store for later use *
209  ************************************************************************/
210  for (j = 0; j < CBANDS; j++) {
211  for (i = 0; i < CBANDS; i++) {
212  temp1 = (cbval[i] - cbval[j]) * 1.05;
213  if (temp1 >= 0.5 && temp1 <= 2.5) {
214  ftemp2 = temp1 - 0.5;
215  ftemp2 = 8.0 * (ftemp2 * ftemp2 - 2.0 * ftemp2);
216  } else
217  ftemp2 = 0.0;
218  temp1 += 0.474;
219  temp3 = 15.811389 + 7.5 * temp1 - 17.5 * sqrt((FLOAT) (1.0 + temp1 * temp1));
220  if (temp3 <= -100)
221  s[i][j] = 0;
222  else {
223  temp3 = (ftemp2 + temp3) * LN_TO_LOG10;
224  s[i][j] = exp(temp3);
225  }
226  }
227  }
228 
229  /* Calculate Tone Masking Noise values */
230  for (j = 0; j < CBANDS; j++) {
231  temp1 = 15.5 + cbval[j];
232  tmn[j] = (temp1 > 24.5) ? temp1 : 24.5;
233  /* Calculate normalization factors for the net spreading functions */
234  rnorm[j] = 0;
235  for (i = 0; i < CBANDS; i++) {
236  rnorm[j] += s[j][i];
237  }
238  }
239 
240  if (glopts->verbosity > 5) {
241  /* Dump All the Values to stderr and exit */
242  int wlow, whigh = 0;
243  fprintf(stderr, "psy model 2 init\n");
244  fprintf(stderr, "index \tnlines \twlow \twhigh \tbval \tminval \ttmn\n");
245  for (i = 0; i < CBANDS; i++) {
246  wlow = whigh + 1;
247  whigh = wlow + numlines[i] - 1;
248  fprintf(stderr, "%i \t%i \t%i \t%i \t%5.2f \t%4.2f \t%4.2f\n", i + 1, numlines[i], wlow,
249  whigh, cbval[i], bmax[(int) (cbval[i] + 0.5)], tmn[i]);
250  }
251  }
252 
253  return (mem);

262  FLOAT r_prime, phi_prime;
263  FLOAT minthres, sum_energy;
264  FLOAT tb, temp1, temp2, temp3;
265  FLOAT *grouped_c, *grouped_e;

284

301  window = mem->window;

303  thr = mem->thr;
304  c = mem->c;
305 
306  snrtmp[0] = mem->snrtmp[0];
307  snrtmp[1] = mem->snrtmp[1];
308 
309  numlines = mem->numlines;
310  partition = mem->partition;
311  tmn = mem->tmn;
312  s = mem->s;
313  lthr = mem->lthr;
314  r = mem->r;

318  }

320 
321  for (ch = 0; ch < nch; ch++) {
322  for (i = 0; i < 2; i++) {
323  /*****************************************************************************
324  * Net offset is 480 samples (1056-576) for layer 2; this is because one must*
325  * stagger input data by 256 samples to synchronize psychoacoustic model with*
326  * filter bank outputs, then stagger so that center of 1024 FFT window lines *
327  * up with center of 576 "new" audio samples. *
328 
329  flush = 384*3.0/2.0; = 576
330  syncsize = 1056;
331  sync_flush = syncsize - flush; 480
332  BLKSIZE = 1024
333  *****************************************************************************/
334  {
335  short int *bufferp = buffer[ch];
336  for (j = 0; j < 480; j++) {
337  savebuf[ch][j] = savebuf[ch][j + mem->flush];
338  wsamp_r[j] = window[j] * ((FLOAT) savebuf[ch][j]);
339  }
340  for (; j < 1024; j++) {
341  savebuf[ch][j] = *bufferp++;
342  wsamp_r[j] = window[j] * ((FLOAT) savebuf[ch][j]);
343  }
344  for (; j < 1056; j++)
345  savebuf[ch][j] = *bufferp++;
346  }
347 
349  psycho_2_fft(wsamp_r, energy, phi);
350  /*****************************************************************************
351  * calculate the unpredictability measure, given energy[f] and phi[f] *
352  *****************************************************************************/
353  /* only update data "age" pointers after you are done with both channels */
354  /* for layer 1 computations, for the layer 2 FLOAT computations, the pointers */
355  /* are reset automatically on the second pass */
356  {
357  if (mem->new == 0) {
358  mem->new = 1;
359  mem->oldest = 1;
360  } else {
361  mem->new = 0;
362  mem->oldest = 0;
363  }
364  if (mem->old == 0)
365  mem->old = 1;
366  else
367  mem->old = 0;
368 
369  new = mem->new;
370  old = mem->old;
371  oldest = mem->oldest;
372  }
373 
374 
375  for (j = 0; j < HBLKSIZE; j++) {
376  r_prime = 2.0 * r[ch][old][j] - r[ch][oldest][j];
377  phi_prime = 2.0 * phi_sav[ch][old][j] - phi_sav[ch][oldest][j];
378  r[ch][new][j] = sqrt((FLOAT) energy[j]);
379  phi_sav[ch][new][j] = phi[j];
380 #ifdef SINCOS
381  {
382  // 12% faster
383  // #warning "Use __sincos"
384  FLOAT sphi, cphi, sprime, cprime;
385  __sincos((FLOAT) phi[j], &sphi, &cphi);
386  __sincos((FLOAT) phi_prime, &sprime, &cprime);
387  temp1 = r[chn][new][j] * cphi - r_prime * cprime;
388  temp2 = r[chn][new][j] * sphi - r_prime * sprime;
389  }
390 #else
391  temp1 = r[ch][new][j] * cos((FLOAT) phi[j]) - r_prime * cos((FLOAT) phi_prime);
392  temp2 = r[ch][new][j] * sin((FLOAT) phi[j]) - r_prime * sin((FLOAT) phi_prime);
393 #endif
394 
395  temp3 = r[ch][new][j] + fabs((FLOAT) r_prime);
396  if (temp3 != 0)
397  c[j] = sqrt(temp1 * temp1 + temp2 * temp2) / temp3;
398  else
399  c[j] = 0;
400  }
401  /*****************************************************************************
402  * Calculate the grouped, energy-weighted, unpredictability measure, *
403  * grouped_c[], and the grouped energy. grouped_e[] *
404  *****************************************************************************/
405 
406  for (j = 1; j < CBANDS; j++) {
407  grouped_e[j] = 0;
408  grouped_c[j] = 0;
409  }
410  grouped_e[0] = energy[0];
411  grouped_c[0] = energy[0] * c[0];
412  for (j = 1; j < HBLKSIZE; j++) {
413  grouped_e[partition[j]] += energy[j];
414  grouped_c[partition[j]] += energy[j] * c[j];
415  }
416 
417  /*****************************************************************************
418  * convolve the grouped energy-weighted unpredictability measure *
419  * and the grouped energy with the spreading function, s[j][k] *
420  *****************************************************************************/
421  for (j = 0; j < CBANDS; j++) {
422  ecb[j] = 0;
423  cb[j] = 0;
424  for (k = 0; k < CBANDS; k++) {
425  if (s[j][k] != 0.0) {
426  ecb[j] += s[j][k] * grouped_e[k];
427  cb[j] += s[j][k] * grouped_c[k];
428  }
429  }
430  if (ecb[j] != 0)
431  cb[j] = cb[j] / ecb[j];
432  else
433  cb[j] = 0;
434  }
435 
436  /*****************************************************************************
437  * Calculate the required SNR for each of the frequency partitions *
438  * this whole section can be accomplished by a table lookup *
439  *****************************************************************************/
440  for (j = 0; j < CBANDS; j++) {
441  if (cb[j] < .05)
442  cb[j] = 0.05;
443  else if (cb[j] > .5)
444  cb[j] = 0.5;
445  tb = -0.434294482 * log((FLOAT) cb[j]) - 0.301029996;
446  cb[j] = tb;
447  bc[j] = tmn[j] * tb + nmt * (1.0 - tb);
448  k = (int) (cbval[j] + 0.5);
449  bc[j] = (bc[j] > bmax[k]) ? bc[j] : bmax[k];
450  bc[j] = exp((FLOAT) - bc[j] * LN_TO_LOG10);
451  }
452 
453  /*****************************************************************************
454  * Calculate the permissible noise energy level in each of the frequency *
455  * partitions. Include absolute threshold and pre-echo controls *
456  * this whole section can be accomplished by a table lookup *
457  *****************************************************************************/
458  for (j = 0; j < CBANDS; j++)
459  if (rnorm[j] && numlines[j])
460  nb[j] = ecb[j] * bc[j] / (rnorm[j] * numlines[j]);
461  else
462  nb[j] = 0;
463  for (j = 0; j < HBLKSIZE; j++) {
464  /* temp1 is the preliminary threshold */
465  temp1 = nb[partition[j]];
466  temp1 = (temp1 > absthr[j]) ? temp1 : absthr[j];
467 #ifdef LAYERI
468  /* do not use pre-echo control for layer 2 because it may do bad things to the */
469  /* MUSICAM bit allocation algorithm */
470  if (lay == 1) {
471  fthr[j] = (temp1 < lthr[ch][j]) ? temp1 : lthr[ch][j];
472  temp2 = temp1 * 0.00316;
473  fthr[j] = (temp2 > fthr[j]) ? temp2 : fthr[j];
474  } else
475  fthr[j] = temp1;
476  lthr[ch][j] = LXMIN * temp1;
477 #else
478  fthr[j] = temp1;
479  lthr[ch][j] = LXMIN * temp1;
480 #endif
481  }
482 
483  /*****************************************************************************
484  * Translate the 512 threshold values to the 32 filter bands of the coder *
485  *****************************************************************************/
486  for (j = 0; j < 193; j += 16) {
487  minthres = 60802371420160.0;
488  sum_energy = 0.0;
489  for (k = 0; k < 17; k++) {
490  if (minthres > fthr[j + k])
491  minthres = fthr[j + k];
492  sum_energy += energy[j + k];
493  }
494  snrtmp[i][j / 16] = sum_energy / (minthres * 17.0);
495  snrtmp[i][j / 16] = 4.342944819 * log((FLOAT) snrtmp[i][j / 16]);
496  }
497  for (j = 208; j < (HBLKSIZE - 1); j += 16) {
498  minthres = 0.0;
499  sum_energy = 0.0;
500  for (k = 0; k < 17; k++) {
501  minthres += fthr[j + k];
502  sum_energy += energy[j + k];
503  }
504  snrtmp[i][j / 16] = sum_energy / minthres;
505  snrtmp[i][j / 16] = 4.342944819 * log((FLOAT) snrtmp[i][j / 16]);
506  }
507  /*****************************************************************************
508  * End of Psychoacuostic calculation loop *
509  *****************************************************************************/
510  }
511  for (i = 0; i < 32; i++) {
512  smr[ch][i] = (snrtmp[0][i] > snrtmp[1][i]) ? snrtmp[0][i] : snrtmp[1][i];
513  }
514 
515  } // next channel
