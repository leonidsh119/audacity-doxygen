104  for (Track *t = iter.First(); t; t = iter.Next()) {
105  if (t->GetName() == name) {

109  }
110

112  addedTrack = (AddAnalysisTrack(name)), lt = addedTrack->get();
113  else

115 
116  int count = 0;
117 
118  // JC: Only process selected tracks.
119  SelectedTrackListOfKindIterator waves(Track::Wave, mTracks);
120  WaveTrack *t = (WaveTrack *) waves.First();
121  while (t) {
122  double trackStart = t->GetStartTime();
123  double trackEnd = t->GetEndTime();
124  double t0 = mT0 < trackStart ? trackStart : mT0;
125  double t1 = mT1 > trackEnd ? trackEnd : mT1;
126 
127  if (t1 > t0) {
128  sampleCount start = t->TimeToLongSamples(t0);
129  sampleCount end = t->TimeToLongSamples(t1);
130  sampleCount len = (sampleCount)(end - start);
131 
132  if (!ProcessOne(lt, count, t, start, len)) {
133  return false;
134  }
135  }
136 
137  count++;
138  t = (WaveTrack *) waves.Next();
139  }
140 
141  // No cancellation, so commit the addition of the track.
142  if (addedTrack)
143  addedTrack->Commit();
144  if (modifiedTrack)
145  modifiedTrack->Commit();
