117 void VSTControl::CreateCocoa()
118 {
119  if ((mLink->callDispatcher(effCanDo, 0, 0, (void *) "hasCockosViewAsConfig", 0.0) & 0xffff0000) != 0xbeef0000)
120  {
121  return;
122  }
123 
124  VstRect *rect;
125 
126  // Some effects like to have us get their rect before opening them.
127  mLink->callDispatcher(effEditGetRect, 0, 0, &rect, 0.0);
128 
129  // Ask the effect to add its GUI
130  mLink->callDispatcher(effEditOpen, 0, 0, mVSTView, 0.0);
131 
132  // Get the subview it created
133  mView = [[mVSTView subviews] objectAtIndex:0];
134  if (mView == NULL)
135  {
136  // Doesn't seem the effect created the subview. This can
137  // happen when an effect uses the content view directly.
138  // As of this time, we will not try to support those and
139  // just fall back to the textual interface.
140  return;
141  }
142 
143  // Get the final bounds of the effect GUI
144  mLink->callDispatcher(effEditGetRect, 0, 0, &rect, 0.0);
145 
146  NSRect frame = { 0, 0, (CGFloat) rect->right - rect->left, (CGFloat) rect->bottom - rect->top };
147 
148  [mView setFrame:frame];
149 
150  [mVSTView addSubview:mView];
151 
152  SetMinSize(wxSize(frame.size.width, frame.size.height));
153 
154  return;
155 }

159 void VSTControl::CreateCarbon()
160 {
161  OSStatus result;
162 
163  Connect(wxEVT_SIZE, wxSizeEventHandler(VSTControl::OnSize));
164 
165  VstRect *rect;
166 
167  // Some effects like to have us get their rect before opening them.
168  mLink->callDispatcher(effEditGetRect, 0, 0, &rect, 0.0);
169 
170  // Suggest a dummy size
171  Rect bounds = { 0, 0, 0, 0 };
172 
173  // And create the window
174  result = CreateNewWindow(kOverlayWindowClass,
175  kWindowStandardHandlerAttribute |
176  kWindowCompositingAttribute |
177  kWindowOpaqueForEventsAttribute,
178  &bounds,
179  &mWindowRef);
180  if (result != noErr)
181  {
182  return;
183  }
184 
185  // Get the root control
186  ControlRef root = HIViewGetRoot(mWindowRef);
187 
188  // Find the content view within our window
189  HIViewRef content;
190  result = HIViewFindByID(root, kHIViewWindowContentID, &content);
191  if (result != noErr)
192  {
193  DisposeWindow(mWindowRef);
194  mWindowRef = NULL;
195 
196  return;
197  }
198 
199  // Some effects (iZotope Vinyl) seem to need an existing subview
200  // of the content view. So just use a "dummy" scrollview.
201  result = HIScrollViewCreate(kHIScrollViewOptionsVertScroll, &mHIView);
202 
203  // Don't want to see the scroll bars
204  HIScrollViewSetScrollBarAutoHide(mHIView, true);
205 
206  // Add it as a subview of the content view
207  HIViewAddSubview(content, mHIView);
208 
209  // Ask the effect to add its GUI
210  mLink->callDispatcher(effEditOpen, 0, 0, mWindowRef, 0.0);
211 
212  // Get the subview it created
213  HIViewRef subview = HIViewGetFirstSubview(content);
214  if (subview)
215  {
216  // The scrollview was used, so leave it.
217  if (subview == mHIView)
218  {
219  subview = HIViewGetFirstSubview(mHIView);
220  }
221  // The effect didn't use our scrollview, so dispose of it.
222  else
223  {
224  HIViewRemoveFromSuperview(mHIView);
225  CFRelease(mHIView);
226  mHIView = subview;
227  }
228  }
229 
230  // Doesn't seem the effect created a subview. This can
231  // happen when an effect uses the content view directly.
232  // As of this time, we will not try to support those and
233  // just fall back to the textual interface.
234  if (subview == NULL)
235  {
236  mLink->callDispatcher(effEditClose, 0, 0, mWindowRef, 0.0);
237  DisposeWindow(mWindowRef);
238  mWindowRef = NULL;
239  mHIView = NULL;
240 
241  return;
242  }
243 
244  // Get the final bounds of the effect GUI
245  mLink->callDispatcher(effEditGetRect, 0, 0, &rect, 0.0);
246 
247  // Set the size of the scrollview to match
248  HIRect r = { 0, 0, (CGFloat) rect->right - rect->left, (CGFloat) rect->bottom - rect->top };
249 
250  // One effect, mutagene lipredemuco, doesn't return a valid rect so
251  // try to detect it and use the created view dimensions instead.
252  if (rect->left < 0 || rect->top < 0 || rect->right <= 0 || rect->bottom <= 0)
253  {
254  HIViewGetFrame(subview, &r);
255  }
256 
257  // Make sure container is the same size as the effect GUI
258  HIViewSetFrame(mHIView, &r);
259  HIViewPlaceInSuperviewAt(mHIView, 0, 0);
260 
261  // Establish the minimum size
262  SetMinSize(wxSize(r.size.width, r.size.height));
263 
264  NSWindow *parent = [mVSTView window];
265  NSWindow *host = [[[NSWindow alloc] initWithWindowRef:mWindowRef] autorelease];
266  [parent addChildWindow:host ordered:NSWindowAbove];
267 
268  ShowWindow(mWindowRef);
269 }
