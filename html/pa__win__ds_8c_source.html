321 static double PaWinDS_GetMinSystemLatencySeconds( void )
322 {
323 /*
324 NOTE: GetVersionEx() is deprecated as of Windows 8.1 and can not be used to reliably detect
325 versions of Windows higher than Windows 8 (due to manifest requirements for reporting higher versions).
326 Microsoft recommends switching to VerifyVersionInfo (available on Win 2k and later), however GetVersionEx
327 is is faster, for now we just disable the deprecation warning.
328 See: https://msdn.microsoft.com/en-us/library/windows/desktop/ms724451(v=vs.85).aspx
329 See: http://www.codeproject.com/Articles/678606/Part-Overcoming-Windows-s-deprecation-of-GetVe
330 */
331 #pragma warning (disable : 4996) /* use of GetVersionEx */
332 
333  double minLatencySeconds;
334  /* Set minimal latency based on whether NT or other OS.
335  * NT has higher latency.
336  */
337 
338  OSVERSIONINFO osvi;
339  osvi.dwOSVersionInfoSize = sizeof( osvi );
340  GetVersionEx( &osvi );
341  DBUG(("PA - PlatformId = 0x%x\n", osvi.dwPlatformId ));
342  DBUG(("PA - MajorVersion = 0x%x\n", osvi.dwMajorVersion ));
343  DBUG(("PA - MinorVersion = 0x%x\n", osvi.dwMinorVersion ));
344  /* Check for NT */
345  if( (osvi.dwMajorVersion == 4) && (osvi.dwPlatformId == 2) )
346  {
347  minLatencySeconds = PA_DS_WIN_NT_DEFAULT_LATENCY_;
348  }
349  else if(osvi.dwMajorVersion >= 5)
350  {
351  minLatencySeconds = PA_DS_WIN_WDM_DEFAULT_LATENCY_;
352  }
353  else
354  {
355  minLatencySeconds = PA_DS_WIN_9X_DEFAULT_LATENCY_;

546 static BOOL CALLBACK CollectGUIDsProcW(LPGUID lpGUID,
547  LPCWSTR lpszDesc,
548  LPCWSTR lpszDrvName,
549  LPVOID lpContext )
550 {
551  DSDeviceNameAndGUIDVector *namesAndGUIDs = (DSDeviceNameAndGUIDVector*)lpContext;
552  PaError error;
553 
554  (void) lpszDrvName; /* unused variable */
555 
556  if( namesAndGUIDs->free == 0 )
557  {
558  error = ExpandDSDeviceNameAndGUIDVector( namesAndGUIDs );
559  if( error != paNoError )
560  {
561  namesAndGUIDs->enumerationError = error;
562  return FALSE;
563  }
564  }
565 
566  /* Set GUID pointer, copy GUID to storage in DSDeviceNameAndGUIDVector. */
567  if( lpGUID == NULL )
568  {
569  namesAndGUIDs->items[namesAndGUIDs->count].lpGUID = NULL;
570  }
571  else
572  {
573  namesAndGUIDs->items[namesAndGUIDs->count].lpGUID =
574  &namesAndGUIDs->items[namesAndGUIDs->count].guid;
575 
576  memcpy( &namesAndGUIDs->items[namesAndGUIDs->count].guid, lpGUID, sizeof(GUID) );
577  }
578 
579  namesAndGUIDs->items[namesAndGUIDs->count].name =
580  DuplicateDeviceNameString( namesAndGUIDs->allocations, lpszDesc );
581  if( namesAndGUIDs->items[namesAndGUIDs->count].name == NULL )
582  {
583  namesAndGUIDs->enumerationError = paInsufficientMemory;
584  return FALSE;
585  }
586 
587  namesAndGUIDs->items[namesAndGUIDs->count].pnpInterface = 0;
588 
589  ++namesAndGUIDs->count;
590  --namesAndGUIDs->free;
591 
592  return TRUE;
593 }

1741  unsigned long *pollingPeriodFrames,
1742  int isFullDuplex,
1743  unsigned long suggestedInputLatencyFrames,
1744  unsigned long suggestedOutputLatencyFrames,
1745  double sampleRate, unsigned long userFramesPerBuffer )
1746 {
1747  unsigned long minimumPollingPeriodFrames = (unsigned long)(sampleRate * PA_DS_MINIMUM_POLLING_PERIOD_SECONDS);
1748  unsigned long maximumPollingPeriodFrames = (unsigned long)(sampleRate * PA_DS_MAXIMUM_POLLING_PERIOD_SECONDS);
1749  unsigned long pollingJitterFrames = (unsigned long)(sampleRate * PA_DS_POLLING_JITTER_SECONDS);
1750 
1751  if( userFramesPerBuffer == paFramesPerBufferUnspecified )
1752  {
1753  unsigned long targetBufferingLatencyFrames = max( suggestedInputLatencyFrames, suggestedOutputLatencyFrames );
1754 
1755  *pollingPeriodFrames = targetBufferingLatencyFrames / 4;
1756  if( *pollingPeriodFrames < minimumPollingPeriodFrames )
1757  {
1758  *pollingPeriodFrames = minimumPollingPeriodFrames;
1759  }
1760  else if( *pollingPeriodFrames > maximumPollingPeriodFrames )
1761  {
1762  *pollingPeriodFrames = maximumPollingPeriodFrames;
1763  }
1764 
1765  *hostBufferSizeFrames = *pollingPeriodFrames
1766  + max( *pollingPeriodFrames + pollingJitterFrames, targetBufferingLatencyFrames);
1767  }
1768  else
1769  {
1770  unsigned long targetBufferingLatencyFrames = suggestedInputLatencyFrames;

1776

1778  Otherwise we use input latency anyway since we do max(in,out).
1779  */

1782  {
1783  unsigned long adjustedSuggestedOutputLatencyFrames =
1784  suggestedOutputLatencyFrames - userFramesPerBuffer;

1787  if( adjustedSuggestedOutputLatencyFrames > targetBufferingLatencyFrames )
1788  targetBufferingLatencyFrames = adjustedSuggestedOutputLatencyFrames;
1789  }
1790  }
1791  else
1792  {
1793  /* maximum of input and output suggested latency */
1794  if( suggestedOutputLatencyFrames > suggestedInputLatencyFrames )
1795  targetBufferingLatencyFrames = suggestedOutputLatencyFrames;
1796  }
1797 
1798  *hostBufferSizeFrames = userFramesPerBuffer
1799  + max( userFramesPerBuffer + pollingJitterFrames, targetBufferingLatencyFrames);
1800 
1801  *pollingPeriodFrames = max( max(1, userFramesPerBuffer / 4), targetBufferingLatencyFrames / 16 );
1802 
1803  if( *pollingPeriodFrames > maximumPollingPeriodFrames )
1804  {
1805  *pollingPeriodFrames = maximumPollingPeriodFrames;
1806  }
1807  }
