147 bool EffectNormalize::Process()
148 {
149  if (mGain == false && mDC == false)
150  return true;
151 
152  float ratio;
153  if( mGain )
154  // same value used for all tracks
155  ratio = DB_TO_LINEAR(TrapDouble(mLevel, MIN_Level, MAX_Level));
156  else
157  ratio = 1.0;
158 
159  //Iterate over each track
160  this->CopyInputTracks(); // Set up mOutputTracks.
161  bool bGoodResult = true;
162  SelectedTrackListOfKindIterator iter(Track::Wave, mOutputTracks);
163  WaveTrack *track = (WaveTrack *) iter.First();
164  WaveTrack *prevTrack;
165  prevTrack = track;
166  mCurTrackNum = 0;
167  wxString topMsg;
168  if(mDC && mGain)
169  topMsg = _("Removing DC offset and Normalizing...\n");
170  else if(mDC && !mGain)
171  topMsg = _("Removing DC offset...\n");
172  else if(!mDC && mGain)
173  topMsg = _("Normalizing without removing DC offset...\n");
174  else if(!mDC && !mGain)
175  topMsg = wxT("Not doing anything)...\n"); // shouldn't get here
176 
177  while (track) {
178  //Get start and end times from track
179  double trackStart = track->GetStartTime();
180  double trackEnd = track->GetEndTime();
181 
182  //Set the current bounds to whichever left marker is
183  //greater and whichever right marker is less:
184  mCurT0 = mT0 < trackStart? trackStart: mT0;
185  mCurT1 = mT1 > trackEnd? trackEnd: mT1;
186 
187  // Process only if the right marker is to the right of the left marker
188  if (mCurT1 > mCurT0) {
189  wxString msg;
190  wxString trackName = track->GetName();
191 
192  if(!track->GetLinked() || mStereoInd)
193  msg = topMsg + _("Analyzing: ") + trackName;
194  else
195  msg = topMsg + _("Analyzing first track of stereo pair: ") + trackName;
196  AnalyseTrack(track, msg); // sets mOffset and offset-adjusted mMin and mMax
197  if(!track->GetLinked() || mStereoInd) { // mono or 'stereo tracks independently'
198  float extent = wxMax(fabs(mMax), fabs(mMin));
199  if( (extent > 0) && mGain )
200  mMult = ratio / extent;
201  else
202  mMult = 1.0;
203  msg = topMsg + _("Processing: ") + trackName;
204  if(track->GetLinked() || prevTrack->GetLinked()) // only get here if there is a linked track but we are processing independently
205  msg = topMsg + _("Processing stereo channels independently: ") + trackName;
206 
207  if (!ProcessOne(track, msg))
208  {
209  bGoodResult = false;
210  break;
211  }
212  }
213  else
214  {
215  // we have a linked stereo track
216  // so we need to find it's min, max and offset
217  // as they are needed to calc the multiplier for both tracks
218  float offset1 = mOffset; // remember ones from first track
219  float min1 = mMin;
220  float max1 = mMax;
221  track = (WaveTrack *) iter.Next(); // get the next one
222  mCurTrackNum++; // keeps progress bar correct
223  msg = topMsg + _("Analyzing second track of stereo pair: ") + trackName;
224  AnalyseTrack(track, msg); // sets mOffset and offset-adjusted mMin and mMax
225  float offset2 = mOffset; // ones for second track
226  float min2 = mMin;
227  float max2 = mMax;
228  float extent = wxMax(fabs(min1), fabs(max1));
229  extent = wxMax(extent, fabs(min2));
230  extent = wxMax(extent, fabs(max2));
231  if( (extent > 0) && mGain )
232  mMult = ratio / extent; // we need to use this for both linked tracks
233  else
234  mMult = 1.0;
235  mOffset = offset1;
236  track = (WaveTrack *) iter.Prev(); // go back to the first linked one
237  mCurTrackNum--; // keeps progress bar correct
238  msg = topMsg + _("Processing first track of stereo pair: ") + trackName;
239  if (!ProcessOne(track, msg))
240  {
241  bGoodResult = false;
242  break;
243  }
244  mOffset = offset2;
245  track = (WaveTrack *) iter.Next(); // go to the second linked one
246  mCurTrackNum++; // keeps progress bar correct
247  msg = topMsg + _("Processing second track of stereo pair: ") + trackName;
248  if (!ProcessOne(track, msg))
249  {
250  bGoodResult = false;
251  break;
252  }
253  }
254  }
255 
256  //Iterate to the next track
257  prevTrack = track;
258  track = (WaveTrack *) iter.Next();
259  mCurTrackNum++;
260  }
261 
262  this->ReplaceProcessedTracks(bGoodResult);
263  return bGoodResult;
264 }

361 //AnalyseDC() takes a track, transforms it to bunch of buffer-blocks,
362 //and executes AnalyzeData on it...
363 // sets mOffset
364 bool EffectNormalize::AnalyseDC(WaveTrack * track, const wxString &msg)
365 {
366  bool rc = true;
367  sampleCount s;
368 
369  mOffset = 0.0; // we might just return
370 
371  if(!mDC) // don't do analysis if not doing dc removal
372  return(rc);
373 
374  //Transform the marker timepoints to samples
375  sampleCount start = track->TimeToLongSamples(mCurT0);
376  sampleCount end = track->TimeToLongSamples(mCurT1);
377 
378  //Get the length of the buffer (as double). len is
379  //used simply to calculate a progress meter, so it is easier
380  //to make it a double now than it is to do it later
381  double len = (double)(end - start);
382 
383  //Initiate a processing buffer. This buffer will (most likely)
384  //be shorter than the length of the track being processed.
385  float *buffer = new float[track->GetMaxBlockSize()];
386 
387  mSum = 0.0; // dc offset inits
388  mCount = 0;
389 
390  //Go through the track one buffer at a time. s counts which
391  //sample the current buffer starts at.
392  s = start;
393  while (s < end) {
394  //Get a block of samples (smaller than the size of the buffer)
395  sampleCount block = track->GetBestBlockSize(s);
396 
397  //Adjust the block size if it is the final block in the track
398  if (s + block > end)
399  block = end - s;
400 
401  //Get the samples from the track and put them in the buffer
402  track->Get((samplePtr) buffer, floatSample, s, block);
403 
404  //Process the buffer.
405  AnalyzeData(buffer, block);
406 
407  //Increment s one blockfull of samples
408  s += block;
409 
410  //Update the Progress meter
411  if (TrackProgress(mCurTrackNum,
412  ((double)(s - start) / len)/2.0, msg)) {
413  rc = false; //lda .. break, not return, so that buffer is deleted
414  break;
415  }
416  }
417 
418  //Clean up the buffer
419  delete[] buffer;
420 
421  mOffset = (float)(-mSum / mCount); // calculate actual offset (amount that needs to be added on)
422 
423  //Return true because the effect processing succeeded ... unless cancelled
424  return rc;
425 }
