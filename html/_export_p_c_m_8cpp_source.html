391 int ExportPCM::Export(AudacityProject *project,
392  int numChannels,
393  const wxString &fName,
394  bool selectionOnly,
395  double t0,
396  double t1,
397  MixerSpec *mixerSpec,
398  const Tags *metadata,
399  int subformat)
400 {
401  double rate = project->GetRate();
402  const TrackList *tracks = project->GetTracks();
403  int sf_format;
404 
405  if (subformat < 0 || subformat >= WXSIZEOF(kFormats))
406  {
407  sf_format = ReadExportFormatPref();
408  }
409  else
410  {
411  sf_format = kFormats[subformat].format;
412  }
413 
414  wxString formatStr;
415  SF_INFO info;
416  SNDFILE *sf = NULL;
417  int err;
418 
419  //This whole operation should not occur while a file is being loaded on OD,
420  //(we are worried about reading from a file being written to,) so we block.
421  //Furthermore, we need to do this because libsndfile is not threadsafe.
422  ODManager::LockLibSndFileMutex();
423  formatStr = sf_header_name(sf_format & SF_FORMAT_TYPEMASK);
424 
425  ODManager::UnlockLibSndFileMutex();
426 
427  // Use libsndfile to export file
428 
429  info.samplerate = (unsigned int)(rate + 0.5);
430  info.frames = (unsigned int)((t1 - t0)*rate + 0.5);
431  info.channels = numChannels;
432  info.format = sf_format;
433  info.sections = 1;
434  info.seekable = 0;
435 
436  // If we can't export exactly the format they requested,
437  // try the default format for that header type...
438  if (!sf_format_check(&info))
439  info.format = (info.format & SF_FORMAT_TYPEMASK);
440  if (!sf_format_check(&info)) {
441  wxMessageBox(_("Cannot export audio in this format."));
442  return false;
443  }
444 
445  wxFile f; // will be closed when it goes out of scope
446 
447  if (f.Open(fName, wxFile::write)) {
448  // Even though there is an sf_open() that takes a filename, use the one that
449  // takes a file descriptor since wxWidgets can open a file with a Unicode name and
450  // libsndfile can't (under Windows).
451  ODManager::LockLibSndFileMutex();
452  sf = sf_open_fd(f.fd(), SFM_WRITE, &info, FALSE);
453  //add clipping for integer formats. We allow floats to clip.
454  sf_command(sf, SFC_SET_CLIPPING, NULL,sf_subtype_is_integer(sf_format)?SF_TRUE:SF_FALSE) ;
455  ODManager::UnlockLibSndFileMutex();
456  }
457 
458  if (!sf) {
459  wxMessageBox(wxString::Format(_("Cannot export audio to %s"),
460  fName.c_str()));
461  return false;
462  }
463  // Retrieve tags if not given a set
464  if (metadata == NULL)
465  metadata = project->GetTags();
466 
467  // Install the metata at the beginning of the file (except for
468  // WAV and WAVEX formats)
469  if ((sf_format & SF_FORMAT_TYPEMASK) != SF_FORMAT_WAV &&
470  (sf_format & SF_FORMAT_TYPEMASK) != SF_FORMAT_WAVEX) {
471  if (!AddStrings(project, sf, metadata, sf_format)) {
472  sf_close(sf);
473  return false;
474  }
475  }
476 
477  sampleFormat format;
478  if (sf_subtype_more_than_16_bits(info.format))
479  format = floatSample;
480  else
481  format = int16Sample;
482 
483  int maxBlockLen = 44100 * 5;
484 
485  int updateResult = eProgressSuccess;
486 
487  const WaveTrackConstArray waveTracks =
488  tracks->GetWaveTrackConstArray(selectionOnly, false);
489  Mixer *mixer = CreateMixer(waveTracks,
490  tracks->GetTimeTrack(),
491  t0, t1,
492  info.channels, maxBlockLen, true,
493  rate, format, true, mixerSpec);
494 
495  {
496  ProgressDialog progress(wxFileName(fName).GetName(),
497  selectionOnly ?
498  wxString::Format(_("Exporting the selected audio as %s"),
499  formatStr.c_str()) :
500  wxString::Format(_("Exporting the entire project as %s"),
501  formatStr.c_str()));
502 
503  while (updateResult == eProgressSuccess) {
504  sampleCount samplesWritten;
505  sampleCount numSamples = mixer->Process(maxBlockLen);
506 
507  if (numSamples == 0)
508  break;
509 
510  samplePtr mixed = mixer->GetBuffer();
511 
512  ODManager::LockLibSndFileMutex();
513  if (format == int16Sample)
514  samplesWritten = sf_writef_short(sf, (short *)mixed, numSamples);
515  else
516  samplesWritten = sf_writef_float(sf, (float *)mixed, numSamples);
517  ODManager::UnlockLibSndFileMutex();
518 
519  if (samplesWritten != numSamples) {
520  char buffer2[1000];
521  sf_error_str(sf, buffer2, 1000);
522  wxMessageBox(wxString::Format(
523  /* i18n-hint: %s will be the error message from libsndfile, which
524  * is usually something unhelpful (and untranslated) like "system
525  * error" */
526  _("Error while writing %s file (disk full?).\nLibsndfile says \"%s\""),
527  formatStr.c_str(),
528  wxString::FromAscii(buffer2).c_str()));
529  break;
530  }
531 
532  updateResult = progress.Update(mixer->MixGetCurrentTime() - t0, t1 - t0);
533  }
534  }
535 
536  delete mixer;
537 
538  // Install the WAV metata in a "LIST" chunk at the end of the file
539  if ((sf_format & SF_FORMAT_TYPEMASK) == SF_FORMAT_WAV ||
540  (sf_format & SF_FORMAT_TYPEMASK) == SF_FORMAT_WAVEX) {
541  if (!AddStrings(project, sf, metadata, sf_format)) {
542  sf_close(sf);
543  return false;
544  }
545  }
546 
547  ODManager::LockLibSndFileMutex();
548  err = sf_close(sf);
549  ODManager::UnlockLibSndFileMutex();
550 
551  if (err) {
552  char buffer[1000];
553  sf_error_str(sf, buffer, 1000);
554  wxMessageBox(wxString::Format
555  /* i18n-hint: %s will be the error message from libsndfile */
556  (_("Error (file may not have been written): %s"),
557  buffer));
558  }
559 
560  if (((sf_format & SF_FORMAT_TYPEMASK) == SF_FORMAT_AIFF) ||
561  ((sf_format & SF_FORMAT_TYPEMASK) == SF_FORMAT_WAV))
562  AddID3Chunk(fName, metadata, sf_format);
563 
564  return updateResult;
565 }

741 void ExportPCM::AddID3Chunk(wxString fName, const Tags *tags, int sf_format)
742 {
743 #ifdef USE_LIBID3TAG
744  struct id3_tag *tp = id3_tag_new();
745 
746  for (const auto &pair : tags->GetRange()) {
747  const auto &n = pair.first;
748  const auto &v = pair.second;
749  const char *name = "TXXX";
750 
751  if (n.CmpNoCase(TAG_TITLE) == 0) {
752  name = ID3_FRAME_TITLE;
753  }
754  else if (n.CmpNoCase(TAG_ARTIST) == 0) {
755  name = ID3_FRAME_ARTIST;
756  }
757  else if (n.CmpNoCase(TAG_ALBUM) == 0) {
758  name = ID3_FRAME_ALBUM;
759  }
760  else if (n.CmpNoCase(TAG_YEAR) == 0) {
761  name = ID3_FRAME_YEAR;
762  }
763  else if (n.CmpNoCase(TAG_GENRE) == 0) {
764  name = ID3_FRAME_GENRE;
765  }
766  else if (n.CmpNoCase(TAG_COMMENTS) == 0) {
767  name = ID3_FRAME_COMMENT;
768  }
769  else if (n.CmpNoCase(TAG_TRACK) == 0) {
770  name = ID3_FRAME_TRACK;
771  }
772  else if (n.CmpNoCase(wxT("composer")) == 0) {
773  name = "TCOM";
774  }
775 
776  struct id3_frame *frame = id3_frame_new(name);
777 
778  if (!n.IsAscii() || !v.IsAscii()) {
779  id3_field_settextencoding(id3_frame_field(frame, 0), ID3_FIELD_TEXTENCODING_UTF_16);
780  }
781  else {
782  id3_field_settextencoding(id3_frame_field(frame, 0), ID3_FIELD_TEXTENCODING_ISO_8859_1);
783  }
784 
785  id3_ucs4_t *ucs4 =
786  id3_utf8_ucs4duplicate((id3_utf8_t *) (const char *) v.mb_str(wxConvUTF8));
787 
788  if (strcmp(name, ID3_FRAME_COMMENT) == 0) {
789  // A hack to get around iTunes not recognizing the comment. The
790  // language defaults to XXX and, since it's not a valid language,
791  // iTunes just ignores the tag. So, either set it to a valid language
792  // (which one???) or just clear it. Unfortunately, there's no supported
793  // way of clearing the field, so do it directly.
794  id3_field *f = id3_frame_field(frame, 1);
795  memset(f->immediate.value, 0, sizeof(f->immediate.value));
796  id3_field_setfullstring(id3_frame_field(frame, 3), ucs4);
797  }
798  else if (strcmp(name, "TXXX") == 0) {
799  id3_field_setstring(id3_frame_field(frame, 2), ucs4);
800  free(ucs4);
801 
802  ucs4 = id3_utf8_ucs4duplicate((id3_utf8_t *) (const char *) n.mb_str(wxConvUTF8));
803 
804  id3_field_setstring(id3_frame_field(frame, 1), ucs4);
805  }
806  else {
807  id3_field_setstrings(id3_frame_field(frame, 1), 1, &ucs4);
808  }
809 
810  free(ucs4);
811 
812  id3_tag_attachframe(tp, frame);
813  }
814 
815  tp->options &= (~ID3_TAG_OPTION_COMPRESSION); // No compression
816 
817  // If this version of libid3tag supports it, use v2.3 ID3
818  // tags instead of the newer, but less well supported, v2.4
819  // that libid3tag uses by default.
820 #ifdef ID3_TAG_HAS_TAG_OPTION_ID3V2_3
821  tp->options |= ID3_TAG_OPTION_ID3V2_3;
822 #endif
823 
824  id3_length_t len;
825 
826  len = id3_tag_render(tp, 0);
827  if (len == 0) {
828  id3_tag_delete(tp);
829  return;
830  }
831  if ((len % 2) != 0) len++; // Length must be even.
832  id3_byte_t *buffer = (id3_byte_t *)malloc(len);
833  if (buffer == NULL) {
834  id3_tag_delete(tp);
835  return;
836  }
837  // Zero all locations, for ending odd UTF16 content
838  // correctly, i.e., two '\0's at the end.
839  memset(buffer, 0, len);
840 
841  id3_tag_render(tp, buffer);
842 
843  id3_tag_delete(tp);
844 
845  wxFFile f(fName, wxT("r+b"));
846  if (f.IsOpened()) {
847  wxUint32 sz;
848 
849  sz = (wxUint32) len;
850  f.SeekEnd(0);
851  if ((sf_format & SF_FORMAT_TYPEMASK) == SF_FORMAT_WAV)
852  f.Write("id3 ", 4); // Must be lower case for foobar2000.
853  else {
854  f.Write("ID3 ", 4);
855  sz = wxUINT32_SWAP_ON_LE(sz);
856  }
857  f.Write(&sz, 4);
858 
859  f.Write(buffer, len);
860 
861  sz = (wxUint32) f.Tell() - 8;
862  if ((sf_format & SF_FORMAT_TYPEMASK) == SF_FORMAT_AIFF)
863  sz = wxUINT32_SWAP_ON_LE(sz);
864 
865  f.Seek(4);
866  f.Write(&sz, 4);
867 
868  f.Close();
869  }
870 
871  free(buffer);
872 #endif
873  return;
874 }
