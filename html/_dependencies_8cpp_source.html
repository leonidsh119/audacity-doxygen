59 // Given a project, returns a single array of all SeqBlocks
60 // in the current set of tracks. Enumerating that array allows
61 // you to process all block files in the current set.
62 static void GetAllSeqBlocks(AudacityProject *project,
63  BlockPtrArray *outBlocks)
64 {
65  TrackList *tracks = project->GetTracks();
66  TrackListIterator iter(tracks);
67  Track *t = iter.First();
68  while (t) {
69  if (t->GetKind() == Track::Wave) {
70  WaveTrack *waveTrack = (WaveTrack *)t;
71  WaveClipList::compatibility_iterator node = waveTrack->GetClipIterator();
72  while(node) {
73  WaveClip *clip = node->GetData();
74  Sequence *sequence = clip->GetSequence();
75  BlockArray &blocks = sequence->GetBlockArray();
76  int i;
77  for (i = 0; i < (int)blocks.size(); i++)
78  outBlocks->push_back(&blocks[i]);
79  node = node->GetNext();
80  }
81  }
82  t = iter.Next();
83  }
84 }

86 // Given an Audacity project and a hash mapping aliased block
87 // files to un-aliased block files, walk through all of the
88 // tracks and replace each aliased block file with its replacement.
89 // Note that this code respects reference-counting and thus the
90 // process of making a project self-contained is actually undoable.
91 static void ReplaceBlockFiles(AudacityProject *project,
92  ReplacedBlockFileHash &hash)
93 {
94  DirManager *dirManager = project->GetDirManager();
95  BlockPtrArray blocks;
96  GetAllSeqBlocks(project, &blocks);
97 
98  int i;
99  for (i = 0; i < (int)blocks.size(); i++) {
100  if (hash.count(blocks[i]->f) > 0) {
101  BlockFile *src = blocks[i]->f;
102  BlockFile *dst = hash[src];
103 
104  dirManager->Deref(src);
105  dirManager->Ref(dst);
106 
107  blocks[i]->f = dst;
108  }
109  }
110 }

112 void FindDependencies(AudacityProject *project,
113  AliasedFileArray *outAliasedFiles)
114 {
115  sampleFormat format = project->GetDefaultFormat();
116 
117  BlockPtrArray blocks;
118  GetAllSeqBlocks(project, &blocks);
119 
120  AliasedFileHash aliasedFileHash;
121  BoolBlockFileHash blockFileHash;
122 
123  int i;
124  for (i = 0; i < (int)blocks.size(); i++) {
125  BlockFile *f = blocks[i]->f;
126  if (f->IsAlias() && (blockFileHash.count(f) == 0))
127  {
128  // f is an alias block we have not yet counted.
129  blockFileHash[f] = true; // Don't count the same blockfile twice.
130  AliasBlockFile *aliasBlockFile = (AliasBlockFile *)f;
131  wxFileName fileName = aliasBlockFile->GetAliasedFileName();
132 
133  // In DirManager::ProjectFSCK(), if the user has chosen to
134  // "Replace missing audio with silence", the code there puts in an empty wxFileName.
135  // Don't count those in dependencies.
136  if (!fileName.IsOk())
137  continue;
138 
139  wxString fileNameStr = fileName.GetFullPath();
140  int blockBytes = (SAMPLE_SIZE(format) *
141  aliasBlockFile->GetLength());
142  if (aliasedFileHash.count(fileNameStr) > 0)
143  // Already put this AliasBlockFile in aliasedFileHash.
144  // Update block count.
145  aliasedFileHash[fileNameStr]->mByteCount += blockBytes;
146  else
147  {
148  // Haven't counted this AliasBlockFile yet.
149  // Add to return array and internal hash.
150  outAliasedFiles->Add(AliasedFile(fileName,
151  blockBytes,
152  fileName.FileExists()));
153  aliasedFileHash[fileNameStr] =
154  &((*outAliasedFiles)[outAliasedFiles->GetCount()-1]);
155  }
156  }
157  }
158 }

160 // Given a project and a list of aliased files that should no
161 // longer be external dependencies (selected by the user), replace
162 // all of those alias block files with disk block files.
163 static void RemoveDependencies(AudacityProject *project,
164  AliasedFileArray *aliasedFiles)
165 {
166  DirManager *dirManager = project->GetDirManager();
167 
168  ProgressDialog progress
169  (_("Removing Dependencies"),
170  _("Copying audio data into project..."));
171  int updateResult = eProgressSuccess;
172 
173  // Hash aliasedFiles based on their full paths and
174  // count total number of bytes to process.
175  AliasedFileHash aliasedFileHash;
176  wxLongLong totalBytesToProcess = 0;
177  unsigned int i;
178  for (i = 0; i < aliasedFiles->GetCount(); i++) {
179  totalBytesToProcess += aliasedFiles->Item(i).mByteCount;
180  wxString fileNameStr = aliasedFiles->Item(i).mFileName.GetFullPath();
181  aliasedFileHash[fileNameStr] = &aliasedFiles->Item(i);
182  }
183 
184  BlockPtrArray blocks;
185  GetAllSeqBlocks(project, &blocks);
186 
187  const sampleFormat format = project->GetDefaultFormat();
188  ReplacedBlockFileHash blockFileHash;
189  wxLongLong completedBytes = 0;
190  for (i = 0; i < blocks.size(); i++) {
191  BlockFile *f = blocks[i]->f;
192  if (f->IsAlias() && (blockFileHash.count(f) == 0))
193  {
194  // f is an alias block we have not yet processed.
195  AliasBlockFile *aliasBlockFile = (AliasBlockFile *)f;
196  wxFileName fileName = aliasBlockFile->GetAliasedFileName();
197  wxString fileNameStr = fileName.GetFullPath();
198 
199  if (aliasedFileHash.count(fileNameStr) == 0)
200  // This aliased file was not selected to be replaced. Skip it.
201  continue;
202 
203  // Convert it from an aliased file to an actual file in the project.
204  unsigned int len = aliasBlockFile->GetLength();
205  BlockFile *newBlockFile;
206  {
207  SampleBuffer buffer(len, format);
208  f->ReadData(buffer.ptr(), format, 0, len);
209  newBlockFile =
210  dirManager->NewSimpleBlockFile(buffer.ptr(), len, format);
211  }
212 
213  // Update our hash so we know what block files we've done
214  blockFileHash[f] = newBlockFile;
215 
216  // Update the progress bar
217  completedBytes += SAMPLE_SIZE(format) * len;
218  updateResult = progress.Update(completedBytes, totalBytesToProcess);
219  if (updateResult != eProgressSuccess)
220  break;
221  }
222  }
223 
224  // Above, we created a SimpleBlockFile contained in our project
225  // to go with each AliasBlockFile that we wanted to migrate.
226  // However, that didn't actually change any references to these
227  // blockfiles in the Sequences, so we do that next...
228  ReplaceBlockFiles(project, blockFileHash);
229 
230  // Subtract one from reference count of NEW block files; they're
231  // now all referenced the proper number of times by the Sequences
232  ReplacedBlockFileHash::iterator it;
233  for( it = blockFileHash.begin(); it != blockFileHash.end(); ++it )
234  {
235  BlockFile *f = it->second;
236  dirManager->Deref(f);
237  }
238 }

543 // Checks for alias block files, modifies the project if the
544 // user requests it, and returns true if the user continues.
545 // Returns false only if the user clicks Cancel.
546 bool ShowDependencyDialogIfNeeded(AudacityProject *project,
547  bool isSaving)
548 {
549  AliasedFileArray aliasedFiles;
550  FindDependencies(project, &aliasedFiles);
551 
552  if (aliasedFiles.GetCount() == 0) {
553  if (!isSaving)
554  {
555  wxString msg =
556 _("Your project is currently self-contained; it does not depend on any external audio files. \
557 \n\nIf you change the project to a state that has external dependencies on imported \
558 files, it will no longer be self-contained. If you then Save without copying those files in, \
559 you may lose data.");
560  wxMessageBox(msg,
561  _("Dependency Check"),
562  wxOK | wxICON_INFORMATION,
563  project);
564  }
565  return true; // Nothing to do.
566  }
567 
568  if (isSaving)
569  {
570  wxString action =
571  gPrefs->Read(
572  wxT("/FileFormats/SaveProjectWithDependencies"),
573  wxT("ask"));
574  if (action == wxT("copy"))
575  {
576  // User always wants to remove dependencies
577  RemoveDependencies(project, &aliasedFiles);
578  return true;
579  }
580  if (action == wxT("never"))
581  // User never wants to remove dependencies
582  return true;
583  }
584 
585  DependencyDialog dlog(project, -1, project, &aliasedFiles, isSaving);
586  int returnCode = dlog.ShowModal();
587  if (returnCode == wxID_CANCEL)
588  return false;
589  else if (returnCode == wxID_YES)
590  RemoveDependencies(project, &aliasedFiles);
591 
592  return true;
593 }
