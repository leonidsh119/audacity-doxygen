34 void ODDecodeTask::DoSomeInternal()
35 {
36  if(mBlockFiles.size()<=0)
37  {
38  mPercentCompleteMutex.Lock();
39  mPercentComplete = 1.0;
40  mPercentCompleteMutex.Unlock();
41  return;
42  }
43 
44  ODDecodeBlockFile* bf;
45  ODFileDecoder* decoder;
46  sampleCount blockStartSample = 0;
47  sampleCount blockEndSample = 0;
48  bool success =false;
49 
50  for(size_t i=0; i < mWaveTracks.size() && mBlockFiles.size();i++)
51  {
52  bf = mBlockFiles[0];
53 
54  int ret = 1;
55 
56  //first check to see if the ref count is at least 2. It should have one
57  //from when we added it to this instance's mBlockFiles array, and one from
58  //the Wavetrack/sequence. If it doesn't it has been deleted and we should forget it.
59  if(bf->RefCount()>=2)
60  {
61  //OD TODO: somehow pass the bf a reference to the decoder that manages it's file.
62  //we need to ensure that the filename won't change or be moved. We do this by calling LockRead(),
63  //which the dirmanager::EnsureSafeFilename also does.
64  bf->LockRead();
65  //Get the decoder. If the file was moved, we need to create another one and init it.
66  decoder=GetOrCreateMatchingFileDecoder(bf);
67  if(!decoder->IsInitialized())
68  decoder->Init();
69  bf->SetODFileDecoder(decoder);
70  ret = bf->DoWriteBlockFile();
71  bf->UnlockRead();
72 
73  if(ret >= 0) {
74  success = true;
75  blockStartSample = bf->GetStart();
76  blockEndSample = blockStartSample + bf->GetLength();
77  mComputedBlockFiles++;
78  }
79  }
80  else
81  {
82  //the waveform in the wavetrack now is shorter, so we need to update mMaxBlockFiles
83  //because now there is less work to do.
84  mMaxBlockFiles--;
85  }
86 
87  //Release the refcount we placed on it if we are successful
88  if(ret >= 0 ) {
89  bf->Deref();
90  //take it out of the array - we are done with it.
91  mBlockFiles.erase(mBlockFiles.begin());
92 
93  //upddate the gui for all associated blocks. It doesn't matter that we're hitting more wavetracks then we should
94  //because this loop runs a number of times equal to the number of tracks, they probably are getting processed in
95  //the next iteration at the same sample window.
96  mWaveTrackMutex.Lock();
97  for(size_t i=0;i<mWaveTracks.size();i++)
98  {
99  if(success && mWaveTracks[i])
100  mWaveTracks[i]->AddInvalidRegion(blockStartSample,blockEndSample);
101  }
102  mWaveTrackMutex.Unlock();
103  }
104  }
105 
106  //update percentage complete.
107  CalculatePercentComplete();
108 }

172  (sampleCount)(tempBlocks[insertCursor]->GetStart()+tempBlocks[insertCursor]->GetClipOffset()) <
173  (sampleCount)((oddbFile->GetStart()+oddbFile->GetClipOffset())))

194 void ODDecodeTask::OrderBlockFiles(std::vector<ODDecodeBlockFile*> &unorderedBlocks)
195 {
196  //we are going to take things out of the array. But first deref them since we ref them when we put them in.
197  for(unsigned int i=0;i<mBlockFiles.size();i++)
198  mBlockFiles[i]->Deref();
199  mBlockFiles.clear();
200  //TODO:order the blockfiles into our queue in a fancy convenient way. (this could be user-prefs)
201  //for now just put them in linear. We start the order from the first block that includes the ondemand sample
202  //(which the user sets by clicking.) note that this code is pretty hacky - it assumes that the array is sorted in time.
203 
204  //find the startpoint
205  sampleCount processStartSample = GetDemandSample();
206  for(int i= ((int)unorderedBlocks.size())-1;i>= 0;i--)
207  {
208  //check to see if the refcount is at least two before we add it to the list.
209  //There should be one Ref() from the one added by this ODTask, and one from the track.
210  //If there isn't, then the block was deleted for some reason and we should ignore it.
211  if(unorderedBlocks[i]->RefCount()>=2)
212  {
213  //test if the blockfiles are near the task cursor. we use the last mBlockFiles[0] as our point of reference
214  //and add ones that are closer.
215  //since the order is linear right to left, this will add blocks so that the ones on the right side of the target
216  //are processed first, with the ones closer being processed earlier. Then the ones on the left side get processed.
217  if(mBlockFiles.size() &&
218  unorderedBlocks[i]->GetGlobalEnd() >= processStartSample &&
219  ( mBlockFiles[0]->GetGlobalEnd() < processStartSample ||
220  unorderedBlocks[i]->GetGlobalStart() <= mBlockFiles[0]->GetGlobalStart()) )
221  {
222  //insert at the front of the list if we get blockfiles that are after the demand sample
223  mBlockFiles.insert(mBlockFiles.begin()+0,unorderedBlocks[i]);
224  }
225  else
226  {
227  //otherwise no priority
228  mBlockFiles.push_back(unorderedBlocks[i]);
229  }
230  if(mMaxBlockFiles< (int) mBlockFiles.size())
231  mMaxBlockFiles = mBlockFiles.size();
232  }
233  else
234  {
235  //Otherwise, let it be deleted and forget about it.
236  unorderedBlocks[i]->Deref();
237  }
238  }
