25 lilv_plugin_instantiate(const LilvPlugin* plugin,
26  double sample_rate,
27  const LV2_Feature*const* features)
28 {
29  lilv_plugin_load_if_necessary(plugin);
30 
31  LilvInstance* result = NULL;
32 
33  const LilvNode* const lib_uri = lilv_plugin_get_library_uri(plugin);
34  const LilvNode* const bundle_uri = lilv_plugin_get_bundle_uri(plugin);
35 
36  const char* bundle_path = lilv_uri_to_path(
37  lilv_node_as_uri(lilv_plugin_get_bundle_uri(plugin)));
38 
39  LilvLib* lib = lilv_lib_open(plugin->world, lib_uri, bundle_path, features);
40  if (!lib) {
41  return NULL;
42  }
43 
44  // Parse bundle URI to use as base URI
45  const char* bundle_uri_str = lilv_node_as_uri(bundle_uri);
46  SerdURI base_uri;
47  if (serd_uri_parse((const uint8_t*)bundle_uri_str, &base_uri)) {
48  lilv_lib_close(lib);
49  return NULL;
50  }
51 
52  const LV2_Feature** local_features = NULL;
53  if (features == NULL) {
54  local_features = (const LV2_Feature**)malloc(sizeof(LV2_Feature));
55  local_features[0] = NULL;
56  }
57 
58  // Search for plugin by URI
59  for (uint32_t i = 0; true; ++i) {
60  const LV2_Descriptor* ld = lilv_lib_get_plugin(lib, i);
61  if (!ld) {
62  LILV_ERRORF("No plugin <%s> in <%s>\n",
63  lilv_node_as_uri(lilv_plugin_get_uri(plugin)),
64  lilv_node_as_uri(lib_uri));
65  lilv_lib_close(lib);
66  break; // return NULL
67  }
68 
69  // Resolve library plugin URI against base URI
70  SerdURI abs_uri;
71  SerdNode abs_uri_node = serd_node_new_uri_from_string(
72  (const uint8_t*)ld->URI, &base_uri, &abs_uri);
73  if (!abs_uri_node.buf) {
74  LILV_ERRORF("Failed to parse plugin URI `%s'\n", ld->URI);
75  lilv_lib_close(lib);
76  break;
77  }
78 
79  if (!strcmp((const char*)abs_uri_node.buf,
80  lilv_node_as_uri(lilv_plugin_get_uri(plugin)))) {
81  // Create LilvInstance to return
82  result = (LilvInstance*)malloc(sizeof(LilvInstance));
83  result->lv2_descriptor = ld;
84  result->lv2_handle = ld->instantiate(
85  ld, sample_rate, bundle_path,
86  (features) ? features : local_features);
87  result->pimpl = lib;
88  serd_node_free(&abs_uri_node);
89  break;
90  } else {
91  serd_node_free(&abs_uri_node);
92  }
93  }
94 
95  free(local_features);
96 
97  if (result) {
98  // Failed to instantiate
99  if (result->lv2_handle == NULL) {
100  free(result);
101  return NULL;
102  }
103 
104  // "Connect" all ports to NULL (catches bugs)
105  for (uint32_t i = 0; i < lilv_plugin_get_num_ports(plugin); ++i)
106  result->lv2_descriptor->connect_port(result->lv2_handle, i, NULL);
107  }
108 
109  return result;
110 }
