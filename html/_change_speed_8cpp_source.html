211 bool EffectChangeSpeed::Process()
212 {
213  // Similar to EffectSoundTouch::Process()
214 
215  // Iterate over each track.
216  // Track::All is needed because this effect needs to introduce
217  // silence in the sync-lock group tracks to keep sync
218  CopyInputTracks(Track::All); // Set up mOutputTracks.
219  bool bGoodResult = true;
220 
221  TrackListIterator iter(mOutputTracks);
222  Track* t;
223  mCurTrackNum = 0;
224  mMaxNewLength = 0.0;
225 
226  mFactor = 100.0 / (100.0 + m_PercentChange);
227 
228  t = iter.First();
229  while (t != NULL)
230  {
231  if (t->GetKind() == Track::Label) {
232  if (t->GetSelected() || t->IsSyncLockSelected())
233  {
234  if (!ProcessLabelTrack(t)) {
235  bGoodResult = false;
236  break;
237  }
238  }
239  }
240  else if (t->GetKind() == Track::Wave && t->GetSelected())
241  {
242  WaveTrack *pOutWaveTrack = (WaveTrack*)t;
243  //Get start and end times from track
244  mCurT0 = pOutWaveTrack->GetStartTime();
245  mCurT1 = pOutWaveTrack->GetEndTime();
246 
247  //Set the current bounds to whichever left marker is
248  //greater and whichever right marker is less:
249  mCurT0 = wxMax(mT0, mCurT0);
250  mCurT1 = wxMin(mT1, mCurT1);
251 
252  // Process only if the right marker is to the right of the left marker
253  if (mCurT1 > mCurT0) {
254  //Transform the marker timepoints to samples
255  sampleCount start = pOutWaveTrack->TimeToLongSamples(mCurT0);
256  sampleCount end = pOutWaveTrack->TimeToLongSamples(mCurT1);
257 
258  //ProcessOne() (implemented below) processes a single track
259  if (!ProcessOne(pOutWaveTrack, start, end))
260  {
261  bGoodResult = false;
262  break;
263  }
264  }
265  mCurTrackNum++;
266  }
267  else if (t->IsSyncLockSelected())
268  {
269  t->SyncLockAdjust(mT1, mT0 + (mT1 - mT0) * mFactor);
270  }
271 
272  //Iterate to the next track
273  t=iter.Next();
274  }
275 
276  if (bGoodResult)
277  ReplaceProcessedTracks(bGoodResult);
278 
279  // Update selection.
280  mT1 = mT0 + (((mT1 - mT0) * 100.0) / (100.0 + m_PercentChange));
281 
282  return bGoodResult;
283 }

467 // ProcessOne() takes a track, transforms it to bunch of buffer-blocks,
468 // and calls libsamplerate code on these blocks.
469 bool EffectChangeSpeed::ProcessOne(WaveTrack * track,
470  sampleCount start, sampleCount end)
471 {
472  if (track == NULL)
473  return false;
474 
475  // initialization, per examples of Mixer::Mixer and
476  // EffectSoundTouch::ProcessOne
477 
478  WaveTrack * outputTrack = mFactory->NewWaveTrack(track->GetSampleFormat(),
479  track->GetRate());
480 
481  //Get the length of the selection (as double). len is
482  //used simple to calculate a progress meter, so it is easier
483  //to make it a double now than it is to do it later
484  double len = (double)(end - start);
485 
486  // Initiate processing buffers, most likely shorter than
487  // the length of the selection being processed.
488  sampleCount inBufferSize = track->GetMaxBlockSize();
489 
490  float * inBuffer = new float[inBufferSize];
491 
492  sampleCount outBufferSize =
493  (sampleCount)((mFactor * inBufferSize) + 10);
494  float * outBuffer = new float[outBufferSize];
495 
496  // Set up the resampling stuff for this track.
497  Resample resample(true, mFactor, mFactor); // constant rate resampling
498 
499  //Go through the track one buffer at a time. samplePos counts which
500  //sample the current buffer starts at.
501  bool bResult = true;
502  sampleCount blockSize;
503  sampleCount samplePos = start;
504  while (samplePos < end) {
505  //Get a blockSize of samples (smaller than the size of the buffer)
506  blockSize = track->GetBestBlockSize(samplePos);
507 
508  //Adjust the block size if it is the final block in the track
509  if (samplePos + blockSize > end)
510  blockSize = end - samplePos;
511 
512  //Get the samples from the track and put them in the buffer
513  track->Get((samplePtr) inBuffer, floatSample, samplePos, blockSize);
514 
515  int inUsed;
516  int outgen = resample.Process(mFactor,
517  inBuffer,
518  blockSize,
519  ((samplePos + blockSize) >= end),
520  &inUsed,
521  outBuffer,
522  outBufferSize);
523  if (outgen < 0) {
524  bResult = false;
525  break;
526  }
527 
528  if (outgen > 0)
529  outputTrack->Append((samplePtr)outBuffer, floatSample,
530  outgen);
531 
532  // Increment samplePos
533  samplePos += inUsed;
534 
535  // Update the Progress meter
536  if (TrackProgress(mCurTrackNum, (samplePos - start) / len)) {
537  bResult = false;
538  break;
539  }
540  }
541 
542  // Flush the output WaveTrack (since it's buffered, too)
543  outputTrack->Flush();
544 
545  // Clean up the buffers
546  delete [] inBuffer;
547  delete [] outBuffer;
548 
549  // Take the output track and insert it in place of the original
550  // sample data
551  double newLength = outputTrack->GetEndTime();
552  if (bResult)
553  {
554  SetTimeWarper(new LinearTimeWarper(mCurT0, mCurT0, mCurT1, mCurT0 + newLength));
555  bResult = track->ClearAndPaste(mCurT0, mCurT1, outputTrack, true, false, GetTimeWarper());
556  }
557 
558  if (newLength > mMaxNewLength)
559  mMaxNewLength = newLength;
560 
561  // Delete the outputTrack now that its data is inserted in place
562  delete outputTrack;
563 
564  return bResult;
565 }

717 void EffectChangeSpeed::Update_Vinyl()
718 // Update Vinyl controls from percent change.
719 {
720  // Match Vinyl rpm when within 0.01% of a standard ratio.
721  // Ratios calculated as: ((toRPM / fromRPM) - 1) * 100 * 100
722  int ratio = wxRound(m_PercentChange * 100);
723 
724  switch (ratio)
725  {
726  case 0: // toRPM is the same as fromRPM
727  if (mFromVinyl != kVinyl_NA) {
728  mpChoice_ToVinyl->SetSelection(mpChoice_FromVinyl->GetSelection());
