242  ReleaseMouse();
243  }
244  else if( !HasCapture() )

246  wxRect rect = GetRect();

473 void ToolManager::Reset()
474 {
475  int ndx;
476 
477  // Disconnect all docked bars
478  for( ndx = 0; ndx < ToolBarCount; ndx++ )
479  {

484 
485  // Disconnect the bar
486  if( bar->IsDocked() )
487  {
488  bar->GetDock()->Undock( bar );
489  floater = NULL;
490  }
491  else
492  {
493  floater = bar->GetParent();
494  }
495 
496  if (ndx == SelectionBarID
497 #ifdef EXPERIMENTAL_SPECTRAL_EDITING
498  || ndx == SpectralSelectionBarID
499 #endif
500  )
501  {
502  dock = mBotDock;
503 
504  wxCommandEvent e;
505  bar->GetEventHandler()->ProcessEvent(e);
506  }
507  else
508  {
509  dock = mTopDock;
510  bar->ReCreateButtons();
511  }
512 
513  bar->EnableDisableButtons();
514 #if 0
515  if( bar->IsResizable() )
516  {
517  bar->SetSize(bar->GetBestFittingSize());
518  }
519 #endif
520 
521  if( ndx == MeterBarID
522 #ifdef EXPERIMENTAL_SPECTRAL_EDITING
523  || ndx == SpectralSelectionBarID
524 #endif
525  )
526  {
527  expose = false;
528  }
529 
530  if( dock != NULL )
531  {
532  // when we dock, we reparent, so bar is no longer a child of floater.
533  dock->Dock( bar );
534  Expose( ndx, expose );
535  //OK (and good) to DELETE floater, as bar is no longer in it.
536  if( floater )
537  floater->Destroy();
538  }
539  else
540  {
541  // The (tool)bar has a dragger window round it, the floater.
542  // in turn floater will have mParent (the entire App) as its
543  // parent.
544 
545  // Maybe construct a NEW floater
546  // this happens if we have just been bounced out of a dock.
547  if( floater == NULL ) {
548  floater = safenew ToolFrame( mParent, this, bar, wxPoint(-1,-1) );
549  bar->Reparent( floater );
550  }
551 
552  // This bar is undocked and invisible.
553  // We are doing a reset toolbars, so even the invisible undocked bars should
554  // be moved somewhere sensible. Put bar near center of window.
555  // If there were multiple hidden toobars the ndx * 10 adjustment means
556  // they won't overlap too much.
557  floater->CentreOnParent( );
558  floater->Move( floater->GetPosition() + wxSize( ndx * 10 - 200, ndx * 10 ));
559  bar->SetDocked( NULL, false );
560  Expose( ndx, false );
561  }
562 
563  }
564  // TODO:??
565  // If audio was playing, we stopped the VU meters,
566  // It would be nice to show them again, but hardly essential as
567  // they will show up again on the next play.
568  // SetVUMeters(AudacityProject *p);
569  LayoutToolBars();
570  Updated();

961 //
962 // Handle toolbar dragging
963 //
964 void ToolManager::OnMouse( wxMouseEvent & event )
965 {
966  // Go ahead and set the event to propagate
967  event.Skip();
968 
969  // Can't do anything if we're not dragging. This also prevents
970  // us from intercepting events that don't belong to us from the
971  // parent since we're Connect()ed to a couple.
972  if( !mDragWindow )
973  {
974  return;
975  }
976 
977 #if defined(__WXMAC__)
978  // Disable window animation
979  wxSystemOptions::SetOption( wxMAC_WINDOW_PLAIN_TRANSITION, 1 );
980 #endif
981 
982  // Retrieve the event position
983  wxPoint pos =
984  ( (wxWindow *)event.GetEventObject() )->ClientToScreen( event.GetPosition() ) - mDragOffset;
985 
986  // Button was released...finish the drag
987  if( !event.LeftIsDown() )
988  {
989  // Release capture
990  if( mParent->HasCapture() )
991  {
992  mParent->ReleaseMouse();
993  }
994 
995  // Hide the indicator
996  mIndicator->Hide();
997 
998  // Transition the bar to a dock
999  if( mDragDock && !event.ShiftDown() )
1000  {
1001  // Trip over...everyone ashore that's going ashore...
1002  mDragDock->Dock( mDragBar, mDragBefore );
1003 
1004  // Done with the floater
1005  mDragWindow->Destroy();
1006  mDragBar->Refresh(false);
1007  }
1008  else
1009  {
1010  // Calling SetDocked() to force the grabber button to popup
1011  mDragBar->SetDocked( NULL, false );
1012  }
1013 
1014  // Done dragging
1015  mDragWindow = NULL;
1016  mDragDock = NULL;
1017  mDragBar = NULL;
1018  mLastPos.x = mBarPos.x = -1;
1019  mLastPos.y = mBarPos.y = -1;
1020  mTimer.Stop();
1021  }
1022  else if( event.Dragging() && pos != mLastPos )
1023  {
1024  // Make toolbar follow the mouse
1025  mDragWindow->Move( pos );
1026 
1027  // Remember to prevent excessive movement
1028  mLastPos = pos;
1029 
1030  // Calc the top dock hittest rectangle
1031  wxRect tr = mTopDock->GetRect();
1032  tr.SetBottom( tr.GetBottom() + 10 );
1033  tr.SetPosition( mTopDock->GetParent()->ClientToScreen( tr.GetPosition() ) );
1034 
1035  // Calc the bottom dock hittest rectangle
1036  wxRect br = mBotDock->GetRect();
1037  br.SetTop( br.GetTop() - 10 );
1038  br.SetBottom( br.GetBottom() + 20 );
1039  br.SetPosition( mBotDock->GetParent()->ClientToScreen( br.GetPosition() ) );
1040 
1041 
1042  // Add half the bar height. We could use the actual bar height, but that would be confusing as a
1043  // bar removed at a place might not dock back there if just let go.
1044  // Also add 5 pixels in horizontal direction, so that a click without a move (or a very small move)
1045  // lands back where we started.
1046  pos += wxPoint( 5, 20 );
1047 
1048 
1049  // To find which dock, rather than test against pos, test against the whole dragger rect.
1050  // This means it is enough to overlap the dock to dock with it.
1051  wxRect barRect = mDragWindow->GetRect();
1052  ToolDock *dock = NULL;
1053  if( tr.Intersects( barRect ) )
1054  dock = mTopDock;
1055  else if( br.Intersects( barRect ) )
1056  dock = mBotDock;
1057 
1058  // Looks like we have a winner...
1059  if( dock )
1060  {
1061  wxPoint p;
1062  wxRect r;
1063 
1064  // Calculate where the bar would be placed
1065  mDragBefore = dock->PositionBar( mDragBar, pos, r );
1066 
1067  // If different than the last time, the indicator must be moved
1068  if( r != mBarPos )
1069  {
1070  wxRect dr = dock->GetRect();
1071 
1072  // Hide the indicator before changing the shape
1073  mIndicator->Hide();
1074 
1075  // Decide which direction the arrow should point
1076  if( r.GetTop() >= dr.GetHeight() )
1077  {
1078  p.x = dr.GetLeft() + ( dr.GetWidth() / 2 );
1079  p.y = dr.GetBottom() - mDown->GetBox().GetHeight();
1080  mCurrent = mDown;
1081  }
1082  else
1083  {
1084  p.x = dr.GetLeft() + r.GetLeft();
1085  p.y = dr.GetTop() + r.GetTop() + mLeft->GetBox().GetHeight() / 2;
1086  //JKC ( ( r.GetHeight() - mLeft->GetBox().GetHeight() ) / 2 );
1087  mCurrent = mLeft;
1088  }
1089 
1090  // Change the shape while hidden and then show it if okay
1091  mIndicator->SetShape( *mCurrent );
1092  if( !event.ShiftDown() )
1093  {
1094  mIndicator->Show();
1095  mIndicator->Update();
1096  }
1097 
1098  // Move it into position
1099  // LL: Do this after the Show() since KDE doesn't move the window
1100  // if it's not shown. (Do it outside if the previous IF as well)
1101  mIndicator->Move( dock->GetParent()->ClientToScreen( p ) );
1102 
1103  // Remember for next go round
1104  mBarPos = r;
1105  }
1106  }
1107  else
1108  {
1109  // Hide the indicator if it's still shown
1110  if( mBarPos.x != -1 )
1111  {
1112  // Hide any
1113  mIndicator->Hide();
1114  mBarPos.x = -1;
1115  mBarPos.y = -1;
1116  }
1117  }
1118 
1119  // Remember to which dock the drag bar belongs.
1120  mDragDock = dock;
1121  }
1122 
1123 #if defined(__WXMAC__)
1124  // Reinstate original transition
1125  wxSystemOptions::SetOption( wxMAC_WINDOW_PLAIN_TRANSITION, mTransition );
1126 #endif
1127 }
