145 void ExportMultiple::CountTracksAndLabels()
146 {
147  mLabels = NULL;
148  mNumLabels = 0;
149  mNumWaveTracks = 0;
150 
151  const Track* pTrack;
152  TrackListConstIterator iter;
153  for (pTrack = iter.First(mTracks); pTrack != NULL; pTrack = iter.Next())
154  {
155  switch (pTrack->GetKind())
156  {
157  // Count WaveTracks, and for linked pairs, count only the second of the pair.
158  case Track::Wave:
159  {
160  if (!pTrack->GetMute() && !pTrack->GetLinked()) // Don't count muted tracks.
161  mNumWaveTracks++;
162  break;
163  }
164 
165  // Only support one label track???
166  case Track::Label:
167  {
168  // Supports only one LabelTrack.
169  if (mLabels == NULL) {
170  mLabels = (LabelTrack*)pTrack;
171  mNumLabels = mLabels->GetNumLabels();
172  }
173  break;
174  }
175  }
176  }

620 int ExportMultiple::ExportMultipleByLabel(bool byName,
621  const wxString &prefix, bool addNumber)
622 {
623  wxASSERT(mProject);
624  bool tagsPrompt = mProject->GetShowId3Dialog();
625  int numFiles = mNumLabels;
626  int l = 0; // counter for files done
627  ExportKitArray exportSettings; // dynamic array we will use to store the
628  // settings needed to do the exports with in
629  exportSettings.Alloc(numFiles); // Allocate some guessed space to use.
630 
631  // Account for exporting before first label
632  if (mFirst->GetValue()) {
633  l--;
634  numFiles++;
635  }
636 
637  // Figure out how many channels we should export.
638  int channels = mTracks->GetNumExportChannels(false);
639 
640  wxArrayString otherNames; // keep track of file names we will use, so we
641  // don't duplicate them
642  ExportKit setting; // the current batch of settings
643  setting.destfile.SetPath(mDir->GetValue());
644  setting.destfile.SetExt(mPlugins[mPluginIndex]->GetExtension(mSubFormatIndex));
645  wxLogDebug(wxT("Plug-in index = %d, Sub-format = %d"), mPluginIndex, mSubFormatIndex);
646  wxLogDebug(wxT("File extension is %s"), setting.destfile.GetExt().c_str());
647  wxString name; // used to hold file name whilst we mess with it
648  wxString title; // un-messed-with title of file for tagging with
649 
650  const LabelStruct *info = NULL;
651  /* Examine all labels a first time, sort out all data but don't do any
652  * exporting yet (so this run is quick but interactive) */
653  while (l < mNumLabels) {
654 
655  // Get file name and starting time
656  if (l < 0) {
657  // create wxFileName for output file
658  name = (mFirstFileName->GetValue());
659  setting.t0 = 0.0;
660  }
661  else {
662  info = mLabels->GetLabel(l);
663  name = (info->title);
664  setting.t0 = info->selectedRegion.t0();
665  }
666 
667  // Figure out the ending time
668  if (info && !info->selectedRegion.isPoint()) {
669  setting.t1 = info->selectedRegion.t1();
670  }
671  else if (l < mNumLabels-1) {
672  // Use start of next label as end
673  const LabelStruct *info1 = mLabels->GetLabel(l+1);
674  setting.t1 = info1->selectedRegion.t0();
675  }
676  else {
677  setting.t1 = mTracks->GetEndTime();
678  }
679 
680  if( name.IsEmpty() )
681  name = _("untitled");
682 
683  // store title of label to use in tags
684  title = name;
685 
686  // Numbering files...
687  if (!byName) {
688  name.Printf(wxT("%s-%02d"), prefix.c_str(), l+1);
689  } else if (addNumber) {
690  // Following discussion with GA, always have 2 digits
691  // for easy file-name sorting (on Windows)
692  name.Prepend(wxString::Format(wxT("%02d-"), l+1));
693  }
694 
695  // store sanitised and user checked name in object
696  setting.destfile.SetName(MakeFileName(name));
697 
698  wxASSERT(setting.destfile.IsOk()); // scream if file name is broke
699 
700  // Make sure the (final) file name is unique within the set of exports
701  FileNames::MakeNameUnique(otherNames, setting.destfile);
702 
703  /* do the metadata for this file */
704  // copy project metadata to start with
705  setting.filetags = *(mProject->GetTags());
706  // over-ride with values
707  setting.filetags.SetTag(TAG_TITLE, title);
708  setting.filetags.SetTag(TAG_TRACK, l+1);
709  // let the user have a crack at editing it, exit if cancelled
710  if (!setting.filetags.ShowEditDialog(mProject,_("Edit Metadata Tags"), tagsPrompt))
711  return false;
712 
713  /* add the settings to the array of settings to be used for export */
714  exportSettings.Add(setting);
715 
716  l++; // next label, count up one
717  }
718 
719  int ok = eProgressSuccess; // did it work?
720  int count = 0; // count the number of sucessful runs
721  ExportKit activeSetting; // pointer to the settings in use for this export
722  /* Go round again and do the exporting (so this run is slow but
723  * non-interactive) */
724  for (count = 0; count < numFiles; count++) {
725  /* get the settings to use for the export from the array */
726  activeSetting = exportSettings[count];
727 
728  // Export it
729  ok = DoExport(channels, activeSetting.destfile, false, activeSetting.t0, activeSetting.t1, activeSetting.filetags);
730  if (ok != eProgressSuccess && ok != eProgressStopped) {
731  break;
732  }
733  }
734 
735  return ok;
736 }
