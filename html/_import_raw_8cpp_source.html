97  TrackFactory *trackFactory, Track ***outTracks)
98 {
99  int encoding = 0; // Guess Format
100  int numChannels = 0;
101  sampleFormat format;
102  sf_count_t offset = 0;
103  sampleCount totalFrames;
104  double rate = 44100.0;
105  double percent = 100.0;
106  SNDFILE *sndFile = NULL;
107  SF_INFO sndInfo;
108  int result;
109 
110  try {
111  // Yes, FormatClassifier currently handles filenames in UTF8 format only, that's
112  // a TODO ...
113  FormatClassifier theClassifier(fileName.utf8_str());
114  encoding = theClassifier.GetResultFormatLibSndfile();
115  numChannels = theClassifier.GetResultChannels();
116  offset = 0;
117  } catch (...) {
118  // Something went wrong in FormatClassifier, use defaults instead.
119  encoding = 0;
120  }
121 
122  if (encoding <= 0) {
123  // Unable to guess. Use mono, 16-bit samples with CPU endianness
124  // as the default.
125  encoding = SF_FORMAT_RAW | SF_ENDIAN_CPU | SF_FORMAT_PCM_16;
126  numChannels = 1;
127  offset = 0;
128  }
129 
130  ImportRawDialog dlog(parent, encoding, numChannels, (int)offset, rate);
131  dlog.ShowModal();
132  if (!dlog.GetReturnCode())
133  return false;
134 
135  encoding = dlog.mEncoding;
136  numChannels = dlog.mChannels;
137  rate = dlog.mRate;
138  offset = (sf_count_t)dlog.mOffset;
139  percent = dlog.mPercent;
140 
141  memset(&sndInfo, 0, sizeof(SF_INFO));
142  sndInfo.samplerate = (int)rate;
143  sndInfo.channels = (int)numChannels;
144  sndInfo.format = encoding | SF_FORMAT_RAW;
145 
146  wxFile f; // will be closed when it goes out of scope
147 
148  if (f.Open(fileName)) {
149  // Even though there is an sf_open() that takes a filename, use the one that
150  // takes a file descriptor since wxWidgets can open a file with a Unicode name and
151  // libsndfile can't (under Windows).
152  sndFile = sf_open_fd(f.fd(), SFM_READ, &sndInfo, FALSE);
153  }
154 
155  if (!sndFile){
156  // TODO: Handle error
157  char str[1000];
158  sf_error_str((SNDFILE *)NULL, str, 1000);
159  printf("%s\n", str);
160 
161  return 0;
162  }
163 
164  result = sf_command(sndFile, SFC_SET_RAW_START_OFFSET, &offset, sizeof(offset));
165  if (result != 0) {
166  char str[1000];
167  sf_error_str(sndFile, str, 1000);
168  printf("%s\n", str);
169  }
170 
171  sf_seek(sndFile, 0, SEEK_SET);
172 
173  totalFrames = (sampleCount)(sndInfo.frames * percent / 100.0);
174 
175  //
176  // Sample format:
177  //
178  // In general, go with the user's preferences. However, if
179  // the file is higher-quality, go with a format which preserves
180  // the quality of the original file.
181  //
182 
183  format = (sampleFormat)
184  gPrefs->Read(wxT("/SamplingRate/DefaultProjectSampleFormat"), floatSample);
185 
186  if (format != floatSample &&
187  sf_subtype_more_than_16_bits(encoding))
188  format = floatSample;
189 
190  WaveTrack **channels = new WaveTrack *[numChannels];
191 
192  int c;
193  for (c = 0; c < numChannels; c++) {
194  channels[c] = trackFactory->NewWaveTrack(format, rate);
195 
196  if (numChannels > 1)
197  switch (c) {
198  case 0:
199  channels[c]->SetChannel(Track::LeftChannel);
200  break;
201  case 1:
202  channels[c]->SetChannel(Track::RightChannel);
203  break;
204  default:
205  channels[c]->SetChannel(Track::MonoChannel);
206  }
207  }
208 
209  if (numChannels == 2) {
210  channels[0]->SetLinked(true);
211  }
212 
213  sampleCount maxBlockSize = channels[0]->GetMaxBlockSize();
214  int updateResult = eProgressSuccess;
215 
216  SampleBuffer srcbuffer(maxBlockSize * numChannels, format);
217  SampleBuffer buffer(maxBlockSize, format);
218 
219  sampleCount framescompleted = 0;
220 
221  wxString msg;
222 
223  msg.Printf(_("Importing %s"), wxFileName::FileName(fileName).GetFullName().c_str());
224 
225  /* i18n-hint: 'Raw' means 'unprocessed' here and should usually be tanslated.*/
226  ProgressDialog progress(_("Import Raw"), msg);
227 
228  long block;
229  do {
230  block = maxBlockSize;
231 
232  if (block + framescompleted > totalFrames)
233  block = totalFrames - framescompleted;
234 
235  if (format == int16Sample)
236  block = sf_readf_short(sndFile, (short *)srcbuffer.ptr(), block);
237  else
238  block = sf_readf_float(sndFile, (float *)srcbuffer.ptr(), block);
239 
240  if (block) {
241  for(c=0; c<numChannels; c++) {
242  if (format==int16Sample) {
243  for(int j=0; j<block; j++)
244  ((short *)buffer.ptr())[j] =
245  ((short *)srcbuffer.ptr())[numChannels*j+c];
246  }
247  else {
248  for(int j=0; j<block; j++)
249  ((float *)buffer.ptr())[j] =
250  ((float *)srcbuffer.ptr())[numChannels*j+c];
251  }
252 
253  channels[c]->Append(buffer.ptr(), format, block);
254  }
255  framescompleted += block;
256  }
257 
258  updateResult = progress.Update((wxULongLong_t)framescompleted,
259  (wxULongLong_t)totalFrames);
260  if (updateResult != eProgressSuccess)
261  break;
262 
263  } while (block > 0 && framescompleted < totalFrames);
264 
265  sf_close(sndFile);
266 
267  int res = updateResult;
268  if (block < 0)
269  res = eProgressFailed;
270 
271  if (res == eProgressFailed || res == eProgressCancelled) {
272  for (c = 0; c < numChannels; c++)
273  delete channels[c];
274  delete[] channels;
275 
276  // It's a shame we can't return proper error code
277  return 0;
278  }
279 
280  for (c = 0; c < numChannels; c++)
281  channels[c]->Flush();
282  *outTracks = (Track **)channels;
283 
284  return numChannels;
285 }
