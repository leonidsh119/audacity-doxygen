249  float lower = mParameters[p].minValue;

260  good = parms.Read(key, &val);

401 bool VampEffect::Process()
402 {
403  if (!mPlugin)
404  {
405  return false;
406  }
407 
408  TrackListOfKindIterator iter(Track::Wave, mTracks);
409 
410  int count = 0;
411 
412  WaveTrack *left = (WaveTrack *)iter.First();
413 
414  bool multiple = false;
415  int prevTrackChannels = 0;
416 
417  if (GetNumWaveGroups() > 1)
418  {
419  // if there is another track beyond this one and any linked one,
420  // then we're processing more than one track. That means we
421  // should use the originating track name in each NEW label
422  // track's name, to make clear which is which
423  multiple = true;
424  }
425 
426  std::vector<std::shared_ptr<Effect::AddedAnalysisTrack>> addedTracks;
427 
428  while (left)
429  {
430  sampleCount lstart, rstart = 0;
431  sampleCount len;
432  GetSamples(left, &lstart, &len);
433 
434  WaveTrack *right = NULL;
435  int channels = 1;
436 
437  if (left->GetLinked())
438  {
439  right = (WaveTrack *)iter.Next();
440  channels = 2;
441  GetSamples(right, &rstart, &len);
442  }
443 
444  size_t step = mPlugin->getPreferredStepSize();
445  size_t block = mPlugin->getPreferredBlockSize();
446 
447  bool initialiseRequired = true;
448 
449  if (block == 0)
450  {
451  if (step != 0)
452  {
453  block = step;
454  }
455  else
456  {
457  block = 1024;
458  }
459  }
460 
461  if (step == 0)
462  {
463  step = block;
464  }
465 
466  if (prevTrackChannels > 0)
467  {
468  // Plugin has already been initialised, so if the number of
469  // channels remains the same, we only need to do a reset.
470  // Otherwise we need to re-construct the whole plugin,
471  // because a Vamp plugin can't be re-initialised.
472  if (prevTrackChannels == channels)
473  {
474  mPlugin->reset();
475  initialiseRequired = false;
476  }
477  else
478  {
480  Init();
481  }
482  }
483 
484  if (initialiseRequired)
485  {
486  if (!mPlugin->initialise(channels, step, block))
487  {
488  wxMessageBox(_("Sorry, Vamp Plug-in failed to initialize."));
489  return false;
490  }
491  }
492 
493  addedTracks.push_back(AddAnalysisTrack(
494  multiple
495  ? wxString::Format(wxT("%s: %s"),
496  left->GetName().c_str(), GetName().c_str())
497  : GetName()
498  ));
499  LabelTrack *ltrack = addedTracks.back()->get();
500 
501  float **data = new float *[channels]; // ANSWER-ME: Vigilant Sentry marks this as memory leak, var "data" not deleted.
502  for (int c = 0; c < channels; ++c)
503  {
504  data[c] = new float[block];
505  }
506 
507  sampleCount originalLen = len;
508  sampleCount ls = lstart;
509  sampleCount rs = rstart;
510 
511  while (len)
512  {
513  int request = block;
514  if (request > len) request = len;
515 
516  if (left)
517  {
518  left->Get((samplePtr)data[0], floatSample, ls, request);
519  }
520 
521  if (right)
522  {
523  right->Get((samplePtr)data[1], floatSample, rs, request);
524  }
525 
526  if (request < (int)block)
527  {
528  for (int c = 0; c < channels; ++c)
529  {
530  for (int i = request; i < (int)block; ++i)
531  {
532  data[c][i] = 0.f;
533  }
534  }
535  }
536 
537  Vamp::RealTime timestamp = Vamp::RealTime::frame2RealTime(ls, (int)(mRate + 0.5));
538 
539  Vamp::Plugin::FeatureSet features = mPlugin->process(data, timestamp);
540  AddFeatures(ltrack, features);
541 
542  if (len > (int)step)
543  {
544  len -= step;
545  }
546  else
547  {
548  len = 0;
549  }
550 
551  ls += step;
552  rs += step;
553 
554  if (channels > 1)
555  {
556  if (TrackGroupProgress(count, (ls - lstart) / double(originalLen)))
557  {
558  return false;
559  }
560  }
561  else
562  {
563  if (TrackProgress(count, (ls - lstart) / double(originalLen)))
564  {
565  return false;
566  }
567  }
568  }
569 
570  Vamp::Plugin::FeatureSet features = mPlugin->getRemainingFeatures();
571  AddFeatures(ltrack, features);
572 
573  prevTrackChannels = channels;
574 
575  left = (WaveTrack *)iter.Next();
576  }
577 
578  // All completed without cancellation, so commit the addition of tracks now
579  for (auto &addedTrack : addedTracks)
580  addedTrack->Commit();
581 
582  return true;
583 }

796
