221 int ExportFLAC::Export(AudacityProject *project,
222  int numChannels,
223  const wxString &fName,
224  bool selectionOnly,
225  double t0,
226  double t1,
227  MixerSpec *mixerSpec,
228  const Tags *metadata,
229  int WXUNUSED(subformat))
230 {
231  double rate = project->GetRate();
232  const TrackList *tracks = project->GetTracks();
233 
234  wxLogNull logNo; // temporarily disable wxWidgets error messages
235  int updateResult = eProgressSuccess;
236 
237  int levelPref;
238  gPrefs->Read(wxT("/FileFormats/FLACLevel"), &levelPref, 5);
239 
240  wxString bitDepthPref =
241  gPrefs->Read(wxT("/FileFormats/FLACBitDepth"), wxT("16"));
242 
243  FLAC::Encoder::File encoder;
244 
245 #ifdef LEGACY_FLAC
246  encoder.set_filename(OSOUTPUT(fName));
247 #endif
248  encoder.set_channels(numChannels);
249  encoder.set_sample_rate(lrint(rate));
250 
251  // See note in GetMetadata() about a bug in libflac++ 1.1.2
252  if (!GetMetadata(project, metadata)) {
253  return false;
254  }
255 
256  if (mMetadata) {
257  encoder.set_metadata(&mMetadata, 1);
258  }
259 
260  sampleFormat format;
261  if (bitDepthPref == wxT("24")) {
262  format = int24Sample;
263  encoder.set_bits_per_sample(24);
264  } else { //convert float to 16 bits
265  format = int16Sample;
266  encoder.set_bits_per_sample(16);
267  }
268 
269  // Duplicate the flac command line compression levels
270  if (levelPref < 0 || levelPref > 8) {
271  levelPref = 5;
272  }
273  encoder.set_do_exhaustive_model_search(flacLevels[levelPref].do_exhaustive_model_search);
274  encoder.set_do_escape_coding(flacLevels[levelPref].do_escape_coding);
275  if (numChannels != 2) {
276  encoder.set_do_mid_side_stereo(false);
277  encoder.set_loose_mid_side_stereo(false);
278  }
279  else {
280  encoder.set_do_mid_side_stereo(flacLevels[levelPref].do_mid_side_stereo);
281  encoder.set_loose_mid_side_stereo(flacLevels[levelPref].loose_mid_side_stereo);
282  }
283  encoder.set_qlp_coeff_precision(flacLevels[levelPref].qlp_coeff_precision);
284  encoder.set_min_residual_partition_order(flacLevels[levelPref].min_residual_partition_order);
285  encoder.set_max_residual_partition_order(flacLevels[levelPref].max_residual_partition_order);
286  encoder.set_rice_parameter_search_dist(flacLevels[levelPref].rice_parameter_search_dist);
287  encoder.set_max_lpc_order(flacLevels[levelPref].max_lpc_order);
288 
289 #ifdef LEGACY_FLAC
290  encoder.init();
291 #else
292  wxFFile f; // will be closed when it goes out of scope
293  if (!f.Open(fName, wxT("w+b"))) {
294  wxMessageBox(wxString::Format(_("FLAC export couldn't open %s"), fName.c_str()));
295  return false;
296  }
297 
298  // Even though there is an init() method that takes a filename, use the one that
299  // takes a file handle because wxWidgets can open a file with a Unicode name and
300  // libflac can't (under Windows).
301  int status = encoder.init(f.fp());
302  if (status != FLAC__STREAM_ENCODER_INIT_STATUS_OK) {
303  wxMessageBox(wxString::Format(_("FLAC encoder failed to initialize\nStatus: %d"), status));
304  return false;
305  }
306 #endif
307 
308  if (mMetadata) {
309  ::FLAC__metadata_object_delete(mMetadata);
310  }
311 
312  const WaveTrackConstArray waveTracks =
313  tracks->GetWaveTrackConstArray(selectionOnly, false);
314  Mixer *mixer = CreateMixer(waveTracks,
315  tracks->GetTimeTrack(),
316  t0, t1,
317  numChannels, SAMPLES_PER_RUN, false,
318  rate, format, true, mixerSpec);
319 
320  int i, j;
321  FLAC__int32 **tmpsmplbuf = new FLAC__int32*[numChannels];
322  for (i = 0; i < numChannels; i++) {
323  tmpsmplbuf[i] = (FLAC__int32 *) calloc(SAMPLES_PER_RUN, sizeof(FLAC__int32));
324  }
325 
326  {
327  ProgressDialog progress(wxFileName(fName).GetName(),
328  selectionOnly ?
329  _("Exporting the selected audio as FLAC") :
330  _("Exporting the entire project as FLAC"));
331 
332  while (updateResult == eProgressSuccess) {
333  sampleCount samplesThisRun = mixer->Process(SAMPLES_PER_RUN);
334  if (samplesThisRun == 0) { //stop encoding
335  break;
336  }
337  else {
338  for (i = 0; i < numChannels; i++) {
339  samplePtr mixed = mixer->GetBuffer(i);
340  if (format == int24Sample) {
341  for (j = 0; j < samplesThisRun; j++) {
342  tmpsmplbuf[i][j] = ((int *)mixed)[j];
343  }
344  }
345  else {
346  for (j = 0; j < samplesThisRun; j++) {
347  tmpsmplbuf[i][j] = ((short *)mixed)[j];
348  }
349  }
350  }
351  encoder.process(tmpsmplbuf, samplesThisRun);
352  }
353  updateResult = progress.Update(mixer->MixGetCurrentTime() - t0, t1 - t0);
354  }
355  f.Detach(); // libflac closes the file
356  encoder.finish();
357  }
358 
359  for (i = 0; i < numChannels; i++) {
360  free(tmpsmplbuf[i]);
361  }
362  delete mixer;
363 
364  delete[] tmpsmplbuf;
365 
366  return updateResult;
367 }
