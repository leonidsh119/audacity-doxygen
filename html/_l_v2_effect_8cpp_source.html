459 bool LV2Effect::SetHost(EffectHostInterface *host)
460 {
461  mHost = host;
462 
463  int numPorts = lilv_plugin_get_num_ports(mPlug);
464 
465  // Fail if we don't grok the port types
466  for (int i = 0; i < numPorts; i++)
467  {
468  const LilvPort *port = lilv_plugin_get_port_by_index(mPlug, i);
469 
470  if (!lilv_port_is_a(mPlug, port, gAudio) &&
471  !lilv_port_is_a(mPlug, port, gControl))
472  {
473  return false;
474  }
475  }
476 
477  // Allocate buffers for the port indices and the default control values
478  float *minimumVals = new float [numPorts];
479  float *maximumVals = new float [numPorts];
480  float *defaultValues = new float [numPorts];
481 
482  // Retrieve the port ranges for all ports (some values may be NaN)
483  lilv_plugin_get_port_ranges_float(mPlug,
484  minimumVals,
485  maximumVals,
486  defaultValues);
487 
488  // Get info about all ports
489  for (int i = 0; i < numPorts; i++)
490  {
491  const LilvPort *port = lilv_plugin_get_port_by_index(mPlug, i);
492  int index = lilv_port_get_index(mPlug, port);
493 
494  // Quick check for audio ports
495  if (lilv_port_is_a(mPlug, port, gAudio))
496  {
497  if (lilv_port_is_a(mPlug, port, gInput))
498  {
499  mAudioInputs.Add(index);
500  }
501  else if (lilv_port_is_a(mPlug, port, gOutput))
502  {
503  mAudioOutputs.Add(index);
504  }
505  continue;
506  }
507 
508  // Only control ports from this point
509  if (!lilv_port_is_a(mPlug, port, gControl))
510  {
511  continue;
512  }
513 
514  LV2Port ctrl;
515  ctrl.mIndex = index;
516 
517  // Get the port name
518  ctrl.mSymbol = LilvString(lilv_port_get_symbol(mPlug, port));
519  LilvNode *tmpName = lilv_port_get_name(mPlug, port);
520  ctrl.mName = LilvString(tmpName);
521  lilv_node_free(tmpName);
522 
523  // Get any unit descriptor
524  LilvNode *unit = lilv_port_get(mPlug, port, gUnit);
525  if (unit)
526  {
527  ctrl.mUnits = LilvString(lilv_world_get(gWorld, unit, gUnitSymbol, NULL));
528  }
529 
530  // Get the group to which this port belongs or default to the main group
531  ctrl.mGroup = wxEmptyString;
532  LilvNode *group = lilv_port_get(mPlug, port, gGroup);
533  if (group)
534  {
535  ctrl.mGroup = LilvString(lilv_world_get(gWorld, group, gLabel, NULL));
536  if (ctrl.mGroup.IsEmpty())
537  {
538  ctrl.mGroup = LilvString(lilv_world_get(gWorld, group, gName, NULL));
539  }
540 
541  if (ctrl.mGroup.IsEmpty())
542  {
543  ctrl.mGroup = LilvString(group);
544  }
545  }
546 
547  // Add it if not previously done
548  if (mGroups.Index(ctrl.mGroup) == wxNOT_FOUND)
549  {
550  mGroups.Add(ctrl.mGroup);
551  }
552 
553  // Get the scale points
554  LilvScalePoints *points = lilv_port_get_scale_points(mPlug, port);
555  LILV_FOREACH(scale_points, j, points)
556  {
557  const LilvScalePoint *point = lilv_scale_points_get(points, j);
558 
559  ctrl.mScaleValues.Add(lilv_node_as_float(lilv_scale_point_get_value(point)));
560  ctrl.mScaleLabels.Add(LilvString(lilv_scale_point_get_label(point)));
561  }
562  lilv_scale_points_free(points);
563 
564  // Collect the value and range info
565  ctrl.mHasLo = !isnan(minimumVals[i]);
566  ctrl.mHasHi = !isnan(maximumVals[i]);
567  ctrl.mMin = ctrl.mHasLo ? minimumVals[i] : 0.0;
568  ctrl.mMax = ctrl.mHasHi ? maximumVals[i] : 1.0;
569  ctrl.mLo = ctrl.mMin;
570  ctrl.mHi = ctrl.mMax;
571  ctrl.mDef = !isnan(defaultValues[i]) ?
572  defaultValues[i] :
573  ctrl.mHasLo ?
574  ctrl.mLo :
575  ctrl.mHasHi ?
576  ctrl.mHi :
577  0.0;
578  ctrl.mVal = ctrl.mDef;
579 
580  // Figure out the type of port we have
581  if (lilv_port_is_a(mPlug, port, gInput))
582  {
583  ctrl.mInput = true;
584  if (lilv_port_has_property(mPlug, port, gToggled))
585  {
586  ctrl.mToggle = true;
587  }
588  else if (lilv_port_has_property(mPlug, port, gEnumeration))
589  {
590  ctrl.mEnumeration = true;
591  }
592  else if (lilv_port_has_property(mPlug, port, gInteger))
593  {
594  ctrl.mInteger = true;
595  }
596  else if (lilv_port_has_property(mPlug, port, gSampleRate))
597  {
598  ctrl.mSampleRate = true;
599  }
600 
601  // Trigger properties can be combined with other types, but it
602  // seems mostly to be combined with toggle. So, we turn the
603  // checkbox into a button.
604  if (lilv_port_has_property(mPlug, port, gTrigger))
605  {
606  ctrl.mTrigger = true;
607  }
608 
609  // We'll make the slider logarithmic
610  if (lilv_port_has_property(mPlug, port, gLogarithmic))
611  {
612  ctrl.mLogarithmic = true;
613  }
614 
615  if (lilv_port_has_property(mPlug, port, gEnumeration))
616  {
617  ctrl.mEnumeration = true;
618  }
619 
620  mControlsMap[ctrl.mIndex] = mControls.GetCount();
621  mGroupMap[ctrl.mGroup].Add(mControls.GetCount());
622  mControls.Add(ctrl);
623  }
624  else if (lilv_port_is_a(mPlug, port, gOutput))
625  {
626  ctrl.mInput = false;
627  if (lilv_port_has_property(mPlug, port, gLatency))
628  {
629  mLatencyPort = i;
630  }
631  else
632  {
633  mGroupMap[ctrl.mGroup].Add(mControls.GetCount());
634  mControls.Add(ctrl);
635  }
636  }
637  else
638  {
639  // Just ignore it for now
640  }
641  }
642 
643  delete [] minimumVals;
644  delete [] maximumVals;
645  delete [] defaultValues;
646 
647  // mHost will be null during registration
648  if (mHost)
649  {
650  mHost->GetSharedConfig(wxT("Settings"), wxT("UseLatency"), mUseLatency, true);
651  mHost->GetSharedConfig(wxT("Settings"), wxT("UseGUI"), mUseGUI, true);
652 
653  bool haveDefaults;
654  mHost->GetPrivateConfig(mHost->GetFactoryDefaultsGroup(), wxT("Initialized"), haveDefaults, false);
655  if (!haveDefaults)
656  {
657  SaveParameters(mHost->GetFactoryDefaultsGroup());
658  mHost->SetPrivateConfig(mHost->GetFactoryDefaultsGroup(), wxT("Initialized"), true);
659  }
660 
661  LoadParameters(mHost->GetCurrentSettingsGroup());
662  }
663 
664  AddOption(LV2_BUF_SIZE__minBlockLength,
665  sizeof(mBlockSize),
