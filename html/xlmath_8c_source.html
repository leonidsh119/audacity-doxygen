40 /* xgcd - greatest common divisor */
41 LVAL xgcd(void)
42 {
43  FIXTYPE m,n,r;
44  LVAL arg;
45 
46  if (!moreargs()) /* check for identity case */
47  return (cvfixnum((FIXTYPE)0));
48  arg = xlgafixnum();
49  n = getfixnum(arg);
50  if (n < (FIXTYPE)0) n = -n; /* absolute value */
51  while (moreargs()) {
52  arg = xlgafixnum();
53  m = getfixnum(arg);
54  if (m < (FIXTYPE)0) m = -m; /* absolute value */
55  for (;;) { /* euclid's algorithm */
56  r = m % n;
57  if (r == (FIXTYPE)0)
58  break;
59  m = n;
60  n = r;
61  }
62  }
63  return (cvfixnum(n));
64 }

91  switch (fcn) {
92  case '-':
93  switch (mode) {
94  case 'I':

96  break;

99  break;
100  }
101  break;
102  case '/':
103  switch (mode) {
104  case 'I':

106  ival = 1 / ival;
107  break;
108  case 'F':
109  checkfzero(fval);

111  break;
112  }
113  break;
114  case 'A':

116  case 'I':
117  mode = 'F';
118  fval = ival;
119  case 'F':
120  fval = atan(fval);
121  break;
122  }
123  break;
124  }
125  }
126 
127  /* handle each remaining argument */
128  while (moreargs()) {
129 
130  /* get the next argument */
131  arg = xlgetarg();
132 
133  /* check its type */
134  if (fixp(arg)) {
135  switch (mode) {
136  case 'I':
137  iarg = getfixnum(arg);
138  break;
139  case 'F':
140  farg = (FLOTYPE)getfixnum(arg);
141  break;
142  }
143  }
144  else if (floatp(arg)) {
145  switch (mode) {
146  case 'I':
147  fval = (FLOTYPE)ival;
148  farg = getflonum(arg);
149  mode = 'F';
150  break;
151  case 'F':
152  farg = getflonum(arg);
153  break;
154  }
155  }

163  case '+': ival += iarg; break;
164  case '-': ival -= iarg; break;
165  case '*': ival *= iarg; break;
166  case '/': checkizero(iarg); ival /= iarg; break;
167  case '%': checkizero(iarg); ival %= iarg; break;
168  case 'M': if (iarg > ival) ival = iarg; break;
169  case 'm': if (iarg < ival) ival = iarg; break;
170  case '&': ival &= iarg; break;
171  case '|': ival |= iarg; break;
172  case '^': ival ^= iarg; break;
173  case 'A': fval = atan2((double) ival, (double) iarg);
174  mode = 'F';
175  xllastarg();
176  break;
177  default: badiop();
178  }
179  break;
180  case 'F':
181  switch (fcn) {
182  case '+': fval += farg; break;
183  case '-': fval -= farg; break;
184  case '*': fval *= farg; break;
185  case '/': checkfzero(farg); fval /= farg; break;
186  case 'M': if (farg > fval) fval = farg; break;
187  case 'm': if (farg < fval) fval = farg; break;
188  case 'E': fval = pow(fval,farg); break;
189  case 'A': fval = atan2(fval, farg);
190  xllastarg();
191  break;
192  default: badfop();
193  }
194  break;
195  }
196  }
197 
198  /* return the result */
199  switch (mode) {
200  case 'I': return (cvfixnum(ival));
201  case 'F': return (cvflonum(fval));
202  }
203 
204  /* This shouldn't fall through, but just in case, this will
205  catch it and make the compiler happy... */
206  xlerror("bad argument type",arg);
207  return NULL;
208 }

253 LOCAL LVAL unary(int fcn)
254 {
255  FLOTYPE fval;
256  FIXTYPE ival;
257  LVAL arg;
258 
259  /* get the argument */
260  arg = xlgetarg();
261  xllastarg();
262 
263  /* check its type */
264  if (fixp(arg)) {
265  ival = getfixnum(arg);
266  switch (fcn) {
267  case '~': ival = ~ival; break;
268  case 'A': ival = (ival < 0 ? -ival : ival); break;
269  case '+': ival++; break;
270  case '-': ival--; break;
271  case 'I': break;
272  case 'F': return (cvflonum((FLOTYPE)ival));
273  case '?': ival = (FIXTYPE)xlrand((int)ival); break;
274  default: badiop();
275  }
276  return (cvfixnum(ival));
277  }
278  else if (floatp(arg)) {
279  fval = getflonum(arg);
280  switch (fcn) {
281  case 'A': fval = (fval < 0.0 ? -fval : fval); break;
282  case '+': fval += 1.0; break;
283  case '-': fval -= 1.0; break;
284  case 'S': fval = sin(fval); break;
285  case 'C': fval = cos(fval); break;
286  case 'T': fval = tan(fval); break;
287  case 'E': fval = exp(fval); break;
288  case 'R': checkfneg(fval); fval = sqrt(fval); break;
289  case 'I': return (cvfixnum((FIXTYPE)fval));
290  case 'F': break;
291  default: badfop();
