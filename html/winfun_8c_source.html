155 void get_xlisp_path(char *p, long p_max)
156 {
157  HKEY hkey;
158  DWORD dwType;
159  LVAL lval;
160  extern LVAL s_search_path;
161 
162  *p = 0; /* for simplicity, we assume if !*p that path was not found */
163  /* therefore, no path is equivalent to an empty string path */
164 
165  /* first, look for path in global variable *SEARCH-PATH* */
166  lval = getvalue(s_search_path);
167  if (lval && stringp(lval)) {
168  strncpy(p, getstring(lval), p_max);
169  p[p_max - 1] = 0; /* make sure string is terminated, even if truncated */
170  }
171  if (*p) return; /* we got search path, so don't look in registry */
172 
173  if (RegOpenKeyEx(HKEY_LOCAL_MACHINE, "SOFTWARE", 0, KEY_READ, &hkey) !=
174  ERROR_SUCCESS) {
175  return;
176  }
177  if (RegOpenKeyEx(hkey, "CMU", 0, KEY_READ, &hkey) !=
178  ERROR_SUCCESS) {
179  return;
180  }
181  if (RegOpenKeyEx(hkey, "Nyquist", 0, KEY_READ, &hkey) !=
182  ERROR_SUCCESS) {
183  return;
184  }
185  if (RegQueryValueEx(hkey, "XLISPPATH", NULL, &dwType, p, &p_max) !=
186  ERROR_SUCCESS) {
187  *p = 0;
188  return;
189  }
190 }

210 * char *prompt: prompt for user
211 * Returns:
212 * opened file pointer
213 * Effect:
214 * opens file, prompts for user input if necessary and warns about
