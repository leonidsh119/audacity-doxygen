204 bool LadspaEffectsModule::RegisterPlugin(PluginManagerInterface & pm, const wxString & path)
205 {
206  // Since we now have builtin VST support, ignore the VST bridge as it
207  // causes duplicate menu entries to appear.
208  wxFileName f(path);
209  if (f.GetName().CmpNoCase(wxT("vst-bridge")) == 0) {
210  return false;
211  }
212 
213  // As a courtesy to some plug-ins that might be bridges to
214  // open other plug-ins, we set the current working
215  // directory to be the plug-in's directory.
216  wxString envpath;
217  bool hadpath = wxGetEnv(wxT("PATH"), &envpath);
218  wxSetEnv(wxT("PATH"), f.GetPath() + wxFILE_SEP_PATH + envpath);
219  wxString saveOldCWD = f.GetCwd();
220  f.SetCwd();
221 
222  int index = 0;
223  LADSPA_Descriptor_Function mainFn = NULL;
224  wxDynamicLibrary lib;
225  if (lib.Load(path, wxDL_NOW)) {
226  wxLogNull logNo;
227 
228  mainFn = (LADSPA_Descriptor_Function) lib.GetSymbol(wxT("ladspa_descriptor"));
229  if (mainFn) {
230  const LADSPA_Descriptor *data;
231 
232  for (data = mainFn(index); data; data = mainFn(++index)) {
233  LadspaEffect effect(path, index);
234  if (effect.SetHost(NULL)) {
235  pm.RegisterPlugin(this, &effect);
236  }
237  }
238  }
239  }
240 
241  if (lib.IsLoaded()) {
242  lib.Unload();
243  }
244 
245  wxSetWorkingDirectory(saveOldCWD);
246  hadpath ? wxSetEnv(wxT("PATH"), envpath) : wxUnsetEnv(wxT("PATH"));
247 
248  return index > 0;
249 }

700 bool LadspaEffect::SetHost(EffectHostInterface *host)
701 {
702  mHost = host;
703 
704  if (!Load())
705  {
706  return false;
707  }
708 
709  mInputPorts = new unsigned long [mData->PortCount];
710  mOutputPorts = new unsigned long [mData->PortCount];
711  mInputControls = new float [mData->PortCount];
712  mOutputControls = new float [mData->PortCount];
713 
714  for (unsigned long p = 0; p < mData->PortCount; p++)
715  {
716  LADSPA_PortDescriptor d = mData->PortDescriptors[p];
717 
718  // Collect the audio ports
719  if (LADSPA_IS_PORT_AUDIO(d))
720  {
721  if (LADSPA_IS_PORT_INPUT(d))
722  {
723  mInputPorts[mAudioIns++] = p;
724  }
725  else if (LADSPA_IS_PORT_OUTPUT(d))
726  {
727  mOutputPorts[mAudioOuts++] = p;
728  }
729  }
730  // Determine the port's default value
731  else if (LADSPA_IS_PORT_CONTROL(d) && LADSPA_IS_PORT_INPUT(d))
732  {
733  mInteractive = true;
734 
735  LADSPA_PortRangeHint hint = mData->PortRangeHints[p];
736  float val = float(1.0);
737  float lower = hint.LowerBound;
738  float upper = hint.UpperBound;
739 
740  if (LADSPA_IS_HINT_SAMPLE_RATE(hint.HintDescriptor))
741  {
742  lower *= mSampleRate;
743  upper *= mSampleRate;
744  }
745 
746  if (LADSPA_IS_HINT_BOUNDED_BELOW(hint.HintDescriptor) && val < lower)
747  {
748  val = lower;
749  }
750 
751  if (LADSPA_IS_HINT_BOUNDED_ABOVE(hint.HintDescriptor) && val > upper)
752  {
753  val = upper;
754  }
755 
756  if (LADSPA_IS_HINT_DEFAULT_MINIMUM(hint.HintDescriptor))
757  {
758  val = lower;
759  }
760 
761  if (LADSPA_IS_HINT_DEFAULT_MAXIMUM(hint.HintDescriptor))
762  {
763  val = upper;
764  }
765 
766  if (LADSPA_IS_HINT_DEFAULT_LOW(hint.HintDescriptor))
767  {
768  if (LADSPA_IS_HINT_LOGARITHMIC(hint.HintDescriptor))
769  {
770  val = exp(log(lower) * 0.75f + log(upper) * 0.25f);
771  }
772  else
773  {
774  val = lower * 0.75f + upper * 0.25f;
775  }
776  }
777 
778  if (LADSPA_IS_HINT_DEFAULT_MIDDLE(hint.HintDescriptor))
779  {
780  if (LADSPA_IS_HINT_LOGARITHMIC(hint.HintDescriptor))
781  {
782  val = exp(log(lower) * 0.5f + log(upper) * 0.5f);
783  }
784  else
785  {
786  val = lower * 0.5f + upper * 0.5f;
787  }
788  }
789 
790  if (LADSPA_IS_HINT_DEFAULT_HIGH(hint.HintDescriptor))
791  {
792  if (LADSPA_IS_HINT_LOGARITHMIC(hint.HintDescriptor))
793  {
794  val = exp(log(lower) * 0.25f + log(upper) * 0.75f);
795  }
796  else
797  {
798  val = lower * 0.25f + upper * 0.75f;
799  }
800  }
801 
802  if (LADSPA_IS_HINT_DEFAULT_0(hint.HintDescriptor))
803  {
804  val = 0.0f;
805  }
806 
807  if (LADSPA_IS_HINT_DEFAULT_1(hint.HintDescriptor))
808  {
809  val = 1.0f;
810  }
811 
812  if (LADSPA_IS_HINT_DEFAULT_100(hint.HintDescriptor))
813  {
814  val = 100.0f;
815  }
816 
817  if (LADSPA_IS_HINT_DEFAULT_440(hint.HintDescriptor))
818  {
819  val = 440.0f;
820  }
821 
822  mNumInputControls++;
823  mInputControls[p] = val;
