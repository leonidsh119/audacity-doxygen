37 /****************************************************************
38 PSYCHO_4 by MFC Feb 2003
39 
40 This is a cleaned up implementation of psy model 2.
41 This is basically because I was sick of the inconsistencies between
42 the notation in the ISO docs and in the sourcecode.
43 
44 I've nicked a bunch of stuff from LAME to make this a bit easier to grok
45 - ATH values (this also overcomes the lack of mpeg-2 tables
46  which meant that LSF never had proper values)
47 - ath_freq2bark() to convert frequencies directly to bark values.
48 - spreading_function() isolated the calculation of the spreading function.
49  Basically the same code as before, just isolated in its own function.
50  LAME seem to does some extra tweaks to the ISO1117s model.
51  Not really sure if they help or hinder, so I've commented them out (#ifdef LAME)
52 
53 NB: Because of some of the tweaks to bark value calculation etc, it is now possible
54 to have 64 CBANDS. There's no real limit on the actual number of paritions.
55 I wonder if it's worth experimenting with really higher numbers? Probably won't make
56 that much difference to the final SNR values, but it's something worth trying
57  Maybe CBANDS should be a dynamic value, calculated by the psycho_init function
58  CBANDS definition has been changed in encoder.h from 63 to 64

91 /* Table covers angles from 0 to TRIGTABLESIZE/TRIGTABLESCALE (3.142) radians
92  In steps of 1/TRIGTABLESCALE (0.0005) radians.
93  Largest absolute error: 0.0005
94  Only create a table for cos, and then use trig to work out sin.
95  sin(theta) = cos(PI/2 - theta)
96  MFC March 2003 */
97 static void psycho_4_trigtable_init(psycho_4_mem * p4mem)
98 {

103  }

106 #ifdef NEWTAN
107 static inline FLOAT psycho_4_cos(psycho_4_mem * p4mem, FLOAT phi)
108 {
109  int index;
110  int sign = 1;
111 
112  index = (int) (fabs(phi) * TRIGTABLESCALE);
113  while (index >= TRIGTABLESIZE) {
114  /* If we're larger than PI, then subtract PI until we aren't each time the sign will flip -
115  Year 11 trig again. MFC March 2003 */
116  index -= TRIGTABLESIZE;
117  sign *= -1;
118  }
119  return (sign * p4mem->cos_table[index]);
120 }

123 /* The spreading function. Values returned in units of energy
124  Argument 'bark' is the difference in bark values between the
125  centre of two partitions.
126  This has been taken from LAME. MFC Feb 2003 */
127 static FLOAT psycho_4_spreading_function(FLOAT bark)
128 {
129 
130  FLOAT tempx, x, tempy, temp;
131  tempx = bark;
132 #ifdef LAME
133  /* MP3 standard actually spreads these values a little more */
134  if (tempx >= 0)
135  tempx *= 3;
136  else
137  tempx *= 1.5;
138 #endif
139 
140  if (tempx >= 0.5 && tempx <= 2.5) {
141  temp = tempx - 0.5;
142  x = 8.0 * (temp * temp - 2.0 * temp);
143  } else
144  x = 0.0;
145  tempx += 0.474;
146  tempy = 15.811389 + 7.5 * tempx - 17.5 * sqrt(1.0 + tempx * tempx);
147 
148  if (tempy <= -60.0)
149  return 0.0;
150 
151  tempx = exp((x + tempy) * LN_TO_LOG10);
152 
153 #ifdef LAME
154  /* I'm not sure where the magic value of 0.6609193 comes from. twolame will just keep using the
155  rnorm to normalise the spreading function MFC Feb 2003 */
156  /* Normalization. The spreading function should be normalized so that: +inf / | s3 [ bark ]
157  d(bark) = 1 / -inf */
158  tempx /= .6609193;
159 #endif
160  return tempx;
161

167 static psycho_4_mem *psycho_4_init(twolame_options * glopts, int sfreq)
168 {
169  psycho_4_mem *mem;
170  FLOAT *cbval, *rnorm;
171  FLOAT *window;
172  FLOAT bark[HBLKSIZE], *ath;
173  int *numlines;
174  int *partition;
175  FCB *s;
176  FLOAT *tmn;
177  int i, j;
178 
179  {
180  mem = (psycho_4_mem *) TWOLAME_MALLOC(sizeof(psycho_4_mem));
181 
182  mem->tmn = (FLOAT *) TWOLAME_MALLOC(sizeof(DCB));
183  mem->s = (FCB *) TWOLAME_MALLOC(sizeof(FCBCB));
184  mem->lthr = (FHBLK *) TWOLAME_MALLOC(sizeof(F2HBLK));
185  mem->r = (F2HBLK *) TWOLAME_MALLOC(sizeof(F22HBLK));
186  mem->phi_sav = (F2HBLK *) TWOLAME_MALLOC(sizeof(F22HBLK));
187 
188  mem->new = 0;
189  mem->old = 1;
190  mem->oldest = 0;
191  }
192 
193  {
194  cbval = mem->cbval;
195  rnorm = mem->rnorm;
196  window = mem->window;
197  // bark = mem->bark;
198  ath = mem->ath;
199  numlines = mem->numlines;
200  partition = mem->partition;
201  s = mem->s;
202  tmn = mem->tmn;
203  }
204 
205 
206  /* Set up the SIN/COS tables */
207  psycho_4_trigtable_init(mem);
208 
209  /* calculate HANN window coefficients */
210  for (i = 0; i < BLKSIZE; i++)
211  window[i] = 0.5 * (1 - cos(2.0 * PI * (i - 0.5) / BLKSIZE));
212 
213  /* For each FFT line from 0(DC) to 512(Nyquist) calculate - bark : the bark value of this fft
214  line - ath : the absolute threshold of hearing for this line [ATH]
215 
216  Since it is a 1024 point FFT, each line in the fft corresponds to 1/1024 of the total
217  frequency. Line 0 should correspond to DC - which doesn't really have a ATH afaik Line 1
218  should be 1/1024th of the Sampling Freq Line 512 should be the nyquist freq */
219  for (i = 0; i < HBLKSIZE; i++) {
220  FLOAT freq = i * (FLOAT) sfreq / (FLOAT) BLKSIZE;
221  bark[i] = ath_freq2bark(freq);
222  /* The ath tables in the dist10 code seem to be a little out of kilter. they seem to start
223  with index 0 corresponding to (sampling freq)/1024. When in doubt, i'm going to assume
224  that the dist10 code is wrong. MFC Feb2003 */
225  ath[i] = ath_energy(freq, glopts->athlevel);
226  // fprintf(stderr,"%.2f ",ath[i]);
227  }
228 
229 
230  /* Work out the partitions Starting from line 0, all lines within 0.33 of the starting bark are
231  added to the same partition. When a line is greater by 0.33 of a bark, start a new
232  partition. */
233  {
234  int partition_count = 0; /* keep a count of the partitions */
235  int cbase = 0; /* current base index for the bark range calculation */
236  for (i = 0; i < HBLKSIZE; i++) {
237  if ((bark[i] - bark[cbase]) > 0.33) { /* 1/3 critical band? */
238  /* this frequency line is too different from the starting line, (in terms of the
239  bark distance) so close that previous partition, and make this line the first
240  member of the next partition */
241  cbase = i; /* Start the new partition from this frequency */
242  partition_count++;
243  }
244  /* partition[i] tells us which partition the i'th frequency line is in */
245  partition[i] = partition_count;
246  /* keep a count of how many frequency lines are in each partition */
247  numlines[partition_count]++;
248  }
249  }
250 
251  /* For each partition within the frequency space, calculate the average bark value - cbval
252  [central bark value] */
253  for (i = 0; i < HBLKSIZE; i++)
254  cbval[partition[i]] += bark[i]; /* sum up all the bark values */
255  for (i = 0; i < CBANDS; i++) {
256  if (numlines[i] != 0)
257  cbval[i] /= numlines[i]; /* divide by the number of values */
258  else {
259  cbval[i] = 0; /* this isn't a partition */
260  }
261  }
262 
263 
264  /* Calculate the spreading function. ISO 11172 Section D.2.3 */
265  for (i = 0; i < CBANDS; i++) {
266  for (j = 0; j < CBANDS; j++) {
267  s[i][j] = psycho_4_spreading_function(1.05 * (cbval[i] - cbval[j]));
268  rnorm[i] += s[i][j]; /* sum the spreading function values for each partition so that
269  they can be normalised later on */
270  }
271  }
272 
273  /* Calculate Tone Masking Noise values. ISO 11172 Tables D.3.x */
274  for (j = 0; j < CBANDS; j++)
275  tmn[j] = MAX(15.5 + cbval[j], 24.5);
276 
277 
278  if (glopts->verbosity > 6) {
279  /* Dump All the Values to STDERR */
280  int wlow, whigh = 0;
281  int ntot = 0;
282  fprintf(stderr, "psy model 4 init\n");
283  fprintf(stderr, "index \tnlines \twlow \twhigh \tbval \tminval \ttmn\n");
284  for (i = 0; i < CBANDS; i++)
285  if (numlines[i] != 0) {
286  wlow = whigh + 1;
287  whigh = wlow + numlines[i] - 1;
288  fprintf(stderr, "%i \t%i \t%i \t%i \t%5.2f \t%4.2f \t%4.2f\n", i + 1, numlines[i],
289  wlow, whigh, cbval[i], minval[(int) cbval[i]], tmn[i]);
290  ntot += numlines[i];
291  }
292  fprintf(stderr, "total lines %i\n", ntot);
293  }
294 
295  return (mem);
296 }

299 void psycho_4(twolame_options * glopts,
300  short int buffer[2][1152], short int savebuf[2][1056], FLOAT smr[2][32])
301 /* to match prototype : FLOAT args are always FLOAT */
302 {
303  psycho_4_mem *mem;
304  unsigned int run, i, j, k, ch;
305  FLOAT r_prime, phi_prime;
306  FLOAT npart, epart;
307  int new, old, oldest;
308  FLOAT *grouped_c, *grouped_e;
309  FLOAT *nb, *cb, *tb, *ecb, *bc;
310  FLOAT *cbval, *rnorm;
311  FLOAT *wsamp_r, *phi, *energy, *window;
312  FLOAT *ath, *thr, *c;
313 
314  FLOAT *snrtmp[2];
315  int *numlines;
316  int *partition;
317  FLOAT *tmn;
318  FCB *s;
319  FHBLK *lthr;
320  F2HBLK *r, *phi_sav;
321 
322  int nch = glopts->num_channels_out;
323  int sfreq = glopts->samplerate_out;
324 
325  if (!glopts->p4mem) {
326  glopts->p4mem = psycho_4_init(glopts, sfreq);
327  }
328 
329  mem = glopts->p4mem;
330  {
331  grouped_c = mem->grouped_c;
332  grouped_e = mem->grouped_e;
333  nb = mem->nb;
334  cb = mem->cb;
335  tb = mem->tb;
336  ecb = mem->ecb;
337  bc = mem->bc;
338  rnorm = mem->rnorm;
339  cbval = mem->cbval;
340  wsamp_r = mem->wsamp_r;
341  phi = mem->phi;
342  energy = mem->energy;
343  window = mem->window;
344  ath = mem->ath;
345  thr = mem->thr;
346  c = mem->c;
347 
348  snrtmp[0] = mem->snrtmp[0];
349  snrtmp[1] = mem->snrtmp[1];
350 
351  numlines = mem->numlines;
352  partition = mem->partition;
353  tmn = mem->tmn;
354  s = mem->s;
355  lthr = mem->lthr;
356  r = mem->r;
357  phi_sav = mem->phi_sav;
358  }
359 
360  for (ch = 0; ch < nch; ch++) {
361  for (run = 0; run < 2; run++) {
362  /* Net offset is 480 samples (1056-576) for layer 2; this is because one must stagger
363  input data by 256 samples to synchronize psychoacoustic model with filter bank
364  outputs, then stagger so that center of 1024 FFT window lines up with center of 576
365  "new" audio samples.
366 
367  flush = 384*3.0/2.0; = 576 syncsize = 1056; sync_flush = syncsize - flush; 480
368  BLKSIZE = 1024 */
369  {
370  short int *bufferp = buffer[ch];
371  for (j = 0; j < 480; j++) {
372  savebuf[ch][j] = savebuf[ch][j + 576];
373  wsamp_r[j] = window[j] * ((FLOAT) savebuf[ch][j]);
374  }
375  for (; j < 1024; j++) {
376  savebuf[ch][j] = *bufferp++;
377  wsamp_r[j] = window[j] * ((FLOAT) savebuf[ch][j]);
378  }
379  for (; j < 1056; j++)
380  savebuf[ch][j] = *bufferp++;
381  }
382 
383  /* Compute FFT */
384  psycho_2_fft(wsamp_r, energy, phi);
385 
386  /* calculate the unpredictability measure, given energy[f] and phi[f] (the age pointers
387  [new/old/oldest] are reset automatically on the second pass */
388  {
389  if (mem->new == 0) {
390  mem->new = 1;
391  mem->oldest = 1;
392  } else {
393  mem->new = 0;
394  mem->oldest = 0;
395  }
396  if (mem->old == 0)
397  mem->old = 1;
398  else
399  mem->old = 0;
400  }
401  old = mem->old;
402  new = mem->new;
403  oldest = mem->oldest;
404 
405 
406  for (j = 0; j < HBLKSIZE; j++) {
407 #ifdef NEWATAN
408  FLOAT temp1, temp2, temp3;
409  r_prime = 2.0 * r[ch][old][j] - r[ch][oldest][j];
410  phi_prime = 2.0 * phi_sav[ch][old][j] - phi_sav[ch][oldest][j];
411 
412  r[ch][new][j] = sqrt((FLOAT) energy[j]);
413  phi_sav[ch][new][j] = phi[j];
414 
415  {
416  temp1 =
417  r[ch][new][j] * psycho_4_cos(mem, phi[j]) -
418  r_prime * psycho_4_cos(mem, phi_prime);
419  /* Remember your grade 11 trig? sin(theta) = cos(PI/2 - theta) */
420  temp2 =
421  r[ch][new][j] * psycho_4_cos(mem, PI2 - phi[j]) -
422  r_prime * psycho_4_cos(mem, PI2 - phi_prime);
423  }
424 
425 
426  temp3 = r[ch][new][j] + fabs((FLOAT) r_prime);
427  if (temp3 != 0)
428  c[j] = sqrt(temp1 * temp1 + temp2 * temp2) / temp3;
429  else
430  c[j] = 0;
431 #else
432  FLOAT temp1, temp2, temp3;
433  r_prime = 2.0 * r[ch][old][j] - r[ch][oldest][j];
434  phi_prime = 2.0 * phi_sav[ch][old][j] - phi_sav[ch][oldest][j];
435 
436  r[ch][new][j] = sqrt((FLOAT) energy[j]);
437  phi_sav[ch][new][j] = phi[j];
438 
439 
440  temp1 = r[ch][new][j] * cos((FLOAT) phi[j]) - r_prime * cos((FLOAT) phi_prime);
441  temp2 = r[ch][new][j] * sin((FLOAT) phi[j]) - r_prime * sin((FLOAT) phi_prime);
442 
443  temp3 = r[ch][new][j] + fabs((FLOAT) r_prime);
444  if (temp3 != 0)
445  c[j] = sqrt(temp1 * temp1 + temp2 * temp2) / temp3;
446  else
447  c[j] = 0;
448 #endif
449  }
450 
451  /* For each partition, sum all the energy in that partition - grouped_e and calculated
452  the energy-weighted unpredictability measure - grouped_c ISO 11172 Section D.2.4.e */
453  for (j = 1; j < CBANDS; j++) {
454  grouped_e[j] = 0;
455  grouped_c[j] = 0;
456  }
457  grouped_e[0] = energy[0];
458  grouped_c[0] = energy[0] * c[0];
459  for (j = 1; j < HBLKSIZE; j++) {
460  grouped_e[partition[j]] += energy[j];
461  grouped_c[partition[j]] += energy[j] * c[j];
462  }
463 
464  /* convolve the grouped energy-weighted unpredictability measure and the grouped energy
465  with the spreading function ISO 11172 D.2.4.f */
466  for (j = 0; j < CBANDS; j++) {
467  ecb[j] = 0;
468  cb[j] = 0;
469  for (k = 0; k < CBANDS; k++) {
470  if (s[j][k] != 0.0) {
471  ecb[j] += s[j][k] * grouped_e[k];
472  cb[j] += s[j][k] * grouped_c[k];
473  }
474  }
475  if (ecb[j] != 0)
476  cb[j] = cb[j] / ecb[j];
477  else
478  cb[j] = 0;
479  }
480 
481  /* Convert cb to tb (the tonality index) ISO11172 SecD.2.4.g */
482  for (i = 0; i < CBANDS; i++) {
483  if (cb[i] < 0.05)
484  cb[i] = 0.05;
485  else if (cb[i] > 0.5)
486  cb[i] = 0.5;
487  tb[i] = -0.301029996 - 0.434294482 * log((FLOAT) cb[i]);
488  }
489 
490 
491  /* Calculate the required SNR for each of the frequency partitions ISO 11172 Sect
492  D.2.4.h */
493  for (j = 0; j < CBANDS; j++) {
494  FLOAT SNR, SNRtemp;
495  SNRtemp = tmn[j] * tb[j] + NMT * (1.0 - tb[j]);
496  SNR = MAX(SNRtemp, minval[(int) cbval[j]]);
497  bc[j] = exp((FLOAT) - SNR * LN_TO_LOG10);
498  }
499 
500  /* Calculate the permissible noise energy level in each of the frequency partitions.
501  This section used to have pre-echo control but only for LayerI ISO 11172 Sec D.2.4.k
502  - Spread the threshold energy over FFT lines */
503  for (j = 0; j < CBANDS; j++) {
504  if (rnorm[j] && numlines[j])
505  nb[j] = ecb[j] * bc[j] / (rnorm[j] * numlines[j]);
506  else
507  nb[j] = 0;
508  }
509 
510  /* ISO11172 Sec D.2.4.l - thr[] the final energy threshold of audibility */
511  for (j = 0; j < HBLKSIZE; j++)
512  thr[j] = MAX(nb[partition[j]], ath[j]);
513 
514  /* Translate the 512 threshold values to the 32 filter bands of the coder Using ISO
515  11172 Table D.5 and Section D.2.4.n */
516  for (j = 0; j < 193; j += 16) {
517  /* WIDTH = 0 */
518  npart = 60802371420160.0;
519  epart = 0.0;
520  for (k = 0; k < 17; k++) {
521  if (thr[j + k] < npart)
522  npart = thr[j + k]; /* For WIDTH==0, find the minimum noise, and later
523  multiply by the number of indexes i.e. 17 */
524  epart += energy[j + k];
525  }
526  snrtmp[run][j / 16] = 4.342944819 * log((FLOAT) (epart / (npart * 17.0)));
527  }
528  for (j = 208; j < (HBLKSIZE - 1); j += 16) {
529  /* WIDTH = 1 */
530  npart = 0.0;
531  epart = 0.0;
532  for (k = 0; k < 17; k++) {
533  npart += thr[j + k]; /* For WIDTH==1, sum the noise */
534  epart += energy[j + k];
535  }
536  snrtmp[run][j / 16] = 4.342944819 * log((FLOAT) (epart / npart));
537  }
538  }
539 
540  /* Pick the maximum value of the two runs ISO 11172 Sect D.2.1 */
541  for (i = 0; i < 32; i++)
542  smr[ch][i] = MAX(snrtmp[0][i], snrtmp[1][i]);
543 
544  } // now do other channel
545
