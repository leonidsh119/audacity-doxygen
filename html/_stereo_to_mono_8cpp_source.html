72 {

75  this->CopyInputTracks(); // Set up mOutputTracks.

77

79  mLeftTrack = (WaveTrack *)iter.First();
80  bool refreshIter = false;
81

84  // create a NEW WaveTrack to hold all of the output
85  AudacityProject *p = GetActiveProject();
86  mOutTrack = p->GetTrackFactory()->NewWaveTrack(floatSample, mLeftTrack->GetRate());
87  }
88 
89  int count = 0;
90  while (mLeftTrack) {
91  if (mLeftTrack->GetKind() == Track::Wave &&
92  mLeftTrack->GetSelected() &&
93  mLeftTrack->GetLinked()) {
94 
95  mRightTrack = (WaveTrack *)iter.Next();
96 
97  if ((mLeftTrack->GetRate() == mRightTrack->GetRate())) {
98  sampleCount leftTrackStart = mLeftTrack->TimeToLongSamples(mLeftTrack->GetStartTime());
99  sampleCount rightTrackStart = mRightTrack->TimeToLongSamples(mRightTrack->GetStartTime());
100  mStart = wxMin(leftTrackStart, rightTrackStart);
101 
102  sampleCount leftTrackEnd = mLeftTrack->TimeToLongSamples(mLeftTrack->GetEndTime());
103  sampleCount rightTrackEnd = mRightTrack->TimeToLongSamples(mRightTrack->GetEndTime());
104  mEnd = wxMax(leftTrackEnd, rightTrackEnd);
105 
106  bGoodResult = ProcessOne(count);
107  if (!bGoodResult)
108  break;
109 
110  mOutTrack->Clear(mOutTrack->GetStartTime(), mOutTrack->GetEndTime());
111 
112  // The right channel has been deleted, so we must restart from the beginning
113  refreshIter = true;
114  }
115  }
116 
117  if (refreshIter) {
118  mLeftTrack = (WaveTrack *)iter.First();
119  refreshIter = false;
120  }
121  else {
122  mLeftTrack = (WaveTrack *)iter.Next();
123  }
124  count++;
125  }
126 
127  if(mOutTrack)
128  delete mOutTrack;
129  this->ReplaceProcessedTracks(bGoodResult);
130  return bGoodResult;

133 bool EffectStereoToMono::ProcessOne(int count)
134 {
135  float curLeftFrame;
136  float curRightFrame;
137  float curMonoFrame;
138 
139  sampleCount idealBlockLen = mLeftTrack->GetMaxBlockSize() * 2;
140  sampleCount index = mStart;
141  float *leftBuffer = new float[idealBlockLen];
142  float *rightBuffer = new float[idealBlockLen];
143  bool bResult = true;
144 
145  while (index < mEnd) {
146  bResult &= mLeftTrack->Get((samplePtr)leftBuffer, floatSample, index, idealBlockLen);
147  bResult &= mRightTrack->Get((samplePtr)rightBuffer, floatSample, index, idealBlockLen);
148  sampleCount limit = idealBlockLen;
149  if ((index + idealBlockLen) > mEnd) {
150  limit = mEnd - index;
151  }
152  for (sampleCount i = 0; i < limit; ++i) {
153  index++;
154  curLeftFrame = leftBuffer[i];
155  curRightFrame = rightBuffer[i];
156  curMonoFrame = (curLeftFrame + curRightFrame) / 2.0;
157  leftBuffer[i] = curMonoFrame;
158  }
159  bResult &= mOutTrack->Append((samplePtr)leftBuffer, floatSample, limit);
160  if (TrackProgress(count, 2.*((double)index / (double)(mEnd - mStart))))
161  return false;
162  }
163 
164  double minStart = wxMin(mLeftTrack->GetStartTime(), mRightTrack->GetStartTime());
165  bResult &= mLeftTrack->Clear(mLeftTrack->GetStartTime(), mLeftTrack->GetEndTime());
166  bResult &= mOutTrack->Flush();
167  bResult &= mLeftTrack->Paste(minStart, mOutTrack);
168  mLeftTrack->SetLinked(false);
169  mRightTrack->SetLinked(false);
170  mLeftTrack->SetChannel(Track::MonoChannel);
171  mOutputTracks->Remove(mRightTrack);
172  delete mRightTrack;
