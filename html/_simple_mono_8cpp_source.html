83 {

89

92  float *buffer = new float[track->GetMaxBlockSize()];
93 
94  //Go through the track one buffer at a time. s counts which
95  //sample the current buffer starts at.
96  s = start;
97  while (s < end) {
98  //Get a block of samples (smaller than the size of the buffer)
99  sampleCount block = track->GetBestBlockSize(s);
100

102  if (s + block > end)
103  block = end - s;
104

106  track->Get((samplePtr) buffer, floatSample, s, block);
107

109  if (!ProcessSimpleMono(buffer, block)) {

113  return false;
114  }
115

117  //onto the track.
118  track->Set((samplePtr) buffer, floatSample, s, block);
