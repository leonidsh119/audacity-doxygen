480 int ControlToolBar::PlayPlayRegion(const SelectedRegion &selectedRegion,
481  const AudioIOStartStreamOptions &options,
482  bool cutpreview, /* = false */
483  bool backwards, /* = false */
484  bool playWhiteSpace /* = false */)
485 {
486  // Uncomment this for laughs!
487  // backwards = true;
488 
489  double t0 = selectedRegion.t0();
490  double t1 = selectedRegion.t1();
491  // SelectedRegion guarantees t0 <= t1, so we need another boolean argument
492  // to indicate backwards play.
493  const bool looped = options.playLooped;
494 
495  if (backwards)
496  std::swap(t0, t1);
497 
498  SetPlay(true, looped, cutpreview);
499 
500  if (gAudioIO->IsBusy()) {
501  SetPlay(false);
502  return -1;
503  }
504 
505  if (cutpreview && t0==t1) {
506  SetPlay(false);
507  return -1; /* msmeyer: makes no sense */
508  }
509 
510  AudacityProject *p = GetActiveProject();
511  if (!p) {
512  SetPlay(false);
513  return -1; // Should never happen, but...
514  }
515 
516  TrackList *t = p->GetTracks();
517  if (!t) {
518  mPlay->PopUp();
519  return -1; // Should never happen, but...
520  }
521 
522  bool hasaudio = false;
523  TrackListIterator iter(t);
524  for (Track *trk = iter.First(); trk; trk = iter.Next()) {
525  if (trk->GetKind() == Track::Wave
526 #ifdef EXPERIMENTAL_MIDI_OUT
527  || trk->GetKind() == Track::Note
528 #endif
529  ) {
530  hasaudio = true;
531  break;
532  }
533  }
534 
535  double latestEnd = (playWhiteSpace)? t1 : t->GetEndTime();
536 
537  if (!hasaudio) {
538  SetPlay(false);
539  return -1; // No need to continue without audio tracks
540  }
541 
542 #if defined(EXPERIMENTAL_SEEK_BEHIND_CURSOR)
543  double init_seek = 0.0;
544 #endif
545 
546  if (t1 == t0) {
547  if (looped) {
548  // play selection if there is one, otherwise
549  // set start of play region to project start,
550  // and loop the project from current play position.
551 
552  if ((t0 > p->GetSel0()) && (t0 < p->GetSel1())) {
553  t0 = p->GetSel0();
554  t1 = p->GetSel1();
555  }
556  else {
557  // loop the entire project
558  t0 = t->GetStartTime();
559  t1 = t->GetEndTime();
560  }
561  } else {
562  // move t0 to valid range
563  if (t0 < 0) {
564  t0 = t->GetStartTime();
565  }
566  else if (t0 > t->GetEndTime()) {
567  t0 = t->GetEndTime();
568  }
569 #if defined(EXPERIMENTAL_SEEK_BEHIND_CURSOR)
570  else {
571  init_seek = t0; //AC: init_seek is where playback will 'start'
572  t0 = t->GetStartTime();
573  }
574 #endif
575  }
576  t1 = t->GetEndTime();
577  }
578  else {
579  // maybe t1 < t0, with backwards scrubbing for instance
580  if (backwards)
581  std::swap(t0, t1);
582 
583  t0 = std::max(0.0, std::min(t0, latestEnd));
584  t1 = std::max(0.0, std::min(t1, latestEnd));
585 
586  if (backwards)
587  std::swap(t0, t1);
588  }
589 
590  int token = -1;
591  bool success = false;
592  if (t1 != t0) {
593  if (cutpreview) {
594  const double tless = std::min(t0, t1);
595  const double tgreater = std::max(t0, t1);
596  double beforeLen, afterLen;
597  gPrefs->Read(wxT("/AudioIO/CutPreviewBeforeLen"), &beforeLen, 2.0);
598  gPrefs->Read(wxT("/AudioIO/CutPreviewAfterLen"), &afterLen, 1.0);
599  double tcp0 = tless-beforeLen;
600  double diff = tgreater - tless;
601  double tcp1 = (tgreater+afterLen) - diff;
602  SetupCutPreviewTracks(tcp0, tless, tgreater, tcp1);
603  if (backwards)
604  std::swap(tcp0, tcp1);
605  if (mCutPreviewTracks)
606  {
607  AudioIOStartStreamOptions myOptions = options;
608  myOptions.cutPreviewGapStart = t0;
609  myOptions.cutPreviewGapLen = t1 - t0;
610  token = gAudioIO->StartStream(
611  mCutPreviewTracks->GetWaveTrackArray(false),
612  WaveTrackArray(),
613 #ifdef EXPERIMENTAL_MIDI_OUT
614  NoteTrackArray(),
615 #endif
616  p->GetRate(), tcp0, tcp1, myOptions);
617  } else
618  {
619  // Cannot create cut preview tracks, clean up and exit
620  SetPlay(false);
621  SetStop(false);
622  SetRecord(false);
623  return -1;
624  }
625  } else {
626  // Lifted the following into AudacityProject::GetDefaultPlayOptions()
627  /*
628  if (!timetrack) {
629  timetrack = t->GetTimeTrack();
630  }
631  */
632  token = gAudioIO->StartStream(t->GetWaveTrackArray(false),
633  WaveTrackArray(),
634 #ifdef EXPERIMENTAL_MIDI_OUT
635  t->GetNoteTrackArray(false),
636 #endif
637  p->GetRate(), t0, t1, options);
638  }
639  if (token != 0) {
640  success = true;
641  p->SetAudioIOToken(token);
642  mBusyProject = p;
643 #if defined(EXPERIMENTAL_SEEK_BEHIND_CURSOR)
644  //AC: If init_seek was set, now's the time to make it happen.
645  gAudioIO->SeekStream(init_seek);
646 #endif
647  }
648  else {
649  // msmeyer: Show error message if stream could not be opened
650  wxMessageBox(
651  _("Error while opening sound device. "
652  "Please check the playback device settings and the project sample rate."),
653  _("Error"), wxOK | wxICON_EXCLAMATION, this);
654  }
655  }
656 
657  if (!success) {
658  SetPlay(false);
659  SetStop(false);
660  SetRecord(false);
661  return -1;
662  }
663 
664  return token;
665 }

807  double t1 = p->GetSel1();

812  * their preferences, and choose sample format based on prefs */
813  WaveTrackArray newRecordingTracks, playbackTracks;
814 #ifdef EXPERIMENTAL_MIDI_OUT

816 #endif

822 #ifdef EXPERIMENTAL_MIDI_OUT
823  midiTracks = t->GetNoteTrackArray(false);

825  }
826  else {

828 #ifdef EXPERIMENTAL_MIDI_OUT
829  midiTracks = NoteTrackArray();
830 #endif
831  }
832 
833  // If SHIFT key was down, the user wants append to tracks
834  int recordingChannels = 0;
835  TrackList tracksCopy{};
836  bool tracksCopied = false;
837  bool shifted = mRecord->WasShiftDown();
838  if (shifted) {
839  bool sel = false;
840  double allt0 = t0;
841 
842  // Find the maximum end time of selected and all wave tracks
843  // Find whether any tracks were selected. (If any are selected,
844  // record only into them; else if tracks exist, record into all.)
845  for (Track *tt = it.First(); tt; tt = it.Next()) {
846  if (tt->GetKind() == Track::Wave) {
847  WaveTrack *wt = static_cast<WaveTrack *>(tt);
848  if (wt->GetEndTime() > allt0) {
849  allt0 = wt->GetEndTime();
850  }
851 
852  if (tt->GetSelected()) {
853  sel = true;
854  if (wt->GetEndTime() > t0) {
855  t0 = wt->GetEndTime();
856  }
857  }
858  }
859  }
860 
861  // Use end time of all wave tracks if none selected
862  if (!sel) {
863  t0 = allt0;
864  }
865 
866  // Pad selected/all wave tracks to make them all the same length
867  // Remove recording tracks from the list of tracks for duplex ("overdub")
868  // playback.
869  for (Track *tt = it.First(); tt; tt = it.Next()) {
870  if (tt->GetKind() == Track::Wave && (tt->GetSelected() || !sel)) {
871  WaveTrack *wt = static_cast<WaveTrack *>(tt);
872  if (duplex) {
873  auto end = playbackTracks.end();
874  auto it = std::find(playbackTracks.begin(), end, wt);
875  if (it != end)
876  playbackTracks.erase(it);
877  }
878  t1 = wt->GetEndTime();
879  if (t1 < t0) {
880  if (!tracksCopied) {
881  tracksCopied = true;
882  tracksCopy = *trackList;
883  }
884 
885  WaveTrack *newTrack = p->GetTrackFactory()->NewWaveTrack();
886  newTrack->InsertSilence(0.0, t0 - t1);
887  newTrack->Flush();
888  wt->Clear(t1, t0);
889  bool bResult = wt->Paste(t1, newTrack);
890  wxASSERT(bResult); // TO DO: Actually handle this.
891  wxUnusedVar(bResult);
892  delete newTrack;
893  }
894  newRecordingTracks.push_back(wt);
895  }
896  }
897 
898  t1 = 1000000000.0; // record for a long, long time (tens of years)
899  }
900  else {
901  bool recordingNameCustom, useTrackNumber, useDateStamp, useTimeStamp;
902  wxString defaultTrackName, defaultRecordingTrackName;
903  int numTracks = 0;
904 
905  for (Track *tt = it.First(); tt; tt = it.Next()) {
906  if (tt->GetKind() == Track::Wave && !tt->GetLinked())
907  numTracks++;
908  }
909  numTracks++;
910 
911  recordingChannels = gPrefs->Read(wxT("/AudioIO/RecordChannels"), 2);
912 
913  gPrefs->Read(wxT("/GUI/TrackNames/RecordingNameCustom"), &recordingNameCustom, false);
914  gPrefs->Read(wxT("/GUI/TrackNames/TrackNumber"), &useTrackNumber, false);
915  gPrefs->Read(wxT("/GUI/TrackNames/DateStamp"), &useDateStamp, false);
916  gPrefs->Read(wxT("/GUI/TrackNames/TimeStamp"), &useTimeStamp, false);
917  /* i18n-hint: The default name for an audio track. */
918  gPrefs->Read(wxT("/GUI/TrackNames/DefaultTrackName"),&defaultTrackName, _("Audio Track"));
919  gPrefs->Read(wxT("/GUI/TrackNames/RecodingTrackName"), &defaultRecordingTrackName, defaultTrackName);
920 
921  wxString baseTrackName = recordingNameCustom? defaultRecordingTrackName : defaultTrackName;
922 
923  for (int c = 0; c < recordingChannels; c++) {
924  WaveTrack *newTrack = p->GetTrackFactory()->NewWaveTrack();
925 
926  newTrack->SetOffset(t0);
927  wxString nameSuffix = wxString(wxT(""));
928 
929  if (useTrackNumber) {
930  nameSuffix += wxString::Format(wxT("%d"), numTracks + c);
931  }
932 
933  if (useDateStamp) {
934  if (!nameSuffix.IsEmpty()) {
935  nameSuffix += wxT("_");
936  }
937  nameSuffix += wxDateTime::Now().FormatISODate();
938  }
939 
940  if (useTimeStamp) {
941  if (!nameSuffix.IsEmpty()) {
942  nameSuffix += wxT("_");
943  }
944  nameSuffix += wxDateTime::Now().FormatISOTime();
945  }
946 
947  // ISO standard would be nice, but ":" is unsafe for file name.
948  nameSuffix.Replace(wxT(":"), wxT("-"));
949 
950  if (baseTrackName.IsEmpty()) {
951  newTrack->SetName(nameSuffix);
952  }
953  else if (nameSuffix.IsEmpty()) {
954  newTrack->SetName(baseTrackName);
955  }
956  else {
957  newTrack->SetName(baseTrackName + wxT("_") + nameSuffix);
958  }
959 
960  if (recordingChannels > 2)
961  newTrack->SetMinimized(true);
962 
963  if (recordingChannels == 2) {
964  if (c == 0) {
965  newTrack->SetChannel(Track::LeftChannel);
966  newTrack->SetLinked(true);
967  }
968  else {
969  newTrack->SetChannel(Track::RightChannel);
970  }
971  }
972  else {
973  newTrack->SetChannel( Track::MonoChannel );
974  }
975 
976  newRecordingTracks.push_back(newTrack);
977  }
978 
979  // msmeyer: StartStream calls a callback which triggers auto-save, so
980  // we add the tracks where recording is done into now. We remove them
981  // later if starting the stream fails
982  for (unsigned int i = 0; i < newRecordingTracks.size(); i++)
983  trackList->Add(newRecordingTracks[i]);
984  }
985 
986  //Automated Input Level Adjustment Initialization
987  #ifdef EXPERIMENTAL_AUTOMATED_INPUT_LEVEL_ADJUSTMENT
988  gAudioIO->AILAInitialize();
989  #endif
990 
991  AudioIOStartStreamOptions options(p->GetDefaultPlayOptions());
992  int token = gAudioIO->StartStream(playbackTracks,
993  newRecordingTracks,
994 #ifdef EXPERIMENTAL_MIDI_OUT
995  midiTracks,
996 #endif
997  p->GetRate(), t0, t1, options);
998 
999  bool success = (token != 0);
1000 
1001  if (success) {
1002  p->SetAudioIOToken(token);
1003  mBusyProject = p;
1004  }
1005  else {
1006  if (shifted) {
1007  // Restore the tracks to remove any inserted silence
1008  if (tracksCopied)
1009  *trackList = std::move(tracksCopy);
1010  }
1011  else {
1012  // msmeyer: Delete recently added tracks if opening stream fails
1013  for (unsigned int i = 0; i < newRecordingTracks.size(); i++) {
1014  trackList->Remove(newRecordingTracks[i]);

1017  }

1021  _("Error"), wxOK | wxICON_EXCLAMATION, this);

1023  SetPlay(false);

1025  SetRecord(false);
1026  }
1027  }

1029 }
