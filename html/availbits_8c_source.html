44 {

46  int adb;
47 
48  slots.extra = 0; /* be default, no extra slots */
49 
50  slots.average = (1152.0 / ((FLOAT) glopts->samplerate_out / 1000.0))
51  * ((FLOAT) glopts->bitrate / 8.0);
52 
53  // fprintf(stderr,"availbits says: sampling freq is %i. version %i. bitrateindex %i slots
54  // %f\n",header->sampling_frequency, header->version, header->bitrate_index, slots.average);
55 
56  slots.whole = (int) slots.average;
57  slots.frac = slots.average - (FLOAT) slots.whole;
58 
59  /* never allow padding for a VBR frame. Don't ask me why, I've forgotten why I set this */
60  if (slots.frac != 0 && glopts->padding && glopts->vbr == FALSE) {
61  if (slots.lag > (slots.frac - 1.0)) { /* no padding for this frame */
62  slots.lag -= slots.frac;
63  slots.extra = 0;
64  header->padding = 0;
65  } else { /* padding */
66  slots.extra = 1;
67  header->padding = 1;
68  slots.lag += (1 - slots.frac);
69  }
70  }
71 
72  adb = (slots.whole + slots.extra) * 8;
73
