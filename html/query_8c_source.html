30 static LilvLangMatch
31 lilv_lang_matches(const char* a, const char* b)
32 {
33  if (!strcmp(a, b)) {
34  return LILV_LANG_MATCH_EXACT;
35  }
36 
37  const char* a_dash = strchr(a, '-');
38  const size_t a_lang_len = a_dash ? (size_t)(a_dash - a) : strlen(a);
39  const char* b_dash = strchr(b, '-');
40  const size_t b_lang_len = b_dash ? (size_t)(b_dash - b) : strlen(b);
41 
42  if (a_lang_len == b_lang_len && !strncmp(a, b, a_lang_len)) {
43  return LILV_LANG_MATCH_PARTIAL;
44  }
45 
46  return LILV_LANG_MATCH_NONE;
47 }
