160 void Importer::ReadImportItems()
161 {
162  int item_counter = 0;
163  wxStringTokenizer toker;
164  wxString item_name;
165  wxString item_value;
166  ExtImportItem *new_item;
167  ImportPluginList::compatibility_iterator importPluginNode;
168 
169  if (this->mExtImportItems != NULL)
170  delete this->mExtImportItems;
171 
172  this->mExtImportItems = new ExtImportItems();
173  /* Rule string format is:
174  * extension1:extension2:extension3\mime_type1:mime_type2:mime_type3|filter1:filter2:filter3\unusedfilter1:unusedfilter2
175  * backslashes are escaped and unescaped internally
176  */
177  for (item_counter = 0; true; item_counter++)
178  {
179  wxString condition, filters, used_filters, unused_filters = wxEmptyString, extensions, mime_types = wxEmptyString;
180  item_name.Printf (wxT("/ExtImportItems/Item%d"), item_counter);
181  /* Break at first non-existent item */
182  if (!gPrefs->Read(item_name, &item_value))
183  break;
184 
185  toker.SetString(item_value, wxT("|"), wxTOKEN_RET_EMPTY_ALL);
186  /* Break at first broken item */
187  if (toker.CountTokens() != 2)
188  break;
189 
190  new_item = new ExtImportItem();
191 
192  /* First token is the filtering condition, second - the filter list */
193  condition = toker.GetNextToken();
194  filters = toker.GetNextToken();
195 
196  /* Condition token consists of extension list and mime type list
197  * mime type list can be omitted entirely (complete with '\' separator)*/
198  toker.SetString(condition, wxT("\\"), wxTOKEN_RET_EMPTY_ALL);
199  extensions = toker.GetNextToken();
200  if (toker.HasMoreTokens())
201  mime_types = toker.GetNextToken();
202 
203  wxString delims(wxT(":"));
204  StringToList (extensions, delims, new_item->extensions);
205 
206  if (mime_types != wxEmptyString)
207  StringToList (mime_types, delims, new_item->mime_types);
208 
209  /* Filter token consists of used and unused filter lists */
210  toker.SetString(filters, wxT("\\"), wxTOKEN_RET_EMPTY_ALL);
211  used_filters = toker.GetNextToken();
212  if (toker.HasMoreTokens())
213  unused_filters = toker.GetNextToken();
214 
215  StringToList (used_filters, delims, new_item->filters);
216 
217  if (unused_filters != wxEmptyString)
218  {
219  /* Filters are stored in one list, but the position at which
220  * unused filters start is remembered
221  */
222  new_item->divider = new_item->filters.Count();
223  StringToList (unused_filters, delims, new_item->filters);
224  }
225  else
226  new_item->divider = -1;
227 
228  /* Find corresponding filter object for each filter ID */
229  for (size_t i = 0; i < new_item->filters.Count(); i++)
230  {
231  for (importPluginNode = mImportPluginList->GetFirst();
232  importPluginNode; importPluginNode = importPluginNode->GetNext())
233  {
234  ImportPlugin *importPlugin = importPluginNode->GetData();
235  if (importPlugin->GetPluginStringID().Cmp(new_item->filters[i]) == 0)
236  {
237  new_item->filter_objects.Add (importPlugin);
238  break;
239  }
240  }
241  /* IDs that do not have corresponding filters, will be shown as-is */
242  if (!importPluginNode)
243  new_item->filter_objects.Add (NULL);
244  }
245  /* Find all filter objects that are not present in the filter list */
246  for (importPluginNode = mImportPluginList->GetFirst();
247  importPluginNode; importPluginNode = importPluginNode->GetNext())
248  {
249  bool found = false;
250  ImportPlugin *importPlugin = importPluginNode->GetData();
251  for (size_t i = 0; i < new_item->filter_objects.Count(); i++)
252  {
253  if (importPlugin == new_item->filter_objects[i])
254  {
255  found = true;
256  break;
257  }
258  }
259  /* Add these filters at the bottom of used filter list */
260  if (!found)
261  {
262  int index = new_item->divider;
263  if (new_item->divider < 0)
264  index = new_item->filters.Count();
265  new_item->filters.Insert(importPlugin->GetPluginStringID(),index);
266  new_item->filter_objects.Insert (importPlugin, index);
267  if (new_item->divider >= 0)
268  new_item->divider++;
269  }
270  }
271  this->mExtImportItems->Add (new_item);
272  }
273 }

275 void Importer::WriteImportItems()
276 {
277  size_t i;
278  wxString val, name;
279  for (i = 0; i < this->mExtImportItems->Count(); i++)
280  {
281  ExtImportItem *item = &(mExtImportItems->Item(i));
282  val.Clear();
283 
284  for (size_t j = 0; j < item->extensions.Count(); j++)
285  {
286  val.Append (item->extensions[j]);
287  if (j < item->extensions.Count() - 1)
288  val.Append (wxT(":"));
289  }
290  val.Append (wxT("\\"));
291  for (size_t j = 0; j < item->mime_types.Count(); j++)
292  {
293  val.Append (item->mime_types[j]);
294  if (j < item->mime_types.Count() - 1)
295  val.Append (wxT(":"));
296  }
297  val.Append (wxT("|"));
298  for (size_t j = 0; j < item->filters.Count() && ((int) j < item->divider || item->divider < 0); j++)
299  {
300  val.Append (item->filters[j]);
301  if (j < item->filters.Count() - 1 && ((int) j < item->divider - 1 || item->divider < 0))
302  val.Append (wxT(":"));
303  }
304  if (item->divider >= 0)
305  {
306  val.Append (wxT("\\"));
307  for (size_t j = item->divider; j < item->filters.Count(); j++)
308  {
309  val.Append (item->filters[j]);
310  if (j < item->filters.Count() - 1)
311  val.Append (wxT(":"));
312  }
313  }
314  name.Printf (wxT("/ExtImportItems/Item%d"), (int)i);
315  gPrefs->Write (name, val);
316  gPrefs->Flush();
317  }
318  /* If we used to have more items than we have now, DELETE the excess items.
319  We just keep deleting items and incrementing until we find there aren't any
320  more to DELETE.*/
321  i = this->mExtImportItems->Count();
322  do {
323  name.Printf (wxT("/ExtImportItems/Item%d"), (int)i);
324  // No item to DELETE? Then it's time to finish.
325  if (!gPrefs->Read(name, &val))
326  break;
327  // Failure to DELETE probably means a read-only config file.
328  // no point continuing.
329  // TODO: Possibly report (once).
330  if( !gPrefs->DeleteEntry (name, false))
331  break;
332  i++;
333  } while( true );
334 }

381 
382  // Not implemented (yet?)
383  wxString mime_type = wxT("*");
384 
385  // First, add user-selected filter
386  bool usersSelectionOverrides;
387  gPrefs->Read(wxT("/ExtendedImport/OverrideExtendedImportByOpenFileDialogChoice"), &usersSelectionOverrides, false);
388 
389  wxLogDebug(wxT("LastOpenType is %s"),type.c_str());
390  wxLogDebug(wxT("OverrideExtendedImportByOpenFileDialogChoice is %i"),usersSelectionOverrides);
391 
392  if (usersSelectionOverrides)

394  importPluginNode = mImportPluginList->GetFirst();
395  while (importPluginNode)
396  {
397  ImportPlugin *plugin = importPluginNode->GetData();
398  if (plugin->GetPluginFormatDescription().CompareTo(type) == 0)
399  {
400  // This plugin corresponds to user-selected filter, try it first.
401  wxLogDebug(wxT("Inserting %s"),plugin->GetPluginStringID().c_str());
402  importPlugins.Insert(plugin);
403  }
404  importPluginNode = importPluginNode->GetNext();
405  }
406  }
407 
408  wxLogMessage(wxT("File name is %s"),(const char *) fName.c_str());
409  wxLogMessage(wxT("Mime type is %s"),(const char *) mime_type.Lower().c_str());
410 
411  for (size_t i = 0; i < mExtImportItems->Count(); i++)
412  {
413  ExtImportItem *item = &(*mExtImportItems)[i];
414  bool matches_ext = false, matches_mime = false;
415  wxLogDebug(wxT("Testing extensions"));
416  for (size_t j = 0; j < item->extensions.Count(); j++)
417  {
418  wxLogDebug(wxT("%s"), (const char *) item->extensions[j].Lower().c_str());
419  if (wxMatchWild (item->extensions[j].Lower(),fName.Lower(), false))
420  {
421  wxLogDebug(wxT("Match!"));
422  matches_ext = true;
423  break;
424  }
425  }
426  if (item->extensions.Count() == 0)
427  {
428  wxLogDebug(wxT("Match! (empty list)"));
429  matches_ext = true;
430  }
431  if (matches_ext)
432  wxLogDebug(wxT("Testing mime types"));
433  else
434  wxLogDebug(wxT("Not testing mime types"));
435  for (size_t j = 0; matches_ext && j < item->mime_types.Count(); j++)
436  {
437  if (wxMatchWild (item->mime_types[j].Lower(),mime_type.Lower(), false))
438  {
439  wxLogDebug(wxT("Match!"));
440  matches_mime = true;
441  break;
442  }
443  }
444  if (item->mime_types.Count() == 0)
445  {
446  wxLogDebug(wxT("Match! (empty list)"));
447  matches_mime = true;
448  }
449  if (matches_ext && matches_mime)
450  {
451  wxLogDebug(wxT("Complete match!"));
452  for (size_t j = 0; j < item->filter_objects.Count() && (item->divider < 0 || (int) j < item->divider); j++)
453  {

455  // this happens when we recompile with --without-ffmpeg and there
456  // is still ffmpeg in prefs from previous --with-ffmpeg builds
457  if (!(item->filter_objects[j]))
458  continue;
459  wxLogDebug(wxT("Inserting %s"),item->filter_objects[j]->GetPluginStringID().c_str());
460  importPlugins.Append(item->filter_objects[j]);
461  }
462  }
463  }

465  // Add all plugins that support the extension
466  importPluginNode = mImportPluginList->GetFirst();
467 
468  // Here we rely on the fact that the first plugin in mImportPluginList is libsndfile.
469  // We want to save this for later insertion ahead of libmad, if libmad supports the extension.
470  // The order of plugins in mImportPluginList is determined by the Importer constructor alone and
471  // is not changed by user selection overrides or any other mechanism, but we include an assert
472  // in case subsequent code revisions to the constructor should break this assumption that
473  // libsndfile is first.
474  ImportPlugin *libsndfilePlugin = importPluginNode->GetData();
475  wxASSERT(libsndfilePlugin->GetPluginStringID().IsSameAs(wxT("libsndfile")));
476 
477  while (importPluginNode)
478  {
479  ImportPlugin *plugin = importPluginNode->GetData();
480  // Make sure its not already in the list
481  if (importPlugins.Find(plugin) == NULL)
482  {
483  if (plugin->SupportsExtension(extension))
484  {
485  // If libmad is accidentally fed a wav file which has been incorrectly
486  // given an .mp3 extension then it can choke on the contents and crash.
487  // To avoid this, put libsndfile ahead of libmad in the lists created for
