74 soxr_quality_spec_t soxr_quality_spec(unsigned long recipe, unsigned long flags)
75 {
76  soxr_quality_spec_t spec, * p = &spec;
77  unsigned quality = recipe & 0xf;
78  double rej;
79  memset(p, 0, sizeof(*p));
80  if (quality > 13) {
81  p->e = "invalid quality type";
82  return spec;
83  }
84  if (quality == 13)
85  quality = 6;
86  else if (quality > 10)
87  quality = 0;
88  p->phase_response = "\62\31\144"[(recipe & 0x30)>>8];
89  p->stopband_begin = 1;
90  p->precision = !quality? 0: quality < 3? 16 : quality < 8? 4 + quality * 4 : 55 - quality * 4;
91  rej = p->precision * linear_to_dB(2.);
92  p->flags = flags;
93  if (quality < 8) {
94  p->passband_end = quality == 1? LOW_Q_BW0 : 1 - .05 / TO_3dB(rej);
95  if (quality <= 2)
96  p->flags &= ~SOXR_ROLLOFF_NONE, p->flags |= SOXR_ROLLOFF_MEDIUM;
97  }
98  else {
99  static float const bw[] = {.931f, .832f, .663f};
100  p->passband_end = bw[quality - 8];
101  if (quality - 8 == 2)
102  p->flags &= ~SOXR_ROLLOFF_NONE, p->flags |= SOXR_ROLLOFF_MEDIUM;
103  }
104  if (recipe & SOXR_STEEP_FILTER)
105  p->passband_end = 1 - .01 / TO_3dB(rej);
106  return spec;
107 }
