466 // the minimum amount of cache entries necessary to warrant a binary search.
467 #define kODFFmpegSearchThreshold 10
468 //also updates data and len to reflect NEW unfilled area - start is unmodified.
470 int ODFFmpegDecoder::FillDataFromCache(samplePtr & data, sampleFormat outFormat, sampleCount &start, sampleCount& len, unsigned int channel)
471 {
472  if(mDecodeCache.size() <= 0)
473  return 0;
474  int samplesFilled=0;
475 
476  //do a search for the best position to start at.
477  //Guess that the array is evenly spaced from end to end - (dictionary sort)
478  //assumes the array is sorted.
479  //all we need for this to work is a location in the cache array
480  //that has a start time of less than our start sample, but try to get closer with binary search
481  int searchStart = 0;
482  int searchEnd = mDecodeCache.size();
483  int guess;
484  if(searchEnd>kODFFmpegSearchThreshold)
485  {
486  //first just guess that the cache is contiguous and we can just use math to figure it out like a dictionary
487  //by guessing where our hit will be.
488  while(searchStart+1<searchEnd)
489  {
490  guess = (searchStart+searchEnd)/2;//find a midpoint. //searchStart+ (searchEnd-searchStart)* ((float)start - mDecodeCache[searchStart]->start )/mDecodeCache[searchEnd]->start;
491 
492  //we want guess to point at the first index that hits even if there are duplicate start times (which can happen)
493  if(mDecodeCache[guess]->start+mDecodeCache[guess]->len >= start)
494  searchEnd = --guess;
495  else
496  searchStart = guess;
497  }
498  }
499 
500  //this is a sorted array
501  for(int i=searchStart; i < (int)mDecodeCache.size(); i++)
502  {
503  //check for a cache hit - be careful to include the first/last sample an nothing more.
504  //we only accept cache hits that touch either end - no piecing out of the middle.
505  //this way the amount to be decoded remains set.
506  if(start < mDecodeCache[i]->start+mDecodeCache[i]->len &&
507  start + len > mDecodeCache[i]->start)
508  {
509  uint8_t* outBuf;
510  outBuf = (uint8_t*)data;
511  //reject buffers that would split us into two pieces because we don't have
512  //a method of dealing with this yet, and it won't happen very often.
513  if(start<mDecodeCache[i]->start && start+len > mDecodeCache[i]->start+mDecodeCache[i]->len)
514  continue;
515 
516  int samplesHit;
517  int hitStartInCache;
518  int hitStartInRequest;
519  int nChannels = mDecodeCache[i]->numChannels;
520  samplesHit = FFMIN(start+len,mDecodeCache[i]->start+mDecodeCache[i]->len)
521  - FFMAX(mDecodeCache[i]->start,start);
522  //find the start of the hit relative to the cache buffer start.
523  hitStartInCache = FFMAX(0,start-mDecodeCache[i]->start);
524  //we also need to find out which end was hit - if it is the tail only we need to update from a later index.
525  hitStartInRequest = start <mDecodeCache[i]->start?len - samplesHit: 0;
526  sampleCount outIndex,inIndex;
527  for(int j=0;j<samplesHit;j++)
528  {
529  outIndex = hitStartInRequest + j;
530  inIndex = (hitStartInCache + j) * nChannels + channel;
531  switch (mDecodeCache[i]->samplefmt)
532  {
533  case SAMPLE_FMT_U8:
534  //printf("u8 in %llu out %llu cachelen %llu outLen %llu\n", inIndex, outIndex, mDecodeCache[i]->len, len);
535  ((int16_t *)outBuf)[outIndex] = (int16_t) (((uint8_t*)mDecodeCache[i]->samplePtr)[inIndex] - 0x80) << 8;
536  break;
537 
538  case SAMPLE_FMT_S16:
539  //printf("u16 in %llu out %llu cachelen %llu outLen %llu\n", inIndex, outIndex, mDecodeCache[i]->len, len);
540  ((int16_t *)outBuf)[outIndex] = ((int16_t*)mDecodeCache[i]->samplePtr)[inIndex];
541  break;
542 
543  case SAMPLE_FMT_S32:
544  //printf("s32 in %llu out %llu cachelen %llu outLen %llu\n", inIndex, outIndex, mDecodeCache[i]->len, len);
545  ((float *)outBuf)[outIndex] = (float) ((int32_t*)mDecodeCache[i]->samplePtr)[inIndex] * (1.0 / (1 << 31));
546  break;
547 
548  case SAMPLE_FMT_FLT:
549  //printf("f in %llu out %llu cachelen %llu outLen %llu\n", inIndex, outIndex, mDecodeCache[i]->len, len);
550  ((float *)outBuf)[outIndex] = (float) ((float*)mDecodeCache[i]->samplePtr)[inIndex];
551  break;
552 
553  case SAMPLE_FMT_DBL:
554  //printf("dbl in %llu out %llu cachelen %llu outLen %llu\n", inIndex, outIndex, mDecodeCache[i]->len, len);
555  ((float *)outBuf)[outIndex] = (float) ((double*)mDecodeCache[i]->samplePtr)[inIndex];
556  break;
557 
558  default:
559  printf("ODDecodeFFMPEG TASK unrecognized sample format\n");
560  return 1;
561  break;
562  }
563  }
564  //update the cursor
565  samplesFilled += samplesHit;
566 
567  //update the input start/len params - if the end was hit we can take off just len.
568  //otherwise, we can assume only the front of the request buffer was hit since we don't allow it to be split.
569  if(start < mDecodeCache[i]->start)
570  len-=samplesHit;
571  else
572  {
573  //we update data pointer too- but it is a typedef'd char* so be careful with the pointer math
574  data+= samplesHit * (SAMPLE_SIZE(outFormat) / sizeof(*data));
575  start+=samplesHit;
576  len -=samplesHit;
577  }
578  }
579  //if we've had our fill, leave. if we've passed the point which can have hits, leave.
580  if(len<=0 || mDecodeCache[i]->start > start+len)
581  break;
582  }
583  return samplesFilled;
584 }
