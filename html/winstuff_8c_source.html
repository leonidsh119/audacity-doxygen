5 /* Changes by Roger Dannenberg, Jan 2006:
6 Previously, the input thread would block on input, so if a command line
7 instantiation of Nyquist called (EXIT), the process would still block
8 in getchar() until the user typed a newline. Now, I only start the
9 input thread if ostgetc is called (input is really needed). This will
10 still read ahead and block, but only in cases where you are already
11 interactive.
12 
13 /* Changes by Roger Dannenberg, April 2004:
14 To support interrupts to Lisp processing, XLISP call oscheck frequently to
15 test for an abort or break condition. This condition can be activated by
16 special handlers, e.g. if a software interrupt is generated by Ctrl-C.
17 Alternatively, the application can read ahead and look for break characters
18 in the input stream. A third approach, implemented by Ning Hu for her
19 Delphi-based IDE, is to send a Windows message to the process. Unfortunately,
20 the Java IDE does not support sending a Windows message, nor can console
21 IO be used to read ahead (console IO does not work because, when started by
22 Java, Nyquist standard IO is redirected through pipes). The two solutions
23 to enable break character prcessing seem to be:
24  1) extend Java with C code to find the process and send Windows messages
25  2) add a thread to perform read ahead and break character processing
26 Option 2 contains the ugliness to Nyquist IO, which is already big and ugly,
27 and leaves Java alone, which is something I don't know much about anyway,
28 so I have chosen option 2: create a thread and read ahead. This uses only
29 about 50 lines of code.
30 
31 A shortcoming of this approach is that, except for Ctrl-C, break characters
32 like ^P, ^B, and ^U require the user to type RETURN to end the input line
33 and allow the character to be processed.
34 
35 The thread will set a signal whenever a line of input is delivered so that
36 Nyquist can block waiting for input. The signal will also be set when a
37 ^C or ^G is detected.
38 
39 For flexibility, compatibility with the Delphi IDE (NyqIDE) is retained by
40 continuing to check for Windows process messages.
41 */

104 void input_thread_run(void *args)
105 {
106  int c;
107  /* this gets called, possible later, in io_init() in userio.c, but
108  * that doesn't seem to prevent this thread from being killed by
109  * CTRL-C, so call it here to be safe.
110  */
111  SetConsoleCtrlHandler(ctrl_c_handler, TRUE);
112  /* printf("input_thread_run\n"); */
113 
114  while (!buffer_eof) {
115  int head;
116  c = getchar();
117  if (c == EOF && abort_flag) {
118  // when user types ^C, an EOF is generated for some reason.
119  // Ignore it...
120  if (abort_flag == ABORT_LEVEL) c = ABORT_CHAR;
121  else c = BREAK_CHAR;
122  } else if (c == ABORT_CHAR) {
123  abort_flag = ABORT_LEVEL;
124  } else if (!abort_flag && c == BREAK_CHAR) {
125  // notice that a break will be ignored until XLISP
126  // handles the ABORT_LEVEL
127  abort_flag = BREAK_LEVEL;
128  } else if (c == BREAK_CHAR) {
129  ; // ignore this because abort_flag is set to ABORT_LEVEL
130  } else if (c == '\005' || c == '\006') { // control-e or control-f
131  ; // ignore these. IDE will send control-f to turn off echo, but
132  // under Windows, echo is already turned off. We filter control-f
133  // here to avoid generating an error message. Maybe the IDE should
134  // not send control-f in the first place, but the IDE is cross-platform
135  // and does not know it's running under Windows, whereas this file
136  // is platform dependent.
137  } else if (c == '\016') { // begin hidden message
138 #define MSGBUF_MAX 64
139  char msgbuf[MSGBUF_MAX];
140  int msgbufx = 0;
141  char type_char = getchar(); // read message type character
142  printf("begin hidden message: %c\n", type_char);
143  if (type_char == EOF) buffer_eof = TRUE;
144  else {
145  // message is terminated by '\021'
146  while ((c = getchar()) != '\021' &&
147  c != EOF &&
148  msgbufx < MSGBUF_MAX - 1) {
149  msgbuf[msgbufx++] = c;
150  }
151  msgbuf[msgbufx++] = 0;
152  printf("message: %s\n", msgbuf);
153  if (c == EOF) buffer_eof = TRUE;
154  else if (msgbufx < MSGBUF_MAX) {
155  if (type_char == 'S') { // slider change message
156  // message format is index<space>value
157  int index;
158  float value;
159  if (sscanf(msgbuf, "%d %g", &index, &value) == 2) {
160  set_slider(index, value);
161  printf("set_slider %d %g\n", index, value);
162  }
163  }
164  }
165  }
166  } else if (c == EOF) {
167  buffer_eof = TRUE;
168  } else {
169  // insert character into the FIFO
170  head = (input_buffer_head + 1) & input_buffer_mask;
171  while (head == input_buffer_tail) Sleep(100);
172  input_buffer[input_buffer_head] = c;
173 #ifdef DEBUG_INPUT
174  if (to_input_buffer) putc(c, to_input_buffer);
175 #endif
176  input_buffer_head = head;
177  }
178  if (c == '\n' || abort_flag || buffer_eof) {
179  SetEvent(input_buffer_ready);
180  // wake up Nyquist if it is waiting for input
181  }
182  }
183  // printf("Input thread exiting\n");
184 }

254 int ostgetc (void)
255 {
256  int c;
257  NEED_INPUT;
258  while (!buffer_eof && (input_buffer_tail == input_buffer_head)) {
259  oscheck();
260  WaitForSingleObject(input_buffer_ready, INFINITE);
261  }
262  if (buffer_eof) c = EOF;
263  else {
264  c = input_buffer[input_buffer_tail];
265  input_buffer_tail = (input_buffer_tail + 1) & input_buffer_mask;
266  }
267  if (c == '\025') { // control-u
268  xlcleanup();
269  } else if (c == '\020') { // control-p
270  xlcontinue();
271  } else if (c == '\024') { // control-t
272  xinfo();
273  }
274  return c;
275 }

299 void oscheck (void) {
300  MSG lpMsg;
301 
302 #if OSC
303  if (nosc_enabled) nosc_poll();
304 #endif
305 
306  // check_aborted(); -- call to userio.c superceded by code here in winstuff.c
307 // printf("Current Thread: %d\n", GetCurrentThreadId()); //for debugging use
308  // look for Windows messages from NyqIDE (a Delphi program)
309  if ((redirect_flag) && (PeekMessage(&lpMsg, NULL, 0, 0, PM_REMOVE)!=0)) {
310  if (lpMsg.message == WM_CHAR) {
311  switch (lpMsg.wParam) {
312  case ABORT_CHAR: abort_flag = ABORT_LEVEL;
313  break;
314  case BREAK_CHAR: // for nyquist, defined to be 2
315  case 7: // NyqIDE sends 7 (BEL) as break character
316  abort_flag = BREAK_LEVEL;
317  break;
318  }
319  // printf("Get message: %d %d %d\n", lpMsg.wParam, BREAK_CHAR, abort_flag); //for debugging use
320  }
321  }
322  if (abort_flag == ABORT_LEVEL) {
323  abort_flag = 0;
324  osflush();
325  xltoplevel();
326  } else if (abort_flag == BREAK_LEVEL) {
327  abort_flag = 0;
328  osflush();
329  xlbreak("BREAK", s_unbound);
330  }
331  run_time++;
332  if (run_time % 30 == 0) {
333  // maybe we should call fflush here like in Unix; I'm not sure if this is
334  // a bug or it is not necessary for Windows - RBD
335  if (run_time_limit > 0 && run_time > run_time_limit) {
336  xlfatal("Run time limit exceeded");
337  }
338  if (memory_limit > 0 &&
339  npools * MAXPOOLSIZE + table_memory + total >
340  memory_limit * 1000000) {
341  xlfatal("Memory limit exceeded");
342  }
343  }
344 }
