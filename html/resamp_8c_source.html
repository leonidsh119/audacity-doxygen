25 /* Algorithm:
26  To resample, we convolve a sinc function with the input stream at
27 times corresponding to the output samples. This requires a sliding
28 window on the input samples. Since samples are accessed a block at a
29 time, the places where the sliding window would span two blocks are
30 too tricky for me. Instead of trying to manage the breaks across
31 blocks, I copy the blocks into another buffer (called X). When the
32 sliding window reaches the end of X, the samples at the end of X
33 are copied to the beginning of X, the remainder of X is filled with
34 new samples, and the computation continues. The trickiest part of
35 all this is keeping all the pointers and phase accumulators correct
36 when the sliding window is relocated from the end of X to the
37 beginning.
38  Although there are different ways to do this, I decided that the
39 output would always go directly to a Nyquist sample block, so the
40 resampling routines (SrcUp and SrcUD) are always called upon to
41 compute max_sample_block_len samples (except that a partial block
42 may be computed when the input sound terminates).
43  To compute max_sample_block_len samples, the input buffer needs:
44 
45  - max_sample_block_len/factor samples, where factor is the ratio of
46  the new sample rate to the old one. I.e. if upsampling by a factor
47  of 2, the input buffer needs half the samples of the output block
48  size.
49 
50  - additional samples the size of the sliding window. Since the
51  output is taken from the center of the window, we can't take
52  samples from the first or last windowsize/2 samples.
53 
54  - to simplify rounding, we throw in some extra samples. This costs
55  only a bit of space and an extra copy for each spare sample.
56 
57  The window size is determined by the particular filter used and
58 by factor (the sample rate ratio). The filter size is Nmult, the
59 number of filter coefficients. When upsampling, this is the window
60 size (the filter acts as a reconstruction filter for the additional
61 samples). When downsampling, the filter is stretched by 1/factor
62 (the filter acts as an anti-aliasing low-pass filter).
63 
64 */

92 static int SrcUp(float X[], float Y[], double factor, double *Time,
93  int Nx, int Nwing, double LpScl,
94  float Imp[], float ImpD[], boolean Interp)
95 {
96  mem_float *Xp, *Ystart;
97  fast_float v;
98 
99  double dt; /* Step through input signal */
100  mem_float *Yend; /* When Y reaches Yend, return to user */
101 
102 /* nyquist_printf("SrcUp: interpFilt %d\n", Interp);*/
103 
104  dt = 1.0/factor; /* Output sampling period */
105 
106  Ystart = Y;
107  Yend = Y + Nx;
108  while (Y < Yend) {
109  long iTime = (long) *Time;
110  Xp = &X[iTime]; /* Ptr to current input sample */
111  /* Perform left-wing inner product */
112  v = FilterUp(Imp, ImpD, Nwing, Interp, Xp, *Time - iTime, -1);
113  /* Perform right-wing inner product */
114  v += FilterUp(Imp, ImpD, Nwing, Interp, Xp+1,
115  (1 + iTime) - *Time, 1);
116  v *= LpScl; /* Normalize for unity filter gain */
117 /* nyquist_printf("SrcUp output sample %g\n", v); */
118  *Y++ = (float) v;
119  *Time += dt; /* Move to next sample by time increment */
120  }
121  return (Y - Ystart); /* Return the number of output samples */
122 }

127 static int SrcUD(float X[], float Y[], double factor, double *Time,
128  int Nx, int Nwing, double LpScl,
129  float Imp[], float ImpD[], boolean Interp)
130 {
131  mem_float *Xp, *Ystart;
132  fast_float v;
133 
134  double dh; /* Step through filter impulse response */
135  double dt; /* Step through input signal */
136  mem_float *Yend; /* When Y reaches Yend, return to user */
137 
138  dt = 1.0/factor; /* Output sampling period */
139 
140  dh = MIN(Npc, factor*Npc); /* Filter sampling period */
141 
142  Ystart = Y;
143  Yend = Y + Nx;
144  while (Y < Yend) {
145  long iTime = (long) *Time;
146  Xp = &X[iTime]; /* Ptr to current input sample */
147  v = FilterUD(Imp, ImpD, Nwing, Interp, Xp, *Time - iTime,
148  -1, dh); /* Perform left-wing inner product */
149  v += FilterUD(Imp, ImpD, Nwing, Interp, Xp+1, (1 + iTime) - *Time,
150  1, dh); /* Perform right-wing inner product */
151  v *= LpScl; /* Normalize for unity filter gain */
152  *Y++ = (float) v;
153  *Time += dt; /* Move to next sample by time increment */
154  }
155  return (Y - Ystart); /* Return the number of output samples */
156 }

159 void resample__fetch(snd_susp_type a_susp, snd_list_type snd_list)
160 {
161  resample_susp_type susp = (resample_susp_type) a_susp;
162  int togo;
163  int Nout;
164  sample_block_type out;
165  /* note that in this fetch routine, out_ptr is used to remember where
166  * to put the "real" output, while X_ptr_reg is used in the inner
167  * loop that copies input samples into X, a buffer
168  */
169  register sample_block_values_type out_ptr;
170  falloc_sample_block(out, "resample__fetch");
171  out_ptr = out->samples;
172  snd_list->block = out;
173 
174 /* Algorithm:
175  Fetch samples until X (the buffered input) is full. X stores enough
176 contiguous samples that a sliding window convolving with the filter
177 coefficients can output a full block without sliding beyond the range
178 of X. Every time we reenter resample__fetch, we take the remaining
179 samples at the end of X, shift them to the beginning, and refill.
180  After X is full, call on SrcUp or SrcUD to compute an output block.
181  The first time resample__fetch is called, the fill pointer Xp will
182 point near the beginning of X, indicating that no previously read
183 samples need to be shifted from the end of X to the beginning.
184 */
185 
186  /* first, shift samples from end of X to beginning if necessary */
187  if (susp->Xp > 2 * susp->Xoff) {
188  int i;
189  int shiftcount = (long) (susp->Time) - susp->Xoff;
190 
191 /* nyquist_printf("shift %d from %d to %lx\n", susp->Xsize + susp->Xoff - susp->Xp, susp->Xp - susp->Xoff, susp->X); */
192  for (i = 0; i < susp->Xp - shiftcount; i++) {
193  susp->X[i] = susp->X[i + shiftcount];
194 /* if (susp->X[i] == 0.0) nyquist_printf("shifted zero to X[%d]\n", i);*/
195  }
196  susp->Time -= shiftcount;
197  susp->Xp -= shiftcount;
198  }
199 
200  while (susp->Xp < susp->Xsize) { /* outer loop */
201  /* read samples from susp->s into X */
202  togo = susp->Xsize - susp->Xp;
203  /* don't run past the s input sample block. If termination or
204  * logical stop info become available, translate to susp->terminate_cnt
205  * and susp->log_stop_cnt.
206  */
207  susp_check_term_log_samples(s, s_ptr, s_cnt);
208  togo = MIN(togo, susp->s_cnt);
209 
210  memcpy(susp->X + susp->Xp, susp->s_ptr, togo * sizeof(sample_type));
211  susp->s_ptr += togo;
212  susp_took(s_cnt, togo);
213  susp->Xp += togo;
214  } /* outer loop */
215 
216  /* second, compute samples to output, this is done in one pass because
217  * we have enough data in X
218  */
219 
220  /* don't run past terminate time */
221  togo = max_sample_block_len;
222  if (susp->terminate_cnt != UNKNOWN &&
223  susp->terminate_cnt <= susp->susp.current + max_sample_block_len) {
224  togo = susp->terminate_cnt - susp->susp.current;
225  }
226  if (!susp->logically_stopped &&
227  susp->susp.log_stop_cnt != UNKNOWN) {
228  int to_stop = susp->susp.log_stop_cnt - susp->susp.current;
229  assert(to_stop >= 0);
230  if (to_stop < togo) {
231  if (to_stop == 0) susp->logically_stopped = true;
232  else togo = to_stop;
233  }
234  }
235  if (togo == 0) {
236 /* stdputstr("resamp calling snd_list_terminate\n"); */
237  snd_list_terminate(snd_list);
238  } else {
239  if (susp->factor >= 1) { /* SrcUp() is faster if we can use it */
240  Nout = SrcUp(susp->X, out_ptr, susp->factor, &susp->Time,
241  togo, susp->Nwing, susp->LpScl, susp->Imp,
242  susp->ImpD, susp->interpFilt);
243  } else {
244  Nout = SrcUD(susp->X, out_ptr, susp->factor, &susp->Time,
245  togo, susp->Nwing, susp->LpScl, susp->Imp,
246  susp->ImpD, susp->interpFilt);
247  }
248  snd_list->block_len = togo;
249  susp->susp.current += togo;
250  }
251 #ifdef RESAMPTEST
252  for (n = 0; n < snd_list->block_len; n++) {
253  if (out->samples[n] == 0.0) {
254  nyquist_printf("resamp: zero at samples[%d]\n", n);
255  }
256  }
257 #endif
258 /*
259  if (susp->logically_stopped) {
260  snd_list->logically_stopped = true;
261  } else if (susp->susp.log_stop_cnt == susp->susp.current) {
262  susp->logically_stopped = true;
263  }
264  */
265 } /* resample__fetch */

294 {

306  susp->LpScl = SMALL_FILTER_SCALE / 32768.0;

314 
315  /* factor in the scale factor of s, since resample is linear */
316  susp->LpScl *= s->scale;
317 
318  susp->terminate_cnt = UNKNOWN;
319  /* initialize susp state */
320  susp->susp.free = resample_free;
321  susp->susp.sr = sr;
322  susp->susp.t0 = s->t0;
323  susp->susp.mark = resample_mark;
324  susp->susp.print_tree = resample_print_tree;
325  susp->susp.name = "resample";
326  susp->logically_stopped = false;
327  susp->susp.log_stop_cnt = logical_stop_cnt_cvt(s);
328  susp->susp.current = 0;
329  susp->s = s;
330  susp->s_cnt = 0;
331  susp->Xoff = (int) (((susp->Nmult + 1) / 2.0) * MAX(1.0, 1.0 / susp->factor) + 10);
332  susp->Xsize = (long) ((max_sample_block_len / susp->factor) + 2 * susp->Xoff);

338

340  susp->susp.sr, 1.0 /* scale factor */);
341 }
