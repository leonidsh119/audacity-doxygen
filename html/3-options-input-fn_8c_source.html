4 /* Example 3: extends example 2 with multiple channels, multiple datatypes,
5  * and other options.
6  *
7  * The application provides an input function, called on demand by libsoxr, in
8  * response to calls to soxr_output(); compared to the `process' approach
9  * (illustrated in example 2) this requires that the application implements
10  * less logic, but one more function.
11  *
12  * The 11 arguments (which are optional, from last to first) are:
13  * INPUT-RATE As example 2
14  * OUTPUT-RATE Ditto
15  * NUM-CHANNELS Number of interleaved channels
16  * IN-DATATYPE# 0:float32 1:float64 2:int32 3:int16
17  * OUT-DATATYPE# Ditto
18  * Q-RECIPE Quality recipe (in hex) See soxr.h
19  * Q-FLAGS Quality flags (in hex) See soxr.h
20  * PASSBAND-END %
21  * STOPBAND-BEGIN %
22  * PHASE-RESPONSE [0,100]
23  * USE-THREADS 1 to use multi-threading (where available)
24  */

31 static size_t input_fn(input_context_t * p, soxr_cbuf_t * buf, size_t len)
32 {
33  /* Read one block into the buffer, ready to be input to the resampler: */
34  len = fread(p->ibuf, p->isize, len, stdin); /* Actual len read may be less. */
35 
36  /* Inform the resampler of the data's whereabouts (which could be anywhere, in
37  * a freshly malloc'd buffer, for example): */
38  *buf = (!len && ferror(stdin))? NULL : p->ibuf; /* NULL if error occurred. */
39 
40  return len; /* # of samples per channel to input. */
41 }
