75 static LV2_Handle
76 instantiate(const LV2_Descriptor* descriptor,
77  double rate,
78  const char* bundle_path,
79  const LV2_Feature* const* features)
80 {
81  (void)descriptor; // Unused variable
82  (void)bundle_path; // Unused variable
83 
84  // Allocate and initialise instance structure.
85  EgScope* self = (EgScope*)calloc(1, sizeof(EgScope));
86  if (!self) {
87  return NULL;
88  }
89 
90  // Get host features
91  for (int i = 0; features[i]; ++i) {
92  if (!strcmp(features[i]->URI, LV2_URID__map)) {
93  self->map = (LV2_URID_Map*)features[i]->data;
94  } else if (!strcmp(features[i]->URI, LV2_LOG__log)) {
95  self->log = (LV2_Log_Log*)features[i]->data;
96  }
97  }
98 
99  if (!self->map) {
100  fprintf(stderr, "EgScope.lv2 error: Host does not support urid:map\n");
101  free(self);
102  return NULL;
103  }
104 
105  // Decide which variant to use depending on the plugin URI
106  if (!strcmp(descriptor->URI, SCO_URI "#Stereo")) {
107  self->n_channels = 2;
108  } else if (!strcmp(descriptor->URI, SCO_URI "#Mono")) {
109  self->n_channels = 1;
110  } else {
111  free(self);
112  return NULL;
113  }
114 
115  // Initialise local variables
116  self->ui_active = false;
117  self->send_settings_to_ui = false;
118  self->rate = rate;
119 
120  // Set default UI settings
121  self->ui_spp = 50;
122  self->ui_amp = 1.0;
123 
124  // Map URIs and initialise forge/logger
125  map_sco_uris(self->map, &self->uris);
126  lv2_atom_forge_init(&self->forge, self->map);
127  lv2_log_logger_init(&self->logger, self->map, self->log);
128 
129  return (LV2_Handle)self;
130 }

180 tx_rawaudio(LV2_Atom_Forge* forge,

182  const int32_t channel,
183  const size_t n_samples,

190  lv2_atom_forge_object(forge, &frame, 0, uris->RawAudio);
191 
192  // Add integer 'channelID' property
193  lv2_atom_forge_key(forge, uris->channelID);
194  lv2_atom_forge_int(forge, channel);
195 
196  // Add vector of floats 'audioData' property
197  lv2_atom_forge_key(forge, uris->audioData);
198  lv2_atom_forge_vector(

201  // Close off object
202  lv2_atom_forge_pop(forge, &frame);

206 static void
207 run(LV2_Handle handle, uint32_t n_samples)
208 {
209  EgScope* self = (EgScope*)handle;
210 
211  /* Ensure notify port buffer is large enough to hold all audio-samples and
212  configuration settings. A minimum size was requested in the .ttl file,
213  but check here just to be sure.
214 
215  TODO: Explain these magic numbers.
216  */
217  const size_t size = (sizeof(float) * n_samples + 64) * self->n_channels;
218  const uint32_t space = self->notify->atom.size;
219  if (space < size + 128) {
220  /* Insufficient space, report error and do nothing. Note that a
221  real-time production plugin mustn't call log functions in run(), but
222  this can be useful for debugging and example purposes.
223  */
224  lv2_log_error(&self->logger, "Buffer size is insufficient\n");
225  return;
226  }
227 
228  // Prepare forge buffer and initialize atom-sequence
229  lv2_atom_forge_set_buffer(&self->forge, (uint8_t*)self->notify, space);
230  lv2_atom_forge_sequence_head(&self->forge, &self->frame, 0);
231 
232  /* Send settings to UI
233 
234  The plugin can continue to run while the UI is closed and re-opened.
235  The state and settings of the UI are kept here and transmitted to the UI
236  every time it asks for them or if the user initializes a 'load preset'.
237  */
238  if (self->send_settings_to_ui && self->ui_active) {
239  self->send_settings_to_ui = false;
240  // Forge container object of type 'ui_state'
241  LV2_Atom_Forge_Frame frame;
242  lv2_atom_forge_frame_time(&self->forge, 0);
243  lv2_atom_forge_object(&self->forge, &frame, 0, self->uris.ui_State);
244 
245  // Add UI state as properties
246  lv2_atom_forge_key(&self->forge, self->uris.ui_spp);
247  lv2_atom_forge_int(&self->forge, self->ui_spp);
248  lv2_atom_forge_key(&self->forge, self->uris.ui_amp);
249  lv2_atom_forge_float(&self->forge, self->ui_amp);
250  lv2_atom_forge_key(&self->forge, self->uris.param_sampleRate);
251  lv2_atom_forge_float(&self->forge, self->rate);
252  lv2_atom_forge_pop(&self->forge, &frame);
253  }
254 
255  // Process incoming events from GUI
256  if (self->control) {
257  const LV2_Atom_Event* ev = lv2_atom_sequence_begin(
258  &(self->control)->body);
259  // For each incoming message...
260  while (!lv2_atom_sequence_is_end(
261  &self->control->body, self->control->atom.size, ev)) {
262  // If the event is an atom:Blank object
263  if (ev->body.type == self->uris.atom_Blank) {
264  const LV2_Atom_Object* obj = (const LV2_Atom_Object*)&ev->body;
265  if (obj->body.otype == self->uris.ui_On) {
266  // If the object is a ui-on, the UI was activated
267  self->ui_active = true;
268  self->send_settings_to_ui = true;
269  } else if (obj->body.otype == self->uris.ui_Off) {
270  // If the object is a ui-off, the UI was closed
271  self->ui_active = false;
272  } else if (obj->body.otype == self->uris.ui_State) {
273  // If the object is a ui-state, it's the current UI settings
274  const LV2_Atom* spp = NULL;
275  const LV2_Atom* amp = NULL;
276  lv2_atom_object_get(obj, self->uris.ui_spp, &spp,
277  self->uris.ui_amp, &amp,
278  0);
279  if (spp) {
280  self->ui_spp = ((const LV2_Atom_Int*)spp)->body;
281  }
282  if (amp) {
283  self->ui_amp = ((const LV2_Atom_Float*)amp)->body;
284  }
285  }
286  }
287  ev = lv2_atom_sequence_next(ev);
288  }
289  }
290 
291  // Process audio data
292  for (uint32_t c = 0; c < self->n_channels; ++c) {
293  if (self->ui_active) {
294  // If UI is active, send raw audio data to UI
295  tx_rawaudio(&self->forge, &self->uris, c, n_samples, self->input[c]);
296  }
297  // If not processing audio in-place, forward audio
298  if (self->input[c] != self->output[c]) {
299  memcpy(self->output[c], self->input[c], sizeof(float) * n_samples);
300  }
301  }
302 
303  // Close off sequence
304  lv2_atom_forge_pop(&self->forge, &self->frame);
305 }
