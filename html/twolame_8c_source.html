194  // If the MPEG version has not been set, then choose automatically

198  if (glopts->version < 0) {

208  if (glopts->mode == TWOLAME_AUTO_MODE) {
209  if (glopts->num_channels_in == 2)

211  else

213  if (glopts->verbosity >= 3) {

241  } else {

243  case 48000:

369 /* Scale the samples in the frame sample buffer
370  using the user specified values
371  and downmix/upmix according to the number of input/output channels
372 */
373 static void scale_and_mix_samples(twolame_options * glopts)
374 {
375  int num_samples = glopts->samples_in_buffer;
376  int i;
377 
378  // apply scaling to both channels
379  if (glopts->scale != 0 && glopts->scale != 1.0) {
380  for (i = 0; i < num_samples; ++i) {
381  glopts->buffer[0][i] *= glopts->scale;
382  if (glopts->num_channels_in == 2)
383  glopts->buffer[1][i] *= glopts->scale;
384  }
385  }
386  // apply scaling to channel 0 (left)
387  if (glopts->scale_left != 0 && glopts->scale_left != 1.0) {
388  for (i = 0; i < num_samples; ++i) {
389  glopts->buffer[0][i] *= glopts->scale_left;
390  }
391  }
392  // apply scaling to channel 1 (right)
393  if (glopts->scale_right != 0 && glopts->scale_right != 1.0) {
394  for (i = 0; i < num_samples; ++i) {
395  glopts->buffer[1][i] *= glopts->scale_right;
396  }
397  }
398  // Downmix to Mono if 2 channels in and 1 channel out
399  if (glopts->num_channels_in == 2 && glopts->num_channels_out == 1) {
400  for (i = 0; i < num_samples; ++i) {
401  glopts->buffer[0][i] = ((long) glopts->buffer[0][i] + glopts->buffer[1][i]) / 2;
402  glopts->buffer[1][i] = 0;
403  }
404  }
405  // Upmix to Stereo if 2 channels out and 1 channel in
406  if (glopts->num_channels_in == 1 && glopts->num_channels_out == 2) {
407  for (i = 0; i < num_samples; ++i) {
408  glopts->buffer[1][i] = glopts->buffer[0][i];
409  }
410  }
411 
412 }

414 /*
415  Encode a single frame of audio from 1152 samples
416  Audio samples are taken from glopts->buffer
417  Encoded bit stream is placed in to parameter bs
418  (not intended for use outside the library)
419 
420  Returns the size of the frame
421  or -1 if there is an error
422 */
423 static int encode_frame(twolame_options * glopts, bit_stream * bs)
424 {
425  int nch = glopts->num_channels_out;
426  int sb, ch, adb, i;
427  unsigned long frameBits, initial_bits;
428  short sam[2][1056];
429 
430  if (!glopts->twolame_init) {
431  fprintf(stderr, "Please call twolame_init_params() before starting encoding.\n");
432  return -1;
433  }
434  // Scale and mix the input buffer
435  scale_and_mix_samples(glopts);
436 
437 
438  // Clear the saved audio buffer
439  memset((char *) sam, 0, sizeof(sam));
440 
441  // Number of bits to calculate CRC on
442  glopts->num_crc_bits = 0;
443 
444  // Store the number of bits initially in the bit buffer
445  initial_bits = buffer_sstell(bs);
446 
447  adb = available_bits(glopts);
448 
449  /* allow the user to reserve some space at the end of the frame This will however leave fewer
450  bits for the audio. Need to do a sanity check here to see that there are *some* bits left. */
451  if (glopts->num_ancillary_bits > 0.6 * adb) {
452  /* Trying to reserve more than 60% of the frame. 0.6 is arbitrary. but since most
453  applications probably only want to reserve a few bytes, this seems fine. Typical frame
454  size is about 800bytes */
455  fprintf(stderr,
456  "You're trying to reserve more than 60%% of the mpeg frame for ancillary data\n");
457  fprintf(stderr, "This is probably an error. But I'll keep going anyway...\n");
458  }
459 
460  adb -= glopts->num_ancillary_bits;
461 
462 
463  /* MFC 26 July 2003 Doing DAB became a bit harder in the reorganisation of the code. Now there
464  is no guarantee that there is more than one frame in the bitbuffer. But DAB requires that
465  the CRC for the *current* frame be written at the end of the *previous* frame. Workaround:
466  Users (Nicholas?) wanting to implement DAB will have to do some work in the frontend. First:
467  Reserve some bits for yourself (options->num_ancillary_bits) Second: Put the encoder into
468  "single frame mode" i.e. only read 1152 samples per channel.
469  (frontendoptions->singleFrameMode) Third: When you receive each mp2 frame back from the
470  library, you'll have to insert the options->dabCrc[i] values into the end of the frame
471  yourself. (DAB crc calc is done below) The frontend will have to keep the previous frame in
472  memory. As of 26July all that needs to be done is for the frontend to buffer one frame in
473  memory, such that the CRC for the next frame can be written in at the end of it. */
474 
475  {
476  int gr, bl, ch;
477  /* New polyphase filter Combines windowing and filtering. Ricardo Feb'03 */
478  for (gr = 0; gr < 3; gr++)
479  for (bl = 0; bl < 12; bl++)
480  for (ch = 0; ch < nch; ch++)
481  window_filter_subband(&glopts->smem,
482  &glopts->buffer[ch][gr * 12 * 32 + 32 * bl], ch,
483  &(*glopts->sb_sample)[ch][gr][bl][0]);
484  }
485 
486  scalefactor_calc(*glopts->sb_sample, glopts->scalar, nch, glopts->sblimit);
487  find_sf_max(glopts, glopts->scalar, glopts->max_sc);
488  if (glopts->mode == TWOLAME_JOINT_STEREO) {
489  // this way we calculate more mono than we need but it is cheap
490  combine_lr(*glopts->sb_sample, *glopts->j_sample, glopts->sblimit);
491  scalefactor_calc(glopts->j_sample, &glopts->j_scale, 1, glopts->sblimit);
492  }
493 
494  if ((glopts->quickmode == TRUE) && (++glopts->psycount % glopts->quickcount != 0)) {
495  /* We're using quick mode, so we're only calculating the model every 'quickcount' frames.
496  Otherwise, just copy the old ones across */
497  for (ch = 0; ch < nch; ch++) {
498  for (sb = 0; sb < SBLIMIT; sb++) {
499  glopts->smr[ch][sb] = glopts->smrdef[ch][sb];
500  }
501  }
502  } else {
503  // calculate the psymodel
504  switch (glopts->psymodel) {
505  case -1:
506  psycho_n1(glopts, glopts->smr, nch);
507  break;
508  case 0: // Psy Model A
509  psycho_0(glopts, glopts->smr, glopts->scalar);
510  break;
511  case 1:
512  psycho_1(glopts, glopts->buffer, glopts->max_sc, glopts->smr);
513  break;
514  case 2:
515  psycho_2(glopts, glopts->buffer, sam, glopts->smr);
516  break;
517  case 3:
518  // Modified psy model 1
519  psycho_3(glopts, glopts->buffer, glopts->max_sc, glopts->smr);
520  break;
521  case 4:
522  // Modified psy model 2
523  psycho_4(glopts, glopts->buffer, sam, glopts->smr);
524  break;
525  default:
526  fprintf(stderr, "Invalid psy model specification: %i\n", glopts->psymodel);
527  return -1;
528  break;
529  }
530 
531  if (glopts->quickmode == TRUE) {
532  // copy the smr values and reuse them later
533  for (ch = 0; ch < nch; ch++) {
534  for (sb = 0; sb < SBLIMIT; sb++)
535  glopts->smrdef[ch][sb] = glopts->smr[ch][sb];
536  }
537  }
538  }
539 
540 
541  sf_transmission_pattern(glopts, glopts->scalar, glopts->scfsi);
542  main_bit_allocation(glopts, glopts->smr, glopts->scfsi, glopts->bit_alloc, &adb);
543 
544  write_header(glopts, bs);
545 
546  // Leave space for 2 bytes of CRC to be filled in later
547  if (glopts->error_protection)
548  buffer_putbits(bs, 0, 16);
549 
550  write_bit_alloc(glopts, glopts->bit_alloc, bs);
551  write_scalefactors(glopts, glopts->bit_alloc, glopts->scfsi, glopts->scalar, bs);
552 
553  subband_quantization(glopts, glopts->scalar, *glopts->sb_sample, glopts->j_scale,
554  *glopts->j_sample, glopts->bit_alloc, *glopts->subband);
555  write_samples(glopts, *glopts->subband, glopts->bit_alloc, bs);
556 
557  // If not all the bits were used, write out a stack of zeros
558  for (i = 0; i < adb; i++)
559  buffer_put1bit(bs, 0);
560 
561 
562  /* MFC July 03 FIXME Write an extra byte for 16/24/32/48 input when padding is on. Something
563  must be going astray with the frame size calcs. This fudge works fine for the moment */
564  if ((glopts->header.samplerate_idx != 0) && (glopts->padding)) // i.e. not a 44.1 or 22kHz
565  // input file
566  buffer_putbits(bs, 0, 8);
567 
568  if (glopts->do_dab) {
569  // Do the CRC calc for DAB stuff if required.
570  // It will be up to the frontend to insert it into the end of the
571  // previous frame.
572  for (i = glopts->dab_crc_len - 1; i >= 0; i--) {
573  dab_crc_calc(glopts, glopts->bit_alloc, glopts->scfsi, glopts->scalar,
574  &glopts->dab_crc[i], i);
575  }
576  }
577  // Allocate space for the reserved ancillary bits
578  for (i = 0; i < glopts->num_ancillary_bits; i++)
579  buffer_put1bit(bs, 0);
580 
581 
582  // Calulate the number of bits in this frame
583  frameBits = buffer_sstell(bs) - initial_bits;
584  if (frameBits % 8) { /* a program failure */
585  fprintf(stderr, "Sent %ld bits = %ld slots plus %ld\n", frameBits, frameBits / 8,
586  frameBits % 8);
587  fprintf(stderr, "If you are reading this, the program is broken\n");
588  fprintf(stderr, "email %s with the command line arguments and other info\n",
589  PACKAGE_BUGREPORT);
590  return -1;
591  }
592  // Store the energy levels at the end of the frame
593  if (glopts->do_energy_levels)
594  do_energy_levels(glopts, bs);
595 
596  // MEANX: Recompute checksum from bitstream
597  if (glopts->error_protection) {
598  unsigned char *frame_ptr = bs->buf + (initial_bits >> 3);
599  crc_writeheader(frame_ptr, glopts->num_crc_bits);
600  }
601  // fprintf(stderr,"Frame size: %li\n\n",frameBits/8);
602 
603  return frameBits / 8;
604 }
