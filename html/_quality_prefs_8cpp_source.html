70 void QualityPrefs::GetNamesAndLabels()
71 {
72  //------------ Dither Names
73  mDitherNames.Add(_("None")); mDitherLabels.Add(Dither::none);
74  mDitherNames.Add(_("Rectangle")); mDitherLabels.Add(Dither::rectangle);
75  mDitherNames.Add(_("Triangle")); mDitherLabels.Add(Dither::triangle);
76  mDitherNames.Add(_("Shaped")); mDitherLabels.Add(Dither::shaped);
77 
78  //------------ Sample Rate Names
79  // JKC: I don't understand the following comment.
80  // Can someone please explain or correct it?
81  // XXX: This should use a previously changed, but not yet saved
82  // sound card setting from the "I/O" preferences tab.
83  // LLL: It means that until the user clicks "Ok" in preferences, the
84  // GetSupportedSampleRates() call should use the devices they
85  // may have changed on the Audio I/O page. As coded, the sample
86  // rates it will return could be completely invalid as they will
87  // be what's supported by the devices that were selected BEFORE
88  // coming into preferences.
89  //
90  // GetSupportedSampleRates() allows passing in device names, but
91  // how do you get at them as they are on the Audio I/O page????
92  for (int i = 0; i < AudioIO::NumStandardRates; i++) {
93  int iRate = AudioIO::StandardRates[i];
94  mSampleRateLabels.Add(iRate);
95  mSampleRateNames.Add(wxString::Format(wxT("%i Hz"), iRate));
96  }
97 
98  mSampleRateNames.Add(_("Other..."));
99 
100  // The label for the 'Other...' case can be any value at all.
101  mSampleRateLabels.Add(44100); // If chosen, this value will be overwritten
102 
103  //------------- Sample Format Names
104  mSampleFormatNames.Add(wxT("16-bit")); mSampleFormatLabels.Add(int16Sample);
105  mSampleFormatNames.Add(wxT("24-bit")); mSampleFormatLabels.Add(int24Sample);
106  mSampleFormatNames.Add(wxT("32-bit float")); mSampleFormatLabels.Add(floatSample);
107 
108  //------------- Converter Names
109  int numConverters = Resample::GetNumMethods();
110  for (int i = 0; i < numConverters; i++) {
111  mConverterNames.Add(Resample::GetMethodName(i));
112  mConverterLabels.Add(i);
113  }
114 }
