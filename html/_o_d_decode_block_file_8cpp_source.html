220 // even if the result is flawed (e.g., refers to nonexistent file),
221 // as testing will be done in DirManager::ProjectFSCK().
222 BlockFile *ODDecodeBlockFile::BuildFromXML(DirManager &dm, const wxChar **attrs)
223 {
224  wxFileName summaryFileName;
225  wxFileName audioFileName;
226  sampleCount aliasStart=0, aliasLen=0;
227  int aliasChannel=0;
228  long nValue;
229  unsigned int decodeType=0;
230 
231  while(*attrs)
232  {
233  const wxChar *attr = *attrs++;
234  const wxChar *value = *attrs++;
235  if (!value)
236  break;
237 
238  const wxString strValue = value;
239  if (!wxStricmp(attr, wxT("summaryfile")) &&
240  // Can't use XMLValueChecker::IsGoodFileName here, but do part of its test.
241  XMLValueChecker::IsGoodFileString(strValue) &&
242  (strValue.Length() + 1 + dm.GetProjectDataDir().Length() <= PLATFORM_MAX_PATH))
243  {
244  if (!dm.AssignFile(summaryFileName, strValue, false))
245  // Make sure summaryFileName is back to uninitialized state so we can detect problem later.
246  summaryFileName.Clear();
247  }
248  else if( !wxStricmp(attr, wxT("audiofile")) )
249  {
250  if (XMLValueChecker::IsGoodPathName(strValue))
251  audioFileName.Assign(strValue);
252  else if (XMLValueChecker::IsGoodFileName(strValue, dm.GetProjectDataDir()))
253  // Allow fallback of looking for the file name, located in the data directory.
254  audioFileName.Assign(dm.GetProjectDataDir(), strValue);
255  else if (XMLValueChecker::IsGoodPathString(strValue))
256  // If the file is missing, we failed XMLValueChecker::IsGoodPathName()
257  // and XMLValueChecker::IsGoodFileName, because both do existence tests,
258  // but we want to keep the reference to the file because it's a good path string.
259  audioFileName.Assign(strValue);
260  }
261  else if (XMLValueChecker::IsGoodInt(strValue) && strValue.ToLong(&nValue))
262  { // integer parameters
263  if (!wxStricmp(attr, wxT("aliasstart")) && (nValue >= 0))
264  aliasStart = nValue;
265  else if (!wxStricmp(attr, wxT("aliaslen")) && (nValue >= 0))
266  aliasLen = nValue;
267  else if (!wxStricmp(attr, wxT("aliaschannel")) && XMLValueChecker::IsValidChannel(aliasChannel))
268  aliasChannel = nValue;
269  else if( !wxStricmp(attr, wxT("decodetype")) )
270  decodeType = nValue;
271  }
272  }
273 
274  return new ODDecodeBlockFile(summaryFileName, audioFileName,
275  aliasStart, aliasLen, aliasChannel,decodeType,
276  0,0,0, false);
277

396 void *ODDecodeBlockFile::CalcSummary(samplePtr buffer, sampleCount len,
397  sampleFormat format)
398 {
399  char* localFullSummary = new char[mSummaryInfo.totalSummaryBytes];
400 
401  memcpy(localFullSummary, bheaderTag, bheaderTagLen);
402 
403  float *summary64K = (float *)(localFullSummary + mSummaryInfo.offset64K);
404  float *summary256 = (float *)(localFullSummary + mSummaryInfo.offset256);
405 
406  float *fbuffer;
407 
408  //mchinen: think we can hack this - don't allocate and copy if we don't need to.,
409  if(format==floatSample)
410  {
411  fbuffer = (float*)buffer;
412  }
413  else
414  {
415  fbuffer = new float[len];
416  CopySamples(buffer, format,
417  (samplePtr)fbuffer, floatSample, len);
418  }
419  sampleCount sumLen;
420  sampleCount i, j, jcount;
421 
422  float min, max;
423  float sumsq;
424 
425  // Recalc 256 summaries
426  sumLen = (len + 255) / 256;
427 
428 
429  for (i = 0; i < sumLen; i++) {
430  min = fbuffer[i * 256];
431  max = fbuffer[i * 256];
432  sumsq = ((float)min) * ((float)min);
433  jcount = 256;
434  if (i * 256 + jcount > len)
435  jcount = len - i * 256;
436  for (j = 1; j < jcount; j++) {
437  float f1 = fbuffer[i * 256 + j];
438  sumsq += ((float)f1) * ((float)f1);
439  if (f1 < min)
440  min = f1;
441  else if (f1 > max)
442  max = f1;
443  }
444 
445  float rms = (float)sqrt(sumsq / jcount);
446 
447  summary256[i * 3] = min;
448  summary256[i * 3 + 1] = max;
449  summary256[i * 3 + 2] = rms;
450  }
451 
452  for (i = sumLen; i < mSummaryInfo.frames256; i++) {
453  // filling in the remaining bits with non-harming/contributing values
454  summary256[i * 3] = FLT_MAX; // min
455  summary256[i * 3 + 1] = -FLT_MAX; // max
456  summary256[i * 3 + 2] = 0.0f; // rms
457  }
458 
459  // Recalc 64K summaries
460  sumLen = (len + 65535) / 65536;
461 
462  for (i = 0; i < sumLen; i++) {
463  min = summary256[3 * i * 256];
464  max = summary256[3 * i * 256 + 1];
465  sumsq = (float)summary256[3 * i * 256 + 2];
466  sumsq *= sumsq;
467 
468  for (j = 1; j < 256; j++) {
469  if (summary256[3 * (i * 256 + j)] < min)
470  min = summary256[3 * (i * 256 + j)];
471  if (summary256[3 * (i * 256 + j) + 1] > max)
472  max = summary256[3 * (i * 256 + j) + 1];
473  float r1 = summary256[3 * (i * 256 + j) + 2];
474  sumsq += r1*r1;
475  }
476 
477  float rms = (float)sqrt(sumsq / 256);
478 
479  summary64K[i * 3] = min;
480  summary64K[i * 3 + 1] = max;
481  summary64K[i * 3 + 2] = rms;
482  }
483  for (i = sumLen; i < mSummaryInfo.frames64K; i++) {
484  summary64K[i * 3] = 0.0f;
485  summary64K[i * 3 + 1] = 0.0f;
486  summary64K[i * 3 + 2] = 0.0f;
487  }
488 
489  // Recalc block-level summary
490  min = summary64K[0];
491  max = summary64K[1];
492  sumsq = (float)summary64K[2];
493  sumsq *= sumsq;
494 
495  for (i = 1; i < sumLen; i++) {
496  if (summary64K[3*i] < min)
497  min = summary64K[3*i];
498  else if (summary64K[3*i+1] > max)
499  max = summary64K[3*i+1];
500  float r1 = (float)summary64K[3*i+2];
501  sumsq += (r1*r1);
502  }
503 
504  mMin = min;
505  mMax = max;
506  mRMS = sqrt(sumsq / sumLen);
507 
508 
509  //if we've used the float sample..
510  if(format!=floatSample)
511  {
512  delete[] fbuffer;
513  }
514  return localFullSummary;
515 }
