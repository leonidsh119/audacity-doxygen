45 /* write N bits into the bit stream */
46 static inline void buffer_putbits(bit_stream * bs, unsigned int val, int N)
47 {
48  static const int putmask[9] = { 0x0, 0x1, 0x3, 0x7, 0xf, 0x1f, 0x3f, 0x7f, 0xff };
49  register int j = N;
50  register int k, tmp;
51 
52  bs->totbit += N;
53  while (j > 0) {
54  k = MIN(j, bs->buf_bit_idx);
55  tmp = val >> (j - k);
56  bs->buf[bs->buf_byte_idx] |= (tmp & putmask[k]) << (bs->buf_bit_idx - k);
57  bs->buf_bit_idx -= k;
58  if (!bs->buf_bit_idx) {
59  bs->buf_bit_idx = 8;
60  bs->buf_byte_idx++;

62  // empty_buffer (bs, minimum);

64  return;
65  }
66  bs->buf[bs->buf_byte_idx] = 0;
67  }
