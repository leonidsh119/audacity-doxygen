103 /*
104 Since Windows 7 by default (callback) exceptions aren't swallowed anymore
105 with native x64 applications. Exceptions can occur in a file dialog when
106 using the hook procedure in combination with third-party utilities.
107 Since Windows 7 SP1 the swallowing of exceptions can be enabled again
108 by using SetProcessUserModeExceptionPolicy.
109 */
110 void ChangeExceptionPolicy()
111 {
112 #if wxUSE_DYNLIB_CLASS
113  gs_changedPolicy = false;
114 
115  wxLoadedDLL dllKernel32(wxT("kernel32.dll"));
116 
117  if ( gs_pfnGetProcessUserModeExceptionPolicy
118  == (GetProcessUserModeExceptionPolicy_t) -1)
119  {
120  wxDL_INIT_FUNC(gs_pfn, GetProcessUserModeExceptionPolicy, dllKernel32);
121  wxDL_INIT_FUNC(gs_pfn, SetProcessUserModeExceptionPolicy, dllKernel32);
122  }
123 
124  if ( !gs_pfnGetProcessUserModeExceptionPolicy
125  || !gs_pfnSetProcessUserModeExceptionPolicy
126  || !gs_pfnGetProcessUserModeExceptionPolicy(&gs_oldExceptionPolicyFlags) )
127  {
128  return;
129  }
130 
131  if ( gs_pfnSetProcessUserModeExceptionPolicy(gs_oldExceptionPolicyFlags
132  | 0x1 /* PROCESS_CALLBACK_FILTER_ENABLED */ ) )
133  {
134  gs_changedPolicy = true;
135  }

240 {

242  {

244  MSWOnInitDialog(hDlg, pOfn);

247  case WM_DESTROY:

455 void FileDialog::FilterFiles(HWND hwnd, bool refresh)
456 {
457  IShellFolder *ishell = NULL;
458  IShellBrowser *ishellbrowser = NULL; // Does not have to be released
459  IShellView *ishellview = NULL;
460  IFolderView *ifolderview = NULL;
461  LPMALLOC imalloc = NULL;
462  HRESULT hr;
463 
464  // Get pointer to the ListView control
465  HWND lv = ::GetDlgItem(::GetDlgItem(hwnd, lst2), 1);
466  if (lv == NULL)
467  {
468  return;
469  }
470 
471  // Get shell's memory allocation interface (must be Release()'d)
472  hr = SHGetMalloc(&imalloc);
473  if ((hr != NOERROR) || (imalloc == NULL))
474  {
475  wxASSERT((hr == NOERROR) && (imalloc != NULL));
476  return;
477  }
478 
479  // Get IShellBrowser interface for current dialog
480  ishellbrowser = (IShellBrowser*)::SendMessage(hwnd, WM_GETISHELLBROWSER, 0, 0);
481  if (ishellbrowser)
482  {
483  // Get IShellBrowser interface for returned browser
484  if (ishellbrowser->QueryActiveShellView(&ishellview) == S_OK)
485  {
486  // Get the IFolderView interface...available on XP or greater
487  ishellview->QueryInterface(IID_IFolderView, (void **)&ifolderview);
488  }
489  }
490 
491  // Init
492  LVITEM lvi;
493  wxZeroMemory(lvi);
494 
495  // Process all items
496  int fltcnt = (int) m_Filters.GetCount();
497  int itmcnt = ListView_GetItemCount(lv);
498  for (int itm = 0; itm < itmcnt; itm++)
499  {
500  // Retrieve the file IDL
501  lvi.iItem = itm;
502  lvi.mask = LVIF_PARAM;
503  if (ListView_GetItem(lv, &lvi) != TRUE)
504  {
505  wxASSERT(FALSE);
506  break;
507  }
508 
509  LPCITEMIDLIST fidl = (LPCITEMIDLIST) lvi.lParam;
510 
511  // On Vista, lParam no longer contains the pidl so retrieve it via the
512  // IFolderView interface. This interface is only available on XP or higher
513  // so if that limitation isn't workable, use IShellView::GetItemObject() to
514  // retrieve items.
515  if (fidl == NULL && ifolderview != NULL)
516  {
517  ifolderview->Item(itm, (LPITEMIDLIST *) &fidl);
518  }
519 
520  if (fidl == NULL)
521  {
522  wxASSERT(fidl != NULL);
523  break;
524  }
525 
526  // Retrieve the IShellFolder interface of the parent (must be Release()'d)
527  if (ishell == NULL)
528  {
529  hr = SHBindToParent(fidl, IID_IShellFolder, (void **)&ishell, NULL);
530  if (!SUCCEEDED(hr))
531  {
532  wxASSERT(SUCCEEDED(hr));
533  break;
534  }
535  }
536 
537  // Get the attributes of the object
538  DWORD attr = SFGAO_FOLDER | SFGAO_BROWSABLE;
539  hr = ishell->GetAttributesOf(1, &fidl, &attr);
540  if (!SUCCEEDED(hr))
541  {
542  wxASSERT(SUCCEEDED(hr));
543  break;
544  }
545 
546  // Allow all folders (things like zip files get filtered below)
547  if ((attr & (SFGAO_FOLDER)) && !(attr & SFGAO_BROWSABLE))
548  {
549  continue;
550  }
551 
552  // Retrieve the parsable name of the object (includes extension)
553  STRRET str;
554  hr = ishell->GetDisplayNameOf(fidl, SHGDN_INFOLDER | SHGDN_FORPARSING, &str);
555  if (hr != NOERROR)
556  {
557  // For some objects, we get back an error of 80070057. I'm assuming this
558  // means there is no way to represent the name (like some sort of virtual name)
559  // or I've not used the correct PIDL. But, in either case, it "probably"
560  // represents some sort of folder (at least in all cases I've seen), so we
561  // simply allow it to display.
562  continue;
563  }
564 
565  // Convert result to wxString
566  wxString filename;
567  switch (str.uType)
568  {
569  case STRRET_WSTR:
570  filename = str.pOleStr;
571  imalloc->Free(str.pOleStr);
572  break;
573 
574  case STRRET_OFFSET:
575  filename = wxString(((char *)fidl) + str.uOffset, wxConvISO8859_1);
576  break;
577 
578  case STRRET_CSTR:
579  filename = wxString(str.cStr, wxConvISO8859_1);
580  break;
581  }
582 
583  // Convert the filename to lowercase (and remember to write filters in lowercase!)
584  filename = filename.Lower();
585 
586  // Attempt to match it to all of our filters
587  bool match = false;
588  for (int flt = 0; flt < fltcnt; flt++)
589  {
590  if (wxMatchWild(m_Filters[flt], filename, false))
591  {
592  match = true;
593  break;
594  }
595  }
596 
597  // Remove it from the display if it didn't match any of the filters.
598  if (!match)
599  {
600  ListView_DeleteItem(lv, itm);
601  itm--;
602  itmcnt--;
603  }
604  }
605 
606  // On Vista and maybe XP, we seem to need to refresh the view after
607  // changing the filters. But, only refresh for type changes and not
608  // selection changes since it causes additional selection change
609  // events to occur.
610  if (ishellview && refresh)
611  {
612  ishellview->Refresh();
613  }
614 
615  // Release the interface
616  if (ifolderview)
617  {
618  ifolderview->Release();
619  }
620 
621  // Release the interface
622  if (ishellview)
623  {
624  ishellview->Release();
625  }
626 
627  // Release the interface
628  if (ishell)
629  {
630  ishell->Release();
631  }
632 
633  // Release the interface
634  if (imalloc)
635  {
636  imalloc->Release();
637  }
638 }
