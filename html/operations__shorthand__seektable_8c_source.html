33 FLAC__bool do_shorthand_operation__add_seekpoints(const char *filename, FLAC__Metadata_Chain *chain, const char *specification, FLAC__bool *needs_write)
34 {
35  FLAC__bool ok = true, found_seektable_block = false;
36  FLAC__StreamMetadata *block = 0;
37  FLAC__Metadata_Iterator *iterator = FLAC__metadata_iterator_new();
38  FLAC__uint64 total_samples = 0;
39  unsigned sample_rate = 0;
40 
41  if(0 == iterator)
42  die("out of memory allocating iterator");
43 
44  FLAC__metadata_iterator_init(iterator, chain);
45 
46  do {
47  block = FLAC__metadata_iterator_get_block(iterator);
48  if(block->type == FLAC__METADATA_TYPE_STREAMINFO) {
49  sample_rate = block->data.stream_info.sample_rate;
50  total_samples = block->data.stream_info.total_samples;
51  }
52  else if(block->type == FLAC__METADATA_TYPE_SEEKTABLE)
53  found_seektable_block = true;
54  } while(!found_seektable_block && FLAC__metadata_iterator_next(iterator));
55 
56  if(total_samples == 0) {
57  flac_fprintf(stderr, "%s: ERROR: cannot add seekpoints because STREAMINFO block does not specify total_samples\n", filename);
58  return false;
59  }
60 
61  if(!found_seektable_block) {
62  /* create a new block */
63  block = FLAC__metadata_object_new(FLAC__METADATA_TYPE_SEEKTABLE);
64  if(0 == block)
65  die("out of memory allocating SEEKTABLE block");
66  while(FLAC__metadata_iterator_prev(iterator))
67  ;
68  if(!FLAC__metadata_iterator_insert_block_after(iterator, block)) {
69  print_error_with_chain_status(chain, "%s: ERROR: adding new SEEKTABLE block to metadata", filename);
70  FLAC__metadata_object_delete(block);
71  return false;
72  }
73  /* iterator is left pointing to new block */
74  FLAC__ASSERT(FLAC__metadata_iterator_get_block(iterator) == block);
75  }
76 
77  FLAC__metadata_iterator_delete(iterator);
78 
79  FLAC__ASSERT(0 != block);
80  FLAC__ASSERT(block->type == FLAC__METADATA_TYPE_SEEKTABLE);
81 
82  if(!grabbag__seektable_convert_specification_to_template(specification, /*only_explicit_placeholders=*/false, total_samples, sample_rate, block, /*spec_has_real_points=*/0)) {
83  flac_fprintf(stderr, "%s: ERROR (internal) preparing seektable with seekpoints\n", filename);
84  return false;
85  }
86

88 
89  if(ok)

91

108 static FLAC__StreamDecoderWriteStatus write_callback_(const FLAC__StreamDecoder *decoder, const FLAC__Frame *frame, const FLAC__int32 * const buffer[], void *client_data)
109 {
110  ClientData *cd = (ClientData*)client_data;
111 
112  (void)buffer;
113  FLAC__ASSERT(0 != cd);
114 
115  if(!cd->error_occurred) {
116  const unsigned blocksize = frame->header.blocksize;
117  const FLAC__uint64 frame_first_sample = cd->samples_written;
118  const FLAC__uint64 frame_last_sample = frame_first_sample + (FLAC__uint64)blocksize - 1;
119  FLAC__uint64 test_sample;
120  unsigned i;
121  for(i = cd->first_seekpoint_to_check; i < cd->seektable_template->num_points; i++) {
122  test_sample = cd->seektable_template->points[i].sample_number;
123  if(test_sample > frame_last_sample) {
124  break;
125  }
126  else if(test_sample >= frame_first_sample) {
127  cd->seektable_template->points[i].sample_number = frame_first_sample;
128  cd->seektable_template->points[i].stream_offset = cd->last_offset - cd->audio_offset;
129  cd->seektable_template->points[i].frame_samples = blocksize;
130  cd->first_seekpoint_to_check++;
131  /* DO NOT: "break;" and here's why:
132  * The seektable template may contain more than one target
133  * sample for any given frame; we will keep looping, generating
134  * duplicate seekpoints for them, and we'll clean it up later,
135  * just before writing the seektable back to the metadata.
136  */
137  }
138  else {
139  cd->first_seekpoint_to_check++;
140  }
141  }
142  cd->samples_written += blocksize;
143  if(!FLAC__stream_decoder_get_decode_position(decoder, &cd->last_offset))
144  return FLAC__STREAM_DECODER_WRITE_STATUS_ABORT;
145  return FLAC__STREAM_DECODER_WRITE_STATUS_CONTINUE;
146  }
147  else
148  return FLAC__STREAM_DECODER_WRITE_STATUS_ABORT;
149 }
