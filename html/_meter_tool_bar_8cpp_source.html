188 void MeterToolBar::OnSize( wxSizeEvent & event) //WXUNUSED(event) )
189 {
190  event.Skip();
191  int width, height;
192 
193  // We can be resized before populating...protect against it
194  if( !mSizer ) {
195  return;
196  }
197 
198  // Update the layout
199  Layout();
200 
201  // Get the usable area
202  wxSize sz = GetSizer()->GetSize();
203  width = sz.x; height = sz.y;
204 
205  int nMeters =
206  ((mRecordMeter ==NULL) ? 0:1) +
207  ((mPlayMeter ==NULL) ? 0:1);
208 
209  bool bHorizontal = ( width > height );
210  bool bEndToEnd = ( nMeters > 1 ) && wxMin( width, height ) < (60 * nMeters);
211 
212  // Default location for second meter
213  wxGBPosition pos( 0, 0 );
214  // If 2 meters, share the height or width.
215  if( nMeters > 1 ){
216  if( bHorizontal ^ bEndToEnd ){
217  height /= nMeters;
218  pos = wxGBPosition( 1, 0 );
219  } else {
220  width /= nMeters;
221  pos = wxGBPosition( 0, 1 );
222  }
223  }
224 
225  if( mRecordMeter ) {
226  mRecordMeter->SetMinSize( wxSize( width, height ));
227  }
228  if( mPlayMeter ) {
229  mPlayMeter->SetMinSize( wxSize( width, height ));
230  mSizer->SetItemPosition( mPlayMeter, pos );
231  }
232 
233  // And make it happen
234  Layout();
235 }
