229 static inline int
230 sord_node_compare(const SordNode* a, const SordNode* b)
231 {
232  if (a == b || !a || !b) {
233  return 0; // Exact or wildcard match
234  } else if (a->node.type != b->node.type) {
235  return a->node.type - b->node.type;
236  }
237 
238  int cmp = 0;
239  switch (a->node.type) {
240  case SERD_URI:
241  case SERD_BLANK:
242  return strcmp((const char*)a->node.buf, (const char*)b->node.buf);
243  case SERD_LITERAL:
244  cmp = strcmp((const char*)sord_node_get_string(a),
245  (const char*)sord_node_get_string(b));
246  if (cmp == 0) {
247  // Note: Can't use sord_node_compare here since it does wildcards
248  if (!a->meta.lit.datatype || !b->meta.lit.datatype) {
249  cmp = a->meta.lit.datatype - b->meta.lit.datatype;
250  } else {
251  cmp = strcmp((const char*)a->meta.lit.datatype->node.buf,
252  (const char*)b->meta.lit.datatype->node.buf);
253  }
254  }
255  if (cmp == 0) {
256  cmp = strcmp(a->meta.lit.lang, b->meta.lit.lang);
257  }
258  default:
259  break;
260  }
261  return cmp;
262 }

315 static inline bool
316 sord_iter_forward(SordIter* iter)
317 {
318  if (!iter->skip_graphs) {
319  iter->cur = zix_tree_iter_next(iter->cur);
320  return zix_tree_iter_is_end(iter->cur);
321  }
322 
323  SordNode** key = (SordNode**)zix_tree_get(iter->cur);
324  const SordQuad initial = { key[0], key[1], key[2], key[3] };
325  while (true) {
326  iter->cur = zix_tree_iter_next(iter->cur);
327  if (zix_tree_iter_is_end(iter->cur))
328  return true;
329 
330  key = (SordNode**)zix_tree_get(iter->cur);
331  for (int i = 0; i < 3; ++i)
332  if (key[i] != initial[i])
333  return false;
334  }
335  assert(false);
336 }

360 static inline bool
361 sord_iter_seek_match_range(SordIter* iter)
362 {
363  if (iter->end)
364  return true;
365 
366  do {
367  const SordNode** key = (const SordNode**)zix_tree_get(iter->cur);
368 
369  if (sord_quad_match_inline(key, iter->pat))
370  return false; // Found match
371 
372  for (int i = 0; i < iter->n_prefix; ++i) {
373  const int idx = iter->ordering[i];
374  if (!sord_id_match(key[idx], iter->pat[idx])) {
375  iter->end = true; // Reached end of valid range
376  return true;
377  }
378  }
379  } while (!sord_iter_forward(iter));
380 
381  return (iter->end = true); // Reached end
382 }

384 static SordIter*
385 sord_iter_new(const SordModel* sord, ZixTreeIter* cur, const SordQuad pat,
386  SordOrder order, SearchMode mode, int n_prefix)
387 {
388  const int* ordering = orderings[order];
389 
390  SordIter* iter = (SordIter*)malloc(sizeof(SordIter));
391  iter->sord = sord;
392  iter->cur = cur;
393  iter->mode = mode;
394  iter->n_prefix = n_prefix;
395  iter->end = false;
396  iter->skip_graphs = order < GSPO;
397  for (int i = 0; i < TUP_LEN; ++i) {
398  iter->pat[i] = pat[i];
399  iter->ordering[i] = ordering[i];
400  }
401 
402  switch (iter->mode) {
403  case ALL:
404  case SINGLE:
405  case RANGE:
406  assert(
407  sord_quad_match_inline((const SordNode**)zix_tree_get(iter->cur),
408  iter->pat));
409  break;
410  case FILTER_RANGE:
411  sord_iter_seek_match_range(iter);
412  break;
413  case FILTER_ALL:
414  sord_iter_seek_match(iter);
415  break;
416  }
417 
418 #ifdef SORD_DEBUG_ITER
419  SordQuad value;
420  sord_iter_get(iter, value);
421  SORD_ITER_LOG("New %p pat=" TUP_FMT " cur=" TUP_FMT " end=%d skip=%d\n",
422  (void*)iter, TUP_FMT_ARGS(pat), TUP_FMT_ARGS(value),
423  iter->end, iter->skip_graphs);
424 #endif
425  return iter;
426 }

449 bool
450 sord_iter_next(SordIter* iter)
451 {
452  if (iter->end)
453  return true;
454 
455  const SordNode** key;
456  iter->end = sord_iter_forward(iter);
457  if (!iter->end) {
458  switch (iter->mode) {
459  case ALL:
460  // At the end if the cursor is (assigned above)
461  break;
462  case SINGLE:
463  iter->end = true;
464  SORD_ITER_LOG("%p reached single end\n", (void*)iter);
465  break;
466  case RANGE:
467  SORD_ITER_LOG("%p range next\n", (void*)iter);
468  // At the end if the MSNs no longer match
469  key = (const SordNode**)zix_tree_get(iter->cur);
470  assert(key);
471  for (int i = 0; i < iter->n_prefix; ++i) {
472  const int idx = iter->ordering[i];
473  if (!sord_id_match(key[idx], iter->pat[idx])) {
474  iter->end = true;
475  SORD_ITER_LOG("%p reached non-match end\n", (void*)iter);
476  break;
477  }
478  }
479  break;
480  case FILTER_RANGE:
481  // Seek forward to next match, stopping if prefix changes
482  sord_iter_seek_match_range(iter);
483  break;
484  case FILTER_ALL:
485  // Seek forward to next match
486  sord_iter_seek_match(iter);
487  break;
488  }
489  } else {
490  SORD_ITER_LOG("%p reached index end\n", (void*)iter);
491  }
492 
493  if (iter->end) {
494  SORD_ITER_LOG("%p Reached end\n", (void*)iter);
495  return true;
496  } else {
497 #ifdef SORD_DEBUG_ITER
498  SordQuad tup;
499  sord_iter_get(iter, tup);
500  SORD_ITER_LOG("%p Increment to " TUP_FMT "\n",
501  (void*)iter, TUP_FMT_ARGS(tup));
502 #endif
503  return false;
504  }
505 }

545 static inline SordOrder
546 sord_best_index(SordModel* sord,
547  const SordQuad pat,
548  SearchMode* mode,
549  int* n_prefix)
550 {
551  const bool graph_search = (pat[TUP_G] != 0);
552 
553  const unsigned sig
554  = (pat[0] ? 1 : 0) * 0x100
555  + (pat[1] ? 1 : 0) * 0x010
556  + (pat[2] ? 1 : 0) * 0x001;
557 
558  SordOrder good[2] = { (SordOrder)-1, (SordOrder)-1 };
559 
560 #define PAT_CASE(sig, m, g0, g1, np) \
561  case sig: \
562  *mode = m; \
563  good[0] = g0; \
564  good[1] = g1; \
565  *n_prefix = np; \
566  break
567 
568  // Good orderings that don't require filtering
569  *mode = RANGE;
570  *n_prefix = 0;
571  switch (sig) {
572  case 0x000:
573  if (graph_search) {
574  *mode = RANGE;
575  *n_prefix = 1;
576  return DEFAULT_GRAPH_ORDER;
577  } else {
578  *mode = ALL;
579  return DEFAULT_ORDER;
580  }
581  case 0x111:
582  *mode = SINGLE;
583  return graph_search ? DEFAULT_GRAPH_ORDER : DEFAULT_ORDER;
584 
585  PAT_CASE(0x001, RANGE, OPS, OSP, 1);
586  PAT_CASE(0x010, RANGE, POS, PSO, 1);
587  PAT_CASE(0x011, RANGE, OPS, POS, 2);
588  PAT_CASE(0x100, RANGE, SPO, SOP, 1);
589  PAT_CASE(0x101, RANGE, SOP, OSP, 2);
590  PAT_CASE(0x110, RANGE, SPO, PSO, 2);
591  }
592 
593  if (*mode == RANGE) {
594  if (sord_has_index(sord, &good[0], n_prefix, graph_search)) {
595  return good[0];
596  } else if (sord_has_index(sord, &good[1], n_prefix, graph_search)) {
597  return good[1];
598  }
599  }
600 
601  // Not so good orderings that require filtering, but can
602  // still be constrained to a range
603  switch (sig) {
604  PAT_CASE(0x011, FILTER_RANGE, OSP, PSO, 1);
605  PAT_CASE(0x101, FILTER_RANGE, SPO, OPS, 1);
606  PAT_CASE(0x110, FILTER_RANGE, SOP, POS, 1);
607  default: break;
608  }
609 
610  if (*mode == FILTER_RANGE) {
611  if (sord_has_index(sord, &good[0], n_prefix, graph_search)) {
612  return good[0];
613  } else if (sord_has_index(sord, &good[1], n_prefix, graph_search)) {
614  return good[1];
615  }
616  }
617 
618  if (graph_search) {
619  *mode = FILTER_RANGE;
620  *n_prefix = 1;
621  return DEFAULT_GRAPH_ORDER;
622  } else {
623  *mode = FILTER_ALL;
624  return DEFAULT_ORDER;
625  }
626 }

628 SordModel*
629 sord_new(SordWorld* world, unsigned indices, bool graphs)
630 {
631  SordModel* sord = (SordModel*)malloc(sizeof(struct SordModelImpl));
632  sord->world = world;
633  sord->n_quads = 0;
634 
635  for (unsigned i = 0; i < (NUM_ORDERS / 2); ++i) {
636  const int* const ordering = orderings[i];
637  const int* const g_ordering = orderings[i + (NUM_ORDERS / 2)];
638 
639  if (indices & (1 << i)) {
640  sord->indices[i] = zix_tree_new(
641  false, sord_quad_compare, (void*)ordering, NULL);
642  if (graphs) {
643  sord->indices[i + (NUM_ORDERS / 2)] = zix_tree_new(
644  false, sord_quad_compare, (void*)g_ordering, NULL);
645  } else {
646  sord->indices[i + (NUM_ORDERS / 2)] = NULL;
647  }
648  } else {
649  sord->indices[i] = NULL;
650  sord->indices[i + (NUM_ORDERS / 2)] = NULL;
651  }
652  }
653 
654  if (!sord->indices[DEFAULT_ORDER]) {
655  sord->indices[DEFAULT_ORDER] = zix_tree_new(
656  false, sord_quad_compare, (void*)orderings[DEFAULT_ORDER], NULL);
657  }
658  if (graphs && !sord->indices[DEFAULT_GRAPH_ORDER]) {
659  sord->indices[DEFAULT_GRAPH_ORDER] = zix_tree_new(
660  false, sord_quad_compare, (void*)orderings[DEFAULT_GRAPH_ORDER], NULL);
661  }
662 
663  return sord;
664 }
