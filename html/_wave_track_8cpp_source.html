638

640  {

642

1019  // The cut line code is not really prepared to handle other situations

1054  // Three cases:
1055

1060  } else

1063  clip->Clear(t0, clip->GetEndTime());

1074  right->Clear(clip->GetStartTime(), t1);

1077 
1078  clipsToDelete.Append(clip);
1079  }
1080  }

1083  * have envelope points, and we need to ensure that the envelope
1084  * outside of the cleared region is not affected. This means
1085  * putting in "glue" points where the clip enters and leaves the

1087  * the region, then one of the glue points will be redundant. */
1088  // clip->Clear keeps points < t0 and >= t1 via Envelope::CollapseRegion
1089  if (clip->GetEnvelope()->GetNumberOfPoints() > 0) { // don't insert env pts if none exist
1090  double val;
1091  if (clip->WithinClip(t0))
1092  { // start of region within clip
1093  val = clip->GetEnvelope()->GetValue(t0);
1094  clip->GetEnvelope()->Insert(t0 - clip->GetOffset() - 1.0/clip->GetRate(), val);
1095  }
1096  if (clip->WithinClip(t1))
1097  { // end of region within clip
1098  val = clip->GetEnvelope()->GetValue(t1);
1099  clip->GetEnvelope()->Insert(t1 - clip->GetOffset(), val);
1100  }
1101  }
1102  if (!clip->Clear(t0,t1))
1103  return false;
1104  clip->GetEnvelope()->RemoveUnneededPoints(t0);
1105  }
1106  }
1107  } else
1108  if (clip->BeforeClip(t1))
1109  {
1110  // Clip is "behind" the region -- offset it unless we're splitting
1111  // or we're using the "don't move other clips" mode
1112  if (!split && editClipCanMove)
1113  clip->Offset(-(t1-t0));
1114  }
1115  }
1116 
1117  for (it=clipsToDelete.GetFirst(); it; it=it->GetNext())
1118  {
1119  mClips.DeleteObject(it->GetData());
1120  delete it->GetData();
1121  }
1122 
1123  for (it=clipsToAdd.GetFirst(); it; it=it->GetNext())
1124  {
1125  mClips.Append(it->GetData());
1126  }
1127 
1128  return true;
1129 }

1207  //
