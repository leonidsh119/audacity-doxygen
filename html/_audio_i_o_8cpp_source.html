347 /*
348 This work queue class, with the aid of the playback ring
349 buffers, coordinates three threads during scrub play:
350 
351 The UI thread which specifies scrubbing intervals to play,
352 
353 The Audio thread which consumes those specifications a first time
354 and fills the ring buffers with samples for play,
355 
356 The PortAudio thread which consumes from the ring buffers, then
357 also consumes a second time from this queue,
358 to figure out how to update mTime
359 
360 -- which the UI thread, in turn, uses to redraw the play head indicator
361 in the right place.
362 
363 Audio produces samples for PortAudio, which consumes them, both in
364 approximate real time. The UI thread might go idle and so the others
365 might catch up, emptying the queue and causing scrub to go silent.
366 The UI thread will not normally outrun the others -- because InitEntry()
367 limits the real time duration over which each enqueued interval will play.
368 So a small, fixed queue size should be adequate.
369 */

463  double Consumer(unsigned long frames)
464  {
465  // Portaudio thread consumes samples and must update
466  // the time for the indicator. This finds the time value.
467 
468  // MAY ADVANCE mTrailingIdx, BUT IT NEVER CATCHES UP TO mMiddleIdx.
469 
470  wxCriticalSectionLocker locker(mUpdating);
471 
472  // Mark entries as partly or fully "consumed" for
473  // purposes of mTime update. It should not happen that
474  // frames exceed the total of samples to be consumed,
475  // but in that case we just use the t1 of the latest entry.
476  while (1)
477  {
478  Entry *pEntry = &mEntries[mTrailingIdx];
479  unsigned long remaining = pEntry->mDuration - pEntry->mPlayed;
480  if (frames >= remaining)
481  {
482  frames -= remaining;
483  pEntry->mPlayed = pEntry->mDuration;
484  }
485  else
486  {
487  pEntry->mPlayed += frames;
488  break;
489  }
490  const unsigned next = (mTrailingIdx + 1) % Size;
491  if (next == mMiddleIdx)
492  break;
493  mTrailingIdx = next;
494  }
495  return mEntries[mTrailingIdx].GetTime(mRate);

509  bool Init(long s0, long s1, long duration, Entry *previous,
510  double maxSpeed, long minStutter, long minSample, long maxSample,
511  bool adjustStart)
512  {
513  if (duration <= 0)
514  return false;
515  double speed = double(abs(s1 - s0)) / duration;
516  bool maxed = false;
517 
518  // May change the requested speed (or reject)
519  if (!adjustStart && speed > maxSpeed)
520  {
521  // Reduce speed to the maximum selected in the user interface.
522  speed = maxSpeed;
523  maxed = true;
524  }
525  else if (!adjustStart &&
526  previous &&
527  previous->mGoal >= 0 &&
528  previous->mGoal == s1)
529  {
530  // In case the mouse has not moved, and playback
531  // is catching up to the mouse at maximum speed,
532  // continue at no less than maximum. (Without this
533  // the final catch-up can make a slow scrub interval
534  // that drops the pitch and sounds wrong.)
535  duration = lrint(speed * duration / maxSpeed);
536  if (duration <= 0)
537  {
538  previous->mGoal = -1;
539  return false;
540  }
541  speed = maxSpeed;
542  maxed = true;
543  }
544 
545  if (speed < GetMinScrubSpeed())
546  // Mixers were set up to go only so slowly, not slower.
547  // This will put a request for some silence in the work queue.
548  speed = 0.0;
549 
550  // Before we change s1:
551  mGoal = maxed ? s1 : -1;
552 
553  // May change s1 or s0 to match speed change:
554  if (adjustStart)
555  {
556  bool silent = false;
557 
558  // Adjust s1 first, and duration, if s1 is out of bounds.
559  // (Assume s0 is in bounds, because it is the last scrub's s1 which was checked.)
560  if (s1 != s0)
561  {
562  const long newS1 = std::max(minSample, std::min(maxSample, s1));
563  if (s1 != newS1)
564  {
565  long newDuration = long(duration * double(newS1 - s0) / (s1 - s0));
566  s1 = newS1;
567  if (newDuration == 0)
568  // Enqueue a silent scrub with s0 == s1
569  silent = true;
570  else
571  // Shorten
572  duration = newDuration;
573  }
574  }
575 
576  if (!silent)
577  {
578  // When playback follows a fast mouse movement by "stuttering"
579  // at maximum playback, don't make stutters too short to be useful.
580  if (duration < minStutter)
581  return false;
582  // Limit diff because this is seeking.
583  const long diff = lrint(std::min(1.0, speed) * duration);
584  if (s0 < s1)
585  s0 = s1 - diff;
586  else
587  s0 = s1 + diff;
588  }
589  }
590  else
591  {
592  // adjust end
593  const long diff = lrint(speed * duration);
594  if (s0 < s1)
595  s1 = s0 + diff;
596  else
597  s1 = s0 - diff;
598 
599  // Adjust s1 again, and duration, if s1 is out of bounds. (Assume s0 is in bounds.)
600  if (s1 != s0)
601  {
602  const long newS1 = std::max(minSample, std::min(maxSample, s1));
603  if (s1 != newS1)
604  {
605  long newDuration = long(duration * double(newS1 - s0) / (s1 - s0));
606  s1 = newS1;
607  if (newDuration == 0)
608  // Enqueue a silent scrub with s0 == s1
609  ;
610  else
611  // Shorten
612  duration = newDuration;
613  }
614  }
615  }
616 
617  mS0 = s0;
618  mS1 = s1;
619  mPlayed = 0;
620  mDuration = duration;
621  return true;
622  }

2707 {

2709  if (mCachedBestRateIn != 0.0 && mCachedBestRateIn == sampleRate) {
2710  return mCachedBestRateOut;
