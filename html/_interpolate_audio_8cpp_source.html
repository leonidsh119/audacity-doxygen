29 // This function is a really dumb, simple way to interpolate audio,
30 // if the more general InterpolateAudio function below doesn't have
31 // enough data to work with. If the bad samples are in the middle,
32 // it's literally linear. If it's on either edge, we add some decay
33 // back to zero.
34 static void LinearInterpolateAudio(float *buffer, int len,
35  int firstBad, int numBad)
36 {
37  int i;
38 
39  float decay = 0.9f;
40 
41  if (firstBad==0) {
42  float delta = buffer[numBad] - buffer[numBad+1];
43  float value = buffer[numBad];
44  i = numBad - 1;
45  while (i >= 0) {
46  value += delta;
47  buffer[i] = value;
48  value *= decay;
49  delta *= decay;
50  i--;
51  }
52  }
53  else if (firstBad + numBad == len) {
54  float delta = buffer[firstBad-1] - buffer[firstBad-2];
55  float value = buffer[firstBad-1];
56  i = firstBad;
57  while (i < firstBad + numBad) {
58  value += delta;
59  buffer[i] = value;
60  value *= decay;
61  delta *= decay;
62  i++;
63  }
64  }
65  else {
66  float v1 = buffer[firstBad-1];
67  float v2 = buffer[firstBad+numBad];
68  float value = v1;
69  float delta = (v2 - v1) / (numBad+1);
70  i = firstBad;
71  while (i < firstBad + numBad) {
72  value += delta;
73  buffer[i] = value;
74  i++;
75  }
76  }

80 // Least Squares AutoRegression (LSAR):
81 void InterpolateAudio(float *buffer, int len,
82  int firstBad, int numBad)
83 {
84  int N = len;
85  int i, row, col;
86 
87  wxASSERT(len > 0 &&

89  numBad < len &&

91 
92  if(numBad >= len)
93  return; //should never have been called!
94 
95  if (firstBad == 0) {
96  // The algorithm below has a weird asymmetry in that it
97  // performs poorly when interpolating to the left. If
98  // we're asked to interpolate the left side of a buffer,
99  // we just reverse the problem and try it that way.
100  float *buffer2 = new float[len];
101  for(i=0; i<len; i++)
102  buffer2[len-1-i] = buffer[i];
103  InterpolateAudio(buffer2, len, len-numBad, numBad);
104  for(i=0; i<len; i++)
105  buffer[len-1-i] = buffer2[i];
106  delete[] buffer2;
107  return;
108  }
109

111 
112  // Choose P, the order of the autoregression equation
113  int P = imin(numBad * 3, 50);
114  P = imin(P, imax(firstBad - 1, len - (firstBad + numBad) - 1));
115 
116  if (P < 3) {
117  LinearInterpolateAudio(buffer, len, firstBad, numBad);
118  return;
119  }

121  // Add a tiny amount of random noise to the input signal -
122  // this sounds like a bad idea, but the amount we're adding
123  // is only about 1 bit in 16-bit audio, and it's an extremely
124  // effective way to avoid nearly-singular matrices. If users
125  // run it more than once they get slightly different results;
126  // this is sometimes even advantageous.
127  for(i=0; i<N; i++)
128  s[i] += (rand()-(RAND_MAX/2))/(RAND_MAX*10000.0);
129 
130  // Solve for the best autoregression coefficients
131  // using a least-squares fit to all of the non-bad
132  // data we have in the buffer
133  Matrix X(P, P);
134  Vector b(P);
135 
136  for(i=0; i<len-P; i++)
137  if (i+P < firstBad || i >= (firstBad + numBad))
138  for(row=0; row<P; row++) {
139  for(col=0; col<P; col++)
140  X[row][col] += (s[i+row] * s[i+col]);
141  b[row] += s[i+P] * s[i+row];
142  }
143 
144  Matrix Xinv(P, P);
145  if (!InvertMatrix(X, Xinv)) {
146  // The matrix is singular! Fall back on linear...
147  // In practice I have never seen this happen if
148  // we add the tiny bit of random noise.
149  LinearInterpolateAudio(buffer, len, firstBad, numBad);
150  return;
151  }

154  Vector a = Xinv * b;
155 
156  // Create a matrix (a "Toeplitz" matrix, as it turns out)
157  // which encodes the autoregressive relationship between
158  // elements of the sequence.
159  Matrix A(N-P, N);
160  for(row=0; row<N-P; row++) {
161  for(col=0; col<P; col++)
162  A[row][row+col] = -a[col];
163  A[row][row+P] = 1;
164  }
165

167  // two pieces. Note that this code could be made to
168  // work even in the case where the "bad" samples are
169  // not contiguous, but currently it assumes they are.
170  // "u" is for unknown (bad)
171  // "k" is for known (good)
172  Matrix Au = MatrixSubset(A, 0, N-P, firstBad, numBad);
173  Matrix A_left = MatrixSubset(A, 0, N-P, 0, firstBad);
174  Matrix A_right = MatrixSubset(A, 0, N-P,
175  firstBad+numBad, N-(firstBad+numBad));
176  Matrix Ak = MatrixConcatenateCols(A_left, A_right);
177 
178  Vector s_left = VectorSubset(s, 0, firstBad);
179  Vector s_right = VectorSubset(s, firstBad+numBad,
180  N-(firstBad+numBad));
181  Vector sk = VectorConcatenate(s_left, s_right);
182 
183  // Do some linear algebra to find the best possible
184  // values that fill in the "bad" area
185  Matrix AuT = TransposeMatrix(Au);
186  Matrix X1 = MatrixMultiply(AuT, Au);
187  Matrix X2(X1.Rows(), X1.Cols());
188  if (!InvertMatrix(X1, X2)) {
189  // The matrix is singular! Fall back on linear...
190  LinearInterpolateAudio(buffer, len, firstBad, numBad);
191  return;
192  }
193  Matrix X2b = X2 * -1.0;
194  Matrix X3 = MatrixMultiply(X2b, AuT);
195  Matrix X4 = MatrixMultiply(X3, Ak);
196  // This vector contains our best guess as to the
197  // unknown values
198  Vector su = X4 * sk;
199 
200  // Put the results into the return buffer
201  for(i=0; i<numBad; i++)
202  buffer[firstBad+i] = (float)su[i];
203 }
