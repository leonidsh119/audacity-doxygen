53 /*
54  new_extension()
55  Puts a new extension name on a file name <filename>.
56  Removes the last extension name, if any.
57 */
58 static void new_extension(char *filename, char *extname, char *newname)
59 {
60  int found, dotpos;
61 
62  // First, strip the old extension
63  dotpos = strlen(filename);
64  found = 0;
65  do {
66  switch (filename[dotpos]) {
67  case '.':
68  found = 1;
69  break;
70  case '\\':
71  case '/':
72  case ':':
73  found = -1;
74  break;
75  default:
76  dotpos--;
77  if (dotpos < 0)
78  found = -1;
79  break;
80  }
81  } while (found == 0);
82 
83  if (found == -1) {
84  strncpy(newname, filename, MAX_NAME_SIZE);
85  }
86  if (found == 1) {
87  strncpy(newname, filename, dotpos);
88  newname[dotpos] = '\0';
89  }
90  // Make sure there is room in the string for the
91  // new filename and the extension
92  if (strlen(newname) + strlen(extname) + 1 < MAX_NAME_SIZE) {
93  strcat(newname, extname);
94  }
95 }

239  build_shortopt_string()
240  Creates a short args string from the options structure
241  for use with getopt_long
242 */
243 static char *build_shortopt_string(struct option *opts)
244 {
245  int count = 0;
246  char *shortstr = NULL;
247  int c = 0, n = 0;
248 
249  // Start by counting the number of options
250  while (opts[count].val != 0) {
251  count++;
252  }
253 
254 
255  // Allocate memory for the string
256  shortstr = malloc((count * 2) + 1);
257 
258  // And loop through the options again
259  for (n = 0; opts[n].val != 0; n++) {
260  if (opts[n].val > 0 && opts[n].val < 127) {
261  shortstr[c++] = opts[n].val;
262  if (opts[n].has_arg == optional_argument) {
263  fprintf(stderr, "gah: can't do optional arguments\n");
264  } else if (opts[n].has_arg == required_argument) {
265  shortstr[c++] = ':';
266  }
267  }
268  }
269 
270  // Finally - terminate the string
271  shortstr[c] = '\0';
272 
273  return shortstr;
274 }

578 int main(int argc, char **argv)
579 {
580  twolame_options *encopts = NULL;
581  audioin_t *inputfile = NULL;
582  FILE *outputfile = NULL;
583  short int *pcmaudio = NULL;
584  unsigned int frame_count = 0;
585  unsigned int total_frames = 0;
586  unsigned int frame_len = 0;
587  unsigned int total_bytes = 0;
588  unsigned char *mp2buffer = NULL;
589  int samples_read = 0;
590  int mp2fill_size = 0;
591  int audioReadSize = 0;
592 
593 
594  // Allocate memory for the PCM audio data
595  if ((pcmaudio = (short int *) calloc(AUDIO_BUF_SIZE, sizeof(short int))) == NULL) {
596  fprintf(stderr, "Error: pcmaudio memory allocation failed\n");
597  exit(ERR_MEM_ALLOC);
598  }
599  // Allocate memory for the encoded MP2 audio data
600  if ((mp2buffer = (unsigned char *) calloc(MP2_BUF_SIZE, sizeof(unsigned char))) == NULL) {
601  fprintf(stderr, "Error: mp2buffer memory allocation failed\n");
602  exit(ERR_MEM_ALLOC);
603  }
604  // Initialise Encoder Options Structure
605  encopts = twolame_init();
606  if (encopts == NULL) {
607  fprintf(stderr, "Error: initializing libtwolame encoder failed.\n");
608  exit(ERR_MEM_ALLOC);
609  }
610  // Get options and parameters from the command line
611  parse_args(argc, argv, encopts);
612 
613  // Display the filenames
614  print_filenames(twolame_get_verbosity(encopts));
615 
616  // Open the input file
617  if (use_raw) {
618  // use raw input handler
619  inputfile = open_audioin_raw(inputfilename, &sfinfo, sample_size);
620  } else {
621  // use libsndfile
622  inputfile = open_audioin_sndfile(inputfilename, &sfinfo);
623  }
624 
625  // Display input information
626  if (twolame_get_verbosity(encopts) > 1) {
627  inputfile->print_info(inputfile);
628  }
629  // Use information from input file to configure libtwolame
630  twolame_set_num_channels(encopts, sfinfo.channels);
631  twolame_set_in_samplerate(encopts, sfinfo.samplerate);
632 
633 
634  // Open the output file
635  outputfile = open_output_file(outputfilename);
636 
637  // initialise twolame with this set of options
638  if (twolame_init_params(encopts) != 0) {
639  fprintf(stderr, "Error: configuring libtwolame encoder failed.\n");
640  exit(ERR_INVALID_PARAM);
641  }
642  // display encoder settings
643  twolame_print_config(encopts);
644 
645 
646  // Only encode a single frame of mpeg audio ?
647  if (single_frame_mode)
648  audioReadSize = TWOLAME_SAMPLES_PER_FRAME;
649  else
650  audioReadSize = AUDIO_BUF_SIZE;
651 
652  // Calculate the size and number of frames we are going to encode
653  frame_len = twolame_get_framelength(encopts);
654  if (sfinfo.frames)
655  total_frames = sfinfo.frames / TWOLAME_SAMPLES_PER_FRAME;
656 
657 
658  // Now do the reading/encoding/writing
659  while ((samples_read = inputfile->read(inputfile, pcmaudio, audioReadSize)) > 0) {
660  int bytes_out = 0;
661 
662  // Force byte swapping if requested
663  if (byteswap) {
664  int i;
665  for (i = 0; i < samples_read; i++) {
666  short tmp = pcmaudio[i];
667  char *src = (char *) &tmp;
668  char *dst = (char *) &pcmaudio[i];
669  dst[0] = src[1];
670  dst[1] = src[0];
671  }
672  }
673  // Calculate the number of samples we have (per channel)
674  samples_read /= sfinfo.channels;
675 
676  // Do swapping of left and right channels if requested
677  if (channelswap && sfinfo.channels == 2) {
678  int i;
679  for (i = 0; i < samples_read; i++) {

708  exit(ERR_WRITING_OUTPUT);
