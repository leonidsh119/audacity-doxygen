412 void PaulStretch::process(float *smps,int nsmps)
413 {
414  //add NEW samples to the pool
415  if ((smps!=NULL)&&(nsmps!=0)){
416  if (nsmps>poolsize){
417  nsmps=poolsize;
418  }
419  int nleft=poolsize-nsmps;
420 
421  //move left the samples from the pool to make room for NEW samples
422  for (int i=0;i<nleft;i++) in_pool[i]=in_pool[i+nsmps];
423 
424  //add NEW samples to the pool
425  for (int i=0;i<nsmps;i++) in_pool[i+nleft]=smps[i];
426  }
427 
428  //get the samples from the pool
429  for (int i=0;i<poolsize;i++) fft_smps[i]=in_pool[i];
430  WindowFunc(3,poolsize,fft_smps);
431 
432  RealFFT(poolsize,fft_smps,fft_c,fft_s);
433 
434  for (int i=0;i<poolsize/2;i++) fft_freq[i]=sqrt(fft_c[i]*fft_c[i]+fft_s[i]*fft_s[i]);
435  process_spectrum(fft_freq);
436 
437 
438  //put randomize phases to frequencies and do a IFFT
439  float inv_2p15_2pi=1.0/16384.0*(float)M_PI;
440  for (int i=1;i<poolsize/2;i++){
441  unsigned int random=(rand())&0x7fff;
442  float phase=random*inv_2p15_2pi;
443  float s=fft_freq[i]*sin(phase);
444  float c=fft_freq[i]*cos(phase);
445 
446  fft_c[i]=fft_c[poolsize-i]=c;
447 
448  fft_s[i]=s;fft_s[poolsize-i]=-s;
449  }
450  fft_c[0]=fft_s[0]=0.0;
451  fft_c[poolsize/2]=fft_s[poolsize/2]=0.0;
452 
453  FFT(poolsize,true,fft_c,fft_s,fft_smps,fft_tmp);
454 
455  float max=0.0,max2=0.0;
456  for (int i=0;i<poolsize;i++){
457  float a=fabs(fft_tmp[i]);
458  if (a>max) max=a;
459  float b=fabs(fft_smps[i]);
460  if (b>max2) max2=b;
461  }
462 
463 
464  //make the output buffer
465  float tmp=1.0/(float) out_bufsize*M_PI;
466  float hinv_sqrt2=0.853553390593f;//(1.0+1.0/sqrt(2))*0.5;
467 
468  float ampfactor=1.0;
469  if (rap<1.0) ampfactor=rap*0.707;
470  else ampfactor=(out_bufsize/(float)poolsize)*4.0;
471 
472  for (int i=0;i<out_bufsize;i++) {
473  float a=(0.5+0.5*cos(i*tmp));
474  float out=fft_smps[i+out_bufsize]*(1.0-a)+old_out_smp_buf[i]*a;
475  out_buf[i]=out*(hinv_sqrt2-(1.0-hinv_sqrt2)*cos(i*2.0*tmp))*ampfactor;
476  }
477 
478  //copy the current output buffer to old buffer
479  for (int i=0;i<out_bufsize*2;i++) old_out_smp_buf[i]=fft_smps[i];
480 }
