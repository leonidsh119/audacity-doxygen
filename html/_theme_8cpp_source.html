779 bool ThemeBase::ReadImageCache( bool bBinaryRead, bool bOkIfNotFound)
780 {
781  EnsureInitialised();
782  wxImage ImageCache;
783  wxBusyCursor busy;
784 
785  // Ensure we have an alpha channel...
786 // if( !ImageCache.HasAlpha() )
787 // {
788 // ImageCache.InitAlpha();
789 // }
790 
791  // IF bBinary read THEN a normal read from a PNG file
792  if( bBinaryRead )
793  {
794  wxString FileName = FileNames::ThemeCachePng();
795  if( !wxFileExists( FileName ))
796  {
797  if( bOkIfNotFound )
798  return false; // did not load the images, so return false.
799  wxMessageBox(
800  wxString::Format(
801  _("Audacity could not find file:\n %s.\nTheme not loaded."),
802  FileName.c_str() ));
803  return false;
804  }
805  if( !ImageCache.LoadFile( FileName, wxBITMAP_TYPE_PNG ))
806  {
807  /* i18n-hint: Do not translate png. It is the name of a file format.*/
808  wxMessageBox(
809  wxString::Format(
810  _("Audacity could not load file:\n %s.\nBad png format perhaps?"),
811  FileName.c_str() ));
812  return false;
813  }
814  }
815  // ELSE we are reading from internal storage.
816  else
817  {
818  wxMemoryInputStream InternalStream(
819  (char *)ImageCacheAsData, sizeof(ImageCacheAsData));
820  if( !ImageCache.LoadFile( InternalStream, wxBITMAP_TYPE_PNG ))
821  {
822  // If we get this message, it means that the data in file
823  // was not a valid png image.
824  // Most likely someone edited it by mistake,
825  // Or some experiment is being tried with NEW formats for it.
826  wxMessageBox(_("Audacity could not read its default theme.\nPlease report the problem."));
827  return false;
828  }
829  }
830 
831  int i;
832  mFlow.Init(ImageCacheWidth);
833  // Load the bitmaps
834  for(i=0;i<(int)mImages.GetCount();i++)
835  {
836  wxImage &Image = mImages[i];
837  mFlow.mFlags = mBitmapFlags[i];
838  if( (mBitmapFlags[i] & resFlagInternal)==0)
839  {
840  mFlow.GetNextPosition( Image.GetWidth(),Image.GetHeight() );
841  // wxLogDebug(wxT("Copy at %i %i (%i,%i)"), mxPos, myPos, xWidth1, yHeight1 );
842  Image = GetSubImageWithAlpha( ImageCache, mFlow.Rect());
843  mBitmaps[i] = wxBitmap(Image);
844  }
845  }
846 
847 // return true; //To not load colours..
848  // Now load the colours.
849  int x,y;
850  mFlow.SetNewGroup(1);
851  wxColour TempColour;
852  const int iColSize=10;
853  mFlow.myHeight = iColSize+1;
854  for(i=0;i<(int)mColours.GetCount();i++)
855  {
856  mFlow.GetNextPosition( iColSize, iColSize );
857  mFlow.RectMid( x, y );
858  // Only change the colour if the alpha is opaque.
859  // This allows us to add NEW colours more easily.
860  if( ImageCache.GetAlpha(x,y ) > 128 )
861  {
862  TempColour = wxColour(
863  ImageCache.GetRed( x,y),
864  ImageCache.GetGreen( x,y),
865  ImageCache.GetBlue(x,y));
870  if( TempColour != wxColour(1,1,1) )
871  mColours[i] = TempColour;
872  }
873  }
874  return true;
