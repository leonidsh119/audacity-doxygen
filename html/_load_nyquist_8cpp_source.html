160 bool NyquistEffectsModule::AutoRegisterPlugins(PluginManagerInterface & pm)
161 {
162  // Autoregister effects that we "think" are ones that have been shipped with
163  // Audacity. A little simplistic, but it should suffice for now.
164  wxArrayString pathList = NyquistEffect::GetNyquistSearchPath();
165  wxArrayString files;
166 
167  if (!pm.IsPluginRegistered(NYQUIST_PROMPT_ID))
168  {
169  RegisterPlugin(pm, NYQUIST_PROMPT_ID);
170  }
171 
172  for (int i = 0; i < WXSIZEOF(kShippedEffects); i++)
173  {
174  files.Clear();
175  pm.FindFilesInPathList(kShippedEffects[i], pathList, files);
176  for (size_t j = 0, cnt = files.GetCount(); j < cnt; j++)
177  {
178  if (!pm.IsPluginRegistered(files[j]))
179  {
180  RegisterPlugin(pm, files[j]);
181  }
182  }
183  }
184 
185  // We still want to be called during the normal registration process
186  return false;
187 }
