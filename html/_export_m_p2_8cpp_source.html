212 int ExportMP2::Export(AudacityProject *project,
213  int channels, const wxString &fName,
214  bool selectionOnly, double t0, double t1, MixerSpec *mixerSpec, const Tags *metadata,
215  int WXUNUSED(subformat))
216 {
217  bool stereo = (channels == 2);
218  long bitrate = gPrefs->Read(wxT("/FileFormats/MP2Bitrate"), 160);
219  double rate = project->GetRate();
220  const TrackList *tracks = project->GetTracks();
221 
222  wxLogNull logNo; /* temporarily disable wxWidgets error messages */
223 
224  twolame_options *encodeOptions;
225  encodeOptions = twolame_init();
226 
227  twolame_set_in_samplerate(encodeOptions, (int)(rate + 0.5));
228  twolame_set_out_samplerate(encodeOptions, (int)(rate + 0.5));
229  twolame_set_bitrate(encodeOptions, bitrate);
230  twolame_set_num_channels(encodeOptions, stereo ? 2:1);
231 
232  if (twolame_init_params(encodeOptions) != 0)
233  {
234  wxMessageBox(_("Cannot export MP2 with this sample rate and bit rate"),
235  _("Error"), wxICON_STOP);
236  twolame_close(&encodeOptions);
237  return false;
238  }
239 
240  // Put ID3 tags at beginning of file
241  if (metadata == NULL)
242  metadata = project->GetTags();
243 
244  FileIO outFile(fName, FileIO::Output);
245  if (!outFile.IsOpened()) {
246  wxMessageBox(_("Unable to open target file for writing"));
247  twolame_close(&encodeOptions);
248  return false;
249  }
250 
251  char *id3buffer = NULL;
252  int id3len;
253  bool endOfFile;
254  id3len = AddTags(project, &id3buffer, &endOfFile, metadata);
255  if (id3len && !endOfFile)
256  outFile.Write(id3buffer, id3len);
257 
258  // Values taken from the twolame simple encoder sample
259  const int pcmBufferSize = 9216 / 2; // number of samples
260  const int mp2BufferSize = 16384 ; // bytes
261 
262  // We allocate a buffer which is twice as big as the
263  // input buffer, which should always be enough.
264  // We have to multiply by 4 because one sample is 2 bytes wide!
265  unsigned char* mp2Buffer = new unsigned char[mp2BufferSize];
266 
267  const WaveTrackConstArray waveTracks =
268  tracks->GetWaveTrackConstArray(selectionOnly, false);
269  Mixer *mixer = CreateMixer(waveTracks,
270  tracks->GetTimeTrack(),
271  t0, t1,
272  stereo? 2: 1, pcmBufferSize, true,
273  rate, int16Sample, true, mixerSpec);
274 
275  int updateResult = eProgressSuccess;
276  {
277  ProgressDialog progress(wxFileName(fName).GetName(),
278  selectionOnly ?
279  wxString::Format(_("Exporting selected audio at %ld kbps"), bitrate) :
280  wxString::Format(_("Exporting entire file at %ld kbps"), bitrate));
281 
282  while (updateResult == eProgressSuccess) {
283  sampleCount pcmNumSamples = mixer->Process(pcmBufferSize);
284 
285  if (pcmNumSamples == 0)
286  break;
287 
288  short *pcmBuffer = (short *)mixer->GetBuffer();
289 
290  int mp2BufferNumBytes = twolame_encode_buffer_interleaved(
291  encodeOptions,
292  pcmBuffer,
293  pcmNumSamples,
294  mp2Buffer,
295  mp2BufferSize);
296 
297  outFile.Write(mp2Buffer, mp2BufferNumBytes);
298 
299  updateResult = progress.Update(mixer->MixGetCurrentTime() - t0, t1 - t0);
300  }
301  }
302 
303  delete mixer;
304 
305  int mp2BufferNumBytes = twolame_encode_flush(
306  encodeOptions,
307  mp2Buffer,
308  mp2BufferSize);
309 
310  if (mp2BufferNumBytes > 0)
311  outFile.Write(mp2Buffer, mp2BufferNumBytes);
312 
313  twolame_close(&encodeOptions);
314 
315  delete[] mp2Buffer;
316 
317  /* Write ID3 tag if it was supposed to be at the end of the file */
318 
319  if (id3len && endOfFile)
320  outFile.Write(id3buffer, id3len);
321 
322  if (id3buffer) {
323  free(id3buffer);
324  }
325 
326  /* Close file */
327 
328  outFile.Close();
329 
330  return updateResult;
331 }

340 int ExportMP2::AddTags(AudacityProject * WXUNUSED(project), char **buffer, bool *endOfFile, const Tags *tags)
341 {
342 #ifdef USE_LIBID3TAG
343  struct id3_tag *tp = id3_tag_new();
344 
345  for (const auto &pair : tags->GetRange()) {
346  const auto &n = pair.first;
347  const auto &v = pair.second;
348  const char *name = "TXXX";
349 
350  if (n.CmpNoCase(TAG_TITLE) == 0) {
351  name = ID3_FRAME_TITLE;
352  }
353  else if (n.CmpNoCase(TAG_ARTIST) == 0) {
354  name = ID3_FRAME_ARTIST;
355  }
356  else if (n.CmpNoCase(TAG_ALBUM) == 0) {
357  name = ID3_FRAME_ALBUM;
358  }
359  else if (n.CmpNoCase(TAG_YEAR) == 0) {
360  // LLL: Some apps do not like the newer frame ID (ID3_FRAME_YEAR),
361  // so we add old one as well.
362  AddFrame(tp, n, v, "TYER");
363  name = ID3_FRAME_YEAR;
364  }
365  else if (n.CmpNoCase(TAG_GENRE) == 0) {
366  name = ID3_FRAME_GENRE;
367  }
368  else if (n.CmpNoCase(TAG_COMMENTS) == 0) {
369  name = ID3_FRAME_COMMENT;
370  }
371  else if (n.CmpNoCase(TAG_TRACK) == 0) {
372  name = ID3_FRAME_TRACK;
373  }
374 
375  AddFrame(tp, n, v, name);
376  }
377 
378  tp->options &= (~ID3_TAG_OPTION_COMPRESSION); // No compression
379 
380  // If this version of libid3tag supports it, use v2.3 ID3
381  // tags instead of the newer, but less well supported, v2.4
382  // that libid3tag uses by default.
383  #ifdef ID3_TAG_HAS_TAG_OPTION_ID3V2_3
384  tp->options |= ID3_TAG_OPTION_ID3V2_3;
385  #endif
386 
387  *endOfFile = false;
388 
389  id3_length_t len;
390 
391  len = id3_tag_render(tp, 0);
392  *buffer = (char *)malloc(len);
393  len = id3_tag_render(tp, (id3_byte_t *)*buffer);
394 
395  id3_tag_delete(tp);
396 
397  return len;
398 #else //ifdef USE_LIBID3TAG
399  return 0;
400 #endif

413  }
414 
415  id3_ucs4_t *ucs4 =
416  id3_utf8_ucs4duplicate((id3_utf8_t *) (const char *) v.mb_str(wxConvUTF8));
417 
418  if (strcmp(name, ID3_FRAME_COMMENT) == 0) {
419  // A hack to get around iTunes not recognizing the comment. The
420  // language defaults to XXX and, since it's not a valid language,
421  // iTunes just ignores the tag. So, either set it to a valid language
422  // (which one???) or just clear it. Unfortunately, there's no supported
423  // way of clearing the field, so do it directly.
424  id3_field *f = id3_frame_field(frame, 1);
425  memset(f->immediate.value, 0, sizeof(f->immediate.value));
426  id3_field_setfullstring(id3_frame_field(frame, 3), ucs4);
427  }
428  else if (strcmp(name, "TXXX") == 0) {
429  id3_field_setstring(id3_frame_field(frame, 2), ucs4);
430  free(ucs4);

433 
434  id3_field_setstring(id3_frame_field(frame, 1), ucs4);
435  }
436  else {
437  id3_field_setstrings(id3_frame_field(frame, 1), 1, &ucs4);
438  }
439 
440  free(ucs4);
441 
442  id3_tag_attachframe(tp, frame);
