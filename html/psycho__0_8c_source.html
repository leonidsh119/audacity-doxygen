35 /* MFC Mar 03
36  It's almost obscene how well this psycho model works for the amount of
37  computational effort that's put in.
38 
39  I got the idea from:
40  Hyen-O Oh et al "Low power mpeg audio encoders using simplified psychoacoustic model
41  and fast bit allocation"
42  IEEE Trans on Consumer Electronics v47 n3 August 2001. p613
43 
44  All this model does is look at the lowest ATH value within the subband, and then looks
45  at the scalefactors. It combines the two in a real dodgy way to get the SMRs.
46 
47  Although the output values aren't really close to any of the other psycho models,
48  the spread of values and the relative sizes of the values for the different subbands
49  is about right
50 
51  Feel free to make any sort of generic change you want. Add or subtract numbers, take
52  logs, whatever. Fiddle with the numbers until we get a good SMR output */

55 static psycho_0_mem *psycho_0_init(twolame_options * glopts, int sfreq)
56 {
57  FLOAT freqperline = (FLOAT) sfreq / 1024.0;
58  psycho_0_mem *mem = (psycho_0_mem *) TWOLAME_MALLOC(sizeof(psycho_0_mem));
59  int sb, i;
60 
61  for (sb = 0; sb < SBLIMIT; sb++) {
62  mem->ath_min[sb] = 1000; /* set it huge */
63  }
64 
65  /* Find the minimum ATH in each subband */
66  for (i = 0; i < 512; i++) {
67  FLOAT thisfreq = i * freqperline;
68  FLOAT ath_val = ath_db(thisfreq, 0);
69  if (ath_val < mem->ath_min[i >> 4])
70  mem->ath_min[i >> 4] = ath_val;
71  }
72 
73  return mem;
74 }

78 void psycho_0(twolame_options * glopts, FLOAT SMR[2][SBLIMIT], unsigned int scalar[2][3][SBLIMIT])
79 {
80  psycho_0_mem *mem;
81  int nch = glopts->num_channels_out;
82  int sfreq = glopts->samplerate_out;
83  int ch, sb, gr;
84  unsigned int minscaleindex[2][SBLIMIT]; /* Smaller scale indexes mean bigger scalefactors */
85 
86  if (!glopts->p0mem) {
87  glopts->p0mem = psycho_0_init(glopts, sfreq);
88  }
89  mem = glopts->p0mem;
90 
91 
92  /* call functions for critical boundaries, freq. */
93  if (!glopts->p0mem) { /* bands, bark values, and mapping */
94 
95  } else {
96 
97  mem = glopts->p0mem;
98 
99  }
100 
101 
102  /* Find the minimum scalefactor index for each ch/sb */
103  for (ch = 0; ch < nch; ch++)
104  for (sb = 0; sb < SBLIMIT; sb++)
105  minscaleindex[ch][sb] = scalar[ch][0][sb];
106 
107  for (ch = 0; ch < nch; ch++)
108  for (gr = 1; gr < 3; gr++)
109  for (sb = 0; sb < SBLIMIT; sb++)
110  if (minscaleindex[ch][sb] > scalar[ch][gr][sb])
111  minscaleindex[ch][sb] = scalar[ch][gr][sb];
112 
113  /* Oh yeah. Fudge the hell out of the SMR calculations by combining the scalefactor table index
114  and the min ATH in that subband There are probably more elegant/correct ways of combining
115  these values, but who cares? It works pretty well MFC Mar 03 */
116  for (ch = 0; ch < nch; ch++)
117  for (sb = 0; sb < SBLIMIT; sb++)
118  SMR[ch][sb] = 2.0 * (30.0 - minscaleindex[ch][sb]) - mem->ath_min[sb];
119 }
