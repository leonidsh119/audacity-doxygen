203 void *BlockFile::CalcSummary(samplePtr buffer, sampleCount len,
204  sampleFormat format)
205 {
206  if(fullSummary)delete[] fullSummary;
207  fullSummary = new char[mSummaryInfo.totalSummaryBytes];
208 
209  memcpy(fullSummary, headerTag, headerTagLen);
210 
211  float *summary64K = (float *)(fullSummary + mSummaryInfo.offset64K);
212  float *summary256 = (float *)(fullSummary + mSummaryInfo.offset256);
213 
214  float *fbuffer = new float[len];
215  CopySamples(buffer, format,
216  (samplePtr)fbuffer, floatSample, len);
217 
218  sampleCount sumLen;
219  sampleCount i, j, jcount;
220 
221  float min, max;
222  float sumsq;
223 
224  // Recalc 256 summaries
225  sumLen = (len + 255) / 256;
226 
227  for (i = 0; i < sumLen; i++) {
228  min = fbuffer[i * 256];
229  max = fbuffer[i * 256];
230  sumsq = ((float)min) * ((float)min);
231  jcount = 256;
232  if (i * 256 + jcount > len)
233  jcount = len - i * 256;
234  for (j = 1; j < jcount; j++) {
235  float f1 = fbuffer[i * 256 + j];
236  sumsq += ((float)f1) * ((float)f1);
237  if (f1 < min)
238  min = f1;
239  else if (f1 > max)
240  max = f1;
241  }
242 
243  float rms = (float)sqrt(sumsq / jcount);
244 
245  summary256[i * 3] = min;
246  summary256[i * 3 + 1] = max;
247  summary256[i * 3 + 2] = rms;
248  }
249  for (i = sumLen; i < mSummaryInfo.frames256; i++) {
250  // filling in the remaining bits with non-harming/contributing values
251  summary256[i * 3] = FLT_MAX; // min
252  summary256[i * 3 + 1] = -FLT_MAX; // max
253  summary256[i * 3 + 2] = 0.0f; // rms
254  }
255 
256  // Recalc 64K summaries
257  sumLen = (len + 65535) / 65536;
258 
259  for (i = 0; i < sumLen; i++) {
260  min = summary256[3 * i * 256];
261  max = summary256[3 * i * 256 + 1];
262  sumsq = (float)summary256[3 * i * 256 + 2];
263  sumsq *= sumsq;
264  for (j = 1; j < 256; j++) { // we can overflow the useful summary256 values here, but have put non-harmful values in them
265  if (summary256[3 * (i * 256 + j)] < min)
266  min = summary256[3 * (i * 256 + j)];
267  if (summary256[3 * (i * 256 + j) + 1] > max)
268  max = summary256[3 * (i * 256 + j) + 1];
269  float r1 = summary256[3 * (i * 256 + j) + 2];
270  sumsq += r1*r1;
271  }
272 
273  float rms = (float)sqrt(sumsq / 256); // the '256' is not quite right at the edges as not all summary256 entries will be filled with useful values
274 
275  summary64K[i * 3] = min;
276  summary64K[i * 3 + 1] = max;
277  summary64K[i * 3 + 2] = rms;
278  }
279  for (i = sumLen; i < mSummaryInfo.frames64K; i++) {
280  summary64K[i * 3] = 0.0f; // probably should be FLT_MAX, need a test case
281  summary64K[i * 3 + 1] = 0.0f; // probably should be -FLT_MAX, need a test case
282  summary64K[i * 3 + 2] = 0.0f;
283  }
284 
285  // Recalc block-level summary
286  min = summary64K[0];
287  max = summary64K[1];
288  sumsq = (float)summary64K[2];
289  sumsq *= sumsq;
290 
291  for (i = 1; i < sumLen; i++) {
292  if (summary64K[3*i] < min)
293  min = summary64K[3*i];
294  if (summary64K[3*i+1] > max)
295  max = summary64K[3*i+1];
296  float r1 = (float)summary64K[3*i+2];
297  sumsq += (r1*r1);
298  }
299 
300  mMin = min;
301  mMax = max;
302  mRMS = sqrt(sumsq / sumLen);
303 
304  delete[] fbuffer;
305 
306  return fullSummary;
307 }

309 static void ComputeMinMax256(float *summary256,
310  float *outMin, float *outMax, int *outBads)
311 {
312  float min, max;
313  int i;
314  int bad = 0;
315 
316  min = 1.0;
317  max = -1.0;
318  for(i=0; i<256; i++) {
319  if (summary256[3*i] < min)
320  min = summary256[3*i];
321  else if (!(summary256[3*i] >= min))
322  bad++;
323  if (summary256[3*i+1] > max)
324  max = summary256[3*i+1];
325  else if (!(summary256[3*i+1] <= max))
326  bad++;
327  if (wxIsNaN(summary256[3*i+2]))
328  bad++;
329  if (summary256[3*i+2] < -1 || summary256[3*i+2] > 1)
330  bad++;
331  }
332 
333  *outMin = min;
334  *outMax = max;
335  *outBads = bad;
336 }

340 void BlockFile::FixSummary(void *data)
341 {
342  if (mSummaryInfo.format != floatSample ||
343  mSummaryInfo.fields != 3)
344  return;
345 
346  float *summary64K = (float *)((char *)data + mSummaryInfo.offset64K);
347  float *summary256 = (float *)((char *)data + mSummaryInfo.offset256);
348 
349  float min, max;
350  int bad;
351  int i;
352 
353  ComputeMinMax256(summary256, &min, &max, &bad);
354 
355  if (min != summary64K[0] || max != summary64K[1] || bad > 0) {
356  unsigned int *buffer = (unsigned int *)data;
357  int len = mSummaryInfo.totalSummaryBytes / 4;
358 
359  for(i=0; i<len; i++)
360  buffer[i] = wxUINT32_SWAP_ALWAYS(buffer[i]);
361 
362  ComputeMinMax256(summary256, &min, &max, &bad);
363  if (min == summary64K[0] && max == summary64K[1] && bad == 0) {
364  // The byte-swapping worked!
365  return;
366  }
367 
368  // Hmmm, no better, we should swap back
369 
370  for(i=0; i<len; i++)
371  buffer[i] = wxUINT32_SWAP_ALWAYS(buffer[i]);
372  }
373 }
